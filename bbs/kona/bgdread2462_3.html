مەن قۇمۇلغا بارغان قۇمۇللۇقلارنىڭ چىراي شەكلى ئاق سېرىق باش شەكلى سوزماق كېلىدىكەن بەدىنى گىروي كېلىدىكەن ئۇلارنىڭ مىجەز قىلىقلىرى غۇلجۇنۇڭ چوڭلىرغا ئوخشۇشۇپ كېتىدىكەن بولۇپمۇ  غۇلجۇنۇڭ يەرلىكلىرىگە بەكلا ئوخشايدىكەن سۆزىدە :مۇنۇ ،ماشۇ،بوكەتتى قىككەتتى ،قىلغىنەۋۇي،بولغىنەۋۇي دىگەندەك سۆزلىرى مېنى تېخىمۇ ھەيران قالدۇردى،مەن بازا ئارلاۋېتىپ مېنىڭ سىراجىدىن چوڭ دادامغا ئوپمۇ ئوخشايدىغان بىر كىشىنى كۆرۈپ تېخىمۇ ھەيران قالدىم، تارىخنى خودايىملا بۈلۈدۇ ،تۇرپاننىڭ قەدىمى تارىخى 3000 يىلدىنمۇ ئېشىپ كېتىدىكەن شىمالى شىنجاڭدىكى ئۇيغۇرلارنىڭمۇ تارىخى قانچە مىڭ يىلغا بارىدۇ بۇنى خۇدايىم بۈلۈدۇ.

قۇمۇللۇقتىن پالۋان چىقسا چىقىدۇكى خوتەنلىكتىن ھەرگىز پالۋان چىقمايدۇ
قاشقا خوتەنىڭ ئادەملىرى خوتۇن پۇرۇش كېلدۇ ئۇلاردا قۇرقۇنچاق بىلەن ۋىجدانسىز تولا  ئۇلارنى پالۋان دىيىش تولمۇ كۈلكۈلۈك ئىش



قەشقەردىن دۇنياغا مەشھۇر ئالىم مەھمۇد قەشقىرى، يۈسۈپ خاس ھاجىپ، ئىسلامنى تۇنجى بولۇپ قوبۇل قىلغان سۇلتان سۇتۇق بۇغراخان قاتارلىقلار چىقتى، ئۇلار تارىخنىڭ سىنىقىدىن ئۆتكەن قەھرىمانلار، خوتەندىن 2مىڭ يىللار بۇرۇنلا غازىباي قاتارلىق تىبابەت ئالىملىرى چىقىشتىن باشلاپ ھازىرغىچە چىققان ئالىملارنى تولۇق بىلىپ بولۇشىمىزمۇ ئاسان ئەمەس، ئەلۋەتتە قومۇل ، غۇلجىلاردىن چىققان قەھرىمانلارنىڭ نامىنىمۇ كۆپ ئاڭلىدۇق ، ئەمىلىيەتتە  ئۇيغۇرلارنىڭ ھەرقايسى يۇرتلىرىدىن چىققان مەشھۇر ئالىم ، قەھرىمانلارنىڭ سانى ھەقىقەتەنمۇ ئاز ئەمەس. ئۇلارنىڭ يۇرت تەۋەسى ۋە نەسەبىنى بىلىشنىڭ ھىچقانداق زىيىنى يوق، ئەڭ مۇھىمى مەشھۇر ئالىم ، قەھرىمانلارنىڭ روھىدىن ئۆگىنىش.

قۇمۇللۇقتىن پالۋان چىقسا چىقىدۇكى خوتەنلىكتىن ھەرگىز پالۋان چىقمايدۇ
قاشقا خوتەنىڭ ئادەملىرى خوتۇن پۇرۇش كېلدۇ ئۇلاردا قۇرقۇنچاق بىلەن ۋىجدانسىز تولا  ئۇلارنى پالۋان دىيىش تولمۇ كۈلكۈلۈك ئىش


ئەبلەخ، مانا سەندەك ئىچكى بۆلگۈنچىلەر ،خەلقىمىزنىڭ ئارىسىغا ئىتىپاقسىزلىق ئۇرۇقىنى چېچىۋاتقان ،بەتبەشىرە مەخلۇقسىلەر ،
قارىسام سەن ئىبلىس بىلەن ئالۋاستىنىڭ ئارلاشمىسىدىن تۆرەلگەن نىمىدەك قىلىسەن .

بىلىپ قوي مەخلۇق ، مەيلى دۇنيانىڭ قانداقلا يېرىدە بولسۇن پەقەت ئۇيغۇرلا بولىدىكەن بىزنىڭ قېرىندىشىمىز ، قەشقەر ،خوتەن ئۇيغۇرلىرى قۇمۇل ئۇيغۇرلىرىغا ئوخشاشلا 32ھەرىپنى ئىشلىتىپ خەت يازىدۇ . بەش ۋاخ نامازدا  سەن يۈزلەنگەن قىبلىگە يۈزلىنىدۇ.
سەن قەشقەر خوتەنگە بېرىپ ياشاپ باقمىغان ،ئۇلار بىلەن ئارلىشىپ كۆرۈپ باقمىغان تۇرۇپ نىمىنى بىلىسەن.

مانا سەندەك  بەدىنىدە قېرىنداشلىق گېنى يوقاپ كەتكەن غەيرى مەخلۇقلار قۇمۇل خەلقنىنىڭ يۈزىنى چۈشۈرۈشمە.
سەندەك غەيرى مەخلۇقلار ،ئۇيغۇر مىللىتىمىزنىڭ ئارىسىدىكى يامان سۈپەتلىك راك ئۆسمىسىگە ئوخشاش نىمىلەر ، سەنلەر سەۋەپلىك ئۇيغۇرلار بىرسى بىرسىنى خاتا چۈشىنىپ تېخىچە قۇمدەك چېچىلىپ يۈرۈۋاتىمىز.

<<ياخشىلىق بوسۇغىدىن چىقىپ بولغۇچە يامانلىق نەچچە تاغدىن ھالقىپتۇ>> دىگەندەك سىنىڭ مۇشۇ بىر ئىنكاسىڭنىڭ قانچىلىك يامان تەسىر پەيدا قلىدىغانلىقىنى بىلەمسەن .

شۇنداق بولغانكەن مەشھۇر شەخىسلەرنىڭ تىگى تەكتىنى سۈرۈشتۈرۈش ئارتۇقچە ئىش ئەمەس
