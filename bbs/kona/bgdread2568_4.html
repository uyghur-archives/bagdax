  خەنزۇ ئەدەبىياتنىڭ بىزدىن خېلى ئۈستۈنلىكى ئېنىق ، خەنزۇ مىللىتىنى يەزغۇچىلىرىمۇ ھازىر بەك يامان ،

2011 ئەڭ نادىر ئوبزورلار . ياكى 2011 دىكى ئەڭ نادىر فىلياتونلار دىگەندەك تەجىمە كىتاپلارنى چىقارسا بىزنىڭ بوزقىردەك يازغۇچى تەرجىمانلىرىمىز  32 ھەرىپىمىز بىلەن ئۇ ئەسەرگەتېخىمۇ جەن كىرگۈزۋېتەتتى ئەمەسمۇ   !
  ئابدۇكېرەم خوجايوفنىڭ ھېلىقى چاقچىقى يەنە ئېسىمگە كەپ قالدى !

ozturuk،ياخشى نيەت بېلەن ئىپادە قىلغان ئىلھاملاندۇرىشىڭىزغا كۆپ رەھمەت.
ئەمەلىيەتتە خەنزۇچە مەتبۇئاتلاردىكى سېياسىي تۈزۈلمە، ئەدىيىلە ئادىللىقىقىغا تاقىلىدىغان مەسىلىلەردە ئاجايىپ ياخشى ئەسەرلەر مەيدانغا كېلىۋاتىدۇ.ھەتتا ئۇلاردا پىكىر يىلسېرى يېڭىلىنىپ بېرىۋاتىدۇ.بۇ يىل داغ-دۇغا قوزغاپ مودا بولىۋاتقان ئەسەرلەر يەنە بىر يىلدا ئۇنىڭدىنمۇ نادىر ئەسەرلەرگە ئورۇن بوشىتىشقا مەجبۇر بولىۋاتىدۇ.ئۇلاردا بىر كىتاب ئەڭ تېز بولغاندا بىر ھەپتىگە قالماي نەشىردىن چىقىپ بولىدىكەن.
ھالبۇكى بىزدە ئەشۇلارنىڭ ئەڭ نادىر ئەسەرلىرىنىڭ يۈزدە %2نى ۋاقتىدا تەرجىمە قىلىپ تونۇشتۇرۇشقىمۇ پۇرسەت بولمايۋاتىدۇ.بۇنىڭ  تۈپكى سەۋەپچىسى
يەنىلا نەشىرىيات ئورۇنلىرىدىكى قاتماللىقنى بۇزۇپ تاشلاپ ئۇنى بازارغا كەڭ ئېچىۋېتىش مېخانىزىمىنىڭ شەكىللىنىپ بولالماسلىقىدۇر.قارىماققا ھۆكۈمەت نەشىرىيات ئورۇنلىرىنى پاتماندا پۇل خەجلەپ بېقىۋاتقاندەك قىلسىمۇ ئەمما ئەشۇ پۇل-خىراجەتلەر شۇ ئورۇنلىرىدىكىلەرنىڭ تەپەككۇرىنى گاللاشتۇرۇپ ، ئۇلارنى تەييارغا ھەييار بولۇپ كۈن ئۆتكۈزىدىغان جانباقتى نان كۈچۈكلىرىگە ئايلاندۇرۇپ قويىۋاتىدۇ.

ئەمدى ئاكادېمىكلار  كوچىسىدىكى پارىخور ئەمەلدارلارغا قاراپ باقايلى:

ئەمەلدارلار بىلەن ئاكادېمىكلار ئۆزلىرىنىڭ يولىدا ماڭسا


ھازىر ئالىي ئەمەلدار ئالىي باشقۇرغۇچىلار ئاكادېمىكلار قاتارىغا كىرمەكتە. بۇ ئاكادېمىكلارنى كۆپەيتىپ سايلاش مەقسىتىگە خىلاپ بولۇپلا قالماستىن، يەنە ئىلىم پەنگە بولغان ناچار خاھىشلارنى پەيدا قىلىدۇ. ئىلمىي ساھە، سىياسىي ساھە، سودا ساھەسى چەك چېگراسى ئايرىپ، مەمۇرىيەت، پۇل مۇئامىلىنىڭ ئىلمىي ساھەگە يۇقۇشىنى توسۇش ئۈچۈن، ئىلمىي ساھەنى ئىلمىي ساھەگە قايتۇرۇپ كەلگەندىلا، ئاكادېمىكلارنى كۆپەيتىپ سايلاشنىڭ ئادىللىقىغا ھەقىقىي كاپالەتلىك قىلغىلى بولىدۇ.

يېقىندا جۇڭگو قۇرۇلۇش ئاكادېمىيەسى كۆپەيتىپ سايلانغان ئاكادېمىكلارنىڭ ئىسىملىكىنى ئېلان قىلدى. بۇ كۆپلىگەن كىشىلەرنىڭ كۆڭلىنى تىنجىتتى. يېڭىدىن سايلانغان 54 ئاكادېمىك ئىچىدە ۋەزىپە ئۆتەۋاتقان ياكى ۋەزىپىدىن بوشانغان ئالىي ئەمەلدارلار يوق، كارخانىنىڭ يۇقىرى دەرىجىلىك باشقۇرغۇچىلىرىدىن ئۈچى بار. لېكىن بۇ يىل 5-ئايدا ئېلان قىلىنغان قۇرۇلۇش ئاكادېمىكلىرى تۈنجى نۆۋەتلىك كۈچكە ئىگە نامزاتلىرى ئىچىدە، ھۆكۈمەت ئەمەلدارلىرى ۋە كارخانىلارنىڭ يۇقىرى دەرىجىلىك باشقۇرغۇچىلىرى 44 كىشىگە يېتەتتى. بۇ ئاممىنىڭ ئومۇمىي يۈزلۈك گۇمانىنى قوزغىغان ئىدى.

كۆپلىگەن ئامما قارشى تۇرۇش سادالىرىنى ياڭراتسىمۇ، يەنە بەزى كىشىلەر ئالىي ئەمەلدارلار، ئالىي باشقۇرغۇچىلار تەرىپىدە تۇرۇپ سۆز قىلىپ، كۆپلىگەن ئالىي ئەمەلدارلار ۋە ئالىي باشقۇرغۇچىلار ئىلىم پەن سەركىلىرى، كارخانىلارنىڭ ئالىي باشقۇرغۇچىلىرىمۇ شۇنداق، ئۇلار ئىلگىرى قۇرۇلۇش پەن تېخنىكا ساھەسىگە زور تۆھپە قوشقان، ئۇلارنى رەت قىلىپ سىرتتا قالدۇرۇش ئادىل ئەمەس، دېيىشتى.

قۇرۇلۇش ئاكادېمىيەسى جۇڭگو قۇرۇلۇش پەن تېخنىكا ساھەسىنىڭ ئالىي شەرەپلىك، مەسلىھەت بېرىش خاراكتېرلىك ئىلمىي ئاپپاراتى. قۇرۇلۇش ئاكادېمىيەسىنىڭ ئاكادېمىكى دۆلەت تەسىس قىلغان قۇرۇلۇش پەن تېخنىكا ساھەسىدىكى ئالى ئىلمىي نام. مەيلى قۇرۇلۇش ئاكادېمىيەسى بولسۇن ياكى قۇرۇلۇش ئاكادېمىكى بولسۇن، ئۇنىڭ ئەڭ مۇھىم ماھىيىتى ئىلىم پەن، باشقۇرۇش ئەمەس. دۆلەتنىڭ قۇرۇلۇش ئاكادېمىيەسى تەسىس قىلىشى ئاساسلىقى، ئىلمىي ساھەدىكى مۇنەۋۋەر ئىختىساسلىقلارنى تېخىمۇ ياخشى جەم قىلىپ، دۆلەتنىڭ پەن تېخنىكا تەرەققىياتىغا چارە-تەدبىر كۆرسىتىش، يۇقىرى سۈپەتلىك مەسلىھەت بېرىش مۇلازىمىتى قىلىشنى مەقسەت قىلىدۇ . ھەر ئىككى يىلدا بىر قېتىم ئاكادېمىكلارنى كۆپەيتىپ سايلاشنىڭ مەقسىتى، بىر تەرەپتىن ئۇلارنىڭ ئىلگىرىكى مۇۋەپپەقىيەتلىرىنى ئېتىراپ قىلىش. لېكىن تېخىمۇ مۇھىمى ئۇلارنى غەيرەت ئۈستىگە غەيرەت قوشۇپ، پەن تېخنىكا تەتقىقاتىدا بۆسۈش ياساشقا ئىلھاملاندۇرۇش. مەيلى ئالىي ئەمەلدارلار ياكى يۇقىرى دەرىجىلىك باشقۇرغۇچىلار بولسۇن،ئىلگىرى ئىلمىي ساھەدە مۇنەۋۋەرلەردىن بولغان بولسىمۇ، لېكىن ئىلمىي ساھەدىن ۋاز كېچىپ ئەمەلدار، ئالىي باشقۇرغۇچى بولدى. ئۇلار كۆپ ۋاقتىنى باشقۇرۇشقا سەرپ قىلىدۇ، پەن تېخنىكىنى ئۆزىنىڭ ئۆتكەلگە ھۇجۇم قىلىش نىشانى قىلالمايدۇ. زور ئىلمىي مۇۋەپپەقىيەتلەرنى قولغا كەلتۈرمىكى قىيىن. شۇنداق ئىكەن ئەمەلدارلار ئەمەلدارلىقىنى، سودىگەرلەر سودىگەرلىكىنى تەكىتلىشى كېرەك. ھۆكۈمەتنىڭ ئالىي ئەمەلدارى ۋە كارخانىنىڭ ئالىي باشقۇرغۇچىسى بولغان ئىكەن، ئۆزىنىڭ مۇنەۋۋەر ياكى ئەمەسلىكىنى ئۆلچىگەندە، ئەلۋەتتە سىياسىي نەتىجىسى ۋە كەسپىي نەتىجىسى بىلەن ئۆلچىشى، ھەرگىزمۇ ئىلمىي مۇۋەپپەقىيەت بىلەن ئۆلچىمەسلىكى كېرەك. شۇڭا ئالىي ئەمەلدار، ئالىي باشقۇرغۇچى بولغاندىن كېيىن ئاكادېمىك كەسكىن رىقابىتىگە قاتنىشىشقا باش قاتۇرسا، دۆلەتنىڭ قۇرۇلۇش ئاكادېمىيەسى قۇرۇش، ئاكادېمىكلارنى تولۇقلاپ سايلاش مەقسىتىگە ئۇيغۇن كەلمەيدۇ. ھۆكۈمەت ئىشلىرى ۋە سودا ئىشلىرى بىلەن شۇغۇللىنىش مەجبۇرىيىتى ۋە بۇرچىغا خىلاپلىق قىلغانلىق بولىدۇ.




ئېلىمىزنىڭ ئىلمىي ساھەسى ئەسلىدىنلا ساغلام ئەمەس، بولۇپمۇ ئىلمىي ساھەنى مەمۇرىيلاشتۇرۇش، مەنپەئەتلەندۈرۈش كەسكىنلىشىۋاتقان رېئال مۇھىتتا، ئالىي ئەمەلدارلار ۋە ئالى باشقۇرغۇچىلارنىڭ ئاكادېمىكلار ساھەسىگە كىرىشى، ئىلىم ساھەسىدە ناچار ئىستىل پەيدا قىلىشى مۇمكىن. يېقىنقى يىللاردىن بۇيان، گەرچە قۇرۇلۇش ئاكادېمىيەسى ئاكادېمىكلاردىن سايلىنىشقا ياردەم تەلەپ قىلىش، پارا بېرىش، مۇناسىۋەتلىشىش، مېھمان قىلىش، سوۋغا قىلىش، مۇۋەپپەقىيەتلىرىنى كۆپتۈرۈش قاتارلىق تۇيۇق يولغا ماڭماسلىقنى كۆپ قېتىم تەلەپ قىلغان بولسىمۇ، لېكىن باھالاش جەريانىدا يەنىلا توسقىلى بولمىدى. ئەگەر ھوقۇق ۋە پۇل ئىشلەتسە، ئادىللىققا كاپالەتلىك قىلىش قىيىنغا توختايدۇ.

رىقابەت كۈنسايىن خەلقئارالىشىۋاتقان، ئۆتكۈرلىشىۋاتقان بۈگۈنكى دۇنيادا ئىلمىي ساھە بولسۇن ياكى سىياسىي ساھە، سودا ساھەسى بولسۇن بىر نىيەت بىر مەقسەتتە بولىدىغانلا بولسا يۇقىرىقىدەك ئەھۋاللار كۆرۈلمەيدۇ. ئىلمىي ساھەنى تاشلاپ ھۆكۈمەت ئىشلىرى، سودا ئىشلىرى بىلەن شۇغۇللانغان ئىكەن شۇ ئىشنى جان دىل بىلەن قىلىش كېرەك. ھەر گىزمۇ ئۆز ئىشىنى قىلىۋېتىپ تۇرۇپ ئۆزگىنىڭ ئىشىغا كۆز تىكمەسلىك لازىم .

خۇددى بەزى ئاكادېمىكلار ئېيتقاندەك: ئىلمىي ساھە، سىياسىي ساھە، سودا ساھەسىنىڭ چەك چىگرىسىنى ئايرىپ، ھوقۇق ۋە پۇلنىڭ ئىلمىي ساھەنى بۇلغىشىنى توسۇپ، ئىلىمنى ئىلىمگە قايتۇرۇپ كېلىدىغانلا بولسا، ئاكادېمىكلارنى كۆپەيتىپ سايلاشنىڭ ئادىللىقىغا ھەقىقىي كاپالەتلىك قىلغىلى بولىدۇ. پەن تېخنىكا ۋە ئىلمىي ساھەنى ساپلاشتۇرغىلى بولىدۇ. شۇڭا قۇرۇلۇش ئاكادېمىيەسىنىڭ بېسىمغا تاقابىل تۇرۇپ، ئەل رايىغا بېقىپ، ئاكادېمىكلارنى كۆپەيتىپ سايلىغاندا ئەمەلدارلاشتۇرماسلىقنى دائىملاشتۇرۇشنى ئۈمىد قىلىمىز.
