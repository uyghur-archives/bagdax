خەت چوڭلىقى : كىچىك نورمال چوڭ

 خوتەنگە كەلسىڭىز چوقۇم ئەسقىتىدۇ،

رەسىم:
خوتەن شەھىرى يۇرۇڭقاش دەرياسى بىلەن قاراقاش دەرياسىدىن ھاسىل بولغان تىنما تۈزلەڭلىكىگە جايلاشقان. شىمالىي تەرىپى لوپ ناھىيىسى بىلەن دەريا ئارقىلىق ئايرىلىپ تۇرىدۇ، غەربىي تەرىپى خوتەن ناھىيىسىگە تۇتىشىدۇ. جەنۇبتىن شىمالغىچە بولغان ئۇزۇنلۇقى 27 كىلومېتىر، شەرقىتىن غەربكىچە بولغان كەڭلىكى 11.5 كىلومېتىر، ئومۇمىي يەر مەيدانى 189 كۋادرات كىلومېتىر. قارمىقىدا بىر بازار ئۈچ يېزا، تۆت مەھەللە ئىش بېجىرىش باشقارمىسى، 28 ئاھالە كومىتېتى، 58 كەنت ئاھالە كومىتېتى بار. شەھەر رايونىنىڭ ئۈرۈمچى شەھىرى بىلەن بولغان ئارىلىقى 1500 كىلومېتىر. خوتەن شەھىرى مۆتىدىل بەلۋاغ قۇرغاق ئىقلىم رايونىغا كىرىدۇ. يىللىق ئوتتۇرىچە تېمپېراتۇرىسى 12.2℃، ئەڭ يۇقۇرى تېمپېراتۇرىسى 40.6℃، ئەڭ تۆۋەن تېمپېراتۇرىسى – 21.6℃
قىشتا ئاساسەن بەك سوغۇق يوق دىيەرلىك،يازدا بەك ئىىىسىق بۇلۇپ كەتسىمۇ سالقىنلىنىدىغان ئورۇنلار شۇنداق كوپ.
ئورۇمچىدىن ماشىنا بىلەن خوتەنگە 18-22 سائەتكىچە يىتىپ بارالايسىز ،خۇدا سالامەتلىكنى بەرسە،
بىلەت باھاسى 260---380 يۇەنگىچە بۇلۇپ ،ئادەتتە ماشىنا قىس ئەمەس ۋاقىتلاردا شوپۇر بىلەن كىلىشەلىسىڭىز 150 250 يۇەنگىچە يولغا چىقالايسىز،يولدا بىر قىتىملا تاماققا توختايدۇ،
ئايرۇپلان بىلىتىمۇ بەك قىممەت ئەمەس بىلەتنىڭ ئەرزان ۋاختىدا ئالسىڭز 500يۇەن ئەتراپىدا بىلەت ئالالايسىز،
خوتەنگە كەلمەكچى بولغان دوستلارنىڭ كوڭۇللۇك ساياھات قىلىشى ئۇچۇن تۇۋەندىكى پايدىلانمىلارنى سۇندۇم،
1.يىمەك ئىچمەك ئورۇنلىرى-لەغمەن يەيمەن دەيدىغانلارغا شەھەر قۇرلۇش ئىدارىسى ئالدىدىكى ئابلىمىت ئاچچىق لەغمەن پادىشاھى،(ئات بازىرىدىكى چېڭجيىەن 1.يول ئاپتۇبۇسىنىڭ بىكىتى ) بۇ يەرنىڭ لەغمىنى خېلى ئوخشاپ قالىدۇ،پولو يەيمەن دەپ قالسىڭىز قىززىق دەۋازىسىدىكى پولوخانىغا بېرىڭ،مەخسۇس پولو ئاشخانىسى ،قۇرۇتۇپ پىشۇرۇلغان ئاسما كاۋاپ، تۇنۇر كاۋىپى،توخۇ كاۋىپى يىمەكچى بولسىڭىز خوتەن چوڭ جامەنىڭ قېشىدىكى قاخشال كاۋاپلىرىنى تېتىپ بېقىڭ،لىكىن خېلى ئۇزاق ساقلايسىز،ئەمما تەمىنى ئۇنتالمايسىز،ئاچچىق-چۇچۇك كالاگۇشى ئوگرىسى ئىچمەكچى بولسىڭىز جىياڭسۇ مىھمانخانىسىنىڭ ئاستىدىكى ‹‹گۇلباغ كالا گوشى ئوگرىىخانا››غا بېرىڭ،يەنە بىر داڭلىق تاماقخاننىڭ ئورنىنى تاس قاپتىمەن دىسە،،ئۇ بولسىمۇ ۋىلايەتلىك سىفەن مەكتەپنىڭ كەينى يولىدىكى ئوتكۇر چەيزە لەغمەنخانىسى،بۇ يەردە ھەممە تاماقلار بار بەك ئوخشايدۇ،دوستلىرىڭىز بىلەن چاي ئىچكەچ مۇڭداشماقچى بولسىڭىز ئارام چايخانىسى،دىپلۇمات چايخانىسىغا بېرىڭ،ئىسىل مۇھىتتا ،يارۇپاچە، مىللىچە تاماقلاردىن تىتىماقچى بولسىڭز شەھەرلىك ئەمگەك كاپالەت ئىدارىسى ئالدىدىكى كىرۇرەن گە كىرىڭ ،يەنە پىيادىلەر كۇچىسىنىڭ كەينى تەرپىدىكى رىۋايەت ئاشخانسىمۇ تاماقلىرىنىڭ تۇرى ،تەمى ،مۇھىتى جەھەتتە خوتەن بۇيىچە ئالدىنقى ئورۇندا تۇرىدۇ، بۇ ئورۇنلارنى خوتەندە ھەرقانداق تاكسىغا چىقىپ دىسڭىزلا ئاپىرىپ قويالايدۇ،تاكسىغا چىقسىڭز كۇندۇزى باشلىنىش باھاسى 5 يۇەن،كىچىسى 7يۇەن،باشلىنىپ ماڭىدۇ،ئۇندىن باشقا توخۇ قورىمىسى، ئوپكە-ھېسىپ،،،،،،چالا قالغانلارنى بىلىدىغان كوپچىلىك تۇلۇقلاپ قويارسىلەر،ئەگەر ياز پەسىلدە كەلسىڭىز خوتەننىڭ چۇققۇ بازارغا بېرىپ قادا ئىچىشنى ئۇنۇتماڭ،
2.ساياھەت جەھەتتە لايىقا يېزىسىنىڭ يازلىق مەنزىرلىرىنى ياڭاق پادىشاھى،ئەنجۇر پادىشاھى، خوتەننىڭ ئالاھىدە نۇقتىلىق ساياھەت ئورۇنلىرى قاتارىدا تۇرىدۇ،ئولۇغ ئاتا مەنزىرە رايۇنى،كوھمارىم سەيلىسى،ئىمامى ئاسىم مازىرى سەيلىسى،قولايلىق بولسا ئۇجمە بازار مازىرى سەيلىسى(مازار ماشايىق تاۋاپ قىلىڭلار دىمىدىم جۇمۇ) يەكشەنبە كۇنىدىكى خوتەن شەھرىنىڭ مەركىزى سودا بازىرى بولغان چۇققۇ بازارغا كىرسىڭىز پۇتۇنلەي يەرلىك ساپ ئۇيغۇرۇمنى كۇرەلەيسىز،شۇنداقلا نىمە ئىزدىسىڭىز تاپالايدىغان بۇ بازارنىڭ چوڭ ئىكەنلىكىگە چوقۇم ھەيران قالىسىز،يۇرۇڭقاش دەريا بويىغا چىقىپ بۇ يەردىكى قاشتىشى بازىرىنى كورگەندە بولسا سىزمۇ ھەم بۇ دەرياغا كىرىپ قاشتىشى ئىزدىگۇڭىز كىلىپ كىتىشى مۇمكىن،ئەمما خۇدايىم بەرمىسە بۇ تاشلارمۇ ئۇنداق ئاسانلا چىقمايدۇ،يۇرۇڭقاش دەرياسىغا چىققانكەنسىز ئەلۋەتتە يۇرۇڭقاش يېزىسىغا (ھازىر خوتەن شەھرىگە قوشۇلۇپ شەھەرنىڭ بىر قىسمى بۇلۇپ كەتتى)چىقىپ بۇ يەرنىڭ نامى مەشھۇر يۇرۇڭقاش گوشگىردىسىنى يىمەي كىرىپ كەتمەڭ،ئوتۇن قالاپ شور تونۇردا پىشۇرۇلغان ئەركەك پاخلاننىڭ گۇشى بىلەن پىشقان بۇ گوشگىردىنى يىىمىسىڭىز ئارماندا قالىسىز،
3.مىھمانخانا جەھەتتە خوتەن گۇجان دەرۋازىسىدىكى تەشنا مىھمانخانىسى قالايمىقان ئىشلار يوق ،شۇنداقلا ئەر –ئايالنىمۇ توي خەت بولمىسا كىرگۇزمەيدىغان پاكىز مىھمانخانا ھىساپلىنىدۇ،ئەتراپىدا ئاشخانا تاللا بازىرى قاتارلىقلار بار،چوڭ بىكەتنىڭ ئالدىدىكى ئاستانە مىھمانخانىسى،،،،،بەزى مىھمانخانلار ئىسىمدە قالماپتۇ،كوپچىلىك تۇلۇقلاپ قوياسىلەر،
4.كىيىم-كىچەك سېتۋالماقچى بولسىڭىز گۇلباغ سودا مەركىزى،چوڭ بازار سودا مەركىزى،ئالتۇن يۇلتۇز سودا مەركىزى،قاتارلىق چوڭ سودا مەركەزلىرى بار،
5.مىۋە –چىۋە ماكانى بولغان خوتەندە ھەقىقى خوتەندىن چىققان مىۋىلەر نىڭ تەمى ئوزگىچە بۇلۇپ باھاسىمۇ سىز ئويلاپ باقمىغان دەرىجىدە قىممەترەك،8.ئاي مەزگىلدە ئەنجۇرنىڭ پىشقان ۋاختى ،يەنە كىلىپ لايىقا يېزىسىنىڭ ئەنجۇرى بەكمۇ يىيىشلىك باھاسى بىر تالنىڭ 3 يۇەندىن 6 يۇەنگىچە ،قىممەتلىكى ئەرزىيىدۇ،خوتەننىڭ يەرلىك شاپتۇلىمۇ 9.ئاي مەزگىلدە پىشىدىغان بۇلۇپ ھەر كىلوسى 20 يۇەندىن 50 يۇەنگىچە،خوتەن ئانىرىنىڭ بىر كىلوسىنىڭ ئۇرۇمچىدە 20يۇەن ئىكەنلىكىنى كۇرۇپ يەپ باقسام پەقەتلا ئوخشىمايدىكەن،ئوتكەن يىل 12.ئايدا ئانارنىڭ كېلوسىنى 65يۇەندىن ئالغان ئىدۇق خوتەندىن،شىمالدىكى تۇققانلارغا ئەۋەتكىلى ،شىمالدىكى بەزى دوستلارغا باھاسىنى دىسەم پەقەتلا ئىشەنمەي قويدى،ئاشخانلاردىكى تاماقلارنىڭ باھاسى ،كىيىم-كىچەك باھاسى جەھەتتە خوتەننىڭ باھا ئۇستۇنلۇكى شىنجاڭ بۇيىچە ئەڭ يۇقۇرى ئورۇندا تۇرىدۇ،(بەزى قىممەتچىلىكلەرنى كەلتۇرۇپ چىقارغىنى يەنىلا شۇ قاشتىشى،ۋە تاش سودىگەرلىرى) ئەمما يەنىلا مىھرى ئىسىىق بۇ ماكان قانداقلا بولمىسۇن بىر كۇرۇپ ھەپتە ئويناپ كىتىشىڭىزغا ئەرزىيىدۇ، چالا قالغان قىسىملارنى بىلىدىغان تورداشلار قالدۇرۇپ قويۇڭلار،
