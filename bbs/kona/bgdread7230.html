ئۇيغۇرنىڭ پاسپورت ھوقۇقى توغرىلىق مۇھاكىمە



ئەسسالامۇ ئەلەيكۇم ئەزىز قېرىنداشلار،ئۇلۇغ ۋەتىنىمىز جۇڭخۇا خەلق جۇمھۇرىيىتى تەۋەلىگىدە ياشاۋاتقان 56 مىللەت ، خۇددى بىر دەستە خۇشپۇراق گۇلدەك بىر تۇتقىغا باغلىنىپ،جاھان ئەھلىنىڭ كوزلىرىنى قاماشتۇرماقتا.بۇنىڭدىن بىز ئوزىمىزنى كاتتا بەختيار ھېساپلايمىز،ئەلبەتتە!
لېكىن،مۇكەممەللىك ئىنسانىيەتكە خاس بولمىغاچقىمىكىن،بەزىبىر يېتىشسىزلىكلەرنىڭ ھېلىھەم مەۋجۇت بولۇپ تۇرىشى،قانۇن ئادالىتىگە تەشنا،سىياسەت مەھكۇملىگىدىن بىزار ئىنسانلارنى ئويلاندۇرماي قالمايدىكەن!
بۇنىڭ تىپىك مىساللىرىدىن بىرى ، ئاپتۇنۇم رايۇنىمىز تەۋەلىگىدە ياشىغۇچى ئاز سانلىق مىللەتلەرنىڭ پاسپورت بېجىرىشىنىڭ تەس بولۇش مەسىلىسى بولسا كېرەك!
پاسپورتنىڭ قانداق ئەھمىيىتكە  ئىگە ئىكەنلىگىنى ھەممىمىز بىلىمىز.ئوخشاشلا چەتئەلگە چىقىشتىن مەقسەتمۇ،ئوزىمىزنىڭ ماددى،مەنىۋى ۋە ھېسسى بايلىقىمىزنى ئاشۇرۇشتىن ئىبارەتتۇر.لېكىن،پاسپورت بېجىرىشنىڭ تەس بولىشى بىلەن،بىز بۇ ھوقۇقلاردىن مەھرۇم بولۇش بىلەنلا قالماي،يەنە،باشقا ئولكە ۋە بىۋاستە قاراشلىق شەھەرلەردىكى ۋەتەنداشلىرىمىزدىن تەڭداش قانۇنى ھوقۇق جەھەتلەردىمۇ خېلىلا كەينىدە قالدۇق.ئەسلىدە دولەت چېگرىدىن چىقىشنى چەكلىگەن ئىنسانلاردىن باشقىلارنىڭ چەكلىمىگە ئۇچرىشى خاتا ئىدى.ئەپسۇس،ھازىر پۇتكۇل ئاپتۇنۇم رايۇنىمىزدىكى جىمى ئاز سانلىق مىللەت پۇقرالىرى بۇ خىل مۇئامىلىگە ئۇچراۋاتىدۇ.بۇنىڭ قانۇنى جاۋاپكارلىقى زادى كىمنىڭ زىممىسىدە؟بۇ ئارقىلىق ئېرىشىلمەكچى بولغان نىشان "ھالقىما تەرەققىيات ۋە ئەبەدى ئەمىنلىك"مۇ  ياكى بىزگە نامەلۇم بولمىغان ئۇلۇغۋار كەلگۇسى نىشانمۇ؟ نىشان ھالقىما تەرەققىيات بولسا،ئۇنداقتا بىز ئاز سانلىق مىللەتلەر تەرەققى قىلمىساق ھالقىما تەرەققىيات ئەمەلگە قانداق ئاشىدۇ؟
ئوقۇرمەنگە ئايان بولغىنىدەك،ئالدىمىزدىكى قۇتلۇق كۇنلەردە پۇتۇن دۇنيا خەلقلىرى تەلمۇرۇپ كۇز سېلىۋاتقان كاتتا مەرىكە___" ئىككى يېغىن" پايتەختىمىز بېيجىڭدا ئېچىلىش ئالدىدا تۇرماقتا!مۇشۇ ھاياجانلىق مېنۇتلاردا ،مەنمۇ پۇتۇن مەملىكەت خەلقى بىلەن بىرگە چەكسىز ئىپتىخارلىق ۋە تەڭدشسىز ئىلتىجا ئىلكىدە يۇرىمەن!!!مەن ئوزۇم  سوز قىلىش ، ھاكىمىيەت ئىشلىرىغا قاتنىشىش سەھنەم بولغان تور سەھىپىىدىن دەس تۇرۇپ شۇنداق خىتاپ قىلىمەنكى،"ئككى يېغىن" غا قەدەم تەشرىپ قىلىش،قىممەتلىك ئالى رەھبەرلەرنى ئوز كۇزى بىلەن كۇرۇش ئىمتىىيازىغا سازاۋەر بولغان ھۇرمەتلىك ئاز سانلىق مىللەت ۋەكىللىرى،سىلەر ئاۋال پارتىيە،ئاندىن خەلق سىلەرگە ئىشىنىپ تاپشۇرغان بىباھا ھوقۇقنى ھەقىقى ئىشلىتىپ،مۇشۇ قېتىملىق "ئككى يېغىن" دا،ئاپتۇنۇم رايۇنىمىزدىكى ئاز سانلىق مىللەتلەرنىڭ پاسپورت ھوقۇقىنىڭ ئالاھىدە زەئىپ ئىكەنلىگىنى ئالاھىدە ئوتتۇرىغا قويساڭلار!؟بىزنىڭمۇ پەقەت كىملىك ۋە نۇپۇس دەپتىرىنى كۇرسىتىپلا، پاسپورت بېجىرگىمىز بارلىغىنى سۇيۇملۇك رەھبەرلىرىمىزگە ئاڭلاتساڭلار!!؟ 
يوقۇرقىلار مېنىڭ شەخسى كوز قارىشىم بولسىمۇ،لېكىن،بۇنى قوللايدىغان ۋىجدانلىقلارنىڭمۇ نۇرغۇن ئىكەنلىگىگە ئىشەنچىم كامىل!تورداشلارنىڭ مۇشۇ توغرىسىدا مۇلاھىزە يۇرگۇزىشىگە تىلەكداشمەن!
ئاخىرىدا،ھەممىڭلارنىڭ پاسپورتلىق بولىشىڭلارنى تىلەيمەن!!!

خەت چوڭلىقى : كىچىك نورمال چوڭ

 

ئەسسالامۇ ئەلەيكۇم ئەزىز تورداشلار،بىزدە ئاتىلىرىمىزدىن قالغان «كوپنىڭ ئەقلى كوپ»دېگەن ماقال بار.شۇڭا ، بۇ ئىشقا ئىمكان بار كوپرەك ئاۋاز قاتناشتۇرساق؟
مېنىڭ شەخسەن تەكلىپىم شۇكى،بىزنىڭ تەلىپىمىز ھەرگىزمۇ چەكتىن ئاشقانلىق ئەمەس،قانۇنسىزلىق تېخىمۇ ئەمەس!!!چۇنكى،بىز جۇمھۇرىيەت پۇقراسى بولۇش سالاھىيىتىمىز بىلەن،ئالدى بىلەن قانۇنغا بويسۇنۇش مەجبۇرىيىتىمىزنى تۇلۇق ئادا قىلغان شەرت ئاساسىدا،ئوزىمىزنىڭ قانۇندا بەلگىلەنگەن ھوقۇقىمىزنىمۇ ئاڭلىقلىق ۋە مەسئۇلىيەتچانلىق بىلەن قوغداش ھوقۇقىمىز بار.بۇنداق قىلمىساق قانداقمۇ دۇنياۋى كۈچلۈك دولەتنىڭ پۇقراسىلىق ئىمتىيازىغا يۈز كېلەلەيمىز؟
ئاپتۇنۇم رايۇنىمىزدائاز سانلىق مىللەتلەرنىڭ  پاسپورت بېجىرىشنىڭ تەس بولىشى ئوبىكتىپ ئەھۋال بولۇپلا قالماي،جەمىيەتتىكى قىزىق نوقتا مەسىلىسى ۋەئاپتۇنۇم رايۇن نوپۇسىنىڭ يېرىمىنىڭ ھوقۇقىنىڭ دەخلى _تەرۇزغا ئۇچراش مەسىلىسى!!!قانچىلىغان ئاقكوڭۇل ئنسانلارنىڭ پاسپورت بېجىرىمەن،دەپ،ئومۇربويى يىققان مال-دۇنياسىنى بىر قىسىم نامەرتلەرگە بېرىپ قويۇپ،قان قۇسۇپ ئارمان بىلەن بۇ دۇنيادىن كەتكەنلىگىنىمۇ ئۇنتۇپ قالمايلى!شۇ بىچارىلەر كىملەر؟ئۇلارنىڭ گۇناھى نېمە؟ قويمىچىلارغا پۇرسەت،شارائىت يارىتىپ بەرگەن يەرلىك سىياسەتنىڭ مەقسىدى زادى نېمە؟ئاپتۇنۇم  رايۇنىمىزنىڭ «ئاپتۇنومىيە»سى ،ئەجىبا،باشقا ئولكىلەرنىڭ،يەنە كېلىپ بىڭتۈەننىڭ «ئاپتۇنومىيە»سىگىمۇ يەتمەمدۇ؟...
تورداشلاردىن ئومىدىم،مۇمكىن بولسا ئۇيغۇر يېزىقىدىكى «جۇڭخۇا خەلق جۇمھۇرىيىتىنىڭ پاسپورت قانۇنى»نى ئىنكاس شەكلىدە يوللاپ قويغان بولساڭلار؟!
تورداشلارنىڭ تېنىگە سالامەتلىك  تىلەيمەن!!!
