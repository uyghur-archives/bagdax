 ماۋزېدۇڭ لىن بياۋنىڭ سىياسى ئۆزگىرىش قىلىش سۈيقەستىنى تارمار قىلغان كۈنلەردە

          دۆلەتنىڭ سابىق مۇئاۋىن رەئىسى ۋاڭ دۇڭشڭنىڭ 1971 - يىلى 9 - ئايدا ماۋجۇشىنىڭ لىن بياۋ ئەكسىلئىنقىلابى گورۇھىنىڭ سىياسى ئۆزگىرىش قىلىش سۇيقەستىنى تار - مار قىلىش جەريانىدىكى ئۆز بېشىدىن ئۆتكۈزگەن كەچۈرمىشلىرى توغرىسىدىكى ئەسلىمىسىدە ، لىن بياۋ ئەكسىلئىنقىلابى گورۇھىنىڭ سىياسى ئۆزگىرىش قىلىش ، ماۋزېدۇڭنى قەستلەپ ئۆلتۈرۈش سۇيقەستىنى ناھايىتى تەپسىلى يازغانىكەن .  لېكىن ماقالە ئۇزۇنراق بولغاچقا ،قىسقارتىپ يوللۇدۇم :
             . . . . . . .      
            1971- يىلى 8 - ئاينىڭ 15 - كۈنى باشلاپ ماۋجۇشىغا ھەمرا بولۇپ جەنۇبقا خىزمەت تەكشۈرۈشكە چىقتۇق . خۇنەن ، جياڭسۇ ، گۇاڭدۇڭلاردا ئۆلكە مەسئۇللىرى ۋە ھەربىي رايۇن مەسئۇللىرى بىلەن كۆپ قېتىم سۆھبەتلەشتى . ماۋجۇشى يول بويىدىكى سۆھبەتلەردە باشتىن ئاياق : " ماركسىزم بىلەن شوغۇللىنىش ، شيۇجىڭجۇيى بىلەن شوغۇللانماسلىق ؛ ئىتتىپاق بولۇش ، بۆلگۈنچىلىك قىلماسلىق ؛ ئوچۇق - ئاشكارە بولۇش ، ھېيلە - مىكىر ئىشلەتمەسلىك كېرەك " دېگەننى تەكىتلەپ كەلدى . ئۇ : 1970 - يىلدىكى لۇشەن يىغىنىدا ، ئۇلار تۇيۇقسىز ھۇجۇم قىلدى . مەخپي ھەرىكەتلەرنى ئېلىپ باردى . ئۇلار نېمىشقا ئوچۇق - ئاشكارە ئىش قىلالمايدۇ ؟ ئۇلارنىڭ قۇسىقىدا جىن بار . ئۇلار دەسلىۋىدە ئىشنى يوشۇرۇپ ، كېيىن تۇيۇقسىز ھۇجۇم قىلىشتى . بۇ ئىشلارنى دائىمىي كومېتىتنىڭ بەش ئەزاسىنىڭ ئۈچىدىن ۋە سىياسى بىيۇرودىكى ئاشۇ بىر قانچە گىنرالدىن باشقا يولداشلاردىن يوشۇرۇنچە ئېلىپ باردى . بۇ گىنراللار خۇاڭ يۇڭشىڭ ، ۋۇفاشيەن ، يې چۈن ، لى زوپېڭ ، چيۇخۇيزولاردىن ئىبارەت . ئۇلارنىڭ بۇ ئىشلارنى قىلىشىدا چوقۇم بىر مەقسەت بار _ دە ! مېنىڭچە ئۇلارنىڭ تۇيۇقسىز ھۇجۇم قىلىش ، يوشۇرۇن ھەرىكەتلەرنى ئېلىپ بېرىشى پىلانلىق ، تەشكىللىك ، پروگراممىسى بولغان ھالدا ئېلىپ بېرىلغان . ئۇلارنىڭ پروگراممىسى دۆلەت رەئىسى قويۇش ، " تالانت ئىگىسى " دەپ ئاتاش . بەزىلەر دۆلەت رەئىسى بولۇشقا ، پارتىيىنى پارچىلاپ ، ھوقۇقنى تارتىۋېلىشقا ئالدىراپ كىتىۋاتىدۇ . لىن بياۋ ھېلىقى نۇتۇقنى سۆزلەشتىن ئىلگىرى مەن بىلەن مەسلىھەتلەشمىدى ، ماڭىمۇ كۆرسەتمىدى ، ئۇلار خۇددى قانداقتۇر  بىر ئىشەنچىسى باردەك ، خۇددى مۇۋەپپەقىيەت قازىنىدىغاندەك ، گېپى بولسا ئاۋۋال ئوتتۇرىغا قويمىدى . لېكىن ئۇلارنىڭ سۆزى بولماي قېلىۋېدى ، ھۇدۇقۇپ تەمتىرەپ قېلىشتى . بۇ قېتىمقى يىغىندا چېن بودانىڭلا مەسىلىسى ئوتتۇرىغا قويۇلدى . مۇئاۋىن رەئىس لىن بياۋ قوغداپ قېلىندى . ئۇنىڭ ئۈستىدە خۇلاسە چىقىرىلمىدى .  . . . . . .     
         9 - ئاينىڭ 3 - كۈنى ماۋجۇشى خاڭجۇغا يېتىپ كەلدى  . . . ماۋجۇشى لۇشەن يىغىنىدىكى مەسىلە ئاساسەن ئىككى ، بىرى ، دۆلەت رەئىسى قويۇش ، يەنە بىرى ، " تالانت ئىگىسى " دەپ ئاتاش مەسىلىسى . تالانت ئىگىسى دېيىشكە قارشى تۇرغانلىق ماڭا قارشى تۇرغانلىق دېيىلدى . ھېلىقى بىر نەچچە سۈپەتلەشنى مەن نەچچە قېتىم سىزىپ چىقىرىۋەتكەن ، دېدى . ماۋجۇشى يەنە : لۇشەندىكى بۇ ئىش تېخى تۈگىمىدى ، تەلتۈكۈس ھەل بولمىدى ، تېخى يەكۈن چىقىرىلمىدى ، - دېدى .       
        خۇددى ماۋجۇشىنىڭ پەرەز قىلغىنىدەكلا ، لۇشەن يىغىنىدا مەغلۇبيەتكە ئۇچىرىغان لىن بياۋ ۋە ئۇنىڭ شېرىكلىرى نىيىتىدىن يېنىش ئۇياقتا تۇرسۇن ، ئەكسىچە ، ماۋجۇشىغا زىيانكەشلىك قىلىش ، ئەكسىلئىنقىلابى قۇراللىق سىياسى ئۆزگىرىش قىلىشتىن ئىبارەت سۈيقەستلىك ھەرىكەتلەرنى باشلىدى .      
             . . . . . .      
          ماۋجۇشى بۇ قېتىم جەنۇبتا خىزمەت تەكشۈرگەن جەرياندا ، لىن بياۋ ۋە ئۇنىڭ شېرىكلىرى ئامالنىڭ بېرىچە ماۋجۇشىنىڭ يۈرۈش - تۇرۇشى ،ۋە ئۇنىڭ يول بويىدىكى سۆھبەتلىرىنىڭ مەزمۇنىنى تىڭ - تىڭلىغان . خۇاڭ يۇڭشىڭ 9 - ئاينىڭ 6 - كۈنى بېيدەيخېدىكى لىن بياۋ ، يې چۈنلەرگە دوكلات قىلغان . ئۇلار ئۆزلىرىنىڭ ئاشكارلىنىپ قالغانلىقىنى ھېس قىلىپ ، تەۋەككۈلچىلىك يولىغا مېڭىپ ، سەپەر ئۈستىدىكى ماۋجۇشىنى قەستلەپ ئۆلتۈرۈش ھەرىكەتلىرىدە بولغان . 
        9 - ئاينىڭ 7 - كۈنى لىن بياۋ لىن لىگوغا " بىرلەشمە فلوت " قا "بىرىنچى دەرىجىلىك "ئۇرۇش تەييارلىقى" ھالىتىگە كىرىشكە بۇيرۇق بېرىش توغرىسىدا يوليۇرۇق بەرگەن .       9 - ئاينىڭ 8 - كۈنى لىن بياۋ : " يولداش لىن لىگو ، يۈيچى يولداشلار يەتكۈزگەن بۇيرۇق بويىچە ئىش كۆرۈشىلەرنى ئۈمىد قىلىمەن " دېگەن بۇيرۇقنى يازغان .  
       شۇنىڭ بىلەن خەتەر ماۋجۇشىغا ھەر ۋاقىت يېقىنلاپ كەلمەكتە ئىدى . شۇ چاغدا خاڭجۇ گارنزوننىڭ چوڭ ھوقۇقى چېن لىيۈننىڭ قولىدا بولۇپ ، ئۇ ماۋجۇشى تۇرىۋاتقان ئورۇننىڭ مۇھاپىزەت خىزمىتىگە بىۋاستە قۇماندانلىق قىلالايتتى . بىزنىڭ خاڭجۇدا تۇرۇشىمىز يولۋاس ئوۋىسىغا كىرگىنىمىز بىلەن ئوخشاش ئىدى . . . . . . . ماۋجۇشى لىن بياۋ قاتارلىقلارنىڭ سۈيقەستىگە قارىتا دەرھال كەسكىن تەدبىرلەرنى قوللاندى ، ئۇ مېنى چاقىرىتىپ مەخسۇس پويىزنى يۆتكىۋېتىشنى ئېيتتى . . . .
       پويىز خاڭجۇدىن ئايرىلدى .       
        . . . . . . ماۋجۇشى : شاڭخەي شەھەر ئەتىراپىدىكى خۇڭچاۋ ئايرودورومىنىڭ مەخسۇس لىنىيىسىدە توختايلى ، گۇجيا باغچىسىغا كېرمەيلى ، - دېدى .       

    . . . . . . .
          « بىرلەشمە فلوت » ئەزالىرىنىڭ كېيىنكى ئىقرارى ۋە بىز ئىگەللىگەن ئەھۋاللارغا قارىغاندا ، ئۇلار ماۋجۇشىنى كۆپ خىل ئۇسۇللاربىلەن قەستلىمەكچى بولغان :
          بىرىنچى خىل ئۇسۇل : مۇبادا مەخسۇس پويىز شاڭخەي خۇڭچياۋ ئايرودورومنىڭ مەخسۇس لىنىيىسىدە  توختىسا ، جەنۇبى لىنىيىنىڭ قۇماندانلىقىغا مەسئۇل بولغان جاڭ تېڭجياۋ مەخسۇس لىنىيەنىڭ يېنىدىكى ئايرودورومنىڭ ماي ئىسكىلاتىنى پارتىلىتىۋېتىش  ياكى ئوت قويۇشقا قۇماندانلىق قىلىش . ۋاڭ ۋېيگونىڭ ئىقرار قىلىشىچە ، ئۇلار ۋاڭ ۋېيگوغا ئوت ئۈچۈرۈش نامى بىلەن « تەلىم - تەربىيە ئەتىرىتى» نى باشلاپ پويىزغا چىقىپ ، قالايماقانچىلىقتىن پايدىلىنىپ ، ئاۋۋال ۋاڭ دۇڭشىڭنى ئۆلتۈرۈپ ، ئاندىن ماۋجۇشىغا قەست قىلىش ياكى تۇتۇپ كېتىشنى ئورۇنلاشتۇرغان .
         ئىككىنچى ئۇسۇل : ئۇلار بۇنى بىرىنچى ئۇسۇل مەغلۇپ بولغاندىن كېيىن قوللانماقچى بولغان . ئۇلار ماۋجۇشىنىڭ مەخسۇس پويىزى شوفاڭ تۆمۈر يول كۆرۈكىدىن ئۆتىۋاتقاندا كۆۋرۈك ۋە مەخسۇس پويىزنى پارتىلىتىپ ، ئىككىنچى «خۇاڭگۇتۇن » ۋاقەسىنى پەيدا قىلىپ ، ئاندىن بۇنى بۇزۇق ئادەملەر قىلدى دەپ ئېلان قىلماقچى بولغان . شوفاڭ كۆۋرۈكى سۇجۇ بىلەن ۋۇشىنىڭ ئارىلىقىدا بولۇپ ، ئۇلار ئۇ يەرگە بېرىپ ، يەر شەكلىنى كۆزەتكەن ، ھەتتا پارتىلاتقۇچ بومبىنى قانداق ئورۇنلاشتۇرۇشنىمۇ ئۆلچەپ لاھىيلىگەن .
         ئۈچىنچى ئۇسۇل :  ئەگەر پارتىلىتىش مۇمكىن بولماي قالسا ، ئۇلار يول بويىدا ئوت چېچىش قۇرالى بىلەن پويىزنى كۆيدۈرمەكچى بولغان . جۇ  يۈچىنىڭ ئېيتىشىچە ، ئوت چېچىش قۇرالى نەچچە سۇڭ قېلىنلىقتىكى پولات تاختىنىمۇ تىشىۋېتەلەيدىكەن . ئۇنىڭ بىلەن پويىزغا ئوت چاچسا پويىزمۇ ، ئادەممۇ بىر دەمدىلا كۈيۈپ تۈگەيدىكەن . ۋاڭ ۋېيگو ، جۇ يۈچىلار تۆمۈر يول بويىغا بېرىپ يەر شەكلىنى كۆزەتكەن . ئۇلار تېخى باشقا جايلاردىن ئوت چېچىش قىسىملىرىنى يۆتكەپ كەلمەكچى بولغان . لېكىن بىز مۇددەتتىن بۇرۇن مېڭىپ كەتكەچكە ئۇلار بۇ قىسىمنى يۆتكەچكە ئۈلگۈرەلمىگەن .
         تۆتىنچى ئۇسۇل :  ئۆزگەرتگەن ئىل - 10 تىپلىق ئايرۇپلان بىلەن خاڭجۇدا ماۋجۇشىنىڭ مەخسۇس پويىزىنى پارتىلىتىش ، ئايرۇپلانغا بومبا قاچىلاشقا چېن لىيۈن ئۆزى مەسئۇل بولۇش . كېيىن چېن لىيۈننىڭ ئىقرار قىلىشىچە ،  يۈ شىنيې ئۇنىڭغا ۋەزىپە تاپشۇرغاندا ، ئۇ خاڭجۇدا ئىشەنچلىكرەك ئۇچقۇچى يوقلىقىنى ئېيىتقان . يۈ شىنيې قايتىپ بارغاندىن كېيىن بۇنى يوقىرىغا دوكلات قىلىپ ،  ئۇچقۇچى ئېۋەتىشكە ماقۇل بولغان . ئۇلار زادى كىمنى ئەۋەتمەكچى ؟ ئۇلار لۇمىڭنى ئەتمەكچى بولغان ، لۇمىڭ شۇ چاغدا ھاۋا ئارمىيىسى قۇماندانلىق شتابى ئۇرۇش قىلىش بۆلىمىنىڭ باشلىقى ئىدى . چېن لىيۈن : بولىدۇ ، شۇنداق قىلايلى ، ئايرۇپلان بىلەن مەخسۇس پويىزنى پارتىلىتىش ئىشەنچلىك بولىدۇ ، دېگەن . 9 - ئاينىڭ 9 - كۈنى يۈ شىيې شاڭخەيدە ۋاڭ ۋېيگوغا مۇنداق دېگەن : بۇ قېتىمقى ئايرۇپلان بىلەن پويىزنى پارتىلىتىشتا ، ئايرۇپلاندىكى قۇراللاردىن باشقا ، يەنە پويىزدىن قېچىپ چىققانلارنى ئېتىش ئۈچۈن زېنىت توپ تەييارلاش كېرەك ، _ دېگەن .

       يۈشىنيې ۋاڭ ۋېيگو بىلەن بۇلارنى پىلانلاۋاتقاندا ، ۋاڭ ۋېيگو مۇبادا ماۋجۇشى مەخسۇس پويىزدا تۇرماي ، گۇجيا باغچىسىدا تۇرسا قانداق قىلىمىز دېگەننى ئوتتۇرىغا قويغان . يۈشىنيې : مەن يەر شەكلىنى كۆردۈم ، ئەگەر ماۋجۇشى گۇجيا باغچىسىغا چۈشسە ، ئاۋۋال ۋاڭ ۋېيگونىڭ « تەلىم - تەربىيە ئەتىرىتى» نى باغچا ئەتىراپىغا موكتۈرۈپ ، پېلىموت بىلەن ئالدى - كەينىدىكى يوللارنى قامال قىلىپ ، ئاۋۋال مۇھاپىزەت قىسىملىرىنى ئۇجۇقتۇرىۋېتىپ ، ئاندىن باستۇرۇپ كىرىمىز ، - دېگەن .
         . . . . . .          
        ۋاڭ فېي شۇ چاغدا ھاۋا ئارمىيىسى قۇماندانلىق شتابىنىڭ مۇئاۋىن شتاب باشلىقى ، « بىرلەشمە فلوت » نىڭ غوللۇق ئەزاسى ئىدى . ئۇلارنىڭ شىمالى سەپتىكى سۈيقەستلىك ھەرىكىتى :  بېيجىڭدا جۇئېنلەي ، جۇدې ، يې جيەنيىڭ ، نيې رۇڭجىن ، شۇي شياڭچيەن ،ليۇبوچىڭ قاتارلىق مارشاللارنى ھەم جياڭ چىڭ ، جاڭ چۈنچياۋ ، ياۋۋېنيۇەنلەرنى ئۆلتۈرۈش ئىدى . ۋاڭ فېي قاتارلىقلار دياۋيۈتەي ، جۇڭنەنخەيلەرنىڭ يەر شەكلىنى كۆزىتىپ چىققان . جۇئېىنلەي شۇ چاغدا جۇڭنەنخەيدە تۇرىۋاتاتتى . ئۇلار جۇڭنەنخەيگە تانكا بىلەن باستۇرۇپ كىرمەكچى بولغان . ۋاڭ فېي : بېيجىڭنىڭ ھاۋا بوشلىقىدا ئۇچۇش چەكلەنگەن ، جۇڭنەنخەينىڭ قۇرۇق تېمىنى تانكا بىلەن ئۆرىۋەتكىلى بولىدۇ ، _ دېگەن . بەزىلەر تېخى جۇڭنەنخەيگە باشقۇرۇلىدىغان بومبا ئېتىش تەكلىپىنىمۇ بەرگەن . ئۇلار تالاش - تارتىش قىلىپ مۇۋاپىقراق بىرەر ئامال تاپالمىغان .
        . . . . . . .
         ھازىر ئويلىسام ، شۇ چاغدىكى ۋەزىيەت ئىنتايىن خەتەرلىك ئىكەن . لېكىن ماۋجۇشى ئۆزى ئىگەللىگەن خەتەرلىك ئەھۋاللارنى ماڭا دېمىگەنكەن . ئۇ ئاتىمىز ئېغىر -بېسىقلىق بىلەن پۇرسەت كۈتكەنئىكەن . شۇ چاغدا مەنمۇ بەزى ئىشلارنىڭ نۇمالسىز ئىكەنلىكىنى سەزگەنىدىم ، شۇڭا خاڭجۇدا تۇرىۋېرىشقا بولمىغاچقا شاڭخەيگە كەتكەنىدۇق . بىزنىڭ ھەرىكىتىمىز بەك تېز بولغاچقا ، چېن لىيۈن ، ۋاڭ ۋېيگو نىمە قىلارىنى بىلەلمەي تەمتىرەپ قالغان . بىز 10 -كۈنى سائەت 15 تىن 35 مىنۇت ئۆتكەندە خاڭجۇدىن قوزغۇلۇپ ، 18 دىن 10 مىنۇت ئۆتكەندە شاڭخەيگە يېتىپ كەلدۇق . بۇ قېتىم ماۋجۇشىغا ھەمرا بولۇپ چىققاندا مەركىزى مۇھاپىزەتچىلەر پولكىنىڭ كادىرلار ئەتىرىتىدىن 100 ئادەم ئېلىۋالغان ئىدىم . ئالدىنقى مۇھاپىزەت پويىزى ۋە مەخسۇس پويىزغا بۇ مۇھاپىزەتچىلەر ئورۇنلاشتۇرۇلغان ئىدى . شاڭخەيگە كېلىپلا مەن يەرلىك مۇھاپىزەتچىلەر قىسمىنى تاشقى مۇداپىئەگە يۆتكەپ ، مەخسۇس پويىزنىڭ ئەتىراپىغا پۇتۇنلەي مەركىزى مۇھاپىزەتچىلەر پولكىنىڭ ئەسكەرلىرىنى ئورۇنلاشتۇردۇم . بىزدىن 150 مېتر نېرىدا خۇڭچياۋ ئايرودورومىنىڭ ماي ئىسكىلاتى بار بولۇپ ، ئۇنىڭغا ئوت كېتىدىغان بولسا مەخسۇس پويىزنىڭ بۇ يەردىن ساق چىقىپ كېتىشى مۇمكىن ئەمەس ئىدى . شۇڭا ئالاھىدە ئىككى مۇھاپىزەتچىنى شۇ يەرگە ئورۇنلاشتۇردۇم .. .. . . . .         
          بۇ ۋاقىتتا ، لىن لىگو قاتارلىقلار ماۋجۇشى ۋە مەركەزدىكى باشقا رەھبەرلەرلەرگە قەست قىلىش پىلانىنى تۈزەش ۋە ئىشقا ئاشۇرۇشنى جىددىلەشتۈرىۋاتاتتى . تۇيۇقسىز ۋاڭ ۋېيگونىڭ شاڭخەيدىن ماۋجۇشىنىڭ مەخسۇس پويىزى كەتتى دەپ بەرگەن تېلېفۇنىنى ئاپتۇ ، بۇ خەۋەر ئۇلارنى قاتتىق قورقۇتۇپتۇ . لىن لىگو ئارقا - ئارقىدىن ئۈزمەي : چاتاق بولدى ، چاتاق بولدى ، _ دەپتۇ .
       لىن لىگو ماۋجۇشىنىڭ 9 ئاينىڭ 12 - كۈنى چۈشتىن كېيىن ھەقىقەتەن جۇڭنەنخەيگە قايتىپ كەلگەنلىكىنى تىڭ - تىڭلاپ ئۇققاندىن كېيىن ، ماۋجۇشىغا قەست قىلىش سۈيقەستىنىڭ بەربات بولغانلىقىنى چوڭقۇر ھېس قىلغان . ئۇ قاتتىق دەككە - دۈككە ئىچىدە بېيدەيخېدىكى يې چۈنگە تېلېفۇن بېرىپ ، ئەھۋالنىڭ جىددىلىكىنى ئېيتقان . ئىككى سائەتتىن كېيىن ئۆزى بىۋاستە ئايرۇپلان بىلەن بېيدەيخېغا بارغان . مېڭىش ئالدىدا ئۆزى كەتكەندىن كېيىن بېيجىڭدىكى ئىشلارغا جۇ يۈيچى قۇماندانلىق قىلىدىغانلىقىنى ئېيتقان . لىن لىگو يەنە جۇيۈيچى قاتارلىقلارغا ھازىر ۋەزىيەت ئۆزگەردى ، دەرھال يۆتكىلىشىمىز كېرەك ، دەرھال يۆتكىلىشنىڭ ھەرىكەت پىلانىنى مۇزاكىرە قىلىشىمىز كېرەك ، دېگەن .
        ئۇلار نەگە يۆتكەلمەكچى ؟ ئۇلار «571 قۇرۇلىشى خاتىرىسى» دىكى پىلانى بويىچە  ماۋجۇشىغا قەست قىلىش سۈيقەستى ئەمەلگە ئاشمىسا ، گۇاڭجۇغا يۆتكىلىپ ، ئۇ يەردە باشقا مەركىزى ھۆكۆمەت قۇرۇپ ، دۆلەتنى پارچىلىماقچى بولغان . مانا بۇ لىن بياۋ ئەكسىلئىنقىلابى گۇرۇھىنى سوتلىغاندا ئېيتىغان ئىككى سۈيقەست :  بىرى ماۋجۇشىغا سۈيقەست قىلىش . ئىككىنچىسى ، خۇاڭ يۇڭشىڭ ، ۋۇ فاشيەن ، لى زوپېڭ ، چيۇخۇيزولارنى ئېلىپ گۇاڭجۇغا قېچىپ بېرىپ ، ئايرىم مەركىزى ھۆكۆمەت قۇرۇپ ، دۆلەتنى پارچىلاش سۈيقەستى ،      يۆتكىلىش ئۈچۈن ئۇلار گۇاڭجۇغا ئۇچىدىغانغا بەش ئايرۇپلان تەييارلىغانىدى :  بۇلاردىن بىرىنچىسى ، لىن بياۋنىڭ 256 - نومۇرلۇق « ئۈچ باشلىق نەيزە » تىپلىق ئايرۇپلانى ، ئىككىنچىسى ، خۇاڭ يۇڭشىڭ قاتارلىقلار ئولتۇرۇشقا ئورۇنلاشتۇرۇلغان « ئۈچ باشلىق نەيزە » تىپلىق ئايرۇپلان ، ئۈچىنچىسى ، ئىل _ 18 تىپلىق ئايرۇپلان ، تۆتىنچىسى ، ماشىنىلارنى قاچىلىغىلى بولىدىغان ئان _ 24 تىپلىق ترانسپوت ئايرۇپلانى ، بەشىنچىسى ،  ئان _ 24 تىپلىق ئايرۇپلان بولۇپ ، ئوق ئۆتمەيدىغان ئاپتۇمۇبىل قاچىلىغىلى بولاتتى . ئۇلار لىن بياۋ ئۈچۈن ئىل _ 18 تىپلىق ئايرۇپلاندىن يەنە بىرنى تەييارلاش كېرەك ، دېگەن . لىن لىگە بۇ ئورۇنلاشتۇرۇشلارنى تېلېفۇندا لىن بياۋغا دوكلات قىلغان ، لىن بياۋ : « دەرھال يۆتكىلەيلى » دېگەن . ئارقىدىنلا لىن لىگو ۋاڭ فېيغا : سىزنىڭ بۇ ئورۇنلاشتۇرۇشىڭىز ياخشى بوپتۇ . مۇئاۋىن رەئىس لىن بياۋ دەرھال گۇاڭجۇغا ئۇچۇشنى قارار قىلدى ، دەپ يەتكۈزگەن . دېگەن .
            . . . . . . .            
          كېيىنكى تەكشۈرۈشتىن قارىغاندا ، لىن بياۋ قاتارلىقلار ئاۋۋال سۆھبەت ئۆتكۈزۈپ ، ئاندىن قۇرال كۈچى ئىشلىتىش پىلانىنى تۈزگەن . ئۇلار گۇاڭجۇغا بارغاندىن كېيىن ، ئاۋۋال شەرت قويۇپ بېيجىڭ بىلەن سۆھبەت ئۆتكۈزمەكچى بولغان . لېكىن سۆھبەتنىڭ مۇۋەپپەقىيەتلىك بولۇش ئېھتىمالىنىڭ ناھايىتى ئاز بولىدىغانلىقىنى پەرەز قىلغان . شۇڭا ، ئۇلار گۇاڭجۇدا دىۋىزىيىدىن يوقىرى كادىرلار يىغىنىنى چاقىرىپ ، ئۇلارنى ھەرىكەتلەندۈرۈپ ، ئايرىم مەركىزى ھۆكۆمەت قۇرىدىغانلىقىنى جاكارلاشنى پىلانلىغان . قۇرال كۈچى ئىشلىتىشتە سوۋېت ئىتتىپاقى بىلەن بىرلىشىپ ، جەنۇب ۋە شىمالدىن قىستاپ ھۇجۇم قىلماقچى بولغان . لىن لىگو گۇاڭجۇ قىسىملىرى ھاۋا ئارمىيىسىنىڭ شتاب باشلىقى گۇتۇڭجۇغا ماشىنا ۋە تۇرالغۇ جاي تەييارلاشنى ئۇقتۇرۇشنى تەلەپ قىلغان . لىن لىگو يۈشىنيېگە : دەرھال شاڭخەيدىكى ۋاڭ ۋېيگوغا تېلېفۇن بېرىڭ ، 9 - ئاينىڭ 13- كۈنى ئەتتىگەندە بىر ئىل _ 18 تىپلىق ئايرۇپلان شاڭخىەي ئايرودورومغا قونىدۇ ، مۇھاپىزەتچىلەر پولىكىنىڭ ئىككىنچى ئوتتۇرا ئەترىتى ئالماشتۇرىۋېتىلسۇن ، ۋاڭ ۋېيگونىڭ ھاۋا ئارمىيىسى 4 - كورپۇسىنىڭ « تەلىم - تەربىيە ئەترىتى » بىلەن شاڭخەيدىكى « بىرلەشمە فلوت » ئەزالىرى تەييارلىقلىرىنى تەق قىلىپ شۇ ئايرۇپلان بىلەن شاڭخەيگە ئۇچسۇن ، _ دېگەن . لىن لىگو يەنە يۈشىنيېگە ھاۋا ئارمىيىسى ھەربىي لازىمەتلىكلەر بۆلۆمىگە تېلېفۇن بېرىپ ، دەرھال 30 تاپانچا ، ئىككى ئاپتۇمات ، كۆپرەك ئوق تەييارلاشنى ئۇقتۇرۇشنى ئېيتقان . بۇ ۋەزىپىلەر ئورۇنلانغاندىن كېيىن ، يۈشىنيې دەرھال ھاۋا ئارمىيىسى قۇرۇسىغا بېرىپ ، جاڭ تىڭجياۋ ، ۋاڭ فېيلارنىڭ كىشىلەرنى يۆتكىلىشىگە تەشكىللەشكە ياردەملەشكەن . جۇيۈچى قوللىرىنى سىلكىپ ، جاڭ تىڭجياۋ ، ۋاڭ فېيلارغا : ئاناڭنى ، غەلبە قىلىش - قىلالماسلىقىمىز مۇشۇ ئېلىشىشقا باغلىق ، _ دېگەن .
       12 - كۈنى كەچ سائەت سەككىزلەر ئەتىراپىدا ، جۇيۈچى ھاۋا ئارمىيسى ئىنىستوتىدا ، ۋاڭ فېي ، يۈشىنيېلارنى يىغىپ مەخپىي يىغىن ئۆتكۈزگەن . يىغىندا ۋاڭ فېي ، يۈشىنيېلارنىڭ كىشىلەرنى تەشكىللەپ لىن بياۋنىڭ جەنۇبقا قېچىشىغا مەسئۇل بولىدىغانلىقى بىكىتىگەن . ئۇلار 13 - كۈنى سەھەر سائەت سەككىزدە ، لىن بياۋ شەنخەيگۇەن ئايرودرومىدىن بىۋاستە گۇاڭجۇغا ئۇچۇش ،  13 - كۈنى سەھەر سائەت يەتتىدە خۇاڭ يۇڭشىڭ ، ۋۇفاشيەن ، لى زوپېڭ ، چيۇخۇيزو قاتارلىقلار بېيجىڭ شىجياۋ ئايرودرومىدىن ئۇدۇل گۇاڭجۇغا ئۇچۇشنى پىلانلىغان .

            بېيدەيخېنىڭ غەربىي دېڭىز ساھىلىدىن ئىككى كىلومېتر يىراقلىقتىكى ليەنفېڭشەن قارىغايلىقىدا ، ئىككى قەۋەتلىك كىچىك بىنا بولۇپ ، ، ئۇ لىن بياۋ ، يې چۈنلەر ئۆز ۋاقتىدا تۇرغان مەركىزى كومىتېت ساناتورىيىسىنىڭ 62 - نومۇرلۇق بىناسى ئىدى . 1971 - يىلى 9 - ئاينىڭ 12 - ئۈنى گۇگۇم چۈشۈشكە باشلىغاندا لىن بياۋ ، يې چۈنلەر يۆتكىلىش ئىشى بىلەن ئالدىراش ئىدى . لىن لىگونىڭ بېيدەيخېغا ئايرۇپلان بىلەن قايتىمەن دېگەن تېلېفۇنىنى ئالغاندىن كېيىن ، ھېيلە ئىشلىتىپ ، شۇ ئاخشىمى قىزى لىن لىخېڭ بىلەن ئۇنىڭ سۆيگىنى جاڭ چىڭلىننىڭ چېيىنى ئىچۈرۈش مۇراسىمى ئۆتكۈزىدىغانلىقىنى جاكارلىدى . بۇنداق قىلىشى ، ئېنىقكى ، خىزمەتچىلەرنىڭ دىققىتىنى باشقا ياققا بۇراپ ، ئۆزلىرىنىڭ سۈيقەستلىك ھەرىكەتلىرىنى نىقابلاش ئىدى . يې چۈن بىلەن لىن لىخېڭ ئوتتۇرىسىدا زىددىيەت بولغاچقا ھەممە ئىشنى ئۇنىڭدىن يوشۇراتتى . لىن لىخېڭ ئەقىللىق قىز بولۇپ ، ئۇ لىن لىگو قايتىپ كەلگەندىن كېيىن ، ئۆيدە ئۆزىدىن يوشۇرۇنچە بىر ئىشنىڭ يۈز بېرىۋاتقانلىقىنى ھېس قىلدى . ئۇ لىن بياۋ خانىسىنىڭ ئىشىك ئالدىغا كېلىپ تىڭ - تىڭلاپ ، لىن بياۋ ، يې چۈن ، لىن لىگولارنىڭ نەگىدۇر بىر يەرگە بېرىش توغرىسىدا ئاراڭلىشىۋاتقانلىقىنى ئاڭلاپ قالدى . ئۇ جىددىلىشىپ شۇ چاغدا بېيدەيخېدا لىن بياۋنىڭ ئامانلىقىنى قوغداۋاتقان 8341 - قىسىمنىڭ مۇئاۋىن پولك كوماندىرى جاڭ خۇڭ بىلەن ئىككىنچى ئەتىرەت باشلىقى جياڭ زۇشۇغا دوكلات قىلدى  .   .   .
        مەن بۇ ئەھۋالنى ئۇقۇپ ، لىن لىخېڭنىڭ ئېيتقانلىرىنى جۇزۇڭلىغا دوكلات قىلدىم .
        .   .   .   .   .   .   .   .
         كەچ سائەت 11 - يېرىمدا ، جۇزۇڭلى ئۆزى بىۋاستە يې چۈنگە تېلىفۇن بەردى  : مۇئاۋىن رەئىس لىن بياۋ ياخشى تۇرىۋاتامدۇ ؟ دەپ سۇرىدى .
         يې چۈن : مۇئاۋىن رەئىس لىن بياۋ ياخشى تۇرىۋاتىدۇ ، دېدى .
         جۇزۇڭلى بېيدەيخېدا بىر مەخسۇس ئايرۇپلان بارلىقىنى بىلدىغان - بىلمەيدىغانلىقىنى سۇرىۋېدى ، يې چۈن دەسلىۋىدە بىلمەيمەن دېدى ، ئەمما سەل تۇرىۋېلىپ ،راست بىر مەخسۇس ئايرۇپلان بار ، ئوغلۇم ئولتۇرۇپ كەپتىكەن ، ئۇنىڭ دادىسى ئەتە ھاۋا ياخشى بولسا ئاسمانغا چىقىپ ئايلىنىپ چۈشسەم دېگەنىدى ، _ دېدى .
        _ باشقا يەرگە بارامدۇ _ نېمە ؟ _ سۇرىدى جۇزۇڭلى .
        يې چۈننىڭ ئىنكاس قايتۇرىشى ئىنتايىن تېز ئىدى : بۇ يەرنىڭ ھاۋاسى سوۋۇپ كەتتى دالىيەنگە بارايلىمىكىن دېۋىدۇق ، _ دېدى .
        _ كېچىسى يول ئۇچۈش خەتەرلىك ، _ دېدى جۇزۇڭلى .
       _ كېچىسى ئۇچمايمىز ، ئەتە ئەتتىگەن ياكى چۈشتىن بۇرۇن ھاۋا ياخشى بولسا ئۇچىمىز ، - دېدى يې چۈن .
        _ بولدى ئۇچماي قويۇڭلار ، بىخەتەر ئەمەس ، چوقۇم ھاۋارايى ئەھۋالىنى ئۇبدان ئىگىلەڭلار ،_ دېدى جۇزۇڭلى .
        ئەگەر زۆزۆر بولسا ئۈزەم بېيدەيخېغا بېرىپ يولداش لىن بياۋنى يوقلاپ كېلىمەن ، _ دېدى جۇزۇڭلى يەنە .
        جۇزۇڭلى بېيدەيخېغا بارىمەن دېۋىدى ، يې چۈن بىردىنلا سەگەگلىشىپ ھۇدۇقۇپ كەتتى . ئەگر جۇزۇڭلى كېلىدىغانلا بولسا ، لىن بياۋنىڭ گۇاڭجۇغا بېرىپ ئايرىم مەركىزى ھۆكۆمەت قۇرۇش سۈيقەستى بەربات بولاتتى ،يې چۈن جۇزۇڭلىنى بېيدەيخېغا كەلمەسلىككە دەۋەت قىلدى  ، ئۇ : ئەگەر سىز بېيدەيخېغا كەلسىڭىز لىن بياۋ تېخىمۇ جىددىلىشىپ ، خاتېرجەمسىزلىنىپ كېتىدۇ . ئەڭ ياخشىسى سىز كەلمەڭ ، _ دېدى .
          ( داۋامى بار )

خەت چوڭلىقى : كىچىك نورمال چوڭ

 

           بۇ چاغدا جۇزۇڭلى خەلق سارىيىدا ، مەن جۇڭنەنخەيدىكى جەنۇبى بىنادا ئىدىم . ئالدىراشلىقتىن پۇتىمىز يەرگە تەگمەي كىتىۋاتاتتى . جۇزۇڭلى لى دېشېڭنى ھاۋا ئارمىيىسى قۇماندانلىق شتابى ئۇرۇش قىلىش بۆلۈمىنىڭ نۆۋەتچىلىك ئىشخانىسىغا ئۆزىگە ياردەملىشىپ ۋاقىتلىق قۇماندانلىق قىلىشقا مەسئۇل قىلىپ ئەۋەتتى ھەمدە ياڭ دېجۇڭنى ۋۇفاشيەنگە ھەمرا قىلىپ شىجياۋ ئايرودورومىغا ئەۋەتتى .
         لىن بياۋ يې چۈننىڭ دوكلاتىدىن جۇزۇڭلىنىڭ بېيدەيخېغا كېلىدىغانلىقىنى ئۇقتى . لىن بياۋ : مەنمۇ بەرىبىر بۈگۈن كەچتە ئۇخلىيالمايدىغان بولدۇم ، نەرسە - كېرەكلەرنى تېز يىغىشتۇرۇڭلار ، دەرھال ماڭايلى ، - دېدى . ئۇمۇ : قانچە تېز ماڭساق شۈنچە ياخشى ، - دېدى .
       بۇيرۇق چۈشىۋېدى ، 62 - نومۇرلۇق بىنادىكىلەر پاتپاراق بولۇپ كەتتى ، لېكىن ئۇلارنىڭ ھېچقايسىسى نېمىشقا مۇنداق تېز كېتىدىغانلىقىنى بىلمەيتتى . لىن بياۋنىڭ ماشىنىسى خانىسىنىڭ ئىشىكى ئالدىغىلا كەلتۈرۈلدى . لىن بياۋ ماشىنىغا چىقىش ئالدىدا يې چۈن ئادەم ئەۋەتىپ لىن لىخېڭنى ئىزدەتتى .
        لىن لىخېڭ لىن بياۋنىڭ باشقا بىر يەرگە بارماقچى ئىكەنلىكىنى دوكلات قىلغاندىن كېيىن ، ئۆيىگە قايتىشقا جۈرئەت قىلالمىدى . بۇ چاغدا ئىككىنچى ئەتىرەتنىڭ پوستا تۇرغان جەڭچىلىرى لىن بياۋنىڭ تۇرالغۇ ئورنىنىڭ قالايمانلىشىپ كەتكەنلىكىنى ، كىشىلەرنىڭ ئۈزۈلمەي بىر نېمە توشۇۋاتقانلىقىنى دوكلات قىلدى . لىن بياۋ ، يې چۈن ، لىن لىگولار ئىلگىرى - كېيىن بولۇپ ماشىنىغا چىقىشتى . لىن بياۋ لىن لىگو ۋە مۇھاپىزەتچى كاتىپتىن :
        _ ئېركوستىغا بېرىش ئۈچۈن قانچىلىك ئۇچىمىز ؟ - دەپ سۇرىدى .
        _ تېزلا بارىمىز ، _  دېدى لىن لىگو .
        لىن لياۋ بۇ گەپنى سۇراپ بولغاندىن كېيىن ماشىنا ماڭدى . ماشىنىنىڭ ئالدى ئورۇندۇقىدا لىن بياۋنىڭ مۇھاپىزەتچى كاتىپى ، كەينىدىكى ئورۇندا لىن بياۋ ، يې چۈن ، لىن لىگولار ئولتارغانىدى . ماشىنا قاراۋۇللۇق پونكىتىغا كەلگەندە ، پوستا تۇرغان جەڭچىلەر ماشىنىنى توستى ، يې چۈن شوپۇرغا بۆسۈپ ئۆتۈپ كەت ، دېدى . مۇھاپىزەتچى كاتىپ تۇيۇقسىزلا نىيىتىنى ئۆزگەرتىپ ، « ماشىنىنى توختات » دەپ توۋلىدى ، شوپۇر بۇ گەپنى ئاڭلىماي  پەقەت ماشىنىنىڭ سۈرئىتىنى ئاستىلاتتى ، مۇھاپىزەتچى كاتىپ ئىشىكنى ئېچىپلا ئۆزىنى سرتقا ئاتتى . بىرسى ماشىنىدىن ئۇنىڭغا ئوق ئاتتى ، بۇ ئەھۋالنى كۆرۈپ تۇرغان جاڭ خۇڭ ، جياڭ زوشۇلار ماشىنا بىلەن ئۇلارنىڭ كەينىدىن قوغلىدى .
        لىن بياۋنىڭ « قىزىل بايراق » ماركىلىق ماشىنىسى سائىتىگە 100 كىلومېترلىق سۈرئەت بىلەن ماڭغاچقا ، جاڭ خۇڭلارنىڭ جىپ ماشىنىسى زادىلا قوغلاپ يېتەلمىدى . ئۇلار قوغلۇشۇپ شەنخەيگۇەن ئايرودورومىغا كەلگەندە ، لىن بياۋ ئاللىقاچان ئايروپىلانغا چىقىپ بولغانىدى . ھۇدۇقۇپ جىددىلىشىپ كەتكەچكە لىن بياۋنىڭ شەپكىسى ۋە يې چۈننىڭ شارپىسى يەرگە چۈشۈپ قالغانىدى . ئايروپىلان تۇلۇق ماي قاچىلانماي تۇرۇپلا ھاۋاغا كۆتۈرۈلدى ، مەخسۇس ئايروپىلاننىڭ ئىككى ليوتچىكىدىن بىرىلا ئايروپىلانغا چىقتى ، يول باشلىغۇچى ، ئالاقىچىنىڭ ھېچقايسىسى ئايروپىلانغا چىقىشقا ئۈلگۈرەلمىدى .
         جاڭ خۇڭ قاتارلىقلار لىن لىگو لايىقلىققا تاللىغان بىر نەچچە ساھىپجامالنى ئايرودورومدا تۇتۇپ قالدى . بۇ ساھىپجاماللارنىڭ قولىدا قۇرال بولۇپ ، مۇھاپىزەتچى جەڭچىلىرىمىزنى ئۆيگە كىرگىلى قويمىدى . جاڭ خۇڭ ئۇلارغا : بۇ سىلەرنىڭ نېمە قىلغىنىڭلار ! بىز مۇئاۋىن رەئىس لىن بياۋنى قوغدايدىغان تۇرساق ، نېمىشقا بۇنداق قىلىسىلەر ، - دېدى .
        مۇھاپىزەتچى جەڭچىلىەر ئۆيگە بېسىپ كىرىپلا ئۇلارنىڭ قۇرالىنى تارتىۋالدى .
        13 - كۈنى كېچە سائەت 0 دىن 32 مىنۇت ئۆتكەندە مەن جاڭ خۇڭنىڭ  شەنخەيگۇەن ئايرودورومىدىن ئايروپىلان ئۇچۇپ كەتتى ، دەپ بەرگەن تېلېفۇنىنى ئالدىم .
        شۇ چاغدا لىن لىخېڭمۇ : ئايروپىلاننىڭ ئاۋازىنى ئاڭلىدىم ، ئايروپىلان ئۇچقان ئوخشايدۇ ، _ دەپ تېلېفۇن بەردى .
         _ سىزنىڭ دوكلاتىڭىز سەل كېچىكىپ قالدى ، _ دېدىم مەن .
        _ مەن ھازىرلا ئايروپىلاننىڭ ئاۋازىنى ئاڭلىدىم ، دېدى ئۇ .
        _ ھازىر مېنىڭ سىزنىڭ تېلېفۇنىڭىزنى ئاڭلايدىغان ۋاقتىم يوق ، - دەپ تېلېفۇننى قويىۋەتتىم .
        .  .  .  .  .  .
         بىز ماۋجۇشىغا ئەھۋاللارنى دوكلات قىلىۋاتقاندا ، ۋۇ فاشيەن شىجياۋ ئايرودورومىدىن ماڭا تېلېفۇن بەردى ، ئۇ لىن بىياۋ ئايروپىلانىنىڭ ئۇچۇپ كەتكىنىگە يېرىم سائەتتىن كۆپرەك بولغانلىقىنى ، ئايروپىلاننىڭ شىمالغا قاراپ ئۇچۇپ ، جياڭجياكو ئارقىلىق خېبېيدىن چىقىپ ، ئىچكى مۇڭغۇلغا كىرگەنلىكىنى دوكلات قىلدى . ۋە ۋۇفاشيەن : ھۇجۇمچى ئايروپىلان ئەۋەتىپ توسىمىزمۇ ؟ _ دەپ سۇرىدى . مەن : تېلېفۇننىڭ يېنىدىن كەتمەي تۇرۇڭ ، مەن دەرھال ماۋجۇشىدىن يوليۇرۇق سۇراي ، - دېدىم .

          _ لىن بياۋ پارتىيە مەركىزى كومتېتىنىڭ مۇئاۋىن رەئىسى ، ئاسماندىن يامغۇر يېغىۋېرىدۇ ، قىزلار تويىنى قىلىۋېرىدۇ . توسىماڭلار ، ئۇچۇپ كېتىۋەرسۇن ، _ دېدى ماۋجۇشى . جۇزۇڭلىمۇ قوشۇلدى ھەمدە مېنىڭ بۇنى ۋۇفاشيەنگە يەتكۈزۈپ قويۇشۇمنى ئېيتتى . مەن دىجۈرنىخانىغا يۈگۈرگەن پېتى ېرىپ ، ۋۇفاشيەنگە ئايرۇپىلان ئەۋەتىپ توسىماڭلار دېگەننى يەتكۈزدۈم ، باشقا مەزمۇننى يەتكۈزمىدىم .
         بۇ 9 - ئاينىڭ 13 -كۈنى تاڭ سەھەر سائەت بىردىن 12 مىنۇت ئۆتكەن چاغ ئىدى . ئايرۇپىلاننىڭ ئۇچۇپ كەتكىنىگە ھازىرغىچە 40 مىنۇت بولغان بولۇپ ، دۆلەت چېگرىسىدىن چىقىپ كېتەي دەپ قالغانىدى . بۇ ئايرۇپلاننى قويۇپ بېرىش ماۋجۇشى بىلەن جۇزۇڭلىنىڭ پىكرى ئىدى . ئۇلارنىڭ بۇ پىكرى توغرا ئىدى . مۇبادا بۇ ئايرۇپىلاننى توسىسىاق ، چاتاق چىقىپ مەملىكەت ئىچىدە يامان تەسىر پەيدا قىلاتتى . لىن بياۋ پارتىيىمىزنىڭ مۇئاۋىن رەئىسى ، بىز شۇ چاغدا ئۇنىڭ ئۇچۇپ نەگە بارىدىغانلىقىنى ، نىمە قىلىدىغانلىقىنى ئۇقمايتتۇق ، ئۇنىڭ مەخسۇس ئايرۇپلانىنى توسۇپ قويۇپ ، بىز پۈتۈن مەملىكەت خەلقىگە نىمە دەپ جاۋاپ بېرىمىز ! كېيىنچە شۇ چاغدىكى ئەمىلى ئەھۋالنى يەنى بۇنىڭ لىن بياۋ، يې چۈنلەرنىڭ ماۋجۇشى ھاياتلا بولسا ، مەيلى ئابروي ، مەيلى ئەلەم ياكى قەلەمدە ماۋجۇشىغا تەڭ كېلەلمەيمىز دەپ قاراپ ، ئۇزۇن مەزگىللەردىن بۇيان پىلانلاپ يۈرگەن يۈرگەن سۈيقەست ئىكەنلىكىنى ئۇقتۇق . شۇ ۋەجىدىن لىن بياۋ مۇنداق ئۈچ پىلان تۈزگەن : 1) ماۋجۇشىغا سۈيقەست قىلىپ ، پارتىيە _ ھۆكۆمەتنىڭ ئەڭ ئالىي ھوقۇقىنى تارتىۋېلىش . 2 ) گۇاڭجۇغا قېچىپ بېرىپ ئايرىم مەركىزى ھۆكۆمەت قۇرۇش . 3 ) شىمالغا قېچىپ ۋەتەنگە ئاسىلىق قىلىش .
        9 - ئاينىڭ 14 _ كۈنى ئەتتىگەن سائەت سەككىز يېرىمدا ، مۇڭغۇڭىيە تاشقى ئىشلار مىنىستىرلىكى جۇڭگونىڭ مۇڭغۇلىيىدە تۇرۇشلۇق باش ئەلچىخانىسىغا تېلېفۇن بېرىپ ، مۇئاۋىن تاشقى ئىشلار مىنىستىرى ئېردىن بىلىكنىڭ  جۇڭگونىڭ مۇڭغۇلىيىدە تۇرۇشلۇق باش ئەلچىسى بىلەن كۆرۈشۈپ ، ئۇنىڭغا جۇڭگونىڭ بىر رىئاكتىپ ئايرۇپىلانىنىڭ مۇڭغۇلىيىدە ھادىسىگە ئۇچىرىغانلىق ئەھۋالىنى خەۋەر قىلىدىغانلىقىنى ئۇقتۇردى . چۈش ۋاقتى سائەت 12 يېرىمدا ،  جۇڭگونىڭ مۇڭغۇلىيىدە تۇرۇشلۇق باش ئەلچىسى  ئايرۇپىلانىنىڭ مۇڭغۇلىيىدە ھادىسىگە ئۇچىرىغانلىق توغرىسىدىكى ئەھۋالىنى جۇڭگو تاشقى ئىشلار مىنىستىرلىكىگە دوكلات قىلدى .   تاشقى ئىشلار مىنىستىرلىكى مۇۋەققەت تاشقى ئىشلار مىنىستىرى جى پېڭفېينىڭ رياسەتچىلىكىدە پارتىيە گۇرۇپپىسى يىغىنى ئېچىپ ، بۇ ئەھۋالنى تېزدىن مەركەزگە دوكلات قىلدى . شۇ چاغدا بىز ھەممىمىز خەلق سارىيىنىڭ شەرقى زالىدا يىغىن ئېچىۋاتاتتۇق ، دوكلاتنى مەركىزى كومىتېت ئىشخانىسىنىڭ مۇئاۋىن مۇدىرى ۋاڭ لياڭئېن تاپشۇرۇۋاپتۇ .
       جۇزۇڭلى دوكلاتنى كۆرگەندىن كېيىن  :  ئىنتايىن مۇھىم بىر خەۋەرگە ئىگە بولدۇق . سىز دەرھال ماۋجۇشىغا دوكلات قىلسىڭىز قانداق ؟ - دېدى .
        _ بولىدۇ ، مەن ھازىرلا باراي ! _ مەن خەۋەرنى دەرھال ماۋجۇشىغا دوكلات قىلدىم .
        .  .  .  .  .

خەت چوڭلىقى : كىچىك نورمال چوڭ

 

          گەرچە بۇ خەۋەر كونكرېت بولمىسىمۇ ، لېكىن ماۋجۇشى ، جۇزۇڭلى ۋە يىغىنغا قاتنىشىۋاتقان سىياسى بيۇرودىكى كۆپ ساندىكى يولداشلارنىڭ كۆڭلىنى ئارامىغا چۈشۈردى .
         باش ئەلچىمىز كېيىن ئايروپىلان ھادىسىگە ئۇچىرىغان نەق مەيدانغا بېرىپ ، ئەھۋال ئىگىلىدى ، ئايروپىلان مۇڭغۇلىيىنىڭ ئۈندۈرخان ئەتىراپىدىكى كېنىت ئۆلكىسىنىڭ بېلخ كانچىلىق رايۇنىدىن 10 كىلومېتر يىراقلىقتىكى جايغا چۈشۈپ كەتكەن . ئايروپىلاندىكى سەككىز ئەر ، بىر ئايالنىڭ ھەممىسى ئۆلگەن . ئايروپىلان چۈشۈپ كەتكەنلىكى ۋە تاشقى ئىشلار مىنىستىرلىكىنىڭ ئۇچىرىشىشلىرى توغرىسىدا باش ئەلچى ۋە تاشقى ئىقتىساد ئىشخانىسىدىكى تاشقى ئىشلار ئەمەلدارلىرىنىڭۇ ماقالىلىرى ئېلان قىلىندى ، بۇلارنىڭ ھەممىسى ئىشەنچلىك ئىدى .
        ئۇزۇن ئۆتمەي ، بىز خۇەيرۇغا قونغان تىك ئۇچار ئايروپىلاندىن قولغا چۈشۈرۈلگەن لىن بياۋنىڭ بىر قىسىم ھۆججەتلىرىنى ، مەسىلەن ، لىن بياۋنىڭ بۇيرۇقى ، ئۇ خۇاڭ يۇڭشىڭغا يازغان خەتلەرنى ئېلىپ كۆرگىنىمىزدە ، نەق مەيداندىكى خۇاڭ يۇڭشىڭ ، ۋۇ فاشيەن ، لى زوپىڭ ، چيۇخۇيزولار داڭ قېتىپ قالدى .
        لىن بياۋ ئاسىيلىق قىلىپ قېچىپ كەتكەندىن كېيىن ، ئۇنىڭ بىلەن مۇناسىۋىتى قويۇق بولغان  خۇاڭ يۇڭشىڭ ، ۋۇ فاشيەن ، لى زوپىڭ ، چيۇخۇيزولارنى قانداق بىر تەرەپ قىلىش توغرىسىدا ماۋجۇشى جۇزۇڭلىغا مۇنداق دېدى : « ئۇلارنى ئىقرار قىلىپ كەڭچىلىككە ئېرىشتۈرۈش ئۈچۈن ئون كۈن قاراپ باقايلى ، پېشقەدەم يولداشلارنىڭ خاتالىق ئۆتكۈزۈشىگە ۋە ئۇنى تۈزۈتۈشىگە يول قويايلى ، مەسىلىنى ياخشى تاپشۇرسىلا بولدى. »
       لېكىن خۇاڭ يۇڭشىڭ قاتارلىقلار بۇ ئون كۈن ئىچىدە يا لىن بياۋنىڭ قىلمىشلىرىنى پاش قىلمىدى ، يا ئۆزلىرىنىڭ مەسىلىسىنى تاپشۇرمىدى . ئۇلار ھېچنەرسىنى ئىقرار قىلمىدى . ئون كۈندىن كېيىن ماۋجۇشى مېنى يېنىغا چاقىرىپ :  خۇاڭ يۇڭشىڭ قاتارلىقلار قانداق بىر تەرەپ قىلىنىپتۇ ، بېرىپ جۇزۇڭلىدىن سۇراپ باقسىڭىز ، _ دېدى . مەن جۇزۇڭلىغا دوكلات قىلدىم . جۇزۇڭلى ماۋجۇشىغا ئۇلارنىڭ ھەدەپ ماتېرىياللارنى كۆيدۈرىشىۋاتقانلىقىنى دوكلات قىلدى . ماۋجۇشى :
        _ شۇنداڭ ، ئۇلار جىنايەت پاكىتلىرىنى يوقاتماقچى ، بۇ كىشىلەر يەنىلا ھەرىكەت قىلىۋېتىپتۇ ، ئۇلار ئاخىرغىچە قارشىلىق كۆرسەتمەكچى ، _ دېدى . جۇزۇڭلى ماۋجۇشىغا :
       _ مەن ھازىرلا بېرىپ بىر تەرەپ قىلاي ، بۈگۈن كەچ بىر تەرەپ قىلالمىسام ، ئەتە ئەتتىگەن چوقۇم بىر تەرەپ قىلىمەن ، _ دېدى .
       جۇزۇڭلى بىلەن ئىككىمىز ماۋجۇشىنىڭ يېنىدىن چىقىۋاتقاندا ، ئۇ ماڭا : سىز جۇڭنەنخەيدىن سىرتقا چىقىپ كەتمەي ، ماۋجۇشىنىڭ بىخەتەرلىكىنى قاتتىق قوغداڭ ، ئىش بولسا ، مەن جاڭ ياۋسى ، ياڭ دېجۇڭلارنى ئىزدەيمەن ، زۆزۆر تېپىلسا ئاندىن سىزنى ئىزدەيمەن ، - دېدى . مەن جۇزۇڭلىغا ئۇلارنى كوللېكتىپ يىغىندا ئەمەس ئايرىم - ئايرىم بىر تەرەپ قىلىش تەكلىپىنى بەردىم .
       كېيىن خەلق سارىيىنىڭ فۇجيەن زالىدا  خۇاڭ يۇڭشىڭ قاتارلىقلارغا ئۇلارنىڭ ئايرىپ تەكشۈرۈلىدىغانلىقى ئۇقتۇرۇلدى . شۇ چاغدا ئۇلارنىڭ قارشىلىق كۆرسىتىشىنىڭ ئالدىنى ئېلىش ئۈچۈن ، فۇجيەن زالىدىكى كۈلدان ، ئىستاكان قاتارلىقلارنىڭ ھەممىسى يىغىشتۇرىۋېتىلدى . جۇزۇڭلى ئۇلارغا مۇنداق دەپ ئۇقتۇردى : سىلەرگە ئىقرار قىلىپ كەڭچىلىككە ئېرىشسۇن دەپ ئون كۈن ۋاقىت بەرسەك قۇلاق سالمىدىڭلار . بۇ سەل قارايدىغان كىچىك ئىشمۇ ، يەنە بۇنىڭدىن چوڭ ئىش بولامدۇ ؟ سىلەر پارتىيە ، خەلق ئالدىدا جىنايەت ئۆتكۈزدۈڭلار ! ھازىر سىلەرگە ئايرىپ تەكشۈرۈلدىغانلىقىڭلارنى ئۇقتۇرىمەن .
       زەھەرخەندىلىك بىلەن پىلانلانغان بىر قېتىملىق ئەكسىلئىنقىلابى سىياسى ئۆزگىرىش مانا موشۇنداق بىت _ جىت قىلىندى .
