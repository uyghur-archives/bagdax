دىنىنىڭ بولۇپمۇ ئۇلۇغ ئىسلام دىنىمىزنىڭ ئۇلۇغ بىر ئىلىم ئىكەنلىگىنى چۇشەنگۇچىلىگىم بار.ئەمما دىنى ئىلىم ئۇچۇنمۇ ئىلمىلىك لازىم.مەن باشقا  گەپ قىلمىدىم پەقەت قارىسىغا ئىنكاس يازماي،ئىسلام دىنىنىڭ دىيارىمىزدىكى تارىخى ھەققىدە ئۇبدانراق ئىزدىنىپ،ئىلىمى دەللىك پىكىر قىلىش ھەققىدە مەسلىھەت بەردىم خالاس. يەنە سىزنىڭ سەممىڭىزگە شۇنى سېلىپ قۇياي،مەن  باغداش تور بېتىگە دىنى مۇنازىرە ئۇچۇن كىرمىگەن.شۇڭا ئاتالمىش«چۇچۇپ كىتىش»مەۋجۇت ئەمەس.قالغىننى ئۆزىڭىز بىلىڭ.
ئەمدى ھۆرمەت گىپىگە كەلسەم،ئۆزىڭىزنىڭ ئەركىنلىكى، مەن ئەزەلدىن باشقىلاردىن ھۆرمەت تاما قىلىپ ئۈگەنمىگەن.كەمتەر ئادەممەن.

بۇ مەزمۇن 3قەۋەتتىكى qarqaniدە2012-03-20 09:48يوللىغان يازمىسىغا نەقىل :
بىزنىڭ مۇشۇ چەرچەننىڭ خېلى يەرلىرى بولىدۇ، ئەمما ئىسلام دىنى بۇ يەرگە ئارقىچىلاپ كىرگەنمىكىن دەيمەن، بۇ يەردىكىلەرنىڭ ئىتقادى بەك سۇس، بىدئەت ئېغىر،ئىختىلاپمۇ خېلى ئېغىر. مەسلەن بۇلتۇرقى چەرچەن شەھەر ئىچىدىكى چوڭ جامە مەسچىتنىڭ مەزىنى جامائەتتىن تاياق يىگەن ئىشلارمۇ بۇنىڭ تىپىك مىسالى، پۇقرالاردا ئاڭ تۈۋەن، دىنى<<زات>> لاردا ساپا دىگەن ئەسلى يوق، قىمارخانا پاھىشخانىلار مەسچىتتىن كۆپ،چەرچەنلىكلەردە ئىقتىساد كۆپ،دىنى مىللى مائارىپقا مەبلەغ سالمايدۇ،بولمىسا چەرچەندە مىللىلار خەنلەردىن جىق،ئەمما چەرچەندە مىللى پۇراق ئانچە كۈچلۈك ئەمەس،چەرچەنلىكتە بار بولغان ئەمەلى شارائىت  ۋە ئقىتىساد قەشقەر خوتەنلىكلەردە بوپ قالسا جەمىيىتىمىز ئۈچۈن تولىمۇ پايدىلىق مۇھم ئىشلارغا ئىشلىتىلەتتى،ئەپسۇس، ئەمدى توختاپ قالاي،مەن قۇرۇلتايغا تەكلىپ بېرىدىغان ۋەكىل بولمىغاندىكىن. ئىشقىلىپ جىق ئىشلارغا ئېچىنىمەن!

يازما نورۇز ئۈستىدە بولغاندىكىن نورۇزنىڭ گېپىنى قىلايلى. بىدئەتكە كەلسەك بىدئەد چەرچەندىلا ئەمەس ئۇيغۇر دىيارىنىڭ ھەممە يېرىدە قەدەمدە بىر ئۇچراپ تۇرىدۇ. مۇسۇلمان قېرىنداشلىرىمىز كۈچىنىڭ يېتىشىچە بىدئەتكە قارشى تۇرىۋاتىدۇ. چەرچەندە كەنىتلەرنىڭ ھەممىسىدە مەسچىت بار، ھەتتا بەزى مەھەللىلەرلەردىمۇ بار. پاھىشخانا مەسچىتتىن جىق ئەمەس، بۇنداق يالغان گەپ قىلماڭ! سىز چەرچەندىكى مىللى پۇراقنى ھېس قىلالماپسىز، مىللى پۇراقنى ئاۋال ئۆزىڭىزدىن ئىزلەپ بېقىڭ سىزدە بارمىكىن؟ چەرچەنلىكلەر نامرات، سىز بۇنى تېخى بىلىپ يەتمەپىسز. قەشقەر، خوتەننىڭ ئىقتىزادى چەرچەننىڭكىدىن كۈچلۈك، شۇڭا ئۇلار سىز قىلالمىغان ئىشلارنى قىلدى. ئەگەر باشقىلار مىللى مائارىپقا مەبلەغ سالمىغان بولسا سىز مەبلەغ سېلىڭ، پۇل بارمۇ؟ بولسا باشقىلاردىن قېيدىماي ئىشنى ئاۋال ئۆزىڭىزدىن باشلاڭ.  مەزىننىڭ تاياق يىيىشىگە كەلسەك مەزىن جامائەتتىن تاياق يىمىگەن. ئەھۋال مۇنداق: مەزىن داۋاملىق مەسچىتكە يالاڭباش كىرگەنلەرنى تىللايدىكەندۇق. بىر كۈنى يەنە بىر قانچە بالا يالاڭباش مەسچىتكە كىرىپتۇ، مەزىن يەنە ئۇلارغىمۇ تىل ياغدۇرۇپتۇ. جامائەت ئارىسىدىكى بىرەيلەن مەزىنگە << تالادا لەيلەپ يۈرگەنلەر ساماندەك، بۇلار يالاڭباش بولسىمۇ مەسچىتكە كىرىپتۇ، ئۇنىڭ ئۈستىگە يالاڭباش مەسچىتكە كىرىش چەكلەنمىگەن >> دەپ ھەدىسلەردىن شېپى كەلتۈرۈپ مەزىنگە ئۇنداق قىلماسلىقىنى تەۋسىيە قىلسا، مەزىن ھېلىقى كىشىنى لۈكچەكتىن-لۈكچەككە تارتىپ، ئاتا-ئانىسىنى قوشۇپ تىللىغان، شۇنىڭ بىلەن ئۇ كىشى ئاچچىقىغا پايلىماي مەزىننى ئۇرۇپ تاشلىغان. ھەرگىزمۇ جامائەت مەزىننى ئۇرمىغان. شۇڭا كۆرمىگەن ئىشنى ئاڭلىۋالغىنىڭىز بويىچە سۆزلەپ پىتنە تارقاتماي بىلگەننى، كۆرگەننى، توغرىنى سۆزلەڭ!!! يۇرتىڭىز كىشىلىرىگە ئېچىنىشنىڭ ئورنىغا يۇرتىڭىز كىشىلىرىنىڭ مەدەنىيەتتە نىمە ئۈچۈن ئارقىدا قالغانلىقى، نىمە ئۈچۈن نامرات ئىكەنلىكى ئۈستىدە ئويلىنىڭ، ئۇلارغا ياردەم بېرىڭ. قىسقىسى، ئەمىلىي ئىشتىن بىرنى قىلىڭ. يەككە ھالدىكى بەزى نەرسىلەرنى ئومۇمىيلىققا سۆرەپ ئەكىرىپ چەرچەننىڭ ئوبرازىنى خۇنىكلەشتۈرمەڭ. سەلبىي نەرسىلەر ھەرگىزمۇ بىر يۇرتنىڭ ياكى بىر قەۋمنىڭ ۋە ياكى بىر توپنىڭ ۋەكىلى بولالمايدۇ.
يەنە بىرگەپ دىن بىرەر يەرگە ئالدىچىلاپ، يېنىچىلاپ، ئارقىچىلاپ، ئاستىنچىلاپ، ئۈستۈنچىلەپ... كىرەمدىكەن؟ سىزنىڭ <<ئىسلام دىنى بۇ يەرگە ئارقىچىلاپ كىرگەنمىكىن دەيمەن، >> دېگەن بۇ داڭقان پۇتى گېپىڭزىنى قانداق چۈشەنسەك بولىدۇ؟...

ئەسسالامۇ ئەلەيكۇم باغداش ئەھلى ، ھەممىڭلانىڭ نورۇز بايرىمىڭلاغا مۇبارەك بولمىسۇن،.؛ مەن چەرچەنى نەچچە كۈندىن بىرى تورغا چىقالماي ئەپسۇسلىنارلىق ئىش بوپتۇ، مەن شۇ تاپتا ھەرقايسىڭلادىن سىلە مۇسۇلمانمۇ دىگەن سۇئالنى سورىسام جاۋاپ بېرىشنىڭ ئورنىغا مەندىن رەنجىشىڭلا مۇمكىن، بۇنى چۈشىنىمەن. چۈنكى مۇنبەردە ماڭا رەدىيەبەرگەن بەزى <<مۇسۇلمانلارنىڭ>> ،<<ئىمانى>> مىڭ گەز ئۆرلەپ، ئۆزىنى ئاران بېسۋالغاندا مىنىڭ سىلەرگە ئىنكاس قايتۇرشۇم، ئوت ئۈستىگە ياغ چاشقانلىق ،يەنى زىددىيەت پەيدا قىلىپ ئىختىلاپ تېرىغانلىق ھېساپلىنىدۇ، جانابى ئاللاھ ئىختىلاپنىڭ ئالدىنى ئېلىشنى پەرىز قىلدى. ئاللاھ ھەممىمىزنى تىنىچ ئىتپاق ياشاشقا نىسىپ قىلسۇن،(ئامىن)  مىنىڭ بۇ مۇنبەرگە مۇنداق ئىنكاسنى يېزىشىمدىكى سەۋەپكە كەلسەك، جانابى ئاللاھنىڭ سۆزى چىقمىغان ھەربىر سورۇنغا ئاللاھنىڭ لەنىتى ياغىدۇ، شۇ ۋەجىدىن ھەممەيلەنگە ئەجىر ساۋاپ بولسۇن دەپ ئىنكاس يازغان ئىدىم، تۇرسۇن بەكرى ئاكىغا دەيدىغان گەپ بۇ مۇنبەرلا ئەمەس 18مىڭ ئالام ئاللاھنىڭ بولغان ئىكەن،  بۇ سورۇنمۇ ئىسلامى سورۇن ئەمما دىن نوقتىسىدىن چىقىپ تۇرۇپ تىما ئىنكاس يوللايمەن دىسىڭىز،  ئۆزىڭىزنى دىندىن چىقىرىپ تۇرۇپ يوللىغان بولىسىز، چۈنكى مۇئمىننىڭ ھەربىر تىنىقىمۇ دىن-ئىسلامنىڭ ئىچىدە بولىدۇ، يەنى قىلغان ھەممە ئىشى ئەمەل بېرلىدىغان ( بۇ يەردە ياخشى-يامان ئەمەل كۆزدە تۇتىلىدۇ) دائىردىن مەڭگۈ چىقىپ كىتەلمەيدۇ،   قالغىنى اللاھ ،ۋە اللاھ نىڭ رەسۇلىنىڭ ھەق يولىنى تۇتۇپ ماڭغانلارغا اللاھنىڭ رەھمىتى ۋە سالامى بولسۇن، ئامىن،.

تورنى ئاچسام بۇ غەلۋە تېخى تۈگىمەپتۇ.چەرچەن ھەققىدە ماقالە يېزىش بۇنچە جىدەللەرگە ھەتتا مىنىڭ شەخسىيىتىمگە تاقىشىدىغان ئىش بولىدىغانلىقىمنى ئويلاپ يەتمىگەن ئىكەنمەن.گەپ قىلسام گەپ ئۇزۇراپ كىتدىغاندەك تۇرىدۇ.پەقەت سۇراپ باقسام،سىز ئۆزىڭىزنى"مۇسۇلىمان' چاغلاپسىز،باشقىلارنى نىمە چاغلايسىز؟بۇ دۇنيادا ھەر بىر بەندىنىڭ سالاھيىتىنى پەقەت بىر ئاللا بىلىدۇ.سىزگە چەرچەنلىكلەر دىنغا يانچە كىرگەن دەپ ۋەھى كەلگەنمۇ؟سىز قەيەردىن كەلگەن "ئەۋلىيە" سىز؟ بىدئەت دىسە سىزدەكلەرنى دىسە بۇلىدۇ. ئىنساننىڭ  تەغدىرىنى ياراتقان ئاللا بەلگىلەيدۇ سىز بەلگىلىيەلمەيسىز. ماڭا ۋە باشقىلارغا دىندىن چىقىپ ماقالە ياز دەپمۇ بۇيرۇق چۇشۇرەلمەيسىز.سىزگە دەپ قۇياي ،گەپنى ئويلاپراق قىلارسىز.ناۋادا قولىڭىزدىن كەلسە،يۇرتىمىز ھەققىدە ماقالە يېزىپ تورغا يوللاڭ، سىز ھەققى تەقىۋادار مۇسۇلمان بولسىڭىز ئالدى بىلەن ئۆز ئانا يۇرتىڭزنى، خەلىقىڭىزنى سۈيۈشنى بىلىڭ.ئۇلارنى  ئىتتىپاقلاشتۇرۇپ توغرا يولغا باشلاشنىڭ يوللىرى ھەقىقىدە ئىزدىنىڭ. ئىسلام دىننىنىڭ بىر ھەدىسىدە" يۇرتنى سۈيۇش ئىماننىڭ جۇملىسىدۇر' دىيىلگەن.مىنىڭ يۇرتىمىز چەرچەننى تور دۇنياسى ئارقىلىق تۇنۇشتۇرۇشۇم ئۆزەمنىڭ ئىشى،سىز نىڭ بۇنىڭغا چات كىرىۋېلىشىڭىز، ھەتتا مىنىڭ سەخسىيىتىمگە چەك قۇيۇش ھوقۇقىڭز يوق. گەپنى قىسقارتاي،سىزنىڭ مىنىڭ ئىتقادىمدىن گۇمانلىنىشقا،ماڭا يا ئۇنداق يا مۇنداق دەپ باھا بىرىشىڭزگە شەرتىڭىز توشمايدۇ.چۇنكى ھەر ئىككىمىزنىڭ تەغدىرىنى ياراتقان ئاللا ئىگەم بەلگىلەيدۇ.بۇ سىزدەكلەرنىڭ قولىدىن كەلمەيدۇ.

خەت چوڭلىقى : كىچىك نورمال چوڭ

 

تەكلىپىڭىزنى قۇبۇل قىلدىم. ئەمما، سىز «مىنىڭ دەۋرگە بېقىپ گەپ قىلىش»دىگەن سۆزۇمنى خاتا چۇشۇنۇپ قاپسىز.دىمەكچى بولغۇنۇم،بىز دۇچ كىلىۋاتقان رىياللىقىنى، ياشاۋاتقان مۇھىتىنى كۆزدە تۇتقان ئىدىم.سىز ۋە مەن بۇ رىياللىقتىن ئۆزىمىزنى چەتكە ئالالمايمىز.ھەم  قولىمىزدىن كەلمەيدۇ. بۇنى سىز بەلكىم چۇشۇنۇسىز .مەن ئەزەلدىن سىز باھا بەرگەندەك،«سۇنىڭ ئېقىشىغا،تۈكنىڭ يېتىشىغا ئىش كۆرىدىغانلار» ئەمەسمەن.ناۋادا شۇنداق بولغان بولسام،30نەچچە يىللىق خىزمەت سىتاژىمنى پەش قىلىپ،ھۆكۈمەت بەرگەن مۇئاشىمنى ئېلىپ ياشاۋەرگەن بۇلار ئىدىم.60كە يېقىنلاشقان ۋاقتىمدا كومپيۇتىر ئالدىدامۇكچۇيۇپ ئولتۇرۇپ ماقالا يازماس ئىدىم.سىز ئويلاپ بېقىڭ قېرىندىشىم، بۇ ماڭا ئاسانمۇ؟ ناۋادا مەن سىز دىگەندەك «سۇنىڭ ئېىشىغا » قاراپ ئىش تۇتقان بولسام يۇرتىمىزنىڭ قەدىمقى تارىخىنى، مەدەنىيتىنى تۇنۇشتۇرۇش بۇ ھەققىدە ئىزدىنىش، يېقىنقى ۋاقىتلارغىچە تامغىلانغان،خاتا بىكىتىلگەن بىر قىسىم قاراشلارنىڭ توغرىسىنى تېپىش ، ئۇنى ئىسپاتلاشنىڭ ماڭا نىمە كېرىكى؟؟كېچە- كۇندۇزلەپ تارىخى ئەسەرلەر بىلەن ھەپىلىشىشنىڭ ماڭا نىمە كېرىكى؟ھۆكۈمەت بېسىپ چىقارغان كىتاپلاردىكىلەرنى «سۇنىڭ ئېقىشىغا قاراپ». توغرا دەپلا كىتىۋەرگەن بۇلار ئىدىم.شۇڭا ئاخىرىدا ماڭا «پىرىسىنپىڭىز ئاشكارىلاندى » دەپ دىياگىنۇز قۇيۇشىڭىزنى قۇبۇل قىلالمىدىم.سىز خالىشىپسىز، مەن سىز دىياگىنۇ ز قويغان قوساق بېقىپ. كۇن ئۆتكۈزىدىغانلاردىن ئەمەس.«سۇنىڭ ئېقىشىغا، تۈكنىڭ يېتىشىغا» ياشايدىغانلادىن تېخىمۇ ئەمەس.
ئەمدى ئىلمىلىك ھەققىدە توختالسام، گەپ ئۇزۇراپ كىتىدۇ، قىسقىچە ئېيىتسام،مىنىڭ ئىلىمىلىك دىگنىم، مەن يازغان ماقالە نورۇز ھەققىدە ، ئىنكاس يازغان توردىشىمنىڭ مۇنازىرىسى بولسا ، باشقىچە مەزمۇندا بولغاچقا، زامان ،ماكان نوقتىسىدىن ئىلىمى ئانالىز ئېلىپ بېرىپ،رىياللىققا ھۆرمەت قىلغان ئاساسىدا ئىنكاس يازمىغانلىققىغا قارىتىلغان بۇنى چۇشۇنۋېلىشىڭىزغا ئىشىنىمەن.
