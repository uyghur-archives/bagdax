     جەمئىيەتتشۇناسلىق ئۇيغۇرلارغا تېخى يېڭى بىر ئۇقۇم ، زۇلپىقار بارات ھازىر  ئامىرىكىدا بۇ پەندە ئوقىۋاتىدۇ  ،ۋە ئۇيغۇر جەمئىيتىدىكى ناچار قىلمىشلارنى ، ناچار ئىدىيە تەسىر قىلىدغان ئاپتور ۋە ئۇلارنىڭ ماقالىلىرىنى قامچىلۋاتىدۇ .  شۇنداقلا بىر قىسىم كىشىلەر تەرىپىدىن قارشىلىققا ئۇچىراۋاتىدۇ ،
  زۇلپىقار بۇ ئاتقا مىنگەنكەن ئەمدى چوقۇم بۇ خىل بېسىملار بىر مەزگىل بەرداشلىق بەرمەي بولمايدۇ ،  بەيگە ئېتىغا مىنگەنلەر ھەممىسى بۇ ئەھۋالغا ئۇچىرايدۇ ،
   لېكىن مىللەت ئۈچۈن سوققان بۇ يۈرەك چوقۇمكى بۇ خىل ئەھۋاللار ئۈستىدىن يەنىلا غالىپ كېلىدۇ ،
  زۇلپىقارنىڭ بەيگە ئېتىنى تېخىەمۇ يىراقلارغا چېپىشىنى ئۈمىد قېلىمەن ، ھارماڭ !
بىزگە يالقۇن روزىدەك ، زۇلپىقار باراتتەك  بىزنىڭ ئوپىراتىسىيە قىلىپ پۇتاپ تۇرىدىغانلار بولمىسا بولمايدۇ

شىنجاڭ ئۇيغۇر ئاپتونوم رايونلۇق مىللەتلەر تىل-يېزىق خىزمىتى كومىتېتى لۇغەت بۆلۈمى تۈزگەن <<ئۇيغۇر تىلىنىڭ ئىزاھلىق لۇغىتى>> (قىسقارتىلمىسى) دە سەرخۇش سۆزىگە مۇنداق ئىزاھات بېرىلگەن:
سەرخۇشi كەيپ قىلغان، كەيپى بار، بىر ئاز مەست، شىركەيپ.
سەرخۇشii ئېلىشاڭغۇ، ساراڭ قېتىش، تەلۋە.

سىز روزى موللا دەپ كىمگە سۈيىقەست قىلىۋاتىسىز؟ كىمگە دارتمىلاۋاتىسىز؟ ھەق گەپ بىلەن مەشھۇرلۇق تاپقان يالقۇن رۇزىنى دارتمىلىغىنىڭىز نىمىسى؟
سىزنىڭ بۇ باغلاش ئەھۋالىڭىزدىن قارىغاندا،روزى موللا نىڭ كىملىكىنى بىلمەيدىغانلىقىڭىز چىقپلا تۇرۇپتۇ.مىللى مەدەنيەت تارىخمىزدىكى قانلىق پاجىئەلەرنىڭ بىۋاستە ئىجادچىسى ۋە ياكى سەۋەپچىسىنى بىلەلمسلىك بۇ ئەيىپ ئەمەس،بىراق ھەر كىشى ئۆزىنىڭ بىلىش دائىرىسى ئىچىدە پائالىيەت ئېلىپ بارغىنى ئەۋزەل.ئۆزباش ئەپەندىم ئۆزى ئېيىتقاندەك نۇرغۇن بەدەل تۆلەش ھېسابىغا مىللىتىمىزنىڭ ئوتىدا كۆيۈپ،سۈيىدە ئېقىپ يات ئەلدە ھىجرەت ھەلەكچىلىكىدە توختاۋسىز ئىلىم تەھسىل قىلىۋاتىدۇ،ھەممىمىز ئىجادىيەت سەپىرىمىز ۋە ياكى تىرىكچىلىك قەدىمىمىزدە ئۆزىمىزگە پۇختا بولساق،ساماننىڭ تېگىدىن سۇ قويۇش،سۇنى لېيىتىپ بېلىق تۇتۇش،ئۆزگىلەرنىڭ غەلبەمىۋىسىگە كۆز ئالايتىش...تەك بىمەنە ئىشلاردىن يىراق تۇرساق،ئۆز-ئارا ھەمكارلىق،ئېناقلىق،ئىتتىپاقلىقىنى كۈچەيىتسەك دېگەن ئۈمۈتتمەن.ئاخىرىدا ئۆزباش ئەپەندىمنىڭ ئىلىم تەھسىل قىلىش ئىشلىرىغاھەم ئىجادىيەت قەدەملىرىگە ئۇتۇق تىلەيمەن.

  • مۇنبەرپۇلى:+5(sebir) ھەق گەپ
    • ھەممە ئىشتا ئوتتۇرھال بولغان ئەڭ ياخشىدۇر.
