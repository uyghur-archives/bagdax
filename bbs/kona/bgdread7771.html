      ئۆلۈمىدىن پۇشايمان يېمىگەن جيەن بوزەن



 شىمال بورىنى قۇملارنى ھەدەپ ئۇچۇرۇپ يەر _ جاھاننى چاڭ - توزان ۋە پاسكىنچىلىققا تولدۇرىۋەتكەنىدى . 
چۈشتىن كېيىن سائەت سەككىز ئەتىراپىدا جيەن بوزەن * مەخسۇس دېلۇ گۇرۇپپىسىدىكى كالتە بۇرۇت يەر ئاستىدىكى 7 - نومۇرلۇق ئۆيگە ئالدىراپ كىرگىنىچە ئېلېكتىر چىراقنى ياندۇرۇپ ، سۇراق خاتىرىسىنى ئاچتى ، قەلەم قەغەز ئۈستىدە شىرتىلداشقا باشلىدى :  گېزەندە __ بېيجىڭ داشۆ تارىخ فاكۇلتېتىنىڭ پروفېسسورى جيەن بوزەن ؛ ۋاقتى :  1966 - يىلى 27 - ئۆكتەبر چۈشتىن بۇرۇن ؛ سۇراقنىڭ مۇھىم مەزمۇنى :  ۋيېتنام ئەمگەكچىلەر پارتىيىسىنىڭ رەئىسى خۇزىمىڭغا ساتقىنلىق قىلىش مەسىلىسى .  .  . 
       بىر پەستىن كېيىن پاقا چاناق لەقەملىك بىرى جيەن بوزەننى بوينىغا قارا تاختا ئاسقان ، قۇلىغا كويزا سالغان ھالدا يالاپ ئەكىردى .
       پاقا چاناق سول قولىدا بېلىنى تېرەجىگەن ، ئوڭ قولىدا مىس كالتەك تۇتقان ھالدا :
       __  جيەن بوزەن ، خۇزىمىڭنىڭ 1942 - يىلى جۇڭگودا قانداق قولغا ئېلىنغانلىقىنى بىلەمسەن ؟  _- دەپ سۇرىدى .
       جيەن بوزەن پاقا چاناقنىڭ سۇرىغان سۇئالىنىڭ تاپقا چۈشمەيدىغانلىقىدىن نېمە ئىش يۈز بەرگەنلىكىنى بىلمەي ھەيران قالغان بولسىمۇ ، بىراق يەنىلا :
        __  خائىننىڭ مەخپىيەتلىكنى ئاشكارىلاپ قويغانلىقى تۈپەيلىدىن قولغا ئېلىنغان ، __ دەپ جاۋاب بېرىشكە مەجبۇر بولدى .
        __  خائىن كىم ؟
        __  ئاڭلىشىمچە رەئىس خۇزىمىڭنىڭ مۇھاپىزەتچىلەر ئەتىرىتىنىڭ مۇئاۋىن باشلىقى شۇنداق قىلغانمىش .
        __  گۇاڭشى قىزىل كېخمىر ئىسيانچىلار تۇەنىدىن كەلگەن تېلېگراممىدا پاش قىلىنشىچە ، __  پاقا چاناق كۆرەڭلىگەن ھالدا قولىغا تېلېگراممىنى ئېلىپ پۇلاڭلىتىپ تۇرۇپ ، __  خۇزىمىڭغا ساتقىنلىق قىلغان ئادەم سەن جيەن بوزەن !  __  دېدى .
        __  سىلەر قاراپ تۇرۇپلا ئادەمگە تۆھمەت قىلىۋاتىسىلەر ! _- جيەن بوزەننىڭ جاۋغايلىرى قاتتىق جىددىلەشكەنلىكتىن لىپىلداپ كەتتى ، __ سىلەر دەلىل ئىسپاتقا ئەھمىيەت بېرەمسىلەر ، يوق ؟ !
        دەرياغا بىر تاش چۈشۈپ كەتسە قاتمۇ - قات دولقۇن ھاسىل بولغاندەك ، بۇ ئىش تۈپەيلىدىن ئۇنىڭ كاللىسىغا ئۆتۈپ كەتكەن ئىشلار بىر - بىرلەپ كېلىپ ، قەلبىدە چەكسىز دولقۇن پەيدا قىلدى .
        1942 _ يىلى ئاۋغۇستنىڭ ئاخىرىقى ئونىنچى كۈنلىرىدىكى بىر كۈنى چۈشتىن بۇرۇنقى ۋاقىت ئىدى . جۇئېنلەي ئۇنى جۇ قەسىرىگە چاقىرىپ كېلىپ ئۇنىڭغا :  ۋيېتنام ئەمگەكچىلەر پارتىيىسىنىڭ رەئىسى خۇزىمىڭنىڭ جۇڭگودا قولغا ئېلىنغانلىقىنى ، ھازىر ئۇنىڭ گۈيلىندىكى جاڭ فاكۇينىڭ 4 _ جەڭ رايۇنىدا قاماقلىق ئىكەنلىكىنى ئېيتىدۇ . جۇئېنلەي ئىنتايىن تىت _ تىت بولغان ھالدا :
        __  مەركەز چۇڭچىڭدىكى يولداشلار __ بىزنىڭ ئامال قىلىپ قۇتقۇزىشىمىزنى قارار قىلدى ، _ دەيدۇ .
        جيەن بوزەن بىر پەس تۇرىۋالغاندىن كېيىن :
        __  مۇئاۋىن رەئىس جۇئېنلەي ۋەزىپىنى ماڭا تاپشۇرۇڭ ، __ دەيدۇ .
        __  ئىش ئەسلىدە مۇنداق ئىكەن ، تاجاۋۇزچىلىققا قارشى خەلقارا جەمئىيەتنىڭ ۋيېتنام شۆبىسى چۇڭچىڭ قانۇن تۈزەش پالاتاسىدىكى سۇن كېغا جىددى تېلېگرامما ئەۋەتىپتۇ ، __ دەيدۇ جۇئېنلەي بېشىنى چايقاپ تۇرۇپ ، __ سۇن كې توخۇ يۈرەك ، قورقۇنچاق ئادەم ، ئۇنىڭ ئۈستىگە بۇ ئىشقا تازا پېيى يوق ، خۇددى گېنرال فېڭيۇڭشياڭ سۈرەتلىگەندەك لايدىن ياسالغان دىۋان بېگىگە ئوخشاش كارغا كەلمەيدىغان ئادەم . ھازىر ئاۋال ھېلىقى تېلېگراممىنى قولغا چۈشۈرىشىمىز كېرەك ، ئاندىن كېيىن مۇۋاپىق ئادەم تېپىپ ، ، تېلېگراممىنى شۇ ئادەم ئارقىلىق جياڭ جيېشىگە يۈز تۇرانە تاپشۇرىشىمىز لازىم ، __ دەيدۇ .
       جيەن بوزەن :
       __ قانۇن تۈزۈش پالاتاسىنىڭ مۇئاۋىن باشلىقى تەن جېن بىلەن مەن يۇرتداش __ تاۋيۇەنلىكمىز ، ئۇنىڭغا كاتىپلىقمۇ قىلغانمەن ، ئارىمىزدا يۇرتداشلىق مېھرىمۇ ، شۇنداقلا شەخسي دوستلۇقمۇ بار ، مېنىڭچە ئۇ بۇ ئىشقا ياردەم قولىنى سۇنىدۇ ، __  دەيدۇ .
       __  توغرا ، مەنمۇ مۇشۇنداق مۇناسىۋەتتىن پايدىلانماقچىمەن .
       جيەن بوزەن شۇئانلا مۇئاۋىن باشلىق تەن جېن بىلەن كۆرۈشۈپ ، ئۆتكەن _ كەچكەن ئىشلار ئۈستىدە بىر مەھەل پاراڭلاشقاندىن كېيىن ، كېلىشتىكى مەقسىتىنى دەيدۇ . تەن جېن گومىنداڭنىڭ مۆتىۋەرلىرىدىن بولۇپ ، ھەققانىيەت تۇيغۇسى كۈچلۈك بولغاچقا ، ئۇنىڭغا پۈتۈنلەي ماقۇل بولىدۇ .
       ئىش ئەسلى مۇنداق بولغان :  ۋيېتنام ئەمگەكچىلەر پارتىيىسىنىڭ رەئىسى خۇزىمىڭ ۋيېتنامنىڭ گاۋپىڭ ئۆلكىىسى بېيپو شەھىرى يەنى ھېندىچىنى پارتىيە مەركىزى كومىتېتى تۇرۇشلۇق جايدىن يولغا چىقىپ گۇاڭشى ئاپتۇنۇم رايۇنىنىڭ جىڭشى دېگەن يېرى ئارقىلىق جۇڭگو تەۋەسىگە ئۆتۈپ ، چۇڭچىڭغا بېرىپ جۇئېنلەي بىلەن كۆرۈشۈپ ياپۇنغا قارشى ئۇرۇش ئەھۋالىنى ئۇقماقچى بولىدۇ . ئۇنىڭ بىلەن بىللە ماڭغان مۇھاپىزەت ئەتىرىتىنىڭ مۇئاۋىن باشلىقى رۇەن ۋېنفانىڭ ساقچى ئىدارىسىگە مەخپىيەتلىكنى ئاشكارىلاپ قويۇشى ئارقىسىدا خۇزىمىڭ ھەمرالىرى بىلەن بەخىتكە قارشى شۇ ۋاقىتنىڭ ئۆزىدىلا قولغا ئېلىنىدۇ .
       جيەن بوزەن تەن جېن ئارقىق تېلېگراممىنى سۇن كېدىن ئەكىلىدۇ . دەسلەپكى جەڭدە غەلبە قىلغاچقا جۇئېنلەي ناھايىتى خۇرسەن بولىدۇ _ دە ، شۇئاننىڭ ئۆزىدىلا قۇتقۇزۇش لاھىيىسىنى مۇزاكىرە قىلىدۇ .
        تېلېگراممىنى قايسى ئابرويلۇق كىشى جياڭ جيېشىگە يۈزتۇرانە تاپشۇرۇپ بەرسە بولار ؟
        جيەن بوزەن ئاۋۋال فېڭ يۇڭشياڭنى خىيالىغا كەلتۈرۈدۇ ، ئەمما شۇ ھامان يەنە ئۇنى مۇۋاپىق كۆرمەيدۇ . فېڭ يۇڭشياڭ بۇرۇندىن تارتىپلا ماتا كۆڭلەك كېيىپ گومىنداڭنىڭ يوقىرى قاتلىمىدىكىلەرنىڭ ھەر خىل سورۇنلىرىغا كىرىپ _چىقىپ يۈرەتتى ، ئۇنىڭ گەپ _ سۆزى قوپالراق ئىدى ، ئۇنىڭ ئۈستىگە كومپارتىيىگە مايىل ئىدى . خې يىڭچىن ، گۇ جېڭگاڭ ئېقىمىدىكىلەر ئۇنىڭغا " قىزىللىشىپ كەتتى " دەپ ھۇجۇم قىلاتتى ، جياڭ جيېشىمۇ بۇ " ھۈرمەتلىك ئاكىسى" نى " يېرىم كوممۇنىست " دەپ قارايتتى . ئەگەر تېلېگراممىنى ئۇنىڭ يۈز تۇرانە تاپشۇرۇپ بېرىشىنى ئۆتۈنسەك جياڭ جيېشى خۇزىمىڭنى " ۋيېتنام كوممۇنىستىك پارتىيىسىنىڭ ئادىمى" مۇ ئەمەسمۇ  دەپ گۇمانلىنارمۇ ، قانداق ؟  بۇ قېرىنداش پارتىيىنىڭ داھىيىسىنىڭ ئامانلىقى ، خېيىم - خەتىرىگە ئالاقىدار ئىش بولغاچقا ئېھتىياتچان بولماي بولمايتتى . ئۇنىڭ كاللىسىغا شۇ زاماتلا :  گۇاڭشىدا ئەمەل كۈچى بار لى زۇڭرېن چۇڭچىڭغا كەلگەن ئەمەسمىدى ؟ دېگەن ئوي كېلىدۇ .
        جيەن بوزەن جىگەرلىك ئادەم ئىدى ، ئەزەلدىن ئوڭايلىقچە باشقىلارغا يېلىنمايتتى . بىراق ، ھازىر خۇزىمىڭنى قۇتۇلدۇرۇش __ پارتىيىنىڭ ئىشى ئىدى ، شۇڭا شەخسى مېجەز بويىچە ئىش قىلىشقا بولمايتتى . ئۇ مۇشۇ بىر قانچە كۈندىن بېرى گومىنداڭنىڭ كاتتىباشلىرى ئارىسىدا ئۇياقتىن بۇياققا موكىدەك چېپىپ يۈردى . ئەسلىدە ئۇلارنى تىللىغۇسى بولسىمۇ ، بىراق يەنىلا كىشىلەرنى كولدۇرلىتىدىغان بەزى سۆزلەرنى تېپىپ ، چىرايلىق سۆزلەرنى قىلىپ يېلىنىپ ، يالۋۇردى ؛ ئەسلىدە يىغلىغۇسى كەلسىمۇ زۇرىغا كۈلۈمسىرىدى ، مانا ئاشۇنداق " ئۇياق _ بۇياققا قاتىراپ يېلىنىش" تىن ئىبارەت بۇرۇقتۇملۇققا چىداش ئەزەلدىن باشقىلارغا باش ئەگمەي كەلگەن جيەن بوزەنگە بەك تەس كەلدى ، تولىمۇ قېيىنغا توختىدى .

خەت چوڭلىقى : كىچىك نورمال چوڭ

 

         __ جيەن ئەپەندى ئېيتقانلىرى قائىدىلىك ، __ دەيدۇ لى زۇڭرېن ئوچۇق - يورۇقلۇق بىلەن ، __ ئەتە ، مەن خۇەنجاڭ ( فېڭ يۇڭشياڭنىڭ يەنە بىر ئىسمى) بىلەن بىللە ۋېييۇەنجاڭ بىلەن كۆرۈشكىلى باراي .
        __ ئۇنداقتا ئىككىلىرىنى ئاۋارە قىلىدىغان بولدۇق !
        دەسلىپىدە جيەن بوزەن فېڭ يۇڭشياڭنى زىيادە « قىزىل » ، كۆزگە يېقىن ئادەم دەپ قارىغانىدى . ئەمدىلىكتە بولسا لى زۇڭرېن ئۇنى بىرگە ئېلىپ بارىمەن دەپ تۇرىۋالغاچقا ، ئارتۇقچە گەپ قىلمىدى .
       ئاشۇ كۈنى چۈشتىن كېيىن ، لى زۇڭرېن ، فېڭ يۇڭشياڭلار ماشىنىغا ئولتۇرۇپ جياڭ جيېشى بىلەن كۆرۈشكىلى جەنۇبىي قىرغاقتىكى داچىغا قاراپ يولغا چىقىدۇ . سالام - سائەت قىلىشقاندىن كېيىن فېڭ يۇڭشياڭ گەپنىڭ پوسكاللىسىنى قىلىپ :
       __  ۋېييۇەنجاڭ ، ۋيېتنام تەرەپتىن تېلېگرامما ئەۋەتىپ ، جاڭ گۈيفانىڭ خۇزىمىڭنى تۇتۇۋالغانلىقىنى ئېيتىپتۇ ، __  دەيدۇ .
       __ جياڭ جيېشى دەرگۇمان بولۇپ فېڭ يۇڭشياڭنىڭ چىرايىغا قاراپ __  بۇ . . . خۇزىمىڭ دېگەن ۋيېتنام كومپارتىيىسىنىڭ ئادىمىمۇ قانداق ؟ __ دەيدۇ .
       فېڭ يۇڭشياڭ ئۇنىڭغا رەددىيە قايتۇرۇپ :
       بورودىن * دىنمۇ سوۋېت كومپارتىيىسىنىڭ ئادىمىغۇ . سىز نېمىشقا ئۇنى تۇتمايسىز ؟ ! __ دەيدۇ .
        __ بۇ . . .  ھەي . . .
       ۋېييۇەنجاڭ ، خۇزىمىڭنى نېمىشقا گۇاڭشىدا تۇتىدۇ ؟ __  لى زۇڭرېن ئاچچىق كۈلۈپ تۇرۇپ ، __  ھازىر ۋيېتناملىقلار بىزنىڭ گۇاڭشىدىن ئادىمىنى قايتۇرۇپ بېرىشنى تەلەپ قىلىۋاتىدۇ . مانا بۇ بالانى ماڭا ئىتتىرىپ قويغانلىق بولماي نېمە ! __  ئۇ گېپىنى قىلىپ بولۇپلا ۋيېتنامدىن كەلگەن تېلېگراممىنى ئۇنىڭغا سۇنىدۇ
       __ خۇزىمىڭنى تۇتقانلىق پىرقى دۇلەتنىڭ سەۋەنلىكى . بۇنداق قىلغاندا ، تاجاۋۇزچىلىققا قارشى خەلقارا جەمئىيەتنىڭ ھېسداشلىقى ۋە قوللىشىدىن مەھرۇم قالىمىز .
       _- خۇەنجاڭ ، دېلىڭ ( لى زۇڭرېن ) دىن ئىبارەت ئىككى گېنېرال خۇزىمىڭغا شۇ قەدەر كۆڭۈل بۆلگەن ئىكەنسىلەر ، ئۇنداقتا . . . مېنىڭ سىلەرنىڭ يۈزۈڭلەرنى قىلىشتىن باشقا ئامالىم يوق _ تە .
       لى زۇڭرېن ، فېڭ يۇڭشياڭلار كەتكەندىن كېيىن جياڭ جيېشى سوغۇققانلىق بىلەن ئويلىنىپ كۆرۈپ خۇزىمىڭنى تۇتۇش ، قويۇپ بېرىشنىڭ ئۆزىگە نىسبەتەن ئېيتقاندا ھېچنىمىگە تەسىر كۆرسىتەلمەيدىغانلىقىنى ھېس قىلىدۇ . شۇنىڭ بىلەن مۇھاپىزەت ئىشخانىسىنىڭ مۇدىرى لىن ۋېيگە جاڭ فاكۈيغا خۇزىمىڭنى قويۇپ بېرىشكە بۇيرۇق قىلىدۇ .
       ئەمما جيەن بوزەن بۇ ئىشتا ئۆزىنىڭ بەدەل تۆلەيدىغانلىقىنى بىلمەيتتى .
       1967 - يىلى 17 - ماي ، ۋيېتنامنىڭ ئامېرىكىغا قارشى تۇرۇپ ۋەتەننى قوغداش ئۇرۇشى راسا كەسكىن بولىۋاتقان چاغدا ، خۇزىمىڭ ئايروپلان بىلەن نەننىڭگە مەخپىي كېلىپ ، ئۇرۇش ئوتلىرى ئىچىدىن « تۇيدۇرماي يېنىپ چىقىپ » تۇغۇلغان كۈنىنى ئارامخۇدا ئۆتكۈزىدۇ .
       خۇزىمىڭنىڭ قەدىناس دوستى __ گۇاڭشى ئاپتۇنۇم رايۇنىنىڭ رەئىسى ۋېي گوچىڭ ئۇنىڭ تۇغۇلغان كۈنىگە ئاتاپ كاتتا ئائىلە زىياپىتى راسلايدۇ .
       19 ماي _  خۇزىمىڭنىڭ 77 ياشقا تولغان خاتىرە كۈنى ئىدى .ئاشۇ كۈنى كەچ كىرىشى بىلەنلا ۋېي گوچىڭنىڭ ئۆيىدىكى چىراقلار ھەممىدىن بۇرۇن يورۇتۇلىدۇ ، شىرەگە خاسىيەتلىك شاپتۇل ، خاسىيەتلىك ئاش شۇنداقلا داڭلىق ماۋتەي ھارىقى تىزىلغانىدى .
       __  خۇزىمىڭنىڭ ئۇزۇن ئۆمۈر كۆرۈشى ئۈچۈن كۆتىرەيلى ! __  كۆپچىلىك رومكىلىرىدىكى شاراپنى بىراقلا ئىچىۋېتىدۇ .
       زىياپەتتە ئۇلارنىڭ قىلىشمىغان گەپلىرى قالمايدۇ ؛ ئامېرىكىغا قارشى تۇرۇپ ۋەتەننى قوغداش ئۇرۇشى ،  جۇڭگونىڭ « مەدەنىيەت زور ئىنقىلابى » . . .  ئۈستىدە پاراڭ قىلىشىدۇ . ئويلىمىغان يەردىن ، ۋېي گوچىڭ 40 - يىللارنىڭ دەسلىپىدە خۇزىمىڭنىڭ جۇڭگودا قولغا ئېلىنغانلىق ئىشىنى تېلغا ئالغاندا ، خۇزىمىڭ قاتتىق ھاياجان ئىلكىدە :
       كېيىن ئۇقتۇمكى ، چۇڭچىڭدىكى جيەن بوزەن ئىسىملىك بىر تارىخشوناس مېنىڭ ھاياتىمنى قۇتقۇزغان ئىكەن . ئۇنىڭغا يۈز تۇرانە تەشەككۈر ئېيتقۇم بار ئىدى ، ئەمما دىدار كۆرۈشۈشكە پۇرسەت بولماي قالدى ، _  دەيدۇ . خۇزىمىڭ گويا روھىي جەھەتتىن قەرزدار بولۇپ قالغاندەك ، بەك ئەپسۇسلىنىپ كېتىدۇ .
       __  ھوي ؟ __ ۋېي گوچىڭنىڭ يۈرىكى جىغغىدە قىلىدۇ . يېقىندا ئۇ بىر قىسقا خەۋەردىن : « جيەن بوزەن مەخپىيەتلىكنى ئاشكارىلاپ قويغانلىقتىن خۇزىمىڭ قولغا ئېلىنغان » دېگەن بىر ماقالىنى كۆرگەنىدى . پارتىيىمىزنىڭ بىر نامسىز قەھرىمانى قاراپ تۇرۇپلا خائىن قىلىپ قويۇلغان . جاھاندا بۇنىڭدىنمۇ كۈلكىلىك ئىش بولامدۇ !
       ئەسەبىيلەشكەن ئاشۇ يىللاردا ، « خائىن » دېگەن بۇ سۆز نېمىلەردىن دېرەك بېرەتتى ؟ ! بۇ گەپنىڭ ۋەزنى ۋېي گوچىڭغا چۈشىنىشلىك ئىدى . شۇڭا ئۇ خۇزىمىڭنىڭ ھېلىقى گېپىنى جۇزۇڭلىغا دوكلاد قىلىدۇ . جۇزۇڭلى ۋېي گوچىڭنىڭ دوكلادىغا : « ئىنكاس قىلىنغان ئەھۋال ئەمەلىيەتكە ئۇيغۇن » دەپ ئىمزاسىنى قويۇپ ، ئۇنى بېيجىڭ ئونۋېرسىتېتى ھەربىي ئىدا قىلىش كومىتېتىغا چۈشۈرۈپ بېرىدۇ ھەمدە جيەن بوزەننىڭ يېقىندىن بۇيانقى ئەھۋالىنى ئۆزىگە مەلۇم قىلىشنى تەلەپ قىلىدۇ .
      « جيەن بوزەن دىلوسى » نىڭ كاتتىبىشى چوڭ يەردىن كەلگەن بۇ دوكلادنى كۆرۈپ مۇزدەك تەرگە چۆمىدۇ _  دە ، « سۇراق خاتىرىسى » نى ھاپلا - شاپلا كۆيدۈرىۋېتىدۇ .
       خۇزىمىڭنىڭ ئىشى ئىككىنچىلەپ تېلغا ئېلىنمىغان بولسىمۇ ، ئەمما ، جيەن بوزەننىڭ ھال - ئەھۋالىدا ياخشىلىنىش بولمايدۇ . شۇڭلاشقا ۋېي گوچىڭ قەدىناس دوستىغا ئۆز ئارزۇسىغا خىلاپ ھالدا :
      __ رەئىس خۇزىمىڭ ، جيەن بوزەن ئېغىر كېسەل بولۇپ قالغاچقا دوختۇرخانىدا يېتىپ قالدى . ھەرىكەت قىلىشى ئەپسىز ، _ دېيىشكە مەجبۇر بولىدۇ . خۇزىمىڭ ھەسىرەتلىك بىر تىنىدۇ _ دە ، كۆڭلىگە بىر نەرسە ئايان بولغاندەك تۇيۇلىدۇ . نەننىڭدە يەتتە كۈن تۇرۇپ ، ئەپسۇسلۇق ئىچىدە ۋيېتنامغا قايتىدۇ . شۇ كەمدە ۋيېتنامدا قاتتىق جەڭ بولىۋاتاتتى . ئاشۇ يەتتە كۈن ئالتۇندىنمۇ قەدىرلىك ئىدى .
        ( داۋامى بار )

        1968 _ يىلى كۈزدە ، چى چۇن توساتتىن بېيجىڭ ئونۋېرسىتېتىغا كېلىپ ماۋجۇشىنىڭ بورژۇئازىيە ئىلىم نوپۇزلىقلىرىغىمۇ « چىقىش يولى بېرش كېرەك » « ئەڭ ئالىي يوليۇرۇقىنى يەتكۈزىدۇ . « بېيجىڭ ئونۋېرسىتېتىدىكى جيەن بوزەن ، فېڭ يۈلەنگە چىقىش يولى بېرىش كېرەك » لىكىنى ئالاھىدە تىلغا ئالىدۇ . جيەن بوزەن بىر مەزگىل ئىچىدە « چىقىش يولى بېرش » نىڭ تىپىغا ئايلىنىدۇ . ئەسلىدىكى يەننەن باغچىسىغا كۆچۈرۈلۇپ ، نازارەت ئاستىدا تۇرىدۇ . ئەمما ، ياجشى كۈن ئۇزۇن داۋام قىلمايدۇ . باش قىشتا « قىزىل بايراق » ماركىلىق پىكاپ جۇڭنەنخەيدىن چىققىنىچە بېيجىڭ ئونۋېرسىتېتى قۇرۇسىغا ئاستىلا كېرىپ كېتىدۇ . پىكاپتىن 40 ياشلار چامىسىدىكى ئىلمەك بۇرۇن ، دىقماق ، تېمەن ، پەيلى يامان بىرەيلەن چۈشىدۇ .
       ئۇنىڭ ئىسمى ۋۇجۇڭ ئىدى . ئۇ ئەلۋەتتە ئىش بېجىرگىلى كەلگەنلەردەك ، ئاۋال پەلەسەپە فاكولىتىتى بىناسىدىكى « تەشۋىقات ئەتىرىتى » قۇماندانلىق شتابىنى ئىزدىمەي ئۇدۇللا تارىخ فاكولتىتىغا كېلىپ ، « جيەن بوزەن مەخسۇس دىلۇ گۇرۇپپىسى » دىكى ھېلىقى كالتە بۇرۇت بىلەن كۆرۈشۈپ ، ئۆزىنىڭ سالاھىيەت كىنىشكىسىنى چىقىرىپ :
      __  مەن ليۇ شاۋچى دىلۇ گۇرۇپپىسىدىن . جيەن بوزەندىن تەكشۈرىدىغان جىددى ئىش بار . بۇ ماۋجۇشى سىلىڭبۇسىنىڭ سېتراتىگىيىلىك ئورۇنلاشتۇرۇشى  ، _  دېدى . ئۇ يەنە :
      __  بۇ ھەقتە سرىتقا قىلچىلىكمۇ شەپە چىقىرىشقا رۇخسەت يوق ، دەپ جىكىلىدى . شۇڭا جيەن بوزەن تاكى ئۆزىنى ئۆلتۈرىۋالغانغا قەدەر بېيجىڭ ئونۋېرسىتېتى ھەربىي ئىدارە قىلىش كومىتېتى ھېچقانداق ئىشتىن خەۋەر تاپالمىدى .
       ئارىدىن ھېچقانچە ۋاقىت ئۆتمەيلا كالتە بۇرۇتنىڭ يېتەكلىشى ئارقىسىدا ۋۇجۇڭ جيەن بوزەننىڭ ئۆيىگە كەلدى . بۇ يەرگە يۆتكىلىپ كەلگەندىن بۇيانقى ئىككى ئايدىن بۇيانقى دەم ئېلىش ، كۈتۈلۈش ئارقىلىق ياشانغان جيەن بوزەننىڭ سالامەتلىكى بىر ئاز ئوڭشىلىپ قالغانىدى .
       __ ۋۇجۇڭ جيەن بوزەننى كۆرۈپلا تەلەتىنى بۇزۇپ :
       __  جيەن بوزەن ئاڭلاپ تۇر : ليۇ شاۋچىنىڭ جىنايىتى ئېنىقلاندى ، مەركەز يەكۈن چىقاردى ، _ دېدى .
      جيەن بوزەن بىزار بولغان ھالدا :
      __  يەكۈن چىقارغان بولساڭلار ، مېنى ئىزدەپ نېمە قىلىسىلەر ؟ ! _ دېدى .
      ۋۇجۇڭ كۆزىنىڭ پاختىسىنى چىقىرىپ :
      __  ئۇ خائىن ، ساتقىن ، ئىشچىلار ئاسىيسى ، بۇ ئېچىلىش ئالدىدا تۇرغان « 9 - قۇرۇلتاي » دا ئېلان قىلىنىدۇ . سەن ماۋجۇشى ئىنقىلابى لۇشيەنى تەرەپتە تۇرامسەن ياكى ليۇشاۋچى تەرەپتە تۇرامسەن ، ئەمدىكى گەپ سىنىڭ ئۈزەڭدە قالدى ! _ دېدى .
       جيەن بوزەن ئۇلارغا ئارقىسىنى قىلىپ ، ئۆزىنى چوڭمۇ تۇتماي ، پەسمۇ چۈشۈرمەي :
       مەن ماركىسىزىمدا چىڭ تۇرغان توغرا رەھبەرلىك تەرەپتە تۇرىمەن ! _ دېدى .
       __ ھىم ؟ ! _- . . . ھۇھ  ئاناڭنى . . . گرانت كاللىسى بىلەن ئەۋرەت ئازگىلىغا كېرىدىغان جاھىل ئىپلاس ! _ كالتە بۇرۇت كۆرەڭلىگەن ھالدا ھەيۋە بىلەن مىس كالتەكنى كۆتۈردى . ۋۇجۇڭ كالتە بۇرۇتنى تۇتۇۋېلىپ ، جيەن بوزەنگە چۈشەندۈرۈپ : ھازىر ليۇ شاۋچىنىڭ 1935 _ ، 1936 _ يىلىدىكى گومىنداڭ بىلەن كومپارتىيىنىڭ نەنجىڭدە ئۆتكۈزگەن بىر نەچچە قېتىملىق سۆھبىتىدە گومىنداڭنىڭ CC ئېقىمىدىكى ئىشپىيۈنلىرى بىلەن تىل بىرىكتۈرۈپ ، ئازات رايۇنلارنى ئەمەلدىن قالدۇرۇشقا ، قىزىل ئارمىيىنى يوقىتىشقا ئۇرۇنغان « سۈيقەست » ىنى ئېنىقلىماقچىمىز ، _ دېدى . ئەمەلىيەتتە بولسا ، بۇ بىر نەچچە قېتىملىق سۆھبەتكە ماۋجۇشى ئۆزى رەھبەرلىك قىغانىدى . دەسلەپتە سۆھبەتكە قاتناشقان ۋەكىل جۇشياۋجۇ بولۇپ ، جيەن بوزەن ئالاقىچى بولغانىدى . ليۇ شاۋچى بۇ سۆھبەتلەرگە ئەسلا قاتناشمىغانىدى . ۋۇجۇڭ داۋاملىق تۈردە « ئەقىل كۆرسىتىپ » : مۇشۇنداق بىر ئىش بار دەپ ئىسپات بېرىپ ، ئۇنىڭغا قول قويۇپ بەرسەڭلا ئىشىڭ تۈگەيدۇ ، _ دېدى .
       جيەن بوزەننىڭ تېنى غۇژژىدە بولۇپ كەتتى .
      _-  شۇنداق بولسۇن ! ئويلىنىپ كۆرگىن ، _ دېدى ۋۇجۇڭ ، _ ساڭا خىزمەت كۆرسىتىپ گۇناھىڭنى يۇيۇش پۇرسىتى بەردىم . سېنىڭ جاۋابىڭنى كۈتىمەن ! _ گېپىنى دەپ بولۇپ ، تاماكىسىنى تۇتاشتۇرۇپ جيەن بوزەننىڭ ئالدىدا ئۇياقتىن بۇياققا ماڭدى . خۇرۇم ئاياق پولغا قاتتىق تېگىپ رېتىملىق ئاۋاز چىقىراتتى .
      20 مىنۇتچە ۋاقىت ئۆتكەندىن كېيىن جيەن بوزەن :
      _- ۋاقىت ئۇزۇراپ كەتكەچكە بۇنداق ئىشنىڭ بولغانلىقىنى ئېسىمگە ئالالمايۋاتىمەن ! _ دېدى .
      ۋۇجۇڭ تاماكىسىنىڭ قالدۇقىنى تاشلىۋېتىپ :
      __  ئېسىمگە ئالالمايۋاتىمەن ؟ ئالالمىساڭ بولمايدۇ ! ئېسىڭگە ئېلىشىڭ شەرت ، بۇ پارتىيىنىڭ ئېھتىياجى ! _ دېدى .
      ئۆيدە جىمجىتلىق ، يەنىلا جىمجىتلىق ھۆكۈم سۈرەتتى . ئاتمۇسفېرا ھەرىكەتتىن توختاپ قېتىپ قالغاندەكلا كۆرىنەتتى .
      ۋۇجۇڭ :
      __ بوپتۇ ، ساڭا ئۈچ كۈن مۆھلەت بېرەي ، ئەمدىغۇ يېتەر ؟ ! _ دېگىنىچە قارا خۇرۇم سومكىسىنى قولتۇغىغا قىستۇرۇپ چىقىپ كەتتى .
      ئۈچ كۈن ئىچىدە جيەن بوزەن ئۆتكەن ئىشلارنى بىر _ بىرلەپ ئەسلىدى . 1936 _ يىلى ئۇ ليۇ شاۋچى بىلەن كۆرۈشمىگەنىدى . ئۆزىنىڭ مەركەزنىڭ تېلېگرامما ئارقىلىق بەرگەن چاقىرىقىغا بىنائەن گومورو ، ماشۈلۇن ، خۇشېڭ قاتارلىقلار بىلەن بىللە 1949 _ يىلى 4 _يانۋار شياڭگاڭدىن ئەينى چاغدا مەركەز تۇرۇشلۇق جايغا كەلگەنلىكى ، ئەتىسى ليۇشاۋچىنىڭ ئۆزىنى تۇنجى قېتىم قوبۇل قىلغانلىقى ، ئۇنىڭ ئۆزىنىڭ تارىخ تەتقىقاتى جەھەتتىكى مۇۋەپپىقىيەتلىرىگە ناھايىتى يوقىرى باھا بەرگەنلىكى ، شۇنداقلا ئۆزىگە ياش ئوقۇغۇچىلارنىڭ ئالاھىدىلىكىگە ماس كېلىدىغان « جۇڭگو ئۇمۇمىي تارىخى » نى يېزىپ بېقىشنى جېكىلىگەنلىكى ئېسىدە ئېنىق ساقلانغانىدى . ئۇ بۇ ۋەزىنلىك ئۈمىدنى ئىزچىل تۈردە كۆڭلىدە ساقلاپ كەلگەنىدى . « مەدەنىيەت زور ئىنقىلابى » باشلانغان دەسلەپكى ۋاقىتقىچە بۇنى ھەرگىز ئېسىدىن چىقارمىغانىدى . . .  ئەجەبا ئۆزىنىڭ ھايات قېلىشىنى كۆزلەپ ئىنقىلاب داھىيىسىغا تۆھمەت قىلىش كېرەكمۇ ؟ ! ياق ، ئۇلارنىڭ ئاغزىمدىن قىلچىلىك نەپكە ئېرىشىمەن دەپ خام خىيال قىلمىسۇن . . .
       ئۈچىنچى كۈنى بولغاندا ۋۇجۇڭ راست دېگەندەك كېرىپ كەلدى - دە ، جيەن بوزەننىڭ  « ئېسىمگە ئالالمايۋاتىمەن  » دېگەبن گېپىنى ئاڭلاش بىلەنلا پۇتى كۆيگەن توخۇدەك سەكرەپ كەتتى . ئۇ جيەن بوزەننىڭ بۇرنىغا قولىنى تەڭلاپ تۇرۇپ :
      _-  ئېنىق تاپشۇرمايدىكەنسەن ، ئالدىڭدا ئۈلۈمدىن باشقا يول يوق ! _ دەپ ھۆكىرىگەندىن كېيىن :
      __  يەنە كېلىمەن ،! سەن ئېنىق تاپشۇرمىغىچە سىنى ھەرگىز قويۇۋەتمەيمەن ! _ دەپ چىقىپ كەتتى .  
      __  شۇنىڭدىن ئېتىبارەن راست دېگەندەك ئۇ ئىككى _ ئۈچ كۈندە بىر نۆۋەت كەلدى . ئۇنىڭ پەيلى بارغانسىرى يامانلاشتى . ھەر قېتىم كەلگەندە ئۈچ _ تۆت سائەت سۇراق قىلىپ ، بارغانسىرى قاتتىق قىستىدى . سۇراقتا ھەمرا بولغۇچى كالتە بۇرۇت زەھەرخەندە بىر نېمە بولغاچقا ، جيەن ئاقساقالنى دائىم قاتتىق كالتەكلىدى . بۇ رېزىنكا يۆگەلگەن ، باشمالتاقتەك توملۇقتىكى ، يېرىم مېتىردىن ئۇزۇنراق مىس كالتەك بولۇپ ، بەدەنگە ئۇرسا جاندىن ئۆتكىدەك تەگسىمۇ تېرىنى زەخىملەندۈرمەيتتى . بۇ _ كالتە بۇرۇتنىڭ ئىجادىيىتى ئىدى .
      ئەنە شۇنداق قىيناپ ئازابلاش نەتىجىسىدە ، ياشانغان ، ئاجىز جيەن بوزەننىڭ جېنى ئارانلا قالغان بولسىمۇ ،ئەمما ۋۇجۇڭنىڭ ھەر قانچە قىستىشىدىن قەتئىينەزەر ئۇ بىرەر ئېغىزمۇ يالغان گەپ قىلمىدى .
       2 _ دىكابر چۈشتە ۋۇجۇڭ يەنە كەلدى .
        جيەن بوزەننىڭ  يەنىلا « ئېسىمگە ئالالمايۋاتىمەن  » دېگەن گېپىنى ئاڭلاپ بىردىنلا يېنىدىن تاپانچىنى چىقىرىپ ، شىرە ئۈستىگە « ۋاققىدە » قويدى :
        __ بۈگۈن تاپشۇرمايدىكەنسەن ، ئاكاڭ قارىغاي مېڭەڭدىن تۈتۈن چىقىرىۋېتىدۇ !
       جيەن بوزەننىڭ چىرايى تامدەك ئاقىرىپ كەتتى .
       ۋۇجۇڭ ئالدىغا بىر قەدەم دېۋەيلەپ كېلىپ  تاپانچىنى جيەن بوزەننىڭ پىشانىسىگە چەنلەپ تۇرۇپ :
       __  ئېيتە ! تېز ئېيت ! ئېيتمىساڭ ئاكاڭ قارىغاي ھازىرلا سىنى ئېتىپ تاشلايمەن ! دەپ سۈرەن سالدى .
        توساتتىن يۈز بەرگەن بۇ ئىشتىن جيەن بوزەن ھاڭ - تاڭ قېتىپ قالدى . پۈتۈن ۋۇجۇدىغا تىتىرەك ئولاشتى ، بىراق ، ئۇ پاك ۋىژدانىنى ساقلاپ ئۈلۈشكە رازىمەنكى ، ئار - نومۇسنى سېتىش بەدىلىگە جاننى ساقلايدىغانلاردىن ھەرگىز بولمايمەن ، دەپ ئويلاپ ، باشقىلارغا تۆھمەت قىلىدىغان گەپنى باشتىن - ئاخىر ئاغزىدىن چىقارمىدى .
        ۋۇجۇڭ بىر ھازا ئەسەبىيلەشكەندىن كېيىن ھېچنېمىگە ئېرىشەلمەي كېتىپ قالدى .
        ئاشۇ كۈنى بىر كېچىنى ئەر _  خوتۇن ئىككىيلەن بىر _ بىرىگە قارىشىپ ، گەپ _ سۆز قىلىشماي تۈنەپ ئۆتكۈزدى .
       ئەتىسى جيەن بوزەن باشقۇرۇش خادىمى لاۋۋۇدىن ئۇيقۇم ياخشى ئەمەس دەپ ئۇيقۇ دورىسى سۇرىدى . باشقۇرۇش خادىمى ئۇنىڭغا بىر كۈنلۈك دورا بەردى . ئەمما جيەن بوزەننىڭ ئاش - تامىقى ۋە يېتىپ - قوپۇشىغا نازارەت قىلىدىغان ھېلىقى ئىشچى ئۇنىڭ دورىسىنى ۋاقتىدا ئىچىشىنى كۆزەتمىدى . شۇنىڭدىن كېيىنكى يېرىم ئايدا جيەن بوزەن ھەر كۈنى ئۇيقۇ دورىسى ئالدى . . . 
    ( داۋامى بار )

خەت چوڭلىقى : كىچىك نورمال چوڭ

 

      17 _  دىكابر قاتتىق قار يېغىۋاتاتتى . جيەن بوزەن ئەسكى جۇۋىسىنى كىيىپ ، قەلىمىنى تارمىدىن ئېلىپ بىر نېمە يازماقچى بولۇپ قەغەزگە بىر نەچچە قۇر جىجىلىدى . بىراق ، خەت چىقمىدى . رېزىنكىسىنى چىقىرىپ قارىسا ، قەلەمنىڭ سىياھى تۈگىگەنكەن . شۇنىڭ بىلەن ئۇ ھەسىرەت چىكىپ :
        _-  قەلەم بىلەن خەت يازسام خەت چىقمايۋاتىدۇ ، مەنمۇ تۈگىشىدىغان ئوخشىمامدىمەن ! _ دەپ تاشلىدى .
        __  سېنىڭ ئىدىيالىزىمىڭنى ھەر كۈنى تەنقىت قىلىۋاتساقمۇ ، __  دېدى ئۇنىڭغا نازارەت قىلىدىغان لاۋۋۇ ئۇنىڭ ھېلىقى گېپىنى ئاڭلاپ قېلىپ ، _  ئۆزگەرمەيدىكەنسەن ، يەنىلا ئىدىيالىزىمنى تەرغىپ قىلىۋاتىسەن !
       جيەن بوزەن ئۈندىمەي جىم تۇرىۋالدى .
       لاۋۋۇ يەنە :
