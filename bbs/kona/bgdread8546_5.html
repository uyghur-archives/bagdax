、ئەسسالامۇ ئەلەيكۇم قېرىنداشلىرىم مەن ئۆزەم بىر مۇخبېر .مەن بىزنىڭ بۇيەردە بولۇپ ئۆتكەن بىر ئىشنى سىلەرگە سۆزلەپ بىرەي بىزنىڭ بۇيەردىمۇ بۇ نەرسە بەك ئەۋەج ئېلىپ كەتكەن،ھەتتا لوبىنى ئشكىرىنىڭ چوڭ بىر شەھىرىدە بىر يىلدا تاپقان پۇلنى بىزنىڭ بۇ ئاران 40مىڭ نوپوسى بار كىچىككىنە يەردەئۈچ ئايدا تاپقانلىقى توغرىسىدا گەپمۇ بولۇنغان ،بىر كۇنى ئىدارىدىن ساقچىلا تۇيۇقسىز تەكشۈرۈش ئېلىپ بارىدىغانلىقى، بىزنىڭمۇ بىرگە بىرىشىمىز  توغۇرىسىدا  ئۇقتۇرۇش قىلدى شۇنىڭ بىلەن ئۇيەرگە بارساق يەنىلا شۇ بىزنىڭ مىللى بالىلار خەنلەر يوق دىيەرلىك .تەكشۈرۈش جەريانىدا ساقچىلار نۇرغۇن پۇللارنى مۇسادىرە قىلىپ ھەم ئۇيەرنىڭ بىر،ئىككى خىزمەتچىسىنى ئېلىپ كېتىپ ئۇيەرنىمۇ تاقىتىۋەتكەن.  لېكىن ئەتىسى يەنە شۇ ساقچى ئىدارىسىنىڭ بىر تېلفۇنى بىلەن تېلىۋىزوردىمۇ ئېلان قىلمىدۇق  ئىككى كۈندىن كىيىن يەنە ئوخشاش ئېچىۋەدى.

بۇ قانداق گەپ بوپ كەتتى مەيخور ئەپەندى تىمىنى يوللاپ قۇيۇپ ئۇزڭىز يۇقاپ كەتتىڭىز مۇ ؟

-------------------------
مەن بار نىدا ،چۈشتىن كىيىن شىركەتتە بەك ئالدىراش ئىشلەپ كەتتۇق.مەن بۇ ئىشنىي ھەرگىز تاشلاپ قويمايمەن.
يۇقىرىدا بەك ياخشى ئىنكاسلار چۈشۈپتۇ.خۇرسەن بولدۇم.ئەمما يەنىلا يىغلىغۇم كەلدى.بۇ تىمىنى يوللىغاندىن بۇيان يىراق يىقىندىكى دوستلىرىم مۇشۇ تىما ھەققىدە تىلىاۇن قىلدى.خەت يازدى.بۈگۈن بىر غۇلجىلىق ئاغىنەم تىلىاۇن قىلىپ ئۆزىنىڭ بىر ئاي بۇرۇن 18 مىڭ يۇەن ئۇتتۇرىۋەتكەنلىكىنى ئىيتتى.ئۇنىڭغا نىمە دىيىشىمنى بىلەلمىدىم.
     ئويلاپ باقايلى قىرىنداشلار بىز ئۆزىمىزنىڭ نىمىمىزگە ئىگە بولالايمىز.خوتۇن قىزلىرىمىز چوڭ شەھەرلەردە يات مىللەتنىڭ ياكى چەتئەللىكنىڭ تىگىدە ياتسا،يىگىتلىرىمىز ئەسلى-ۋەسلىنى ياتلار ئاچقان قىمارخانىغا سىلىپ بەرسە.بالىلىرىمىز ئاچ-يالىڭاچ ئۇقۇشسىز قىلىپ بالا جىنايەتچىلەرگە ئايلانسا.قىرى ئاتا -ئانىمىز تېخى غىرىبچىلىقتا.

بىزنىڭ يېزىدىمۇ تۆتى بار،ساپلا ئۇيغۇر بالىلار،مەن پات-پات ئىشىكتىن ئويناۋاتقان ياشلارنى كۆرۈپ قويىمەن،لېكىن،ئۇ نەرسىگە قاراشقىمۇ چولام يوق.
60ياشلىق بىر ئادەم بار،ھەر كۈنى شۇ ئېلىكتۇرۇنلۇق ئۇيۇنچۇق ئوينايدىغان،ئىچىگە ئادەم پاتماي قالىدۇ،بۈگۈن بىر مۇھىم ئىش بىلەن بىزنىڭ يېزا بىلەن ئارىلىقى 3كېلۇمىتىر كېلىدىغان دېلىۋىزىيەنىڭ بازىرىغا بېرىپ قالدىم،ئۇ يەردە 3-4سائەت چاممىسىدا تۇرۇپ قالدىم،خەنزۇلار تېراكتۇر ئوڭشىتىش،ئۇغۇت سېتىۋىلىش بىلەن ئالدىراش،ئۇيۇن ئوينايدىغان،تاماشا يىغىلدىغان يەردىن پەقەت خەنزۇ بازىرىدا ماجاڭخانىدىن بىرنىلا كۆرۈپتىمەن،لېكىن،ئەتراپتا ئېلىكتۇرۇنلۇق ئۇيۇنچۇقخانا،تورخانا،بېليارتخانا دېگەن نەرسىلەرمۇ يوق ئىكەن،خەنزۇلار ئۆز ئىشى بىلەن ئالدىراش ئىكەن.
بىزنىڭ يېزا بازىرىدا تورخانىدىن 3بار،بۇنداق مۇنبەر كۆردىغان ئىش ئەسلا يوق،ساپلا ئۇيۇن ئويناش بىلەن ئاۋارە ھەممە ياشلىرىمىز ئۇيۇن ئوينايدۇياكى مۇڭداشقۇدا رومانتىك مۇھەببەت قۇربانى بولغانلار،ئېلىكتۇرۇنلۇق ئۇيۇنچۇقخانىدىن تۆتى بار،ئادەم پاتمايلا قالىدۇ،كەينى تەرەپتە خېلى كۆپ بېليارت(كالاڭچۇر)دەمدۇ،شۇ نەرسە بەك تولا،ئۇ يەردىمۇ نۇرغۇن بىكارچى ياشلىرىمىز بار،تاماكىنى قىڭغىر چىشلەپ،كالتەكنى تۇتۇپ،كۆك دۇخاۋىنىڭ ئۈستىدە يۇمىلاق تاش سوقىدىغان ئەزىمەتلەر،كۈندە بىكارچىلىقتا قىلدىغانغا ئىش تاپالماي،مۇشۇ نەرسىنى ئوينايدۇ.
40ياشتىن ئاشقان مەن تۇنۇيدىغان 20دەك ئادەملىرىمىز بار،ئۇلار ھەر كۈنى بىر خەنزۇنىڭ دۇكىنىغا كىرىپ،ئاق ھاراق ئىچىدۇ،پۇلى قالمىسا،1.5يۈەنلىك(二锅头（دەپ بىر ھاراق بار،شۇنى ئىچىدۇ.
بىر ئىككى قەۋەتلىك بىر بىنا بار،2-قەۋەت داڭلىق قىمارخانا سورۇنى،10-20يۈەن ئوينايدىغان ئىش يوق،ئەڭ تۆۋەن بولسا،1000سوم،يۇقىرسى 100مىڭ يۈەنگىچە ئوينايدۇ.نۇرغۇن كىشىلەر ئۆي-مۈلىكىدىن،يېرىدىن ئايرىلدى،خۇتۇنى تاشلىۋاتقان ئەرلىرىمىزمۇ بار.
يېزا بازىرىغا بارسام،شۇ يەردە بىكارچىلىقتا گەپ ساتىدىغان نۇرغۇن گۆدەك ياشلار،ئەتە قانداق ئۆي تۇتارمەن،مەن نېمىش قىلىۋاتىمەن،ئاتا-ئانام ماڭا مەڭگۈ باقىۋەندە ئەمەسقۇ دەپ ئويلايدىغانلار بەكمۇ-بەك ئاز.
ئاتا-ئانىنىڭ بوشلۇقى،ئىشتا ھۇرۇنلۇقى بالىنىڭ مىجەزىدە ھەرخىل خارەكتىرنى شەكىللەندۈردىكەن.نۇرغۇن يەرلەر خەنزۇنىڭ قولىغا ئۆتتى.
ئاددىيسى،بىزنىڭ مەھەللىدىمۇ ھازىر %60يەر خەنزۇنىڭ قولىدا،كېيىن،شۇلارنىڭ قولىدا ئىشلەيدىغان ئىشلارمۇ بولدىغان ئوخشايدۇ.
بىزنىڭ يېزا گەپنىڭ ئۇچۇقى،ئۇيغۇرلار يېرىنى سېتىپ،ئۇيۇن-تاماشا،ھازىرقى باياشاتچىلىق،ئېسىل ماشىنا ئېلىپ،كىرا قىپ پۇل تېپىش....
بۇ يەردىكى كۆچمەن خەنزۇلارنىڭ ئويلايدىغانى كېيىنكى ئەۋلادلىرىم ئۈچۈن يەر ئالىمەن،تېرىقچىلىق قىلىمەن،ئىگىلىك تىكلەيمەن،كېيىنكى ئەۋلادلىرىمنىڭ بەختى مېنىڭ مەڭگۈلۈك ئارمىنىم........
شۇ يېرىنى سېتىۋاتقان ئۇيغۇرلىرىمىز گەنسۇنىڭ نەچچە مىڭ مېتىر ئېگىزلىككە ئەپكەش بىلەن سۇ تۇشۇپ،تاغنىڭ ئۈستىگە تېرىقچىلىق قىلىۋاتقىننى كۆرگەن بولسا،ئازراق ئويغىنىش بولارمىكىن دەيمەن....مەن شۇ گەنسۇ ۋە ئىچكىردىكى خەنزۇ دېھقانلىرىنىڭ ھەقىقىي ياشاش مۇھىتىنى كۆرگەندىن كېيىن،نېمە ئۈچۈن شىنجاڭغا كېلىپ بېيىپ كەتكەنلىكى،قانداق ياشىسىا بولدىغانلىقى توغۇرلۇق كۆپ ئويلاندىم.شۇڭا،خەنزۇلار ئىچكىردىن كەلسە،بۇ يەرنى جەننەتتەك كۆردىغىننى ھەقىقىي چۈشىنىپ يەتتىم.

مىنىڭ بىلىشىمچە، بۇ خىلدىكى ئىلىكتىرونلۇق ئويۇنچۇقخانىلار قارا  گورۇھ  تۈسىنى ئالغان.  پۈتۈن مەملىكەت مىقياسىنى ئىزدىسىڭىز بۇ ئىشنىڭ بېشى ئىككى ئۈچ  ئادەمگە باغلىنىشى مۇمكىن. ئاشۇ يەرگە دائىم كىرىپ-چىقىدىغانلاردىن ئىگىلىنىشىچە، ناھىيە بازىرىغا جايلىشىۋالغان بىر ئويۇنچۇقخانىنىڭ ئايلىق كىرىمى نەچچە مىليۇن، ھەتتا ئۇنىڭدىنمۇ ئېشىپ كېتىدىكەن.  بىز تەرەپلەردىمۇ بۇ نىجىسنىڭ قولىدا ۋەيران بولغانلار ھەم بولىۋاتقانلار ئاز ئەمەس. ئەڭ يامان بولغىنى ئاشۇلارنىڭ شۈلۈپ يېگىنى ئازلىق قىلىۋاتقاندەك، ئۆز قەۋمىمىزدىن  بىر تۈركۈم جازانىخورلار يېتىشىپ چىقىپ، بىلىنى رۇسلىيالمايۋاتقان بىچارە (ئەقلىگە پوق يېگەن) قىمارۋازلارنى ئەجەل پاتقىقىغا پاتۇرماقتا. (جازانىخوردىن قەرز ئالغان بىر تونۇشۇمنىڭ ئېيتىپ بىرىشىچە، مىڭ سومنىڭ كۈنلۈك ئۆسۈمى 100 يۈەن ئەتراپىدا بولارمىش).
قانۇن ئورۇنلىرى بىر نەچچە قېتىم ئومۇميۈزلۈك ھەرىكەت قوزغاپ بۇنداق چىكىدىن ئاشقان سورۇنلارنى تاقىتىۋەتكەن. لېكىن، ھەيران قالارلىق يېرى شۇكى ئۇلار بىر نەچچە كۈن ئۆتەر-ئۆتمەيلا يەنە تىجارىتىنى ئەسلىگە كەلتۈرىۋالغان مىساللارمۇ ناھايىتىمۇ كۆپ.
ئۇنىڭدىن باشقا، ئاشۇ يەرلەردە پات-پات جىدەل-ماجرا چىقىپ تۇرىدۇ. قانۇن ئورۇنلىرى بۇ ئىشقا ئارلىشىدۇ.  تېخىمۇ ھەيران قالارلىق يېرى، بىر نەچچە كۈن ئۆتكەندە، دۇكان ئۆز پېتىچە قىلىپ، ئۇنىڭ خوجايىنلىرى ھەم قوغدىغۇچىلىرى پۈتۈنلەي ئالمىشىپ بولىدىكەن.
مېنىڭ سىرتتىن بېلىشىمچە، بۇ مانا مۇشۇنداق سىرلىق بىر قورغان. بۇنىڭ كەينىدە قانداق ئويۇن ئوينىلىۋاتىدۇ؟ بۇلارغا كىم باشچىلىق قىلىۋاتىدۇ؟  يەنە قانداقتۇر  ئاللا كىملەر يانتاياق بولۇپ، قانات ئاستىغا ئېلىۋاتىدۇ؟ يەنە كېملەر ئۇلارنىڭ يولىنى راۋان قىلىۋاتىدۇ؟ بۇ ھازىرچە ھەممىگە قاراڭغۇ.
شۇڭا، مېنىڭ ئۈمىدىم، ئالدى بىلەن ئاشۇ قىمار سورۇنلىرىغا ئۆگىنىپ قالغان دوسلىرىمىزنى، ئۇرۇق-تۇغقانلىرىمىزنى، خانىم-قىزلىرىمىز ئەرلىرىنى قايتۇرۇپ كېلەلىگەندە، ئۇلارمۇ ئاستا ئۆز يولىغا راۋان بولىشى مۇمكىن.
ئەگەر، ھۆكۈمەت دائىرىلىرىنىڭ دىققىتىنى قوزغايمىز دەيدىكەنمىز، چوقۇم تېخىمۇ يۇقىرى دەرىجىلىك ھۆكۈمەت ئاپپاراتلىرىنىڭ دىققىتىنى قوزغىشىمىز كىرەك.
گۆھەرگە سەدەپ لازىم، ئادەمگە ئەدەب.
