مۇشۇ مۇسابىقە باھانىسى بىلەن شىنجاڭدىمۇ پۈتۈن شىنجاڭ بويىچە ھەر يىلى پۇتبول بىرلەشمە مۇسابىقىسى ئۆتكۈزۈپ تۇرسا بولىدىكەن، ھەر قايسى ۋىلايەت شەھەرلەردىن ئاز دېگەندىمۇ بىردىن كوماندا قاتناشقان بولسا، ھەرقايسى شەھەرلەر ئوتتۇرىسىدا تېخنىكا ئالماشتۇرۇش كۈچۈيۈپ، شىنجاڭ پۇتبولچىلىرىنىڭ كەسپى سەۋىيىسىنىڭ يوقۇرى كۆتۈرىلىشىگە چوڭ ياردىمى بولۇپ، شىنجاڭنىڭ پۇتبولچىلىقىدىكى بىر چوڭ ئىلگىرلەش بولۇپ قالاتتى.
ئىقتىساد مەسىلىسىدە ھۆكۆمىتىمىز باش بولۇپ چوڭ ياردەم قىلغان بولسا (مەركەزدىن شىنجاڭ پۇتبولچىلىقى ئۈچۈن ئىلتىماس قىلسا تەستىقلىنىپ قېلىنىشىنىمۇ يوق دېگىلى بولمايدۇ، مەركەزدە پۇل ئازمىتى!)  ئاندىن پۈتۈن شىنجاڭ بويىچە ئەڭ ياخشى ماھىرلارنى تاللاپ بىر كوماندا قۇرۇپ، شىنجاڭغا ۋەكالىتەن "جۇڭگو دەرىجىدىن تاشقىرى بىرلەشمە مۇسابىقىسى" ئويناشنى قولغا كەلتۈرگەن بولسا، ئاندىن ئۆزىمىزنىڭ ھەقىقى تالانتىمىز بارمۇ-يوقلىقىنى پۈتۈن جۇڭگو شۇنداقلا دۇنياغا ئىسپاتلاپ بېرەلىگەن بولاتتۇق. بىز پەقەت ئاشۇنداق سەھنىلەردىلا ئۆزىمىزنى نامايەن قىلالايمىز!
ئازراق خام خىيال قىلىپ قويدۇم، تورداشلارنىڭ قانداق خىياللىرى باركىن؟

توۋا دەيمەن مۇشۇ مۇنبەر باشقۇرغۇچىلارغا مۇشۇنداق ياخشى پۇتبول خەۋەرلىرىنى ۋاقتى-ۋاقتىدا ئېلان قىلىپ تۇرمامدۇ دەيمەن!بىزغۇ يولنى يىراق كۆرمەي چىقىپ مۇسابىقە كۆرۈپ كەلدۇق،لېكىن قومۇلدا خەۋەر كۈتۈپ تۇرغان قېرىنداشلىرىمىز نۇرغۇن ئەمەسمۇ!
يەنە بىر گەپ قومۇل بىرلەشمە مۇسابىقىسىمۇ شۇنداق ياخشى ئېلىپ بېرىلىۋاتىدۇ يا بۇ مۇسابىقىنىڭ تۈزۈك بىر خەۋىرى يوق،<<يىلتىز >>جەمىيىتىدىكلەردىن ئۇقۇشىمىز چە مۇسابىقە خەۋەرلىرىنى يوللاپ تۇرۇپتۇ بولمىسا،مۇشۇنداق تۇرۇپمۇ رازى بولغىدەك بىر خەۋەر يوق.بۇئىشلارنى كىمدىن كۆرىمىز؟يەنىلا شۇ مۇنبەر باشقۇرغۇچىلاردىن سورايمىزغۇ!باغداش تورى قومۇل دىيارىدا خېلى نوپۇزلۇق دەپ قارايمەن ئەجەبا سىلەرنىڭ قومۇل پۇتبولچىلىقى توغرىسىدىكى بۇخەۋەرلەر بىلەن ئانچە بەك خوشۇڭلار يوقمۇ-يا؟پۇتبول خەۋەرلىرىنىڭ تەڭدىن تولىسى <<جەسسۇر>>كوماندىسى توغرىسىدا،گەپنىڭ سېسىغىنى قىلغاندا <<جەسسۇر>> كوماندىسى مۇشۇ توردا تەشۋىق مۇسابىقىسى قىلغاندىن باشقا پۇتبول سۆيەرلەرگە نىمە قىلىپ بەردى؟قايسى كۈندىكى قاتتىق بوراننى قايرىپ قويۇپ پۈتۈن كۈن كۆرگەن مۇسابىقەلەر،ئەتىيازنىڭ ئىسسىقىدا كۈنگە كۆيۈپ تۇرۇپ مۇسابىقە كۆرگەن كۆرىنىشلەرنى مۇشۇ باغداش تورىدا كۆرگەن بولساقتى ئازراق بولسىمۇ ھاردۇقىمىز چىقىپ كۆڭلىمىز ئارام تېپىپ قالغان بولاتتى!قاچانلا تورغا چىقىپ قارىساق ئۇنى ماختاپ بۇنى چۇشۇرگەن، پۇت تېپىشىۋاتقان كۆرىنىشلەر،ئەجەبا بىزنىڭ ئويىمىزدىكى پۇتبول مۇشۇنداقمىتتى؟ئەجەبا باغداش تورىدا پۇتبول توغرىسىدا مەخسۇس بىر سەھىپە ئېچىش بەسى مۈشكۈلمۇ؟!باشقىسىنى قويۇپ تۇرغاندىمۇ قومۇلدا ئېلىپ بېرىلىۋاتقان بىرلەشمە مۇسابىقىنى كۆرۈپ تۇرۇش ھەممىزنىڭ ئارزۇسىغۇ!!!
قوشۇمچە ئېيتىپ قويماقچى <<تۇرپاندىكى پوتبول مۇسابىقى باشلىنىدۇ>> دىگەن تېمىدا  ﻗﻮﻣﯘل «ﺟﻪﺳﯘر» ﭘﯘﺗﺒﻮل ﻛﯘﻟﯘﺑﻰ ۋە «ﻳﯩﻠﺘﯩﺰ» ﭘﯘﺗﺒﻮل ﻛﯘﻟﯘﺑﻰ دەپ ئېلىنغان. بۇ ئەسلى<<يىلتىزپۇتبول جەميىتى>>ۋەكىللەر كوماندىسى ۋە <<جەسسۇر>>(<<جەسۇر>>ئەمەس)
كوماندىسى دىگەن تەرتىپتە كىلىشى كېرەك.چۈنكى ئالدىنقىسى 11كوماندىدىن قۇرۇلغان قانۇنىي تەشكىلاتتىنىڭ  بىرلەشمە كوماندىسى، دەرىجىسى يوقىرىسى ئالدىدا كىلىشى كېرەك.

مەنمۇ مۇسابىقە نەتىجىسىگە تۆت كۆزۈم بىلەن تەشنا ئىدىم. يە يوللانمىدى بۇ نەتىجىسى. قارىسام ھۇررا تورىدا مۇسابىقە نەتىجىسى باركەن، يوللاپ قويدۇم.قارىسام قومۇل كوماندىلىرى بەك ئۇتتۇرۋەتكەندەك قىلىدۇ،ئۇتۇش -ئۇتتۇرۇش دېگەن بولۇپ تۇرىدۇ، مۇھىمى تەجرىبە - ساۋاقلارنى يەكۈنلەش.


ئەسسالاممۇ ئەلەيكۇم ، ھۇررا ئەھلى :
                كەڭ خەلق ئاممىسىنىڭ ﺋﯩﺸﺘﯩﻦ ﺳﯩﺮﺗﻘﻰ ﻣﻪﻧﯩﯟى ﺗﯘرﻣﯘﺷﯩﻨﻰ ﻳﻪﻧﯩﻤﯘ ﺑﯧﻴﯩﺘﯩﺶ، ﻳﺎﺷلار - ﺋﯚﺳﻤﯜرﻟﻪرﻧﯩﯔ ﺳﺎﻏلاﻣﻠﯩﻖ ﺳﺎﭘﺎﺳﯩﻨﻰ ﻳﻪﻧﯩﻤﯘ ﺋﺎﺷﯘرۇش ﺋﯜﭼﯜن، ﻣﺎي ﺑﺎﻳﺮﯨﻤﻰ ﻣﻪزﮔﯩﻠﯩﺪە  تۇرپان ۋىلايىتىدە «ﭘﻪﺗﺘﺎر ﺋﻪﺳﻠﯩﮭﻪﻟﯩﻚ ﻳﯧﺰا ﺋﯩﮕﯩﻠﯩﻜﻰ ﻟﻮﯕﻘﯩﺴﻰ» 12 - ﻧﯚۋەﺗﻠﯩﻚ ﻳﺎﺷلار ﭘﯘﺗﺒﻮل ﻣﯘﺳﺎﺑﯩﻘﯩﺴﻰ ئۆتكۈزۈلدى . مۇسابىقە ئورۇنلاشتۇرۇشى بويىچە مۇسابىقە 28-ئاپرىل باشلايدىغان بولغاچقا ، مۇسابىقىگە قاتنشىىدىغان 8 كوماندا 27-ئاپرىل كەچ تۇرپان دىيارىغا پەيدا بولۇپ بولدى ، ئەڭ ئاخرىدا ھۇررا كۇلۇبى بىلەن قۇمۇل يىلتىز كوماندىسىنىڭ مىھمانخانىغا چۈشىشى بىلەن كۇلۇب ۋە كوماندا يىتەكچىلىرى يىغىنى ئېچىلىپ مۇسابىقە ، تاماق ۋە ياتاق ئورۇنلاشتۇرۇشلىرى سۆزلەندى ھەم ئۆزئارا  پىكىرلىشىپ ، بەزى نۇقتىلاردىكى مەسلىلەرنى ھەل قىلىپ ، يىغىن مۇۋاپىقىيەتلىك ئاياقلاشتى . ئاخىرىدا كوماندا يىتەكچىلىرى چەك تارتىپ ، كوماندىلار گۇرۇپقا ئايرىلدى ، بىزنىڭ كوماندا ئا گۇرۇپغا بۆلىندى . نەتىجىسى تۆۋەندىكىچە :
ئا گۇرۇپ :    ھۇررا كۇلۇبى ، تۇرپان بارمان كوماندىسى ، قۇمۇل جەسۇر كۇلۇبى ۋە غۇلجا ئېلى يۇلتۇز
ب گۇرۇپ : شىنجاڭ ئونۋىرسىتىتى ، قۇمۇل يىلتىز كوماندىسى ، قاراماي دولان ، تۇرپان ئىتتىپاق
بۇ نەتىجە بىز ئۈچۈن بولۇپمۇ ئا گۇرۇپ ئۈچۈن پايدىلىق بولغىنى شۇكى 4 كۇنلۇك مۇسابىقىدە ب گۇرۇپتىكىلەر بەزى كۈنلىرى بىر كۈندە 2 مەيدان مۇسابىقە ئوينايتى ، لېكىن مۇسابىقە ۋاقتىنى 90 مىنۇتتىن 80 مىنۇتقا ئۆزگەرتكەن ئىدى  ، ە ئەڭ دەھشەت يېرى يېرىم ھەل قىلغۇچ بىلەن ھەل قىلغۇچ مۇسابىقە بىر كۈندىلا ئېلىپ بېرىلەتتى . بۇ ئورۇنلاشتۇرۇش كۆپ يىتەكچىلەرگە بېسىم بولدى شۇنداقدىمۇ ھەممىسى رازىلىق بېرىشتى . يىغىن تۈگىگەندە كەچ سائەت 23:00 بولغان بولسىمۇ  لېكىن كوماندا يىتەكچىلىرى ياتاقلىرىغا قايتىشىپ كوماندا ئەزالىرى يىغىنلار ئېچىپ ، تۇرپاندىكى بارلىق پائالىيەت ۋە مۇسابىقىلەرنى ئورۇنلاشتۇرۇپ ، دەم ئېلىشتى ......
28-ئاپرىل مۇسابىقە كۈنى . بۇ كۈنى ئا گۇرۇپتىكىلەر مۇسابىقە چۈشەتتى ، ھۇررا كۇلۇبىمىز ئېچىلىش مۇراسىمىدىن كىيىنكى بىرىنچى مەيدان مۇسابىقىنى غۇلجا ئىلى يۇلتۇزى كوماندىسى بىلەن ئوينايتتى . چۈشتىن بۇرۇنقى دەم ئىلىشلار ئاياقلىشىپ چۈشتىن كىيىن 2:00 دە  «ﭘﻪﺗﺘﺎر ﺋﻪﺳﻠﯩﮭﻪﻟﯩﻚ ﻳﯧﺰا ﺋﯩﮕﯩﻠﯩﻜﻰ ﻟﻮﯕﻘﯩﺴﻰ» 12 - ﻧﯚۋەﺗﻠﯩﻚ ﻳﺎﺷلار ﭘﯘﺗﺒﻮل ﻣﯘﺳﺎﺑﯩﻘﯩﺴنىڭ ئېچىلىش مۇراسىمى ئېلىپ بېرىلدى ھەرقايسى كوماندىلار  پاراتتىن ئۆتۈپ ، باشلىقلار ۋە مەرىپەتۋەرلەر سۆز قىلغاندى ، ئاندىن دولان كوماندىسى ، ئىلى يۇلتۇزى كوماندىسى قاتارلىق كوماندا ۋە شەخسىلەر تۇرپاندا يېقىندا يۈز بەرگەن بوران ئاپىتىگە ئىئانە قىلىشتى ، دولان كۇلۇبى تۇرپانلىق پۇتبۇل خانىشى ئەقىدەگە 1000 سوم قىممىتىدىكى  پۇتبۇل كىيىمى ۋە  قوراللىرىنى بەردى . ئېچىلىش مۇراسىمى شۇنداق داغدۇغىلىق ئاياقلشىپ ، مۇسابىقە باشلاندى .
ھۇررا كۇلۇبى كوماندىسى مەشىقمۇ قىلالماي ، ئالدىراپلا تۇرپانغا بارغىنى بىرىنچى مەيداندىلا ئۆزىنى كۆرسەتتى ، بىزنىڭ كوماندا ئەزالىرى قىزىپ ئېچىلىپ بولغىچە ئىلى يۇلتۇزى كوماندىسى توپ ئۇرۇپ بىزنىڭ بالىلارنى ئالدىرتىپ قويدى ، غۇلجا كوماندىسىنىڭ ھۇجۇمى كۆپ ئەمەس لېكىن بىكلىرى بەك ياخشى ئوينىدى ، بىزنىڭ سەپ شەكلىنىڭ خاتا ئورۇنلاشتۇرغانلىقىدىن ئوتتۇرا مەيدان ئاجىزلىشىپ ، 80 مىنۇت بېسىپ ئوينىساقمۇ لېكىن ئالدىغا ياخشى توپ كەلمەي ، نەتىجىدە مۇسابىقىنى 2-1 ئۇتتۇرۇپ قويدۇق ، بۇ نەتىجە كوماندا ئەزالىرىغا بەك ئېغىر كەلدى . ئىككىنچى كۈندىكى مۇسابىقىدە بىز قۇمۇل جەسۇر كۇلۇبى بىلەن ئۇچراشتۇق ، بىز گۇرۇپتىن چىقىش ئۈچۈن ھەمدە ئارتۇق توپ يىغىش ئۈچۈن قاتتىق ھۇجۇم ئويناپ ، 9-0 نەتىجە بىلەن قۇمۇل جەسۇر كۇلۇبى ئۇتتۇق ، كوماندىمىزنىڭ ھۇجۇمچىسى تاھىر 5 توپ ئۇرۇپ قويدى ، لېكىن يولۋاس ياردەمچى پاستىن ئالتىنى بەردى . ئۈچىنچى كۈنىدىكى مۇسابىقىدە ساھىپخان كوماندا تۇرپان بارمان كوماندىسى بىلەن قاتتىق چىشلىشى ئوينىدۇق بۇ كوماندا بۇلتۇر  بىز بىلەن چىمپىيۇنلۇق تالاشقان كوماندا ئىدى گەرچە بىز كۆپ ھۇجۇم قىلغان بولساقمۇ لېكىن توپ ئۇرالماي قالدۇق ، تاكى مۇسابىقە 75 مىنۇتقا كەلگەندە يولۋاس ۋە ئىليار تولىستەكلا يالغۇز توپتىن ئىككىنى ۋارتانى ئالداپ ئۆتۈپ ئۇرۇپ ، نەتىجىنى 2-0 قىلىپ ئاياقلاشتۇرۇپ ، ئا گۇرۇپپىدىن ئىككىنچى بولۇپ چىقىپ ، يېرىم ھەل قىلغۇچتا ئەڭ كۈچلۈك كوماندا دەپ قارالغان قاراماي كوماندىسى بىلەن ئۇچرىشىپ قالدۇق ، ئىككىلا كوماندا بىر بىرى بىلەن بەكلا تونۇش يەنە كىلىپ  بىزنىڭ كوماندىدا يولۋاس ، ئىليار ۋە .ئىنايەت قاتارلىق  قارامايلىق 3 يىگىت بولغاچقا بۇ مۇسابىقىنىڭ ئوت ئۇچقۇنى تېخىمۇ ئۇلغۇيۇپ كەتتى ، يەنە كېلىپ قاراماي دولان كوماندىسى بارلىق تەييارلىقنى تۇلۇق قىلىپ كەلگەن ئىدى ، بىزنىڭ كوماندىدا يارلىنىش كۆپ بولسىمۇ لېكىن يەنىلا ياخشىراق سەپ شەكلىنى تەشكىللەپ ئوينىدۇق ، مۇسابىقە يېرىم مەيدان 0-0 ئاياقلاشقان بولسىمۇ ، كىيىنكى مەيدان رىپىرنىڭ ئوتتۇرا بىكىمىزگە بەرگەن ئارتۇق سېرىق كارتىسى بىلەن 10 ئادەم ئوينىدۇق ، بۇنىڭ بىلەن ئۇلارنىڭ ئىككى يان ھۇجۇمچىسى كۆپ قىتىم بىكلەردىن بېسىپ ئۆتۈپ ، توپ ئۇرىۋالدى شۇنىڭ بىلەن بىز ئەڭ كۈچلۈك كوماندا قارماي كوماندىسىغا تۇنجى قىتىم ئۇتتۇرۇپ قويدۇق ، بۇ نەتىجىگە ھۇررا كۇلۇبىدىكى بارلىق ئەزالار قاتتىق نارازى بولدى لېكىن چۈشتىن كىيىن ئۈچىنچىلىكنى تالىشىدىغان مۇسابىقە بار بولغاچقا يەنىلا كەيپىياتنى تەڭشەپ ، 2 سائەتتىن كىيىن يەنە مەيدانغا يىتىپ كەلدۇق .  ب گۇرۇپتىن چىققان شىنجاڭ ئونۋىرسىتىتى (ئىزچىلار )كوماندىسى غۇلجا ئىلى يۇلتۇزى كوماندىسىغا ئۇتتۇرۇپ قويغاچقا ، ئۈچىنچىلىكنى تالىشىدىغان مۇسابىقە ھۇررا كۇلۇبى بىلەن شىنجاڭ ئونۋىرسىتىتى (ئىزچىلار ) كوماندىسى  ئوتتۇرىسىدا ئېلىپ بېرىلدى ، ئىككىلا تەرەپ چۈشتىن بۇرۇن مۇسابىقە ئوينىغاچقا ، مەيدان بەكمۇ كەسكىن ئەمەس ئىدى ئۇنى ئاز دەپ يارىدار بەكلا كۆپ بولۇپ كەتتى ، ئۇنى بۇنى قىلىپ يۈرۈپ ، ئاران بىر نۇقتا توپقا ئېرىشسەك ھۇجۇمچىمىز تاھىر توپنى باراڭغا تىپىۋەتتى ..... مۇسابىقە يەنە داۋاملىق ئېلىپ بېرىلىپ ، 70 مىنۇت بولغاندا سول كىراي ئىنايەت ياردامچى پاستىن بىرنى بەردى ئوڭ كىراي يولۋاس توپنى بىكتىن بۇرۇن ئېلىپ ۋارتارغا ئۇرۇپ ، نەتىجىنى 1-0 قىلىپ ئاياقلاشتۇردى . شۇنداق قىلىپ ھۇررا كۇلۇبى بۇ قىتىملىق مۇسابىقىدە ئۈچىنچى بولدى . ھەل قىلغۇچ مۇسابىقە قاراماي دولان كوماندىسى بىلەن ئىلى يۇلتۇز كوماندىسى ئوتتۇرىسىدا دەھشەت كەسكىن ئېلىپ بېرىلدى ، دولان كوماندا ئەزالىرىنىڭ ماسلىشىشچانلىقى بەك ياخشى  ، ئىلى يۇلتۇزى كوماندىسىنىڭ ئەزالىرى ھەم چىڭ ھەم پۇت ئاستى ياخشى بولغاچقا ، ئىككىلا تەرەپ ھۇجۇم ۋە مۇدائىپىيىنى ياخشى قىلىپ ، ئەڭ ئاخىرىدا 2-2 بولۇپ ئاياقلاشتى ، ئىلى يۇلتۇزى كوماندىسنىڭ ئەزاسى بىر ئىختىيارى توپنى بەك چىرايلىق ئۇرۇپ ، تۇرپان خەلقىنى ھەيران قالدۇردى .....بۇ قىتىملىق مۇسابىقە قائىدىسى بويىچە ھەل قىلغۇچ مۇسابىقىدىكى 80 مىنۇتتا تەڭلىشىپ قالسا بىۋاستە نۇقتا توپ تىپىدىغان بولغان ، بۇ  قاراماي كوماندىسى ئۈچۈن بەك پايدىلىق ئىدى چۈنكى ئۇلارنىڭ ۋارتارسى شىنجاڭ كوماندىسىنىڭ ۋارتارى زۇليار ئىدى ، دىگەندەك زۇليار 3 نۇقتا  توپنى تۇسۇپ قاراماي دولان كوماندىسىنىڭ چىمپىيۇن بولىشىغا زور تۆھپە قوشتى .  لېكىن مۇسابىقە ئاياقلىشىپ ئىلى يۇلتۇز كوماندىسى مەردانە ھالدا  بىر بىرى بىلەن قول تۇتۇشۇپ ، ھۇررا غۇلجا دەپ توۋلاپ بارلىق مەستانلارغا ئېغىر سالام بەردى ، تۇرپان خەلقىمۇ ئۇلارغا قاتتىق چاۋاقلار بىلەن سالام قايتۇرۇپ ، ئىككىلا كوماندىغا قىقاس چۇقان ياڭرىتىپ ، ئۇلارغا رەھمىتىنى ئىيتىشتى ..... چىمىپىۇنلۇق  تالىشىش ئاياقلاشقاندىن كىيىن ھەرقايسى كوماندىلار مەيدانغا كىرىپ يىپىلىش مۇراسىمى ئېلىپ بېرىلدى ، باشلىقلار سۆز قىلىپ بولغاندىن كىيىن ، مۇكاپات تارقىتىلدى .  شىنجاڭ ئونۋىرسىتىتى كوماندىسى ، قارماي دولان كوماندىسى تەنتەربىيە ئەخلاقى مۇكاپاتىغا ، قۇمۇل جەسۇر ، قۇمۇل يىلتىز ، ئىلى يۇلتۇزى ۋە ھۇررا كۇلۇبى  تەشكىللەش مۇكاپاتىغا ئېرىشتى ، ئاخىرىدا قاراماي دولان كوماندىسىنىڭ بىرىنچى ، ئىلى يۇلتۇزى كوماندىسىنىڭ ئىككىنچى ، ھۇررا كۇلۇبىنىڭ ئۈچىنچى ۋە ئىزچىلار كوماندىىسنىڭ تۆتىنچى بولغانلىقىنى جاكارلاپ ، 7000، 5000، 3000، ۋە 1000 سوم مۇكاپاتلارنى بەردى .
ئەڭ مۇنەۋەر ھۇجۇمچى يەنى ئالتۇن كەش مۇكاپىتىنى بىزنىڭ ھۇررا كۇلۇبىنىڭ ھۇجۇمچىسى تاھىرغا بەردى ئۇمۇ 1000 سوم بىلەن مۇكاپاتلاندى .  ئۇ بۇ مۇسابىقىدە 5 توپ ئۇرغان ئىدى .
مىنىڭچە بۇ قىتىمىق مۇسابىقە تۇرپاندا ئۆتكۈزۈلگەن 12 نۆۋەت مۇسابىقە ئىچىدىكى سەۋىيىسى ئەڭ يۇقىرى بولغان ، ئەڭ قىززىغان ، ئەڭ  چىرايلىق بىر مۇسابىقە بولغان مەيلى مۇسابىقىنىڭ تەشكىللىنىشى ، مەيلى كوماندىلارنىڭ سەۋىيىسى ۋە تەييارلىق جەھەتتىن ئالدىدىكى 11 نۆۋەتتى خېلىلا يۇقىرى ئىدى.بۇنىڭدىن بۇ  مۇسابىقىنى تەشككىلىگەن رەھبەرلەر ، مۇناسىۋەتلىك ئادەملەر  بولۇپمۇ مەستانلار بەك خوشال بولدى بۇلۇپمۇ كىچىك بالىلارنىڭ روھى دۇنياسىغا چوڭ تەسىر قىلدى دەپ ئويلايمەن ، ئۇلارنىڭ توپچىلاردىن  كىيىملىرىنى سورىشى ، چولپانلارغا ئىمزا قۇيغۇزۇپ  خاتىرە رەسىملەرگە چۈشىشىدىن ، تۇرپان دىيارىدىكى مەستانلارنىڭ پۇتبۇل ئىڭىنى تېز يۇقىرى كۆتىرىلگەنلىكىنى ، خەلقئارادىكى مەستانلار قىلغان ئىشلارنى بۇلارنىڭ قىلالايدىغانلىقىنى ھىس قىلدۇردى .....
ئاخىرىدا مەن بۇ مۇسابىقىنى تەشكىللىگەن ۋە باشتىن ئاخىر نازارەت قىلىپ ئىشلارنىڭ يۈرىشىگە ئېلىپ بېرىلىشىغا كۆپ كۈچ چىقارغان مەخمۇت ئاكىمىز ، دىلشات ئاكىمىزغا ، مەرىپەتپەرۋەر ساخاۋەتچى ﺗﯘرﭘﺎن ﺷﻪﮬﻪرﻟﯩﻚ «ﭘﻪﺗﺘﺎر ﺋﻪﺳﻠﯩﮭﻪﻟﯩﻚ ﻳﯧﺰا ﺋﯩﮕﯩﻠﯩﻜﻰ ﻣﯘلازﯨﻤﻪت ﭼﻪﻛﻠﯩﻚ ﺷﯩﺮﻛﯩﺘﻰ» ﻧﯩﯔ ﺑﺎش دﯨﺮﯦﻜﺘﻮرى ﭘﻪﺗﺘﺎر ﻗﯧﻴﯘم ئاكىمىزغا بولۇپمۇ  بۇ قىتىملىق مۇسابىقىنى قوزغىغان ۋە باشتىن ئاخىر ئەڭ كۆپ كۈچ چىقارغان تۇرپان پۇتبۇل جەمئىيىتىنىڭ باش كاتىپى ھەم مۇئاۋىن رەئىسى ئىھساقجانغا چىن كۆڭلىمدىن رەھمەت ئىيتىمەن ، سىلەرنىڭ قوللىشىڭلار ، تىرىشچانلىقلار ۋە تۆھپەڭلەر بولمىسا ، بۇ قىتىملىق مۇسابىقە ئاللبۇرن بۆشىكىدە  ئۆلگەن بولاتتى ، رەھمەت سىلەرگە ھارمىغايسىلەر ، سىلەر تۇرپان پۇتبۇلچىلىقى ئەمەس بەلكى شىنجاڭ پۇتبۇلچىلىقىنىڭ يۈكسىلىشىگە  چوڭ ياردەم قىلدىڭلار ، بۇ قىىتملىق مۇسابىقىگە شىمالىي شىنجاڭ رايۇنىدىكى ئەڭ چوڭ بىر مۇسابىقە بولشىغا تۈرتكە بولدۇڭلار . سىلەرنىڭ تىنىڭلارنىڭ سالامەت ئىشلىرىڭلارنىڭ ئوڭۇشلۇق ئائىلەڭلارنىڭ بەخىتلىك بولۇشىنى تىلەيمەن ، ئەڭ ئاخىرىدا مۇشۇ قىتىملىق مۇسابىقىگە كوماندا يىتەكلەپ كەلگەن يىتەكچى ، تىرنىلارغا ھەم مۇسابىقىگە قاتناشقان بارلىق پۇتبۇلچىلارغا ئەڭ ئالىي ئىھتىرام بىلدۈرىمەن سىلەر  ھەقىقىي ئېسىل يىگىتلەر ، ئۇلۇغ ئاللا سىلەرگە ياخشى ئامەتنى يارقىلغۇسى ، ھۇررا تۇرپان پۇتبۇلچىلىقىن ، ھۇررا شىنجاڭ !
