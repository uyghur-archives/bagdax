بۈگۈن سىز نىمە ئىش قىلدىڭىز ؟




ئەسسالامۇ ئەلەيكۈم   ئەزىز قېرىنداشلىرىم !!!
 بۈگۈن 2011- يىلى 5- ئاينىڭ 5- كۈنى ئۇيغۇر دوپپا مەدەنىيەر بايرىمى، بايرمىڭىزغا موبارەك بولسۈن !
سىز بۈگۈنكى بايرامدا نىمە ئىش قىلدىڭىز ؟ نىمە ئىشلارنى قىلىشنى ئويلىشىۋاتىسىز ؟ سىز بۈگۈنكىدەك مىللىي مەدەنىيەت بايرام كۈنىدە سىز ئۆزىڭىزنىڭ ئۇيغۇر قەۋمىنىڭ بىر ئەزاسى ئىكەنلىگىڭىزنى ئەسكە ئالدىڭىزمۇ ؟
سىز بۇگۈن دوپپا كىيدىڭىزمۇ ؟
ئۇنتۇلۇپ قېلىۋاتقان دوپپىڭىزنى قولىڭىزغا ئېلىپ، توپا چاڭلىرىنى سۇرتۇپ،  ئويغا چومدىڭىزمۇ ؟
سىز بۈگۈنكى كۈندە، مىللىتىمىزنىڭ ئەنئەنىسى بويىچە ئالدى بىلەن مىھرىبان ئاتا - ئانىڭىزنى ئەسكە ئالدىڭىزمۇ ؟ ئۇلارنىڭ قېشىغا بېرىپ ، ياكى تېلفوندا بولسىمۇ ئەسسسالامۇ ئەلەيكۈم ئەزىزانە ئاتا - ئانىلىرىم دىدىڭىزمۇ؟
ئۇلارنىڭ قولىدىكى سۈپۇرگۈنى ئېلىپ ئۇلارنىڭ ھويلا ئارمىنى بىر كۈن بولسىمۇ سۇپۈرۈپ بەردىڭىزمۇ ؟
 ئاتىڭىزنىڭ قولىدىكى كەتمەننى ئېلىپ، باشتېن ئېگىز كۆتۈرۈپ ! يا ئاللاھ ! دەپ يەرگە چوڭقۇر بىر قېتىم پاتۇرۇپ باقتىڭىزمۇ ؟ .......
ئەي ئانا !سىز بۇگۈن قىزىڭىز بىلەن مۇڭداشتىڭىزمۇ؟ ئۇنىڭغا چىمەن دوپپىنىڭ ، ئەتلەسنىڭ ، ئۇيغۇر قىزلىرىنڭ 40 تال ئۆرۈم چېچىنىڭ ھىكايىلىرىنى ئېيتىپ بەردىڭىزمۇ ؟ ئۇنىڭ يۇمران قاشلىرىغا ئۆسما قويۇپ قويدىڭىزمۇ ؟ ئۈنىڭغا ئىپپەت قۇربانلىرى نۇزۇگۈم ، ئىپارخان  ... لارنىڭ ھىكايىسىنى سۆزلەپ بەردىڭىزمۇ ...
ئەي ئاتا ، سىز بۈگۈن ئوغلىڭىز بىلەن مۇڭداشتىڭىزمۇ ؟ ئۇنىڭ بېشىدىكى دوپپىنى ئۆز قولىڭىز بىلەن تۈزەشتۈرۈپ قويدىڭىزمۇ ؟ ئۇنى قوتانغا ئەكىرىپ، كىشنەپ تۇرغان ئارغىماق ئېتىڭىزنى كۆرسىتىپ، ئۇنىڭ ئۇستىگە بىر مىنگەشتۇرۇپ باقتىڭىزمۇ ؟
بىر چاغلاردا بوۋاڭ مۇشۇنداق ئارغىماقلارنى مىنەتتى دەپ بوۋىلىرىنىڭ ھىكايىسىنى ئېتىپ بەردىڭىزمۇ ؟
  ئوغلىڭىزنى ئېلىپ چىقىپ يۇرتىڭىزنىڭ كەڭ ، گۈزەل ، پۈتمەس تۈگىمەس بايلىققا تولغان زىمىنى كۆرسىتىپ مانا بۇ سېنىڭ يۈرتۈڭ ، بۇ سىنىڭ ۋەتىنىڭ ، بۇ يەرلەر سېنىڭ بارلىغىڭ ، بۇ يەرلەر سېنىڭ جېنىڭدىن ئەزىز، ۋەتەننى قوغداش ئەرنىڭ بۇرچى! ۋەتەن ئۈچۈن جاد تەسەددۇق قىلىش ئەرنىڭ مەجبۇرىيىتى! دىدىڭىزمۇ ؟
 باتۇر سادىر بوۋىسىنىڭ ئات ئۈستىدىكى كارامىتىنى ، سېپىل پارتىلىتىشتىكى ئەقىل پاراسىتىنى، سۆزلەپ بەردىڭىزمۇ ؟
 ئەلشىر ناۋائى، يۈسۈپ خاس خاجىف ، ماخمۇت قەشقىرى قاتارلىق بىباھا گۆھەرلەرنى قالدۇرغان بۆۋىلىرىنىڭ كىتاپلىرىنى كۆرسەتتىڭىزمۇ ؟
 لوتپوللا ، ئابدۇخالىق ئۇيغۇر ، تەۋپىق، ئەخمەتجان قاسىمى .... قاتارلىق ئۆت يۈرەكلەرنىڭ ھىكايىسىنى سۆزلەپ بەردىڭىزمۇ ؟
 ئابدۇرەھىم ئۆتكۈر ، زوردون سابىر، ئابۇشۇكۈر مۇھەمەت ئىمىن .... قاتارلىق ئەدىپلىرىمىزنى تونۇشتۈردىڭىزمۇ ؟
 نەبىجان تۈرسۈن ، ئەركىن سىدىق ... قاتارلىق ھازىرقى زاماندىكى بىلىم ساھەسىدىكى بۈركۈتلەرنىڭ ھىكايىسىنى ئېيتىپ بەردىڭىزمۇ ؟ ....
بۈگۈن سىز، خوشنا خولۇملىرىڭىزغا قاردىڭىزمۇ ؟ ئۇنىڭ ئوغۈل قىزلىرى بېشىغا دوپپا كىيىپتىمۇ ؟ ياكى بېشىدا دوپپا يوق، پۈتلىرى يالاڭ ئاياق، ئوغلىڭىزنىڭ يىپيىڭى كىيىملىرىگە ، بېشىدىكى ئوتقاشتەك دوپپىلىرىغا مەيۈسلۈك بىلەن تەلمۇرۇپ قاراۋاتامدۇ ؟
ئاۋۈ يالغۈز ئايىمخان موماينى يوقلىدىڭىزمۇ ؟ ئۇنىڭغا  بۈگۈن بىر چىلەك سۈ ئەكىرىپ تۇڭىغا قۇيۇپ بەردىڭىزمۇ ؟
ئاۋۈ ئاۋۈتاخۇن بوۋاي بۈگۈن مەسچىتكە چىقالىدىمۇ؟
ئاپلا ماۋۇ خوشنامنىڭ بالىسى يىقىلىپ چۈشتى ....   ئۈنى دادىل بېرىپ بىر قېتىم يۆلەپ قويدىڭىزمۇ .........؟

بۇگۈن سەھەر ئايالىم بىلەن ئىككىمىز بىر قانچە كۈندىن بۇيان دوپپا بايرىمى ئۈچۈن تەييارلىغان چىرايلىق دوپپىلىرىمىزنى كىيىپ ئىككى چوڭ ئۆيدىكى ياشانغان ئاتا - ئانىلىرىمىزنى ۋە بىر قانچە كېسەل بىلەن ئائىلىدە كۈتۈنىۋاتقان دوستلىرىمىزنى يوقلاش ئۈچۈن سىرىتقا چىقتۇق. مەھەللىمىزدىكى كىشىلەرنى بىزنى كۆرۈپ خۇددى بۇرۇن كۆرۈپ باقمىغاندەك ئىككىمىزگە باشتىن ئاياغ قارىشىپ كەتتى .... توۋا...
بىر قانچە خۇشنىلىرىمىز بىنا ئالدىدا پاراڭلىشىپ ئولتۇرۇپتىكەن ئۇلار بىزگە:
_ ئوھۇي، نېمە ئىش بۇ ؟ بىر يەرگە ئۆلۈپ پەتىسىگە ماڭدىڭلارمۇ دېگەن ئىدى. مەن جاۋاب بەرگىچە ئايالىم:
_ شۇنداق، ئۇزۇندىن بۇيان كىيىلمەي ئېغىر كېسەل بولۇپ قالغان ئۇيغۇر دوپپىلىرىدىن بىر قانچىسىنىڭ ئۆيىگە پەتىسىگە بېرىپ، ئۇلارنى قولۇمىزدىن كېلىشىچە داۋالاپ ساقايتىۋالايلى دېگەن دەپ جاۋاب بەردى. ئۇلار ئايالىمنىڭ نېمە دەۋاتقانلىقىنى بىلمەي بىر - بىرىگە پىچىرلىشىپ كەتتى. پەرىزىمچە ئايالىمنى ساراڭ بولۇپ قاپتۇ دىدى ھەقىچان ....
