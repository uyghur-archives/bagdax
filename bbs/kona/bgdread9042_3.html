    بەلكىم قەشقەردە ئۇزۇن يىل خىزمەت قىلغان جۇتاۋ ئەپەندى قەشقەر خەلقىنىڭ يېرىم ئەسىردىن ئارتۇق ۋاقىتتىن بۇيان قەشقەر خەلق مەيدانىنىڭ ئالدىغا تىكلەنگەن ماۋزېدوڭنىڭ پۈتۈنسۈرىك ھەيكىلىنى ھازىرغىچە ياخشى ئاسراپ ساقلاپ كەلگەنلىكىنى كۆرۈپ  ماۋزېدوڭ خاتىرە سارىيىنى شىنجاڭغا يۆتكىۋەتسە تېخىمۇ بىخەتەر بولىدۇ، دېگەن تونۇشقا كەپ قالغان ھەم بۇ ئويىنى قۇملۇق ئىگىلىكى دېگەندەك باشقىچە ئۇسۇل بېلەن نىقاپلاپ ئىپادىلىگەن بولۇشىمۇ مۇمكىن.
   دېمىسىمۇ خۇددى ۋۇجىياخۇ ئەپەندى «شىنجاڭلىقلار ھەققىدە قايتا ئويلۇنۇش»ناملىق ماقالىسىدا ئېيىتقاندەك «شىنجاڭلىقلارنىڭ قۇلىقى بەك يۇمشاق، ئۇلار پارتىيىنىڭ گېپىنى ھەممىدىن بەك ئاڭلايدۇ، ئۇلار پارتىيە سېياسىتىنى ئىجرا قىلىشتا ئەزەلدىن بويۇنتاۋلىق قىلمايدۇ»ئەمەسمۇ؟
    شۇڭا جەمئىيەتتىكى «سوتسىيالىزم شىنجاڭدىلا قالدى»دېگەن سۆزى بىكاردىنلا تارقالمىغان بولسا كېرەك ئەتىمالىم.

   مۇبادا يۇقۇرقى سۆرەتتىكى ئەھۋاللار قەشقەررايونىدا يۈز بەرسە شۇ ھامان «مىللىي بۆلگۈنچىلىك»،«تېرورلۇق»تۈسىنى ئېلىپ ، قەشقەردىكى ھەربىي -ساقچى گازارمىلار شۇ ھامان تەڭلا ھەركەتكە كېلىشى مۇمكىن ئىدى!
