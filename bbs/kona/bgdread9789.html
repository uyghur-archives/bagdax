 تاڭ سۇلالىسى نېمەئۈچۈن تۈركلەرگە باش ئەگكەن؟

 تاڭ سۇلالىسى نېمەئۈچۈن تۈركلەرگە
باش ئەگكەن؟

      تۈركلەرئىلىمىزنىڭ شىمالىدىكى قەدىمى مىللەت بۇلۇپ، جەنۇبى-شىمالى سۇلىلار دەۋرىدە تەدرىجى زورۇيۇپ، نەچچەيۈزمىڭ كىشلىك قۇشۇنغا ئىگە بولغان ئىدى. سۈي سۇلالىسىگە كەلگەندە،تۈركلەر ئىچكى نىزار ۋە ئۇرۇش تۈپەيلى ھۈكۈمران گۇروھلار ئىچىدە  بۆلۈنۈش پەيدا بۇلۇپ، غەربى تۈرك ۋە شەرقى تۈرك دەپ ئىككىگە بۈلۈنۈپ كېتىدۇ،غەربى تۈركلەر ئالتاي تېغىنىڭ غەربىنى ماكان تۇتتى، شەرق تۈركلەر بولسا شەرىقتىن ھىنگان تاغلىرى ( ھازىرقى خىيلۇڭجاڭ تەۋەسىدە)دىن باشلاپ غەرىپتەئالتاي تاغلىرىغىچە بولغان زىمىنغا ھۈكمىران بولدى. ئۇلار بۆلۈنۈپ ئۇزۇن  بولماي، شەرىق تۈركلەرنى سۈي سۇلالىسى قۇشۇنلىرى مەغلۇپ قىلدى، غەربى تۈركلىرى ئىچىدىمۇ يەنە ئىچكى نىزا يۈز بېرىشكە باشلىدى، سۈي سۇلالىسىنىڭ ئاخىرقى يىللىرىغا كەلگەندە، سۈي سۇلالىسى زاۋاللىققا يۈز تۇتتى، بۇ چاغدا، شەرقى- غەربى تۈركلەر يەنە قايتىدىن بىرلىشىپ، جانلىنىشقا باشلىدى، ھەمدە موبىي( ھازىرقى مۇڭغۇل ئىگىزلىكى) دىن تارتىپ، غەربى دىيارغىچە بولغان زىمىننى كونتۇرۇل قىلىپ، ئوتتۇرا تۈزلەڭلىككىچە تەھدىت سېلىشقا باشلىدى.
سۈي سۇلالىسىنىڭ ئاخىرىغا كەلگەندە، دېھقانلار قوزغۇلاڭلىرى يۈز بىرىپ، 611-يىلىدىن باشلاپ، سۈي سۇلالىسىدا تەخمىنەن 100دىن كۆپرەك دېھقانلار قوزغىلاڭ قۇشۇنى بارىلىققا كەلدى. 617-يىلى 7-ئايدا سۈي سۇلالىسىنىڭ يىمىرىلىشى كۈرۈنۈپ قالدى، بۇچاغدا، ۋەزىيەت كۈتۈپ تۇرغان لىيۈەن ۋەئۇنىڭ ئوغلى لى شىمىن ۋەزىيەتنىڭ پىشىپ يىتىلگەنلىكىنى پەمىلەپ، جىن ياڭ ( ھازىرقى شەنشى تەييۈەن) دىن30مىڭ كىشىلىك قۇشۇن تارتىپ، دېھقانلار قوزغۇلىڭىنىڭ جەڭ مېۋىسىنى تارتىۋېلىشقا ئۇرۇندى.
«كونا تاڭ نامە» دە خاتىرلىنىشچە، لىيۈەن دادا- بالا ئىككىسى قۇشۇن تارتقاندا، تۈركلەر پۇرسەتتىن پايدىلىنىپ جىن ياڭغا ھۇجۇم قىلغان، بۇنىڭ بىلەن لىيۈەن دادا-بالا ئىككىسى تۈركلەرنىڭ قورقۇنۇشلۇق بىر كۈچ ئىكەنلىكىنى ھېس قىلىپ، تۈركلەر بىلەن بولغان مۇناسىۋەتنى ياخشىلىمىغاندا، ئۈزىنىڭ ئەمىدىلا شەكىللەنگەن كۈچىنىڭ يىمىرىلىشىنى تۇنۇپ يېتىدۇ، تۈركلەرنىڭ تەھدىتىنى يۇقۇتۇش ئۈچۈن،لىيۈەن ئۈزى تۈرك خاقانىغا مەكتۇپ يېزىپ، سوۋغات تەقدىم قىلىش قا ۋەدە بىرىپ، ھەمدە نۇرغۇن سوۋغا-سالاملار بىلەن لىيۇۋىنجىن ئىسىملىك يېقىنىنى تۈركلەربىلەن سۆھبەتكە يولغا سالىدۇ.
لىيۇۋىنجىن تۈركلەرگە لىيۈەننىڭ مەقسىدىنى يەتكۈزگەندىن كىيىن، تۈرك خاقانى خۇشال بۇلۇپ،2000كىشىلىك ئاتلىقنى قۇشۇپ، لى يۈەنگە دوستلىقىنى ئىزھارقىلىدۇ. لىيۈەن تۈرك خاقانىنىڭ مال-دۇنياغا ھېرىسمەن بۇلۇشتەك ئاجىزلىقىدىن تۇلۇق پايدلىنىش مەقسىدىگە يىتىدۇ، شۇنىڭ بىلەن ئۈزىنىڭ ئوتتۇرا تۈزلەڭلىكتىكى ئورنىنى تىكلىشىگە پۇرسەت يارتىلىدۇ.
ئەمما تۈركلەرنىڭ ئىشتاھاسى بارغانسىرى يۇغۇنايدۇ، ھەرخىل باھانىلەربىلەن لى يۈەندىن مال-مۈلۈك تەلەپ قىلىدۇ، ھاكاۋۇرلىق بىلەن يۇغانچلىق قىلىدۇ، ئەمما ئوتتۇرا تۈزلەڭلىك ۋەزىيىتى تېخى مۇقۇم بولمىغاچقا، لى يۈەن تۈركلەرنىڭ يۇغانچلىقىغا چىداپ كېلىدۇ، گەرچە سوۋغات تەقدىم قىلىش تۈۋەننىڭ يۇقىرىغا قىلىدىغان ئىشى بولسىمۇ، ئەمما لى يۈەن چوڭ ئۇمۇمىلىقنى كۆزدە تۇتۇپ، تۈركلەرنىڭ ھەرخىل يولسىز تەلەپلىرىگە ماقۇل بۇلىدۇ، تۈرك خاقانى ئالەمدىن ئۆتكەندە، لى يۈەن تارىختىكى پادىشاھلار ئالەمدىن ئۆكەندىلا بۇلىدىغان ئەڭ زور ماتەم مۇراسىمى بىلەن ئىتائىتىنى بىلدۈرىدۇ. لى يۈەن « بىزنىڭ كۈچىمىز تۈركلەرنىڭكىگە يەتمەيدۇ، ئۇنىڭ ئۈستىگە تېخى زىمىنىمىزنى مۇقۇم قىلىپ بۇلالمىدۇق شۇڭا تۈركلەرنىڭ قىلمىشىغا چىداپ تۇرىشىمىزكېرەك» دەيدۇ.
لى يۈەن ئوتتۇرا تۈزلەڭلىكنى بىرلىككە كەلتۈرۈپ، تاڭ سۇلالىسىنى قۇردى،تۈركلەر بۇنى كۈرۈپ، ئۆزلىرىنىڭ ئەمدى ئىلگىرىكىگە ئوخشاش خالىغانچە سوۋغات تەلەپ قىلالمايدىغانلىقىنى ھېس قىلىپ،تاڭ سۇلالىسىنىڭ يېڭىدىن قۇرۇلغان، كۈچىنىڭ تېخى ئانچە چوڭ ئەمەسلىكىدىن پايدىلىنىپ، يىل بۇيى ھۇجۇم قىلىپ، مال- مۈلۈك بۇلاڭ- تالاڭ قىلىدۇ. ئاخىرى 150مىڭ كىشلىك زور قۇشۇن بىلەن جىن جۇ ( ھازىرقى شەنشى تەۋەسى ۋە خېبىي، ئىچكى مۇڭغۇل رايۇنلىرى) غا ھۇجۇم قىلىپ كىرىپ،5000 نۇپۇسنى بۇلاپ كېتىدۇ. ئارقىدىنلا يەنە100مىڭدىن ئارتۇق ئەسكەر بىلەن تەييۈەنگە ھۇجۇم قىلىدۇ، 626-يىلى لى شىمىن ئەمدىلا پادىشاھلىققا ئولتۇرغاندا، تۈركلەر 200مىڭ كىشىلىك قۇشۇن بىلەن تاڭ سۇلالىسى پايتەختى چاڭ ئەنگە40 كىلومىتىركەلگۈدەكلا جايغىچە باستۇرۇپ كىرىدۇ. لى شىمىن نۇرغۇن سوۋغا سالام يوللاپ، تۈركلەر بىلەن ئىتىپاق تۈزۈپ، ئىتائەت قىلىدىغانلىقىنى بىلدۈرىدۇ، شۇندىلا تۈركلەر ئەسكەرلىرىنى چىكىندۈرۈپ قايتىدۇ، بۇ ۋەقە تارىختا « ۋىي شۈيدىكى ئىتىپاقداشلىق» دەپ ئاتالدى.
لى شىمىن تەختكە ئولتۇرغاندىن كېيىن ئىقتىساسلىقلارنى ئىشلىتىپ، ئىقتىساتنى تەرەققى قىلدۇرۇپ،ھەربى كۈچنى كۈچەيتتى، 629-يىلىدىن كىيىن، تاڭ سۇلالىسىنىڭ تۈركلەرگە بولغان باش ئىگىش ۋەزىيىتىدە ئۆزگۈرۈش بۇلۇپ، تاڭ سۇلالىسىنىڭ ھەربى كۈچى زور دەرىجىدە كۈچەيدى. ئەكىسچە، تۈركلەر ئاجىزلىشىپ، ئىچكى نىزار ۋە قوزغۇلاڭ ئىچىدە قالدى، بۇچاغدا ۋەزىيەت تاڭ سۇلالىسىغا پايدىلىق بولغان ئىدى، 617-يىلىدىن 629-يىلىغىچە بولغان 20يىل ئىچىدە لىشىمىن تۈركلەرنى قانداق ھەل قىلىش ھەققىدە ئويلاندى، تۈركلەرنىڭ قىلمىشىغا چىشىنى چىشلەپ چىداپ كەلدى،629-يىلى لىشىمىن ۋەزىيەتنى تەھلىل قىلىش ئارقىلىق،تۇلۇق ھەربى ھازىرلىق كۈرۈش نەتىجىسىدە،تۈركلەرنى تەل-تۈكۈس ھەل قىلىش پەيتىنىڭ يېتىپ كەلگەنلىكىنى پەرەز قىلىپ،تۈركلەرگە ھۇجۇم قوزغاپ، ئۈزىنىڭ يىگىرمە يىللىق ئار-نۇمۇسىنى يۇيماقچى بولدى. لى جىن ئىسىملىك سانغۇنغا بۇيرۇق بىرىپ، تۈركلەرگە ھۇجۇمغاماڭدۇردى، تاڭ قۇشۇنى تۈركلەرنى مەغلۇپ قىلىپ، خاقانىنى تىرىك تۇتتى. شۇنىڭدىن تارتىپ، تاڭ سۇلالىسىنىڭ تۈركلەرگە باش ئىگىپ، سوۋغا- سالام ماڭدۇرۇشى ئاخىرلىشىپ، تارىخ باشقىدىن يېزىلدى.

مەنبە: «تارىختىكى پادىشاھلار»نامىلىق ئەسەر، ئەسەرنىڭ ئەسلى ئاپتۇرى لىيۇجىشىن، بىيجىن ئاۋىئاتسىيەنەشرىياتى.

 تاڭ سۇلالىسى نېمەئۈچۈن تۈركلەرگە
باش ئەگكەن؟

      تۈركلەرئىلىمىزنىڭ شىمالىدىكى قەدىمى مىللەت بۇلۇپ، جەنۇبى-شىمالى سۇلىلار دەۋرىدە تەدرىجى زۇرۇيۇپ،نەچچەيۈزمىڭ كىشلىك قۇشۇنغا ئىگە بولغان ئىدى سۈي سۇلالىسىگە كەلگەندە،تۈركلەر ئىچكى نىزار ۋە ئۇرۇش تۈپەيلى ھۈكمىران گوروھى ئىچىدە  بۈلۈنۈش پەيدا بۇلۇپ، غەربى تۈرك ۋە شەرقى تۈرك دەپ ئىككىگە بۈلۈنۈپ كىتىدۇ،غەربى تۈركلەر ئالتاي تېغىنىڭ غەربىنى ماكان تۇتتى، شەرق تۈركلەر بولسا شەرىقتىن ھىنگانتاغلىرى ( ھازىرقى خىيلۇڭجاڭ تەۋەسىدە)دىن باشلاپ غەرىپتەئالتاي تاغلىرىغىچە بولغان زىمىنغا ھۈكمىران بولدى. ئۇلار بۈلۈنۈپ ئۇزۇن  بولماي، شەرىق تۈركلەرنى سۈي سۇلالىسى قۇشۇنلىرى مەغلۇپ قىلدى، غەربى تۈركلىرى ئىچىدىمۇ يەنە ئىچكى نىزار يۈزبىرىشكە باشلىدى، سۈي سۇلالىسىنىڭ ئاخىرقى يىللىرىغا كەلگەندە، سۈي سۇلالىسى زاۋاللىققا يۈزتۇتتى، بۇچاغدا، شەرقى- غەربى تۈركلەر يەنە قايتىدىن بىرلىىشىپ، جانلىنىشقا باشلىدى، ھەمدە موبىي( ھازىرقى مۇڭغۇل ئىگىزلىكى) دىن تارتىپ، غەربى دىيارغىچە بولغان زىمىننى كونتۇرۇل قىلىپ، ئوتتۇرا تۈزلەڭلىككىچە تەھدىت سېلىشقا باشلىدى.
سۈي سۇلالىسىنىڭ ئاخىرىغا كەلگەندە، دېھقانلار قوزغۇلاڭلىرى يۈزبىرىپ، 611-يىلىدىن باشلاپ، سۈي سۇلالىسىدا تەخمىنەن 100دىن كۆپرەك دېھقانلار قوزغىلاڭ قۇشۇنى بارىلىققا كەلدى،617-يىلى 7-ئايدا سۈي سۇلالىسىنىڭ يىمىرىلىشى كۈرۈنۈپ قالدى، بۇچاغدا، ۋەزىيەت كۈتۈپ تۇرغان لىيۈەن ۋەئۇنىڭ ئوغلى لى شىمىن ۋەزىيەتنىڭ پىشىپ يىتىلگەنلىكىنى پەمىلەپ، جىن ياڭ ( ھازىرقى شەنشى تەييۈەن) دىن30مىڭ كىشىلىك قۇشۇن تارتىپ، دېھقانلار قوزغۇلىڭىنىڭ جەڭ مىۋىسىنى تارتىۋېلىشقا ئۇرۇندى.
«كونا تاڭ نامە» دە خاتىرلىنىشچە، لىيۈەن دادا- بالا ئىككىسى قۇشۇن تارتقاندا، تۈركلەر پۇرسەتتىن پايدىلىنىپ جىن ياڭغا ھۇجۇم قىلغان، بۇنىڭ بىلەن لىيۈەن دادا-بالا ئىككىسى تۈركلەرنىڭ قورقۇنۇشلۇق بىركۈچ ئىكەنلىكىنى ھېسقىلىپ،تۈركلەربىلەن بولغان مۇناسىۋەتنى ياخشىلىمىغاندا، ئۈزىنىڭ ئەمىدىلا شەكىللەنگەن كۈچىنىڭ يىمىرىلىشىنى تۇنۇپ يىتىدۇ، تۈركلەرنىڭ تەھدىتىنى يۇقۇتۇش ئۈچۈن،لىيۈەن ئۈزى تۈرك خاقانىغا مەكتۇپ يېزىپ، سوۋغات تەقدىم قىلىش قا ۋەدە بىرىپ، ھەمدە نۇرغۇن سوۋغا-سالاملاربىلەن لىيۇۋىنجىن ئىسىملىك يېقىنىنى تۈركلەربىلەن سۆھبەتكە يولغا سالىدۇ.
لىيۇۋىنجىن تۈركلەرگە لىيۈەننىڭ مەقسىدىنى يەتكۈزگەندىن كىيىن، تۈرك خاقانى خۇشال بۇلۇپ،2000كىشىلىك ئاتلىقنى قۇشۇپ، لى يۈەنگە دوستلىقىنى ئىزھارقىلىدۇ. لىيۈەن تۈرك خاقانىنىڭ مال-دۇنياغا ھېرىسمەن بۇلۇشتەك ئاجىزلىقىدىن تۇلۇق پايدلىنىش مەقسىدىگە يىتىدۇ، شۇنىڭ بىلەن ئۈزىنىڭ ئوتتۇرا تۈزلەڭلىكتىكى ئورنىنى تىكلىشىگە پۇرسەت يارتىلىدۇ.

باش ئەگكەن؟
