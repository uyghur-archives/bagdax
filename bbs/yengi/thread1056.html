         باغداش مۇنبىرى خىزمەتچى قوبۇل قىلىش توغرىسىدىكى ئۇقتۇرۇشىنى چىقارغاندىن كېيىن نۇرغۇن كونا يېڭى تورداشلار ئاكتىپ ئاۋاز قوشۇپ باغداش مۇنبىرىنىڭ تەرەققىياتى ئۈچۈن بىر كىشلىك كۈچ چىقىرىشنى خالايدىغانلىقىنى بىلدۈرۈشتى. ھەم باشقۇرغۇچى ، مۇخبىر ، تەرجىمان قاتارلىقلارغا ئۆزىنى ۋەياكى باشقا مۇنبەرداشلارنى نامزات كۆرسەتكەنمۇ بولدى . بۇلارنىڭ ئىچىدە ئەڭ كۆپ مۇلاھىزە قىلىنغىنى باشقۇرغۇچىلىق نامزاتى بولدى ، باشقۇرغۇچىلىق ۋەزىپىسىنى ئۆز زىممىسىگە ئېلىشنى خالايدىغان يېڭى نامزاتلارمۇ خېلى كۆپ بارلىققا كەلدى ، ھەم شۇنداقلا مۇنبىرىمىزنىڭ پېشقەدەم باشقۇرغۇچىلىرىنى داۋاملىق ئۆز خىزمىتىنى قىلسا دېگەن ئۈمىدتە ئۇلارمۇ قايتا نامزاتلىققا كۆرسىتىلدى ، ئەلۋەتتە بۇ ئەھۋال مۇنبەرداشلىرىمىزنىڭ، پېشقەدەم باشقۇرغۇچىلارنىڭ  ئىلگىركى خىزمەتلىرىنى مۇئەييەنلەشتۈرگەنلىكنى ھەم ئۇلارنىڭ نەتىجىلىرىدىن مەمنۇن بولغانلىقىنى چۈشەندۈرۈپ بېرىدۇ.
         مۇنبەر باشقۇرغۇچىلىقى بولسا پۈتۈنلەي خالىسانە ، ئىشتىن سىرت ئېلىپ بېرىلىدىغان خىزمەت بولغاچقا باشقۇرغۇچى بولغان ئادەم چوقۇم ئۆزىنىڭ بىرىنچى كەسپىدىن باشقا ئايرىم ۋاقىت چىقىرىپ تېما تەستىقلاش ، تەھرىرلەش قاتارلىق خىزمەتلەرنى ئېلىپ بېرىشقا توغرا كېلىدۇ. شۇڭلاشقا پېشقەدەم باشقۇرغۇچىلىرىمىزدىن كۈندىلىك خىزمىتى بىرقەدەر ئالدىراشلار ، ئۆزىنىڭ مۇنبەرنىڭ ئىشلىرى ئۈچۈن يېتەرلىك ۋاقىت چىقىرالماي باشقۇرغۇچىلىق ۋەزىپىسىنى ياخشى ئورۇندىيالماسلىقىدىن ئەنسىرەپ داۋاملىق باشقۇرغۇچى بولۇشنى چىرايلىق رەت قىلىدىغانلارمۇ بولدى . ئەلۋەتتە مۇنبەرداشلار بۇتورداشلارنىڭ قىيىنچىلىقىنى چۈشىنىمىز .
       مەن يوقاردىكى ئەھۋاللانى ئويلىنىپ كۆرۈپ مۇنبەر باشقۇرغۇچىلا گۇرۇپپىسىنى تۆۋەندىكىدەك تەشكىللەشنى توغرا تاپتىم .
باشقۇرغۇچىلارنى قاتلام بويىچە ئورۇنلاشتۇرۇش :
      بېكەت قۇرغۇچىلار ئەلۋەتتە ئەڭ ئالىي تولۇق ھوقۇقلۇق باشقۇرغۇچى بولىدۇ ، مۇنبەرنىڭ ئاساسلىق تەرەققىيات يۆلىنىشىنى بەلگىلەيدۇ.
      تۈر باشقۇرغۇچىلىرى:
ساناپ باقسام مۇنبىرىمىزدە جەمئىي ئالتە تۈر باركەن(مۇنبەر ئىشخانىسىدىن باشقا) ، بىز ھەربىر تۈرگە بىردىن سەھىپە باشقۇرغۇچى قويساق بۇۋەزىپىنى مېنىڭچە يەنە شۇ بىزنىڭ پېشقەدەم مۇنەۋۋەر باشقۇرغۇچىلىرىمىز ئۆز ئىختىيارلىقى بىلەن ئۆززىممىسىگە ئالسا ياخشى بولاتتى.چۈنكى  تۇرمۇش ھەم كەسىپكە بىرقەدەر پىشقان ئەزالار بۇ ئورۇنغا قويۇلسا مۇنبەر قۇرغۇچىلىرى بىلەن ئاكتىپ ماسلىشىپ سەھىپە باشلىقلىرىنىڭ تېما تەستىقلىشىغا توغرا يېتەكچىلىك قىلىپ مۇنبەرنىڭ ساغلام تەرەققىي قىلىشغا تېخىمۇ ياخشى شارائىت ھازىرلانغان بولاتتى.
     سەھىپە باشقۇرغۇچىلىرى :
ھەربىر تۈر ئاستىدا بىرقانچە سەھىپە بار بولۇپ ، بۇسەھىپىلەرگە يەنە ئايرىم باشقۇرغۇچى بەلگىلەنسە بولىدۇ.  ئادەتتە تېما تەستىقلاش ، تەھرىرلەش ئىشلىرىنى سەھىپە باشلىقلىرى ئۆز ئۈستىگە ئالسا بولىدۇ. ئەگەردە سەھىپە باشلىقلىرى تەستىقلاشتا ئىككىلىنىپ قالغان تېمىلار بولسا تۈر باشلىقىغا خەۋەرقىلىپ مۇزاكىرە قىلىنغاندىن كېيىن قارار چىقىرىلسا ، بۇنداق بولغاندا مۇنبىرىمىزنىڭ تېما سانى ھەم سۈپىتىنىڭ يوقىرى كۆتۈرۈلىشىگە بەلگىلىك ئاكتىپ تەسىرى بولاتتى دەپ قارايمەن .
    يوقاردا ئۆزەمنىڭ مۇنبەر باشقۇرغۇچىلىقىغا بولغان ئازتولا ئويلىغانلىرىمنى تاقا-تۇقا قىلىپلا سۆزلەپ قويدۇم . ئەمدى تۆۋەندە ئۆزەمنى باشقۇرغۇچىلىق نامزاتىغا كۆرسىتىپ ئازتولا ئارقا ئىشىك قىلىۋېلىشىمغا يول قويۇڭلار ھە !
          مەن ئەلقوندى يوقاردا دىگەنلىلىرىم بويىچە تۈر « بىلم غەزىنىسى » ئاستىدىكى سەھىپە«كومپيوتېر ئۇچۇرلىرى»،«كومپىيوتىردىن سوئال-جاۋاب»،«يۇمشاق دېتال باغچىسى » قاتارلىقلارنىڭ خالىغان بىرىگە سەھىپە  باشلىقى بولۇشنى خالايمەن . ئۆزەم باشقىلارغا ياردەم بېرىشنى خۇشاللىق دەپ بىلىدىغان ، مېنىڭ گەپ-سۆز ، ئىش - ھەرىكەتلىرىمدىن باشقىلار زەررىچە نەپكە ئېرىشسە ئالەمچە خۇشال بولىدىغان(نۇرغۇن دوسلۇرۇم ھايات لېيفىڭ دەپ ئاتمۇ قويۇۋالغان تېخى) كىشى بولغاچقا ، تورداشلارغا كومپىيوتىربىلىملىرىنى كېڭەيتىشنى ، كومپيوتېر توغرىسىدا سورىغان سوئاللىرىغا ، خەنزۇچە بېكەتلەردىن چۈشىنەلمىگەن مەسىلىلەرگە كۈچۈمنىڭ يېتىشىچە ئاددى تىللار بىلەن جاۋاب بېرىپ كوپىيوتىردىن ئىبارەت بۇ ھازىرقى زاماندىكى ئەڭ ئىلغار ئەڭ مودا پەن تېخنىكا مەھسۇلاتىنىڭ تېخىمۇ كۆپ ئۇيغۇر ئائىلىلىرىگە سىڭىپ كىرىپ تېخىمۇ كۆپ ئۇيغۇر خەلقىنىڭ كومپىيوتىردىن پايدىلىنىپ يىراقنى يېقىن ، قىيىن ئىشنى ئاسان قىلىشى ئۈچۈن كۈچ چىقىرىش مەقسىتىدە مۇشۇ ۋەزىپىنى ئۆز ئۈستۈمگە ئېلىشنى قارار تاپتىم . ئەلۋەتتە باغداش مۇنبىرىمىزدە كومپيوتېر بىلىملىرى بىلەن خەلقىمىزنىڭ ئېھتىياجىنى قامداۋاتقان باشقا ئەزالىرىمىزمۇ بار ئۇلارنىڭمۇ بۇسەپكە قوتۇلىشىنى ئۈمىد قىلىمەن.

Re:مۇنبەر  باشقۇرغۇچىلار كوللىكتىپىنى قۇرۇشقا بولغان تەكلىپ

 مەن ئەلقوندى يوقاردا دىگەنلىلىرىم بويىچە تۈر « بىلم غەزىنىسى » ئاستىدىكى سەھىپە«كومپيوتېر ئۇچۇرلىرى»،«كومپىيوتىردىن سوئال-جاۋاب»،«يۇمشاق دېتال باغچىسى » قاتارلىقلارنىڭ خالىغان بىرىگە سەھىپە  باشلىقى بولۇشنى خالايمەن . ئۆزەم باشقىلارغا ياردەم بېرىشنى خۇشاللىق دەپ بىلىدىغان ، مېنىڭ گەپ-سۆز ، ئىش - ھەرىكەتلىرىمدىن باشقىلار زەررىچە نەپكە ئېرىشسە ئالەمچە خۇشال بولىدىغان(نۇرغۇن دوسلۇرۇم ھايات لېيفىڭ دەپ ئاتمۇ قويۇۋالغان تېخى) كىشى بولغاچقا ، تورداشلارغا كومپىيوتىربىلىملىرىنى كېڭەيتىشنى ، كومپيوتېر توغرىسىدا سورىغان سوئاللىرىغا ، خەنزۇچە بېكەتلەردىن چۈشىنەلمىگەن مەسىلىلەرگە كۈچۈمنىڭ يېتىشىچە ئاددى تىللار بىلەن جاۋاب بېرىپ كوپىيوتىردىن ئىبارەت بۇ ھازىرقى زاماندىكى ئەڭ ئىلغار ئەڭ مودا پەن تېخنىكا مەھسۇلاتىنىڭ تېخىمۇ كۆپ ئۇيغۇر ئائىلىلىرىگە سىڭىپ كىرىپ تېخىمۇ كۆپ ئۇيغۇر خەلقىنىڭ كومپىيوتىردىن پايدىلىنىپ يىراقنى يېقىن ، قىيىن ئىشنى ئاسان قىلىشى ئۈچۈن كۈچ چىقىرىش مەقسىتىدە مۇشۇ ۋەزىپىنى ئۆز ئۈستۈمگە ئېلىشنى قارار تاپتىم . ئەلۋەتتە باغداش مۇنبىرىمىزدە كومپيوتېر بىلىملىرى بىلەن خەلقىمىزنىڭ ئېھتىياجىنى قامداۋاتقان باشقا ئەزالىرىمىزمۇ بار ئۇلارنىڭمۇ بۇسەپكە قوتۇلىشىنى ئۈمىد قىلىمەن.
مەن بۇ كۆز قارىشىڭىزنى قوللايمەن .   سىز باشقۇرغۇچى بولۇپ سايلىنىڭ ياكى سايلىنالماي قېلىڭ بۇنىڭ كارايىتى چاغلىق . مېنىڭچە سىز يەنىلا بۇرۇنقىدەكلا كومپىيۇتېرغا ئائىت بىلىملىرىڭىزنى مۇنبەرگە يوللاپ تۇرۇشىڭىزنى ئۈمىد قىلىمەن . چۈنكى مەن ئۇزۇندىن سىزنىڭ بۇ ھەقتە يوللىغان قىممەتلىك بىلىملىرىڭىزدىن بەھىرلىنىپ كېلىۋاتىمەن .
