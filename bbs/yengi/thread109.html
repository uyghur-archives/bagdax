ئەسسىلامۇئەلايكۇم ،ھۆرمەتلىك باغداشنىڭ قىزغىن ئىشتىراكچىلىرى :
       ئالدى بىلەن باغداش تورىنىڭ بارلىق كونا -يېڭى ئەزالىرىغا قىزغىن سالام يوللايمىز ،ھەممىمىزگە مەلۇم  ئۆتكەن بىر يىلدا شىنجاڭ رايونىدىكى ۋەزىيەتنى كونتىرول قىلىش ئۈچۈن مۇناسىۋەتلىك ئورۇنلار شىنجاڭ رايونىدا ۋاقتىچە تورنى قامال قىلغان ئىدى ، شۇندىن كېيىن تورلارنى قايتا تەرتىپكە سېلىنىپ قايتىدىن كەڭ خەلق ئاممىسىغا ئېچىۋېتىلدى. تور بېكىتىمىزمۇ مۇناسىۋەتلىك ئورۇنلارنىڭ چوڭقۇر غەمخۇرلىقى ئاستىدا قايتىدىن تىزىمغا ئالدۇرۇلۇپ رەسمىي ئېلان قىلىندى!
             
   
  مۇشۇ مىنۇتلاردا ناھايىتى ھاياجانلىق ئىلكىدە سىلەرگە باغداش تور بېكىتىنىڭ قايتا ئېچىلغانلىقىنى ئۇقتۇرۋاتىمىز ،  ئەگەر مۇشۇ  مىنۇتلاردا سىلەرمۇ بىزگە ئوخشاش ھاياجان ۋە خوشاللىققا چۆمگەن بولساڭلار بۇ باغداشنىڭ ئۆتكەن ئىككى يىلدىكى غەلبىسى  . شۇ تاپتا ھەممىمىز خوشالمىز ، چۈنكى بىز ياقتۇردىغان، سىز ۋە بىزدەك نۇرغۇن تورداشلارنى بىر يەرگە جەم قىلغان ، نۇرغۇن ئەسلىمىلەرنى قالدۇرغان <باغداش تورى> تور دۇنياسىدا قايتا نامايەن بولدى ، نۇرغۇن ئەسلىمىلەر ھېلىمۇ يادىمىزدا، ئەنەئاشۇ قاينام تاشقىنلىققا چۆمگەن ،گۈلخان ئەتراپىدا قومۇل سەنىمىگە چۈشكەن نەۋرۇزلار ، مۆھتىرەم ،جاپاكەش ئانىلىرىمىزغا ئاددىغىنە داستىخان سېلىپ ئۇلارغا بولغان ھۆرمىتىمىزنى ،قەلبىمىزنى بىلدۈرۈپ كۆزىمىزگە ئىسسىق ياشلارنى ئالغان تەسىرلىك مىنۇتلار،  مىللىتىمىزنىڭ مۇنەۋۋەر پەرزەنتى ئۆتكۈر ئەپەندىنى خاتىرلەپ ئالىملىرىمىزغا بولغان ھۆرمىتىمىزنى  بىلدۈرگەن ئۇنتالماس مىنۇتلار  ھېلىمۇ ھەممىزنىڭ يادىدا تۇرۇپتۇ . بىز شۇ چاغلارنى ئەسلەيمىز ،ھەم يەنەشۇنىڭدەك ئەھمىيەتلىك پائالىيەتلەرنى قىلىش ئۈچۈن تىرىشىمىز .
لېكىن تورداشلار كۆرۈپ تۇرۇپسىلەر ، تور بېكىتىمىزدە ئەسلىدە بار بولغان ماتىرىياللار مەلۇم سەۋەپلەر تۈپەيلى يوقاپ كەتتى ، بىز تورداشلارنىڭ ماتىرىياللىرىنى ساقلاپ قالالمىدۇق ، بارلىق تورداشلاردىن تۆۋەنچىلىك بىلەن كەچۈرۈم سورايمىز . يەنە شۇنداقلاتور بېكىتىمىزنىڭ ئەسلىدىكى نامى بولغان bagdax.com تور ئېتىك مەزگىلدە چەتئەلدىكى كىشىلەر تەرىپىدىن ئېلىپ كېتىلدى ، شۇڭلاشقا بىز bagdax.cn نامىدىكى تور نامىنى ئىشىلتىشكە مەجبۇر بولدۇق ، شۇ سەۋەپلىك تورداشلارغا ئېلىپ كەلگەن قولايسىزلىقلاردىن كەچۈرۈم سورايمىز.تورداشلارنىڭ بۇرۇنقىدەك قىزغىنلىق بىلەن تور بېكىتىمىزنى قوللىشىنى چىن دىلىمىزدىن ئۈمىد قىلىمىز. بارلىق كونا تورداشلارنىڭ باغداشتا قايتا جەم بولۇشىنى  شۇنداقلا  بارلىق ئەزالارنىڭ بۇرۇنقى ئەزالىق ئىسمىنى ئىشلىتىشىنى تەۋسىيە قىلىمىز !

باغداشنى  قوللاڭ ، سىز بىزنى  قوللىسڭىز بىز سىز ئۈچۈن ئەڭ ياخشى بولغان تور مۇھىتى يارىتىپ بېرىشكە تىرىشىمىز ! تور بېكىتىمىزنىڭ باشقا سەھىپىلىرى تولۇق ئىشلىنىپ بولمىغاچقا ھازىرچە مۇنبەرنى ئېلان قىلىپ تۇردۇق ،تېز سۈرئەتتە باشقا خىلمۇ-خىل سەھىپە ۋە رەڭگارەڭ قاناللارنى كەڭ تورداشلار بىلەن يۈز كۆرۈشتۈرۈشكە تىرىشىمىز !


مۇكەممەللىك مەۋجۇت ئەمەس ، لېكىن بىز تىرىشىمىز !

  تور بېكىتىمىزنىڭ كېيىنكى تەرەققىياتى ۋە ساغلام، نورمال ھالەتتە يۈرگۈزىلىشى ئۈچۈن تۆۋەندىكىدەك دىققەت قىلىشقا تېگىشلىك نۇقتىلارنى ئالاھىدە ئۇقتۇرىمىز:
1.تور بېكىتىمىزدە دۆلىتىمىزنىڭ تۈرلۈك سىياسەتلىرىگە قارشى ماقالىلەرنى يوللاشقا بولمايدۇ .
2.مىللەتلەر ئىتتىپاقلىقى ۋە مۇقىملىققا زىيان يەتكۈزىدىغان ماقالىلەرنى يوللاشقا بولمايدۇ.
3.شەخس ياكى ئورۇنلارنى ھاقارەتلەيدىغان.مەنپەئەتىگە تەسىر يەتكۈزىدىغان تېمىلارنى يوللاشقا بولمايدۇ.
4.كېلىش مەنبەسى ئېنىق بولمىغان ماقالە - ئەسەرلەرنى يوللاشقا بولمايدۇ.
توربېكىتىمىزنىڭ ساغلام تەرەققىياتى ئۈچۈن يوقىرىقى ئۇقتۇرۇشنى ئالاھىدە ئۇقتۇردۇق،ئەگەر يوقىرقىدەك ئەھۋاللار كۆرۈلسە يوللىغۇچى بارلىق مەسئولىيەتنى ئۆز ئۈستىگە ئالىدۇ .

دىققىتىڭىز داۋاملىق باغداشنىڭ كېيىنكى تەرەققىياتىدا بولسۇن !

مەن تەپسىلىي چۈشەندۈرۈپ قويمىساممۇ ھەممە ئىش ھەممىمىزگە ناھايىتى ئېنىققۇ
  جانابى ئالاھىدە باشقۇرغۇچى ئەپەندىم  


كۆكلەي دىسەڭ ئۇيغۇرۇم ، سۇ بولايمۇ يىلتىزمۇ ! دىگەن ماقال -تەمسىلنى مېڭىڭىزگە تىقىپ كىرگۈزۋەتكىلى بولغان بولسا كاشكى ....
ھېچقانداق باش ئاخىرى يوق ئىنكاسنى يېزىپ مەقسىتىڭىزنىڭ نىمىلىكنى بىلگىلى بولمايدۇ ،ھەر قانداق تەنقىدى پىكرىڭىز بولسا ئېنىق ئوتتۇرىغا قويۇڭ ، بۇ يەردە باشقىلارغا خاتا چۈشەنچە بېرىشىڭىزنى ئۈمۈد قىلمايمىز !   ھەرقانداق پىكرىڭىز بولسا ئوتتۇرىغا قويسىڭىز  جاۋاپ بېرىشكە قوبۇل قىلىشقا ھەر ۋاقىت تەييارمىز !
