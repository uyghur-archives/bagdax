يوللىغان ۋاقتى 2012-10-3 09:30:00 |ھەممە قەۋەتنى كۆرۈش
  سىز بۇ يەردە مىللەتچىلىك قىلماڭ !. مەن ئالدىنقى بىرنەچچە ئىنكاسىمدىمۇ ئېنىشتىيىننىڭ ھەقىقەتەنمۇ خاتالاشقىنىنى دەپ ئۆتتۈم ، ھەم يەنە مەنمۇ بىر يېڭى ئۆگەنگۈچى سىزگە ئوخشاش فىزىكا ئالىمى ئەمەس  . نىسپىيىلىك نەزەرىيىسىنى تولۇق چۈشەنمەيدىغىنىم ئۇ مۇقەرەر ، سىزمۇ شۇنىڭ ئىچىدە . بەلكىم سىز مەنچىلىكمۇ چۈشەنمەسلىكىڭىز مۇمكىن ئۇ نەزەرىيەنى .
مەن تورغا كىرىشنى ئۆگەنگەندىن تارتىپ مۇشۇ تېمىغا چۈشكەن ئىنكاسقىچە ئىشتىن سىرتقى فىزىكا ئالىملىرىنىڭ پۈتۈن ھۇجۇدى بىلەن ئېنىشتىيىننىڭ خاتالاشقىنىنى ئىسپاتلىماقچى ياكى ئىسپاتلىغانلىغانلىقى توغرىسىدىكى ماقالە ۋە ئىنكاسلارنى بەكمۇ تولا كۆرۈپ كەتتىم .  ئۆتكەنكى  80 يىلدىن بۇيان نىسپىيىلىك نەزەرىيىسى باشتىن ئاخىرى غەلبە قىلىپ كەلدى . تىخى مۇشۇ چاغقۇچە ھېچقانداق بىر تەجرىبە نىسپىيىلىك نەزەرىيىسى ئوتتۇرىغا قويغان يەكۈنلەرنىڭ خاتالىقى بارلىقىنى ئىسپاتلىيالمايۋاتىدۇ .
‹باشقىچە پىكرى بارلار مىللەتچىلىك قىلماي ، پىكرىڭلارنى دوستانە ئوتتۇرىغا قويارسىلەر›--بىرەر نەزىريەنىڭ خاتالىغىنى كۇرۇپ تۇرۇپ بىلمەسكە سېلىش،بۇ ئىلمى قاراش بولالمايدۇ،بۇنداق گۇدەكلەرچە قاراش تېخىمۇ ئەسقاتمايدۇ،مەيلى قانداق مىللەت بولۇشىدىن قەتتى نەزەر ئىلىم -پەن ساھاسىدە ئالدامچىلىق قىلسا بولمايدۇ،يەھۇدىلارمۇ دەل بۇنىڭ ئىچىدە،نىسپىلىك نەزىريەسى قانداق قىلىپ غەلبە قىلىپ كەلدى،بۇنى قايسى مىسال ئارقىلىق چۇشەندۇرۇشنى لايىق تاپتىڭئىزكىن يوپىتېر ئەپەندىم،بىز بۇ يەردە مەن ھەۋەسكارمەن دىگەن شۇئارنى تاۋلىساق بولىدۇ،بۇ بىزنىڭ ئىلگەرلەش ئىمكانىيىتى بولمىغاندا چېكىنىشكە قالدۇرغان يولىمىز،چۇشۇنۇشكە بولىدۇ،لېكىن نىسپىلىك نەزىريەسى بىلەن كۇۋانىت مېخانىكىسى ھازىرقى زامان فىزىكىسىنىڭ ئاساسى،دەپ قارالغىلى خېلى زامانلار بولدى،مەن بىر ھەۋەسكار دىگەن سۇزنى جوڭگۇ پەنلەر ئاكادىمىيىسى نەزىريەۋى فىزىكا تەتقىقات ئۇرنىدىكى ئاكادىمىك ،فېرافېسسورلارمۇ شۇنداق بايان قىلغان،شىنجاڭ فىزىكا جەمىيىتىغۇ بۇنى بۇرۇنلا دەپ بولغان،باق تاپشۇرغۇچىلارنىڭ پۇلىنى بۇ فىزىكلارغا تەتقىقات خىراجىتى قىلىپ بېرىۋاتقان مۇشۇ زاماندا،دوكتۇرلار ئەڭ ئادەتتىكى دەپ قارىلىۋاتقان نەزىريەنىمۇ چۇشۇنەلمىسە،بۇنى ئۇقۇغۇچىلارغا دەرسلىك قىلىپ ئوتسە،ئەلۋەتتە بۇ دەرسلىكلەر ئۇچۇن ئوقۇغۇچىلار ئوقۇش پۇلى تۇلەيدۇ،بۇ يەردە كىشىنى سەسكەندۇرىدىغان بىر ئالدامچىلىقنىڭ بارلىغىنى كۇرۇپ يەتكىلى بولماسمۇ،نەچچە كۇن بۇرۇن شىنجاڭ ئۇنىۋېرىستېتىنىڭ فىزىكا كەسپىدىكى بىر فېرافېسسور بىلەن بىر يەردە بولۇپ قالدىم،بۇمۇ تاساددىبى توغرا كېلىپ قېلىش،گەپتىن -گەپ چىقىپ،ھەقىقى ئەھۋالنى يۇشۇرمايلا بىر سۇزنى قىلىشتۇق،{شىنجاڭ ئۇنىۋېرىستېتىدىن كارغا يارىدۇدەك ،يا كىلاسسىك فىزىكىدىكى يا ھازىرقى زامان فىزىكىسىدىكى ئاساسى نەزىريەنى چۇشۇنىدىغان بىرەر ئوقۇتقۇچى يوقكەن،قىلىۋاتقان ئىشلىرىنىڭ بىر قىسمى ساپلا ئالدامچىلىق،بۇنى قانداق چۇشەندۇرۇسىزكىن}فېرافېسسور ،بۇنى كېيىن چۇشەندۇرۇدىغان بولدى،مەن بۇنىڭ مۇمكىن ئەمەسلىگىنى شۇ يەردىلا دىدىم،ئاخىرقى سۇزدە بىز يەنە ئالاقە قىلىدىغان بولۇشتۇق،كۇرۇشكە بولۇدىكى بۇ نەزىريەنى بىرەر-ئىككى ئادەمنىڭ خاتا دىيىشى بىلەن ھىچ بىر ئىشنى قىلغىلى بولاسمىكىن،مويىمى دەلىل ئىسپات،يەنى پاكىت.

بۇ مەزمۇن 37قەۋەتتىكى ھاكىمدە2012-09-18 22:08يوللىغان يازمىسىغا نەقىل :

 سىز شۇنداق قارىغان بولسىڭئىز بۇ سىزگىلا ۋەكىللىك قىلىشى مۇمكىن ،{ئۇچ پادىشاھلىق دەۋرىدە،دوڭجۇ پادىشاھنى ئالماشتۇرماقچى بولغاندا،يۇئەنشاۋ شۇ زامان قارشى چىققان ئىكەن،يەنى  ئۇ دوڭجوغا {پادىشاھنى قانداقمۇ خالىغانچە ئالماشتۇرغىلى بولسۇن،پادىشاھ دىگەننى سەن ئالماشتۇرۇدىغان نەرسىمۇ ئۇ؟}بۇندىن 320نەچچە يىل بۇرۇن نيوتون تەڭرىنىڭ {اللاھ}نىڭ ئورنىغا ئۇزىنى ئالماشتۇرغان تۇرسا،بۇنى شۇ فىزىكا ساھاسىدىكىلەر ياۋاشلىق بىلەن قۇبۇل قىلغان،بۇ ھازىرمۇ داۋاملىشىۋاتىدۇ،سىز بۇ ئىشنى تەنقىت قىلماي،قانداق تاقەت قىلدىڭئىز،ئادەم نىمىگىلا ئېتىقات قىلسا بولىۋېرەمدۇ قانداق،ئۇنداقتا ئىمان دىگەن سۇزنى قانداق چۇشۇنۇش كېرەك،يەنە كېلىپ بۇ  تورداشنىڭ دىگىنىنى تامامەن ئورۇنسىز دىگىلى بولماس،مەن بانكىدا ئىشلەيمەن،بۇ تەتقىقات بەكلا مويىم بولغانلىغى ئۇچۇن موسكۇۋا ئۇنىۋېرىستېتىنى پۇتتۇرۇپ،فىزىكا ئىلىمى بويىچە دوكتۇرلۇقنى تاماملىغان ،شىنجاڭ تەبىى پەنلەر ئاكادىمىيىسىدە ئىشلەۋاتقان تۇردى  ئۇمەر ئەپەندىمنىڭمۇ بۇ ماقالىگە قوشقان توھپىسى ئاجايىپ زورغۇ،،يەنە كېلىپ بۇ ئەپەندىم ھازىر خوتەن قارىقاش ناھىيەسىدە مۇقۇملۇق خىزمىتى قىلىۋاتقانلىغى ئۇچۇن دائىم تورغا چىقىشمۇ تەسكە توختايدۇ،سۇزىڭئىزنى ھامان بىر كۇنى كۇرىدۇ،مەن ئالدى بىلەن يازغان ئىنكاسىڭئىزنى بۇ ئەپەندىمنىڭ خەت ساندۇغىغا يوللاپ كوياي،كۇلۇپ كېتىدىغان بولدىدە،لېكىن بۇ تۇردى قارشى پىكارلەرگە ناھايىتى ئەستايىدىل قارايدۇ،چۇنكى بۇ دۇنيادا تەڭرىلا خاتالاشمايدۇ،ئادەم بولغاندىن كېيىن خاتالىقتىن ساخلىنىش تەس،خاتالىقنى بىلگەندە ئۇنى داۋاملىق ئوتكۇزمىگەن تۇزۇكمىكىن،ئەلۋەتتە،سىز نيوتوننىڭ تەڭرى ئىكەنلىگىنى داۋاملىق ئېتىراپ قىلسىڭئىز بولىۋېرىدۇ،اللاھمۇ بۇنى بەك سۇرۇشتۇرۇپ كەتمەس،بۇ جاھاندا ئازغان بەندە دىگەن تۇلا گەپ.يەنە كېلىپ بۇ ماقالىگە نۇرغۇن ئادەم ياردەمدە بولغان،شۇنىڭ ئۇچۇن  ئەينى چاغدا 5مىليون ئامىرىكا دوللىرى دو تىكىپ،پۇتۇن دۇنياغا بۇ ماقالىنىڭ بۇرۇنقى نۇسخىلىرىنى ئەۋەتكەن،  بىزدە {يالغۇز ئاتنىڭ ھەرقانچە قىلسىمۇ چېڭى چىقماس دىگەن سۇز بارغۇ،يەنە كېلىپ ئۇيغۇر دىگەن سۇزنىڭ مەنىسى ئىتپاقلاشماق دىگەن مەنىنى بىرىدىغۇ دەيمەن]  

ھاكىم ئەپەندى.بۇلارنىمۇ توغرا چۈشىنىڭ. نېمىشقا دېسىڭىز، جاھان ئۇچقاندەك تەرەققىي قىلىۋاتقان بۈگۈنكى كۈندە،ئۆزلىكىنى ساقلاپ قېلىش ئۈچۈن توختىماي تىپچەكلەۋاتقان بىز ئۇيغۇرلار،ئەنە شۇنداق مەشھۇرلارنىڭ كۆپلەپ چىقىشىغا مۇھتاج.سەنئەت جەھەتدە ، ياكى ئۆزگىچە مۆجىزىلىك ماھارەت جەھەتدە ئارقا-ئارقىدىن مەيدانغا كەلگەن ماھىرلار، قانچىلىرىمىزنى ھاياجانلاندۇرۇپ "ئۇيغۇرۇمنى تۇنۇتتى!"دەپ ئىسسىق كۆز ياشلىرىمىزنى ئاققۇزغان بولسا،ئەمدى تەببىي پەن ساھەسىدە بىر ئالىمنىڭ چىقىشى، يەنە كېلىپ ئۇنىڭ ئايال كىشى بولۇشى، نۇرغۇنلىرىمىزنى ھەتتا "ياشىسۇن ساھىبجامال، ياشىسۇن، تۈمەن مىڭ يىل ياشىسۇن!"دەپ شۇئار توۋلاتقۇزۇۋېتىش دەرىجىسىگىمۇ بارغۇزىۋەتتى.نۇرغۇنلىرىمىز "ئۇ زادى نېمىنى تەتقىق قىلدى؟بۇ تەتقىقاتنىڭ ئەھمىيىتى نېمە؟"دېگەننى ئويلاپمۇ قۇيۇشمىدۇق ھەم ئويلىيالمايمىز!لېكىن، بۇ خانىمنىڭ مەشھۇرلۇق ئىندوكسىيىسى نۇرغۇن ياشلارغا كۈچ-مەدەت، ئىلھام بولدى،قىسقىسى مىللىتمىزنىڭ روھىنى ئۇرغۇتتى.كۆكنۇر شىركىتىمۇ بەلكىم ياشلىرىمىزنىڭ قىزغىنلىقىغا ئاۋاز قوشۇپ،ئىقتىساد جەھەتدە قوللاپ، بۇ پائالىيەتنى ئويۇشتۇرغان بولۇشى مۇمكىن.بۇ ئەلۋەتتە نۇرغۇن ياشلىرىمىزغا ياقتى، كۆڭلىدىكىدەك ئىش بولدى!دەرۋەقە،بۇ ئىش مىللىي پىسخىكىمىزغا بەكلا ياقتى. بۇ ئەلۋەتتە كۆكنۇر شىركىتىنىڭ ئىناۋىتىگە ئاكتىپ تەسر كۆرسىتىپ، ئىشلىرى تېخىمۇ يۆكسىلىشى مۇمكىن.ئىشقىلىپ،چۈشەنسىڭىز ھەممىسى بىر-بىرىگە ئوڭ تاناسىپ ئىشلار بۇ.ھەم مىللىتىمىزنىڭ "قول قولنى يۇسۇسا، قول قوپۇپ يۈزنى يۇيىدۇ"پەلسەپىسىگىمۇ ماس كېلىدۇ.شۇڭا،ھېچ بولمىغاندا مۇشۇ نوقتىدىن، سىزنىڭ تەتۈر مۇقام توۋلىشىڭىز ،ياشلىرىمىزنىڭ قىزغىنلىقىغا سوغۇق سۇ سېپىشىڭىز،تەببىيكى نۇرغۇن تورداشلارغا ياقمىدى.ئەمدى "راس مۇشۇ ساھىبجامال خانىمنىڭ قولغا كەلتۈرگەن نەتىجىلىرىنىڭ ماھىيىتى نېمە؟ توغرىمۇ خاتامۇ ؟"دېگەندەك مەسىلىلەر، بۇ مۇنبەردىكى كۆپ قىسىم تور ئەھلىنىڭ كاللا قاتۇرىدىغان ، ھەم كاللا قاتۇرالايدىغان دائىرىسىدىكى ئىش بولمىغاچقا،نۇرغۇنلىرى سىزگە ئاچچىق گەپ قىلىپ، زەردە گۆش بولۇپ ئولتۇرۇشتىن باشقىنى بىلمەيدۇ.شۇڭا  تەتقىقات نەتىجىلرىڭىز بولسا ، بىر-بىرلەپ ئوتتۇرىغا قۇيۇڭ.بىز چۈشەنمىسەكمۇ ،چۈشىنىپ مۇناسىپ مۇنازىرەلىشىدىغانلار چىقىپمۇ قالار.بىر بىلىپ باقايلى.بولمىسا مۇنازىرىڭىز بۇ تەرىزدە داۋاملاشسا،ھېچكىمنى قايىل قىلالمايلا قالماستىن،ئەكسىچە ئۆزىڭىزگە "كۆرەلمەس، پىتنە-پاساتچى، غەيۋەتخور"دىگەندەك سېسىق قالپاقلارنى يۈكلىۋېلىشىڭىز مۇمكىن. قالغىنىنى ئۆزىڭىز ئويلاڭ.

ھاكىم ئەپەندى.بۇلارنىمۇ توغرا چۈشىنىڭ. نېمىشقا دېسىڭىز، جاھان ئۇچقاندەك تەرەققىي قىلىۋاتقان بۈگۈنكى كۈندە،ئۆزلىكىنى ساقلاپ قېلىش ئۈچۈن توختىماي تىپچەكلەۋاتقان بىز ئۇيغۇرلار،ئەنە شۇنداق مەشھۇرلارنىڭ كۆپلەپ چىقىشىغا مۇھتاج.سەنئەت جەھەتدە ، ياكى ئۆزگىچە مۆجىزىلىك ماھارەت جەھەتدە ئارقا-ئارقىدىن مەيدانغا كەلگەن ماھىرلار، قانچىلىرىمىزنى ھاياجانلاندۇرۇپ "ئۇيغۇرۇمنى تۇنۇتتى!"دەپ ئىسسىق كۆز ياشلىرىمىزنى ئاققۇزغان بولسا،ئەمدى تەببىي پەن ساھەسىدە بىر ئالىمنىڭ چىقىشى، يەنە كېلىپ ئۇنىڭ ئايال كىشى بولۇشى، نۇرغۇنلىرىمىزنى ھەتتا "ياشىسۇن ساھىبجامال، ياشىسۇن، تۈمەن مىڭ يىل ياشىسۇن!"دەپ شۇئار توۋلاتقۇزۇۋېتىش دەرىجىسىگىمۇ بارغۇزىۋەتتى.نۇرغۇنلىرىمىز "ئۇ زادى نېمىنى تەتقىق قىلدى؟بۇ تەتقىقاتنىڭ ئەھمىيىتى نېمە؟"دېگەننى ئويلاپمۇ قۇيۇشمىدۇق ھەم ئويلىيالمايمىز!لېكىن، بۇ خانىمنىڭ مەشھۇرلۇق ئىندوكسىيىسى نۇرغۇن ياشلارغا كۈچ-مەدەت، ئىلھام بولدى،قىسقىسى مىللىتمىزنىڭ روھىنى ئۇرغۇتتى.كۆكنۇر شىركىتىمۇ بەلكىم ياشلىرىمىزنىڭ قىزغىنلىقىغا ئاۋاز قوشۇپ،ئىقتىساد جەھەتدە قوللاپ، بۇ پائالىيەتنى ئويۇشتۇرغان بولۇشى مۇمكىن.بۇ ئەلۋەتتە نۇرغۇن ياشلىرىمىزغا ياقتى، كۆڭلىدىكىدەك ئىش بولدى!دەرۋەقە،بۇ ئىش مىللىي پىسخىكىمىزغا بەكلا ياقتى. بۇ ئەلۋەتتە كۆكنۇر شىركىتىنىڭ ئىناۋىتىگە ئاكتىپ تەسر كۆرسىتىپ، ئىشلىرى تېخىمۇ يۆكسىلىشى مۇمكىن.ئىشقىلىپ،چۈشەنسىڭىز ھەممىسى بىر-بىرىگە ئوڭ تاناسىپ ئىشلار بۇ.ھەم مىللىتىمىزنىڭ "قول قولنى يۇسۇسا، قول قوپۇپ يۈزنى يۇيىدۇ"پەلسەپىسىگىمۇ ماس كېلىدۇ.شۇڭا،ھېچ بولمىغاندا مۇشۇ نوقتىدىن، سىزنىڭ تەتۈر مۇقام توۋلىشىڭىز ،ياشلىرىمىزنىڭ قىزغىنلىقىغا سوغۇق سۇ سېپىشىڭىز،تەببىيكى نۇرغۇن تورداشلارغا ياقمىدى.ئەمدى "راس مۇشۇ ساھىبجامال خانىمنىڭ قولغا كەلتۈرگەن نەتىجىلىرىنىڭ ماھىيىتى نېمە؟ توغرىمۇ خاتامۇ ؟"دېگەندەك مەسىلىلەر، بۇ مۇنبەردىكى كۆپ قىسىم تور ئەھلىنىڭ كاللا قاتۇرىدىغان ، ھەم كاللا قاتۇرالايدىغان دائىرىسىدىكى ئىش بولمىغاچقا،نۇرغۇنلىرى سىزگە ئاچچىق گەپ قىلىپ، زەردە گۆش بولۇپ ئولتۇرۇشتىن باشقىنى بىلمەيدۇ.شۇڭا  تەتقىقات نەتىجىلرىڭىز بولسا ، بىر-بىرلەپ ئوتتۇرىغا قۇيۇڭ.بىز چۈشەنمىسەكمۇ ،چۈشىنىپ مۇناسىپ مۇنازىرەلىشىدىغانلار چىقىپمۇ قالار.بىر بىلىپ باقايلى.بولمىسا مۇنازىرىڭىز بۇ تەرىزدە داۋاملاشسا،ھېچكىمنى قايىل قىلالمايلا قالماستىن،ئەكسىچە ئۆزىڭىزگە "كۆرەلمەس، پىتنە-پاساتچى، غەيۋەتخور"دىگەندەك سېسىق قالپاقلارنى يۈكلىۋېلىشىڭىز مۇمكىن. قالغىنىنى ئۆزىڭىز ئويلاڭ.---- كوكنۇر شىركىتىنىڭ بۇ پائالىيەتنى ئەسلى پەرھات مۇئەللىم ئۇيۇشتۇرغان ئىكەن،نەچچە كۇن بۇرۇن بىر خىزمەتدىشىمىزنىڭ دادىسى قازا قىلغاندا، ئىدارىدىكى خىزمەتداشلار بىلەن ئىشتىن چۇشكەندە  پەتىلەپ بارغان،شۇ ئۇيدە پەرھات دوكتۇرنى كۇرۇپ قالدىم،ئوغلى بىلەن بىللە ئىكەن،ھەم ئۇي ئىگىسىنىڭ تاغىسى ئىكەن،پارھات مۇئەللىم بىلەن تېلىفوندا قانچە قېتىملاپ كۇرۇشكەن،يۇز تۇرانە كۇرۇشۇمۇ ئۇنىڭ ئىشخانىسىدا بولغان،گەپتىن گەپ چىقىپ ساھىپجامالنىڭ گېپى چىققاندىن كېيىن ئاندىن ئۇرۇمچى تېلۋىزىيە ئىستاسىسىنىڭ پۇرۇگىراممىسى ھەققىدىكى سۇز چىقىپ قالدى،ساھىپجامالنىڭ 3-ئايدا مۇنازىرىگە قوشۇلۇپ،4-ئاينىڭ ئاخىرى گېپىدىن يېنىپ قالغانلىغىنى ئېيتقىنىمدىن كېيىن ،پەرھات ئەپەندىم ھەيران قالدى،ھەم بۇ مۇمكىن ئەمەس،ساھىپجامالنىڭ ئىلىم -پەن روھى كۇچلۇك،نيوتون قانۇنلىرىمۇ خاتا ئەمەس،بۇنى تارىخ قايا -قايتا ئىسپاتلىغان،تۇرسا،يەنە كېلىپ بىز بۇ قانۇنلاردىن پايدىلىنىپ نۇرغۇن ئىشلارنى قىلغان،...ۋاھاكازا،لېكىن بىرلا مەسىلە،خوش ئۇنداق بولسا،پەرھات ئەپەندىم،سىز بۇ مۇنازىرىگە چۇشەلەملا،.....،چەشەلمەيدىغان بولدى،قاراڭ،بىزنىڭ نەزىريە راستىنلا ئىلمى ئاساسى بولمىسا،پۇتۇن دۇنيادىكى فىزىكا ئالىملىرىغا 5مىليون دوللار دو تىكىش بۇ ھاماقەتلىك بولىدۇ،بۇ پۇل دۇلىتىمىزنىڭ پۇلىغا سۇندۇرغاندا ،42مىليون خەلق پۇلى بولىدۇ،بۇ پۇلنى راستىنلا ئۇتتۇرۇپ قويساق،بۇنى تولىيەلمەيمەن دىسەك ،بۇ قاملاشماسمىكىن،پەرھات دوكتۇرنىڭ دىيىشىچە،خەلقىمىز ئىچىدە تەبىى پەننى چۇشۇنىدىغان ئادەم ئاز سانلىقنى ئىگىلەيدۇ،بەكلا ئاز،شۇنىڭ ئۇچۇن مەن ساھىپجامالنى تەشۋىق قىلغان،سىز ماقالىنى خەلقئارالىق ژورنالدا ئېلان قىلىپ،ئېلىپ كەلسىڭئىز ،قالغان ئىشنى مەن قىلاي دىدى،ئەلۋەتتە،رەخمەت دىدىش زۇزۇر ئىدى،لېكىن مەن،بىز بۇنداق قىلساق،خەلقىمىزنى يەنە بىر مەزگىل ئالدىغان بولىمىز،بىز خەلقئارالىق  ژورنالدا ئېلان قىلىپ بولدۇق،بولمىسا،ئۇرۇمچى تېلۋىزىيە ئىستانسىسى بۇنى تەشۋىق قىلىشتا تەۋەككۇل قىلمىغان بولاتتى دىدىم،ئومومەن،خۇلاسە شۇكى،پەرھات دوكتۇرمۇ بۇ مۇنازىرىگە چۇشەلمەيدۇ،تورداشلار ئالدانغان بولغاچقا ،پىتنىخور،كۇرەلمەس،غەيۋەتخور دىگەن سۇزلەرنى ئىشلەتكەن بولۇشى مۇمكىن،بۇ خۇددى بىر تازنىڭ چېچى بار ئادەمنى تاز دىگىنىدەك ئىش تۇرسا،بۇنىڭغا كۇلۇپ قويسىڭئىزلا بولىدۇ،يەنە كېلىپ ،بۇ تازغا ئىچىڭئىز ئاغرىپ قالىدۇ،چۇنكى بۇ كىشى تاز بولۇپلا قالماي،كۇرۇش قۇۋىتىمۇ ئاجىزلاپ كەتكەن،بۇ ئۇنىڭ نېرۋىسىنىڭ تەرەقىياتى بىلەن ئوڭ تاناسىپ بولۇپ قالغان،قاراڭ،گاچابولىۋالما ئەپەندىم،بۇ خىلدىكى كىشىلەر ئۇزلىرىنىڭ ئالدانغانلىغىنى مەڭگۇ بىلمەي قالۇرمۇ؟بۇ پەقەت ۋاقىتلىق ئەھۋال،ھەققانىيەت دىگەننى بىرنەچچە كىشىنىڭ كۇچىگە تايىنىپ توسۇپ قېلىش مۇمكىن ئەمەسقۇ،

 ئۇنتۇپ قاپتىمەن،بىز ئۇيغۇرلار ئۇزۇناراق خوشال بولساق بولاتتى،چۇنكى بىزنىڭ ئەدەبىيات ساھاسىدىكى نەتىجىلىرىمىز كوپ،تىىبى ساھادىكى كىتاپلارنىڭ ھەممىسىنى ئاففاق غۇجا كۇيدۇرىۋەتكەن ئىكەن،شۇنىڭدىن قالغانلىرى ،كېيىنكى تەتقىقاتلار ئارقىلىق قولغا كەلگەنلىرى ھازىر چىقىۋاتىدۇ،قولىمىزدا ئالتۇندەك پاقىراپ تۇرىۋاتقان نەتىجىلەر بولغان بىلەن بۇنىڭدىن پايدىلىنىپ،ئالى مەكتەپنى پۇتتۇرگەن ئۇغۇل قىزلىرىمىزنى كوپەرەك ئىشقا ئورۇنلاشتۇرۇدىغان ئىشلارنىمۇ قىلالمايۋاتىمىز،تەبىى پەندە ساھىپجامالدەك بىرەيلەننىڭ چىقىشى كىشىنى خوشال قىلىدۇ،لېكىن بۇ ئۇزۇنراق بولسا ياخشى بولاتتى،ئىلىم -پەن ساختىپەزلىكنى كۇتۇرمەيدۇ،مېنىڭ ئەندىشە قىلىدىغىنىممۇ مۇشۇ،

 ئۇنتۇپ قاپتىمەن،بىز ئۇيغۇرلار ئۇزۇناراق خوشال بولساق بولاتتى،چۇنكى بىزنىڭ ئەدەبىيات ساھاسىدىكى نەتىجىلىرىمىز كوپ،تىىبى ساھادىكى كىتاپلارنىڭ ھەممىسىنى ئاففاق غۇجا كۇيدۇرىۋەتكەن ئىكەن،شۇنىڭدىن قالغانلىرى ،كېيىنكى تەتقىقاتلار ئارقىلىق قولغا كەلگەنلىرى ھازىر چىقىۋاتىدۇ،قولىمىزدا ئالتۇندەك پاقىراپ تۇرىۋاتقان نەتىجىلەر بولغان بىلەن بۇنىڭدىن پايدىلىنىپ،ئالى مەكتەپنى پۇتتۇرگەن ئۇغۇل قىزلىرىمىزنى كوپەرەك ئىشقا ئورۇنلاشتۇرۇدىغان ئىشلارنىمۇ قىلالمايۋاتىمىز،تەبىى پەندە ساھىپجامالدەك بىرەيلەننىڭ چىقىشى كىشىنى خوشال قىلىدۇ،لېكىن بۇ ئۇزۇنراق بولسا ياخشى بولاتتى،ئىلىم -پەن ساختىپەزلىكنى كۇتۇرمەيدۇ،مېنىڭ ئەندىشە قىلىدىغىنىممۇ مۇشۇ،  

دىگىنىڭزدەك بولسۇن , مەن فىزىكىنى ئانچە بەك چۈشەنمەيمەن , قارىسام سىزنى نۇرغۇن تورداشلار ئەيىلەۋېتىپتۇ , ئەگەر سىزنىڭ پىكىرلىرىڭىز ھەقىقەتەن خاتا بولسا ھەق يولدا مېڭىڭ , ئەمىلىيەتكە يۈزلىنىڭ كۈپچىلىك بىلەن بىر يولدا مېڭىپ ياخشى نەتىجە قازىنىڭ

 پەقەت ئىلمى ئۆگىنىش پوزىتسىيىسى بولغاندىلا ئاندىن ئىلىم-پەننىڭ پەللىسىگە يەتكىلى بولىدۇ. ئۇنداقتا، ئىلمى ئۆگىنىش پوزىتسىيىسى دېگەن نىمە؟ ئىلمى ئۆگىنىش پوزىتسىيىسى بولسا بىلىدىغاننى بىلىمەن دېيىش بىلمەيدىغاننى بىلمەيمەن دېيىش، بىلمەيدىغان نەرسىنى بىلىدىغان قىياپەتكە كىرىۋالماسلىق، مۇستەھكەم پۇىتا قەدەم بىلەن ئىلگىرلەش، ئاز-ئازدىن ئۆگىنىش ئارقىلىق ھەممە نەرسىنى ئايدىڭلاشتۇرۋىلىش دېمەكتۇر.----مەنغۇ ساھىپجامال مۇئەللىمنىڭ تېخخمۇ يۇقۇرى نەتىجىلەرنى قولغا كەلتۇرۇشىگە تىلەكداشمەن،لېكىن ئالدامچىلىق قىلماسلىغىنى،بولۇپمۇ ئۇز مىللىتىنى ئالدىماسلىغىنى ئومۇت قىلاتتىم،خانىم بىلىدىغان بىلەن بىلمەيدىغانلىغىنى تازا پەرق قىلىپ كېتەلمەيدۇ، لېكىن بۇ ماقالىدە بۇلار تىلغا ئېلىنماپتۇ،نيوتون ۋە ئېينىشتىيىنلار ئۇز نەزىريەلىرىنى تۇلۇق چۇشەنگەن ئەمەس،لېكىن خۇشامەت تۇسىنى ئالغان مۇشۇ خلدىكى نۇرغۇن ماقالە كىتاپلار يېزىلغان،بۇ ماقالىمۇ شۇنداق تۇس ئېلىپ قاپتۇ،ئىنسانىيەتنى بىر مەزگىل ئالدىغىلى بولىدۇ،لېكىن ئالدامچىلىقنىڭ ئۇمۇرى ئۇنداق ئۇزۇن بولالىشى تەس،توغلۇچ ئەپەندىمنىڭ نىمە ئۇچۇن ئىلمى پوزىتسىيەنى قوللانماي،گۇپپاڭچىلىق ئۇسۇلىنى قوللانغىنىغا ،ئىلمى پۇزىتسىيە قوللۇنۇشنى خالىمىغانلىغىنى چۇشىنگىلى بولمايدىكەن،  





ياخشىمۇ سىز؟ ھاكىمدەنىڭ 6-قانۇنىنى ئوتتۇرا قويغانلىغىڭىزنى قىزغىن تەبرىكلەيمەن.
