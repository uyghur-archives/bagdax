　　خەلق گېزىتىدە بۈگۈن ئىستىلدا بۇرۇلۇش ياساشنىڭ ئۈنۈمى ھەققىدىكى «ساختا پىكىر ساندۇقىنى تۈگىتىپ، ھەقىقىي مەسىلىنى ھەل قىلايلى» ماۋزۇلۇق ئوبزورچى ماقالىسى ئېلان قىلىندى. ماقالىدا مۇنداق دەپ كۆرسىتىلدى: تورداشلارنىڭ ئىنكاس قىلىشىچە، بەزى جايلاردىكى شەھەر باشلىقى خەت ساندۇقى، تارماقلارنىڭ سۆز قالدۇرۇش تاختىسى تەكەللۇپ قىلغان سۆزلەرگە ئايلىنىپ قالغان. بۇنىڭغا جاۋاب ۋاقتىدا قايتۇرۇلۇپ، سەمىمىي پوزىتسىيە تۇتۇلۇپ، «بەرگەن پىكرىڭىز ئىنتايىن ياخشى، جەزمەن يۈكسەك ئەھمىيەت بېرىش كېرەك» دەپ جاۋاب قايتۇرۇلغان. لېكىن ھەقىقىي مەسىلە ھەل قىلىنمىغان. ئىستىلدا بۇرۇلۇش ياساشنى چوڭقۇر ئىلگىرى سۈرۈشتە جايلار تارماقلار ھەقىقىي تەدبىر قوللىنىپ، ياخشى ئۈنۈم ھاسىل قىلدى، ئەمما يەنە يۇقىرىقىدەك «ساختا پىكىر ساندۇقى» ئەھۋاللىرىمۇ ساقلانماقتا.

　　ماقالىدا مۇنۇلار تەكىتلەندى: «ساختا پىكىر ساندۇقى» ئەھۋالىنى تۈگىتىشنىڭ ئاچقۇچى ھەقىقىي مەسىلىنى ھەل قىلىشتا. ھەر دەرىجىلىك كادىرلار ئىستىلدا بۇرۇلۇش ياساشنى ئامما ئۈچۈن ئەمەلىي ئىش، ياخشى ئىش قىلىش، ئاممىنىڭ قىيىنچىلىقىنى ھەل قىلىش بىلەن بىرلەشتۈرۈشى كېرەك. ئىستىلدا بۇرۇلۇش ياساشنىڭ چىقىش نۇقتىسى ۋە ئاخىرقى نۇقتىسى خەلق ئۈچۈن تېخىمۇ ياخشى مۇلازىمەت قىلىشتا. ئىستىلدا بۇرۇلۇش ياساش ئۈچۈنلا بۇرۇلۇش ياسىغاندا ئاممىنىڭ نارازىلىقىنى تېخىمۇ بەكرەك قوزغاپ قويىمىز. ئىستىلدا بۇرۇلۇش ياساشتا جايلار، تارماقلار چوڭقۇر بۇرۇلۇش ياسىشى، ئەمەلىي بۇرۇلۇش ياسىشى، ئەمەلىي ئۈنۈم ئارقىلىق خەلقنىڭ ئىشەنچىسىگە ئېرىشىشى كېرەك.

تۆنۇگۇن دەۋىر رېستۇرانى  ئالدىدىكى يول قاشالىنىپ  بىر نەچچە  چىشلىق ماشىنا يەر تاتىلاپ يۇرۇيدۇ ، بۇ ئەمدى دەۋىر مەيدانى ئالدىدىكى يولنى ياساۋاتامدۇ ، ياكى ، يەرنىڭ ئاستىدىكى پار ،سۇ توك يوللىرىدىن چىققان چاتاقنى ئوڭشىماقچىمۇ ئىشقىلىپ بىلگىلى بولمايدۇ ،

ناۋادا يولنى ئوڭىشىسا ئۇنداقتا ، قومۇل شەھەرلىك ھۆكۆمەتنىڭ  سۆرەمپالاس گەۋدىسى بۇ ئىش ئىنكاس قىلىنىپ ئەمىليەتتە مەندىن بۇرۇن ئىنكاس قىلغانلارمۇ نۇرغۇن ، كام بولغاندا بەش ئايدىن كىيىن ، رەسمى ھەركەتكە كەلگەن بولدى ،

بىز قومۇل شەھرىنىڭ پۇقرالىرى بولۇش سۇپۇتىمىز بىلەن شەھەرلىك ھۆكۆمەتنىڭ  ئۇيغۇر ئاممىسى كۆپ ماڭىدىغان بۇ يولغا ئاخىرى تۇتۇش قىلغانلىقىدىن سۆيۇندۇق ، ئانچە مۇنچە ئاغرىنىپمۇ قويدۇق ، تىزراق بولسا بولاتتى ئەسلى ،

مەيلى ئەمدىلىكتە بىز بۇ يولنىڭ سۇپەتلىك پۇتۇشىنى ھەم تىز پۇتۇشىنى كۆزىتەيلى ،
