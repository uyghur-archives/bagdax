بۇ ئۇقۇتقۇچى(ئايال لۇكچەك)نىڭ قىلمىشى ئىشتىن بۇشۇتۇش بەلگىلىمىسىگە ناھايىتى تۇلۇق جايىدا چۇشىدۇ؟؟؟؟؟؟؟؟؟؟؟؟؟؟؟؟؟؟؟؟؟؟؟؟؟..........؟؟؟؟
بۇ قايسى سوتچىنىڭ <<ئادىل كىسىمى>>،
مىنىڭچە ،بۇ ئاغزىدىن ئانا سۈتى كەتمىگەن <<سوتچى>>ئوخشايدۇ،ھازىرقى جەمىيەتتە %90 ئاتا ئانىلار باللىرىنى مەكتەپكە <<پۈتۈنلا>>تاپشۇرۇپ بېرىدۇ ،باللىرىنىڭ ئەخلاقىنىڭ قانداق بۇلۇشى،دەرىسلەرنى قانچىلىك ئۆگىنىپ ماڭغانلىغى بىلەن كارى بولمايدۇ ،مەكتەپ ئائىلە باشلىغى ئۈچۈن باللىرىنى ئەدەپ-ئەخلاق،قانۇن-تۈزۈم،سالامەتلىك ........جەھەتلەردە %100
    مەسئۈلىيەتنى ئۈز ۈستىگە ئالغان <<سۇغۇرتا شىركىتى>>.ئاتا-ئانىلار باللىرىنىڭ قورسىقى،كىيىم كېچىگىدىن باشقىسىغا مەسئۇل بولمايدىغان ،باللىرىدىن چاتاق  چىقسا ۋاقىراپ-جاقىراپ مەكتەپنى بېشىغا كىيىدىغان ،ئۇقۇتقۇچىنى ئۇرۇپ دۇمبالايدىغان (يىنىك بولغاندا )،يۇقىرى دەرىجىلىك ئورگانلارغا ئەھۋالنى كۆپتۈرۈپ ئەرىز قىلىپ ............
ئاتا-ئانىلار ھەم يۇقارقى سۆرەتنى تارتقان ئۇكاش،سىز ئۇقۇتقۇچىنى قانچىلىك چۈشۈنىسىز،؟
مۇنبەردە دەرىس بىرىۋاتقان ئۇقۇتقۇچىلارنىڭ %99 پىداگوكىكا،پىسخىلوگىيە ئۇقۇغان،قانۇننى بىلىدىغان ئادەملەر،ئەمما مەن ئۇقۇغۇچىلارنى ئۇرۇشنى ،تەن جازاسى بىرىشنى توغرا دەپ قارىمايمەن،ئەمما بىر قىسىم قەتئىلا  چىرايلىق گەپ ئاڭلىمىغان ئۇقۇغۇچىلارنى مۇۋاپىق دەرىجىدە قورقىتىپ قۇيۇشنى خاتا ئىش دەپ قارىمايمەن ،بولمىسا بىر قىسىم بالدۇر يىتىلىپ قالغان،چەتئەلنىڭ كىنولىرىنى دوراپ ئۆگىنىپ قالغان ئۇقۇغۇچىلارنى ئايال ئۇقۇتقۇچىلار قەتئىلا باشقۇرالمايدۇ ،بۇلۇپمۇ يېڭىدىن خىزمەتكە چىققان ئايال ئۇقۇتقۇچىلارنى ئۇقۇغۇچىلار بوزەك قىلىدۇ،
يەنە بىر گەپ ،بىر قىسىم ئۇقۇتقۇچىلار ئۆز كەسپىگە سادىق بولغانلىغى،ئۇقۇغۇچىلارغا تەلەپنى قاتتىق قويغانلىغى ئۈچۈن بالىلارنى ئازدۇر -كۆپتۇر تەنقىتلەپ قۇيغان بۇلۇشى مۇمكىن،ئۇقۇتقۇچىلارمۇ سىلەرگە ئوخشاش ئادەم ،باشقىلار بىلەن ياخشى ئۈتۈشنى خالايدۇ،ئەمما
باللىرىنى مەكتەپتىن ئىبارەت<< سۇغۇرتا شىركىتى>>گە ئۆتكۈزىپ ئېتىپ ،باللىرىغا ئادەم بۇلۇش يۇللىرىنى ئۆگىتەلمىگەن ئاتا-ئانىلارنىڭ <<كىيىنكى ئەۋلاتلىرى>>نى مۇئەللىم ئادەم بۇلۇش يۇلىغا سالمىسا بۇ ئۇقۇغۇچىلار قانداق بۇلۇپ كېتەر؟؟؟؟؟
ئىسىڭىزدە بولسۇن :بىزدە تېخى ئېشىپ-تېشىپ تۇرغىدەك ئۇقۇتقۇچىلار قۇشۇنى  يوق،ئۇنى ئاق،بۇنى قارا دەپ ئىشتىن بوشاتقىلى تەگسىڭىز ،بازاردىن ئۇقۇتقۇچى  تاپالمايسىز.
سىزگە ياقمىغان ئۇقۇتقۇچى بولسا ،سىز ئۇنىڭ بىلەن سىردىشىڭ،باللىرىڭىزنىڭ ئەھۋالىنى ئېگەللەڭ،ئۇنىڭغا تەكلىپ بىرىڭ.......

Re:بۇ ئوقۇتقۇچىنىڭ قىلمىشى  ئوقۇغۇچىلارغا تەن جازاسى بەرسە ئىشتىن ھەيدىلىدىغانغا چۈشەمدۇ؟

بۇ مەزمۇن 22قەۋەتتىكى taklimakan9216دە2012-11-09 22:30يوللىغان يازمىسىغا نەقىل :
مىنىڭ بۇ رەسىملەرنى يوللاشتا باشقا مەقسىدىم يوق، پەقەت تەربىيە قىلىدىغان ئورۇننى ، ۋاستىنى توغرا قوللىنىش كىرەك دىمەكچىمەن. مەنمۇ بىر ئوقۇتقۇچى مەنمۇ گەپ ئاڭلىمىغان ئوقۇغۇچىغا  قاتتىق قول بولغان ،ئەمما ئايرىم تەربىيە قىلغان ،ئۇنداق كۆپچىلىكنىڭ ئالدىدا بۇ  بۇ ئوقۇتقۇچىدەك  بالىلارنىڭ غورۇرىغا تەگمىگەن. بۇ رەسىمدىكى بالا سىلەر كۆرگەندەك تاياق يىسىمۇ كۈلۈپ نايناقشىپ تۇرىدۇ، بۇنىڭ سەۋەبى بەكلا ئاددى ،يەنى بۇ بالا بۇرۇندىن باشلاپ بۇنداق تىل ئاھانەت،تاياق يىيىشلەرگە كۈنۈپ كەتكەن ،بۇ ئوقۇتقۇچى مەخسۇس تەرىيە كالتىكى تەييارلىۋالغىنىغا قارىغاندا بۇ بىر ئىككى كۈن بولغان ئىش ئەمەستەك ،يەنە كىلىپ ئۇرۇش ئۇسۇللىرىمۇ  بۇ ئوقۇغۇچىلارغا تۇنۇشكەن ،بىرنىمە دەپ تولىغاندەك قىلغان بۇ بالا ئالقىنىنى چىقاردى ،ئوقۇتقۇچى ئورماقچى بولغانتى يەنە تىزلا قۇلىنى تارتىۋالدى. بۇنىڭدىن بۇ تاياقنىڭ بوش تەگمەيدىغىنىنى پەرەز قىلىش تەس ئەمەس. مەن دەيمەن  قاتتىق قوللۇقمۇ جايىدا قولىنىلسا ئاندىن ئۈنۈمى ياخشى بۇلىدۇ .ئۇستاز زالىم بولسا شاگىرت ئالىم بولىدۇ دىگەن گەپ  بۇ ئوقۇتقۇچىغا ئوخشاش سورۇن تاللىماي ،ۋاستە تاللىماي تەربىيە ئېلىپ بېرىڭ دىگەن گەپ بولمىسا كىرەك .  سىلەر ئابلېز ھەسەن مۇئەللىمنىڭ پەرزەنت تەربىيسى ھەققىدىكى لىكسىيەسىنى ئاڭلىغانمۇ؟ ئۇنىڭدا بالىلارغا ئائىلىدە ئاتا-ئانىنىڭ ،مەكتەپتە ئوقۇتقۇچىلارنىڭ قانداق تەربىيە بەرگنلىكىنى ھەم قانداق تەربىيە بىرىش كىرەكلىكىنى ناھايىتى ياخشى چۈشەندۈرگەن. بۇ  ۋىدىودىكى مەن تىلىفۇنۇم ئارقىلىق يۇشۇرۇنچە تارتىۋالغان نەق مەيدان، بۇ بالىنىڭ ئالدىدا يەنە بىر بالىنى ئالقىنىنى چىقارغۇزۇپ خېلى قاتتىق ئۇرىۋەتتى،  ئەمما ئۇنى سۈرەتكە ئېلىۋېلىشقا ئۈلگۈرمىدى .بىر دەمدىن كىيىن يان تەرەپتە تۇرغان بۇ بالا بىلەن بىر مىللەتىكى بىر نەچچە ئوقۇتقۇچى چىداپ تۇرالماي  بۇ ئىشقا ئارلاشتى، بالىلارغا بۇندىن كىيىن گەپ ئاڭلاشنى ئوقۇتقۇچىسىدىن كەچۈرۈم سوراشنى ئېيتىپ تۇراتتى ئەمما بۇئوقۇتقۇچى ئانچە قارشى ئالمىغاندەك پۇزۇتسىيەدە يەنە ئۆز ئۇسۇلى بىلەن تەربىيسىنى داۋاملاشتۇرغىلى تۇردى،مىنىڭ ئۇ يەردە تۇرۇۋىرىشكە ئامالىم بولمىغاچقا ھەم ئارلىشىشقا ئىمكان بولمىغاچقا ئىچىم ئاچچىق بۇلۇپ  كىتىپ قالدىم......چۈنكى تاياق يەۋاتقىنى مەن بىلەن ئوخشاش تىلدا گەپ قىلىدىغان بالا ،ئۇرىۋاتقىنى بولسا خەن مۇئەللىم...... كۈڭلۈمدە   تىل بىلەن ئېيتقۇسىز بىرخىل تۇيغۇ.    بۇ ۋەقەنىڭ ئورنى خوتەن ۋېلايىتى قاراقاش ناھىيىلىك 2-ئوتتۇرا مەكتەپ يەنى خەنزۇ مەكتەپ، بۇ ئوقۇتقۇچىمۇ خەنزۇ ئوقۇتقۇچى.  

سىز بۇ ئىشنى قاراقاشتىكى مائارىپ ئىدارىسغا مەلۇم قىلىڭ ، ئەگەر پەرۋا قىلمىسا خوتەندىكى مائارىپ ئىدارىسغا مەلۇم قىلىڭ ، قولىڭىزدا ئىسپات بار ئەمەسمۇ ، ئەگەر قومۇلدا شۇنداق بولغان بولسا ، سىزنىڭكىدەك ئىسپات بولسا مەن مەلۇم قىلغان بولاتتىم ،
