تاڭسىڭنىڭ سالام خېتى

(فىليەتۇن)
سېغىنىشلىق سالام خەت،
سېغىنغاچقا يازغان خەت،
سېغىنمىسام يازمايتىم،
بەك سېغىنغاچ يازغان خەت.
    سۆيۈملۈك،قەدىرلىك،ۋاپادار،ئەخلاقلىق،باتۇر،سەمىمىي،راستچىل،كەمتەر،بەزىدە ئازراق ياغاچ قۇلاق چوڭ شاگىرتىم ۋۇكۇڭ،ياخشى تۇردۇڭمۇ؟ ئۈرۈمچىدە ئىشلىرىڭ بەلەن، ئوبدان، ئوڭۇشلۇق، يۈرۈشلۈك، كۆڭلۈڭدىكىدەك كىتىپ بارامدۇ؟ مەن بۇدخانىدا كۈندە كۈجە كۆيدۈرۈپ ئاۋالىقسوارادىن سېنىڭ بەخت-سائادىتىڭنى نىسىگە سوراپ ساڭا ئامەت تېلەپ تۇرىۋاتىمەن،ئىي.....ئامىتابا..!
  مېنى ھازىر كىشىلەر، قېرىپ بەك كوت-كوت،قانداق گەپ بولسا ئىخچام قىلماي لىڭتاسمىلىق قىلىپ سوزۇپ سۆزلەيدۇ دىيىشىدۇ،مىنى پەقەت باجېدىن قالسا سەنلا چۈشىنىسەن شۇنداققۇ مايمۇنۇم، مېنىڭچە بۇ كىشىلەر ئۆزى ئالجىپتۇ. ئەمدى مەقسەتكە كەلسەك، مەن بۇ خەتنى ئالدىرماي، تەمتىرىمەي، ئولتۇرۇپ تۇرۇپ تېگۇئەنيىن چېيىنى ئېچكەچ مىسىلداپ يېزىۋاتىمەن، چۈنكى سەن خەتنى ئىتتىك ئوقۇيالمايسەن، ھە راست، بىز ئۆي كۆچتۇق، لېكىن ئادرىسى ئۆزگەرمىدى، چۈنكى ئۆي كۆچكەندە نومۇر تاختىسىنىمۇ بىللە ئالغاچ كەلدۇق، بىزنىڭ بۇ يەردە ئۆتكەن ھەپتىدە ئىككى قېتىم يامغۇر ياغدى، بىرىنچى قېتىم ئۈچ كۈن، ئىككىنچى قېتىم تۆت كۈن ياغدى، تۈنۈگۈن باجىې بىلەن بۇتخانا ئالدىدىكى كەچلىك بازاردا گۆشنان يىدۇق، خوجايىن گۆشناننى سەككىز پارچە قىلايمۇ ياكى ئون ئىككى پارچە قىلايمۇ دېگەن ئىدى، مەن سەككىز پارچە قىلىڭ، ئون ئىككى پارچە قىلسىڭىز بەك جىق، بىر چوشقا بىلەن يەپ بولالمايمەن دىدىم. توغرا، ئالدىنقى ھەپتە ساڭا بىر پەلتۇ ئەۋەتتىم، پوچتىدا ئېغىرلىقى ئېشىپ كەتمىسۇن دەپ، تۈگمىسىنى كىسىپ شۇ پەلتۇنىڭ يانچۇقىغا سېلىپ قويدۇم، باجىېنىڭ كىچىك خوتۇنى تۇغدى، لېكىن ئوغۇل قىزلىغىنى بىلەلمىگەچكە، سېنىڭ تاغا ياكى يەڭگە بولىشىڭنى ئۇقالمايۋاتىمىز. راس...ئۇرۈمچىدە ۋاقتىڭ بولۇپ قالسا كىچىك بالىلارغا بۇرۇنقى، بەك بۇرۇنقى زاماندا... دېگەن چۆچەكلەرنى ئېيتىپ بەرگىن، ئۇلارغا شۇ ۋاقىتلاردا ئاسماننىڭ كۆپ-كۆك، سۇلارنىڭ سۈپسۈزۈك ئىكەنلىكىنى، توخۇنىڭ تۇخۇمنى ئۆزى تۇغىدىغانلىقىنى، قوي-كالا گۆشلىرىنى خاتىرجەم يىگىلى بولىدىغانلىغىنى، چاشقاننىڭ مۈشۈكتىن قورقىدىغانلىقىنى، سوت ئورۇنلىرىنىڭ ناھايىتى ئادىللىقىنى، تاماكا-ھاراقلارنى ئەرلەرنىڭلا چىكىپ-ئىچىدىغانلىقىنى، ئەرلەرنىڭ ئاياللارنى قوغدايدىغانلىقىنى، ئەر-ئاياللارنىڭ پەرقى بولىدىغانلىقىنى، توي قىلىش ئۈچۈن باشتا پەقەت مۇھەببەتلىشىشلا بولۇپ، سىنىشىپ يۈرىيدىغان ئىشنىڭ يوقلىقىنى، ساتىراچخانىلارنىڭ چاچلا ياسايدىغانلىقىنى، دورىلارنىڭ ئاغرىغان يەرنى ساقايتىدىغانلىقىنى، دوختۇرلارنىڭ ئادەملەرنى قۇتۇلدۇرىدىغانلىقىنى، كىنو ئىشلەش ئۈچۈن باشتا ئارتىستلارنى تەربىيلەندۈرىدىغانلىقىنى، رەسىمگە كېيىم كىيىپ چۈشىدىغانلىقىنى، بالىنىڭ دادىسىنىڭ ئىنىقلىقىنى، مەكتەپلەرنىڭ قالايمىقان ھەق ئالمايدىغانلىقىنى، لەقۋانىڭ دوكتور ياكى باشلىق بولالمايدىغانلىقىنى، قۇيرۇق ئېسىپ ئۆپكە ساتمايدىغانلىقىنى، تويدىن كېيىن ئاشنا تۇتسا بولمايدىغانلىقىنى، بىر ئادەمنىڭ بىرلا خوتۇنى بولىدىغانلىقىنى، مۇنازىرىلەرنىڭ ناھايىتى ئەھمىيەتلىك بولىدىغانلىقىنى، كىشىلەردە مۇھەببەت-نەپرەت ۋە ھەق ناھەق كۆز قارىشى دەيدىغان بىر خىل گۈزەل تۇيغۇ بولىدىغانلىقىنى، قىل بىلەن پىلنىڭ پەرقى بارلىقىنى، نەرسە سېتىۋالسا پۇل تۆلەيدىغانلىقىنى،ئىشەكنىڭ ئىشلىتىش ئورنى بېكخامدىن يۇقۇرلىقىنى، ئىنكاسنى يازغۇچىغا ئەمەس،تېمىغا قارىتا يازىدىغانلىقىنى.....سۆزلەپ بەرگىن..
    مايمۇنۇم، ئايالىڭ مايمۇنەم بىلەن ئۆينى ياخشى تۇتقىن، ھەرقانداق قېيىنچىلىق بولسا مېنى ئىزدە، باجىي بىلەن ئۈرۈمچىدىكى ئۆيۈڭگە بېرىپ بىرنەچچە ئاي ئارام ئالغاچ ساڭا ياردەم بېرىمىز، شۇڭا خاتىرجەم بولغىن. بەك كەچ بولۇپ كېتىپتۇ،ئەسلى بىۋاستە تېلىفون ئۇراي دېگەن، بىلىسەن مىجەزىمنى، بەك سپايە سۆزلىگەچكىمۇ، سەل-پەل ئۇزاق كوتۇلداپ قويىمەن، پىنسىيە پۇلۇم تېلىفون ھەققىغە يەتمىگۈدەك. جاۋاپ خەت يازمىساڭمۇ بولىدۇ، مېنىڭكى كىرسە ھەق ئالمايدىغان مەشرەپ كارتىسى، شۇڭا بىراقلا تېلىفون قىلغىن. ئىچىڭ پۇشسا مۇشۇ مۇنبەرگە كىرىپ، ماڭا ئاتاپ ئازتولا كۈجە كۆيدۈرۈپ قويغىن. مېنىڭ كارتامغا پۇل سالمىساڭمۇ بولىدۇ، كارتا نومۇرۇم(856-658-586-658)،خەيىر ئاۋالىقسوراغا ئامانەت.
ھۆرمەت بىلەن تاڭ دىيارىدىكى ئۇستازىڭ تاڭسىڭدىن.
ئېنتىرنەت يىلى-مانچىنچى ئاينىڭ مانچىنچى كۈنى.

باجىېنىڭ تاڭسىڭغا خېتى
(فىليەتۇن)
ئالپامىش
   قەدىرلىك،سۈيۈملۈك،مىھرىبان ئۇستازىم تاڭسىڭ،تەنتەك،جايىل،ئاتكاچى چوڭ ئاغام ۋۇكۇڭغا،يۇمۇلاق تاۋۇز،مەيدانى ئوچۇق ئەمەس،ئېغىر بىسىق،سەھرالىق ئىنىم شاسىڭغا پېقىر باجىې دىن ئوتلۇق سالام؛
  ئۇستاز،ئاغا،ئىنىلەر..ئوبدان تۇردۇڭلارمۇ؟راسنى دىسەم يازغان خىتىڭلارنى كۆرۈپ،سىلەرنى ھەقىقەتەن سېغىنغانلىغىمنى ھىس قىلىۋاتىمەن،گەرچە كۈنلەر مۇشەقەت بولسىمۇ شۇ بىرگە بولغان كۈنلىرىمىزنى ئويلىسام كۆزلىرىمگە ئىختىيارسىز ياش كىلىدۇ،ھىلىمۇ بۇزۇرۇكۋارىمىزغا شۈكرى،شۇ كۈنلەرنىڭ لەززىتىنى مانا ئەمدى كۆرىۋاتىمەن،بۇنىڭ ئۈچۈن ئۇستازىم تاڭسىڭغا كۆپ تەشەككۈر.
  ئۇستاز،مەن ھازىر شىنجاڭدىكى مەلۇم بىر شەھەردە مۇئاۋىن شەھەر باشلىقى.يېقىندا مىنى يەنە چوڭ شەھەرگە يۆتكىمەكچى،كۈنلىرىم خېلى يامان ئەمەس ئۆتىۋاتىدۇ،شەھەرنىڭ سىرتىدا شەخسى ئۈچ قەۋەتلىك داچام بار،قايەرگە بارماي ئاپىرىپ ئەكىلىدىغان پىكاۋىم بار،خوتۇن-بالا دەمسىز؟سەككىز بالام بار،ئەڭ كىچىكى قىشىمدا،قالغىنى ئامىرىكا دىگەن شەھەردىكى چوڭ مەكتەپلەردە،ئۈچ خوتۇن ۋە تاڭ دۆلىتىگە يېغىنغا بارغاندا تونۇشۇپ قالغان توققۇز  ئۇسۇلچى ئاشنام بار.
  ئۇستاز،سىزنىڭ سايىڭىزدە غەرىپكە نوم ئالغىلى بارغاننىڭ پايدىسىنى مانا ئەمدى كۆرىۋاتىمەن،غەرىپتىن قايتىپ كىلىپ مىنى خىزمەتكە بۆلگەندە كىيىنكى كۈنلىرىمنى ئويلاپ كۆزۈمگە ياش ئالغانتىم،لىكىن چوشقا بولۇپ قالغانلىغىمدىنمۇ تەلىيىم كەلگىلى تۇردى،يىلدىن يىلغا مەرتىۋەم ئۆستى،بۇ يەردە بىلىمنىڭ ئاز-كۆپلىكىگە ئەمەس،قورساقنىڭ چوڭ-كىچىكلىگە قاراپ مەنسەپ بىرىدىكەن،بۇنداق بولىشىنى سىزمۇ ئويلاپ باقمىغانسىز ھە؟ئۆز ۋاقتىدا بۇزۇرۇكۋارىمىز ماڭا'سپپايە'ئۇنۋانىنى بەرگەندە نارازى بولغانىدىم،لىكىن شۇ سپپايەم بىلەن باشلىقلارنىڭ پوردىقىنى سىپايىلىق بىلەن ئاۋايلاپ كۆتۈرۈپ،مانا شەھەر باشلىقى بولۇپ قالدىم دىسە..بۇزۇرۇكۋارىمىز ھەقىقەتەن ئوقۇغان ئادەمكەن.بۇ شەھەرگە يېڭى كەلگىنىمدە تەلىيىمگە ئىتلار چاشقان تۇتۇپ يەيدىغان بولۇپ قاپتۇ،چۈنكى مۈشۈكلەر چىرىكلىشىپ كەتكەن گەپ،مۇشۇ پۇرسەتتىن ئۈنۈملۈك پايدىلىنىپ ئاممىنىڭ ياخشپ چوشقىسى بولۇپ قالدىم،ئۇنۇڭ ئۈستىگە سىز بىلەن نوم ئەكەلگىنىم،مىنىڭ چوشقا جەمەتىدىن بولغىنىم مەرتىۋەمنىڭ كەينى-كەينىدىن ئۆسۈشىگە تۈرتكە بولدى.
   ئۇستاز...ئۆزۋاقتەدا سىزمىنى بەكلا ئاچكۆز،ھورۇن،بۇزۇق دەپ تىللايتىڭىز،ئەنە قاراپ تۇرۇپسىز،چوڭ ئاغام باتۇر،ئەخلاقلىق،راسىتچىل ئىدى،لىكىن ھازىر كۆرگەن كۈنىچۇ؟ھەرنىمە بولسا،كۆڭلىڭىز يېرىم ۋاقىتلاردا شاسىڭ ئىنىمنىڭلا سۆزىنى تىڭشايتىڭىز،ئۇنى سوغۇق قان،سەۋىرچان دەپ ماختايتىڭىز،ئۇنىڭ كۈنى چوڭ ئاغامنىڭكىدىنمۇ بەتەر تۇرمامدۇ،ۋاقىت ھەممىنى بەلگىلەيدىكەن ئۇستاز،ھەممىدىن كۆرۈمسىز،ئاچكۆز،ئاسمانغا مەڭگۈ بىشىنى كۆتۈرۈپ قارىيالمايدىغان چوشقا بولغان مەن،مانا ھازىر جاھاننىڭ پەيزىنى سۈرىۋاتىمەن،مىنى دۆت دەيتىڭىز،مانا مەندىنمۇ دۆت بولغان ئاممىغا باشلىق بولۇپ يۈرىپتىمەن ئۇستاز.
   ئۆزۋاقتىدا سىزنىڭ غىمىڭىزنى يەپ،سەكسەن بىرمىڭ مۇشەقەتنى باشتىن ئۆتكۈزگەن بولسام ھازىر بۇ شەھەردە خەلىقنىڭ غىمىنى داچامدا يىتىپ-قوپۇپ،كۈندە ئەيشى ئىشارەت بىلەن يەۋاتىمەن،توغراق قىلىپ ئېيىتقاندا غېمىنى ئەمەس،نېنىنى يەپ قېنىنى ئىچىۋاتىمەن،كىم ئۇلارنى غەرپىكە بېرىپ نوم ئەكەلمىسۇن دەپتىكەن،ھەر نىمە بولسا مەن دىگەن تاڭ دۆلىتىگە چوڭ تۆھپىلەرنى قوشقان چوشقا تۇرسام،يەتتە ياشتىن يەتمىش ياشقىچە ھەممىسى مۇبارەك نامىمنى بىلىدىغان تۇرسا،ھەتتا تاڭ دۆلىتىنىڭ مۇخبىرلىرى پات-پات كىلىپ زىيارەت قىلىپ كىتىدۇ دىسە.بىزنىڭ بۇيەردە'ھەممە ئادەمنىڭ بالىلىرى مەكتەپتە ئوقۇش كېرەك''ھەممە گۈللەر بەس-بەستە ئىچىلسۇن''ھەممە ئېقىمدىكىلەر بەس-بەستە يايرىسۇن'دىگەندەك مۇھىتتىن قىلچە ئەسەر بولمىسىمۇ،لىكىن 'ئىسلاھات ئىچىۋىتىش' كەڭ جارى بولىۋاتقان ياخشى كۈنلەر بولغاچقا،مىنىڭ بۇ راھەت-پاراغىتىم ئەيىبلىك بۇلۇش ئۇياقتا تۇرسۇن،رەھبەرلەرنىڭ ئوبدانلا مەدەتلىرىگە سازاۋەر بولىۋاتىدۇ،شۇڭا مەن مۇشۇ ئۇسۇلدا داۋاملىق تېرىشىپ،چوشقىلىغىمنى ياخشى جارى قىلدۇرۇپ،قولۇم بىلەن كۆتۈرگەن پورداقنى دۈمبەمدە يۈدۈپ كۈتۈرۈپ(ئۆز ۋاقتىدا كېلىن يۆتكىگەندەك)بۇزۇرۇكۋارىمىز بۇيرىسا تاڭ دۆلىتىگە باشلىق بولىمەن.شۇ ۋاقىت كەلسۇن قەنى ئۇستاز،ۋۇكۇڭ ئاغام بىلەن شاسىڭ ئىنىمنى قىشىمغا ئەكىلىۋالىمەن،چوڭ ئاغامغا ئۆزەم يازغان 'پورداق كۆتۈرۈشنىڭ يۈز تەدبىرى'دىگەن كىتابىدىن بىرنى مۇشۇ خەت بىلەن ئەۋەتىۋىتەي،ۋاقتى بولسا كۆرۈپ قويسۇن،شاسىڭ ئىنىمدىن غەم قىلمايمەن،ئۇ بۇرۇنلا ئەپكەش كۆتۈرۈپ كۆنۈپ كەتكەن،ئۇنۇڭ ئۈچۈن پورداق كۆتۈرۈش چوڭ ئىش ئەمەس،سىز بولسىڭىز پېنسىيەگە چىققاندىكىن ئاز تولا چوڭلار ئىشلىتىدىغان ئاتمار دورىلىرىنىڭ ئېلانىنى ئىشلەتكۈزۈپ بىر كىچىدىلا ئابرويىڭىزنى كۆتۈرۋىتىمەن ئۇستاز.
    خەتنى كاتىپىمدىن يازغۇزغاچقا كۆڭلۈمدىكى نۇرغۇن سۆزلەرنى دىيەلمىدىم ئۇستاز(چۈنكى مەن خەت تونىمايمەن،بۇنى بىلىسىزغۇ ھە)يېقىندا ئورۇنلاشتۇرغان 'چوڭ رەھبەرلەرنىڭ ئۈچ ئايلىق كۇرسى'دا بىزگە خەت ئوقۇش-يېزىشنى ئۈگەتكۈدەكمىش،شۇ چاغدا ئۆز قۇلۇم بىلەن سالام خەت يازارمەن،كەچتە ساپا مۇئارىپى ۋە ئۇقۇشسىز قالغان بالىلارنىڭ ئىقتىسات مەسلىسىنى مۇزاكىرە قىلىش توغرۇلۇق چوڭ بىر زىياپەت بولماقچى ئىدى،كىچىكىپ قالماي،شۇڭا خەتنى مۇشۇ يەرگىچە يازاي،خەيىر،بۇزۇرۇكۋارىمغا ئامانەت.
كامالپ ئېھتىرام بىلەن پەخىرلىك چوشقا جەمەتىدىن جۇباجې.

تاڭسىڭنىڭ لايىق ئىزدىشى

جۇباجىي خوتۇنىنى ئېلىپ ئۇستازىنى كۆرگىلى كەپتۇ.تاڭسىڭ ئۇلارغا ھەۋەس قىلدىمۇ ياكى ھەسەت قىلدىمۇ توي قىلىشنى نىيەت قىلىپ، لايىق تونۇشتۇرۇش ئورنىغا كەپتۇ. بۇ يەردە بىر قېرى مۇماي باركەن.
موماي: ئىسمىڭىز؟
تاڭسىڭ: تاڭسىڭ، سەنزاڭ، راھىپچاق دىسىڭىزمۇ بولىدۇ ......
موماي: زادى نىمە؟
تاڭسىڭ: TOM
موماي: يېشىڭىز؟
تاڭسىڭ: چوڭ ئەمەس.
موماي: قاچان تۇغۇلغان؟
تاڭسىڭ: تاڭ سۇلالىسىدا.

موماي: كەسپىڭىز؟
تاڭسىڭ: قانۇنلۇق بۇلاڭچى.
موماي: بۇ نىمە گەپ؟
تاڭسىڭ: تىلەمچى دىمەكچىمەن.
موماي: ئۆيىڭىز بارمۇ؟
تاڭسىڭ: يوق.
موماي: نەدە تۇرىسىز؟
تاڭسىڭ: جوكۇلاڭگا ئېبادەتخانىسى 

موماي: قاتناش قورالىڭىز بارمۇ؟
تاڭسىڭ: بار.
موماي: قانداق قورال؟
تاڭسىڭ:بىر قوتۇر ئاق ئات.

موماي: بۇرۇن قىز دوستىڭىز بولغانمۇ؟
تاڭسىڭ: شۇنداق.
موماي: ئۇلار نىمە قىلاتتى.
تاڭسىڭ: پاسىق سۆڭەك جىن، ئۆمۈچۈك جىن،تۈلكە جىن..
موماي: خېلى ئويناپتىكەنسىز ھە؟
تاڭسىڭ: ئادەتتىكى ئىشلار بۇ، مۇھەببەتتە چىقىشالماي قالدۇق.
موماي: نىمىشقا ئايرىلىپ كەتكەن؟
تاڭسىڭ: ھەممىسى گۆشۈمنى يىمەكچى بولۇشقان.

موماي: لايىق تاللاشتىكى تەلىپىڭىز نىمە؟
تاڭسىڭ: مىنى يىمەيدىغان بولسۇن.
موماي: ئەمىسە راھىبە بولسا قانداق؟
تاڭسىڭ: ياق، ساۋاقداشلار توي قىلسا بالىسى غەلىتە تۇغۇلۇپ قالارمىش.
موماي: ئەمىسە ئاۋالوكسىۋاراچۇ؟
تاڭسىڭ: بولماسمىكىن، ئۇ مىنىڭ رەھبىرىم، ئۇنىڭ ئۈستىگە ئىشخانىدا مۇھەببەتلىشىشنى خالىمايمەن.
موماي: بۇزرۇك ئانا (يۇرۇڭ سۇلتاننىڭ ئايالىنى دىمەكچى) قانداق بولار؟!
تاڭسىڭ: بۇمۇ بولمايدۇ، ئىرى بەك چىڭ تۇتىدۇ، ئۇنىڭ ئۈستىگە مەن پىشقان ئاياللارنى ياقتۇرمايمەن.
موماي: ئەمىسە مەنچۇ؟!
تاڭسىڭ:ئۇف بولدىلا،يەنىلا راھىبە بولسۇن......
بصر كۈنصى تاڭسىڭ باشلىق 4 ئەزا ئايروپىلانغا ئولتۇرۇپ سەپەرگە چىقتى، يولدا ئايروپىلاندىن چاتاق چىقتى، ئەمما ئايروپىلاندا پەقەت 3دانە پاراشوت بار ئىدى.
تاڭسىڭ ئېيىتتى،مەن سۇئال سوراي، كۆپچىلىك جاۋاب بېرىڭلار، جاۋاب بېرەلمىگەن ئادەم يەرگە سەكرەپ چۈشىدۇ .
ةتاڭسىڭ سۇئال سوراشقا باشلىدى، ۋۇكۇڭ دەپ باقە، ئاسماندا نەچچە قوياش بار ؟

ئۇلارغا شۇ ۋاقىتلاردا ئاسماننىڭ كۆپ-كۆك، سۇلارنىڭ سۈپسۈزۈك ئىكەنلىكىنى، توخۇنىڭ تۇخۇمنى ئۆزى تۇغىدىغانلىقىنى، قوي-كالا گۆشلىرىنى خاتىرجەم يىگىلى بولىدىغانلىغىنى، چاشقاننىڭ مۈشۈكتىن قورقىدىغانلىقىنى، سوت ئورۇنلىرىنىڭ ناھايىتى ئادىللىقىنى، تاماكا-ھاراقلارنى ئەرلەرنىڭلا چىكىپ-ئىچىدىغانلىقىنى، ئەرلەرنىڭ ئاياللارنى قوغدايدىغانلىقىنى، ئەر-ئاياللارنىڭ پەرقى بولىدىغانلىقىنى، توي قىلىش ئۈچۈن باشتا پەقەت مۇھەببەتلىشىشلا بولۇپ، سىنىشىپ يۈرىيدىغان ئىشنىڭ يوقلىقىنى، ساتىراچخانىلارنىڭ چاچلا ياسايدىغانلىقىنى، دورىلارنىڭ ئاغرىغان يەرنى ساقايتىدىغانلىقىنى، دوختۇرلارنىڭ ئادەملەرنى قۇتۇلدۇرىدىغانلىقىنى، كىنو ئىشلەش ئۈچۈن باشتا ئارتىستلارنى تەربىيلەندۈرىدىغانلىقىنى، رەسىمگە كېيىم كىيىپ چۈشىدىغانلىقىنى، بالىنىڭ دادىسىنىڭ ئىنىقلىقىنى، مەكتەپلەرنىڭ قالايمىقان ھەق ئالمايدىغانلىقىنى، لەقۋانىڭ دوكتور ياكى باشلىق بولالمايدىغانلىقىنى، قۇيرۇق ئېسىپ ئۆپكە ساتمايدىغانلىقىنى، تويدىن كېيىن ئاشنا تۇتسا بولمايدىغانلىقىنى، بىر ئادەمنىڭ بىرلا خوتۇنى بولىدىغانلىقىنى، مۇنازىرىلەرنىڭ ناھايىتى ئەھمىيەتلىك بولىدىغانلىقىنى، كىشىلەردە مۇھەببەت-نەپرەت ۋە ھەق ناھەق كۆز قارىشى دەيدىغان بىر خىل گۈزەل تۇيغۇ بولىدىغانلىقىنى، قىل بىلەن پىلنىڭ پەرقى بارلىقىنى، نەرسە سېتىۋالسا پۇل تۆلەيدىغانلىقىنى،ئىشەكنىڭ ئىشلىتىش ئورنى بېكخامدىن يۇقۇرلىقىنى، ئىنكاسنى يازغۇچىغا ئەمەس،تېمىغا قارىتا يازىدىغانلىقىنى.....سۆزلەپ بەرگىن..

ماۋۇ يېرى پەچەتكەن.
