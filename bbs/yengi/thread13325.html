ئانىلار ئۇلۇغدۇر،دىنىمىزدا ئاللاھ تائالنىڭ ھەققىدىن قالسا ئاتا ئانىنىڭ ھەققى كېلىدۇ،ئاتا ئانىمىزنىڭ قەدرىنى قىلايلى،ئۇلار بىزگە ئۆمۈرلۈك باقىۋەندە ئەمەس، بىركۈنى كىتىدۇ يىغلاپ قالىمىز،ئامالىمىز بولمايدۇ،ئەمما بۇ ئاچچىق ھەقىقەتنى قۇبۇل قىلماي ئامال بولمايدۇ،شۇڭا قېرىنداشلار ئاتا ئانىمىزنىڭ قەدرىنى قىلايلى!
 

كەتكەندە ئاناڭ بىر كۈنى
(مۇسئەب)
كەتكەندە ئاناڭ بىر كۈنى
يىغلايسەن ئەلەم بىلەن شۇ كۈنى
ئاھ ئۇرىسەن دادلايسەن ئامالسىز
لېكىن ئاناڭ جىمجىت چىقماس ئۈنى
شۇندا بىلىسەن خاتالىقىڭنى
تەقدىرنىڭ ئاچچىق سىنىقىنى
كاللاڭنى سىلكىۋەت ئويلانغىنسەن
بىشىڭغا كىلىشتىن بۇرۇن شۇنداق بىر كۈنى
******
ئاناڭ كۈتەر سېنى كىچە كېچىلەپ
بالام بالام كەلسە باغرىمغا بېسىۋالسام دەپ
قىززىقىدا تامىقىنى يېۋالسىكىن دەپ
سەن ۋاپاسىزنى ئويلاپ غېمىڭنىلا يەپ
ئاغرىپمۇ قالدى بالام بالام دەپ
سەن ئويلايسەن ئانام تېخى ياش دەپ
كەتكەندە ئاناڭ بىر كۈنى
يىغلايسەن ئەلەم بىلەن شۇ كۈنى
******
بىچارە ئاناڭ ئولتۇرىدۇ بولۇڭدا
كۆزلىرى دائىما سېنىڭ يولۇڭدا
غېمىڭنىلا يەيدۇ چۈشىدە ياكى ئوڭىدا
لېكىن سەن ئەركىن ئوينايسەن تويدا
خۇددى ھېچكىمىڭ يوق بۇ دۇنيادا
كەتكەندە ئاناڭ بىر كۈنى
يىغلايسەن ئەلەم بىلەن شۇ كۈنى
*****
تويلار تويلار ئۈزۈلمەس تويلار
ئەگىشىپ ئۇنىڭغا رەڭگارەڭ چايلار
ياغار ئۆيۈڭگە يامغۇردەك باغاقلار
غېرىپ كۈلبەڭدە يوقتۇر كىتابلار
كىتابلاردىن ئويغىنىدۇ ۋىجدانلىق روھلار
سېنى تېخىمۇ پەپىلەر ئېسىل باغاقلار
تۈگىمەس باغاقلىرىڭدىن بىزار ئانىلار
كەتكەندە ئاناڭ بىر كۈنى
يىغلايسەن ئەلەم بىلەن شۇ كۈنى
******
سەن تۇغۇلغاندا مەجبۇر ئىدىڭ
ھېچ ئىشىڭنى قىلالماس ئاجىز ئىدىڭ
چىچالمىساڭ ئىنجىقلاپ يىغلار ئىدىڭ
يىغاڭ بىلەن ئاناڭنى ئۇخلاتماس ئىدىڭ
ئەمدى ئەھۋال ئۆزگەردى چۈنكى چوڭ بولدۇڭ
ئانا دېگەن ئۇلۇغ زات قەدرىنى بىلمىدىڭ
كەتكەندە ئاناڭ بىر كۈنى
يىغلايسەن ئەلەم بىلەن شۇ كۈنى
*****
ياشىدىڭ دۇنيادا دوستلىرىڭ ئۈچۈن
قايغۇرسا قايغۇرۇپ ئازابى ئۈچۈن
سەن بەك ئالدىراش بىچارە ئاناڭ ئۈچۈن
ئويلۇنۇپ باقمىدىڭ ئاناڭنىڭ ھېسياتى ئۈچۈن
سۇئال قوي نېمە قىلدىم مېھرىبانىم ئۈچۈن؟
كەتكەندە ئاناڭ بىر كۈنى
يىغلايسەن ئەلەم بىلەن شۇ كۈنى
*****
ئانىلار يىغلىسا يىغلايدۇ ئالەم
بىر قولىدا بۈشۈك بىرىدە ئالەم
نۇر چاچىدۇ قاراڭغۇ دىللارغا
ئاجىز بۇۋاقلارنى قىلىىپ چىن ئادەم
قۇربانلىق بېرىدۇ بولۇپ ئۇ بەردەم
كەتكەندە ئاناڭ بىر كۈنى
يىغلايسەن ئەلەم بىلەن شۇ كۈنى
*****
سۆزلىرىم جىق ئىدى ئاجىز كەلدى قەلىمىم
ئىچىم كۈچلۈك ئوخشايدۇ يۇتىۋەتتىم ئەلىمىم
ئاخىرىدا ساڭا باردۇر بىر سەمىمى تەكلىپىم
ئويلۇنۇپ باققايسەن توغرا كۆرسەڭ تەكلىپىم
ئاناڭنىڭ ھېسياتى بىلەن ئويناشما ھېچقاچان
ئاناڭ قارغىمايدۇ ئەمما كۆز ياشلىرى يامان
بىلمەي قالىسەن تۇتقاندا يۈرىكىڭنى قىلىپ قان
ئانىسىنى قاخشاتقان بولغان ئەمەس ھەرگىز ئامان
بارىدا خىزمىتىنى قىل دۇئاسىنى ئال بولغىن سەن ئامان
كەتكەندە ئاناڭ بىر كۈنى
يىغلايسەن ئەلەم بىلەن شۇ كۈنى
