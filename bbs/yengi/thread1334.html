           ئوقۇتقۇچىنىڭ جەمىيەتتىكى ئورنىنى تىلغا ئالساق بەزى كىشىلىرىمىز ئاغزىنى بۇزۇپ تىللايدىغان بولدى، ھەممەيلەن دەيدۇ، ۋاي ھازىرقى ئوقۇتقۇچىلاردا ساپا يوق،ئۇنداغەي بۇنداغەي. نېمىشقا ئوقۇغۇچىلاردىن ئۆز پەرزەنتلىرىمىزدىن سەۋەب ئىزدەپ باقمايمىز؟
    مەن بۇيەردە مۇئەللىملەر توغرىلىق بىر نەرسە دېيىشىم مېنىڭ ئوقۇتقۇچى بولغىنىمدىن ئەمەس بەلكى ئۇيغۇر مائارىپىغا كۆڭۈل بۆلۈشۈمدىن بولدى.
پەرزەنتىڭىزگە قاراپ بېقىڭ ،ئاندىن بالىڭىز بىلەن بىر سىنىپتىكى  دەرستە ياخشى باشقىلارنىڭ پەرزەنتىگە قاراپ بېقىڭ ، نېمە ئۈچۈن باشقىلارنىڭ بالىسى دەرستە ياخشى ئىمتىھاندا يۇقىرى نومۇر ئالدى؟لېكىن سىزنىڭ بالىڭىز ئىمتىھاندىن ئۆتەلمىدى؟
          ئوخشاش بىر سىنىپتىكى بىر ئوقۇتقۇچىنىڭ ئىككى ئوقۇغۇچىسى نېمە ئۈچۈن ئوخشىمىغان نەتىجىگە ئېرىشتى؟ گۇناھ ئوقۇتقۇچىدىمۇ؟  مېنىڭچە گۇناھ بالىڭىزدا ، ھازىر ئوچۇق ئاشكارا دەۋر ،ئوقۇتقۇچى قەستەن قالدۇرۇۋەتتى دېگەن تەقدىردىمۇ بۇ مۇمكىن ئىش ئەمەس، ھازىر ئوقۇتقۇچىلارنىڭ ئۇنچىلىك چوڭ ھوقۇقى يوق. ئىمتىھاندا يۇقىرى نومۇرئالغان ئوقۇغۇچى دەرستىىن سرت ئائىلە ئوقۇتقۇچىسى تەكلىپ قىلغان بولۇشى مۇمكىن،ۋە ياكى ئۇ تىرىشچان، ئەگەر بالىڭىزدا تىرىشىش ،ئالغا ئىلگىرلەش روھى رىقابەت ئېڭى بولمايدىكەن ئۇنى ئېينىشتىيىن،نوبېل، ئېدىسوندەك ئالىملار ئائىلە ئوقۇتقۇچىسى بولۇپ يېتەكلىسىمۇ ئۇ يەنىلا ئىمتىھاندىن ئۆتەلمەيدۇ.
        ھازىر ئوقۇغۇچىلارنىڭ يۈكى ھەقىقەتەن ئېغىرلاپ كەتتى، ئوقۇتقۇچىلارنىڭ تېخىمۇ شۇنداق، لېكىن بىر قىسىم ئاتا ئانىلار بالىسنىڭ نەتىجىسى ناچار بولسا ئوقۇتقۇچىدىن كۆرىدىغان بولدى، بۇ ئوقۇتقۇچىلارغا بولغان ئادالەتسىزلىك.  بەزى ئوقۇچىلارنىڭ نەتىجىسىنىڭ ناچارلىقى ئۇنىڭ ئۈگىنىشكە قىزىقماسلىقى ۋە ئويۇنغا بېرىلىپ كېتىشى ۋەياكى باشقا سىرتقى ئامىللارغا مۇناسىۋەتلىك. شۇنىڭ ئۈچۈن بىز پەرزەنت تەربىيىسدە مەكتەپ ،ئائىلە،جەمىيەت ئورتاق بولغان ئۇنۋېرسال كۆڭۈل بۆلۈش سېستىمىىسى شەكىللەندۈرسەك بىزنىڭ ياش ئۆسمۈرلىرىمىزگە چوقۇم پايدىسى بولىدۇ.
مېنڭچە كونىلارنىڭ ئۇستاز زالىم بولسا شاگىرىت ئالىم بولىدۇ دەيدىغان گېپى ناھايىتى توغرا گەپكەن. ھە دېسە بالىڭىزغا بولۇشۇپ ئوقۇتقۇچىنى تىللىماڭ،ئوقۇتقۇچىنى ھاقارەتلىگەنلىك ئىلىم ئىگىلىرىگە قىلنغان ھاقارەتتۇر. بۇنداق ئاتا ئانا تەربىيلىگەن پەرزەنتنىڭ ياخشى ئادەم بولۇپ يېتىلىشى ناتايىن.
 
          ھۈرمەتلىك ئاتا ئانىلار:
        ئازراق ۋاقتىمىزنى چىقىرىپ پەرزەنتلەرنىڭ ئۈگىنىشىگە ھەقىقىي كۆڭۈل بۆلەيلى. پەرزەنتىڭىزگە ئانا تىل تەربىيىسى بېرىشنى ھەرگىز ئۇنتۇپ قالماڭ(پەرزەنتلىرى قېرىنداش مىللەتلەر تىلىدا ئوقۇۋاتقانلار)،چۈنكى تەپەككۇرى ئۆلگەن ئادەم جەسەتكە ئوخشايدۇ. پەرزەنتىڭىزنىڭ دەرستە ناچار بولۇشى ئەيىپ ئەمەس، مۇھىمى تىرىشىشتا، قانداق ئۈگىنىشتە، پوزىتسىيەدە، پوزىتسىيە ھەممىنى بەلگىلەيدۇ. بالىڭىز مەكتەپتە بىلىم ئۈگىنەلمىسىمۇ قانداق ئادەم بولۇشنى ئۈگىنەلىسە بۇنىڭدىنمۇ مۇھىم بايلىق يوق..

بالىغا كۆڭۈل بۆلۈش، ھەرگىزمۇ ئۇنىڭ كۆزىگە كىرىۋېلىپ ئۇنىڭ بېسىمىنى ئېغىرلىتىش، دەپ قارىمايمەن. بالىلارنىڭ يۈرۈش-تۇرۇشى، ئوي-ئىدىيسى دىگەندەكلەرگە كۆڭۈل بۆلۈپ قويايلى. ھازىرقى بالىلارنىڭ نىمە ئويلايدىغىنىنى بىلىپ بولغىلى بولمايدۇ. ئوقۇشقا قىلچىلىك قىزىقمايدۇ، دەيدۇ ئوقۇتقۇچى دوستلىرىم. بالىڭىزنىڭ ئىدىيسىگە تەسىر قىلىۋاتقان ناچار ئىدىيلەرنى ۋاقتىدا بايقاپ، ئۇنىڭغا توغرا يولنى كۆرسىتىپ بەرسىڭىز، بالا پۈتۈن دىققىتىنى ئوقۇشقا قاراتسا، قالغان ئىشقا مەكتەپ، ئوقۇتقۇچىلار بار. قىزىقمىغان ئىشنى زورلىغان بىلەن ھېچ نەتىجە چىقمايدۇ. ھازىر بالىلاردا بۇنداق قاشاڭلىقنىڭ پەيدا بولىشىدا ئائىلە، جەمئىيەت، مەكتەپنىڭ ئورتاق تەسىرى بولىشى مۇمكىن، ئوقۇغاننىڭ پايدىسى يوق، دىگەن ئىدىيە بالىلاردىكى ئەڭ چوڭ خاتا قاراشتەك قىلىدۇ ھازىر. ئوقۇپ خىزمەت تاپالماسلىق بىلەنلا ئوقۇشنى ھېچنىمىگە ئەرزىمەس قىلىۋېتەمدۇق. ئوقۇپ خىزمەت تېپىش، پۇل تېپىش ئۇ بىر ۋاستە ھەرگىزىمۇ ئاخىرقى نىشان ئەمەس، ئاخىرقى نىشانىمىز بەختلىك، ھۆرمەتكە سازاۋەر تۇرمۇشقا ئېرىشىش. بەزىلەر بەلكىم پۇل بولمىسا بۇ ئۇلۇغ غايە نىشانلىرىمىز يوققا چىقىدۇ، دىيىشى مۇمكىن، ئەمما تاغدەك پۇلىڭىز بولسىمۇ، شۇنى ئىشلىتىپ شۆھرەت قازانغۇدەك ئەقىل-ئىدراكىڭىز بولمىسا، مەنىڭچە بۇ تۇرمۇش، بۇ ئاقىۋەت ھەرگىزمۇ بىز ئارزۇ قىلغان ئاقىۋەت بولمايدۇ.

بەك ياخشى يېزىپسىز. بىز ئوقۇتقۇچىلارنىڭ كۆڭلىدىنى ئوتتۇرغا قويۇپسىز. بىر قىسىم ئاتا- ئانىلار بار بالىسىنى خۇددى پادىغا قوشىۋەتكەندەك، مەكتەپكە بىرىۋەتسە بالىسى نىمە ئۆگىنىۋاتىدۇ، بالىسىنىڭ ئەھۋالى نەدىن نەگىچە  دىگەنلەرنى ئويلاپمۇ باقمايدۇ. ھەتتا بالىسىنىڭ قانچىنچى سىنىپتا  ئوقۇۋاتقانلىقىنى بىلمەيدىغان ئاتا-ئانىلارمۇ بار.   مانا موشۇنداق ئاتا-ئانىلار ئوقۇتقۇچىلاردىن قۇسۇر ئىزدەيدۇ. ماڭا شۇنداق بىر ئائىلە باشلىقى ئۇچرىغان. بىر سورۇندا ئۇچۇرشۇپ قېلىپ ئوقۇتقۇچىلارنى  چۈشۈرۈپ سۆزلەپ كەتتى، ئۇ كىشىنىڭ بالىسى دەل بىزنىڭ مەكتەپتە ئوقۇيتتى. مەن ئۇ كىشىدىن بالىلىرى قانچىنچى سىنىپتا ئوقۇيدۇ دەپ سورىسام 5-يىللىقتىمۇ ياكى 6-يىللىقتىمۇ دەپ بىلمەيدۇ، ئاخىرى سىنىپ مەسئولى كىمدۇ دەپ سورىسام ئۇنىمۇ ئۇقمايدۇ، شۇڭا ئاچچىقىم كىلىپ ئۆزلىرىنىڭ ئوقۇتقۇچىلارنى ئۇنداق ياكى مۇنداق دىيىشكە سالاھىيەتلىرى توشمايدۇ.  بالىسىنىڭ نەچچىنچى يىلللىقتا ئوقىۋاتقانلىقىنى ھەمدە سىنىپ مەسئولىنىڭ كىم ئىكەنلىكىنى بىلمەي تۇرۇپ پىكىر قىلىشلىرى خاتا. ئەگەر ھەقىقى بالىلىرىغا كۆڭۈل بۆلسىلە مەكتەپكە بېرىپ بالىلىرىنىڭ ئوقۇتقۇچىسىنى تونىۋالسىلا ھەمدە ۋاقىتلىرى بولسا باللىرى بىلەن بىللە ئولتۇرۇپ شۇ ئوقۇتقۇچىلارنىڭ دەرسىنى ئاڭلاپ باقسىلا ، ئاندىن پىكىرلىرى بولسا قىلسىلا بولىدۇ دىگەنتىم، ئەتراپتىكىلەرمۇ ئۇ كىشىنى ئەيىپلەپ كەتتى. شۇنىڭ بىلەن ئاغزى بېسىققانتى.
