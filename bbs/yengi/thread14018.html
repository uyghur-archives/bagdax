ئۈندىدارغا ھەمبەھىرلەش ياردەم
يوللىغان ۋاقتى 2012-12-27 13:27:32 |ھەممە قەۋەتنى كۆرۈش
ئەسسالاممۇئەلەيكۇم:

بۇ راس بولغان ئىشلار، تورداشلارنىڭ تەپسىلىي ئوقۇپ ئاكتىپ پىكىر قاتناشتۇرىشىنى تۆۋەنچىلىك بىلەن سورايمەن.



 مەن يىڭى يىل ئۆرۈلسە 20ياشقا كىرىمەن،خوتەن شەھەر ئىچىدە ئاشپەزلىك قىلىمەن،تۈنۈگۈن چۈشتىن كېيىن گۈلباغدىكى قىزىل يۇلتۇز سودا سارىيىغا كىيىم-كېچەك سېتىۋالغىلى چىققانتىم،سودا سارىيىدا توساتتىن كۆرۈشمىگىلى 4يىل بولغان،باشلانغۇچ ۋە تۇلۇقسىزدا 9يىل بىللە ئوقۇغان قىز ساۋاقدېشىمنى ئۇچۇرتۇپ قالدىم،قىزغىن سالاملاشتۇق،ئۇ بىر قىز تۇنۇشى بىلەن بازار ئايلانغىلى كەپتىكەن،،نىمە ئىش قىلۋاتىسىز دەپ سورىسام، بىر تاللا بازىرىدا ئىشلەۋاتىمەن دىدى،ھەم مىنىڭ تىلفۇن نۇمۇرۇمنى سورىۋېلىپ ،ئۆزىنىڭ ئالدىراش ئىكەنلىكىنى ئېيتىپ كىتىپ قالدى..

تۈنۈگۈن كەچ سائەت 9لەر ئەتراپىدا يانفۇنۇم سايرىدى،تىلفۇننى ئالسام ھېلىقى سودا بازىرىدا ئۇچراشقان ساۋاقدېشىم ئىكەن،
ئەسسالاممۇ ئەلەيكۇم؟
ۋەئەلەيكۇم ئەسسالام،سىز ھازىر نەدە؟
ئاشخانىدا،ئىشتىن چۈشەي دىدىم،
ھە ئۇنداق بولسا ئۇزۇن يوللۇق بىكەت ئەتراپىدىكى خۇڭشىڭ مىھمانخانىسىنىڭ ئالدىغا كىلىڭە؟
نىمە ئىشتى؟
ۋاي تىز بۇلۇڭە كەلگەندە دەي،
ھە ماقۇلە مەن ھازىر باراي...
تىلفۇننى قۇيىۋېتىپ تىز-تىز تازلىقنى قىلىشقا باشلىدىم،ئالاھەزەل 20مىنۇت ئەتراپىدا تازلىقىمنى قىلىپ بۇلۇپ ئۇ دىگەن جايغا باردىم،..ئۇ ھېلىقى قىز تۇنۇشى بىلەن يول بۇيىدا تۇراتتى،ئۇلار بىلەن كۆرۈشتۈم.ئۇ قىسقا سالامدىن كىيىنلا ماڭا تاكسى توسۇشۇمنى ئېيتتى،سەۋەبىنى سوراي دەپ بولغۇچە ھېلىقى تۇنۇشى ۋاي تىز بۇلۇڭە دەپ ئالدىراتتى،،مەنمۇ ئارتۇق گەپ قىلماي تاكسى توسۇدۇم..ئۈچىمىز تاكسىغا ئولتۇرۇپ ۋىلايەتلىك مىتىرولوگىيە ئىدارىسىنىڭ ئارقىسىدىكى تار كوچىغا كىردۇق،ئۇ ئىككىسى تاكسىدىن چۈشۈپ مىنىڭ ساقلاپ تۇرۇشۇمنى تاپىلاپ بىر قۇروغا كىرىپ كەتتى،

ھېلىقى تاكسى شوپۇرى(مەن بىلەن تەڭ دىمەتلىك)ماڭا سىرلىق بىر قارىۋەتكەندىن كىيىن سورىدى:سىلە ئۇ ئىككىسى بىلەن تۇنۇشقىلى قانچىلىك بولدى،،بۇ سۇئالدىن سەل غەلىتىلىك ھىس قىلدىم ۋە يالغاننى تۇقۇپلا 1ئايدەك بوپ قالدى دىدىم،،،ئۇ قىزلاردىن ئۆزۈڭلارنى تارتىڭلار،،بۇنداق غەلىتە گەپتىن سەل چۆچۈدۈم..نىمىشقا دەپ سورۇدۇم..ئۇ ئىككىسىنىڭ ھېلىقى قۇروغا نىمىشقا كىرگىنىنى بىلەمسىلە؟ياق بىلمەيدىكەنمەن...
ئۇلار خىروئىن سېتىۋالغىلى كىرۋاتىدۇ ئۇ قۇروغا..

نىم،،نىمە؟خىروئىن؟كاللامغا بىر چىلەك سۇغۇق سۇ قۇيغاندەك بۇلۇپ كەتتىم،،ۋاي يالغان ئېيتمىساڭلىيا ئاداشوي،،چىك ۋاييييييييي،،،غۇدۇردۇم،ھەم ئىشەنمىدىم..يۈرۈگۈم ئەنسىز سوقۇشقا باشلىدى،،ئاللىقانداق ۋەھىمىلىك تۇيغۇ تىنىمگە تىتىرەك ئالاشتۇردى،،
ھىم بوپتۇ ئەمىسە مەن يالغان ئېيتقان بولسام ئۆزىدىن سوراڭلار..ئاڭغىچە ھېلىقى ئىككىسىمۇ قۇرودىن چىقىپ كەلدى،،ۋە شوپۇرغا بايا كەلگەن مىھمانخاننىنىڭ ئالدىغا ئاپىرىپ قۇيۇشنى ئېيتتى،،ئۇزاق ئۆتمەي يىتىپ كەلدۇق،،ئۇ ئىككىسى مىنى ياتاق ئۆيىگە باشلىدى،، يەنى شەكلى ئۆزگەرەەن مىھمانخانىغا باشلىدى....
مەن سەل غەلىتىلىك ھىس قىلىپ بىز بۇ  پاھىشخانىغا كىرىپ نىمە قىلىمىز؟ مەن ئالدىراشتىم،ھازىر كىتەي دىدىم؟ ساۋاقدېشىم ئۇنىمىسام ئۇنىماي زورلاپ ئەكىردى،،ئۈچىمىز بىر ئۆيگە كىردۇق،،ئۇلار ئىشىكنى ئىچىدىن تاقىۋېتىپ پارىڭىنى باشلىدى..

بىز ئىككىمىز مۇشۇ پاھىشخانىدا ئىشلەيمىز،، دىدى ئۇلار،نىمە،،سىلەر مۇشۇ يەردە تىنىڭلارنى ساتامسىلەر؟ئۇلار ھەئە دىگەندەك بېشىنى لىڭشىتتى،،چوڭقۇر بىر ئۇھسىندىم،ۋاي توۋا ھېي بىر تۈزۈكرەك گەپ قىلىڭە،،ئادەمنىڭ قىلىقىنى قىلسىڭىز بولمامدۇ؟توۋا خۇدايىم،،سىز نىمانچە ئۆزگىرىپ كەتتىڭىز؟ قېنى بۇرۇنقى ئۇماق،مۇلايىم ھالىتىڭىز؟سىز بىر مۇسۇلمان پۇشتىغۇ؟بۇ ئىشلار مىللىتىمىز قىزلىرىغا خاس ئەمەسقۇ؟

ئوھوش،،گەپ قىلمىسام نىمانداق تۇلا قاينايسىز،،خۇمارىم تۇتۇپ كەتكەندە،،ئۇ گىپىنى تۈگۈتۈپلا يېنىدىن تاماكا چىقىرىپ ئىككىسى تۇتاشتۇرۇپ چىكىشتى ۋە ماڭا تەڭلىدى ياق،،ياق،،مەن چەكمەيمەن،،،ھا ھا ھا ئوغۇل بالىمىش تېخى،،تاماكا چەكمىگەن نىمىسىگە،،ئۇ ماڭا مەسخىرلىك قارىدى..

ھەي بايا خۇمارىم تۇتۇپ كەتتى دەيسىز نىمە خۇمار ئۇ؟
مۇشۇ تاماكىنىڭ خۇمارى بولمامدۇ؟
يالغان ئېيتمىسىڭىزا،بايا ھېلىقى قۇروغا نىمىگە بارغانتىڭىز؟ئۇلار سەل ئەندىكتى بولغاي چىرايى تۇتۇلدى،،مىنىڭ ئۇنىڭ ئاق تاماكا چىكىدىغانلىقىنى بىلۋالغانلىقىمنى سەزگەن بولسا كىرەك،،ئىشنى يۇشۇرمايلا ئېيتتى،،ھەئە چىكىمىز..چەكسەك نىمە بوپتۇ؟ھىم ھېلىقى شوپۇرنىڭ گىپى راسكەندە؟نىمە ؟سىزگە بۇ ئىشنى ھېلىقى شوپۇر دىگەنما؟ھەئە شۇ دىگەن..
شوپۇرنى تىللاشقا باشلىدى،
ۋاي ئانناڭنى ھارامدىن بولغان ھېجىقىز،،ۋاي چىرايلىق گەپ قىلىڭە،،دىدىم مەن...

ئۇلار مىنىڭ ئۇلارنىڭ زەھەر چىكىدىغانلىقىنى بىلىپ بولغانلىقىمنى بىلگەندىن كىيىن پاراڭلىرىنى مەندىن يۇشۇرماي دىيىشىشكە باشلىدى،،

ئاداش بايا كىيىن قاپتۇق ئەمەسمۇ؟ ،خىروئىن تۈگەپ كىتىپتۇ ؟
ئەمدى قانداق قىلىمىز؟
قانداق قىلساق بۇلاركىنتاڭ...
مەنمۇ ئەتە ئەتتىگەنگىچە چىدايمەن،،
سەنچۇ؟دىدى ھېلىقى قىز ساۋاقدېشىمغا؟
ساۋاقدېشىم،،،ئاداش مىنىڭ ھازىرلا خۇمارىم تۇتۇپ كىتىۋاتىدۇ،،يە بولمىسا ھېلىقى جايغا بارايلىمۇيا؟
قايسى جايغا(ئۇلارنىڭ دىمەكچى بولغىنى خوتەن شەھرىگە ئانچە يىراق بولمىغان ئىمام ئەزىم مازىرى ئەتراپىدا ئىكەن،،يەنى خەلقىمىزگە ئاپاق خوجا زامانىسىدىن ئۇدۇم قالغان ئىسلام دىنى ئىشھاقىيە سۈلۈكىدىكىلەرنىڭ زىكرى قىلىپ ئاللاھقا ئىبادەت قىلىدىغان جايى ئىدى)

ئۇ ئارلىقتا ساۋاقدېشىمنىڭ زەھەر خۇمارى تۇتۇپ، بۇلالمايلا قالدى،،ئۇنىڭغا ئىچى ئاغرىغان دوستى،،بىز ئىككىمىز ئىمام ئەزىم مازىنىڭ ئەتراپىدا بىز سودىگەردىن خىروئىن ئەپكىلەيلى دىدى ماڭا؟
ياق مەن قەتئىي بارمەيمەن،،مىنىڭ تىنىچ ياشىغۇم بار،،بۇ ئىش بەكلا خەتەرلىك دىدىم،،ۋاي خاتىرجەم بۇلۇڭ بارايلى تىز بۇلۇڭ ئەزمىلىك قىلماي تۇرۇڭە،،،،،،،ئىككىلىنىپ قالدىم،،،،

ساۋاقدېشىم يىغلاپ ئۆزىنى كارۋاتقا ئېتىشقا باشلىدىم،،ئۇنىڭغا ئىچىم ئاغرىدى،،ۋاي خۇش بوپ قالاي ئۇ يالغۇز بارسا بولمايدۇ،،سىز قوغدىغۇچى بۇلۇپ بىللە بېرىشىپ بىرىڭ،،بىرەر ئىش بوپ قالسا مەن ئىگە دىدى،،،ئۇنىڭ نالە-پەرياتلىرىغا چىدىماي بارسام باراي دىدىمدە ئورنۇمدىن تۇرۇپ ئۇنىڭ بىلەن سىرتقا ماڭدىم،،، 


يولغا چىقىپ تاكسىي توسۇدۇق،،ئالاھەزەل قىرىق مىنۇتتا ئۇجايغا بېرىپ بولدۇق،،مەن يەنىلا ئاۋۋالقىدەك تاكسىدا ئۇنىڭ چىقىشىنى كۈتۈپ ئولتۇردۇم،،ئۇ كىرىپ كىتىپ 10مىنۇتلاردىن كىيىن چىقتى،،،يەنە ياتاققا قايتتۇق،،ئىشلىرىمىز بىخەتەر ئۇڭۇشلۇق بولدى...

ئۇ ئەسلى بىر سۇيۇقلۇق زەھەر ئىكەن،،ئۇنى شىپىرىسقا تارتىپ بىلىكىگە ئۇردىكەن،،زەھەرنى ئۇرۇپ بۇلۇپلا ،،ئۇھ دەپ ئاز-تۇلا تىنىچلاندى،،ئىككىلىسى بىلىكىگە  بىر قېتىمدىن ئۇرۇپ كۆڭلى تىنىچلاندى،،،

مەن شۇ تاپتا قانداقلا بولمىسۇن،،ئۇ ئىككىسىنى بۇ يامان يولدىن قۇتۇلدۇرۇپ چىقىرىش كىرەك دەپ ئويلىدىم،،ئۇلارنىڭ بۇ يولغا كىرىپ قېلىشىنىڭمۇ مەلۇم سەۋەبى بارلىقىنى،بۇرۇن چوڭقۇر ئازاپ-ئوقۇبەت تارتىپ ئۆزىنى مۇشۇنداق  تاشلىۋەتكەنلىكىنى ھىس قىلدىم..

ئۇنىڭغا نەسىھەت قىلىشنى كۆڭلۈمگە پۈكۈپ،ئامال بار ئۇنىڭغا تەسىر قىلىدىغان گەپلەرنى دەپ ئۇنىڭ يامان يولدىن چىقىشىغا تۈرتكە بۇلۇشۇمنى ئويلاپ ،ئۇ ئىككىسىگە ئېغىز ئاچتىم..تېخى ئۆزىنىڭ ياش ئىكەنلىكىنى ھاياتىنى بۇنداقلا نابۇت قىلۋەتمەسلىك كىرەكلىكىنى، بۇ يولدىن قايتىپ چىقىشىنى،،ئەگەر خالىسا زەھەر تاشلىشىغا ياردەم قىلىدىغانلىقىمىنى ئېيتىپ،،خېلى نۇرغۇن تەربىيە ئىشلىدىم،،بۇرۇنقى كۈنلەرنى ئۇنتۇپ گۈزەل كەلگۈسىنى يارىتىشنى،ئاللاھقا ئىبادەت قىلىپ ياخشى يولدا مېڭىشىنى دىدىمن،،ھەم ئۇلارنىڭ ئازار يىگەن كۆڭلىگە تەسەللىي بەردىم،،،

ساۋاقدېشىم،، مىنى چىڭ قۇچاقلاپ ئىسەدەپ يىغلاپ كەتتى،،مەن نېمىدىگەن شور پىشانە ھە؟ مەن ئۆلسەم بولماسمىدى؟مەن بۇ دۇنياغا مۇشۇنداق خورلۇق تارتىشقىلا كەلگەن بولغىيمىتتىم دەپ زارلاندى،،،بىز مەكتەپتە ئوقۇغان چاغلاردىمۇ ئۇنىڭ ئائىلىسىنىڭ قانداق ئىكەنلىكىنى بىلەتتىم،،ئۇنىڭ ئانىسى ئۈلۈپ كەتكەن،،ئۆگەي ئانىسى ئۇنى ھەدىسىلا خورلايتتى،،

مەن ئويلاپ قالدىم،،ئۇنىڭ يامان يولغا كىرىپ قېلىشى بەلكىم ئائىلىىسىدىن كۆڭلى ئىللىقلىق ھىس قىلمىغان چېغى ،

ئۇ كۆزلىرى تارام-تارام ياش ھالدا كىچكىدىن تارتىپ بولغان ئىشلارنى،،تارتقان خورلۇقلىرىنى بىر-بىرلەپ سۆزلەشكە باشلىدى،،ئۇنىڭغا ھۆرمەت قىلىپ گىپىنى ئەستايىدىللىق بىلەن ئاڭلىدىم،،ئۇ ئەسلى ئوغۇل دوستىنىڭ بىرلا چىكىپ باققىن،،بىر چەككەنگە ىۇمار بۇلۇپ قالمايسەن دىگەن سۆزىگە ئىشىنىپ   بىلمەي چەككەنلىكىنى دىدى،،ھەم ئۇنى چىكىشكە كۆپ پۇل بولمىسا بولمايدىغانلىقىنى بىلىپ ئامالسىز پاھىشخانىدا ئىشلەپ تاپقان پۇلغا خىروئىن چىكىۋاتقانلىقىنى دىدى...

مەن ئۇنىڭغا دىدىم..مەن ئەتىلا قانداق قىلغاندا زەھەر تاشلىغىلى بۇلىدىغانلىقىنىڭ ئامالىنى ئىزدەپ باقاي،،،ھەراس ھۆكۈمەت ئەيدىز داۋالاش ئورنىدا زەھەر تاشلاتقۇزىدىغۇ؟ دىدىم،،مىنىڭ ھاياتىم ئاخىرلىشىش ئالدىدا تۇرۋاتىدۇ دىدى ئۇ چوڭقۇر ئۇھسىنغا ھالدا،،،،،
نىمە سىز ئەيدىز ئەمەسسىز ھە؟
ۋاي ياقەي مەن ئەيدىز ئەمەس،،لىكىن ھەر قېتىم زەھەر خۇمارىم تۇتۇپ كەتكەندە پۇت-قوللىرىم قېتىپلا قالىدۇ،،ئەگەر زەھەر ۋاختىدا يىتىپ كەلمىسە،،قېتىپ قېلىپ ئۈلۈپ كىتىمەن،،ئۆتكەندىمۇ تاسلا قالدىم ئۆلگىلى دىدى،،،،

ئۇنداقتا بىز ئەيدىز داۋالاش ئورنىغا بېرىپ باقايلى،،زەھەرنى تاشلاتقۇزسا ئەجەپ ئەمەس دىدىم،،،

ئۇ،، ياق،،ھۆكۈمەت تارماقلىرى بۇ ئىشنى بىلىپ قالسا يەنە تۈرمىگە كىرىپ كىتىمەن،،ئۇندىن كۆرە قېتىپلا ئۆلگىنىم ياخشى دىدى،،،

ياق سىز ئۆلسىڭىز بولمايدۇ،،بۇنىڭغا تېخىمۇ ياخشىراق ئامال چىقىپ قالىدۇ،،،سىز تۇرۇپ تۇرۇڭ مەن ھەرقايسى مۇنبەرلەرگە بۇ ئىشنى تىما قىلىپ يوللاپ باقاي،، ياردەم قىلىدىغانلار،يەنى زەھەر تاشلاشنىڭ ئامالىنى چوقۇم بىلىدىغان تورداشلار چىقىدۇ ..دىدىم...

ھۆرمەتلىك تورداشلار ئۇ قىزنىڭ ئەھۋالى سىزلەرگە ئايان بولدى،،،ئۇ قىز قانداقلا بولمىسۇن مىللىتىمىزنىڭ كەلگۈسى ئانىسى،،بىز ئۇيغۇر يىگىتى بولغانكەنمىز ئۇنى بۇ يولدىن قۇتۇلدۇرۇپ چىقىرىشىمىز كېرەك؟
