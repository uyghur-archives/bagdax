   غەيرىتىڭىزگە بارىكاللا ، قىلغان سۆزلىرىڭىزنى ئاساسەن ياقتۇردۇم . لېكىن سىز مەن قارىسام ئاپتونومىيە قانونى بەرىبىر ئاقمايدۇ ،شۇڭا ئويغۇرلار ئۆز چىقىش يولىنى ھۆكۈمەتكە قاراپ ئوتۇرماي ئۆزى تېپىشى كېرەك دەپ ئويلاۋاتقاندەك قىلىسىز . مىللىتىمىز  سىزگە ئوخشاش تىرىشچان ، غايىلىك كىشىلەرگە ئېھتىياجلىق ، شۇنداقلا يەنە ئاپتونومىيە قانونىغا تېخىمۇ ئېھتىياجلىق ،چۈنكى مىللىتىمىز ئۆزىدىن 130 ھەسسە كۆپ بولغان مىللەت بىلەن رىقابەتكە چۈشسە بۇ ئادىللىق بولمايدۇ . نۇرغۇن ئىشلارنى يەنىلا دانا ھۆكۈمىتىمىز تۈزگەن قانۇنلار ئارقىلىق ھەل قىلىشقا توغرا كېلىدۇ .  
 
 سالام holmis !
توغرا دەيسىز،بىز نۇرغۇن ئىشلاردا ئاساسى قانۇن ۋە ئاپتونومىيە قانونىغا تېخىمۇ ئېھتىياجلىق ،بىراق ھازىرقى ئەھۋالىدىن قارىغاندا  بۇ قانۇنلارنىڭ  ئەمىلىشىگە خېلى ۋاقىت كېتىدۇ ،مەن ئويلايمەن بىزنىڭ ئىختىسادى ئېڭىمىز يۇقۇرى كۆتۇرلۇپ ،كوپلەپ كارخانچىلار قۇشۇنى بارلىققا كەلسە،ئىقتىسادى ئېڭىمىز يۇقۇرى كۆتۇرۇلسە  شۇ مۇناسىۋەتلىك قانۇنلارنىڭ ئەمىلىشىگە ئاكتىپ تەسىر كۆرسىتىدۇ،ئاددىسى باشقىلارنىڭ ئالدىدا سوزلەش پۇرسىتىگە ئىرىشەلەيمىز(چۇنكى كونىلاردا گاچا بولسىمۇ باينىڭ بالىسى سوزلىسۇن دىگەن گەپ بار ئەمەسمۇ)،بىزمۇ يۇرتۇمىزدىكى بەزى تۇرلەرگە مەبلەخ سالالايدىغان ،تۇرلەرگە قاتنىشالايدىغان دەرىجىگە يىتەلەيمىز؛مەنچە جەمىيەتنىڭ تەرەقىقى قىلىشىغا ئەگىشىپ باشقىلار پۇلۇمىز يوق چاغدا ئىرەن قىلمىغان ئىشلارغا ئىرەن قىلىدىغان بۇلۇدۇ،يەنە بىر جەھەتتىن ئاشۇ ئىشسىزلىق دەستىدىن تۇيۇق يولغا كىرىپ قېلىۋاتقان ياشلارنى ،نامراتلىق دەستىدىن ئىچكىرلەردە سەرگەردان بۇلۇۋاتقان قېرىنداشلارنى،قىينچىلىق دەستىدىن ئالى مەكتەپلەردە ئۇقۇيالمايۋاتقان سىتودىنتلارنى، بۇزەك قىلىشقا ئۇچراۋاتقان ئاشۇ جاپاكەش نامرات دىھقانلىرىمىزنى قۇتقۇزاغىلى بۇلىدۇ، پۇلسىزلىق ۋە باشقا سەۋەپلەر تۇپەيلى ياسالماي كىلىۋاتقان‹‹زۇردۇن سابىرنىڭ خاتىرە ئۇيى››گە ئوخشاش مەشھۇرلىرىمىزنىڭ خاتىرە ئۆيلىرنى ياسىيالايمىز،قىسقىسى مىلىتىمىزنىڭ مۇشۇ رىقابەت دەۋرىدە شاللانماي، ئاتا-بوۋلىرىمىزنىڭ يۇلىنى داۋاملاشتۇرۇپ كىتەلەيمىز.ئەلۋەتتە بۇ بىر مۇشكۇل جەريان ،بىراق بىز بەل قۇيۇۋەتمەسلىگىمىز كېرەك،ئىقتىدارىمىز بارلىرىمىز ئىقتىدارىمىزنى چىقىرايلى،پۇلۇمىز بارىمىز ئاز تولا پۇلۇمىزنى چىقىرايلى،قۇلۇمىزدا ھىچنىمە بولمىسا شۇ تىرىشىۋاتقانلارغا ئىككى قوللاپ چاۋاق چىقىرايلى،شۇنداق قىلساق يېقىن كەلگۇسىدە يۇرتۇمىزدا  ھەر قايسى جەھەتلەردىن يىتىلگەن غوللوق ئادەملەركۆپلەپ يىتۇشۇپ چىققۇسى .
ئەگەر سۆزۇمدە بەزى نوخسانلار بولسا تەنقىدى پىكىرلەدە بۇلۇشىڭىزنى ئۇمىت قىلىمەن،(راستىنى دىسەم سەككىز يىل بۇپتىكەن مۇنداق نەرسىلەرنى يازمىغىلى)
