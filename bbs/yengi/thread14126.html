     بۇ تىمىنى كورۇپ ئىچىڭىز بىر خىل غەزەپكە تۇلىۋاتىدۇ ھە ئىزىز كۇرۇرمەنلىرىم،لىكىن ئالدىرىماي يازمىغا بىر قىتىم كۇز يۇگۇرتۇپ بىقىڭ،ئاندىن يۇزۇمگە 10 كاچات ئۇرسىڭىزمۇ تۇرۇپ بىرىمەن،،،
سىز ئۇيغۇرمۇ؟؟؟،،
سىز ئۇيغۇرنى قانچىلىك چۇشۇنىسىز؟؟؟
ئۇيغۇر دىگەن سۇزنىڭ مەنىسى زادى نىمە؟؟؟؟
بۇ سۇئاللارغا جاۋاپ بىرىشكە ئالدىرىماي تۇرۇڭ،،،چۇنكى ئالدىڭىزدىكى شۇنچىلىك بىر ھەقىقەتنى تۇنۇشقا ئاجىز كەلدىڭىز،،،،چۇنكى سىز ئۇيغۇرلار كۇپرەك تۇرىۋاتقان يۇرىتلاردىن باشقا يۇرىتلاردا مەلۇم ۋاقىت،ھۇنەر -تىجارەت،ياكى ئۇقۇش قاتارلىق ئىشلار بىلەن شۇغۇللانغان ۋاقتىڭىزدا ،، بۇ سۇئالىمنىڭ جاۋابىغا ئىرىشىپ قالىسىز،،،
سىز ئۇيغۇرلار مەرگەزلىك ئولتۇراقلاشقان رايۇنلاردا ،بىر قىسىملارنىڭ قىلىۋاتقان يامان ئىش -ھەرىكەتلىرىنى،كورۇپ نەپرىتلىنىۋاتقان بولۇشىڭىز مۇمكىن،،،لىكىن قانداقلا بۇلمىسۇن ئۇلار سىزنىڭ ئەتىراپىڭىزدىكى كىشىلەر،،يۇلىڭىز ۋە قىلىۋاتقان ئىشلىرىڭىز بىر بولمىسىمۇ لىكىن يەنىلا تۇرىۋاتقان يىرىڭىز بىر،،،ئىچىۋاتقان سۇيىڭىز بىر،نەپەسلىنىۋاتقان ھاۋا-ئوكسىگىنىڭىز بىردۇر،،چۇنكى سىز ئۇز نوۋىتى كەلگەندە شۇ سىز ئانچە ياقتۇرمايۋاتقان،كۇزىڭىزگە سىغمايۋاتقان كلىشىلەرنىڭ يارىدىمىگىمۇ مۇھتاج بۇپ قىلىشىڭىز مۇمكىن،،،قىنى بىر مىسال كەلتۇرەيمۇ؟؟؟
 
سىزنى بىر يۇك ماشىنا شوپۇرى دەپ قارايلى،قاتتىق قار يىغىۋاتقان سوغۇق قىش كۇنلىرىنىڭ بىرسىدە ،سىزنىڭ ماشىنىڭىز تۇيۇقسىز،،شەھەرگە يىراق بىر ئۇيغۇر مەھەللىسىدە بۇزۇلۇپ قالدى،قورسىقىڭىز ئاچ،ماشىنا بۇزۇق،،،،سىز قانداق قىلىسىز؟؟؟
تىزدىن ئۇزىڭىزگە ئەڭ يىقىن بىر ئائىلىلىكنى ئىزلەيسىز دە ئىشىكنى قاقىسىز،،
چاچ ساقاللىرى چارلاشقان،،پۇتىغا كونىراپ -ئاقىرىپ كەتكەن مەسە-كالاچ كىيۋالغان بىر دىھقان بۇۋاي چىقىدۇ،سىز :
__ئەسسالامۇ ئەلەيكۇم،،ماشىنام تۇيۇقسىز بۇزۇلۇپ قالدى قاراڭ،،شۇڭا،،،،
سىز دەپ بولغۇچە شۇ ئۇيغۇر بوۋاي مۇنداق دەيدۇ،،،
___قىنى ئاۋال ئويـگە كىرسىلە،،،بەك سوغۇق بۇپ كەتتى،،سىلە بىر ئىسىنىۋالسىلا،،ئاڭغۇچە،،،
شۇنداق قىلىپ ئوي ئىگىسى سىزنى قىلچە يات ئالماي،باي ياكى كەمبىغەللىكىگە قارىماي سىزنى مەش قويۇلغان،ئىسسىق ئويىگە باشلايدۇ،،ھال كۇنى ئانچە ياخشى بولمىغان بىر ئۇيغۇر ئائىلىسى سىزنى تۇنۇمىسىمۇ ،،سىزگە قىززىق بىر پىيالە چاي بىلەن قاتتىقراق بولسىمۇ بىر ناندا چاي ئەكىلىدۇ،،سىزغۇ ياراتمايسىز ھەقىچان،،بۇنداق ئاددى داستىخانلارنى،كۇنلىلىكى گۇش قورۇمىلىرىنى يەپ خام كىكىرىپ يۇرگەندىكىن،،،لىكىن ئامالىڭىز يۇق،،بۇ ۋاقىتتا يىنىڭىزدىكى ياكى بانكا كارتىڭىزدىكى پۇلنىڭ پەقەتلا كۇچى يۇق،،،،لىكىن شۇنداقتىمۇ سىز چاي ئىچكۇچە شۇ ئۇيغۇر سىزدىن گەپ سوراپ،ئارىدىكى جىمجىتلىقنى بۇزىدۇ،،،
___ئۇكام،،،ئۆزلىرى شەھەردىنمۇ يا؟؟؟؟
بۇ گەپ سىزنىڭ قۇلىقىڭىزغا تىتىمايدۇ،،چۇنكى سىز ساپ-ساقلا ماشىنىڭىزنىڭ بۇ كورۇمسىز يۇرىتتا بۇزۇلۇپ قىلىپ سىزگە ئاۋارىچىلىق ئەكىلىشىنى ئويلاپمۇ باقمىغان ئىدىڭىز،،،شۇنداقتىمۇ جاۋاپ بەرمەي ئامالىڭىز يۇق،،،،
---شۇنداق،،ئا *******نىڭ ئويىسىگە چىققان،،كەچ قىلىپ ماشىنا بۇزۇلۇپ قىلىپ****
دىھقان دەيدۇ،،،
---ئا ،،،،، ھىچقىسى يۇق ئۇكام،،بىللە قاراپ باقايلى،ساقىيىپ قالغۇدەك بولسا شەھەرگە كىرىپ كىتەرلا،،،بولمىسا ئويلىرىمىز شەھەردىكىدەك ئىسىل بولمىسىمۇ ،لىكىن كوڭۇل -كوكسىمىزدەكلا كەڭ جۇمۇ،،،لىكىن خاتىرجەم بولسىلا ،،،ئورۇن -كورپىلەردە پىت بارمىكىن دەپمۇ ئەنسىرەپ كەتمىسىلە،،ھازىر بىز دىھقانلارنىڭمۇ ھال ئەھۋالىمىز ياخشىلىنىپ،،پىتلارنى ئۇرۇقلۇق ئۇچۇنمۇ تاپالمايۋاتمىز دىسىلە؟؟؟
سىز ئەگەر ئۇزىڭىزنى ھەقىقى ئۇيغۇر دەپ قارىسىڭىز ،،،مانا بۇ قىزىقارلىق چاقچاقلارغا تە بولماي ئاشۇ ئاق كوڭۇل ،،ساددا دىھقان بىلەن ئاسمان باقتى بولۇپ كۇلۇپ كىتىسىز،،، ،
سىز نىمە دىيىشىڭىزنى بىلمەي قالىسىز،،سىزدە راستىنلا ۋىجدان بولسا،،،،،،
سىز ئۇيغۇر بولسىڭىز، چۇقۇم ۋۇجۇدىڭىزدا ئۇزىڭىزنىڭ ئۇيغۇر مىللىتىدىن ئىكەنلىكىڭىزدىن چەكسىز پەخىرلىنىسىز،،،
بۇۋاينىڭ ئوغۇللىرى كىلىپ ماشىنا يىنىغا چىقتىڭلار،،،لىكىن بۇ تىشى --پارقىراپ تۇرغان بىلەن ،،يىلىكىدىن كەتكەن ماشىنا ،، ‹‹راس گىپىنى ئىيتىپ ›› سىزنى خىجىللىق دەرياسىغا باشلىغاندا ،بوۋاي سىزدىن سەمىمىلىك بىلەن سورايدۇ::
ئۇكام،،بەك زورۇل ئىشلىرى بولمسا ،،ماشىنىنى ئەتە كۇندۇزدە رىمۇنۇت قىلدۇرسىلا ،ئويدىكىلەرگە تىلپۇن قىلىپ خاتىرجەم قىلىپ،،ئۆيگە بەك قايىتمىسىلا بولمايدىغان ئىش بولسا،، كىچىك ئوغلۇم قوشنا كەنىتكە بىرىپ ،كىچىك ماشىنىسى بار تۇرسۇن قارىنى چاقىرىپ كەلسۇن،،ئارىلىقى 10 كىلۇمىتىرچە كەلسىمۇ ئوغلۇم بىر سائەتتىلا بىرىپ كىلەلەيدۇ ،،ئۇ جەرەندەك تىز يۇگىرەلەيدۇ،،،( بۇۋاينىڭ ھويلىسىدا ئوتقاشتەك قىپ-قىزىل بىر مۇتسىكلىت بار ئىدى،لىكىن قار كوپرەك بولغاچقا ،، يۇللاردا مۇتسىكلىت مىنىشكە بولمايتى ئەلبەتتە)،ماشىنىلىرىنىڭ قىشىدا مەن ياتاي كىچىچە،،مەھەللمىزدىغۇ ئوغرى-يالغانلار يۇق دىيەرلىك،،بىراق سىلە خاتىرجەم بۇلالمايدىغان ئوخشايلا،،،
سىز خاتىرجەم بۇلۇپ قونۇپ قالماقچى بولدىڭىز،،،كەچ بولدى،،بۇۋاينىڭ ئاددى --ساددا كىيىنگەن ،،خۇش پىچىم كىلىن قىزى ئەتكەن،ئاش كۆكى گۇپۇلدەپ پۇراپ تۇرغان،تىرناقچىلىك قىلىپ تۇگۇلگەن،قىيمىلىق چۆچۇرىدىن ئىككى چىنە ئىچىپ ،،دىھقان ئۇزى تىرىغان پاختىدىن قىلىنغان،،،تا بۇگۇنگىچە ھىچ كىمنىڭ ئۇستىگە يىپىنىپ بىقىشقا پىتىنالمىغان قىلىن ۋە يۇقمشاق يۇتقان -- كورپىدە راھەتلىنىپ 4 يۇلتۇز دەرىجىلىك مىھمانخانىدىكىدەك ياتقىنىڭىزدا،،، ئاشۇ ئاق كۆڭۇل،سەمىمى دىھقان، كونىراق بىر پارچە تىكمەت كىگىز بىلەن يۇڭلۇق جۇۋىسىنى ، ماشىنىڭىز قىشىغا ئاچىقىپ يىغىۋاتقان قارغىمۇ پەرۋا قىلماي مۇگۇدەشكە باشلىدى،،،
مەنە بۇ بىر ئويغۇر تۇرمۇشىنىڭ كىچىككىنە كارتىنىسى،،،ئۇ سىزنىڭ پۇلىڭىزنى كۇزلىمىدى،،سىزدىن ھەق تەلەپ قىلمىدى،،لىكىن بىر مىللەتلىك-مۇسۇلمانلىق رۇلىنى ھەقتادىغا يەتكۇزۇپ ئورۇنلىدى،،
 
ئۇيغۇرلىرىمىز ياشاۋاتقان يۇرىتلارنىڭ ھەممىسىدە ،،ئوغرى-يالغانچى،،كاززاپ-ئالدامچى،،،زىناخۇر --بۇزۇقلار ۋە قارى-ئولىمالار ماۋجۇت،،،شۇنىڭلىق بىلەن ما يۇرىتتىكىلەر مۇنداق،،،ئا يۇرتتىكىلەر ئانداق،،،،دەپ بىر كىشنى بىر يۇرتقا ۋەكىل قىلىشقا ھىچ قايسىمىزنىڭ ھوقۇقى ۋە ھەققىمىز يۇق،ئەمما يۇرتىمىزنى قىزغىن سويۇش،خەلقىمىزگە قولىمىزدىن كىلىشچە ياخشىلىق قىلىش مەجبۇرىيتىمىز بار،،، تۇرمۇشۇمىزغا كۇنسىرى يىقىنلىشىۋاتقان ئاتالمىش ‹‹شەھەرلىشىش مەدەنىيەت ھادىسىسى ›› بىر بىناغا جايلىشىپ ياشاۋاتقان 40-50 ئائىلىلىك 100 مىتىر دائىرىدە ياشاۋاتقان بولساقمۇ لىكىن ،،بىزدە باش كوتۇرۇشكە باشلىغان ‹‹ئۆزۇڭنى بىل ،ئۆزگىنى قوي ›› ئىددىيەسى بىز ئۇيغۇرلارنىڭ ئەسلىدىكى ‹‹ئۇيۇشقاق ››لىق ئىسمىمىزگە نىسبەتەن بىر قارا داغ بولۇشقا باشلىدى،،،،،،،،،،،،،
 
ۋىجداننىڭ گىپى بولسا،،كوچىلارغا چىقىپ قالغان قىز-ئاياللارنىڭ مەسلىسى ئوتتىرىغا قويۇلغان تىمىلار ئوتتۇرىغا قويۇلسا ، تۇكۇرۇكلىرىمىزنى چاچىرىتىپ تۇرۇپ ئىنكاس يىزىشىمىز،،،دومبۇلاق چوگۇندەك قايناۋىرىپ بويىنمىزغا تاشىمىزكى،،،
بۇ مەىسلىنى مانداق يۇل بىلەن ھەل قىلساق بولاتتى ؟؟؟؟؟
دەپ بىر ئىجادى پىكىرىمىزنى بىرەلمەيمىز،شەھەرلەردىكى تىرىلغۇ يىرىدىن ئايرىلىپ قىلىپ،،ئۇششاق ھوپىگەر-تىجارەتچىلىك بىلەن شۇغۇللىنىۋاتقانلارنىڭ قىشىدىن ئوتكۇنىڭىزدە،،يۇللارنىڭ قىستىلىشىپ كەتكەنلىكىدىن بىزار بولۇپ ئاشۇ ئۇز تىرىكچىلىكىنى ھالال يۇللاردىن تىپىۋاتقانلارغا كۇڭۇل بولۇش-قۇللاش ئۇياقتا تۇرسۇن،،ئۇلاردىن بىزار بولسىز،،ئۇلاردىن نۇمۇس قىلىسىز،،كوڭۇل بۇلۇش دىگەنلىك ئۇلارغا پۇل بىرىڭ ،،،دىگەنلىك ئەمەس،،،ئۇلار سىز ئىھتىياجلىق بولغان ، كۇندىلىك تۇرمۇشىمىزدا كۇپ ئىشلىتىدىغان ماللارنى ساتىدۇ،،بەلكىم سۇپىتى سىز تولا كىرىپ ‹‹پۇل دىگەننى خەجلەپ،شاما قىلىۋىتىدىغان ››چوڭ تىپتىكى تاللا بازارلىرىنىڭكىدەك سۇپەتلىك بولماسلىقى مۇمكىن،،لىكىن ساقتا ئەمەس،،چۇنكى شۇ ئۇيغۇر تىجارەتچىلەرنىڭ بۇ بويۇملارنى ئىشلەپچىقىرىدىغان زاۋۇتلىرى يۇق،،ئۇلار ئوغۇرلۇق قىلسا---تۇتىۋالىدۇ،،قىمار ئوينىسا ---ئۇتىۋالىدۇ،،،،
سىز ئاشۇلارنىڭ ھۇجۇدىدىكى قىيىنچىلىقلارغا باش ئەگمەس رۇھىنى ،،، چەكسىز تەكلىماكان قۇملۇقىدىكى نەچچە مىڭ يىللىق تۇغىراقلاردىن كەلمىگەن،،،
ئاشۇ تۇغراقلارغا ھۇجۇدى سىڭىپ كەتكەن ، ئۇيغۇرلارنىڭ ۋارىسلىرى ئەمەس مەن

شۇنداق،،،ئۇيغۇرلار ئوبىكتىپ ۋە سوبىكتىپ سەۋەپلەر تۇپەيلىدىن ،مەدىنىيەت -مائارىپ،پەن تىخنىكا ،،قاتارلىق بىر قىسىم نازۇك تىخنىكا ئىش ھالقىلىرىدا ،،نۇرغۇنلىغان ئوڭۇشسىزلىقلارغا ئۇچىراپ كەلدى،،ئىچكى ئولكىلەردىن كىلىپ تۇرىۋاتقان ‹‹ماتاڭچىلار ۋەقەسى ››گە ئوخشاش ، ئەخلەت -ماز پاراڭلارغا ئىچىمىز ئاچچىق بولدى، ئابدۇكىرەم ئابلىز دەك،،ماۋۇ ئۇسۇل خانىشى دىلنار ئابدۇللادەك ‹‹خەلىق سويگەن سەنئەتكارلىرىمىز››نى يەتتە ياشتىن يەتمىش ياشقۇچە ھەممىمىز بىلىمىز يۇ، دۇنيا فىزىكا ساھەسىدىكى 30 يىلدىن ئارتۇق تالاش تارتىشنى ھەل قىلىپ ،، دۇنيا بويىچە ئەڭ مەشھۇر ‹‹ئارخانۇۋ-كەشىر ›› دىن ئىبارەت ئىككى فىزىكا ئالىمىنىڭ خاتالىقىنى تۇزۇتۇپ بەرگەن،، ئىلمى ماقالىلىرى دۇنيا بويىچە ئەڭ زورۇر ‹‹نەزىريەۋى فىزىكا ئىلمى ›› دەرىسلىك پايدىلىنىش ماتىرياللىرى قاتارىغا تىزىلغان، ئۇيغۇر قىزىكا ئالىمى ساھىپجامال دۆلەت خانىمنى بىلىدىغانلار زادى قانچىلىك؟؟؟
ئامىرىكىدىكى ‹‹ دولەتلىك ئالەم قاتىنىشى›› ئىدارىسىدا خىزمەت قىلىدىغان 10 مىڭغا يىقىن مۇتەخەسىس ئالىملار قاتارىدا،، ئەركىن سىدىق ئىسىملىك بىر ئۇيغۇر ئالىمى بار،،، ئەركىن سىدىق ئەپەندى مۇشۇ ئورۇندا ئىشلەيدىغانلار ئىچىدىكى بىردىنبىر تۇركى تىلدا سۇزلىشىدىغان مىللەتكە تەۋە كىشىدۇر، 1985-يىلىنى پۇتكۇل مەمىلىكەت بويىچە ‹‹ئەركىن سىدىق ئۇگۇنۇش ھادىسىسى ››نى كەلتۇرۇپ چىقارغان ئىكەن،(ئەركىن سىدىق ئاقسۇدىن )،،،بۇ كىشىنى تۇنۇيدىغانلاردىن زادى قانچىمىز بار؟؟؟؟
مۇشۇنداق بىزنىڭ ئۇيغۇر مىللىتىدىن ئىكەنلىكىمىزنى دۇنياغا تۇنۇتۇپ تۇرغان كىشىلەرنىڭ ئىشلىرنى نىمىشقا ئىسىمىزدە ساقلىيالمايمىز؟؟؟
بىز مەھمۇد كاشىغەرى ۋە يۇسۇپ خاس ھاجىپ بوۋىمىز قالدۇرۇپ كەتكەن دۇنياۋى تەۋەرۇك ئەسەر ‹‹ تۇركى تىللار دىۋانى ››ۋە ‹‹قۇتادقۇ بىلىك ›› تىن ئىبارەت بۇ كىتاپلارنى بىشىمىزغا ياستۇق قىلىپ يىتىپ،،،دۇنيا مەدىنىيەت مىراسلىرى قاتارىدىن ئورۇن ئالغان قىممەتلىك سەنئەت ئەنگۇشتەرىمىز ‹‹ئويغۇر ئون ئىككى مۇقامى››نى تىڭشاپ مۇگىدەپ ياتساق بۇلامۇ يا؟؟؟؟
مەنغۇ ئۇيقۇغا بەكلا ئامراق،،چۇنكى مىنىڭ قانلىق قامچىلار بىلەن ،، قاتتىق ساۋاپ تۇرۇپ ئىيتىلغان ‹‹يىقىملىق ئەللەي ناخشىسى ›› دىن قانچىلىك ھوزۇر ئالالايدىغانلىقىمنى پەقەت ۋە پەقەت ئۆزۇملا بىلىدىغان تۇرسام،،،؟
،

 چۈنكى ماڭا ئوخشىغان  ياقا يۇرتتا ئۈزۈن  مەزگىل  يات مىللەتنىڭ قېتىدا ياشىغان  بىرلىرى ،  مۇساپىرچىلىقنىڭ دەردىنى . ئۇيغۇرنىڭ بىر بىرىگە بولغان مۇھەببىتىنى سۆيگىسىنى تېخىمۇ ياخشى ھىس قىلالايدۇ دەپ ئويلايمەن .  مەن مىللەتچى ئەمەس . ئەمما ئۇيغۇرنىڭ ئاق كۈڭۈللىكىنى مەن ھازىرغىچە كۆرگەن ھېچبىر مىللەتكە تەڭ قىلمايمەن .
