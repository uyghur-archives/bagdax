ئەسسالامۇ ئەلەيكۇم ئەزىزلەر، ئالدى بىلەن يېڭى يىلىڭلارغا مۇبارەك بولسۇن. كومپىيۇتېر مەشغۇلات سېستىمىسى بولغان Windows8 گە ئۇيغۇر كومپىيۇتېر ئىشلىرىدىكى تۆھپىكارىمىز ئالىم ئەھەتنىڭ كۈچ چىقىرىشى ۋە مىھنەت قىلىشى بىلەن ئۇيغۇرچە تىل بولىقى قوشۇلۇپ ، ئۇيغۇر كومپىيۇتېر ئىشلىرىنىڭ تەرەققىياتى ئۈچۈن يېڭى بىر سەھىپە ئېچىلدى . ئۇزۇندىن بۇيان Windows مەشغۇلات سېستىمىسىنى ئانا تىلىدا ئىشلىتىشنى ئارزۇ قىلىپ كەلگەن بارلىق قېرىنداشلارغا نىسبەتەن ناھايتى زور خۇشاللىق ئىش بولغان بولسىمۇ ، زور كۆپ ساندىكى قېرىنداشلىرىمىزنىڭ تورغا چىقىش شارائىتىنىڭ يوقلىقى ياكى بۇ ئىشقا مۇناسىۋەتلىك ئۇچۇرلارنىڭ تارقىلىش دائىرىسىنىڭ چەكلىكلىكى ، شۇنداقلا پىسخىكىمىزدىكى پەرۋاسىزلىق ، بىخۇتلۇق تۈپەيلىدىن Windows مەشغۇلات سېستىمىسىنىڭ ئۇيغۇرچە بولۇپ كەتكەنلىكىدىن ئىبارەت ئالەمشۇمۇل ئىشلاردىن بىخەۋەر يۈرمەكتە . بىلسىمۇ پەرۋاسىز يۈرمەكتە . مەزكۇر سېتىما كىلەر ئاينىڭ بىرىنچى كۈنىدىن باشلاپ نەچچە مىڭدىن سېتىلىشقا باشلايدۇ، دېمەك ئەڭ ئاخىرقى 30 كۈنلۈك مۆھلەت قالدى…

‏بۇ ئىشلار مۇشۇنداقلا ئۆتۈپ كەتسىمۇ مەيلىتى . بىراق ، مىكروسوفت شېركىتىنىڭ سىتاتسكىلىق سانلىق مەلۇماتىدا ئۇيغۇرلارنىڭ ئۇيغۇرچە تىل بولىقى قوشۇلغان مەشغۇلات سېستىمىسىنى ئىشلىتىش دەرىجىسى تۆۋەن بولۇپ قالسا ، مۇنداقچە ئېيتقاندا ، بىزنىڭ بۇ ئىشقا تۇتقان پوزىتسىيەرىمىز تازا قىزغىن بولمىسا ، شۇنداقلا ، مىكروسوفتنىڭ Windows مەشغۇلات سېستىمىسىغا ئۇيغۇرچە تىل بولىقى كىرگۈزۈش ئۈچۈن سالغان مەبلىغى كىرىمى چېقىمىنى قاپلىيالمىسا كىيىنكى قېتىملىق مەشغۇلات سېستىمىسىنى ئىشلەيدىغان ۋاقىتتا ئۇيغۇر تىلىنى قۇشۇشنى ئويلاشمايدۇ - دە ، شۇنىڭ بىلەن ئۇيغۇر كومپىيۇتېر تەرەققىياتى تىبەت ، موڭغۇل ... قاتارلىق مىللەتلەرنىڭ ئارقىسىدا قېلىش خەۋپى كېلىپ چىقىدۇ . ئەلۋەتتە ، بۇ ھىچقايسىمىز كۆرۈشنى خالىمايدىغان ئاقىۋەت .
‏يەنە بىر جەھەتتىن ، ئىشەنچىلىك مەلۇماتلاردىن قارىغاندا 2013 - يىلى 2 - ئاينىڭ 1 - كۈنىدىن باشلاپ ئۇيغۇرچە تىل بولىقى قوشۇلغان بۇ مەشغۇلات سېستىمىسى تەخمىنەن 1280 يۈەنگە ئۆسۈپ كېتىش ئېھتىماللىقى ئىنتايىن چوڭ . ھالبۇكى ، 2-ئاينىڭ 1-كۈنىگە 30 كۈنلا قالدى!....

‏تىزرەك ھەرىكەتكە كېلىڭ ! پۇرسەت تىزلا ئۆتۈپ كەتمەكتە . كىيىنكى پۇشايمان ئۆزۈڭگە دۈشمەن ! ئۇيغۇر تەرەققىياتى ئۈچۈن بىر كىشىلىك ھەسسىڭىزنى قوشۇپ ، ئۇيغۇرچە مەشغۇلات سېستىمىسىدا ئانا تىلىڭىزنىڭ ھوزۇرىنى سۈرۈڭ !

مەن بۈگۈن سىناق تەرىقىسىدە پايپالدىكى 1دوللار پۇلنى قايتۇرۋالغان ئىدىم.قايتۇرۇش مۇۋەپپىقىيەتلىك بولدى دەپ پايپال ھېسابىمدىن بىر دوللار ئازلىدى،لېكىن پۇل پايپال بىلەن باغلىغان بانكا كارتامغا كىرمەپتۇ.ياكى كارتامدا دوللار بويىچە ساقلانغىنىنى تاپالمىدىم.دېمەكچىمەنكى پايپالدىكى پۇلنى قايتۇرىۋالساق كارتىمىزغا خەلق پۇلى شەكىلىدە قايىتامدۇ ياكى دوللار بويىچىمۇ ؟بىلىدىغانلا يول كۆرسەتسەڭلار.
پايپالغا قاچىلاپ بولغان بولسىڭىز ، ئۆزىڭىز سېتىۋالسىڭىزلا بولىدىكەنغۇ ، قايتۇرۋالىمەن دەپ ئولتۇرمايلا
