    چۇبابايۇۈ ئابباسۇپقا ئەتە 50 كىشلىك شەرقى تۇركىستان توغۇرلۇق ئىلمى مۇئاكىمە يىغىنى ئىچىلىدۇ دەپ ئۇختۇرىدۇ .يىغىندا روس تارىقچى ؛شەرقى تۇركىستان ئەزەلدىن جوڭگۇنىڭ زىمىنى ئەمەس . 7 _ ئەسىردىن باشلاپ روسنىڭ كونتۇرۇللىغىدا بولغان . يەنە بىر روس تارىقچى 7 _ ئەسىردە روس ئىمپىرىيىسى قۇرۇلمىغان ، روس مىللىتىمۇ ئەمدى شەكىللىنىۋاتقان تۇرسا كانداكسىگە ، شەرقى تۇركىستان روسنىڭ كونتۇرۇللىغىدا بولىدۇ . نۇرغۇن سوھبەتلەر بولىدۇ .
    بۇيى ئىگىز كەلگەن روس ، ئابباسۇپنى ھەربى مەشىققە ئورۇنلاشتۇرىدۇ ، ئايرۇپىلان بىلەن تاشكەننىڭ شەرىق جەنۇپ تەرەپتىكى ھەربى مەشىق كىلىش لاگىرىغا بارىدۇ كىچىك تىپتىكى كورالدىن تاتىپ چوڭ تۇپتىكى كوراللارنىڭ مەشىقىنى ئۇگۇنىدۇ ، ئاساسلىق ئۇگۇنىدىغىنى كوماندانلىق شىتابىدا كوماندانلىق قىلىش ، 35  كون مەشىق قىىلدۇ .
  جۇجۇبايۇف   ئابباسۇپقا موسكىۋادىن تىلگىراما كەلدى تىزدىن يولغا چىق ھەر ئىكتىمالغا قارشى 3 پىكاپ ئورونلاشتۇرىدى 2 ئوپىسەر 1 ھەربى دوختۇر بىللەن يولغا چىقىدۇ ئالمۇتۇ ئايرىدورۇم ئىستانسسادا ئابباسۇپنى ساقلاپتۇرغان ئىگىز بويلۇق روس ؛ خۇداغا شۇكرە كەلدىڭما ؟ تىز ئايرۇپىلانغا چىق . سەن يەنە كىچىككەن بولساڭ كاللامدىن ئايرىلتىم .مىنىڭ ئىسمىم ياقۇپ .
    7 تە كىشلىك سوكبەتتىكى ئوتتۇرا بويلۇق روس شۇنداق مۇلايىملىق بىلەن ئابباسۇپنىڭ ئالدىغا كىلىپ ،يولداش سىتالىن ئالدىڭىزغا ئالايىتەن مىنى ئەۋەتتى . مىنىڭ .ئىسمىم خىسلوپ . ئەسلى تاشكەنگە ئەۋەتىدىغان تىلگىرامىنى ئالمۇتاغا ئەۋەتكەن ، كىچىكىپ كەلگەن گەپ. تىلىگىراپ ئىدارىسىدىكى قايسى ھايۋاننىڭ قىلغان ئىشى دەپ ياقۇپ خاپۇلاپ كىتىدۇ بۇلار موسكىۋاغا يىقىن بىر داچىغا كىلىدۇ .ئابباسۇپنى باشتىن ئاخىرى قوغدىغۇچىلار رەھبەرلەر شۇنداق ياخشى كۇتىدۇ . ،تەسۋىرلىغۇسىز . بۇلار پاي پىتەك بۇلۇپ كىرىمىن سارىيىغا بارىدۇ . ئۇستىخانلىق بىر روس ئابباسۇپ بىلەن مۇلايىم كورىشىدۇ ئابباسۇپنىڭ پۇتۇن ئارخىپ تەرجىمالىنى تەپسىلى سوراپ بولاپ ؛سەن نىمە ئۇچۇن ئىلىخان . تورە دىننى رەھبەر بىلەن بىرلىشىسەن جاۋاپ ؛ غۇلجا كونسىلخانىسىنىڭ ئورۇن لاشتۇرىشى ۋە مۇسكىۋانىڭ ئورۇنلاشتۇرىشى سۇئال ؛بىرلىشىشتىكى مەخسىتىڭنى دەپ بىرەمسەن جاۋاپ ؛ئىنقىلاپنىڭ مەخسىدى ئىتىپاقلىشىدىغان بارلىق ئىنقىلاپچىلار بىلەن بىرلىشىش ئۇلۇق دىننى دايى ئىلىخان . تۇرى بىلەن بىرلەشسەك پۇتۇن مۇسۇمان بىلەن بىرلەشكەن بولىمىز .

يوللىغان ۋاقتى 2013-1-10 02:32:40 |ھەممە قەۋەتنى كۆرۈش
     سۇئال ؛ ئىلىخان تۇرەم قانداق ئادەم . جاۋاپ؛ شەرقىى تۇركىستاندا ئىناۋىتى شۇنداق يۇقىرى كىشى . سۇ.ئال ؛ راس دەيسەن ، ئۇ كىشى ھەقىقەتەن شەرقىى تۇركىستاننى مۇستەقىل قىلىشنى ئويلامدۇ . جاۋاپ ؛ مىنىڭچە شۇنداق . سۇئال ؛ مۇستەكىل بولغاندىن كىيىن ، ئۇ كىشى قانداق سىياسەتلەرنى يۇرگۇزۇدۇ . جاۋاپ ؛ ئۇ كىشى نەتىجە قازىنىش ئۇچۇن ھەرقانداق ئادەملەر بىلەن بىرلىشەلەيدۇ ، نەتىجە قازىنىپ بولغاندىن كىيىن ھۇز ئورۇشدىن يانمايدۇ . مەسلەن ؛ مۇخاممەت پەيغەمبىرىم قانچىلىك ئادەمتى ئۇ مىنىڭ ئوقۇغۇچىم ، قۇرانى كىتاپنى مەن يىزىپ ئۇنىڭغا بەرگەن .
     ئابباسۇپنى باشلاپ چوڭ بىر زالغا كىرىدۇ ، بۇ يەردە موللى توۈ ، مالىن كوۈ ، ياقۇپ،قاتارلىق رەبەرلەر بىلەن سوكبەت ئوتكۇزىدۇ ؛ موسكىۋاغا قاچان كەلدىڭ ؛3 كۇن بولدى 10 _ ئاينىڭ 14 _ كۇنى نەلەگە بىرىپ سايائەت قىلدىڭ ، مالىنكۇۈ گەپ قىستۇرۇپ بۇنى ئورۇنلاشتۇمىدۇق . ؛ بۇ جەھەتتە بىز ياخشى قىلماپتۇق بۇ دا كەلگەندە سىنى دولەت رەھبىرىدەك كۇتىۋالىمىز  ؛روس تىلىنى بىلەمسەن ؛ ئازىراق بىلىمەن داۋاملىق ئۇگىنىۋاتىمەن ؛ سىنىڭ ئوبۇكتاڭ ياڭ ۈىڭ يى روسچىنى بىلەمدۇ ؛ بىلمەيدۇ ؛ ئۇيغۇرچىنى بىلىدىغاندۇ ئۇنىمۇ بىلمەيدۇ . سورۇن كەيپىياتى سۇسلىشىدۇ  ؛ ھازىر ئۇرۇش مەزگىلى قىيىنچىلىق ۋاقىت شۇنداق بولسىمۇ بىز سىلەنىڭ ئىنقىلاۋىڭلانى كوللايمىز ھەمدە كورال ياراق بىرىمىز . سىنىڭ يەنە نىمە تەلىۋىڭ بار ؛ سىتالىن بىلەن كورەشكىم بار  سورۇندىكىلەر كۇلۇشۇپ كىتىدۇ . ئۇلارنىڭ ئارىسىدىن بىرسى چىقىپ سىتالىننىڭ قىياپىتىگە كىرىدۇ ئابباسۇپ چۇشەنمەي ئولتاغاندا مالىنكۇۈ ئابباسوپقا ئىشەرەت قىلىدۇ ئابباسۇو جۇگۇرەپ بىرىپ سىتالىن بىلەن كورەشگەن بولىدۇ . ياخشىمۇ سىز يولداش سىتالىن ئاخىرى سىز بىلەن كورەشتىم ،ئۇلۇق ئۇستاز ، ئۇلۇق دايى ئۇلۇق رەھبەر سىزنى بەك ھورمەتلەيمەن مۇشۇ گەپلەرنى 3 قىتىم دەيدۇ سىتالىن؛ مەنمۇ سىز بىلەن كورەشكىنىمگە خۇرسەن مەن ؛ پۇتۇن دۇنيا ئىزىلگۇچى خەلىقنىڭ ئۇستازى سىتالىن ياشىسۇن . سىتال ؛شەرقىى تۇركىستان جۇمھۇرىيەتتىكى خەلىق ياشىسۇن !
        سۇئال ؛ سەن قاچاندىن باشلاپ كولاڭغا كورال ئىلىپ گومىنداڭنى يوقىتشنى ئويلىدىڭ . جاۋاپ ؛ ماو جۇ شىنىڭ . چاقنىغان قىزىل يۇلتۇز . قاتارلىق كىتاپلارنى ئوقۇغاندىن باشلاپ،؛ سۇئال سەن ماو جۇ شىنى چوڭ بىلىدىكەنسەندە ؛ شۇنداق ھەقىقەتەن چوڭ بىلىمەن ؛ ھازىرچۇ ؛ لىنىن ، سىتالىننىڭ كىتاۋىنى ئوقىغاندىن كىيىن ئەسلىدە ماو زى دوڭ سىتالىننىڭ ئوقۇغۇچىسى ئىكەن ، موللىتوۈ ئابباسۇپقا سەن بىزنىڭ دىگەن يىرىمىزدىن چىقتىڭ سەن روس پارتىيەسىگە ئىلقىماس يازساڭ بولىدۇ سىنى كوبۇل قىلىمىز .مەن ھازىرچە ئوتمەي پىشىپ يىتىلمىدىم كەلگۇسىدە ئوتەي . ياقۇپنىڭ ئاچچىغى كىلىپ ئاناڭنى سەن بۇيەرگە نىمە دەپ كەلدىڭ . سىنىڭ تىگى تەكتىڭدىن جۇڭ گۇ خەنزۇلىرىنىڭ ئىشپىيونى كەنسەن سەن بىكاغا ئۇلارنىڭ تەرجىمالى بومماپتىكەنسەن ، ئىگىز بويلۇق روس كىرىپ ھەرھالدا چوڭ ئىش بولمىدى سىتالىن ؛ئابباسۇپنى شەرقىى تۇركۇستاننىڭ 1 قول رەھبىرى بولالمايدۇ دىدى
     ئەتىسى ئابباسۇپقا ئۇرۇش پىلانىنى ئۇختۇردى.
    1_  ئۇرۇشنى غۇلجىدىن باشلاش كورال ياراق ئادەم يوتكەشگە كولايلىق  
   2 _ ئۇرۇشنى 11 _ ئاينىڭ 4 _ كۇنىدىن بۇرۇن بۇلىشى كىرەك جۇڭ گۇ ئارمىيەسىنىڭ كۇراللىرى قەخرىتان سوغاقتا كاغا كەلمەيدۇ بىزنىڭ قۇرالىمىز ئەكسىچە .
   3 _ ئىنقىلاپنىڭ مەخسىتى بىرلەشكىلى بولىدىغان بارلىق ئىنقىلاپچىلار بىلەن بىرلىشىش ئىنقىلاپ غەلبە قىلغاندىن كىيىن دىكتاتۇر يۇر گۇزسە بولىدۇ .
   4 _ شەرقىى تۇركىستان جۇمۇرىيىتى كورۇلۇپ ئوزلىرىنىڭ سىياسىتىنى يۇرگۇزسە بولىدۇ .
   5 _ شەرقىى تۇركىستان جۇمۇرىيىتىنىڭ 1 _ قول رەھبىرى ۋاقىتلىق ئىلىخان تۇرەم بولىدۇ ، جۇمۇرىيەتنىڭ ساقچى ، ئىچكى ئىشلار ، تەشۋىقات ئىشلىرىغا ئابباسۇپ مەسۇل بولىدۇ ،
   6 _ شەرقىى تۇر كىستان خەلىق پارتىيەسىگە ئابباسۇپ مەسۇل .
     يەنە ، 7 مىڭ كىشىلىك قۇرال ياراق 2 ئەتىرەت قوراللىق خادىملارنى قوشاپ ياردەم قىلىمىز .
     ئىسقاق بىك لىنتانىت مىللى ئارمىيەنىڭ باش قۇماندانى بولىدۇ . مايۇر پىدىنوۈ روس ئوپىسەر مەسلىھەتچىلىرى بىلەن غۇلجا كونسۇل خانىدا تۇرىدۇ
       داۋامى بار .
