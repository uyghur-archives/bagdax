    دانىكۇۈ بىلەن ياكۇپ ، ئابباسۇپقا مۇسكىۋا ۋە سىتالىن سىنىڭ تۇرمۇشىڭغا كوڭۇل بۇلۇپ ساڭا لايىق تەپبىرىشگە ئىككىمىزنى مەسۇل قىلدى . ئابباسۇپ، مىنىڭ لايىق تىپىشىمگىمۇ سىتالىن ئارلىشامدىكەن دەپ ئويلايدۇ ياقۇپ ؛تاشكەن ئىنىستىتوتىنىڭ ئەلا ئوقۇغۇچىسى ئەنگىلىس ، روس ، ئۇيغۇر ، تىللىرىنى پىششىق بىلىدۇ ھەم چىرايلىق قىز ئاننا . ئۇنىڭ دادىسى ئوتتۇرا ئاسىيادىكى شاڭ جىئاڭ دەرىجىلىك ئوپىسسەر يىقىندا مارشاللىققا ئوسىدۇ سىتالىننىڭ ئەڭ يىقىن ئادىمى ؛كوڭۇلبولگىنىڭلاغا رەخمەت مىنىڭ لايىقىم بار ياڭ ۈىڭ يۇ . ياكۇپ ؛ بۇ نىمە دىگىنىڭ ياڭ ۈىڭ يۇ خەنزۇ تۇرسا سىلەنىڭ ئىنقىلاۋىڭلا خەنزۇ ھوكۇمىتىنى ئاغدۇرۇپ تاشلاش  ئۇ روسچىنى يا ئۇيغۇرچىنى بىلمىسە بىزنىڭ ئاڭلىشىمىزچە ئۇنى سەت دەيدۇ ساڭا يۇك بولمامدۇ . ئابباسۇپ ؛ مەن ئونىڭدىن ئايرىلسام بولمايدۇ . ياقۇپ ؛ سەن ھەقىقەتەن ئىپتىدايى خەنزۇ پەرەزكەنسەن .
   دانىكۇۈ بولدى تالاش تارتىش قىلمايلى سىتالىن تەستىقلىغان مۇيوم بىر تىزىملىكنى ئابباسۇپقا بىرىدۇ ؛ كورۇپ باق ، شەرقىى تۇركىستان جۇمھىرىيىتىنىڭ رەھبەرلەر  تىزىملىكى ئابباسۇپ بۇنى كورۇپ ئاچچىغى كىلىپ كوڭلى يىرىم بولىدۇ بۇتىزىملىكتىكى كىشىلەر خەنزۇ ياكى جۇڭگۇ كومۇنۇسلىرى بىلەن ئالاقىسى يوق ئۇنىڭ ئۇستىگە بىرمى خەنزۇ يوق ، روس مەسلىھەچچى خادىملىرىنىڭ ئىچىدە پارتىيەلىكلەر ناھايىتى ئاز ئاساسلىق كىشىلەر، ئىلىخان تۇرەمنىڭ ئادەملىرى. ئاچچىغى كەلسىمۇ ئابباسۇپ چاندۇرماي خەنزۇ ب جۇڭگۇ كومۇنۇسلىرى بىلەن ئالاقىسى يوق كىشىلەركەن دەپ جاۋاپ بىرىدۇ .  
     كەش بولغاندا ئابباسۇپ بىلەن ئاننانى بىر ئويگە سولاپ قويىدۇ . ياقۇپ ئىتتىگەندە ئويگە كىرسە ئاننا كارۋاتتا ياتقان ئابباسۇپ كىرىسلودا ياتقان ياقۇپ ؛ ۋاي ئەخمەقلا دەپ ئابباسۇپنى باشلاپ چىقىپ كىتىدۇ . بۇلار ئىسقاق.بىك بىلەن كورۇشۇپ ئۇرۇش پىلانىنى تۇزۇپ كورال ياراقنى ئىلىپ جۇڭگۇ چىگىرىسىغا قاراپ يولغا چىقىدۇ . ئاننا ،ئابباسۇپنى چىگىرا بويىغىچە ئۇزۇتۇپ كىلىپ ئابباسۇپقا قاراپ مەن سىزنى ياخشى كورىمەن سىزنى مەڭگۇ ساقلايمەن قاچان كەل دىسىڭىز شۇ ۋاقىتتا قىشىڭىزغا كىلىمەن ، سىزدىن باشقا ھىچكىم بىلەن توي قىلمايمەن .
     داۋامى بار .

      1944 _ يىلى 10 _ ئاينىڭ 21 _ كۇنى بىرىنچى تۇركۇمدىكى كۇرال ياراقنى چىگىرىدىن ئوتۇپ ، غىنى . باتۇرغا تاپشۇرۇپ بىرىدۇ .
     1944 _يىلى 11 _ ئاينىڭ 6 _كۇنى ئابباسۇپ قوراللىق پاتىزانلانى باشلاپ غۇلجا شەھەرگە ھۇجۇم قىلىدۇ . غۇلجا شەھەردە گومىنداڭنىڭ بىر دىۋىزىيە ئەسكىرى بۇلۇپ . ھەرەمباغ ۋە بۇت خانىدىن باشقا جايلارنى ئازات قىلىپ .شەرقىى تۇركۇستان جۇمھۇرىيىتى قۇرۇلىدۇ بۇ ئۇرۇشتا ئاز_تولا خاتالىق يۇز بىرىدۇ بىر قىسىم خەنزۇ پۇخرالىرىمۇ ئولۇپ كىتىدۇ .
   ھەرەمباغ ئازات بۇلۇپ 5كۇندىن كىيىن ئابباسۇپ ياڭ ۈىڭ يىنىڭ ئويىگە بارىدۇ . ئابباسۇپ شۇمەزگىلدە نۇرغۇن خەنزۇلارنى قۇتقۇزۇپ قالىدۇ .ئابباسۇپ ياڭ ۈىڭ يى بىلەن كورۇشكەندە ، سىلەنىڭ بۇ ئىنقىلاۋىڭلانى بىزمۇ قوللىغانتۇق خەلىق پەرۋەر دەپ ، بىگۇنا خەنزۇ پۇخرالىرىنى ئولتىرىۋەتتىڭلاغۇ . ئابباسۇپ ؛ئىنقىلاپتا خاتالىق يۇز بىرىدۇ بەزەن ئىشلانى مۇتلەق دىگىلى بولمايدۇ مەسلەن ئارىمىزگە كىرىۋالغان غىنى.باتۇر ئاسىتۇۈغا ئوخشاش ئادەملەمۇ بار .
  ياڭ ۈىڭ يى يۇگۇرەپ بىرىپ ئابباسۇپنىڭ تاپانچىسىنى تارتىۋىلىپ ئۇزىنىڭ مىڭىسىغا تەڭلەيدۇ . ئابباسۇپ ؛قىززىق قانلىق قىلما مەن چۇشەندۇرەي ، ياڭ ۈىڭ يى ؛ ئاغزىڭنى زۇم كومۇنۇس ئىنتىرناتسىونالنى ئەمەلگە ئاشۇرۇپ ، پۇتۇن جۇڭگۇ خەلقىنى ئازات قىلىمىز دەپ ۋەدە قىلغانتىڭغۇ . دەپلا ئۇزىنى ئىتىۋالىدۇ .ئابباسۇپ ئۇلگۇرەممەي قالىدۇ . ئابباسۇپنىڭ كوڭلى ناھايىتى يىرىم بولىدۇ بۇ ئىشقا ئاسىىتۇف قاتارلىقلا سەۋەپچى .
    ئابباسۇپنىڭ ئاپىسى بالام كوڭلىڭنى يىرىم قىلما ياڭ ۈىڭ يى بولسا ياخشى كىلىن بولاتتى . باشقا خەنزۇ قىزدىن بىرنى ئالساڭ بولدىغۇ دەپ ئابباسۇپنىڭ كوڭلىنى ياسايدۇ .
    ئابباسۇپ جەنۇبى شىنجاڭغا يۇرۇش قىلىش ئارمىيەنىڭ باش قوماندانى بولۇپ،باي ، ئاقسۇ كونا شەھەردىكى جەڭلەگە قاتنىشىدۇ .  
  1945 _ يىلى 10 _ئايدا گومىنداڭ ھوكىمىتى جاڭجىجوڭ ئەپەندىنىڭ شىنجاڭغا كىلىشى .روسنىڭ ئارلىشى بىلەن ئۇرۇش توختايدۇ . ئابباسۇپ جەنۇبى شىنجاڭدىن قايتىپ كىلىپ .ئۇرۇش ئەھۋاللىرىنى ئىلىخان. تۇرەمگە دوكىلات قىلىدۇ . .ئىلىخان . تۇرەم ئاڭلىسام سەن لۇسۇشىڭ بىلەن توي قىلماقچى بوپسەن بىرىنچى ئالماقچى بولغان ئايالىڭ خەنزۇ ئۇ ئولدى يەنە خەنزۇنى ئالماقچىمۇ ؟ ئالدىنقى سەپكە ماڭغاندا نىمە ئۇچۇى مۇيىم نەرسىلەرنى لۇسۇشىڭغا بىرىسەن ئاپاڭغا ياكى .ئىنىڭغا بەرمەي . ئابباسۇپ ؛ شۇنداق قىلدىم  ئىلىخان. تۇرەم خاپۇلاپ ؛ مەن سىنىڭ چوڭ داداڭ بىلەن تەڭ كىلىمەن ، دوسلۇق مۇناسىۋىتى بىلەن ساڭا كۇيۇنۇپ گەپ قىلىۋاتىمەن قانداق بىر سىرلىق كۇچ سىنى بۇيولغا باشلاۋاتىدۇ ھىچ چۇشىنەلمىدىم . ئابباسۇپ ؛ مىنىڭ ئايال تاللاش ھوقۇقۇن يوقما . چىرايلىق ئۇيغۇر قىزلىرى بار شۇنى تاللىساڭ بولمامدۇ. پاسكىنا چوشقىنى تاللىغىچە . سىنى ئاددى بىر پۇقرادىن گىنىراللىققا كىم ئوستەدى ئۇنتۇپ قالما. ئابباسۇپ ؛ مىنىڭ ئالغىنىم ئالغان. ئابباسۇپ لۇ سۇشىڭ بىلەن كورىشىپ ئولەپكەتكەن، لۇ سۇ شىڭنىڭ ئاپىسىنىڭ رەسىمىگە تازىم قىلىدۇ
    1945 _ يىلى 12 _ ئاي  غىنى باتۇر بىلەن ، ئاسىتۇۈ ئابباسۇپنىڭ ئويىگەكىلىپ ؛بىز ساڭا ئەلچىلىككە كەلدۇق ئىنقىلاپ غەلبە قىلدى ، ئەمدى سەن لايىقتىن بىرنى تاپساڭ بولىدۇ ، ئىلىخان تۇرەم يۇقىرى قاتلامدىكى كىشىلەر ساڭا كۇڭۇل بولىۋاتىدۇ؛ ئابباسۇپ  ؛سىلەنىڭ ئوياڭلادا كىمنى ئالسام بولا ؛ ئۇيغۇرلانىڭ ئىچىدە ئەڭ چىرايلىق پەرىزات مىلىيونور قىز ئابباسۇپ ؛ ھوسەن باينىڭ قىزىمۇ ؛ شەرقى تۇركىستاندا ئەڭ چوڭ باينىڭ قىزىئابباسۇپ ؛ بۇ سىلەنىڭ ئوياڭلامۇ ياكى ھۇسەن باي ،قىزنىڭ ئويىمۇ ؛ ھەممىمىزنىڭ ئورتاق پىكىرى . ئابباسۇپ گىلەمگە تۇكىرىدۇ . غىنى باتۇر ؛ نىمە قىلماقچىسەن ئاسىتۇۈ گەپ قىستۇرۇپ ؛ سىنىڭ بالا ساۋاقتىشىڭ تۇرسا ئابباسۇپ ؛ ئۇنى ھەرگىز ئالمايمەن ؛ نىمىشقا ئابباسۇپ ؛ شۇ چاغلاردا مەن ئۇنىڭدىن ئەھۋال سورىسام ،ئۇ يوقال قەلەندەر بىرىپ ئىشەكنىڭ سۇدىكىنى .ئىچە دىگەن ، ئۇ دىگەن بالىلىق ۋاقىت ھازى ساڭا يال بىرىپ كەلدىغۇ .ئابباسۇپ ئۇنىمىدى ؛ خىتتاي قىزۋالا سىنى ئوقىتىۋالدىمۇ يا سىنىڭ باشقا مەخسىتىڭ بارمۇ ئابباسۇپ ؛ مىنىڭ مەخسىتىم چوقوم ئالىمەن . غىنى باتۇر ھەددىڭدىن ئىشىپ كەتمە ئابباسۇپ ؛ئىلىخان تۇرەم نىمە قىلالايدۇ ۋاقتى كەلگەندە ئىلىخان تۇرەمنىڭ ماشىنىسى بىلەن توي يوتكەيمەن
