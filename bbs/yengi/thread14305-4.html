   بۇرۇن دەرىسلىك كىتابلاردىن « يېرىم كېچىدە چىللىغان توخۇ »ناملىق بىر تېكىسىتنى ئوقۇغانىدىم.تېكىسىتتە يىللىقچىلارنى كۆپرەك ئىشلىتىپ بايلىق شىلىۋېلىش مەخسىدىدە پوممىشىكنىڭ ھەر كۈنى يېرىم كېچىدە توخۇ كاتىكىنىڭ يېنىغا بېرىپ توخۇنى دوراپ چىللايدىغانلىقى،يىللىقچىلارنىڭ ئالمان-تالمان ئورنىدىن تۇرۇپ ئېتىزغايۈگرەيدىغانلىقى يېزىلغانىدى.شۇ چاغدا ئوقۇتقۇچىمىز پومىششىكنىڭ ئادەم قېلىپىدىن چىققان رەھىمسىز، ئاچكۆزلۈكىنى بولۇشىغا تىللاپ بەرگەنىدى.

   ئەمدىلىكتە ئويلاپ باقسام گۇناھ پومىشىكتىلا ئەمەس توخۇنىڭ يېرىم كېچىدە چىللىمايدىغانلىقىدىن ئىبارەت ئەقەللىي قانۇنىيەتنىمۇ بىلمىگەن دۆت، كالۋا يىللىقچىلاردىمۇ يەتكۈچە باركەن.

   قارىغاندا ئىنسان قانچىكى نادان قېلىپ ، قەلبىنى كىر بېسىپ كەتسە ئەپسانىگە شۇنچە چوقۇنىدىغان ئوخشايدۇ. بۇنى پۇرسەت بىلگەن ھاكىمىيەت بېشىدىكىلەر توختىماي ئەپسانە توقۇپ، پۇخرالىرىنى كولدۇرلاتسا كېرەك.

چوڭ ئاچامنىڭ كۈيۈسى مەلۇم ناھىيىنىڭ تەشۋىقات بۆلۈمىدە ئىشلەيتى. ئىختىيارى مۇخبىر دەپ قويسا بەك خوش بولۇپ كېتەتتى. ئىختىيارىي مۇخبىرلىق ئۈچۈنلا يارىتىلغاندەك ئادەم ئىدى، خەۋەر يېزىشقا ئالامەت ئامراق ئادەم ئىدى. بىر كۈنى ئاپاملارنى ئۆيىگە قولىدا گېزىت خوشال كىرىپ كەلدى. مېنى كۆرۈپ بىر پارچە سۈرەتلىك خەۋەرنى چىقىرىۋالغانلىقىنى دېگەچ گېزىتنى ماڭا بەردى. كۆڭلىگە يارىشا ئېلىپ قارىسام، «پەن- تېخنىكىغا تايىنىپ بىيىش ماھىرى»دېگەن ماۋزۇ ئاستىدا سېمىز، ھاپا قۇۋۇز بىر دېھقاننىڭ سۈرىتى تۇرىدۇ. ماقالىسى يېرىم بەت يېزىلىپتۇ. بۇ ئادەم ماڭا تونۇشتەك بىلىنىپ« بۇ ئادەم بىزنى ھېلىقى چاغدا مېھمان قىلغان كىشى شۇغۇ» دېسەم« شۇ »دېدى. رەسىمگە سىنچىلاپ قارىسام بۇ كىشىنىڭ قولىدا بىر كىتاب تۇرىدۇ، لىكىن كىتابنىڭ ئاستىنى ئۈستىگە، ئۈستىنى ئاستىغا قىلىپ تۇتۇپ تۇرۇپتۇ. كۈلۈپ كەتسەم، يىزنەم سەۋەبىنى سورىدى. رەسىمنى كۆرسىتىپ قويسام، مۇخبىرلىقتا بۇنداق ئىشلار بولۇپ تۇرىدۇ دېدى. كىيىن بىر ئىش بىلەن شۇ تەرەپكە ئۆتۈپ، ھېلىقى كۈلكىلىك رەسىم ئىسىمگە كەلدى. سۈرۈشتۈرۈپ ھېلىقى ماھىرنىڭ ئىنىسىنى تاپتىم. ئاكا- ئۇكا ئەپ ئۆتمەيدىكەن. سەۋەپ ئاكىسى ھازىر كەنت مۇدىرى ئىكەن، بىيىپ كېتىپتۇ، بۇنىڭدا ئىنىسىنىڭ تۆھپىسى چوڭكەن، چۈنكى ئىنىسى يېزىدىن نەخ مەيدان يىغىنى ئاچىدىغان كۈنى ئاخشىمى خاماندىكى چەشنىڭ ئۈستىگە ئۆزىنىڭ خامىنىدىن 2 تاغار داننى ھېچكىمگە تۇردۇرماي ئەكىلىپ قوشۇپ قويۇپتىكەن. كىيىن ئاكىسى پەن- تېخنىكىغا تايىنىپ بىيىش ئۆلگىسى بوپتۇ. لىكىن ئىنىسى ماڭا تۈزۈك ياردەم قىلمىدى دەپ ئاكىسىنى تىللاپ يۈرىدىكەن. شۇڭا گەپ كوچىلىسام «بۇ ساۋادسىز ئوغرى كىتابمۇ ئوقۇيالمايدۇ، ئىمزامۇ قويالمايدۇ، يىنىغا ئېسىۋالغان بىر تامغىسى بار شۇنى باسىدۇ» دەپ غەيۋەت قىلىپ بەردى. شۇ چاغدىلا  پارتىكومنىڭ ئاۋازى بولغان گېزىتتىكى سۈرەتتە كىتابنىڭ تەتۈر تۇتۇلىشىنىڭ سەۋەبىنى چۈشەندىم.

چوڭ ئاچامنىڭ كۈيۈسى مەلۇم ناھىيىنىڭ تەشۋىقات بۆلۈمىدە ئىشلەيتى. ئىختىيارى مۇخبىر دەپ قويسا بەك خوش بولۇپ كېتەتتى. ئىختىيارىي مۇخبىرلىق ئۈچۈنلا يارىتىلغاندەك ئادەم ئىدى، خەۋەر يېزىشقا ئالامەت ئامراق ئادەم ئىدى. بىر كۈنى ئاپاملارنى ئۆيىگە قولىدا گېزىت خوشال كىرىپ كەلدى. مېنى كۆرۈپ بىر پارچە سۈرەتلىك خەۋەرنى چىقىرىۋالغانلىقىنى دېگەچ گېزىتنى ماڭا بەردى. كۆڭلىگە يارىشا ئېلىپ قارىسام، «پەن- تېخنىكىغا تايىنىپ بىيىش ماھىرى»دېگەن ماۋزۇ ئاستىدا سېمىز، ھاپا قۇۋۇز بىر دېھقاننىڭ سۈرىتى تۇرىدۇ. ماقالىسى يېرىم بەت يېزىلىپتۇ. بۇ ئادەم ماڭا تونۇشتەك بىلىنىپ« بۇ ئادەم بىزنى ھېلىقى چاغدا مېھمان قىلغان كىشى شۇغۇ» دېسەم« شۇ »دېدى. رەسىمگە سىنچىلاپ قارىسام بۇ كىشىنىڭ قولىدا بىر كىتاب تۇرىدۇ، لىكىن كىتابنىڭ ئاستىنى ئۈستىگە، ئۈستىنى ئاستىغا قىلىپ تۇتۇپ تۇرۇپتۇ. كۈلۈپ كەتسەم، يىزنەم سەۋەبىنى سورىدى. رەسىمنى كۆرسىتىپ قويسام، مۇخبىرلىقتا بۇنداق ئىشلار بولۇپ تۇرىدۇ دېدى. كىيىن بىر ئىش بىلەن شۇ تەرەپكە ئۆتۈپ، ھېلىقى كۈلكىلىك رەسىم ئىسىمگە كەلدى. سۈرۈشتۈرۈپ ھېلىقى ماھىرنىڭ ئىنىسىنى تاپتىم. ئاكا- ئۇكا ئەپ ئۆتمەيدىكەن. سەۋەپ ئاكىسى ھازىر كەنت مۇدىرى ئىكەن، بىيىپ كېتىپتۇ، بۇنىڭدا ئىنىسىنىڭ تۆھپىسى چوڭكەن، چۈنكى ئىنىسى يېزىدىن نەخ مەيدان يىغىنى ئاچىدىغان كۈنى ئاخشىمى خاماندىكى چەشنىڭ ئۈستىگە ئۆزىنىڭ خامىنىدىن 2 تاغار داننى ھېچكىمگە تۇردۇرماي ئەكىلىپ قوشۇپ قويۇپتىكەن. كىيىن ئاكىسى پەن- تېخنىكىغا تايىنىپ بىيىش ئۆلگىسى بوپتۇ. لىكىن ئىنىسى ماڭا تۈزۈك ياردەم قىلمىدى دەپ ئاكىسىنى تىللاپ يۈرىدىكەن. شۇڭا گەپ كوچىلىسام «بۇ ساۋادسىز ئوغرى كىتابمۇ ئوقۇيالمايدۇ، ئىمزامۇ قويالمايدۇ، يىنىغا ئېسىۋالغان بىر تامغىسى بار شۇنى باسىدۇ» دەپ غەيۋەت قىلىپ بەردى. شۇ چاغدىلا  پارتىكومنىڭ ئاۋازى بولغان گېزىتتىكى سۈرەتتە كىتابنىڭ تەتۈر تۇتۇلىشىنىڭ سەۋەبىنى چۈشەندىم.
