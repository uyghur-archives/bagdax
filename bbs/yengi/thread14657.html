    نەچچە كۈن بۇرۇن مەلۇم ئىش بىلەن شىنجاڭ ئۇنۋېرستىتى فىلولوگىيە ئىنىستوتىنىڭ مۇدىرى ،تىلشۇناس ، ئالىم ، ئارسلان ئابدۇللا تەھۇر ئەپەندىمنى شىنجاڭ ئۇنۋېرستتىغا ئىزدەپ بېرىپ مۇئەللىمنىڭ بىر سائەتلىك دەرسىنى ئاڭلاش شەرىپىگە  ئېرىشتىم، مەن بارغان چاغدا ئارسلان مۇئەللىم ئاسپىرانت ئوقۇغۇچىلارغا ئىشخانىسىدا دەرس ئۆتىۋاتقان ئىكەن ، مېنى بىردەم ساقلىشىمنى ئېيتتى ،ئەزەلدىنلا ئارسلان مۇئەللىمگە ھۆرمۈتۈم چوڭقۇر ئىدى.بۇ قېتىمقى پۇرسەتتە قىسقىغىنە بىر سائەتلىك دەرس ئاڭلاش جەريانىدا ئارسلان مۇئەللىمنىڭ دەرس ئۆتىۋاتقان چاغدىكى خۇشخۇيلىقى، ۋە دەرس ئۆتۈش مىتوتىتىغا ھەقىقەتەن قايىل بولدۇم ۋە ھۆرمۈتۈم تېخىمۇ ئاشتى .    
  مېنىڭ كەسپىم ئەدەبىيات ئەمەس شۇنداقلا بۇ ئاسپىرانتلارغا سۆزلىنىۋاتقان بىر قەدەر كەسپىي دەرس بولسىمۇ مېنى تولىمۇ قىزىقتۇردى . بۇ دەرس ئۇيغۇر شېئىرىيتىدىكى ئوخشۇتۇش توغۇرلۇق سۆزلىنىۋاتقان دەرس ئىكەن ، ئارسلان مۇئەللىم ئوخشالغۇچى توغۇرلۇق سۆزلەپ كېلىپ شېئىرنىڭ بىرنچى مىسراسىدا  شەيئى مەلۇم نەرسىگە ئوخشالغاندىن كېيىن ئاندىن كەينى مىسرادىكى ئوخشالغۇچى ئالدىنقى مىسرادىكى ئوخشالغۇچىغا قارىتا ئاندىن ئوخشىشىپ كېلىدۇ . ئەگەر ئالدىنقى مىسرادىكى ئوخشالغۇچى بومىسا كەينى مىسرادا ھېچقانداق مەنە چىقمايدۇ  .
    خەنزۇلارنىڭ داڭلىق يازغۇچىسى گومۇرو يەنە بىر داڭلىق يازغۇچىغا لۈشۈن جۇڭگو جەمئىيتى ئۈچۈن بىر ئۆمۈر كالا ئورنىدا خىزمەت قىلدى ، مەن بۇ ئۆمرۈمدە شۇ كالىنىڭ قۇيرۇقى بولۇشنى ئارزۇ قىلىمەن دىگەندە يەنە بىر يازغۇچى ئەگەر سىز كالىنىڭ قۇيرىقى بولسىڭىز مەن ئاشۇ قۇيرۇقتىكى بىر تال تۈك بولۇشنى ئارزۇ قىلىمەن . دىگەن  . ئەگەر دەسلەپتىكى لۈشۈن كالىغا ئوخشىتىىلمىغاندا بىۋاستە ھالدا گومورۇي مەن كالىنىڭ قۇيرۇقى دىسە بۇنىڭدا ھېچقانداق مەنە چىقمايدۇ ، دەپ ناھايىتى ياخشى چۈشەندۈردى ، دەل مۇشۇخىل چۈشەندۇرۇش مېنىڭغا چوڭقۇر ئەسلىمە قالدۇردى . 
      بۇ سائەتلىك دەرسنى ئاڭلىغاندىن كېيىنكى نەچچە كۈندە كۆڭلۈمدە نۇرغۇن خىياللار كەچتى ، توغرا لۇشۈن ھەقىقەتەن جۇڭگو جەمئىيىتى ئۈچۈن كالا بولدى ،گومۇرو شۇ كالىنىڭ قۇيرۇقى بولۇشنى تاللىدى ، يەنە بىر قىسىملار كالىنىڭ قۇيرۇقىدىكى تۈك بولۇشنى ئارزۇ قىلدى .
   ئۇنداقتا ئۇيغۇر جەمئىيتىدىكى كالا ئوبرازى كىم ، كالىنىڭ قۇيرۇقىچۇ ؟ كالىنىڭ قۇيرۇقىدىكى تۈك بولۇشنى كىملار خالايدۇ، بۇ بىر ئويلىنىشقا تېگىشلىك مەسىلە ؟


  ئۆزتۈرۈك نامىمدىكى ھەرقانداق تېما ۋە ئىنكاس باغداش  تورىغا خاس ، ھەرقانداق ئورۇنغا يۆتكەپ ئىشلەتمەڭ !

بىزگە مۇشۇنداق ئىتتىپاقلىشالايدىغان،ھەمكارلىشالايدىغان،بىر-بىرىمىزنى ئېتىراپ قىلىپ قوللاپ ئەتىۋارلايدىغان ،ئىتتىپاقلىشىپ بىر ياقىدىن باش چىقىرالايدىغان،بىر -بىرىنى سىغدۇرالايدىغان، ئومۇمنىڭ مەنپەئەتى ،ئۆملۈك -بىرلىك ئۈچۈن شەخسى مەنپەئىتىدىن ۋاز كېچىپ ئومۇم ئۆچۇن خىزمەت قىلىشنى خالايدىغان ئۆزىنى بېغىشلاش روھى  ،بېشىنى باشلاپ بەرسە ،ئىزىدىن مېڭىپ  داۋاملاشتۇرالايدىغان روھ  ناھايتى لازىم، لېكىن بىزدە ھازىر يەنىلا قېيداش  پىسخىكىسى،بىر-بىرىنى ئېتىراپ قىلماسلىق، بىر-بىرى بىلەن چىقىشالماسلىق،بىر-بىرىنى تىللاش مەسىلىسى مەۋجۇت، ئۈرۈمچىدە ساۋاقداشلار خەلقئارا چوڭ بازاردا ئۇچراشماقچى بولۇپ ئالتىمىز شەھەرنىڭ ئالتە بۇرجىكىدىن ئالتە يول بىلەن مېڭىپ بازاردا ئۇچراشتۇق، تاكسى شوپۇرىمۇ مەندىن قايسى يول بىلەن ماڭاي دەپ سوراپ تۇرىدۇ،مەن ھازىر قايسى يول راۋان بولسا شۇ  يول بىلەن ماڭغىن دەيمەن، يولنى بىر نەچچە يول ئىچىدىن تاللاپ ئوخشاش بىر مەنزىلگە بارغىلى بولىدۇ، يولنى تېز  ماڭغىلىمۇ بولىدۇ،ئاستا ماڭغىلىمۇ بولىدۇ، بەرىبىر بارماقچى بولغان مەنزىلگە بارىمىز.
