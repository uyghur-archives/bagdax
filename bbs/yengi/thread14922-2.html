    داۋامىنى كېچىكتۈرۈپ قويدۇم ھەم ئىنكاس شەكلىدە يازغانلىقىم ئۈچۈن ماۋزۇلارنى چوڭايتالمىدىم، شۇڭا تورداشلاردىن كەچۈرۈم سورايمەن ھەم ئىنكاس يازغاندا ھەرخىل بەلگە ۋە خەت شەكلىنى چوڭايتىش، رەڭلەش قاتارلىق مەشغۇلاتلارنى قىلغىلى بولسا ئۈگىتىپ قويۇشىڭىزلارنى سەمىمىي ئۈمىد قىلىمەن.

                           1. تاڭبالى يايلىقى           (2-سائەت)
                                       ئابدۇۋەلى مۇقىيىت

      ئوقۇتۇش نىشانى:
1. تېكىستنى راۋان ، ھېسسىياتلىق ئوقۇپ ، نادىر جۈملىلەرنىڭ مەنىسىنى چۈشىنىش.
2. ئوقۇغۇچىلارغا يايلاقنىڭ گۈزەل مەنزىرىسىنى ۋە ئۆزگىچە مىللىي ئالاھىدىلىكىنى ، شۇنداقلا ئۇيغۇر، قازاق مىللەتلىرىنىڭ چوڭقۇر دوستلۇقىنى ھېس قىلدۇرۇش، غەربىي دىيارنى ئەتراپلىق چۈشىنىشكە بولغان قىزىقىشىنى قوزغاش.
3. تېكىستتىن تەسۋىرىي جۈملىلەرنى تېپىپ مەنىسىنى چۈشىنىۋېلىش ۋە كۆچۈرۈپ يېزىۋېلىش. تېكىستتىكى ئىپادىلەش ئالاھىدىلىكىنى ھېس قىلىش، ئاپتورنىڭ ھېسسىياتىنى ئىپادىلەش ئۇسۇلىنى ئۆگىنىش.


        ئوقۇتۇش جەريانى:

دەرسكە باشلاپ كىرىش:
ئوقۇغۇچىلارنىڭ تاڭبالى يايلىقى، قازاق خەلقىنىڭ تۇرمۇش ئادىتى توغرىسىدا توپلاپ كەلگەن ماتېرىياللىرىنى ئوتتۇرىغا قويۇش ئارقىلىق دەرس باشلىنىدۇ.

    ئوقۇتقۇچى: تاڭبالى يايلىقى ئىلى قازاق ئاپتونۇم ئوبلاستىنىڭ نىلقا ناھىيىسىدە ، ئۇ قاش دەرياسىنىڭ يوقىرى ئېقىن  ۋادىسىغا جايلاشقان بولۇپ ، پايانسىز يايلىقى، شىددەت بىلەن ئاقىدىغان ئېقىن-جىرالىرى، سان-ساناقسىز جىلغا، ئارىشاڭ، قىيالىرى، قويۇق قارىغايلىرى، توپ-توپ قوي-كالىلىرى بىلەن مەشھۇر، تەبىئىيلىكى قوغدىلىدىغان رايوندۇر. << تاڭبالى >> دېگەن سۆز قازاق تىلىدا << چوڭ مۆھۈر>> دېگەن مەنىنى بىلدۈرىدۇ. بۇ يايلاقتىن خەت ۋە مۆھۈر ئويۇلغان نۇرغۇن تاشلار تېپىلغان، شۇڭا بۇ يايلاق << تاڭبالى>> دەپ ئاتالغان. (دوسكىغا يېزىلىدۇ)

ئوقۇتقۇچى: ساۋاقداشلار، توپلاپ كەلگەن ماتېرىياللىرىڭلاردىن يايلاقنىڭ گۈزەل مەنزىرىسى ۋە قازاق خەلقىنىڭ تۇرمۇش ئادەتلىرى بىلەن خېلىلا پىششىق تونۇشۇپ قالدۇق. ئەمىسە بىز بۇ تېكىستنى يەنە بىر قېتىم ئوقۇپ، ئوبرازلىق جۈملىلەرنى تېپىپ تەھلىل قىلىپ باقايلى. ( ئالدىنقى سائەتتە تاللانغان << ھېسسىياتلىق ئوقۇش ماھىرى>> لىرىغا ئوقۇتۇلىدۇ)
1. << يايلاقنىڭ گۈل-گىياھلىرىدىن كېلىۋاتقان خۇش ھىدلار مېنى ئېيتقۇسىز لەززەتكە چۆمدۈردى>>____ مەنىسى:  ئاپتورنىڭ يايلاقنىڭ گۈزەل ۋە ساپ ھاۋاسىدىن ئىنتايىن ھوزۇرلانغانلىقىنى بىلدۈرىدۈ.
2. ئېكراندىن ياپيېشىل يايلاق ۋە قاتار كەتكەن كېگىز ئۆيلەرنىڭ سۈرىتى كۆرسىتىلىپ سورىلىدۇ:
ساۋاقداشلار، تېكىسىتتىن  سۈرەتتىكى كۆرۈنۈش تەسۋىرلەنگەن جۈملىنى تېپىپ بېرەلەمسىلەر؟

ئوقۇغۇچى: يايلاقتىكى ئاق ئوتاۋلار خۇددى چەكسىز كەتكەن تىنىق كۆل ئۈستىگە قونغان ئاققۇلاردەك چېچىلغانىدى.

ئوقۇتقۇچى: ئاپتور يايلاقتىكى كېگىز ئۆيلەرنى كۆلدىكى ئاققۇلارغا ئوخشىتىپ تەسۋىرلەش ئارقىلىق بۇ مەنزىرىنى كۆز ئالدىمىزدا ئېنىق گەۋدىلەندۈرۈپ  بېرىپتۇ، سىلەر يەنە قانداق نەرسىلەرگە ئوخشىتالايسىلەر؟

ئوقۇغۇچى: ....يېشىل مەخمەلگە كەشتىلەنگەن ئاق گۈللەردەك...

<< ئورمانلىرى قويۇق، چۆپلىرى بولۇق، خۇشپۇراق گۈللىرى بەرق ئۇرۇپ ئېچىلغان، بۇلاقلىرى مۆلدۈرلەپ تۇرغان بۇ كەڭ زېمىن داڭلىق تاڭبالى يايلىقى ئىدى>> ، << كۈندۈزى ئاجايىپ گۈزەل كۆرۈنىدىغان تاغلار كېچىدە يانپاشلاپ ياتقان غايەت زور مەخلۇققا ئوخشاپ قالىدىكەن>> ، << ئىككى ئىت قاۋىغىنىچە كىرىچتىن ئايرىلغان ئوقتەك بىز تەرەپكە قاراپ يۈگۈردى >>... قاتارلىق جۈملىلەر تەھلىل قىلدۇرىلىدۇ.

ئوقۇتقۇچى: تېكىستنىڭ يېزىلىش ئالاھىدىلىكىدىن نېمىلەرنى بايقىدىڭىز؟

ئوقۇغۇچى: 1. بۇ تېكىست ۋەقەلىك تەرتىپى بويىچە يەنى تەرتىپلىك باياندا يېزىلغان . 2. تېكىستنىڭ تىلى ناھايىتى گۈزەل بولۇپ ، مول ئوبرازچانلىققا ۋە تەسۋىرىيچانلىققا ئىگە.
3. ھېسسىيات بىلەن مەنزىرىنى كىرىشتۈرۈپ يازغان، بۇ تېكىستنىڭ ئاساسىي يېزىلىش ئالاھىدىلىكى . 4. ئاپتور ئويلىغىنىنى بىۋاستە ئىپادىلىگەن. 5. يايلاق مەنزىرىسىنى ئىنچىكە تەسۋىرلىگەن.

ئوقۇتقۇچى: ئۆزىڭىز ئىلگىرى مەنزىرە تەسۋىرلەپ يازغان ماقالىلىرىڭىز بىلەن بۇ تېكىستنى سېلىشتۇرۇپ بېقىڭ، قايسى تەرەپلىرىگە ئەھمىيەت بەرسىڭىز بولغۇدەك؟

ئوقۇغۇچى: ماقالىنىڭ باشلىنىشى ، ئاخىرلىشىشى ۋە ئوبرازلىق تىللارنى ئىشلىتىشكە ئەھمىيەت بەرسەم بولغۇدەك.

ئوقۇتقۇچى:  تېكىستتىكى << مەن >> ئائىلىسى بىلەن << مۇقان ئاكا >> ئائىلىسىنىڭ دوستلۇقى قانداق دوستلۇقكەن؟ بۇنىڭدىن نېمىنى ھېس قىلدىڭىز؟

ئوقۇغۇچى: ئۇلارنىڭ دوستلۇقى ئۈچ ئەۋلادتىن بۇيان داۋاملىشىپ كەلگەن ، چوڭقۇر مېھرى-مۇھەببەت سىڭگەن دوستلۇق. بۇنىڭدىن شىنجاڭدا ياشاۋاتقان ھەرمىللەت خەلقىنىڭ بولۇپمۇ ئۇيغۇر، قازاق مىللەتلىرىنىڭ ناھايىتىمۇ ئىجىل-ئىناق ئىكەنلىكىنى ھېس قىلدىم.

ئوقۇتقۇچى : بۇ تېكىستتىن قانداق تەسىرات ئالدىڭىز؟

ئوقۇغۇچى: يۇرتىمىز شىنجاڭنىڭ گۈزەل مەنزىرىلىرى ناھايىتى كۆپكەن، مەسىلەن ئالدىنقى يىللىقلاردا ئىلى دەرياسى، تەكلىماكان قۇملۇقى، ئالتاي تاغلىرى قاتارلىق گۈزەل ھەم ئۆزگىچە مەنزىرىلەر تەسۋىرلەنگەن تېكىستلەرنى ئۈگەنگەن ئىدۇق، بۈگۈن يايلاق مەنزىرىسىنى ۋە يايلاقتا ياشايدىغان خەلقلەرنىڭ تۇرمۇش ئادەتلىرىنى ئۆگىنىپ ناھايىتىمۇ ھاياجانلاندىم. چۈنكى يۇرتىمىز شىنجاڭ ھەقىقەتەنمۇ گۈزەلكەن، كۆرۈپ باقمىغانلارنى بىر ئاڭلاش بىلەنلا ئۆزىگە جەلب قىلىۋالىدىكەن، مەن موشۇنداق بىر گۈزەل يۇرتتا تۇغۇلۇپ ئۆسكەنلىكىم ئۈچۈن ناھايىتى پەخىرلىنىمەن.

خۇلاسە: (ئوقۇتقۇچى) يوقىرىدىكى ساۋاقداشلار ناھايىتى ياخشى سۆزلىدىڭلار، مېنىڭچە قالغان ساۋاقداشلارنىڭمۇ، جۈملىدىن مېنىڭمۇ ھەم ھېسسىياتىم سىلەرنىڭكى بىلەن ئوخشاش ، ھەممىمىز گۈزەل ئانا يۇرتىمىز شىنجاڭنى چىن كۆڭلىمىزدىن سۆيىشىمىز، پەخىرلىنىشىمىز، تىرىشىپ ئوقۇپ ، غەربىي دىيارنىڭ ئېچىلمىغان سىرلىرىنى ئېچىشىمىز، يېزىلمىغان مەنزىرىلىرىنى يېزىشىمىز شۇ ئارقىلىق پۈتۈن جاھانغا تونۇتۇشىمىز كېرەك.

                             2. يىپەك يولى

ئوقۇتۇش مەزمۇنى:  

8-13 بەتلەر

ئوقۇتۇش نىشانى:

1. تېكىستنى ھېسسىياتلىق ئوقۇپ، يېڭى سۆزلەرنىڭ مەنىسىنى چۈشىنىۋېلىش.
2. مەنىسى چوڭقۇر جۈملىلەرنى تەھلىل قىلىپ، غەربىي دىيارنىڭ پارلاق تارىخىنى بىلىۋېلىش. يىپەك يولىنىڭ مۇھىم رولىنى ھېس قىلىش.
3. غەربىي دىيارنىڭ تارىخىنى چۈشىنىشكە بولغان قىزىقىشىنى قوزغاش . يىپەك بولىغا دائىر ھېكايىلەر ئۈستىدە پىكىر ئالماشتۇرۇش.

مۇھىم نۇقتا:

       ئوقۇغۇچىلارنى تېكىستنى مۇستەقىل ئوقۇشقا يېتەكلەش. تېكىستنىڭ ئاساسىي مەزمۇنىنى چۈشىنىۋېلىش.

قىيىن نۇقتا:

يىپەك يولىنىڭ مۇھىم رولى، غەربىي دىيارنىڭ تارىخى ۋە ئاپتورنىڭ ئىپادىلەش ئۇسۇلىنى چۈشىنىۋېلىش.

ئوقۇتۇش ۋاستىلىرى:

دۇنيا خەرىتىسى ( گلوبۇس ) ، كۆپ ۋاستىلىق ئوقۇتۇش دېتالى

ئوقۇتۇش ئۇسۇلى:

1. دۇنيا خەرىتىسى ( گلوبۇس) تىن يىپەك يولى بېسىپ ئۆتكەن مۇساپىنى تاپقۇزۇش ۋە يىپەك يولى ھەققىدە نېمىلەرنى بىلىدىغانلىقىنى سوراش.
2.تېكىستنى مۇستەقىل ئوقۇتۇش ، يېڭى سۆز جۈملىلەرگە ئىزاھات بېرىش.
3. ھەمكارلىشىپ ئۆگىنىش ئاساسىدا ئۆز-ئارا پىكىر ئالماشتۇرۇش، تەھلىل قىلىشقا يېتەكلەش.
ئۆگىنىش ئۇسۇلى:

1. ئىزدىنىپ ئۆگىنىش ئاساسىدا يىپەك يولى توغرىسدا ئالدىن ماتېرىيال توپلاش.
2. مۇستەقىل ، كوللىكتىپ ئاۋازسىز ئوقۇش، روللارغا بۆلۈنۈپ ئوقۇش.
3.ھەمكارلىشىپ ئۆگىنىش ، سوئاللارنى لايىھىلەش.
4. پىكىر ئالماشتۇرۇش، تەھلىل قىلىش.

ئوقۇتۇش جەريانى:

دەرسكە باشلاپ كىرىش:

ئوقۇتقۇچى: ساۋاقداشلار  بۈگۈن  سىلەرگە بىر كۇبلېت شېئىر ئوقۇپ بېرەي، ئاڭلاپ بولۇپ ، ھېس قىلغانلىرىڭلارنى دەپ بېرىڭلار ، بولامدۇ؟

            ئىېلىمىزدە بىر مەشھۇر يول بار ئىكەن بىلەمسىز؟
            ‹‹يىپەك يولى›› دىيىلىپ كەلگەن ئىكەن تارىختىن.
           ماڭغان ئىكەن شۇ يولدىن جاڭ چىيەن،بەن چاۋ، تاڭ شۈەنزاڭ،
           شەرق، غەرب رىشتىنى تۇتاشتۇرۇپ يۈرەكتىن .

ئوقۇغۇچى: مەن بۇ شېئىردىن ‹‹ يىپەك يولى ›› نىڭ ناھايىتى ئۇزۇندىن بېرى بار ئىكەنلىكىنى ، جاڭ چىيەن ، بەن چاۋ، تاڭ شۇەنزاڭ قاتارلىق مەشھۇر شەخسلەرنىڭمۇ موشۇ يول ئارقىلىق شەرق، غەرب ئارىسىدا دوستلۇق مۇناسىۋىتى ئورناتقانلىقىنى بىلدىم.


( داۋامى بار)
