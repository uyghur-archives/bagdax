مەنغۇ مۇشۇ خەلقئارا ۋەزىيەتتىكى بەزىبىر ئۇچۇرلاردىن گۇمانلىنىپمۇ قالىمەن. ئەمما ھەممىمىزگە مەلۇملۇق بىر ئىش، مەرھۇم كازافىي ھايات ۋاقتىدا (ئۇچۇرلاردىن ئېسىمدە قېلىشىچە 1989- يىلىدىكى بىر قېتىملىق خەلقئارالىق يىغىندىغۇ دەيمەن) ھەرقايسى دۆلەت باشلىقلىرى بىلەن قول ئېلىشىپ كۆرۈشىدىغان چاغدا ئوڭ قولىغا ئاپئاق پەلەي كەيگەن (سول قولىغا كەيمىگەن) ھالدا يىغىن مەيدانىدا پەيدا بولغان. بۇنىڭغا كۆپچىلىك ھەيران قالغان ھەمدە بىر مۇخبىر بۇنىڭ نېمە ئىش ئىكەنلىكىنى سورىغاندا ئۇ: «مەن ئامېرىكىنىڭ غالچىلىرى (ئامېرىكا) بىلەن قول ئېلىشىپ كۆرۈشكەندە قولۇمنىڭ مەينەت بولۇپ كەتمەسلىكى ئۈچۈن مۇشۇنداق قىلدىم» دەپ جاۋاب بەرگەن. دېمەك، بۇنى ئاڭلىغان ھەرقانداق بىر ئادەم كازافىينىڭ ئامېرىكا باشچىلىقىدىكى غەرب ئەللىرىگە بولغان پوزىتسىيىسى ۋە مەيدانىنى بىلىۋالايدۇ. ئەمدى كازافىينىڭ ئۆلۈپ كېتىش جەريانىغا قارايدىغان بولساق ئۇ تاكى ئۆلۈپ كەتكۈچە دۆلىتىدىن ئايرىلمىدى (گەرچە بىرقىسىم ئارقا كۆرۈنۈش بىزگە قاراڭغۇ بولۇشى مۇمكىن)، ئۇ باشقا دۆلەتلەرگە سەرگەردان بولۇپ چىقىپ كەتسە تامامەن ھاياتىنى ساقلاپ قېلىشى مۇمكىن ئىدى..... مانا بۇلاردىن بىز مەرھۇم كازافىينىڭ سىياسىي مەيدانىنىڭ قانچىلىك ئىكەنلىكىنى بىلىۋېلىش تەس ئەمەس.
ئەمدى كازافىينىڭ سەۋەنلىكلىرى ۋە باشقا سەلبىي تەرەپلىرى نوقتىسىدىن ئېيتقاندا كۆپ گەپ قىلىشنىڭ ھاجىتى يوقكىن ھەرقانداق بىر ھاكىمىيەت يۈرگۈچى (سىياسىيۇن) ھامان بىرقسىم سەۋەنلىكلەردىن ساقلىنالىشى مۇمكىن ئەمەس. كۈنلەر ئۆتۈپ بىز شۇنداق ياخشى كۆرۈپ قوللايدىغان ئادەملەرنىڭ شۇنچە رەزىل ئىكەنلىكىنى ھېس قىلىمىز. بۇ زاماندا خەلقئارا ۋەزىيەت كۆپ قۇتۇبلىشىشقا قاراپ يۈزلىنىۋاتقان، شۇنچە مۇرەككەپ ۋەزىيەتتە ھىچكىم ئەتە يەنە قانداق ئىشلارنىڭ يۈز بىرىدىغانلىقىنى ئېنىق بىرنەرسە دەپ بولالمايدۇ.

سۈرىيەنىڭ زۇڭتۇڭ جەمەتى بەسر ئاسادمۇ ۋاقتىدا قورال تاشلاپ ياخشىلىقچە تەسلىم بولمىسا ،ئۇنىڭ كۈنىمۇ ئاۋۇ ئىككى <قەھرىمان>دوستىنىڭكىدەك ھەتتا ئۇنىڭدىنمۇ بەتتەر بولىدۇ...ئۇ ھېلىمۇ سۈرىيەدە يۈزمىڭغا يېقىن كىشىنىڭ ئۆلۈشىنى كەلتۈرۈپ چىقاردى...
  توغرا دەيسىز تاخان قېرىندىشىم. بىزنىڭ ئۇيغۇر تىلىمىز نېمىشقا ئاشۇ ئاتالمىش قەھرىماننىڭ ئىسمىنى توغرا ئاتاپ بېرەلمەيدۇ؟ كازاڧڧڧفىي ئەمەس قەززافى. قوشتىللىق بولدۇق دەپ ئەمدى ئەتە ئۆگۈن ئۆزىمىزنىڭ ئىسمىمىز بولغان ئابدۇرېشىت دېگەننى ئابۇدۇرېشىتى، مۇھەممەد نى مەيمەيتى، سەمەت نى سەمەيتى، ئالىم نى ئالىمۇ.... نېمە دېگەن بىمەنىلىك!!! ما تېلېۋىزىيە ئىستانىسىدىكى قارا قورساق تەرجىمان، دىكتورلار قاچاندا قارىسا ئاشۇنداق بۇززۇۋانچە گەپ قىلىپ بىزنىڭ تىلىمىزنى ئەبجەش قىلىۋەتتى. ئاپتونوم رايون دەيدۇ، قايسى دۆلەتتىكى، قايسى مىللەتنىڭ ئاپتونوم رايونى ئىكەنلىكىنى ئېنىق ئايرىماستىنلا ئاپتونوم رايونىمىز دەپ چۈشۈشلىرى... شىنجاڭ ئۇيغۇر ئاپتونوم رايونى دېسە تىلى كېسىلىپ كېتىدىغاندەك!!! ئېغزى مايماقلار . ئۈچ يىل بولغىلى تۇردى، مۇشۇ كۆرگۈ(تېلېۋىزور) دېگەننى تازا كۆرۈپ كەتمىدىم، ئادەملىرىمىزنىڭ تىلى ئاشۇنداق كەمتۈك، ساز بولغاچقا...   مۇشۇ مەركىزى خەلق رادىيو(_ئۈنتارت) ئىستانىسىدىكى دىكتور رىياسەتچىلەرگە قاتتىق ھۆرمىتىم بار. ئۇلار زېرىكمەي شىنجڧڭ ئۇيغۇر ئاپتونوم رايونى دەپ ئاتايدۇ. شۇڭلاشقىمىكىن  ئۈنتارتقۇ بىلەن كومپيۇتېر مېنىڭ ئۇچۇر مەنبەيىم بولۇپ كېلىۋاتىدۇ.
