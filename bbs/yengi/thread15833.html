يوللىغان ۋاقتى 3 كۈن ئالدىدا |ھەممە قەۋەتنى كۆرۈش
تور ئارلاۋېتىپ بوزقىر ئەپەندىنىڭ " يىڭى بىر بىمەنە ئەپسانە " ناملىق تەرجىمىسى ۋە ئۇنىڭغا چۇشكەن ئىنكاسلارنى ناھايتى قىزىقىش ئىچىدە ئوقۇپ چىقتىم . شۇنداقلا ئۆزىمىزنىڭ ئەپسانىلىرى توغرىلىق ئۆيلىنىشقا باشلىدىم .
     كىچىڭىمدىن تارتىپ ماخمۇت قەشقىرى  ۋە " تۇرك تىللار دىۋانى "، يۇسۇپ خاس ھاجىپ ۋە "قۇتاتقۇ بىلىك " ، " 12 مۇقام " ... ئىشقىلىپ قانداقتۇر  ئۇيغۇرلارلا "قىلالايدىغان " باشقا مىللەتلەر "قىلالمايدىغان " ئىشلار توغرىسىدىكى تۇرلۇك ئەپسانىلەر ، ئۇيغۇرلارنىڭ بەك ئەقىللىق ئىكەنلىكى باشقا مىللەتلەرنىڭ بىزدەك ئەقىللىق ئەمەسلىكى بۇنىڭ سەۋەبىنى تەكشۇرسە ئەلئېھرامنىڭ قۇرۇلمىسىغا ئوخشىشىپ كېتىدىغان دوپپا كىيدىغانلىمىز ئۈچۈنلىكى  توغرىسىدىكى ئەپسانىلىلەردىن روھىم كۈتۇرۇلۇپ ئۆزەمنى ھەقىقەتەنمۇ باشقا مىللەتلاردىن ئەقىللىق چاغلايدىغان ئىدىيە شەكىللەنگەن ئىدى مەندە .
    ئاز كام قىرىق ياشنىڭ قارىسىنى ئاپتىمەن . شۇنچە يىللىق تۇرمۇش تەجىرىبىلىرىمدىن ، باشقا مىللەتلەردىن تۇنۇشقان دوستلىرىمدىن ، ئونۇڭدىن باشقا ئۇيغۇرلارنىڭ ھازىرقى ئومومى ئەھۋالىدىن قارىسام بۇلار پەقەت بىر ئەپسانىلىكەن خالاس .
    ئەگەر بىز ھەقىقەتەن ئەقىللىق بولساق ، نىمە ئۈچۈن شۇنچە ياخشى شارائىتتا ئوقۇتقۇچىلارنىڭ ۋە بالىلىرىمىزنىڭ يالۋۇرۇشلىرىغا پەرۋامۇ  بالىلىرىمىزنى مەكتەپتىن توختىتىپ ، بۇرنىمىزنىڭ ئۈچىنىلا كورۇپ ئازىراق پۇل تاپىدىغان ئىشلارغا سالىمىز ؟
   ئەگەر بىز راستىنلا ئەقىللىق بولساق ، ئىقتىسادى جەھەتتە نىمىشقا باشقا مىللەتلەردىن يۇقۇرى ئورۇندا تۇرالمايمىز ؟ بىزنىڭ ئىقتىسادىمىز ئەمدىلا يىمەك - ئىچمەكنى ئاساس قىلغان كارخانىلاشقان دەۋىرىگە قەدەم قويدى ، باشقىلار بولسا ئاللىبۇرۇن سانائەتلەشكەن ، ماشىنىلاشقان ، ئىلىكتىرۇنلاشقان دەۋىردە . جۇڭگۇ دىكى ئۆلكە ، ئاپتۇنۇم رايۇنلار ئىچىدە ئارقىدىن سانىغاندا بىرىنچى دىسەك يامان كەتمەيدۇ . شەخىسلەرنىڭ ئەھۋالىنى دىمەيلا قۇياي ، بىزدە نىمە جىق دىۋانە ، ئەگەر ئەقىللىق بولساق نىمىشقا ئۆزىمىزنىڭ كۇنىنى ئۆزىمىز مۇستەقىل ئالالمايمىز ؟
   ئەگەر بىز راستىنلا شۇنچە ئەقىللىق بولساق ، نىمە ئۈچۈن ئۆزىمىزنىڭ يامان ئىللەتلىرىنى تىزلا يۇقۇتۇپ باشقىلارنىڭ ناچار ئىللەتلىرىنى شۇنچە تىز يۇقتۇرۋالىمىز ؟
   ئون نەچچە ياش ۋاقىتلىرىم بولسا كېرەك ، مەھەلىمىزدە بىرى كۈندۇزى ھاراق ئىچىپ دەلدەڭشىپ قاپتكەن ، مەھەلىدىكىلەرنىڭ گەپ - سوزىدىن قېچىپ بىر ئايغىچە قارىسىنى كورسەتمىكەن . ھازىر مەھەلىمىزدىكى بوتكىدا نەچچە سورۇندا ،كىچە - كۈندۇز دىمەي  ھەر خىل ياشتىكىلەر ئارلىشىپ ھاراق ئىچىپ ، ئۆزىلىرىنىڭ كۆپ ئىچەلەيدىغانلىقىنى ماختىنىپ سۆزلىشىپ ، تىللىشىپ ، ئۇرۇشۇپ يۇرۇشلەرگە ھىچكىم ھەيران قالمايدۇ . ئۇندىن باشقا تاماكىنىغۇ دىمەيلا قۇياي تەرەققى قىلىپ نەشە ، خىروئىن چىكىدىغانلارمۇ كۈنسىرى كوپەيمەكتە ، قىمارنىغۇ ئوينىمايدىغان ئىمام بىلەن مەزىن قالدىمۇ بىلمىدىم بۇ توغرىسىدا سوزلىمەي  . بۇزۇقچىلىق قىلمىغانلار نورمال سانالمايدىغان ھالاتكە يەتتۇق .( شىنجاڭنى جۇڭگونىڭ ئەيدىز سەركەردىسىگە ئايلاندۇرۇپ بولدۇق )... ئاياللىرىمىز بولسا شۇنچە  شۇنچە لاتاپاتلىك تەبىئى گۇزەللىكىدىن ۋاز كېچىپ چاچلىرىنى كىسىپ ،ئەبجەش رەڭلەردە بۇياپ ، ئالۋاستىدەك گىرىم قىلمىسا تويغا ، چايغا كىرەلمەيدىغان بۇلۇپ كەتتى ، كۈچىدىكى نۇرغۇن قىز -ئاياللاردىن  ، پاكىزلىق ، پاكلىق  ، ئىسلام نۇرى كورىۋېلىش تەس .  بىلىمسىزلىك ، ئۆزىنى قارغۇلارچە باشقىلار بىلەن سېلىشتۇرۇش ، نىمىنىڭ ھەقىقى بايلىق ئىكەنلىكىنى چۇشۇنۇش ئىدىيىسى كەمچىل .
    ئەگەر بىز راستىنلا شۇنچە ئەقىللىق بولساق ، نىمە ئۈچۈن ئۆزىمىزنىڭ ئانا تىلىمىزنىمۇ قوغداپ قالالمايمىز ؟
بىز ھەممىمىز ئۇيغۇرنىڭ ئۇيىدا تۇغۇلدۇق ، ئۇيغۇرنىڭ ئۇيىدە ، يۇرتىدا چوڭ بولدۇق بۇ كۆز يۇمغۇسىز ھەقىقەت . ئەنە ئەشۇ ئائىلە ، ئەشۇ يۇرت بىزنىڭ ئۇيغۇرچىنى ئۆگىنىشىمىزگە يىتىپ ئاشىدۇ .يا بىر پارتىيە - ھۆكۇمەت بۇ تىلىڭدا سوزلىسەڭ بولمايدۇ دىمىدى . يا بىر يېزىقىمىزنى ئىناۋەتسىز دىمىدى بىزنىڭ تىلىمىزغا باشقا تىلنى ئارلاشتۇرۇپ سوزلەيدىغانغا ھىچقانداق باھانىمىز يوق . .  باھانە بار دىيىلسە ئۆزىمىزنى كۆز - كۆز قىلىش مەقسىدىدە ئۆگىنۋالغان نەچچە ياد مىللەنىڭ تىلىنى ئارلاشتۇرۇپ سۆزلەپ ئۆزىمىزنىڭ " بىلىملىك " ئىكەنلىكىنى كۆز - كۆز قىلىش خالاس . بار دىيىلسە ئاز ساندىكى غورورىنى ئىت يەپ كەتكەن كەمغۇرور ئاتا - ئانا بولغۇچىنىڭ بالىلىرىغا ئۇيغۇر تىلىنىڭ مۇھىملىقى ، چوقۇم ئۇيغۇرچە سوزلەشنى تەشەببۇس قىلمىغانلىقى ، مەجبورلىمىغانلىقىدىن كورۇشكە بولىدۇ . ...يەنە نۇرغۇن باھانىلىرىڭىز بولسىمۇ ھەممىسى ئۇسۇرۇققا تەڭ !
  ئەگەر بىز راستىنلا ئەقىللىق بولساق ، نىمىشقا ئۆزىمىزنىڭ قانۇنلۇق ھوقوق مەنپەئەتىنىمۇ قوغداپ قالالمايمىز ؟
جۇڭگو  قانۇن تەرتىپلىرى ئېنىق ،قانۇن - تۇزۇم مۇككەممەل بىر دۆلەت . مۇشۇنداق بىر دۆلەتتە ياشاۋېتىپمۇ دىھقانلىرىمىزنى دىمەي تۇرايلى خېلى ئۇقۇمۇشلۇق زىيالىيلىرىمىزمۇ ئۆزىنىڭ ھۆقۇق - مەنپەئەتىنى ، مائاشىنى ھەتتا ئۈيىنىمۇ قانۇن بۆيىچە قوغداپ قالالمايدىغان ئەھۋاللار ئىنتايىن نۇرغۇن .
نۇرغۇن - نۇرغۇن مىساللانى ئالغۇم بار ئىدى بىراق ، سەۋىيەمنىڭ ، قەلەم كۈچۈمنىڭ تۈۋەنلىكىدىن كاللام قۇرۇق  .  
ئەمدى ھەرگىز ئۇ ئەپسانىگە ئىشەنمەيمەن ، ئادەمنىڭ ئەقىللىق يا دۆت بۇلۇشى ھىچقانداق مىللەت ياكى دىن بىلەن مۇناسىۋەتسىزكەن . ئادەم بىلىمنى ، ئەقىلنى ئەتراپتىكى مۇھىتتىن ، ئەتىراپتىكى شەيئىلەرىن قۇبۇل قىلىدىكەن . ئەقىللىق بۇلۇش ئۈچۈن شەئىلەر بىلەن كوپرەك ئۈچرىشىش ، كۈزىتىش ،تەھلىل قىلىش كېرەك ئىكەن . پاكىتنى ، ئەمەلىيەتنى ھۆرمەت قىلىش كېرەككەن ، ھىسياتقا تايىنىپ ھۆكۇم چىقىرىشتىن ساقلىنىش كېرەككەن .
قىسقىسى ، ئۇيغۇرۇمنىڭ دوپپىسىنىڭ مېنى ۋە باشقىلارنى ئەقىللىق قىلىش رولى پەقەتلا يوقكەن .

‏سىز قايسى مەھەللىدىنكى بىلمىدىم ، ھەممىنى بىر تاياقتا ھەيدىمەڭ . ماۋۇ خوتەن تەرەپتە ھاراق ئىچىپ لەيلەپ يۈرگەنى ،ياكى ئىچتىم دەپ ماختانغانلارنى كۆرۈپ باقمىغىلى ئۇزۇن زامان بوپ كەتتى .
‏سىز بەك ئۈمۈدسىزلىنىپ كەتتىڭىزمۇ قانداق؟
‏جەمىيەتكە دىققەت قىلىپ بېقىڭ  بىر نەچچە يىل بولدى خەلىقىمىز  ئويغاندى ،يەنى مىللى ئاڭ ئويغاندى،دىنىي گۈللىنىش دەۋرى باشلاندى ،ئىلىم تەلەپ قىلىش كۈچەيدى بۇلارغىمۇ دىققەت قىلىڭ .
‏يەنە بىر گەپ     كىچىڭىمدىن تارتىپ ماخمۇت قەشقىرى  ۋە " تۇرك تىللار دىۋانى "، يۇسۇپ خاس ھاجىپ ۋە "قۇتاتقۇ بىلىك " ، " 12 مۇقام " ... ئىشقىلىپ قانداقتۇر  ئۇيغۇرلارلا "قىلالايدىغان " باشقا مىللەتلەر "قىلالمايدىغان " ئىشلار توغرىسىدىكى تۇرلۇك ئەپسانىلەر ، ئۇيغۇرلارنىڭ بەك ئەقىللىق ئىكەنلىكى باشقا مىللەتلەرنىڭ بىزدەك ئەقىللىق ئەمەسلىكى بۇنىڭ سەۋەبىنى تەكشۇرسە ئەلئېھرامنىڭ قۇرۇلمىسىغا ئوخشىشىپ كېتىدىغان دوپپا كىيدىغانلىمىز ئۈچۈنلىكى  توغرىسىدىكى ئەپسانىلىلەردىن روھىم كۈتۇرۇلۇپ ئۆزەمنى ھەقىقەتەنمۇ باشقا مىللەتلاردىن ئەقىللىق چاغلايدىغان ئىدىيە شەكىللەنگەن ئىدى مەندە .
    ئاز كام قىرىق ياشنىڭ قارىسىنى ئاپتىمەن . شۇنچە يىللىق تۇرمۇش تەجىرىبىلىرىمدىن ، باشقا مىللەتلەردىن تۇنۇشقان دوستلىرىمدىن ، ئونۇڭدىن باشقا ئۇيغۇرلارنىڭ ھازىرقى ئومومى ئەھۋالىدىن قارىسام بۇلار پەقەت بىر ئەپسانىلىكەن بۇ گېپىڭىزنى يۈتىۋېتىڭ.سىز 40 ياشقا كىرگەن بىلەن ئەقلى تەپەككۇرىڭىز تاكامۇللاشماپتۇ .بىز (مەخمۇد قەشقەرى ،يۈسۈپ خاس ھاجىپ ،12 مۇقاملاردىن ) نىمىشقا پەخىرلەنمىگۈدەكمىز ،تازا پەخىرلىنىشىمىز كېرەك .
‏خەلىقىمىز سىز دېگەندەك ئۇنچىۋېلا دۆت ئەمەس پەقەت شۇ ئەقىلنى جارى قىلدۇرۇشقا ،قېزىپ چىقىرىشقا شارائىت بۇلسا ئاندىن بۇلاتتى .قالغىنىنى ئۆزىڭز ئويلاپ بېقىڭ .

يوللىغان ۋاقتى 3 كۈن ئالدىدا |ھەممە قەۋەتنى كۆرۈش
بۇ دۇنيا ئىمتىھان مەيدانى بولغانلىقى ئۈچۈن ئاللاھ تەئالا بۇ دۇنيادا ياخشى يولنىمۇ بار قىلغان، يامان يولنىمۇ بار قىلغان. ياخشى يولغا ئەقىلنى ھەيدەكچى ۋەھىينى دەۋەتچى قىلغان، يامان يولغا نەپىسىنى ھەيدەكچى شەيتاننى دەۋەتچى قىلغان. ئاندىن ئىنسانغا ئىككىسىنىڭ قايسىسىنى تاللاش توغرىسىدا تولۇق ئەركىنلىك بەرگەن، شۇنداقلا ھەر نەرسىدە قىلغاندەك بۇ ئىشتىمۇ بىر تەڭپۇڭلۇق ئورناتقان. ئىلاھىي ئادالەت مانا بۇدۇر.
باشقىچە قىلىپ ئېيتقاندا ئاللاھ تەئالا ئىنسانغا ياخشى-ياماننى تونۇشى ئۈچۈن ئەقىل ئاتا قىلغان. ئەقىلگە يېتەكچىلىك قىلىشى ئۈچۈن كىتاب نازىل قىلغان، كىتابنى چۈشەندۈرۈش ئۈچۈن پەيغەمبەر ئەۋەتكەن. يەنى ئىنساننى ئىچىدىن بىر نەرسە (ئەقىل)، تېشىدىن بىر نەرسە (ۋەھىي) توغرا يولغا دەۋەت قىلىپ تۇرىدۇ.
ئىمان توغرىنىڭ توغرىلىقىنى ئەقىل بىلەن چۈشىنىش ئەمەس، قەلب بىلەن تەستىقلاش بولغانلىقى ۋە ئىنساننىڭ قارار مەركىزى ئەقلى ئەمەس قەلبى بولغانلىقى ئۈچۈن قەلب تولاراق ئەقىلنىڭ مەسلىھەتى بىلەن ۋەھىينىڭ يوليورۇقىغا ئەمەس، نەپسنىڭ ئارزۇسى بىلەن شەيتاننىڭ ۋەسۋەسىسىگە بويسۇنىدۇ. چۈنكى قەلب مەنپەئەتپەرەستۇر ۋە تۇراقسىزدۇر.
نەپىس تېزلا قولغا كېلىدىغان مەنپەئەتكە قىزىقىدۇ، شەيتان قەلبنى ئەقىلگە ئەمەس، نەپىسىگە بويسۇنۇشقا چاقىرىدۇ. بەدەن ئەقىلگە ئەمەس، قەلبكە بويسۇنىدىغانلىقى ئۈچۈن قەلبنىڭ خاتا قارارىنى ئىجرا قىلىپ گۇناھ ئۆتكۈزىدۇ. شۇنىڭ ئۈچۈن پەيغەمبىرىمىز مۇنداق دېگەن:
«بىلىڭلاركى، ئىنساننىڭ بەدىنىدە بىر پارچە گۆش بار. ئۇ گۆش تۈزەلسە پۈتۈن بەدەن تۈزۈلىدۇ، ئۇ گۆش بۇزۇلسا پۈتۈن بەدەن بۇزۇلىدۇ. بىلىڭلاركى، ئۇ گۆش قەلبتۇر»- بۇخارى ۋە مۇسلىم رىۋايەت قىلغان.
ھەر ئىنسان تولۇق ئىختىيارلىق بولغان بۇ ساھەدە ھېچكىمنىڭ ھېچكىمگە بېسىم قىلىش ئالاھىدىلىكى بولمايدۇ. شۇنىڭ ئۈچۈن بۇ دۇنيادىكى ئىشلار ياخشى كىشىلەرنىڭ ئويلىغىنىدەك بولمايدۇ.
ھېساب ئاخىرەتتە ئېلىنىدىغانلىقى ئۈچۈن، شۇنداقلا يامانلىقنىڭ جازاسى ۋە ياخشىلىقنىڭ مۇكاپاتى ئاخىرەتتە بېرىلىدىغانلىقى ئۈچۈن ئىنسانلارنىڭ كۆپىنچىسى بۇ دۇنيادا «ئەتىكى دۈمبىدىن بۈگۈنكى ئۆپكە ياخشى» دېگەن خاتا مەنتىقىسى بويىچە ئىش قىلىدۇ.
توغرىنى چۈشەنگەن بىر مۇسۇلماننىڭ ۋەزىپىسى يولدىكى ھەمراھلىرىنىڭ قالايمىقان ھەرىكەتلىرىگە قاراپ ئۈمىدسىزلەنمەستىن ئۇلارغا توغرىلارنى ئىلمىي يوسۇندا، تاتلىق بىر تىل ۋە چىرايلىق بىر مۇئامىلە بىلەن چۈشەندۈرۈشتۇر، شۇنداقلا ئاللاھ ئالدىدىكى مەسئۇلىيەتلىرىنى ئەسلىتىشتۇر.
چۈنكى بىر يامان ئىشنى كۆرگەن مۇسۇلماننىڭ ۋەزىپىسى ئۇ يامان ئىشنى قىلغانلارغا چىرايلىقچە نەسىھەت قىلىپ قويۇشتۇر، خالاس. ئۇلار ئەمەل قىلامدۇ، قىلمامدۇ؟ ئۆزلىرىنىڭ ئىشى. شۇنى ئۇنتۇماسلىق لازىمكى:
ھەر كىشىنىڭ دوزاخقا كىرىش ئەركىنلىكى بار. بۇ ئەركىنلىكنى ئۇلارغا ئاللاھ بەرگەنلىك ئۈچۈن ھېچكىم ھېچكىمنى ھىدايەتكە زورلىيالمايدۇ. ياخشىلار يامانلارغا قولىدىن كېلىشىچە نەسىھەت قىلسا ۋە ئۆزلىرى توغرا يولدا مۇستەھكەم تۇرسا ئاللاھ ئالدىدىكى ۋەزىپىسىنى ئادا قىلغان بولىدۇ، شۇنداقلا يامانلارنىڭ بۇلارغا ھېچقانداق زىيىنى بولمايدۇ. ئاللاھ تەئالا مۇنداق دەيدۇ:
«يَا أَيُّهَا الَّذِينَ آمَنُواْ عَلَيْكُمْ أَنفُسَكُمْ لاَ يَضُرُّكُم مَّن ضَلَّ إِذَا اهْتَدَيْتُمْ إِلَى اللّهِ مَرْجِعُكُمْ جَمِيعاً فَيُنَبِّئُكُم بِمَا كُنتُمْ تَعْمَلُونَ»
«ئى مۇئمىنلار! ئۆزۈڭلارنى گۇناھتىن ساقلاڭلار، سىلەر قاچانىكى توغرا يولدا بولساڭلار، باشقىلارنىڭ ئاداشقىنى سىلەرگە زىيان يەتكۈزەلمەيدۇ. ھەممىڭلار ئاللاھنىڭ دەرگاھىغا قايتىسىلەر، ئاللاھ قىلغان ئەمىلىڭلارنىڭ سىلەرگە ئېيتىپ بېرىدۇ»- مائىدە 5/105.
ھەر كىشى ئۆزىنىڭ ۋە ئۆزىنىڭ ئائىلىسىنىڭ توغرا يولدا بولۇپ بولمىغانلىقى ھەققىدە ئويلىنىشى لازىم. «جەمئىيەت بۇزۇلۇپ كەتتى، ھەممە ئادەم گۇناھ قىلىۋاتىدۇ. پالانى ناماز ئوقۇمايدۇ. پۇستانى روزا تۇتمايدۇ…» دېگەندەك ئۆزىنىڭ يا دۇنيالىقىغا يا ئاخىرەتلىكىگە قىلچە پايدىسى يوق گەپلەرنى قويۇپ ئۆزىنىڭ ئەھۋالىغا قارىشى، ئۆزىنىڭ ئائىلىسىنىڭ ئەھۋالىغا قارىشى ۋە مەن قانچىلىك مۇسۇلمان؟ مەن قانچىلىك ئىنسان؟ مېنىڭ ئائىلە-تاۋابىئاتىمنىڭ مۇسۇلمانچىلىقى قايسى دەرىجىدە؟… دەپ ئۆزىگە سوئال قويۇشى كېرەك. ئاللاھ تەئالا مۇنداق دەيدۇ:
«يَا أَيُّهَا الَّذِينَ آمَنُوا قُوا أَنفُسَكُمْ وَأَهْلِيكُمْ نَاراً وَقُودُهَا النَّاسُ وَالْحِجَارَةُ عَلَيْهَا مَلَائِكَةٌ غِلَاظٌ شِدَادٌ لَا يَعْصُونَ اللَّهَ مَا أَمَرَهُمْ وَيَفْعَلُونَ مَا يُؤْمَرُونَ»
«ئى مۇئمىنلار! سىلەر ئۆزۈڭلارنى ۋە ئائىلەڭلەرنى ئوتتىن ساقلاڭلار. ئۇ ئوتنىڭ يېقىلغۇسى ئىنسانلاردۇر ۋە تاشلاردۇر. ئۇ ئوتنىڭ بېشىدا قوپال ۋە قاتتىق قول پەرىشتىلەر بار. ئۇلار ئاللاھنىڭ ئەمرىگە ئاسىيلىق قىلمايدۇ، ئۆزلىرىگە بېرىلگەن ئەمرلەرنى ئىجرا قىلىدۇ»- تەھرىم 66/6.
