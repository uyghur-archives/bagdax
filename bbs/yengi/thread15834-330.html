ھەي نېمە دەي! شۇنداقتىمۇ ئۆز قارىشىمنى دەپ باقاي: شىنجاڭ ئۇيغۇر ئاۋتونۇم رايونىنىڭ مەركىزى ئۈرۈمچى شەھىرىدىكى (يەنى جۇڭگونىڭ شىنجاڭ ئۇيغۇر ئاۋتونۇم رايونىلىق خەلق ھۆكۈمىتى خىزمەت ئورگىنى جايلاشقان جايىدىكى)بىر ناھىيە-رايون دەرىجىلىك ئاساسى قاتلام مائارىپىقا مەسئۇل مەمۇرى ئورگان رەھبىرىنىڭ ئاغزاكى ھالدا «ئۇيغۇرچە ئېلىتىماسىڭنى قوبۇل قىلىشنى رەت قىلىمىز» دىيىشىنى ھەرقانداق كىشى(ئەقلىنى تاپقان مەيلى ئۇ قىززىققان يا سوغۇققان كىشى، ئانا تىللىق ياكى قوشتىللىق ئوقۇغان كىشى بولۇشىدىن قەتئىنەزەر) قانداقسىغىلا ئويلاپ باقسا، «كاللامدىن ئۆتتى» دىيەلەرمۇ؟ مەن تۆنۈگۈن ئاخشام  باغداش مۇنبىرىدىن مۇشۇ خەۋەرنى ئوقۇپ ، تا ھازىرغىچە ئويلاپمۇ ھېچبىر كاللامدىن ئۆتكۈزەلمىدىم. بىلمىدىم ئۇيغۇر تىلىدىكى تۈرلۈك تاراتقۇ - نەشىريات ئورۇنلىرى(قايسى بىر خەۋەردىن جۇڭگودىكى ئاز سانلىق مىللەتلەر تىل-يېزىقىدا گېزىت -ژورنال ئەڭ كۆپ چىقىدىغان جاي شىنجاڭ ئۇيغۇر ئاۋتونۇم رايونى دەپ ئاڭلىغان ئىدىم)، جۈملىدىن ئۇيغۇر تىلىنى قوللىنىدىغان ھەر بىر جان ئىگىسى قانداق ئويلىدىكىن؟ قائىدە تەرتىپ بويىچە كەلگەن بىر پۇقرانىڭ تەرتىپلىك، يوللۇق ئېلىتىماسىغا قارىتا « ئېلتىماسىڭ قوبۇل قىلىنمايدۇ» دىگەن ئەشۇ باشقۇرغۇچى دەۋلەت مەمۇرى  ئامما ئۈچۈن قىزغىن مۇلازىمەت قىلىش پوزىتسىيەسى بىلەن ئۆز پىكرى(مەن بۇ شۇ يولداشنىڭ شەخسى پىكرى بولماستىن بەلكى، قانۇن ئارقىلىق ئىدارە قىلىندىغان جەمىيەت قۇرۇش ئۈچۈن زور تىرىشچانلىق كۆرسىتىۋاتقان جۇڭگودەك بىر قۇدرەتلىك دەۋلەتنىڭ ئاساسىي قاتلام باشقۇرغۇچى مەمۇرىنىڭ دەۋلەت نامىدا بەرگەن جاۋاپ پىكرى دەپ قارىدىم)نىڭ قانۇنىي ئاساسىنى  ئوتتۇرىغا قويغان بولسا، قانداق ياخشى بولار ئىدى ھە؟ بىز ئاددى پۇقرالارنىڭ نەزىرىدە، ئاساسىي قاتلامدىكى دەۋلەت مەمۇرلىرى   كەڭ خەلىق ئاممىسى ئۈچۈن دەۋلەت قانۇن-نىزاملىرىنى ئادىل يۈرگۈزگۈچىلەرئەمەسمىدى! مۇكەممەل قانۇنچىلىق دەۋلىتى قۇرۇش ئۈچۈن تىرىشۋاتقان قۇدرەتلىك دەۋلىتىمىز جۇڭگودە،  ئەگەر ھەرقانداق پۇقرا قانۇندىن چىقىپ كەتسە، يەنى قانۇنسىزلىق قىلسا، قانۇن-نىزام بويىچە جازالىنىشى كېرەك. ئەمما قانۇن ئىجرا قىلغۇچى-قانۇننى نازارەت قىلغۇچىلار قانۇن- تەرتىپتىن چىقىپ كەتسە،ياكى ئۆز مەجبۇرىيەتلىرىنى قانۇن بويىچە ئادا قىلمىسا قانداق قىلىش كېرەك؟ مېنىڭچە بۇ چوقۇم قانۇن ئىدارە قىلىدىغان جەمىيەت قۇرۇش قاتتىق تەكىتلىنىۋاتقان ھازىرقى دەۋردە ھەرقانداق پۇقرا ، قانۇنىي ئىگە ۋە تەشكىلات ئەستايىدىل ئويلىنىشقا تېگىشلىك ئىنتايىن سەزگۈر مەسىلە بولسا كېرەك. مەن شىنجاڭ ئۇيغۇر ئاۋتونۇم رايونىدا ياشاۋاتقانلىقىم ئۈچۈن، بىر جۇڭگو پۇقراسى بولۇش سالاھىيىتىم بىلەن، قانۇنىي نازارەتچى ئورگانلارنىڭ بۇ خىل قىلمىشلارنى قانۇنىي ھوقۇق دائىرىسى بويىچە، تېزدىن ئەستايىدىللىق بىلەن تەكشۈرۈپ كۆرۈش  تەكلىپىنى بېرىمەن. مېنىڭچە بۇ ۋەزىيەت تەقەززاسى بولسا كېرەك.

ئەگەر يىغىن ئېچىپ ئۇيغۇر تىلىدىكى ئىلتىماسنى قۇبۇل قىلماسلىقنى قارار قىلغان بولسا،بۇ ھۆكۈمەت ھەركىتى بولىدۇ،ھەم يىغىن خاتىرىسىگە يېزىلىپ ئارخىپتا ساقلىنىدۇ.
سىز چۇقۇم قۇبۇل قىلماسلىق ھەققىدىكى قارار توغرىسىدىكى يازما ئۇقتۇرۇشنى تەلەپ قىلىڭ،بۇ شۇ قارارنى چىقارغان ئورۇننىڭ مەجبۇرىيىتى.
ھۆكۈمەت تارماقلىرى ئۆزىگە تەۋە مەمۇرىيەت يۈرگۈزۈشتە،ئىلتىماس قىلغۇچىنىڭ يازما ئىلتىماسىنى تاپشۇرۇپ ئېلىپ 10كۈن  (بەزى تارماقلار15كۈن)ئىچىدە ئىلتىماسنى قۇبۇل قىلىپ ئىجىرا قىلىش-قىلماسلىق ھەققىدە قارار چىقىرىدۇ،ئەگەر قۇبۇل قىلمايدىغان بولسا قۇبۇل قىلىنماسلىق سەۋەبى ھەققىدە ئىلتىماس قىلغۇچىغا يازما ئۇقتۇرۇش قىلىدۇ.
ئاغزاكى ئۇقتۇرۇشنىڭ قانۇنى كۈچى بولمايدۇ،ئۇلار يەنىلا سىزگە جاۋاپ بەرمىگەنگە ھىساپ،سىز يەنىلا جاۋاپ بېرىشنى تەلەپ قىلىڭ،يەنە ئۇيغۇرچە ئەلتىماسنى قۇبۇل قىلمايدىغانلىغى ھەققىدە يىغىندا قارار قىلىغانلىغىنى ئوتتۇرىغا قويسا،بۇ ھەقتىكى يازما ئۇقتۇرۇشنى تەلەپ قىلىڭ،ئۇ ئورۇن سىزگە بۇنداق يازما ئۇقتۇرۇشنى چىقىرىپ بېرىشكە ھەرگىزمۇ جۈرئەت قىلالمايدۇ،ئۇلارنىڭ بۇنداق قىلىدىغانغا بېشى ئون ئەمەس،ئەگەر چىقىرىپلا بەرسە ئۆزلىرىنىڭ بالا تېررىيدىغانلىغىنى،سىزنىڭ مۇشۇ ئۇقتۇرۇشنى كۆتىرىپ ئۆزلىرىنى بەكلا رەسۋا قىلىۋېتىدىغانلىغىنى بىلىدۇ.چۈنكى «جۇڭخۇا خەلق جۇمھۇرىيىتى ئاساسى قانۇنى»ھەم «جۇڭخۇا خەلق جۇمھۇرىيىتى مىللى تېررىتورىيىلىك ئاپتونومىيە قانۇنى»ئۇلارنىڭ بۇنداق قىلىشىغا چەك قويغان،لېكىن ئۇلار ئۆزلىرىنىڭ قانداق ئاقىۋەت پەيدا قىلىدىغانلىغىنى،ھۆكۈمەتنىڭ يۈزىگە قارا سۈرىتىۋاتقانلىغىنى ئانچە بەك ھىس قىلالمىغان ئوخشايدۇ.
