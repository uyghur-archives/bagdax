يېقىنقى بىېر مەزگىللەردە باغداش تورىدا  بىز بۇرۇن بىلمىگەن ، بىلەلمىگەن تارىخ ، شەخىس ، نۆۋەتتىكى جەمىيەت ئەھۋالى ، ئجتىمائى كىرزىس،مللى كىملىك ، مائارىپ ................ھەققىدە بىر بىرىدىن ئىسىل يازمىلار، تەرجىمە ئەسەرلەر يېزىلىپ (ئېلان قىلىپ ) كەڭ جامائەتنىڭ قىزغىن ئىشتىراك قىلىشىغا  مۇيەسسەر بۇلىۋاتىدۇ. بۇندىن بۇرۇن بىز بىلمىگەن ، بىلەلمىگەن تارىخلار نەبىجان تۇرسۇندەك تۆھپىكارلىرىمىزنىڭ ئەجرى بىلەن بىز بىلەن يۈزكۈرۈشۈپ ئۆتمۈشىمىزنىڭ قاراڭغۇ تەرەپلىرى كۆز ئالدىمىزدا نامايەن بۇلىۋاتىدۇ. بوز قىر دەك تالانتىلىق تەرجىمانلىرمىزنىڭ مىننەتسىز ئەجرى بىلەن مىللەت، جەمىيەت، سىياسى ، مائارىپ ، ئىقتىسادقا ئائىت ئېسىل يازمىلار تەرجىمە قىلىنىپ جەمىيىتىمىزدە (دۆلىتىمىزنىڭ ئىچكىرى جايلىرى ،) قىزغىن بەس -مۇنازىرە قوزغاۋاتقان قىزىق نوقتىلار ئويغۇر تور ئەھلىنىڭ مۇلاھىزىسىگە قۇيۇلۇپ كۈچلۈك بەس - مۇنازىرە قوزغاۋاتىدۇ( بەزى نازۇك تېمىلار ئىچكىرى جايلاردا ئاللىقاچان نۇپۇزلۇق ئاخبارات ۋاستىلىرىدا ئېلان قىلىنغان بولسىمۇ تىل توسالغۇسى سەۋەبلىك نۇرغۇن كىىشلەر كۈرەلمىگەنلىك سەۋەبىدىن خەۋەرسىز قېىلىپ تولىمۇ ئېھتىياتچانلىق بىلەن مۇئامىلە قىلنۋاتقان بولغاچقا بوزقىر ئەپەندىمنىڭ ئىشلىرىغا بەرىكەتلەر بولسۇن، يامان كۆزلەردىن ئاللاھ ئۆز پانايىدا ساقلىسۇن). ..............................
............ يۇقارقىدەك بىر بىرىدىن ئىسىل تېمىلارنى كۈرۈپ ساقلىۋېلىشقا ئۈلگۈرمەيلا تور يۈزىدىن غايىپ بۇلىۋاتىدۇ. ئىنكاس ئارقىلىق ساقلىۋالغان تېمىلارنى قايتا ئىزدىسەك تېما ئۈچۈرۈلگەنلىك سەۋەبىدىن تاپقىلى بولمايۋاتىدۇ. شۇڭا باشقۇرغۇچىلارغا بىر نەچچە تەكلىپ سۇنماقچىمەن :
1. خەنزۇچە تور بەتلەردە ساقلاش ، بېسىپ چىقىرىش دىگەندەك ئىقتىدارلار بار  مۇشۇ باغداش تورىغىمۇ بېسىپ چىقىرىش ئىقىتدارى قۇشۇشقا بۇلامدۇ ( مىنىڭچە بەت لايھەلىگۈچى ۋە باشقۇرغۇچىلارنىڭ تېخنىكا سەۋېيىسى بۇنى ئىشقا ئاشۇرۇشقا تامامەن مۇمكىنچىلىك بېرىدۇ ).
2. مەلۈم سەۋەپلەر تۈپەيلىدىن ۋاقىتسىز ئۈچۈرۈلگەن تېمىلار بىراقلا ئۈچۈرلۈپ يوقامدۇ ياكى سانلىق مالۇمات ئامبىرىدا ساقلىنامدۇ؟ ئەگەر شۇ تېما زۇرۇر لازىم بۇلۇپ ئەسلى تېما ئىگىسى بىلەن ئالاقىلىشش ئىمكانىيىتى بولاي باشقۇرغۇچىلار بىلەن ئالاقىلەشسەك  ئەۋەتىپ بىرىش ئىمكانىيىتى بارمۇ ؟
3. باغداش تورى قايسى خىل خەت نۇسخسى ئىشلىتىدۇ؟  ياقتۇرغان تېما ئىنكاسلارنى بوياپ كۇچۇرۇپ چاپلاش  ئارقىلىق ۋورد ھۆججىتى قىلىپ ساقلاپ لازىم بولغاندا ئۈزىمىزنىڭ كومپىيۇتېرىدا ئاچساق خەت نۇسخسىسنى تۇنۇماسلىق سەۋەبىدىن نۇرمال كۆرگىلى بولمايدىكەن، مۇشۇ مەسىلىنى بەت يۈزىدىلا ھەل قىلىش مۇمكنچىلىكى بارمۇ ؟ ھىچ بولمىغاندا خەت نۇسخىسى بىلەن تەمىنلەشكە بۇلارمۇ ؟
