  مەن ھەر كۈنى ئاق مەسچىت ئالدىدىكى يول بىلەن ماڭىمەن ، بۇ ئەتىراپتا ھەر كۈنى دىگۈدەك ئون نەچچە ئادەمنىڭ بىر يەرگە توپلىشىۋېلىپ ئۆزىنى ئۇنۇتقان ھالەتتە بىر خىل ئويۇن بىلەن مەلىكە بولىۋاتقىنىنى كۆرگىلى بولىدۇ . يېققىن بېرىپ قارىسىڭىز ئۇ باشقا ئويۇن بولماستىن ئون نەچچە يىل ئىلگىرىلا ئوينىلىپ ھەممە بىلىپ پاخشەك بولۇپ كەتكەن ئويۇن يەنى - كىچىك تەخسە ئىچىدىكى قوناق دانىسى ياكى شىمىشكىنىڭ قانچە تال ئىكەنلىگىنى تېپىش ئويۇنى ، گەرچە كۆپۈنچە تورداشلار بۇ ئويۇننى بىلسىمۇ ، مەن يەنىلا بۇ ئويۇننى بۇ يەردە چۈشەندۈرۈپ ئۆتۈشنى توغرا تاپتىم . بۇ ئويۇن تۆت ياكى بەش كىشىنىڭ رىژىسسورلىقىدا باشلىنىدۇ ، بۇ كىشىلەر ئەمگەكنى ناھايىتى ئېنىق تەقسىم قىلىۋېلىشقان بولۇپ ئىچىدىكى بىرى موشۇ توپنىڭ ئەتىراپىنى چۆرگىلەپ سىرتنى كۈزىتىدۇ ، بازار باشقۇرغۇچى ياكى بىرەر ساقچىنى كۆرگەن ھامان دەرھال بۇيرۇق قىلىپ ئويۇننى يىغىشتۇرىدۇ ، قالغان ئۈچى يالغاندىن پۇل تىككەن بولۇپ ئەتىراپتىكىلەرنى قىزىقتۇرۇپ سورۇنغا چىللىشىدۇ ، قالغان بىرى سول قولىنىڭ بىگىز بارمىقىغا ماگىنىت ياكى تۆمۈر پارچىسىدىن بىرنى چاپلاشتۇرۋېلىپ سول قولىدا تۇتقان نېپىز تاختىنىڭ دۇمبە قىسمىغا مەھكەم بېسىپ تۇرىدۇ تاختىنىڭ قۇساق تەرىپىدە بولسا ئىچىگە ماگىنىت ياكى تۆمۈر پارچىسى سېلىنغان قوناق دانچىسى ياكى شىمىشكا  تاختىنىڭ دۈمبە تەرىپىدىكى ماگىنىتنىڭ ئۆزىگە تارتىشى ئارقىسىدا چۈشۈپ كەتمەي چاپلىشىپ تۇرىدۇ . ئوڭ قولىدا بولسا ئىككى -ئۈچ تال شىمىشكىنى تاماشىبىنغا كۆرسىتىپ تۇرۇپ كىچىك تەخسە ئىچىگە سېلىپ كۆپچىلىكتىن تەخسە ئىچىدە قانچە تال شىمىشكە بارلىقىنى سورايدۇ .  ئۇنىڭ قانچە تال شىمىشكە سېلىشى شۇنچە ئېنىقكى كۆزى سەل ئاجىز ئادەممۇ توغرا جاۋاپ بىرەلەيدۇ . ئەنە شۇ چاغدا ئۇنىڭ گۇپپاڭچىلىرى ئەڭ ئاز بولغاندا 50 يۈەن پۇل تىكىپ تەخسە ئىچىدىكىنى ئىككى - ئۈچ قېتىم توغرا تېپىپ بىرمۇنچە پۇلنى ئۇتقان بولىدۇ ، تۆتىنجى قېتىمدا بولسا قەستەن خاتا تېپىپ ئۇتتۇرۇپ ئەتىراپتىكى قىزىققۇچىلارنى تېرىكتۈرىدۇ ، بەشىنجى قېتىمدا بولسا ئېننىق قىلىپ ئۈچ تال شىمىشكىنى تەخسە ئىچىگە تاشلايدۇ ، بۇ چاغدا ئەتىراپتىكى ئۇتتۇرىدىن بىرسى جاۋابىنى بىرىدۇ .  گۇپپاڭچىلار ئۇنى رىغبەتلەندۈرۈپ پۇل تىكىشكە ئۈندەيدۇ ، شۇنىڭ بىلەن ئۇ كىشى ئۆزىنى ئۇنۇتقان ھالدا تەييار كەلگەن پۇلغا ئېرىشىش ئۈچۈن پۇل تىكىدۇ ، شۇنىڭ بىلەن ئۇ تەخسە ئىچىدىكى شىمىشكىنى ئۈچ تال دىگەن بولسا سول قول كۆرسەتكۈچ بارماقنىڭ كۆتۈرىلىشى بىلەن تۆتىنجى شىمىشكە تەخسە ئىچىدە پەيدا بولىدۇدە  يانچۇقتىكى پۇل ئۇچۇپ چىقىپ كىتىدۇ ، يېڭىلگەن ئادەم ئۆزىنىڭ كۆرۈش سېزىمىنى ئەيىپلىگەچ باشقىدىن پۇل تىكىدۇ . ئويۇن موشۇ تەرەققىسىدە داۋاملىشىدۇ . بۇ ئويۇننى ھەممە ئادەم دىگۈدەك بىلىپ كەتكەن ھازىرقى زاماندىمۇ ھامان ئالدىندىغانلارنىڭ چىققىنىغا قاراپ ھەيران قالىمەن . بۇ ئويۇننى ئورۇنلاشتۇرغۇچىمۇ ئۇيغۇر ، قويۇلغۇچىمۇ ئۇيغۇر ، بۇ كىشىلەرنىڭ ئۆز مىللىتىنى قويۇشتىن باشقا ئىشى يوقمىدۇ ؟ بۇ كەسىپتىن كىرگەن پۇل ئۇلارغا قانچىلىك تاتلىق تېتىيدىغاندۇ . بۇلارنىڭ قىلغان ئەتكەنلىرىگە باشتا شۇنداق نەپرەتلەنگەن بولسام،ئاخرىدا ئۇلارنى ماختىغۇم كەلدى ، نىمىشقا دىسىڭىز ھازىر دىگەن ھەممە كىشىنىڭ ئېڭى ئۇيغۇنۇپ كەتتى ھەتتا كىچىك بالىلارمۇ ھەممە نەرسىنىڭ ئەجىر-مېھنەتسىز قولغا كەلمەيدىغىنىنى بىلىپ كەتتى ، مانا موشۇنداق چاغدىمۇ بەزەن تەييارتاپلار ئۆزىچە تەر تۆكمەي ھالاۋەتكە ئېرىشىشنى ئۈمىد قىلىشىدىكەن يوقۇرقىدەكلەرنىڭ ئورىكىگە چۈشۈپ بۇرنىغا يىمەي قالمايدۇ . يوقۇرقىلار ھەرگىز ئەتىراپتىكىلەرگە مىلتىق تىرەپ پۇل تىكىشكە مەجبۇرلىمايدۇ ، ئويناش - ئوينىماسلىق ھەر كىمنىڭ ئىختىيارىكى ئىش شۇنداق تۇرۇقلۇق بىز ئۇلاردىن ئاغرىنماي ، ئۆزىمىز ھالال يولدا مېڭىپ ، ھەر قانداق نەرسىگە ئەجىر بىلەن ئېرىشسەكلا بولدى . ئاخىرىدا دەيدىغىنىم ،بىزدە" تۈلكە ئاكاڭ يۈرگىلى ئۈچ كۈن بولدى ئاچتا، بىر سەۋەبى بولمىسا قۇيرۇق تۇرامدۇ تاشتا " دەيدىغان بىر تەمسىل بار ، دۇنيادا ھەرگىزمۇ بىكارلىق نەرسە يوق ، شۇنىڭ ئۈچۈن ھەر قانداق ھالاۋەتكە ئۆزىمىزنىڭ ئەجىر - مېھنىتى ۋە ساپ تەرى بىلەن ئېرىشكەيمىز.
