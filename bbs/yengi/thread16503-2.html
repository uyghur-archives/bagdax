بىر قۇرمۇ ئوقۇمايمىكىن دېگەن موشۇ بىر گۈرجەك فوقنى ،يەنە بۇرنۇمنى تۇتۇپ بولسىمۇ ئۇ يەربۇيەرلىرىنى مالتىلاپ قالدىم .ئاجايىف ھە ئاجايىف .قانداقمۇ كاللىڭىزدىن تېشىف كەتمەي سىغىف تۇرغاندۇ موشۇنچىۋالا كاتتا ئىلىم -مەرىفەت ؟

نيوتون تەڭرىلىك ھوقوقىنى يېرىم سائەت ئىجارە ئېلىف ،يەرشارىنى بىرنى ئىتتىرىۋەتكەن دەل شۇ تارىخى مىنۇتلاردا بۇ مۆجىزاتنى ئېنگىلىسجان ھە ئېنگىلىسجان بىر تۈف گلاس دەرىخىنىڭ كەينىدە تۇرۇف كۆرۈف قافتۇدە ،شۇنىڭ بىلەن بۇنى كۆرگەن ۋاشنگتون قارىم ئاچچىقىغا فايلىماي دادىسىنىڭ ئۇنىمىغىنىغا قاتتىق نارازىلىق بىلدۈرگەن ھالدا ،قولىدىكى فالتا بىلەن گلاسنى چېفىف يىقىتىفتۇ . گلاس چېفىلماق بىلەن تەڭ ،ھە ئەمدى ۋاشىنگتونجاننىڭمۇ داڭقى چىقىدۇ دە ،ئاز كۈن ئۆتمەي ئەمرىكىغا دازۇڭتۇڭ بوفتۇ . ماياقتىكى ئۈچ فادىشاھلىقنى بىرلەشتۈرۈلمەي بېشى ئوچاق ئېتىلىف تۇرغان جۇگېلىياڭ قاراجىم جاڭفيىني چاقىرىف كىرىف قاتتىق تەنقىد قىفتۇ .ھەي فالاكەت چۇفۇر ساقال يۇڭلۇق قافاق ،قالاڭ خەق كىچىككىنام فالتىسى بىلەن نى ئىشلارنى قىۋاتىدۇ ،سىلىگە قايلىسا سېفى يوق كەتمەندەك فالتىنى كۆتۈرىۋالغان بىلەن تۇرغان تۇرقىلىرى ئەنە ،دېسىلىيا ،قاندا قىلاتىمىز دەفتۇ . بۇنىڭدىن ئىزا تاتىف كەتكەن جاڭفىي قارىم .بۇنچىۋالا فاراڭلاردا ئۆزىنىڭ تىلغا ئېلىنىفمۇ قويۇلمىغانلىقىغا قورسىقى كۆفكەن قالغان گالىلىي فالىلىي تاراق تۇرۇق چوڭ كىچىك ئالىملار ھار ئەف كېتىف ،بىرلىشىف تەڭرىگە خەت تۇتۇفتۇ ،ماشتا بىرنى كۆزگە ئىلمىدى ،ئىگە بولۇف بېرىڭ دەف .تەڭرى ئاندىن كۆتۈرە بۇ داڭقان فۇتى دەتالىشىڭنى ،بېشىمنى چۆرگىلىتىشمەي ،ھە قايسىڭغا قانچە قېتىم دەيمەن ھە ؟ ماتېماتىكىلىق  ھېسافلايدىغان بىرنېملەرنى كۆتۈرۈف كېلىشمە دەف .ھە ،سەن نېمە يانا چىرايىڭنى مانـــــــــــــــــــــــــــــداق قىلىف چەكچىيىف قارايسەن ماڭا ؟ چىقىشە ۋا تەڭرى ئىشخانامنى يىف بازىرى قىلىشماي دەف ھەيدەف چىقىرىفتۇ .مۇنۇ تۇرفانلىقمىدۇ فىچانلىقمۇ ئەيتاۋۇر ئالەملىك ئىجازەتنامىلەرگە ئىگە بولمىش ئىلىمدار ھاكىم ئەفەندى يەرلىك داڭلىق مەھسۇلاتلاردىن مەجۈن قاتارلىقلارنى سەل نورمىدىن ئېشىفراق يەف قېلىپ ،قايناقسۇ قايناتقىلى ئوتۇن يوق تېزەك تەرگىلى ماڭغان يولدا بۇ جېدەلنى ئۆز كۆزى بىلەن كۆرۈف كەفتۇدەك دەيدۇ .بۇنى ئاڭلىغان بىرلەشكەن دۆلەتلەر تەشكىلاتىدا نەچچە كۈندىن قاتتىق  كۈلۈف كېتىف  ئېسىگە كېلەلمەيۋاتقانلار سانى داۋاملىق ئېشىۋېتىفتىمىش.......
