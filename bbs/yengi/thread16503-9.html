بىيز ئادەتتە فىزىكىلىق مەسىللەرنى تەھلىل قىلغاندا ماكان،زامان ۋە جىسىم ئۇقۇمىنى بىرلىكتە تەھلىل قىلىمىز،نيوتون قانۇنلىرىدا بۇ ئۇقۇملار يوق،مەسىلەن،نيوتون قاچان ،نەدە تۇرۇپ ئالەمنىڭ  چوڭ بىر سائەتكە ئوخشايدىغانلىغىنى بايقىغان،بۇ ۋاقىتتىكى جىسىم -دەل چوڭ-كىچىكلىگى ئالەم بىلەن باراۋەر بولغان چوڭ سائەتمۇ ياكى ئەمەسمۇ؟ بۇنىڭ قانداق دەلىللىرى بار ؟،ۋاقىت يەنى زامان بولسا نيوتون دەۋرىدىكى قاراش بويىچە  تەڭرىنىڭ  ئالەمنى ياراتقان ۋاختىغا 5689يىل بولغان،لېكىن بۇ ۋاقىت نۇرغۇن يىللار بۇرۇنلا ئاغدۇرۇلۇپ كەتكەن،،ھازىرقى ۋاقىت بويىچە تەخمىنەن 13مىليارت700مىليون يىلدىن 320نەچچە يىل ئازراقلا ۋاقىت بولغان،مانا بىز بۇ پەرققە ئاساسەنمۇ نيوتوننىڭ تامامەن خاتالاشقانلىغىنى بىلەلەيمىز،يەنە كېلىپ ئۇنىڭ  ئالەملىك تارتىشىش كۇچى دىگىنى يەنە شۇ زامان بىلەن باغلىنىدىغان كۇچ،بۇ كۇچمۇ نيوتون تەڭرىنى چاقىرىشتىن بۇرۇن يوق ئىدى،يەنى ئالەم پەيدا بولۇپ 5689يىلغىچە ئالەملىك تارتىشىش كۇچى مەۋجۇت ئەمەس ئىدى دەپ قارىلىدۇ.نىمە ئۇچۇن بۇ قاراش شۇنچە ئۇزۇن يىلدىن بۇيان قانۇن تەرىپىدىن قوغدۇلىدۇ؟بۇ قانۇننى يەنە كىملەر قوغدايدۇ؟نىمە ئۇچۇن قوغدايدۇ؟ئەڭ مويىمى بۇ توردا نيوتوننى ياقلىغۇچىلارغا يۇقۇرى مۇئاش،يۇقۇرى تەمىنات بېرىلىدۇ،بۇنىڭدا 300يىلدىن ئاشتى؟بۇمۇ بىر مەبلەغ ئىسىراپچىلىغى؟بىز قەغىز يولۋاس بىلەن ھەقىقى يولۋاسنى پەرق قىلىشىمىز كېرەك.ھەقىقى يولۋاسنى قەغەز يولۋاس دەپ قاراپ قالساق،ھەقىقى يولۋاسقا ئەلۋەتتە يەم بولۇپ كېتىمىز.ئەلۋەتتە بىزلا ئەمەس،

كۇرۇشكە بولۇدىكى نيوتون قانۇنلىرىنى تارىختىن بۇرۇن ھىچ بىر كىشى تۇلۇغى بىلەن چۇشەنگەن ئەمەس،نيوتون قانۇنلىرى چىققاندىن بۇيان نيوتون تەڭرىنىڭ ئورنىغا ئالماشقان،بىز پەقەت دەرسلىكلەردىلا بولۇپمۇ دولىتىمىزنىڭ ئەمىلى دەرسلىكلىرىدە خۇدا يوق دەيمىز،لېكىن كوڭلىمىزدە خۇدا بار دەپ چۇشۇنىمىز،بۇمۇ بىر توقۇنۇش،بىز كۇنۇپ كەتتۇق،لېكىن بۇنى خۈۋكىڭدەك پەقەت خۇدانى ئېتىراپ قىلمايدىغان ئالىملار بولسا،سىپاگەرچىلىك بىلەن {نيوتون نەزىريەلىرىگە تەڭرىنى ئارىلاشتۇرغان،ھازىر تەڭرىنى ئارىلاشتۇرمايمۇ ئالەمنىڭ قانۇنيەتلىرىنى چۇشەندۇرگىلى بولىدۇ}دەپ قويغان،لېكىن قارىسىڭئىز{ۋاقىتنىڭ قىسقىچە تارىخى}،{شاكال ئىچىدىكى ئالەم}دىگەن كىتاپلىرىغا {تۇنجى تۇرتكە كىتاپلىرى} دىگەن سۇزنى قوشۇپ قويغان،خۇدانى ئېتىراپ قىلمايدىغان ئادەم يەنىلا خۇدانى ئۈز نەزىريەلىرىدىن چۇشۇرمىگەن}،دۇنيا بويىچە 90نەچچە دولەتنىڭ ئاكادىمىگى،پەخرى ئاكادىمىگى ،پەخرى دوكتۇرلىغى،پەخرى فېرافېسسورى دىگەندەك شەرەپلەرگە ئېرىشكەن ئادەم ھازىر دىگەن سوزىنى ھازىر ئىنكار قىلىدۇ،شۇنىڭ ئۇچۇن دەرسخانىدا دىيىلگەن سۇزلەرنىڭ راسلىغىغا كاپالەتلىك قىلىش ناھايىتى تەسكە توختايدۇ،بۇ خۇددى يولۋاس تىرىسىدىن جۇۋا كىيىپ تۇرۇپ،يولۋاس قاتارلىق ياۋايى ھايۋانلارنى ئاسىراش تەشكىلاتىغا ئەزا بولۇپ كىرگەندەكلا بىر ئىش،دۇنيادا ھەممىلا ئادەم دىگۇدەك يالغان سۇزلەشكە باشلىغاندا،ھەقىقى راس گەپنىڭ بازىرى يەنىلا چىقىدۇ.

مىنىڭ بۇ يەردە ئوتتۇرىغا قويغان ''مەڭگۈلىك دىۋىگاتىل ''دىگىنىم،ھەقىقى مەنىدىكى دىۋىگاتىل ئەمەس.يەنى بىر جىسىم  سىرتقى كۈچنىڭ تەسىرىگە ئۇچرىمىغان ئەھۋال ئاستىدا ھەركەت قىلامدۇ قىلمامدۇ؟ئەگەر ئۇ ھەركەت قىلمىسا دەل نىيۇتۇننىڭ بىرىنچى ھەركەت قانۇنىغا چۈشكەن بولىدۇ.ئەگەر ھەركەت قىلسا دەل ئەكسچە بولىدۇ.مىنىڭ بۇ يەردە كۆزدە تۇتقىنىم مەڭگۈلىك ھەركەت قىلىدىغان جىسىم مەۋجۇتمۇ ئەمەسمۇ؟ئەگەر مەۋجۇت بولمىسا پۈتكۈل ئالەمدىكى ئالەم جىسىملىرى نىمىنىڭ تەسىرىدىن ھەركەت قىلىدۇ.ئالەمدىكى ھەركەت قىلىۋاتقان جىسىملار مەڭگۈلىك ھەركەتكە كىرەمدۇ ياكى مەلۇم ۋاقىتتىن كىيىن ھەركەتتىن توختامدۇ؟ مەن بۇ مەسىلىنى قايسى بىر ماتىريالدىن نەقىل كەلتۈرمىدىم.

                                                                                                                                       {1}نيوتون  شۇ دەۋىردە پۇتكۇل ئەلەمنى غايەت زور سائەتكە ئوخشاشقان،بۇ سائەتنىڭ سىرتىدا ھىچ نەرسە يوق،خىرىستىئان دىنىنىڭ نەزىرىدە بۇ سائەتنىڭ سىرتىدا تەڭرى بار،نيوتوننىڭ چوڭ سائەت نەزىريەسىنى ئوتتۇرغا قويغان ۋاختى 1685يىللار ئەتىراپىدا بولۇپ،بۇنى جىسىم ،ماكان،زامان نوختىسىدىن چۇشەندۇرسەك،جىسىم دەل چوڭ ئالەم،{يەنى چوڭ سائەت،}،سائەت تېخى ھەرىكەت قىلمىغان،لېكىن ئالەم ھەرىكەت قىلغىلى خىرىستىئان دىنىنىڭ شۇ ۋاختىدىكى نەزىريەسىگە ئاساسلانغاندا،5689يىلدەك بولغان،بۇنى زامان دەيمىز،نيوتون دەل بۇ ۋاقىتتا تەڭرىنى چاقىرغان،بىزنىڭ قارىشىمىزدا تەڭرى نيوتونغا ياردەم قىلمايدۇ،لېكىن 300يىلدىن كوپەرەك ۋاقىتتىن بۇيان پۇتكۇل ئىنسانىيەت تەڭرى نيوتونغا ياردەم قىلغان دەپ چۇشەنگەن،مەسىلىنىڭ ئەڭ ئاچقۇچلۇق يېرىمۇ مۇشۇ،تەڭرى ئەمەس،ئادەتتىكى ئىنسانلار بىلىدىغان 5689يىلنىڭ ئۇتۇپ كەتكەنلىگىنى بىلمەسلىگى مۇمكىنمۇ؟                                                                                               {2}  سىزنىڭ جىسىم دىگىنىڭئىز ئادەم ۋە ھايۋاناتلارنى كۇزدە تۇتامدۇ -تۇتمامدۇ؟ئۇلارنى فىزىكا جىسىم دەمدۇ- دىمەمدۇ؟بىز كۇرگەن فىزىكا ماتىرياللىرىدا جىسىم دىيىلىدۇ،قارىسىڭئىز ئادەم ۋە ھايۋاناتلار سىرتقى كۇچ بولمىسىمۇ ھەركەت قىلىدۇ،شۇنىڭ ئۇچۇن بۇ جىسىمنىڭ ھەركىتى نيوتون قانۇنلىرىغا چۇشمەيدۇ،{پۇتكۇل ئالەمدىكى ئاسمان جىسىملىرى نيوتون دەۋرىدە ئالەملىك تارتىشىش كۇچىنىڭ تەسىرىدە ھەركەت قىلاتتى، نيوتوننىڭ تەڭرىنى چاقىرىشىدا،دەل مۇشۇ ئالەملىك كۇچىنى بىرەرمىكىن دەپ چاقىرغان،ئەلۋەتتە ،قەھۋە ئىچىشكە چاقىرمىغان،يەنى ئالەمنىڭ ھەركەت قىلغانلىغىنى ماھىيەتتە سائەتنىڭ ھەركەت قىلغانلىغىى بولۇپ ھىساپلىنىدۇ،كۇرۇشكە بولۇدىكى،بىز نورمال لوگىكا بىلەن قارىساق،ئالەم ھەرىكەت قىلاتتى،بۇ يەردە تارتىشىش كۇچى ھىچكىمنىڭ ئىگىدارچىلىغىدا ئەمەس،نيوتون دەل تەڭرىنى چاقىرىش ئۇيۇنىنى ئويلاپ تېپىپ،بۇ كۇچنى ئۈزىنىڭ سەخسى كۇچى قىلىۋالغان،ھازىر كۇچنىڭ بىرلىگى دەل نيوتون،بۇنى ئىلىم -پەن ساھاسىدىكى زور ئالدامچىلىق دەپ قارايمىزمۇ ياكى تەڭرى نيوتونغا ياردەم قىلغان دەپ قارايمىزمۇ؟بىزغۇ مۇسۇلمان،بىز تەڭرىنى بىز بىلمىگەن نەرسىلەرنى بىلىدۇ دەيمىز،لېكىن فىزىكىدا بۇنىڭ تەتۇرسى،ئوقۇتقۇچىلار ئوقۇغۇچىلارنى بالىلىرىمىز دەپ قارايدۇ،شۇ سەۋەپتىن بالىلارنىڭ ساختا تەربىيەدە بولۇپ قېلىشىنى ئەلۋەتتە خالىمايدۇ،لېكىن ئۇلارغا ئامال يوق،بۇنى ئۇلارنىڭ ۋىژدانى كۇتۇرمەيمۇ ئامالى يوق،بىز نيوتوننى ئاقلاپ چىقالايدىغان پاكىتنى تاپالماي قالدۇق،لېكىن بەزى تورداشلار بۇ ماقالىنى ئوقۇماسلىق،تەھلىل يازماسلىق،پىكىر بەرمەسلىك تەكلىۋىنى قويۇپتۇ،نىمە ئۇچۇنلىگىنى بىلمايلا قالدىم،راستىنى دىگەندە،تورداشلارنىڭ قارشى پىكىرلىرىنىڭ بولۇشى نورمال ئەھۋال،لېكىن سىز ئوتتۇرغا قويغان مەسىلىگە ئاسان جاۋاپ بەرگىلى بولمىدى،                                                                  {3}سىز ئوتتۇرغا قويغان مەسىلە بەكلا مۇرەككەپ بولغاچقا بىر كۇن قايتا قايتا ئويلاپ،مانا جاۋابىنى يازدۇق،سىزنىڭ ،مېنىڭ ،تۇققانلىرىمىزنىڭ ،دوستلىرىمىزنىڭ .....پەرزەنتلىرىنى مەكتەپكە ئاپىرىپ باشقىلارنىڭ ئالداپ كېتىۋاتقانلىغىنى،بۇ خىل ئالدامچىلىقنىڭ قانۇننىڭ قوللۇشىغا ئېرىشىۋالغانلىغىنى كۇرۇپ،يەنىلا ھەق سۇزلەيلى دەپ قارىدۇق،{دۇنيا پەلساپە گۇرۇپىسى}دىن ھەر كۇنى دىگۇدەك ئۇچۇرلار كېلىپ تۇرۇدۇ،دۇنيا بويىچە نۇرغۇن ئادەم بۇ ئىشقا قىزىقىپ قالدى،لېكىن ئېنگىلس تىلىنى ئۇلار بىلەن پىكىر ئالماشتۇرغۇدەك دەرىجىدە بىلمىگەچكە،قىيىنچىلىق كوپ،بىز ھازىر نيوتون،ئېينىشتىيى،سىتىفون خوك قاتارلىق ئالىملارنىڭ ئەجەللىك خاتالىقلىرىنى بايقاپ قالدۇق،يەنە كېلىپ بۇ قاراشلىرىمىزنى دۇنيانىڭ ئالدىنقى قاتارىدا تۇرۇدىغان ژورناللارغا ئەۋەتىپمۇ بولدۇق،دۇنيادا ھەممىلا ئادەمنىڭ باشقىلارغا ياخشىچاق بولۇش ئۇچۇنلا ياشىمايدىغانلىغىغا ئىشىنىشكە بولىدۇ.مۇشۇ ئايدا ئىمتىھان بار بولغانلىغى ئۇچۇن تازا دەل كەلتۇرۇپ جاۋاپ بېرەلمىدىممىكىن دەيمەن،چۇنكى ئىمتىھاندىن ئۈتمەي بولمايدۇ،كەم قالغان يەرلەرنى كېيىن  يەنە يولۇقلاي.
