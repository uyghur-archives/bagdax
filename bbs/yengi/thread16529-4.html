    ئەلھاسىل شۇقەدەر يامان كورۇپ ھەبىس قىلماق ئۇچۇن ھوكۇمگە لايىق بولغان غۇجىنىياز.ھاجىلارغا 3000 تال مىلتىق بىلەن 300 ساندۇق ئوق ئىلتىپات قىلماقچى  بولغانلىغىڭىزنىڭ سەۋەپلىرى نىمىدۇ؟ دىمەك بىرتەرەپتىن تۇتۇپ ھەبىس قىلماقچى بولىسىز، يەنە بىر تەرەپتىن 3000 مىڭ تال مىلتىقلار ئىلتىپات قىلماقچى بولسىڭىز، ئەجىبا جىمىساردا ئوز ھەققىمىز بولغان 1800 تال مىلتىقلىرىمىزنى زورلۇق بىلەن تارتىپ ئالىسىز،ئەمدى 3000 مىڭ تال مىلتىق بىرىمەن دەپسىز،بۇ ئالدامچىلىقىڭىزغا نىمە دىمەك قانداق جاۋاپ قىلماق لازىم! مۇنداق قىلىشقا سىزنى نىمە مەزبۇر قىلدى؟ ياكى دىنداشلىرىڭىز سەپداشلىرىڭىزنى قان يىغلىتىپ زورلۇق بىلەن بۇلاپ كەتكەن مىلتىقلار بىلەن قۇراللاندۇرغان،ئەسىرگە چۇشكەن خەنزۇ ئەسكەرلەر زىنىچۇئەنزى سوقۇشىدا پولۇك-روتا كوماندىرلىرىڭىزنى ئولتۇرۇپ تاشلاپ دوشمەن بولغان شىڭ دۇبەن تەرەپكە ئوتۇپ قۇشۇلۇپ كەتكەندىن كىيىن دوس بىلەن دۇشمەننى پەرىق ئەتتىڭىزمۇ -يا! دىمەك مەزكۇر قۇراللار ئوز دىنداشلىرىڭىزنىڭ قولىدا ياكى ئامبىرىڭىزدا تۇرغان بولسا ئىدى، زىنچۇئەنزىدا بولغان ئىغىر كۇنلەر بولمىغان بولاتتى !ئەپسۇسكى،ئوز سەپداشلىرىڭىزغا  شۇقەدەر خائىنلىقلار قىلدىڭىز ۋە قىلماقچى بولىسىز، خاتىرىڭىزدە بولسۇنكى ،ئوز سەپداشلىرىڭىزغا قىلغان خىيانەتلىرىڭىزگە يارىشا دوشمەنلىرىڭىزدىن كوگۇلۇكنى ئەلۋەتتە كورىسىز ! جىق ياشلىق،گۇدەكلىك قىلدىڭىزۋە قىلماقچى بولىسىز، ئەنە شۇنداق ھىلە-مىكرلەر بىلەن بىزلەرنى تۇتماقچى بولسىڭىز ، ھەبىس قىلماقچى، ۋە ئولتۇرمەكچى بولسىڭىز،ئىنشا ئاللا.....يەرلىك ئىنقىلاپچىلار لازىم بۇلۇپ قالىدىغان كۇنلارمو بۇلوپ قالۇر. تەكرار ئاگا قىلىمىزكى ، بىر يۇرت ۋە مىللەت ئىگىسى بولغان يەرلىك ئىنقىلارچىلار ،سىز ماجۇڭيىغا قارشى ئەمەس ۋە دوشمەن ئەمەس، ئىھتىمال سىز بىزلەرنى شىڭشىسەي بىلەن ئىتىپاقلاشتى دىگەن بانا بىلەن گۇنا قويسىڭىز كىرەك . ئەمما سىز ماجۇڭيىڭ ياڭ سەنمۇ ئارقىلىق مەخپى ھالدا شىڭ دۇبەن بىلەن بۇلدورىلمىغان بىتىمنى بىزلەر ئاشكارە قىلدۇق، ئەمما بىزلەرنى سىزدىن ئايرىلىپ شىڭ دۇبەن بىلەن ئىتىپاق تۇزمەككە مەزبۇر قىلغان يىگانە سەۋەپلەر ،سىز ماجۇڭيىڭنىڭ خىيانەت. جىنايەت ۋە رەڭۋازلىغىڭىزدۇر،خاتىرىڭىزدە بولسۇنكى بۇ توغرىدا تۇكۇلگەن قانلار ۋە بۇندىن كىيىن تۇكۇلىدىغان قانلارنىڭ مەسئۇلىيىتى ۋە جاۋاپكارلىغى سىزدە،قىيامەت كۇنى ئاللا زولجالال ھوزۇرىدا سىز ماجۇڭيىڭ ئەلۋەتتە مەسئۇل بولىسىز.
   جەنۇبى شىنجاڭ يەرلىك ھوكۇمەت باشلىقى:غۇجىنىياز . مىڭگونىڭ 22 - يىلى 7 - ئاينىڭ 4 - كۇنى.
    شۇنىڭدىن كىيىن ماجۇڭيىڭ 3 پولۇك ئەسكەر ئاجىرتىپ  مافۇيون ئىسىملىك بىر ئادىمىنى بىركادا كوماندىرىغا تەيىنلەپ غۇجىنىياز.ھاجى قىسىملىرىغا قوغلاپ زەربە بىرىشكە ئەۋەتتى . مۇشۇ سەۋەپتىن غۇجىنىياز.ھاجى قىسىملىرى بىلەن ماجۇڭيىڭ قىسىملىرى ئوتتۇرىسىدا قارا شەھەر ، باش ئەگىم،بۇگۇر، كۇچا، باي، قارا يۇلغۇن، مارالبىشى، پەيزىۋات-چوكانيار قاتارلىق جايلاردا
چوڭ-كىچىك قاتتىق سوقۇشلار بولدى. شۇنۇڭ بىلەن بىر مەزگىل غۇجىنىياز.ھاجى قوشۇنلىرى ئۇرۇشلاردا مەغلۇپ بولۇپ قاچقان، ماجۇڭيىڭ قىسىملىرى غەلبە قازىنىپ قوغلاشقان ۋەزىيەت شەكىللەندى ،غۇجىنىياز.ھاجى قوشۇنلىرى قارا يۇلغۇندىن بىر سوقۇشتا مەغلۇپ بولۇپ ئۇچتۇرپان قاقشال  يولى بىلەن قەشقەرگە ئوتۇپ كەتتى.
    1934 - يىلى 1 - ئاينىڭ ئاخىرلىرى مافۇيۇئەن قىسىملىرى قەشقەردە مۇھاسىرىيە ئىچىدە قالغان ماجەنشىئاڭغا ياردەم بىرىش ئۇچۇن قەشقەرگە باستۇرۇپ باردى، بۇ چاغدا قەشقەردە تۇرىۋاتقان غۇجىنىياز.ھاجى ۋە سابىت داماللاملار شەھەردە ئۇرۇش قىلماسلىق تاكتىكىسىنى قوللىنىپ يەكەن-يىڭىسار تەرەپكە چىكىنىپ كەتتى،شۇ كۇنلەردە ئۇرۇمچى ھوكۇمىتى ، ئۇرۇمچىگە ھۇجۇم قىلىۋاتقان ماجۇڭيىڭ قىسىملىرىنى چىكىندۇرۇپ قوغلاپ كالمەكتە ئىدى ."ئالتايىسكا" تارباغاتايىسكا" نامىدا شىنجاڭغا ياردەمگە كەلگەن سىۋىت قىزىل ئارمىيەسىنىڭ ئەجەللەك زەربە بىرىشىگە بەرداشلىق بىرەلمەي ماجۇڭيىڭ قەشقەرگە قىچىپ كىلىپ . ئارقا تەرەپتىن شىڭ دىبەن ئەسكىرى قوغلاپ كىلىۋاتقان. ئالدى تەرەپتە غۇجىنىياز.ھاجى قىسىملىرى تۇسۇپ تۇرىۋاتقان قىيىن ئەھۋالدا قالدى،مۇشۇنداق ئەھۋال ئاستىدا ئۇ سىۋىت ئىتىپاقىنىڭ قەشقەردە تۇرۇشلۇق ئەلچىخانىسىنى ۋاستە قىلىپ شىڭ دۇبەن ھوكۇمىتى بىلەن كىلىشىم يازدى. كىلىشىمدە تۇزۇلگەن بەلگىلىمىلەر بويىچە ماجۇڭيىڭ بىر قىسىم ئادەملىرى بىلەن سىۋىت ئىتىپاقىغا كەتتى. ئۇنىڭ قالدۇق قىسىملىرى دىۋىزىيە كوماندىرى ماخۇسەننىڭ قۇماندانلىقىدا خوتەن ۋىلايىتىدە تۇرىدىغان بولدى،بۇ چاغدا غۇجىنىياز.ھاجى قىسىملىرى مەكىت ئارقىلىق دەريا ياقىلاپ ئاقسۇغا كەتتى.
