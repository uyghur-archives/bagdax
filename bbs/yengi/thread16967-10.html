مەھمۇد بوۋىمىز تۈركلەر توغرىسدا‹‹ تۈركى تىللار دىۋانى ››دا مۇنداق يازغان :


1.  دىۋاننىڭ باش قىسمىدا تەڭرىگە ۋە ھەزرىتى مۇھەممەد پەيغەمبەرگە ھەمدۇ سانا ئوقۇپ بولۇپلا مۇنداق يازىدۇ: ‹‹ مەن تەڭرىنىڭ دۆلەت قۇياشىنى ( ئىلمى نۇجۇم) تۈركلەر بۇرجىدا تۇغدۇرغانلىقىنى ۋە پەلەكنىمۇ تۈركلەرنىڭ زېمىنى ئۈستىدە چۆرگىلەتكەنلىكىنى كۆردۈم ، تەڭرى ئۇلارنى تۈرك دەپ ئاتىدى  ۋە سەلتەنەتكە ئىگە قىلدى،دەۋرنىڭ خاقانلىرىنى تۈركلەردىن قىلدى ۋە زاماننىڭ ئىختىيارىنى شۇلارنىڭ قولىغا تاپشۇردى،ئۇلارنى ئىنسانلارغا باش قىلدى››
دىۋان تۈركى تىللىرى بىلەن ئەرەپ تىلىنى سېلىشتۇرۇپ يېزىلغان ئىدى، بوۋىمىز يەنە يېتىشكەن ئەرەپ تىلشۇناسى ئىدى. بۇ دەۋردە تېخى ئوسمان تۈرك ئېمپىرىيىسى  بارلىققا كەلمىگەن ئىدى، پەقەت سالجۇقىلار بار ئىدى.

2. ئىشەنچلىك بىر بۇخارالىق ئالىم ( بۈيۈك ھەدىسشۇناس ئىمامى بۇخارىمۇ بۇخارالىق ئالىم ئىدى، ئەرەپ ئەمەستى) بىلەن نىشاپۇرلۇق باشقا بىر ئالىمنىڭ پەيغەمبىرىمىزدىن نەقىل كەلتۈرۈپ ئېيتقان تۆۋەندىكى سۆزلىرىنى ئېنىق ئاڭلىغان ئىدىم ‹‹ پەيغەمبىرىمىز قىيامەت بەلگىلىرىنى ، ئاخىر زامان پىتنىلىرىنى ، ۋە ئوغۇز تۈركلىرىنىڭ مەيدانغا چىقىدىغانلىقىنى سۆزلىگەندە ‹ تۈرك تىلىنى ئۆگىنىڭلار،چۈنكى ئۇلارنىڭ ھۆكۈمرانلىقى ئۇزاق داۋام قىلىدۇ › دەپ ئېيتقان ئىكەن.››.

3. تۈرك- تەڭرى يارلىقىغۇچى نوھنىڭ ئەۋلادىنىڭ ئىسمى،بۇ تۈرككە تەڭرى تەرىپىدىن قويۇلغان ئىسىم.ئۇنىڭ ئەۋلادىمۇ تۈرك دەپ ئاتىلىدۇ.

4.بىز تۈرك دىگەن ئىسىمنى ئۇلۇغ تەڭرى ئۆزى قويغان دىدۇق،چۈنكى قەشقەرلىك خەلەف ئوغلى تەقۋادار ئىمام ئۇستاز ھۈسەيىننىڭ ئىبىن ئەل غەرقى دىگەن ئادەمدىن ئاڭلاپ ماڭا ئېيتىپ بېرىشىچە ، ئىبىن ئەبىد دۇنيا دىگەن نام بىلەن تونۇلغان جەرجارالىق ئۇستاز ئەبۇبەكرى مۇفىد ئاخىر زامان توغرىسىدا يازغان بىر كىتابىدا ، پەيغەمبىرىمىزدىن نەقىل كەلتۈرۈپ مۇنداق بىر ھەدىسنى رىۋايەت قىلغان :‹‹ ئۇلۇغ تەڭرى ئېيتىدۇكى ، مېنىڭ بىر تائىپە قوشۇنۇم بار، ئۇلارنى تۈرك دەپ ئاتىدىم  ، ئۇلارنى كۈنچىقىشقا ئورۇنلاشتۇردۇم، بىرەر قوۋمغا غەزەپلەنسەم تۈركلەرنى ئۇلارنىڭ ئۈستىگە ئەۋەتىمەن.››

بوۋىمىز دىۋانۇ لۇغاتىت تۈركنى باغداتتا ئابباسىلار ئوردىسىدا يېزىپ چىققان ۋە پەيغەمبىرىمىزنىڭ مۇقەددەس ئورنىدا ئولتۇرغان ھاشىم ئۇرۇقىدىن بولغان ئابباسىلار خەلىپىسى ئەبۇلقاسىم ئەلمۇقتەدى بىئەمرۇللاھ ئابدۇللا بىن مۇھەممەد بىن ئابدۇللاھ ئەل ئابباسىيغا ( خەلىپە قائىم بىئەمرۇللاھنىڭ نەۋرىسى ) غا تەقدىم قىلغان !

رەھمەتلىك  مومام  مۇنداق بىر رىۋاىيەتنى دەپ بىرىدىغان ........
بۇرۇن  ئىسكەندەرنىڭ  زامانىدا  يەجۇجى  مەجۇجىلەر  دىيارىغا  كەپتۇ  ،يۇقارقى تورداشلار دىگەندەك  سىپىل سوقماقچى بولۇپتۇ  ،ئىسكەندەرنىڭ ئەسكەرلىرى  پولات  تاۋلاش ۋە بىناكارلىق ئۇستىكارلىرى ئىكەن ،ئەمما ئۇنچە  ئۇزۇن ۋە ئىگىز  سىپىلنى  تىز سۇرئەتتە  سىلىشقا  كوپلەپ  ئادەم كۇچى كىتىدىكەن ،بۇنى قانداق ھەل قىلىش كىرەك ؟ ئاخىرى تىخنىك ئىسكەندەردىن ،قۇللار-مەدىكارلار  ھەقىقەتكە قايىتقان بىر تۇركۇم يەجۇجى -مەجۇجىلاردىن   بولىدىغان  بولۇپتۇ ،ئىش باشلىنىپتۇ ،ئادەم بەك كوپ ئىكەن ،سىپىل پۇتكەندە  ئىكەندەر  راۋاققا چىقىپ  بۇ مەدىكارلارنى  شۇنداق  كوزدىن كوچۇرۇپتۇ ،ۋە ئىش بىشىدىن  سوراپتۇ ،بۇنچە كوپ ئادەملەرنىڭ  يىمىكىنى  -تامىقىنى قانداق  ھەل قىلدىڭ ؟ئۇسسۇزلۇقىنى  قانداق قاندۇردۇڭ ؟؟؟؟
ئىش بىشى مەدىكارلارنى يىغىپتۇ ۋە ئىسكەندەرنىڭ ئالدىدىن ئوتكۇزۇپتۇ ،ئىسكەندەر  شۇنداق سەپ سىلىپ قارىسا  بۇ مەدىكارلارنىڭ  قولىدا  ئىككى  تال  چوكا  ،دۇمبىسىدە  قاچا  ،بىلىدە  سۇ ئىچىدىغان قاپاق تۇرغان ،ئالدىدىكى ھالىدىن كىتىپ  يىقىلىپ قالغانلىرىنىڭ  جەسىدىنى  كىيىنكىلەر  ئولاش-چولاش پاك-پاكىز   
يىىۋەتكۇدەك ،ئەيھونناس  !!!!ئىسكەندەرنىڭ چىكىسىدىن  تەر قۇيۇلۇپتۇ ،ۋۇجۇدىنى  تىتىرەك بىسىپتۇ .........
ئىسكەندەر كىتەر  ۋاختىدا   ھىلىقى  يەجۇجى  ئىش بىشى  ئىسكەندەرنىڭ ئالدىغا كىلىپ  :‹‹ئى  جاھاننىڭ سەركەردىسى -پاسىبانى  بىز ساڭا  شۇنچە كوپ  ئىشلىدۇق ،بىزنىڭ  ئىش ھەققىمىز قىنى ؟؟››دەپتۇ ،ئىسكەندەر قوينىدىن بىر  كوك تاشنى  (كوك مەر-مەر تاشنى)   چىقىرىپ :‹‹ مۇنۇ سىنىڭ ئىش ھەققىڭ ،ئاخىر زاماندا كوك  تاشقا مىنگەيسەن ››  دەپلا  ئىتىپتۇ   ،...............
رىۋايەتلەردىن  قارىغاندا  ئۇ كوك  تاش  سەمەرقەنىتكە  چۇشكەنمىش ،،،،يەنە  بىر رىۋايەتلەرگە  قارىغاندا  ئۇ  كوك تاش خوتەنگە  چۇشكەنمىش
ئارخىلوگلار  ھازىرغىچە  ئۇ كوك تاشنى ئىزدەۋىتىپتىمىش  ، تاشچىلار مۇ   بوش كەلمەي  يورۇڭقاشنى  قىزىپ  سۇيىنى  تەتۇر ئاقتۇرۋىتىپتىمىش  
ھىلىقى  قۇرۇلۇشچى  يەجۇجىلەر  داۋاملىق  قۇرۇلۇش قىلىۋىتىپتىمىش ،،،ۋەزىپىسى تىخى  تۇگىمەپتىمىش ،ۋەزىپىنى تۇگەتكەندە  ئىسكەندەرنىڭ  بەرگەن  ھەققىگە  ئىرىشىپ كوك تاشقا  مىنىدىكەن دە ئاخىر زامان بولىدىكەن
