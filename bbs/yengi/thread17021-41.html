شەكىل نەزەريىسدىكى تەشەببۇسلارنى قوللايدىغان تەجىربە نەتىجىلىرى ئاز ئەمەس،ھازىر پەقەت بۇ تەجىربىلەرنى تەكشۈرۈپ كۇرىدىغان،توغرا-خاتالىغىنى قايتا دەلىلەپ ئىشلەپ باقىدىغان ئۇرۇن ۋە سالاھىيەتلىك شەخىسلەر چىقمايۋاتىدۇ.دىمەكچىمەنكى شەكىل نەزەريىسى ھەرگىز تەجىربە ئاساسى يوق نەزەريە ئەمەس.ئەقەللىسى 29-تىئورما تەجىربىسىنى بولسىمۇ  قايتا ئىشلەپ ئىرىشكەنلىرىنى ئوتتۇرىغا قويىدىغان ياكى بىر ئىلمى ئورگان ياكى بىر سالاھىيەتلىك ئىلمى خادىم ئوتتۇرىغا چىقمايۋاتىدۇ.مەن تېمىلىرىمدا ۋە بەزى بېۋاستە ئالاقىلىرىمدا بۇ تەجىربە ئاساسلىرىنى تىلغا ئالغان بولساممۇ ھازىرغىچە قۇلاق يوپۇرۇلۇپ كېلىۋاتىدۇ.تەجىربە پاكىتلىرىنى شۇنچە كۆپ تونۇشتۇرساممۇ يەنىلا نەتىجىسى يوق دىگەن گەپلەر بېسىقمايۋاتىدۇ.بەزىلەر مىنىڭ تېمىلىرىمغا ۋە ماڭا ئالاقىدار تېمىلارغا تېما بىلەن مۇناسىۋىتى بولمىغان يامان غەرەزلەرنى مەنبە قىلغان ئىنكاسلارنى يېزىپ قەستەن بۇزغۇنچىلىق قىلىۋاتىدۇ.ھەقىقى كەسىپ ئىگىلىرىنىڭ بۇ تېمىلارغا كىرىپ مۇھاكىمە ئېلىپ بېرىشىغا كاشىلا پەيدا قىلىۋاتىدۇ....دەپ كەلسەم گەپ كۆپ....ئالاقىدار ھەممە تەجىربىلەرنى تەكشۈرۈشكە ۋاقتى بولمىسىمۇ،پەقەت بىرلا تەجىربە نەتىجىسىنى پۈتۈن يەر شارىدىكى مەن يېزىۋاتقان بۇ قۇرلارنى ئوقۇيالايدىغان كەسىپ ئىگىلىرى ھازىرقى بار بولغان فىزىكا بىلىملىرى بىلەن چۈشەندۆرۆپ،مەن ئوتتۇرىغا قويغان 29-تىئورمىنى ئاغدۇرۇپ تاشلىيالامدۇ،قېنى ئۇرۇنۇپ باقمىسۇنمۇ؟!.ماۋۇ ئىنكاسنى قايتا چاپلايدىغان بولدۇم،مانا:

شەكىل نەزەريىسىنى تەتقىق قىلىشنىڭ  زۇرۇرىيىتى بار-يوقلىغىنى  ئەڭ ئاسان بىلىدىغان بىر ئۇسۇل بار.ئۇ بولسىمۇ 29-تىئورما تەجىربە نەتىجىىسىنى ھازىرقى فىزىكا بىلىملىرى بىلەن چۈشەندۆرۈش.چۈشەندۇرگىلى بولسا بۇ نەزەريىگە قىزىقمىساق،كۈچ ئويۇشتۇرۇپ ئىلگىرلەپ تەتقىق قىلمىساقمۇ بولىدۇ.چۈشەندۆرگىلى بولمىسا بىز بۇ نەزەريىنى ۋايىغا يەتكۈزۈپ،كۈچ ئويۇشتۇرۇپ تەتقىق قىلماسلىقنىڭ ھېچقانداق باھانىسىنى تاپالمايمىز.

29-تىئورمىچە ئېغىش ھازىرقى زامان فىزىكىسىنى چوڭ ئىسلاھاتقا،ھەتتا بىر قېتىملىق فىزىكا ئىنقىلاۋىغا ئېلىپ كىرىدۇ.بۇ مەن دەۋاتقان گەپ ئەمەس.ئەھۋالدىن خەۋەردار بەزى مۇتەخەسىسلەر دەۋاتقان گەپ.ھازىر بار بولغان بارلىق فىزىكا بىلىملىرى نۇقتىسىدا تۇرۇپ 29-تىئورمىچە ئېغىشنى چۈشەندۇرۆپ بەرگىلى بولمىسا دەل مۇشۇنداق بولىدۇ.


ئەڭ ئەقەللى ۋە ئاددى قەدەم شۇكى،ئالاقىدار ئورۇن ۋە شەخىسلەر ئالدى بىلەن تۆۋەندىكى تېمىدا تونۇشتۇرۇلغان تەجىربىلەرنى قايتا بېۋاستە ئۇزلىرى ئىشلەپ كۇرۈش.ياكى باشقىلار ئىشلىگەن تەجىربىنى نەق مەيداندا كۇرۈش.راسىت-يالغانلىغىنى بېۋاستە كۆزىتىش.تونۇشتۇرۇلغان تەجىربىلەر مانا ماۋۇ تېمىدا:
