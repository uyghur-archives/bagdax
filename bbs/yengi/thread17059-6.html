  مەن تونۇيدىغان تۇرسۇنكام ئىسىملىك 65 ياشلاردىكى بىر دىھقان ئاكىمىز بدر ئىدى . ئۇ كىشى بىر كۈنى مۇنداق بىر ئىشنى سۆزلەپ بەردى .
     بىر كۈنى بۇ كىشىگە سۇ تۇتۇش نۆۋىتى كىلىپ ( كەچ چۈشەي دىگەندە ) تازا ئالدىراش سۇ تۇتىۋاتقاندا كەنت ئىشخانىسىدىن ئادەم كىلىپ دەرھال كەنت ھويلىسىغا دۈجۈرنىلىك ( ئامانلىق قوغداش ) قىلىشقا بېرىشقا چاقىرىپتۇ . نىمىلا دىسە ھېلىقى ئادەم ئۇنىماي « سېكىرتار » نىڭ ئۆزىگە ئېيتىشنى دەپتۇ . ئۇ كىشى ئامالسىز سۇنى باشلاپ قويۇپ سېكىرررتارنى ئىزدەپ بېرىپ : جېنىم ئۇكام ، ماڭا بى سۇ تۇتۇش نۆۋىتى كەپتىكەن ، ئۆيدە خوتۇننىڭ مىجەزى يوق ياتىدۇ ، باللىرىم تېخى كىچىك ،( بۇ ئادەمنىڭ خوتۇنى تۇغماي كىيىن پەرزەنتلىك بولغان ) شۇڭا بى رۇخسەت قىيسلىرى سۇنى تۇتىۋىلىپلا كەلسەم بوپتىكەن . ھىلى تېشىپ كەتسە پاكەن تۆلەپ قالىمەن ...
    ___  نىمە ماڭا بۇندا دەپ بېرىسە ئاكا !! ما بى مىنىڭ ئىشىمما ! غاچ كەسەڭ كەل ، بوممىسا كاپالەت پۇلۇڭنى ئالىمەن دەپ ئويلىما !! گەپ شۇ !! ( ئىشىكنى يۈزىگە يېپىپ ئىشخانىغا كىرىپ كىتىپتۇ )  بىچارە بوۋاي ھۆڭرەپ يىغلاپ كىتىپتۇ . چۈنكى ئۇ سولتەك سېكىرتار 25 ياشلاردىكى نىمىمىش . ئامال يوق قايتا كەينىدىن كىرىپتۇ . مىڭ يالۋۇرۇپ ئاران ئىككى سائەتلىك سوراپ سۈيىنى تۇتىۋاپتۇ .
   كىيىن ماڭا دەپ بېرىشىچە __ :  شۇ چاغدا شۇ نائەھلىنىڭ كەينىدىن چىرىپ ئاشۇ ،  مىڭىسىگە پىچاقنى تىقىپ قوچىۋەتكىم كەپ كەتتى !__ دىدى .
   مانا مۇشۇنىڭدەك پارتىيىنىڭ مۇئاشىنى ھارام ئېلىپ خەلقنى قاقشىتىدىغان ئىبلىسلارنىڭ تاناۋىنى راسا تارتىپ ، خەلق ئۈچۈن خىزمەت قىلىشنى تازا بىر ئۆگىتىپ قويسا ، ئۇنىڭ ھېلىقى چوڭچىلىق كېسىلى ساقايغان بولاتتى .
     ئۇ بوۋاينىڭ دىيىشىچە تېخى ،ئەگەر كىم ئۇنىڭ چىشىغا تەگسە ئۇنى بىر ئامال قىلىپ « باپلارمىش » .
شۇڭا ھەر دەرىجىلىك رەھبىرىي كادىرلارنىڭ قەرەللىك قەرەلسىز ھالدا بۇنداق پارازىت مۇناپىقلارنى ۋاقتىدا بايقاپ قاتتىق بىر تەرەپ قىلىشىنى ، خەلققە خەلق ئۈچۈن خىزمەت قىلىدىغان ھەقىقىي كادىرلاردىن قويۇشىنى سەمىمىي ئۈمىد قىلىمىز ..

ماۋۇ بايدىن شۇنچە بايلىق چىقىدىكىنۇ مىنىڭ دادامنىڭ ئانامنىڭ دەيدىكەنسەنۇ نىمىگە ئىگە بولالايسەن ساڭا ماۋۇ ئاقسۇلىقكىن ماۋۇ بايلىقكىن دەپ يۈندىسنىمۇ بەمەيدۇ ھە قاچان ھەممىسنى ھىلىقى جانجىگەر جاڭ ۋاڭ دەيدىغان ئاكىلىرىڭغا تەۋە ھەقىچان ئۇلىغا ما بىزنىڭ باينىڭكىتى دىيەلمەيسەن ، بىز بارغۇ خوتۇننى ئىچكىرگە ئەچىقىپ بەرگەن يىرىمىز يوق لازىم بولسا ئەكەلگۈچىلىكىمىز با نەدىكى ئىككى تۆتتال يىزىنىڭ گىپىنى ئاڭلىۋىلىپ پۈتۈن قەشقەرگە ئاتما،ما خوتەنلىك سىلنىڭ خوتەنگىزە قەشقەلىكلام بىرىپ يەر ئىگەللىۋاپتىدە ماۋۇ ئاكىلىرىڭ ئىگەللىگەن يە كۆزەڭگە كۆرەنمەيۋاتامدا قەشقەلىق بىر ئۆي سالسا ماۋۇ ئاكىلىرىڭ نەچچە يۈز مو يەرنى ئىگەللەپ ئولتۇرىدۇ ئانىڭغا گەپ يوق شۇنداما يا ئانىڭ بىرنىمىسى يوغانراقمىكىن گەپ قىلغىلى بولمايدىغان سەت گەپ قىلماي دەپ تىم ئاشۇ، ماۋۇ بىزنىڭ قەشقىدىمۇ نى خوتەنلىكلەر با بىزمۇ قىرىندىشىمىز دەپ پاتۇرۇپ سىغىشىپ يۈرىۋاتىمىز ئاڭلاشتىڭما، بۇندىن كىيىن يۇرتۋازلىق قىلىشما جىما ھەقاسىڭ بى باشقا مىللەت بولمىغاندىكىن.
