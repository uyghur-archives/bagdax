  ‹‹ئەڭ ئاخىرقى كۆل ››ناملىق تېلىۋىزىيە تىياتىرى يېقىنقى زامان تارىخمىزدا ئىشلەنگەن مەيلى سۈرەتكە ئېلىش تېخنىكىسى ياكى ئاكتىيورلارنىڭ ئوبراز يارىتىش ماھارىتى جەھەتلەردىن بولسۇن بىر قەدەر ياخشى ئىشلەنگەن فىلىم .يېقىندا بۇ فىلىمنى تېلىۋىزوردا قايتا كۆرۈپ مەن ئىلگىرى بۇ فىلىم ھەققىدە ھېس قىلغانلىرىمنى تورداشلار بىلەن ئورتاقلىشىپ بېقىش نىيىتىگە كەلدىم .چۈنكى بۇ فىلىم تەڭرىتاغ كىنو سىتۇدىيەسىنىڭ يېقىنقى 10 يىل ئىچىدە ئىشلىگەن خېلى يامان تەسىر قوزغىغان ئەسەرلىرىنىڭ بىرى .ھالبۇكى بۇنداق ئۇيغۇر تۇرمۇشىغا بىر قەدەر ئەتراپلىق چۆككەن تېلىۋىزىيە تىياتىرلىرىنى ئىشلەش بۇ سىتۇدىيەگە ياكى باشقىلارغىمۇ ئانچە نېسىپ بولمىدى .شۇڭا بۇندىن كېيىنكى ئاكتىيورلار ۋە رېژېسورلارنىڭ بۇرۇنقى فىلىملەرنىڭ مۇۋەپپىقىيىتى ۋە مەغلۇبىيىتى ھەققىدە ئويلىنىش ھەم ھەل قىلىش چارىسىنى ئىزدەش مەجبۇريىتى بار .
       بۇ تېلىۋىزىيە تىياتىرىدا تەكلىماكان يېزىلىرىدىكى ئۇيغۇر تۇرمۇشىنىڭ جانلىق كارتىنالىرى ئەختەم ئۆمەرنىڭ ‹‹قۇرتلاپ كەتكەن كۆل››ناملىق پوۋىستىنى ئاساس قىلىپ تۇرۇپ ئىشلەنگەن .فىلىمدە رول ئالغان ئاكتىيورلارمۇ ئوبرازلارنى ياخشى يارىتىپ  ئۇيغۇر تېلىۋىزىيە تىياتىرچىلىقىنىڭ تەرەققىيات ئىمكانىيىتىنى كۆرسىتىپ بەرگەن .بۇ فىلىم يۇقارقىدەك ئالاھىدىلىكلىرى بىلەن تەرىپلىنىشكە ھەقلىق بولسىمۇ ئەمما فىلىم مەزمۇن جەھەتتىن بىرقانچە نۇقسانلاردىنمۇ خالى ئەمەس .
.    1.فىلىمدە قۇمقۇيرۇق ناملىق يېزىدىكىلەر سېسىپ كەتكەن كۆلنىڭ سۈيىنى ئىچىشگە رازىكى قۇدۇق قېزىپ چىقىرىلغان سۇنى ئىچىشكە قارشى چىقىدۇ .بۇ كىشىلەر بولسا پاكىزلىقنى ھەممىدىن ئەلا بىلىدىغان مۇسۇلمان خەلق ھەرگىزمۇ سېسىق نەرسىلەرگە ئۆزىنى ئۇرىدىغان ،سېسىق كۆلدىن مەدەت تىلەيدىغان  ياۋايىلار ياكى ھايۋانلار ئەمەس ئىكەن.تەكلىماكاننىڭ مەلۇم بىر كەنتىدە شۇنداق ئىش يۈز بەرگەن بولسا قېنى بىرەرسى چىقىپ پاكىتلىرى بىلەن ئوتتۇرىغا قويۇپ باقمىسۇنمۇ ؟!بۇ پۈتۈنلەي سەپسەتە بولماي نىمە ؟كىممۇ پاكىز سۇ تۇرغان يەردە مەينەت كۆلگە ئېسىلىۋىشنى خالىسۇن ...ئەلۋەتتە ئەدەبى ئەسەردە توقۇلما قىلىشقا يول قويىلىدۇ ،ئەمما توقۇلما چوقۇم تۇرمۇش چىنلىقىغا ئىگە بولغاندا ئاندىن پۇت تىراپ تۇرلايدۇ ،كۆرۈرمەننى تەسىرلەندۈرەلەيدۇ .فىلىمدە كۆل يۈزىدە ئەخلەتلەر لەيلەپ يۈرگەن ،كالا قويلارمۇ شۇ كۆلدىن سۇ ئىچىۋاتقان تۇرۇپ كىشىلەر بۇ كۆلدىن ۋاز كېچەلمەيدۇ .پاسكىنا سۇنىڭ زىيانلىقن ئىكەنلىگىنى بىلىش ئۈچۈن قانداقتۇر ئونۋىرسىتىت پۈتتۈرگەن زىيالى بولۇش كەتمەيدۇ .بۇنى ئەقلى ھوشۇ نورمال كىشىلەر بىلىپ يىتەلەيدۇ .ئەجىبا فىلىمدە 15 ياشلىق نۇسرەت بىلەن ئامانگۈل قاتارلىق ئۆسمۈر قىزلار بۇ نۇقتىنى تونۇپ يەتكەن يەردە ،شۇ يېزىدىكى باشقا كىشىلەرنىڭ ئەقلىنى شامال ئۇچۇرۇپ كەتكەنمىدۇ ؟ئۇلار نىمە دەپ كۆلگە ئېسىلىۋالىدۇ چوقۇم بىر سەۋەبى بولىشى كېرەققۇ .
      2.فىلىمدە يەنىلا دىندارلارنى خۇراپى ،يېزىلىقلارنى ئەخمەق كۆرسىدىغان مەدىنىيەت ئېنقىلاۋىدىن  قالغان فىلىم ئىشلەش ئۇسلۇبى ئىشلىتىلگەن .ئەختەم ئۆمەرنىڭ ئەسلى ئەسىرىدە ئۇيېزىدىكى كىشىلەرنىڭ روھىدىكى ئەگرى توقايلىقلار ،ئىجابى ھەم سەلبىلىكلەر نۇسرەتنى مەركەزقىلغان ھالدا ئوتتۇرىغا قويۇلغان بولسا بۇ تىلىۋىزىيە تىياتىرىدا كىشىلەرنى ئىككى قۇتۇپقا بۆلۈپ ياخشى يامان ئادەم ئوبرازى يارىتىلغان .بۇ سەنئەتنىڭ چىنلىقىغا يازغۇچى رېژىسورنىڭ ئوبراز يارىتىش خىسلىتىگە ئۇيغۇن ئەمەس .ئەسلى ئەسەردە نۇسرەتمۇ زىدىيەتكە ئىگە ئىجابى تىپ .ئەمما فىلىمدە بولسا ئۇ ھېچ نۇقسانسىز بىر زىيانكەشكلىككە ئۇچرىغۇچى .ئادەتتە يامان كىشىلەرنىڭمۇ ياخشى تەرىپى ياخشى كىشىلەرنىڭمۇ يامان تەرىپى بولىدۇ . كىشىلەر خاراكتېرىنىڭ بۇنداق ئىككى خىللا قېلىپتا بولمايدىغانلىقىنى يازغۇچى جالالىدىن بەھرام توردىكى سۆھبەتتىمۇ ئەسكەرتىپ ئۆتكەن .ئەجىبا ئۇ يېزىدىكى كىشىلەرنىڭ روھى دۇنياسىدىكى زىدىيەتلەر ،دۇنيا قارىشىدىكى ئۆزگىچىلىكلەر فىلىمدىكىدەك شۇنداق ئاددى ھەم ياسالما بولىشى كېرەكمۇ ؟!كىشىلەرنى بۇنداق قۇتۇپقا بۆلۈپ ئوبراز يارىتىش كىمگە كېرەك بولۇپ قالدى ؟!
     3.ئەختەم ئۆمەر يازغان ئەسلى ئەسەردە قۇدۇق قېزىش ئەترىتى يوق ،ئەسكەرلىكتىن كېلىپ يېزىغا قۇدۇق قېزىش ئەترىتىنى باشلاپ كېلىدىغان سىدىقمۇ يوق ،نۇسرەت بىلەن ئامانگۈلمۇ قۇدۇق قېزىش ئەترىتىنىڭ ئەتراپىدا لاغايلاپ يۈرمەيدۇ .نۇسرەت  بولسا چىنلىققا ئىگە يارىتىلغان جېنى بار ،يېزىلىقلارغا خاس خىياللىرى ھەم ئاجىزلىقلىرى بار مەسۇمە قىزنىڭ ئوبرازى .بۇ ئەختەم ئۆمەرنىڭ باشقا ئەسەرلىرىدىمۇ ئىزچىل قوللىنىپ كەلگەن تىرىك رېئالىزىملىق تىپ يارىتىش ئۇسۇلى .لېكىن فىلىمدىكى نۇسرەتنىڭ بىر قانچە تەرىپى ئەسەردىكىگە ئوخشايدىغانلىقىنى ھېساپقا ئالمىغان نۇرغۇن يەرلىرى يەنىلا كونا ھېندىستان فىلىملىرىدىكى باش قەھرىماندەك كىشىلەرنىڭ ھېسداشلىقىنى قوزغاپ تەسىرلەندۈرۈش ئۈچۈن ئوتتۇرىغا چىقىرىلغان سۈنئى پىرسوناژدىن ئىبارەت .ئەسلى ئەسەر بىلەن فىلىمنى كۆرگەنلەرگە پەرىقنىڭ قانداقلىغى مەلۇم.
     قىسقىسى فىلىم يۇقارقىدەك سېسىقچىلىقنى ئۆزىگە ھەمرا قىلغان ،بۇ بابكارنىڭ كۆلىدىن چىققان پۇراقنىمۇ بېسىپ چۈشىدىغان مەنىۋى ئەخلەتلەر ئىدى .فىلىم ئىشلەنگەندە پۈتۈنلەي تەشۋىقاتقا ئېسىلىۋالمىغان ،ئەختەم ئۆمەرنىڭ ئەسلى ئەسىرىنى ئاساس قىلغان بولسا ئەلۋەتتە‹‹ قۇدۇق قېزىش ئەترىتى ››نىڭ ئېسىل خىسلەتلرىىنىمۇ گەۋدىلەندۈرۈش بىلەن بىرگە يېزىدىكى خەلقنىڭ سېسىق كۆلگە ئېسىلىۋىلىشىغا بىر سەۋەپنى قوشۇپ قويغان بولسا .نۇسرەت ئامانگۈل قاتارلىق يېزا قىزلىرىنىڭ ئۆزىگە خاس ئىندۇۋۇداللىقىنى ۋە كەمچىلىكلىرىنىمۇ قوشۇپ ئىپادىلىگەن بولسا ،فىلىمدىكى دىندارلار پۈتۈنلەي يامان قىلىۋىتىلمەي ئۇلاردىمۇ ئېنسانغا خاس ئارتۇقچىلىق ۋە كەمچىلىكلەر بولغان بولسا بۇ تېلىۋىزىيە تىياتىرى تېخىمۇ تۇرمۇشچانلىققا ئىگە رېئالىزىملىق خۇسۇسيىتى يۇقىرى ئېسىل تېلىۋىزىيە تىياتىرى بولۇپ چىققان بولاتتى .
