يوللىغان ۋاقتى 4 كۈن ئالدىدا |ھەممە قەۋەتنى كۆرۈش
شىمال تەتۈر،غەرپ ئوڭ،جەنۇپ ئوتتۇرىدا،شەرق تۆۋەندە،مەن نەدە،نىمىشقا ئۆلگۈم كەلدى،ھاراق بارمۇ ئىچىشىپ بىرەي،نىشە بارمۇ چىكىشىپ بىرەي،يىمىگەن ئىشەك بىلەن چوشقىنىڭ گۆشى قالدىمۇ،دىيارىمدا مۇسۇلمانچە تاماق ئىزلەۋاتىمەن،ئوغىرى پاك،ئىمام گۇناھكار،قۇلاق پاڭ،كۆز قارىغۇ،ئۆي ياتاق،ياتاق ئۆي،كىنشكىسى يوق شوپۇر يولدا،رەسمىيەت تولۇق يول يوق،مەن شوپۇر ئەمەس،قىزى نەدە،ئانىسى نەدە،بالىسنى جىنم بالام دەپ مىھىرى بىلەن سۆيۈپ نەۋىرىسى بولغاندىمۇ 20كويغا ئىپپىتىنى ساتقان بۇزۇق رەسۋالار،ئۇراي دىسە قوغىدىلىدىغان ھايۋان قاتارىدا،جان دىگەن شۇنداق قىينالدى،ئە سكىنىڭ گىپىنى ئاڭلاپ چىدىمىدىم،ماڭىمۇ گۇناھ يىزىلدى،ماڭا ھىچكىم ئىشەنمىسۇن،يەرنى قىغلاپ تىرىغلى خوش ياقمىسا،ئاھ دىھقىنىم دەپ جەرىمانە ئالسا دىھقانمۇ گاچا ،گاچا بولمىسا دىھقان بولامدۇ،بۇ سۆزۈمگە ئىسلىۋالماڭ،بالامغا يا گىپمنى چۈشەندۈرەلمىسەم،چاي ئورنىغا پىۋا ئىچسە ئىمان تۇرامدۇ،ساتىراچمۇ ياخشى ئادەم ئەمەس،نىمىشقا ساقالنى چۈشۈرىدۇ ماز ساتىراچ،كىچىسى چۈش كۆرەلمىدىم،كۈندۈزى ئۇخلىالمىدىم،ئەتىراپىمغا شۇنداق نەزەر سىلىپ قارىدىم،ياخشىلار كۆرۈندى،ئەسكىلەر بۇغدايغا چۈشكەن ئەمەندەك دورا چاچسا ئۆلمىسە،يۇلىۋەتسە بىرسى مىڭ بوپ ئۈنسە ئامال يوق،خوتەن بۇددىستلىرىدىن نوكتى رەشد  بىلەن جوكتى رەشىد نامازغا تۇرغان 72 سەپ مۇسۇلماننىڭ بىشىنى كىسپتىكەنمىش،شۇڭا ئەسكى خۇمپەر ئۇ،ھاراقنى نىمىشقا ئىچىدۇ،قىمار ئوينىسا ئۆزىنىڭ پۇلى لىكىن ئاقۋەتەنى بىلسە ئوينىماس،تانسىنى كىم ئىجاد قىلغان،ھايانى نىمىشقا ساتىمىز،مەن ھايا ئالىمەن ساتىدىغانلار بارمۇ،ماتىماتىكا بەك ياخشى كىتاپ،بىرىدن ئونغىچە بولغان ساننى ماڭا ئۆگەتكەن،توخۇلىرىمنى ئاتىكاچلىقى،چاشقان باققانتىم،ئوغۇرلۇققا سىلىپ تىللا يىغسا باي بولۇپ كەيىپ سۈرەي دەپ،لىكىن چاشقان يىغىپ ماڭا بەرمەي ھەممىنى پاقىغا يەم قىلدى،ۋاپاسى يوق چاشقاندىن مەن ياخشى،يوقالسام بولىدىغۇ بىشڭىزنى ئاغىرىتماي،توك نىمانداق ئەرزان ھە،دەريا سۈيى بۆرەكتىكى تشنى داۋالايمىش،يىنمدا دەريا تۇرسا سۈيى ھەقسىز تۇرسا نمىشقا مىنىرال سۇ ئىچىمەن،مەن شۇنداق گاچىمەنكى پوق قوساقتۇرمەن،ئەقلىم گەجگەمدە،گاچا بولمىسا سىكىرتار بولىمەنمۇ،ئۆزۈمنىڭ دەردى،بىر كەنىتنىڭ دەردى،يەر دىگەننى قەتئى ئەتىۋارلاڭ،ھىچ بولمىسا يەرلىكڭىزگە چۈشلۇق يەرنى ئەپقىلىڭ،ئاسماندا ئۇچسىڭىزمۇ يەرگە كىرىسىز،ئات بىلەن ئىشەكنىڭ بالىسى قىچىر،نەسلى يوق دىمە مىنى جۇمۇ،نەسلىم ساپ ئۇيغۇرمەن،ئانام بىلەن دادام چەتئەل تىلىدىن ئەرەپچىنىلا بىلىدىكەن،ئۇيغۇرچىنىغۇ راۋرۇس ئۆگەنگەنكەن،مازاق قىلماڭ شۇ سەۋىيىسى شۇنچىلككەن،مەنمۇ شۇ،ئەمما قىزىم مەندىن جىق تىل بىلىدۇ،10 ياشقا كىرىدى،بىر ۋاق ئوقۇيالمىسىمۇ بەشۋاق بار،تىرناق ئالغۇچنىڭ ناچارلىقى ھەيييي،تاسما باغلاپ تاسمىدەك بوپ قالدۇق،قىزلارنىڭ ئىشتان بەلبىغىنى بىر ئۆزگەرتسەم دەيمەن،تولا ئىشتىنى تىقىمىغا چۈشۈپ خاپا قىلدى،ئەتلەسنى چەتتە كەيسە نىيۇزەيكۇنى ئىتىزدا كىيىدۇ،ماڭا باشقا گىپىڭىز يوقتۇ،ئاپەتمۇ ئامەتمۇ ئاللانىڭ ئىلكىدە  ،پىشانىگە پۈتۈلگەنى كۆرۈڭ،ئەخلاقلىق قانۇن-ئىنتىزامغا رىئايە قىلىدىغان پۇقرا بولۇڭ،قوناقتىن ياغ چىقىدۇ،بۇغدايدىن چىقمايدۇ،كونا شەھەرمۇ ياخشى ئىدى،نى-نى سەۋىيىسزلەر باشقۇرغۇچى ،ياق سەۋىيلكلەر ئۇ ،تىما يىزىپ تىل ئاڭلاش مەن ئۈچۈن ھوزۇر،ئەتىراپىمغا قاراپ مۇڭلاندىم،ئىشەكمۇ ماڭا قاراپ پىسسڭىدە كۈلۈپ قويۇپ كەتتى،كىيىن ھاڭىراشقا مادارى يەتمەي ئوسۇرۇپ سالدى،چىدىماس ئىشەكلەر،ماز ئىشەكلەر،ئىچكەن سۇدىمۇ شەك بارمىكى دەپ ئىشەنچىسىز گۇيلار ئىچىۋەرمەمسە،بۇغدايغا ئىككى قىتىم دەريا سۈيى قويدۇق،كىشلەر دەريا سۈيىنى تالىشىپ ئۇرۇق-تۇققانلارمۇ كەتمەن كۆتىرىشىپ ياتلاشىتى،سۇنى قەدىرلەڭ،زىمن سۇ بىلەن ياشىرىدۇ،مەنغۇ بىر كالامپاي كىشىدۇرمەن.پۇل خەجلەپ گۇناھنى قىلىپ بىر سائەتلىك ھوزۇرنى دەپ قىيامەتتە قارا يۈز بولغڭىز بارمۇ،قىزلار ھەي ياخشى قىزلار،نىمىشقا ساق قىزلار جالاپ بوپ قالدى،پۇلىسى يىتىپ ئىشىپ تۇرسىمۇ جالاپلىق قىلدى بىر خانىم توۋا،بالىسى ئالى مەكتەپتە ئوقۇسىمۇ ياتاققا چىقىتى دىدى،شۇڭا مەن سىلەرگە دىسەم مەن بۇ دۇنيادىن تازا رازى ئەمەسكەنمەن،تەتۈركەنمەن،خۇراپىمىشمەن،ماڭا ئۇنداق ئادەم زەزھەرلەيدىغان ئاتالمىش مەدەنيەتنىڭ لازىمى يوق،چاچنى تاراپ ئاياقنى مايلاپ گالىستۇك بىلەن كۆزەينەك تاقاپ ماڭسا مەدەنيەتلىك بولامدا ئىشەنمەنمەيمەن،كوچىغا ئۆرە تۇرۇپ سىيدىغان زەيپانىلەرمۇ گالىستۇك تاقايدىغۇ،ئەر تۇرۇپ ئەرگە تىگىدىغان ئەرنىڭ تىگىدە ياتىدىغان پۇلنى دەپ ئەرنى خوتۇن قىلىدىغان بەچچە شوۋىلار ئاۋۇماقتا،زىممنى بۇزماقتا،ئۆلەسەن دەپ قاغىدىم،لىكن قاغىشىم ئۆتمىدى،شۇڭا دەيمەن بەزىدە،تاغقا چىقىپ بىرىنىڭ قويىنى باقسام بوينۇمغا رادىئودىن بىر ئىسىپ خەۋەر ئاڭلىغاچ،ياكى ئوڭ دولامغا چاياننىڭ رەسىمنى سول دولامغا يىلاننىڭ رەسىمنى  چەكتۈرۈپ ئاسما كۆينەك بىلەن كۇسار كىيىپ ئىت يىتىلسەم  ۋاي توۋا دەي  نىمبولدى ماڭا،ئەپۇ قىلڭلار مۇشۇنداق ئاشىدىغان بوپ قالدىم،مەن ماڭاي،مىنى ئۆلتۈرىۋىتڭلار،چۈنكى مەن كىرەكسىز ئەخلەت ،زىمىنغا يۈك،گاچا بولسىڭىز ئنكاس يىزىڭ،دۆت ياكى ئەخمەق بولسىڭىز ئوقۇپ قويۇڭ،،
