بۇنى خەرىتىگە بىرلام قارىۋەتسىڭز ئىنتايىن روشەن بىلەلەيسىز ... ئوتتۇر ئاسىيا تارىختا شانلىق مەدىنىيەت ياراتقان رايۇن تارىختا ئوتتۇر ئاسىيادىن دۇنيانىڭ توت تەرەپىگە نۇرغۇن يېڭىلىقلار تارقالغان ئىجادىيەتكە باي رايۇن... يىپەك يۇلىنىڭ پەيدا بۇلىشى بىلەن ئوتتۇرا ئاسىيا خەلىقىنىڭ تارىختىكى ئىجادىلىق رولى بارا بارا سۇسلىشىپ باردى ئۇزىنىڭ ئاكتىپ يىڭىلىق يارىتىش روھىنىڭ ئورنىغا شەرىقنىڭ تەييار ماللىرىنى غەرىپكە غەرىپنىڭ تەييار ماللىرىنى شەرىققە تۇشۇيدىغان تۇشۇغۇچىلىق بىكىنمىچىلىك ھالەتكە مۇپتىلا بولدى ... بۇنىڭ بىلەن ئوتتۇرا ئاسىيانىڭ دۇنيان تۇتقان ئورنىغا شەرىق بىلەن غەرىپ ئالمىشىپ شەرىق بىلەن غەرىپ كۇچلەندى ... كىيىنكى كۇنلەردە ئوتتۇرا ئاسىيادىكى ھاكىمىيەت ئالمىشىشلار  ھەم بۇ رايۇندىكى ھەرىىل مۇقۇمسىزلىقلار تۇپەيلىدىن شۇنداقلام كارۋان سودىگەرلەرنى مەخسۇس تۇسۇپ بۇلاڭچىلىق قىلىدىغان ئەكۋاللار يىپەك يۇلىنىڭ بىخەتەرلىكىگە تەسىر يەتكۇزدى ...  شۇنىڭ بىلەن ئۇزىنىڭ  ماللىېرىنى چىقىرىدىغان دولەتلەر باشقا بىخەتەر يول ئىزدەشكە مەجبۇر بولۇپ زور تەۋەككۇلچىلىك بىلەن دىڭز يۇلى ئېچىپ چىقتى شۇنىڭدىن باشلاپ ئوتتۇرا ئاسىيا تاشلىنىپ قالدى ...  دىڭىز بۇيىدىكى شەھەرلەر سودا ئالاقە كۇچەيگەنسىرى تەرەقى قىلىپ ماڭدى ھەم ھازىرمۇ دىىڭز بۇيى رايۇنلار تەرەقىياتتا باشقا ئىچكى قۇرۇقلۇق رايۇنلىرىغا سېلىشتۇرغاندا غايەت زور تەرەقىيات نەتىنجىلىرىگە ئېرىشتى ... ئەمدى نوۋەت ئىچكى قۇرۇقلۇق رايۇنلىرىنى ئېچىش كان بايلىقلىرىنى تىز سۇرئەتتە يوتكەپ پايدىلىنىش باسقۇچىغا كىردى ...ھازىرقى دەۋىرگە كەلگەندە  جۇڭگۇنىڭ باش كۇتۇرىشىگە ئەگىشىپ جۇڭگۇنىڭ مىللەتچىلىك ۋە كىڭەيمىچىلىك  كولەڭگىسىندىن دۇنيا ئېھتىيات قىلىشقا باشلىدى جۇڭگۇغا نىسبەتەن دىڭز يۇلى ئانچە بىخەتەر ئەمەس پۇتۇن دىڭز يوللىرىنى غەرىپ دۆلەتلىرى ۋە ئىتىپاقداشلىرى كونتىرول ئاستىغا ئېلىپ جۇڭگۇنى قورشاپ بولدى ...  
بۇنداق بىر شارائىتتا تارىختىكى يىپەك يۇلىنى تىرىلدۇرۇپ بۇ يولدىن پايدىلىنىش جۇڭگۇنىڭ بىردىن بىر تاللىشى .. شۇنداق قىلغاندىلام دىڭىز يۇلىغا تايىنىپ قېلىش ھالىتىنى ئوزگەرتىپ ئۇزىنىڭ ماللىرىنى يىڭى يول ئارقىلىق سىرىتقا چىقىرىدىغان ۋە سىرىتتىكى خام ماددىلارنى بىخەتەر تۇشۇپ ئەكىلەلەيدىغان  ئەۋزەل شارائىتقا ئىگە بۇلالايدۇ ... ئۇنىڭدىن باشقا ئوتتۇرا ئاسىيا  يەنى قازىقىستان  قىرغىسىزتان ئۆزبىكىستان ئاففانىستان دىىگەندەك دولەتلەر مول قېزىلما بايلىققا ئىگە رايۇن بۇ بايلىقلار جۇڭگۇ كۇزىنى قىزارىتماي قالمايدۇ ... جۇڭگۇ ئاففانىستاندىكى زور مىس كىېنىنى قېزىش ھوقوقىنى قۇلىغا ئېلىپ بولدى قازىقىستاندىكى نىفىت تەبىئى گاز تۇربىلىرى ئاللىبۇرن ياسىلىپ ئۇيغۇر دىيارىغا تۇتاشتۇرۇپ بولدى .. ھەم نۇرغۇن خىتتاي كۆچمەنلىرىنى قازىقىستانغا چىقىرىپ ئۇ يەردە بوز يەر ئېچىپ دىىخانچىلىق قىلىۋاتىدۇ ..  ئوزبىكىستان بىلەنمۇ تەبىى گاز كىلىشىمى بار .. شەرىق تەرەپتە تاشقى كۇچلەرنىڭ قورشاۋىغا چۇشۇپ قالغان خىتتاي غەرىپكە سۇرۇلۇپ ئوتتۇرا ئاسىياغا يۇپۇرۇلۇپ كىلىۋاتىدۇ  ... ئورۇستىن پارچىلىنىپ چىقىپمۇ روس تەسىرىدىن قۇتۇلالمايۋاتقان بۇ ئوتتۇرا ئاسىيا دولەتلىرىمۇ خىتتاينىڭ كىلىپ مەبلەخ سېلىشىغا ھەم  ئوز دولىتىنىڭ قېزىلما بايلىقلىرىنى ئەرزان باھادا يوتكەپ كىتىشىگە يېشىل چىراق يىقىپ بېرىۋاتىدۇ  شۇ ئارقىلىق ئۇزىنىڭ روسلاغا تايىنىۋېلىشنى ھالىتىنى ئاجىزلاشتۇرماقچى بۇلىشىۋاتىدۇ ...ھەر كىمگە ئۇزىنىڭ مىللى مەنپەتى ۋە دولەت مەنپەتى ئۇستۇن ئورۇندا تۇرىدۇ  .. ئەپسۇس بىز بولساق خىتتاينىڭ رەزىل زۇلۇم سىياسىتى ئاستىدا يوقۇلۇشقا قاراپ كىتىپ بارىمىز  ...
