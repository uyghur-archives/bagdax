ئۇلار سۈكۈت قىلىدۇ .ئۇلارلا ئەمەس قولىدا كەتمەن كۆتۈرۈپ يۈرگەن دېھقان بىلەن پاراڭلاشسىڭزىمۇ بىر قىسىم زىيالىيلارىدىن ئورۇنلۇق گەپ قلىدۇ ،لىكىن ئاشكارە سۆزلەشتىن ئۆزىنى قاچۇرىدۇ .قانۇندا ئېنىق قىلىپ بەلگىلەنگەن بولسىمۇ ،ئۇلار يەنىلا تەپ تارتىدۇ
«جۇڭخۇا خەلق جۇمھۇريتى ئاپتۇنۇمىيە قانۇنى»دا 1.ئاپتۇنۇم رايۇن ،دۆلەت قانۇنى بۇيىچە مەمۇرى ئورنى نۇرمال ئۆلكىلەردىن بىر دەرىجە يۇقىرى بولغان ،مەركەز بىلەن بىۋاستە مۇناسىۋەتتە بۇلالايدىغان ،شۇ يەرلىك مىللەتنىڭ باشقۇرشىدا بولغان ،مەركەزگە باج تۈلىگەندىن باشقا بارلىق ھۇقۇقلىرى ئۆزىدە بۇلىدىغان ،ئۆز ئالدىغا ئاپتۇنۇمىيە قانۇنى يۈرگۈزەلەيدىغان ،شۇ ئاپتۇنۇم رايۇن تەۋەسىدىكى بارلىق ھۈكىمەت ۋە مەمۇرى ئورگانلارنىڭ بىرىنچى قول باشلىقى ۋە شۇ ئاپتۇنۇم رايۇن دائىرسىدىكى بارلىق مەمۇرى ئورگانلاردا ئىشلەيدىغان كىشلەرنىڭ 60%تىن ئارتۇقى شۇ يەرلىك مىللەتتىن بولغان ئالاھىدە تۇپراق ۋە رايۇن .
2.ئاپتۇنۇم رايۇن ،شۇ زىمىندا ئەزەلدىن ياشاپ كەلگەن ،شۇ زىمىننىڭ ئىگىسى بولغان مىللەتنىڭ نامىدا بۇلىدۇ.مەسلەن ،شىنجاڭ ئۇيغۇر ئاپتۇنۇم رايۇنى ،شىزاڭ زاڭزۇ ئاپتۇنۇم رايۇنى ،نىڭشىيا خۇيزۇ ئاپتۇنۇم رايۇنى ....دىگەندەك .
3.ئاپتۇنۇم رايۇننىڭ يەرلىك مىللىتى ،ئاپتۇنۇم رايۇنى ئىچىدە ئەڭ ھۇقۇقلۇق مىللەتتۇر.
4.ئاپتۇنۇم رايۇن ئىچىدە ،شۇ يەرلىك مىللەتنىڭ دىنى ،تىلى ،يېزىقى ،مائارىپى ،ئۆرىپ-ئادىتى ،مەدەنىيەت -سەنئىتى ،ئىكولوگىيە -مۇھىتى ،نۇپۇسۇ ،بايلقى ،تۇپرىقى قاتارلىقلار ئالاھىدە قوغدىلىدۇ.
5.ئاپتۇنۇم رايۇندا ،شۇ يەرلىك مىللەت كشلىرى ئالدى بىلەن ئاپتۇنۇم رايۇن ئىچىدىكى ئەۋزەل شارائىتلاردىن بەھرىمان بۇلىدۇ .
6ئاپتۇنۇم رايۇن ئىچىدىكى  يەرلىك نۇپۇستىن كۆچمە نۇپۇس مەڭگۈ ئېشىپ كەتمەسلىكى كىرەك .كۆچمە نۇپۇس ناھايتى كۈچلۈك دەرىجىدە كونتىرول قىلنىشى كىرەك .
7.ئاپتۇنۇم رايۇن ئىچىدىكى بارلىق ھۈكمەت ۋە بارلىق مەمۇرى ئورگانلاردا شۇ يەرلىك مىللەتنىڭ تىلى ،يېزىقى بىرىنچى ئورۇندا قوللىنىلىدۇ ،شۇنداقلا شۇ رايۇننىڭ ۋاختى ھۈكمەت ۋە خەلىق ،ارسىدا بىردەك قوللىنىلىشى كىرەك .
8.ئاپتۇنۇم رايۇن ئىچىدىكى يەرلىك مىللەت خەلىقنىڭ مەكتەپلىرىدە ،باشلانغۇچتىن ئالى مەكتەپكىچە بارلىق دەرىسلەر شۇ يەرلىك مىللەتنىڭ تىل يېزىقىدا بۇلىشى كىرەك .
9.ئاپتۇنۇم رايۇندىكى كۆچمە نۇپۇسلار ،چۇقۇم شۇ يەرلىكنىڭ تىلىنى تۇلۇق ئۆگىنىشى ،ئەگەر خالىسا قوللىنىشى كىرەك.
ئاپتۇنۇمىيە قانۇنىدا مانا مۇشۇنداق بەلگىلەنگەن بولسىمۇ ،لىكىن بىر باشلىق ياكى زىيالى نۇرغۇن ئىشنىڭ قانۇندا دىيىلگىنىدەك ئىجرا بولمايۋاتقانلىقىنى ئوتتۇرغا قۇيۇشقا جۈرئەت قىلالمىدى .
ئەمەلىيەتتە پۈتكۈل پارتىيە ئىچىدە جوڭگۇ كوممونىستىك پارتىيسىدەك ئادىل پارتىيە يوق دەپ ئويلايمەن .پەقەت ئۆز ئىچىمىزدىن چىققان ۋە پارتيە ئىچىگە يۇشۇرنىۋالغان بىر قىسىم چىرىك ئەمەلدارلار ھەرخىل يوللار ئارقىلىق قانۇننى بۇرمىلاپ كىتىۋاتىدۇ .
