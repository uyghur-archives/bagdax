                             مەشھۇرلىرىمىز مۇنبەر كۆرەمدۇ؟


ئىلاۋە:يېقىنىقى يىللاردىن بۇيان كەڭ ئومۇملاشقان  ئنتىرنىت تورى،مىكرو بىلوگ ۋە مۇنبەر ....قاتارلىق تارتقۇلار  خىزمەت ۋە تۇرمۇشمىزغا ئاز بولمىغان قۇلايلىقلارنى ئېلىپ كەلدى. ئەمما بىر قىسىم ئاتالمىش مىللەتنىڭ سەركىسى بولغاب زىيالىلار قاتلىمى  مەشھورلۇق كىبىر، ئەمەلدارلىق سۇر-ھەيۋىسىنى كۆرسىتىپ ئۆزنى قاچۇردى . ياكى كۆرمىگەنگە سېلىپ قۇلاق يوپۇرۇپ يۇردى . ئۇلار راستىنلا  كۆرمەمدىغاندۇ ياكى كۆرسىمىۇ  كۆرمەسكە سېلىپ يۇرەمدىغاندۇ دىگەن خىيال بىلەن ئۇزۇمچە تەكشۇرۇش ئېلىپ باردىم ھەم بۇ تەكشۇرۇش نەتىجىسنى تورداشلار بىلەن ئورتاقلىشىنى مۇۋاپىق كۆردۇم  .


   <<بىرناھەق ئىشىنى كۆرۈپ تۇرۇپ توسمىغۇچى ياكى ھەقىقەتنى سۆزلىمىگۇچى ئوخشاشلا شۇ ناھەق ئىشقا قاتناشقۇچى گۈناھكار >>
                                                                                                                                                                        (قۇرئان كەرىم )دىن

2012-يىلى جەمىيەتشۇناس زۇلپىقار بارات ئۆزباشنىڭ <<ئاۋانگارتلىق گۇمراھلىقمۇ  ئە .ئوسمانغا تەنقىد>> <<رادىل ئابلاغا دوستانە تەنقىد >>.. قاتارلىق بىر قىسىم ماقالىلرى  ئەلكۇيى مۇنبىردە ئېلان قىلنىپ كۈچلۈك مۇنازىرە قوزغىدى ھەم مۇنبەر ئەزالىرنىڭ <ئاق تاغلىق ، قارا تاغلىق>قا بۆلىنىپ كىتىشىگە سەۋەب بولدى .  مۇنازىرنىڭ ئاخىرى چىقمدى . دەل شۇ كۇنلەردە  مەن تۇلۇق ئوتتۇردىكى بىر يازارمەن مۇئەللىمنىڭ ھاۋالىسى بىلەن نوپۇزلۇق بىر تەھىرىر بۆلىمىگە بىر پارچە ئورگىنالىنى ئاپرىپ بېرىش ئۈچۈن بېرىپ قالدىم ، ئىشخانىغا كىرگىنىمدە   مۇھەرىر خۇش چىراي قارشى ئالدى، ۋە كومپىوتىر ئالدىن قوزغىلىپ  يان تەرپكە قويۇلغان كىرسىلوغا ئوتۇپ مەن بىلەن قىزغىن پاراڭلاشتى ، شۇ ئەسنادا مەن ئىكرانى سەل يانتۇ بولۇپ قالغان كومپىيوتىر ئىكرانىدىن  ئۆزباش ئەپەندىنىڭ <<ئاۋانگارتلىق گۇمراھلىقمۇ  ئە .ئوسمانغا تەنقىد>>   مۇنازىرسىنىڭ باش بېتىنى كۆرۇپ قالدىم ۋە گەپ ئارلىقىدا  بىلمىگەن كىشى قىياپىتىدە  مۇھەررىنىڭ شۇ مۇنازىرنى كۆرگەن ،كۆرمىگەنلىكنى سوراپ باقتىم ، مۇھەرىرىگە بۇسۆزۇم ياقمىدمۇ ياكى باشقا سەۋەپ بارمۇ گېپىمنى ئاڭلىمىغانغا سېلۋالدى ، مەن يەنە قايتا سورغىمىندا ئۇ خوش ياقماسلىق بىلەن   _ مەنمۇ بۈگۈن باشقىلاردىن ئاڭلىدىم تىخې كۆرگېچە بولمىدىم .دېدى .
ئېنىقىكى ئۇ قىپقىزىل يالغان گەپ قىلدى . مەنمۇ باشقا گەپ قىلمىدىم ۋە شۇ ماقالىنى بىر كۆرۇپ بېقىشنى ، مۇمكىن بولغاندا ئۆزنىڭ كۆز قارشىنى يېزىپ بېقىشنى ئېيتىپ چقىپ كەتتىم . ئەمما خېلىغىچە ئۇنىڭ نىمە ئۈچۈن بۇ مۇنازىردىن ئۆزنى قاچۇرغىنى كاللامدىن ئوتمىدى .


<<بىر ئۇقۇتقۇچىنىڭ ئاھۇ زارى >> دىگەن مۇنازىر بولنىۋاتقان كۇنلەردە مەن قەشقەردە تەتىل مەزگىللىك تەربىيلەش مەكىزى  قۇرۇش ئۈچۈن مەكتەپتىن رۇخسەت سۇراپ قەشقەرگە باردىم . تەربىلەش مەركىزنى تەستىقلىتىش ئۈچۈن شەھەرلىك مائارىپ ئىدارىسگە كىرىشكە تورغا كەلدى  ، مىڭ بىر جاپالاردا بىر تۇنۇشۇمنىڭ شۇ ئىدارىدە ئىشلەيدىغان بىر دوستىنى تېپىپ ئىشخانىسغا باردۇق ، ھەيران قالدىمكى مەن ئىزدەپ بارغان شۇ كىشىمۇ<<بىر ئۇقۇتقۇچىنىڭ ئاھۇ زارى >>  نامىلىق مۇنازىرىنى كۆرۇپ ئولتۇرۇپتۇ .
بۇ مەسۇم تېخى بىزنىڭ پىرافىسور ئونۋانى بار بىر مۇئەللىم  <<ئۇيغۇرلار ۋە سېسىق تۇققۇزىنچى>><<ئۇيغۇرلار ۋە سېسىق تۇققۇزىنچىغا رەدىيە >>  نامىلىق ماقالىنى ھاياجان بىلەن ئۇقۇپ بەردى . ھەيران قالدىمكى بىزنىڭ مەشھورلىمىزمۇ مۇنبەردىكى مۇنازىرلەرنى كۆرۇپ تۇردىكەن . ئاڭلاپ تۇردىكەن ، ئەجەبلىنەرلىكى شۇكى مۇنازىرگە پىكىر قاتناشتۇرمايدىكەن .


خۇلاسە : ھازىر دانا پارتىمىزنىڭ توغرا رەھبەرلىكىدە ، قىزىل بايراقنىڭ ئىللق ھارارىتى ئاستىدا ئىشچى خىزمەتچىللىرمىزنىڭ  (دۆلەت شىتاتىدىكى)  خىزمەت ئورنى كۆپ ياخشىلاندى ، ھەربىر ئادەمنىڭ بىردىن تورغا ئۇلاقىلىق كومپىيوتىرى بار بولدى ، ئۇلار ئاسان بىكار ۋاقتىنى  تورغا چىقىش  ئويۇن ئويناش بىلەن زامانىۋى ئارام ئېلىشنى بىر لەشتۇردى، قىقىسى <<شىنجاڭ يازغۇچىلار مۇنبىر>>دە ھەرقايسى گېزىت - ژوناللارنىڭ مەسئۇل مۇھەررىرلىرى، يېراق يېزا قىشلاقلاردىكى ئاددى ئۇقۇتقۇچىلار ، بىلىمىك دوكتۇر ئۇتۇپ كەتتى ئالىملار بار ، ئەجەبا  شىنىجاڭ يازغۇچىلارمۇنبىرنى تېپىپ كۆرەلىگەن ، يازما يوللىيالىغان ئەشۇ ئاتالمىش زىيالىلىرمىز باغداش مۇنبىرىگە ئوخشاش مۇنازىرە مۇنبىرنى تاپالماي قالارمۇ ؟؟  كۆرمەي قالارمۇ ....؟؟  
سائەت-سائەتلەپ كومپىيوتىر ئالدىدا ئولتۇرغان ھەرساھەدىكى ئالى مەلۇماتلىق زىيالىرىمىز مۇنازىرە تور بىكەتلىرنى كۆرمەسلىكى مومكىن ئەمەس ، ئۇلار كۆردۇ ، ئەمما ئۇلارنىڭ قەلبى ھەقىقەتنى سۆزلەشتىن كۆپ يېراق . ئۇلارنىڭ كۆرۇپ تۇرۇپمۇ ئىككى ئېغىز ھەق گەپ قىلالماسلىقىنى پاجىئە دىمەي ئامالىمىز يوق .

ئىدارىدە ئولتۇرۇپ تورغا چىقىدىغان مەشھۇرلار، زىيالىيلار، ئەمەلدارلارنىڭ تورغا چىقىش ھەققىدە مەندىمۇ ئاز تولا چۈشەنچە بار:
بىر قىسىمى قەرت، قىمار دېگەندەك ئويۇنلارنى ئوينايدۇ، ئۇيغۇرچە خەت بېسىشنى ئۇقمايدۇ
بىر قىسىمى چوكان (ياكى قىز) ئوۋلايدۇ، تۆھمەت دېمەڭلار، بۇ راست...
بىر قىسىمى ئۇيغۇرچە تورلارغا ئەسلا كىرمەيدۇ....باشقا تىلدىكى تور بەتلەرنى ئېچىشنى شەرەپ بىلىدۇ، ياكى ئۆزىنى كۆرسىتىشنى مەقسەت قىلغان؛
بىر قىسىمى ئۇيغۇرچە تور بەتلەرگە چىقسىمۇ، ئاساسەن كۆڭۈل ئېچىش خاراكتېرىدىكى تور بەتلەرنى كۆرىدۇ، ناخشا، كىنو چۈشىرىدۇ...
بىر قىسىمى ئۇيغۇرچە مۇنبەرلەرنى كۆرىدۇ، لېكىن ھېچقانداق ئىپادە بىلدۈرمەيدۇ، كىچىك بالىلار ، ئىش ئۇقمايدىغان شۇمتەكلەر دەپ ياراتمايدۇ...
بىر قىسىملار كۆرۈپ تۇرسىمۇ، لېكىن سىرتقا چىققاندا مەن ئەزەلدىن ئۇيغۇرچە توربەت ئاچمايمەن، نېمە بار ئۇيغۇرچە تور بەتلەردە دەپ خۇپسەنلىك قىلىدۇ...
بىر قىسىم مەشھۇرلار ئۇيغۇرچە تور بەتلەرگە دەسلەپتە چىقىپ باققان، لېكىن پىكىر ئەركىنلىكى بولغانلىقى ئۈچۈن، ئۆزىنى ھېچكىمنىڭ ئاياپ قويمايدىغانلىقىنى، «ھۆرمەتسىزلىك» قىلىدىغانلىقىنى كۆرۈپ يېتىپ چىقماس بولغان.
ئاز بىر قىسىملىرى كەڭ ئۇيغۇر تورچىلىرى بىلەن ھەمنەپەس بولۇپ، ئاكتىپ ئىنكاس قايتۇرۇپ، ئۆز قاراشلىرىنى ئەركىن ئىپادىلەيدۇ....
