ئاتا ئانىلا ئۆزلىرىنى ئالدىراش دەپ بالىلىرىنى كومىنىسنىڭ يەسلىلىرىگە ئاپىرىپ بېرىشتى  نۇمۇس ھايا نى بىلمەيدىغان بېچارە گۆدەكلىرىمىز  شۇ مۇھىتتا مالىملىرىنىڭ كۆرستىشى بىلەن بىرىنىڭ قولىنى بىر برىنىڭ بېلىنى تۇتۇپ ئويناشتى، بەزىدە ھەم بىربرىنى بېسپ ئويناشتى ،چۈنكى ئۇلا ھايا نى ئۇخمايدۇ.. ئېچىنىشلىقى ئۇلارغا بۇنىڭ ئەخلاقسىزلىق ھاياسىزلىق ئىكەنلىكىنى ئۇقتۇرۇشى چۈشەندۈرشى كىېرەك بولغان ئوقۇتقۇچىلىرىمۇ ھايانى ئۇخمايدۇ، ئۇلا ئۈچۈن پۇل مۇھىم گۆدەھلەرنىڭ ئەخلاقى ئەمەس، بۇ گۆدەھلەرنىڭ يىغلىماي  كۈننى كەچ قىلىش بۇ ئوقۇتقۇچىلانىڭ خوشللىقى،،،،،،،      ئەنە شۇنداق قىلىپ پەرزەنىلرىمىز يەسلىدكى چېغىدىلا قىز ئوغۇللارنىڭ قول ۋە بەل تۇتۇشنى ئادەتتىكى ئىش دەپ بىلىشتى چۈنكى ئۇلر كۆندى...باشلانغۇچقا، تولۇقسىز غا چىقتى ئۇلا يەنىلا كۆنگىنى بويىچە ئىش قىلدى ئىلگىرى قول  تۇتۇشنى ئادەتتىكى ئىش دەپ بىلگەن بولسا  ئەمدى قۇچاقلاپ سۆيىشنى ئۈگەندى.بۇنىڭغا ياندىشىپ ئۇلار ئۆيدىن ئاتا ئانىسىدىن ئايرىلپ تۈزۈم ۋە تەربىيسى چىڭ بولمىغان ئەركىن ياتاقلاغا ئورۇنلاشتى بۇ ھال ئۇلارغا تېخىمۇ پۇرسەت يارتىپ بەردى......  بالا توي قىلىش يىشىغا يەتكەندە ئاتا ئانا ۋە ئۇرۇق تۇققانلرى كومىنىس ئىدىيىسى بويىچە بالا كىچىك جېنىنى باقالمايدۇ بىرە ئشنىڭ بىشىنى تۇتۇپ يا ئوقۇپ خزمەتكە چىقىپ ئاندىن تويىنى قىلسۇن دېدې ئەممە ئۇلا رىزىقىنى كومىنىسلا ئەمەس ئىنسان ئەمەس اللە نڭ بېردىغىننى ئۇنتىشتى  چۈنكى كومىنىسلا شۇنداق سۆزلەرنى بىزنىڭ كاللىمزغا سىڭدۇرۋەتتىكى بىزلەر ئۇ گەپلەرنى راس شۇنداق دىيشىپ ئەمەل قىىپ كېلۋاتىمىز يۇ، ئەممەئۆز  ئېمانىمىز ئەمەلسىز قېلۋاتىدۇ...مانا شۇ سەۋەپ  تولۇق ئوتتۇرنى ئىمان ئەخلاق بىلەن پۈتتۈرگەن نې ياشلارمۇ مۇشۇ قىز ئوغۇل ئارىسىدىكى ھېس تۇيغۇدا ئېزىپ كېىتىۋاتىدۇ.... ئالى مەكتەپكە كەلگىنىگە ئىككى يىل بوپتۇ كۆرگەنلرىمىگە بەزىدە ئىشەنمەي قالىمەن ھەر قىتىم شۇنداق مەنزىرىلىەرنى كۆرگىنىمدە شۇ چەت يېزىلاردىكى بالىلىرىنى ئالى مەكتەپكە ئەۋەتىپ ئۆيلىرىدە خاتىرجەم ئولتۇرۋاتقان ئاتا ئانىلاغا ئىچىم ئاغرىيدۇ،،،اللە خالىسا بۇ كۆرگەن ھەم كۆرۋاتقانلىرىمنى تورداشلارغا ئىيتىمەن....خەنزۇلا سىنىپتىن بىرىنىڭ بىلىنى تۇتۇپ دەرىستىن چۈشىدغانلىقىنى ئاتا ئانىلانىڭ بالىلىرىنى تا يەسلىدىن باشلاپ مۇشۇندا ئاتا بۇۋاڭ مايمۇن دەيدىغان مائارىپقا تاشلاپلا قويماسىلىقى كېرەكلىكىنى جاكارلايمەن، ئىن شا ئاللا!!!

شۇنداق ئاتا ئانىلار بالىلىرىنى 24سائەت مەكتەپكە ئۈتكۈزۈپ قويغىنى يوق ، لېكىن شۇنى بىلڭككى ‹‹مۇدىر ئەپەندى ›› ئوقۇغۇچى مەكتەپتە ھەرجەھەتتىن بىلىم ئالىدۇ ، مەكتەپتە بىلىم ئۈگىنىپلا قالماي ئەدەپ ئەخلاق جەھەتتىنمۇ ياخشى تەربىلىنىشى كىرەك ، مەكتەپ دىگەن ئوقۇغۇچىلارنىڭ يېرىم ئاتا ئانىسى دېسەك ھەرگىزمۇ ئارتۇق گەپ ئەمەس سىز ناۋادا ئوقۇتقۇچى بولسىڭزى ئالدى بىلەن بالىلارنى ئەخلاق جەھەتتىن تەربىلەش جەھەتتىكى ئىدىيىڭىزنى توغۇرلاڭ ، بۇ سىزنىڭ ۋە بارلىق ئوقۇتقۇچىلارنىڭ باش تارتىشقا بولمايدىغان شەرەپلىك مەجبورىيتى ، بۇ مەجبورىيەتتنىڭ ھۈددىسىدن چىقالمايايدىغان ئوقۇتقۇچىنى ھەرگىزمۇ ياخشى ئوقۇتقۇچى دىگىلى بولمايدۇ ،
