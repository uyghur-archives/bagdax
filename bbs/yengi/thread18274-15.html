    تۈنۆگۈن مەكتەپ رەھبەرلىرى ئوقۇتقۇچىلار ئىشتىن چۈشكەندىن كىيىن قىسقا يىغىن ئېچىپ ، 10 نەچچە تۈرلۈك بىر ئۇقتۇرۇشنامىنى ئوقۇتقۇچىلارغا يەتكۈزدى . ( مەن ئىشخانىدا مەلۇم ئىشلار بىلەن بولۇپ قېلىپ كىيىن قېلىپ تولۇق ئاڭلىيالمىدىم )
     ئاڭلىغان تەرەپلىرىم :__ مائارىپ ئىدارىسى يەنە يىغىن ئېچىپ شەنبە - يەكشەنبىلىك دەم ئېلىشنىڭ ۋاقتىنچە ئەمەلدىن قالغانلىقى ، ئوقۇتقۇچىلارنىڭ توغرا چۈشىنىشىنى ، دەم ئېلىش كۈنىدە كىلىپ سىياسىي ئۆگىنىش قىلىش ( مارالبىشىدىكى 23 - ئۈمىد ( ئاپرىلنىڭ ئۇيغۇرچە ئاتىلىشى ) ۋەقەسى مۇناسىۋىتى بىلەن سىياسىي ئۆگىنىش قىلغۇدەكمىز ) ۋە كەسپى رەسمىيەتلىرىمىزنى تولۇقلاش ئىشلىرىنى قىلىغۇدەكمىز .
    ئاجايىپ ئىش ھە ! بىزدە ، « ئوتنى كىم تۇتسا شۇنىڭ قولى كۆيىدۇ » دىگەن گەپ بار .
   مىنىڭچە بۇ مەسىلىنى ھەل قىلىشنىڭ تۈپ يولى ئەمەس . مۇشۇ قارارنى چىقارغان ئەپەندىنىڭ ئالدىغا بېرىپ :
   __ سىز « يىمىگەن مانتىنىڭ پۇلىنى بېرىڭ ! » دىسە ئۇنى غىڭ قىلمايلا تۆلىۋىتەمسىز ؟ دىسە ، ئۇ :
  __ ئەلۋەتتە تۆلەۋىرىمەن ! __ دىمەس ھەقاچان ؟!
    بىز ئوقۇتقۇچىلار قايسىبىر قىلىپ قويغان گۇناھىمىزغا 5 كۈن جاپالىق ئىشلەپ ، مىڭىمىز چارچاپ ، ئەمدى بىر ئارام كۈنى كەلگەندە دۆلەت قانۇندا بەلگىلىگەن دەم ئېلىش كۈنىمىز مۇشۇنداقلا بىكار قىلىنسا ، بۇنىڭغا ئىگە بولىدىغان ئورۇن ياكى مەسئۇل خادىم يوقمۇ ؟!
   ئۇنىڭسىزمۇ بىزنىڭ خىزمەت بىسىمىمىز ئازمۇ ؟
   مىنىڭ بۇنداق دىيىشىم ھەرگىزمۇ سىياسىي ئۆگىنىش توغرا ئەمەس ! كەسپى رەسمىيەتنى ئىشلىمەيمىز ! _ دىگەنلىكتىن دېرەك بەرمەيدۇ . ھەممە ئىش ئۆز يولىدا ، ئۆز لايىقىدا بولغىنى ياخشى .چۈنكى بىزمۇ ئوخشاشلا جوڭخۇا خەلق جۇمھۇرىيىتىنىڭ قانۇنلۇق پۇقراسى ئەمەسمۇ ؟! بىزنىڭ قانۇنلۇق دەم ئېلىش كۈنىدىن بەھىر ئېلىش ھوقوقىمىز يوقمۇ ؟ تەتىل تېخى تەتىلدەك بولمىسا ، دەم ئېلىش كۈنلىرىمىز تېخى ! سەككىز سائەت تولۇق ئىشلەپ يەنە كەسپى باقاۋۇلدەك ( ئامانلىق ساقلىغۇچىدەك ) مەكتەپنى ساقلىساق ، بۇ ئۇزۇن كېچىلەردە كۆزلەر ئۇيقۇغا كەتسە ، نەق شۇ چاغدا يۇقۇرىدىن تەكشۈرۈش كەلسە ئۇنىڭغا تېخى جەرىمانە تۆلىسەك ؟!
    بەزى يېزا - بازاردىكى باشلانغۇچ مەكتەپلەردە ئامانلىق ئۆيىدىن پەقەت بىرسىلا بار بولۇپ ، ئاستى - ئۈستى كارىۋاتتا مەكتەپتىكى ئەر ئوقۇتقۇچىلارمۇ ، ئايال ئوقۇتقۇچىلارمۇ بىللە تۇرىدىكەن . ( بەزى مەكتەپلەردە 24 سائەت تولۇق تۇرىدۇ . بۇ ئىشلارنىڭ ئىسپاتى ئەلۋەتتە بار ) .
   بۇ ئوقۇتقۇچىلارنىڭ ئۆيى يوقمۇ ؟ ئايالى ، ئەرلىرى يوقمۇ ؟ ئۇششاق بالىلىرى يوقمۇ ؟ كېسەلچان ئاتا - ئانىلىرى يوقمىدۇ ؟ !
   مۇناسىۋەتلىك ئورۇنلارنىڭ بىز ئوقۇتقۇچىلارنىڭ ھەقىقىي بىر ئوقۇتقۇچىلىق سالاھىيىتى بىلەن دۆلەتنىڭ باشقا قاتلاملىرىدىكى ھەر ساھە پۇقرالاردەك قانۇنلۇق ھوقۇق - مەنپەئەتىمىزنى كاپالەتكە ئىگە قىلىشىنى چىن كۆڭلىمىزدىن بەكمۇ _ بەكمۇ ئۈمىد قىلىمىز !

شىنجاڭ ئۇيغۇر ئاپتونوم رايونلۇق مائارىپ نازارىتى رەھبەرلىرى ئۇيغۇرچە مىكروبىلوگدىن ئۈنۈملۈك پايدىلانماقتا



     ئۆز مۇخبىرىمىز نىجات نىياز خەۋىرى، فوتوسى: 17-ماي چۈشتىن كېيىن، شىنجاڭ ئۇيغۇر ئاپتونوم رايونلۇق مائارىپ نازارىتى رەھبەرلىرىنىڭ ئۇيغۇرچە «بارمۇ» مىكروبىلوگىنى ئىشلىتىش مۇھاكىمە سۆھبەت يىغىنى مائارىپ نازارىتى يىغىن زالىدا ئېچىلدى، يىغىنغا ئاپتونوم رايونلۇق مائارىپ نازارىتىدىكى ھەر قايسى باشقارمىلىرىنىڭ باشلىقلىرى قاتناشتى.
    يىغىندا تەڭرىتاغ ئۇيغۇرچە تورىنىڭ تېخنىكى ئەكرەمجان ئەركىن يېڭىدىن روياپقا چىققان يېڭى تېزبىلوگنىڭ بارلىق ئىقتىدارلىرىنى تونۇشتۇردى، ئارقىدىن ھەر قايسى باشقارما باشلىقلىرى ئۆز نامىدا مىكروبىلوگقا ئەزا بولۇپ كىرىپ، تورداشلار بىلەن ئۆزئارا ھەمسۆھبەتتە بولدى.


    ئاپتونوم رايونلۇق مائارىپ نازارىتىنىڭ مۇئاۋىن نازىرى تۇڭ جىمىن مۇنداق دېدى: يېقىنقى بىر نەچچە يىلدىن بۇيان مىكروبىلوگ تورداشلار ئارىسىدا كەڭ ئومۇملاشتى، بىز ئالدىنقى يىلى سىنا مىكروبىلوگىدا نازارەتنىڭ شۇنداقلا نازارەتتىكى رەھبەرلەرنىڭ خەنزۇچە مىكروبىلوگىنى ئاچقاندىن كېيىن، نازارەتتىكى ئاز سانلىق مىللەت رەھبەرلىرىنىڭ ئۇيغۇرچە مىكروبىلوگ ئېچىشىنى تەۋسىيە قىلدۇق، بولۇپمۇ نازىر تۇرسۇن ئىبىراھىم ئۆزى باش بولۇپ ئۇيغۇرچە مىكروبىلوگ ئېچىپ، مائارىپ نازارىتىدىكى باشقا رەھبىرىي كادىرلارغا ئۈلگە بولدى. مىكروبىلوگنىڭ كەڭ كۆلەمدە ئومۇملىشىشى نەتىجىسىدە بىز ئۆز خىزمىتىمىزدىكى يېتەرسىزلىكلەرنى تېخىمۇ تولۇقلاپ، مائارىپ سىستېمىسىدا ساقلىنىۋاتقان مەسىلىلەرنى ۋاقتىدا ھەل قىلىپ، مائارىپقا كۆڭۈل بۆلۈۋاتقان، مائارىپنى قوللاۋاتقان بارلىق تورداشلارغا ئۆزىمىزنىڭ ئەمەلىي ھەرىكىتىمىز ئارقىلىق جاۋاپ قايتۇرىمىز.



    ئىگىلىنىشىچە، مائارىپ نازارىتى بۇ قېتىم مائارىپ نازارىتى رەھبەرلىرىنىڭ ئۆز نامىدا مىكروبىلوگ ئېچىشى ۋە كۈندە يوللىغان مىكروبىلوگ سانى، ئىنكاس سانى قاتارلىقلارنى ئۈنۈم مائاش تەرتىپىگە كىرگۈزگەن بولۇپ، مائارىپ نازارىتىنىڭ نازىرى تۇرسۇن ئىبىراھىم شۇنداقلا مائارىپ نازارىتىدىكى ھەر قايسى باشقارمىلارنىڭ باشلىقلىرى مائارىپ سىستېمىسىدىكىلەرگە ئۈلگە بولۇپ ئۇيغۇرچە مىكروبىلوگقا تىزىملاتقان ۋە تورداشلار بىلەن ئۆزئارا پىكىر ئالماشتۇرۇشنى باشلىغان.

RE: دۆلىتىمىزنىڭ « ئوقۇتقۇچىلار قانۇنى » ئەمەلدىن قالدىمۇ ؟ !

[b]تولىمۇ ئەپسۇس ئوقۇتقۇچىلارنىڭ ئاھۇ-زارى پەقەت قول يازمىلاردا، كۇس-كۇس پاراڭلار ئىچىدە، ئىشىكى چىڭ يېپىلغان ئوقۇتۇش گورۇپپا ئىشخانىسىدىكى ئوتتۇرا قاتلام كادىرلىرى يوق بولغان مۇنازىرە پاراڭلىرىدىلا  ۋە يەنە قانداقتۇر قۇلاقنى قۇلاققا يېقىپ تۇرۇپ ئەنسىزلىك ئىچىدىكى ۋايساشلاردىلا جاراڭلايدۇ خالاس.
  سىز ئىش ئۈستىلىگە قەۋەت-قەۋەت كىتاپلارنى تىزىپ قويۇپ زېھنى بىلەن ئوقۇتۇش تەتقىقاتى، ئوقۇغۇچىلارغا بىلىم بېرىش يوللىرىدا ئىزدىنىۋاتقان ئوقۇتقۇچىلىرىمىزنىڭ ئاقىرىپ كەتكەن چاچلىرىغا قاراپ ھېسداشلىق قىلىشتىن كۆرە ئۇلارنىڭ روھى دۇنياسىدىكى بۇرۇن سۆيۈنۈپ ئىشلىگەن ئوقۇتقۇچىلىق خىزمىتىنىڭ ئەمدى زارلىنىپ تۇرۇپ ئىشلەيدىغان خىزمەتكە ئايلانغانلىقىدەك ئىپادىسىنى تۈرۈلگەن قوشومىسىدىن، نۇرسىزلانغان كۆزلىرىدىن ، ئۈمۈدسىزلەنگەن چىرايىدىن كۆرۈپ چۈشىنىپ يەتسىڭىزلا كۇپايە. چۈنكى بۇ چۈشىنىش بەلكىم سىزگە بۇرۇنقىدەك مۇنبەردىن پەخىرلىنىپ ، مۇنبەرنى تەرەققىي قىلدۇرۇپ، مۇنبەر ئۈستىدە ھاياتىنى تەقدىم ئېتىدىغان ئوقۇتقۇچىلارنى ناھايىتىمۇ ئازلاپ كېتىۋاتقانلىقىنىڭ سەۋەبىنى بىلىشكە قىزىقتۇرىدۇ.

  سىز كەچلەردە، كوچىلاردا مەس بولۇپ دەلدەڭشىپ كېتىپ بارغان ياش ياكى ئوتتۇرا ئوقۇتقۇچىلارنىڭ ھاياسىز ، ئۆز سالاھىيىتىگە ماس كەلمەيدىغان قىلىقلىرىنى كۆرسىڭىز، ئۇلارنىڭ ئالدىغا دادىل قەدەم بىلەن بېرىپ ئىككى كەلىمە ئاگاھلاندۇرۇش سۆزى قىلىپ قويۇشنى ئۇنۇتماڭ، پەقەت بولمىغاندا«مۇئەللىم سىزنى مەن ئۆيىڭىزگە ئاپىرىپ قويايمۇ، نەگە بارىسىز» دېگەندەك سۆزلەر، ئۇلارنى قانچە كۆپ مۇئەەللىم دەپ ئاتاپ بەرگەنچە ئۇلار ئۆزلىرىنىڭ مۇئەللىملىكىنى ئۇنتۇپ قالىدىغان يامان ئىللىتىدىن قالىدۇ. خىزمەتتىكى چارچاشنى ھاراق ئىچىش بىلەن ئەمەس ئۆتكۈر تەپەككۈرى ئارقىلىق ئىنكاس ۋە نەتىجە بىلەن چىقىرىدۇ.

سىز يەنە ئوقۇتقۇچىلارنىڭ مەكتەپ مودىرىنى كۆرسە ئىتائەتمەنلىك بىلەن شەرتسىز قول قويۇشتۇرپ تۇرىشىنى كۆرۈپ ئوقۇتقۇچىلارنى «ئەجەپ ياۋاشكەن» دەپ قالماڭ ، مېنىڭ بىلىشىمچە تارىختىن ھازىرغىچە ئوقۇتقۇچىلار باش كۆتۈرۈپ ئىجتىمائىي داۋالغۇش پەيدا قىلغان بىرەر چوڭ تارىخ يازمىلىرى ياكى تىل ئۇزارتمىلىرى يوق ، چۈنكى ئوقۇتقۇچىدا بىرلا غەرەز يەنى پەرزەنتلەرنى ياخشى تەربىيلەش ، ئوقۇغۇچىلارنى ئەلگە ياراملىق قىلىپ تەربىيلەش تەك بىرلا، بىرلا نىشان بار بولغاچقا ، زاماننى چۈشىنىپ ئۇنىڭ يەشمىسىنى ئوقۇغۇچىلارغا يەتكۈزگىنى بىلەن زامانداشلارنى چۈشىنىپ ئۇلارغا  بىرەر ئېغىز پىكىر بېرىشكە قارىتا رىياللىقتىكى نەزەر دائىرىسى تارلىق قىلىدۇ. يەنە بىر ئەمەلىي سەۋەپ مۇدىرنىڭ «خىزمەتتىن بىكار قىلىش، مىقدارلاشتۇرۇپ باھالاشتىكى نۇمۇر تارتىش، ئەڭ يامىنى ھامان بىركۈنى ئۇنداق مۇنداق سەۋەپلەرنى ئارتىپلا تۇيدۇرماي ، يوللۇق ئۆچ ئېلىشلىرى....ئوقۇتقۇچىلارنىڭ ئېڭىدىكى مائاشقا خۇددى خۇداغا تەلمۈرگەندەك تەلمۈرۈپ مائاشسىز قالسا ھازىرلا ئۆلىدىغاندەك ، بىرەر ئېغىز ھەقىقىي گەپ قىلىپ قويسا ھازىرلا ئۇلارنىڭ قوڭىغا تىپىپ ھەيدەپ چىقىرىدىغاندەك خىيالىرى.»  قاتارلىق بىر مۇنچە سەۋەپلەر ئۇلارنى شۇنداق ئىتائەتمەن قىلىپ قويغان.  ئەگەر سىز يۇقارقىدەك ئەھۋاللارنى ئاڭلىسىڭىز ياكى چۈشەنسىزىڭىز ئوقۇتقۇچىلارغا بىر نەچچە كەلىمە سۆز قىلىپ ئۇلارنىڭ زاماننى چۈشىنىپ زامانداشلارنى ۋە زامانداشلارنىڭ باشقا ئىش-ھەركەتلىرىنى چۈشەنمەي قېلىشتىن ساقلىنىشقا،ئۆزنىڭ يەنە نۇرغۇن ئىشلارنى ھوجۇتقا چىقىرالايدىغانلىقىغا قارىتا دەۋەت، مەدەت بېرىپ قويۇشنى ئونۇتماڭ.
ئوقۇتقۇچىلارنىڭ تەربىيىسىدىن جاھاندىكى ھەممە ساھە بارلىققا كەلگەن ۋە تەرەققىي قىلغان ئەمما ئوقۇتقۇچىلارنىڭ قولىدىن پەقەت سىنىپ، مەكتەپ بارلىققا كەلگەن بۇنى قانداق چۈشىنەرمىز ئەمدى
