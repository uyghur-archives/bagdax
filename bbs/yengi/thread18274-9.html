       ئەسسالامۇ ئەلەيكۇم ! ئەزىز ، مۆھتەرەم قېرىنداشلىرىم ! مەن ئۇشبۇ تىمىنى يوللىغان قېرىندىشىڭلار « ئورداكەنت » بولىمەن . مىنى ۋە مەن يوللىغان تىمىدىكى مەزمۇنلارنىڭ مەزمۇنىنى يەنىمۇ ئىچكىرىلەپ چۈشىنىشنى خالىساڭلا ئازراق ۋاقتىڭلارنى سەرپ ئېتىپ نەزىرىڭلارنى تۆۋەندىكى يازمىلىرىمغا ئاغدۇرۇشىڭلارنى سەمىمىلىك بىلەن ئۈمىد قىلىمەن .
   نەچچە كۈندىن بېرى مەكتەپتىكى يېقىن بىر بۇرادىرىمنىڭ تويى ۋە مەكتەپنىڭ « باقاۋۇللۇق » ۋەزىپىسىنى ئورۇنداش سەۋەبى بىلەن ھەر قايسىڭلار بىلەن پىكىر ئالماشتۇرۇشقا ئامالسىز قالدىم . شۇ ۋەجىدىن بەزى قېرىنداشلىرىمنىڭ ئىنكاسلىرىغا قارىتا ۋاقتىدا چۈشەنچە بېرەلمىدىم . بۇ تەرەپلەردىكى نوقسانلىرىمنى ئەپۇ ئېتىكىڭلار بىلەن يېپىۋىتىشىڭلارنىمۇ سەمىمىلىك بىلەن ئۈمىد قىلىمەن .
     خوش ، مەقسەتكە كەلسەك  ، تىما يوللانغاندىن كىيىن كۆپلىگەن قېرىنداشلىرىمنىڭ قىممەتلىك پىكىرلىرى بىلەن يۈزلەشتىم . بۇ ئىنكاسلار ئىچىدە بەزى قېرىنداشلىرىم خۇددى مەكتىپىمدىكى شۇ غەلىتە ئىشلارنى كۆزى بىلەن كۆرگەندەك بايان قىلىشى مىنى ، شۇ قېرىندىشىم ئىشلەۋاتقان مەكتەپتىمۇ شۇنداق ئىشلارنىڭ مەۋجۇتلىقىدىن ۋە ياكى ئەتراپىدىكى مەكتەپ ياكى قېرىنداشلىرى ئىشلەۋاتقان مەكتەپتە مەۋجۇت بولۇشى مومكىن دىگەن قاراشقا كەلتۈردى .
    بىزدە « ئىسنىڭ ئاچچىقىنى مورا بىلىدۇ » دىگەن ھېكمەت بار . ئۇشبۇ گەپكە ئاساسلانغاندا مىنىڭ مەكتىپىمنىڭ ئەمەلىي ئەھۋالىنى مەن ۋە مىنىڭ خىزمەتداشلىرىن ناھايىتى ياخشى بىلىدۇ .
    مەلۇم بىر قەۋم ، مىللەت ، رايۇن ، ۋە ياكى دۆلەت كىشىلىرىنىڭ ھەممىسىنىڭ ياخشى ياكى ھەممىسىنىڭ يامان ( ناچار دىمەكچى ) بولۇشى مومكىنمۇ ؟ ئەلۋەتتە مومكىن ئەمەس . ئېلىمىزدىكى ياكى ئۇيغۇر ئاپتونوم رايۇنىدىكى ۋە ياكى جەنۇبىي شىنجاڭ مەكتەپلىرىدىكى ھەممە مەكتەپ مۇدىرىنىڭ قانۇن بويىچە ئىش قىلىدىغان ، مائارىپقا ئۆزىنى بېغىشلايدىغان مەرىپەتپەرۋەر بولۇشى ياكى ، ھەممىسىنىڭ نەپسانىيەتچى، ئومومنىڭ ھەققىگە خىيانەت قىلىدىغان ، قانۇننى بولۇپمۇ « ئوقۇتقۇچىلار قانۇنى » نى ئاياغ ئاستى قىلىدىغان ئەبلەخ بولۇشى مومكىنمۇ ؟! بۇمۇ ئەلۋەتتە مومكىن ئەمەس .
    خوش ، ئۇنداقتا مەن يوللىغان تىمىدىكى ئۇ مەسىلە بىزنىڭ مەكتەپتە يۈز بېرىۋاتقان تالاي خاتالىقلارنىڭ بىرى- خالاس .
   « ناخشا » دىگەن نامدىكى ( 7- بەت 63- قەۋەت ) بىر قېرىندىشىم مىنى ھەم ماڭا ئوخشىغان ئاتالمىش « قارشى چىققۇچىلار » نى ئۆز ئىنكاسىدا يازغان :__« مۇنداق مەزمۇندىكى ماقالىلار ھۇرۇن ، خىزمەتتە ئەستايىدىل ئەمەس ، ئىزدىنىش روھى كەم ، بىر قىسىم قاخشاش كىسىلى ئېغىر كىشىلەرنىڭ ئۆزىنى خىزمەتتىن قاچۇرۇش ، ئۆزىنى قوغداش ئۇچۇن ئىلىپ بىرىۋاتقان ئاخمىقانە قاراشلىرى ياكى تەسەللىسى خالاس .» __ دىگەن ئەخمىقانە نادانلارچە كۆز قارىشى ، بۇ ئىنكاسقا قارىتا ئېنىق بىر چۈشەنچە بېرىشىمنى تەقەززا قىلدى .
   قېرىندىشىم « ناخشا » ! مەن مىلادى 2004 - يىل شىنجاڭ سەنئەت ئىنىستىتۇتىنىڭ مۇزىكا ئورۇنداش كەسپىنى پۈتتۈرۈپ 2006 - يىلىدىن باشلاپ ئۇشبۇ مەكتەپنىڭ مۇزىكا ئوقۇتقۇچىلىقىنى قىلىپ كېلىۋاتقان بىر ئەر مۇئەللىم . مەن ئاتا - ئانام  ھەتتا بوۋام - مۇمام شۇغۇللانغان بۇ شەرەپلىك خىزمەتنى خېلى ئەتراپلىق چۈشىنىمەن ھەم ئەڭ ياخشى يېرى مەن بۇ كەسىپنى بەكمۇ قىزغىن سۆيىمەن . خىزمەتكە قاتناشقاندىن بېرى مەيلى ماڭا كەسپىمدىن باشقا ئورۇنلاشتۇرۇلغان خەن تىلى ، تارىخ ، ئۇنۋىرسال ئەمەلىيەت ، گۈزەل - سەنئەت ، يېڭىشەھەر دەرىسلىكى ( ناھىيە ئۆزى تۈزگەن دەرسلىك ) قاتارلىق پەنلەردىن دەرس بېرىش ھۆددىسىدىن تولۇق چىقىپ ، ئوقۇ -ئوقۇتۇش جەھەتتە « بىلگىنىمنى ئەلدىن ئايىمىدىم » .
    ئۇندىن باشقا مەكتەپلەرنىڭ تۈرلۈك ئەدەبىيات - سەنئەت پائالىيەتلىرىدە ئەمەلىي ئىشلەپ مەكتەپنىڭ شۇنداقلا باشقا مەكتەپ ۋە بازارلىق ھۆكۈمەتنىڭ بۇ جەھەتتىكى خىزمەتلىرىنى ئۇلار قايىل بولغۇدەك ، رازى بولغۇدەك ئىشلەپ كېلىۋاتىمەن . بۇ جەھەتتە مەن خېلى يامان ئەمەس بولغاچقا بازارلىق ھۆكۈمەت ۋە ناھىيە بويىچە ئۆتكۈزۈلگەن مۇسابىقىلەردە 2- 3 - دەرىجىلىك مۇكاپاتلارغىمۇ ھەم نائىل بولدۇق . ( پۈتۈن بازاردىكى چوڭ پائالىيەتلەرگە چوقۇم مەن بىۋاستە قاتنىشىپ مۇزىكىلارنى ئاكوردلاشتۇرۇش ، تەشكىللەش ، يېتەكلەش قاتارلىق ئىشلارنى چوقۇم مەن ئىشلەيمەن .)

     مەن 2002 - يىلى ۋە 2006 - يىلى ئايرىم - ئايرىم ھالدا مىنىڭ ئورۇندىشىمدا شىنجاڭ ئۈن _ سىن نەشىرىياتى ۋە جوڭگو مىللەتلەر ئۈن - سىن  نەشىرىياتى تەرىپىدىن نەشىر قىلىنغان قاپلىق ئۈن ئالغۇ لېنتىسى ۋە سىن تەخسىلىرىنى نەشىر قىلدۇرغان . ( بۇ ئۈن - سىن لەردە مىنىڭ 10 نەچچە ئىجادى ئاھاڭلىرىم بار ، ھېلىھەممۇ داۋاملىق ئىجادىيەتتىمەن . بۇ ئۈن - سىن بويۇملىرىنىڭ نامى يەنىلا ھازىرچە مەخپى .)
    يەنە مەن ئىمتاھان بەرگەن يىلى مىنىڭ ئوقۇتقۇچى تولۇقلاش ئىمتاھانىدىكى كەسپى نەتىجەمنىڭ قەشقەر ۋىلايىتى بويىچە 1 - بولۇشى ۋە 2012 - يىلى 11 - ئايدا يەكەن ناھىيىسىدە ئېلىپ بېرىلغان « ئالاھىدە كەسىپ ئوقۇتقۇچىلىرىنىڭ دەرس ئۆتۈش مۇسابىقىسى » دە  1- دەرىجىگە ئېرىشكەنلىكىم مىنىڭ سىز ئېيتقاندەك « ھۇرۇن ، خىزمەتتە ئەستايىدىل ئەمەس ، ئىزدىنىش روھى كەم ، بىر قىسىم قاخشاش كىسىلى ئېغىر كىشىلەر» ئەمەسلىكىمنىڭ دەلىلى ئەمەسمۇ ؟!
    خوش ، ئەمدى ئوقۇتقۇچىلار ئەخلاقىغا دائىر مەزمۇنلاردىن ئۇچۇر بېرەي .
   ئوقۇتقۇچىلار قوشۇنى ھەقىقەتەنمۇ بىر چوڭ قوشۇن . بۇنداق كىشىلەر توپى ئىچىدە مۇنداق ياكى ئۇنداق كىشىلەرنىڭ بولۇشى تەبىئي ئەھۋالغۇ ؟ بۇ __«  ساپ ھاۋالىق ، خەندان ناۋالىق باغۇ - بوستانلىقلاردا زەھەرلىك يىلانلارنىڭ بولۇشى ئەجەپلىنەرلىك ئىش بولمىغاندەكلا بىر ئىش » .
    ئەمما مېنىڭ ھايات مىزانىمدا سىزگە ئوخشاش خوشامەتچىلىك ، مەنمەنچىلىك ، نادان ، مەقسىتىگە يېتىش يولىدا ۋىجدانىنى دەپسەندە قىلىۋىتىدىغان ، ئوقۇتقۇچىلارنى چۈشەنمەيدىغان ، ھەم چۈشىنىشنى خالىمايدىغان، ئۆزىڭىز ئېتىراپ قىلغان مىتودى يوق ساپاسىز ئوقۇتقۇچىلار ياكى مەكتەپ مۇدىرلىرى ئۆلۈك بىلەن باراۋەردۇر .
    مەن گەرچە مۇزىكا كەسپىدە ئوقۇغان ھەم شۇ كەسىپتە شۇغۇللانغان بولساممۇ ھاراق - شاراپتىن خالىي ، بىمەنە ۋە تېتىقسى سورۇن ، مەدەنىيەتسىز گەپ - سۆزدىن يىراق ،سەمىمى ، ۋەدەمگە ۋاپا قىلىدىغان ، مەيلى ئوقۇتقۇچى ياكى ئوقۇغۇچىلارنى چۈشىنىشنى ئۇلار بىلەن يېقىنلىشىشنى خالايدىغان ، خۇشچاخچاق ، خۇشپېئىل ، « مەكتەپتىكى بالىلار ئۆزۈمنىڭ بالىلىرى » دىگەن ئۇقۇمغا ئىگە ھەم مەسئۇلىيەتچان ، تەلەپچان ، قولۇمدىن كېلىشىچە باشقىلارغا ياردەن قىلىشنى خالايدىغان ، مەسىلىلەرنى ھېسسىياتقا ئەمەس ، ئەقىلگە ، قانۇنغا تايىنىپ بىر تەرەپ قىلىدىغان ، چوڭلارنى ھۆرمەتلەيدىغان ، كىچىكلەرنى ئىززەتلەيدىغان ھەم ( مېنىڭچە ) بىرقەدەر ئەتراپلىق تەربىيە كۆرگەن بىر خەلق ئوقۇتقۇچىسىمەن .
     ھازىرغا قەدەر بىر قارار ئوقۇغۇچىلىرىم  ( يەنە بىر قارار ئوقۇغۇچۇم بار ئىدى ، رەھبەرلىك تېخى يېقىندا  مىنى « مەكتەپنىڭ ئەدەبىيات - سەنئەت ئىشلىرىنى جانلاندۇرىدۇ »، دىگەن گەپ بىلەن سىنىپ مۇدىرلىق ۋەزىپەمنى ئېلىپ تاشلىدى ، ئەمما دەرسىم ئەسلىدىكى ھەپتىسىگە 8 سائەتتىن 15 سائەتكە كۆپەيتىلدى ) ئوقۇش پۈتتۈردى . ئۇلار ۋە ئۇلارنىڭ ئاتا - ئانىلىرى تا ھازىرغا قەدەر مىنى ئىزدەپ تۇرىدۇ ، مەنمۇ شۇ . ئۇلارمۇ خۇددى ئۆزۈمنىڭ تۇققانلىرىمدەك تۇيۇلىدۇ . بۇ قېتىم سىنىپ مۇدىرلىقتىن قالغاندىن كىيىن ئوقۇغۇچىلىرىم مەكتەپكە 3 قېتىم ئىىلتىماس سۇنۇپ ، مىنىڭ ۋەزىپەمنى ئەسلىگە كەلتۈرۈشنى تەلەپ قىلدى . قايتا - قايتا رەت قىلىشقا ئۇچرىغاندىن كىيىن بۇ بالىلار ناھىيىلىك مائارىپ ئىدارىسىغا بېرىپ تەلەپ قىلماقچى بولغاندا مىنىڭ گىپىم بىلەن توختاپ قالدى . ھەتتا بۇ بالىلارنىڭ ئاتا - ئانىلىرىمۇ مەن بىلەن بىۋاستە كۆرۈشۈپ ياكى تېلىفون قىلىپ « قانداق قىلىپ بولمىسۇن بالىلارنى قولۇمغا ئېلىشىمنى تەلەپ قىلدى » . ئۇندىن باشقا مەكتەپتىكى مەن دەرس بېرىدىغان ياكى بەرمەيدىغان ، مىنىڭ سىنىپىمدا بولمىغان نۇرغۇن ئوقۇغۇچىلار ( ئاز دىگەندە %70 ئوقۇغۇچى ) مىنىڭ دەرس بېرىشىمنى خالايدۇ ۋە ھەر كۈنى نەچچە يۈز ئوقۇغۇچى مەن بىلەن قول ئېلىشىپ كۆرۈشۈشكە كېلىدۇ . مەن بۇنىڭدىن بەكمۇ تەسىرلىنىمەن .ئەجىبا مەن بۇلارنى بۇ ئىشلارغا مەجبۇرلىغاندىمەنمۇ ؟! بۇ ئىشلار مىنىڭ خىزمەتنى قانداق ئىشلىگىنىمگە ، ئۇلارغا قانداق تەسىر بەرگەنلىكىمگە دەلىل بولالا ؟!

     دىمەك ، مەن ھەقىقەتەنمۇ « خەلق رازى بولىدىغان مائارىپ » بىلەن شۇغۇللىنىشقا تېرىشتىم . مەن دۆلەت بەرگەن ئىش ھەققىگە لايىق ئىشلىيەلمىسەم ئۆزۈمدىن ئۇيىلىمەن .
  بىزدە « ئەجدادنىڭ ئىزىدىن تىكەن ئۈنسە ، ئەۋلادنىڭ تاپىنى قاناپتۇ » دىگەن گەپ بار . ئوقۇتقۇچىنىڭ  ئوقۇغۇچىلارغا بىلىم بېرىشىدىن باشقا ، ئۇلارغا قانداق ئادەم بولۇش يولىنى بىلدۈرۈش مەجبۇرىيىتى بار .
    سىز تىلغا ئالغان مەمتىلى ئەپەندى مىنىڭ نەزەرىمدە شىنجاڭ يېقىنقى زامان تارىخىدىكى ئەڭ مۆتىۋەر ھەم ھۆرمەتلىك كىشى . ئەمما ئۇ كىشى 6 ئاي ئىچىدە 24 مەكتەپ قۇرۇشتەك ئۇلۇغۋار ئىشنى سىزگە ئوخشاش خۇشامەتچىلىك يولى بىلەن ئەمەس ، ئۆزىنىڭ مول ھەم ئەتراپلىق بىلىمى ، يۈكسەك ئەخلاق - پەزىلىتى ۋە ھەقىقىي مائارىپچىلاردا بولۇشقا تىگىشلىك ئېسىل روھى بىلەن ئىشقا ئاشۇرغان . سىزنىڭ يۇقۇرقىدەك « ئىش بىجىرىش » ئۇسۇلىڭىزنى ئەتراپىڭىزدىكى ئوقۇغۇچى - ئوقۇتقۇچىلار بىلمەي ، ئاڭلىماي قالارمۇ ؟ سىزدەك بىر مەكتەپنىڭ بېشى بولغان ئادەم باشقىلارغا ئاشۇنداق چاكىنىلىقتا ئۈلگە بولسا ھە ! سىز كىيىنكى ئەۋلادلىرىمىزغا شۇنداق پاسىقلىقنى مىراس قالدۇرماقچىمۇ ؟ مەسىلىنى ھەل قىلىشنىڭ يولى پەقەت بىرلا ئەمەسقۇ ؟ سىز شۇ ئىشلىرىڭىزدىن خىجىللىق ھېس قىلمامسىز ؟!
    كەچۈرۈڭ قېرىندىشىم ، بەلكىم بەزى گەپلىرىم قاتتىق كەتكەندۇ ، ئېغىر ئالمىغايسىز . ئەگەر يۇقارقى مەن توغرىسىدىكى سۆزلىرىمگە ئىشەنمىسىڭىز سىز مەيلى قەيەردىن بولۇڭ مەكتىپىمگە ، مەھەللەمگە كېلىپ ئېنىقلاپ باقسىڭىز بولىدۇ . ئەگەر بىرى يالغان بولسا سىزنىڭ بارلىق چىقىملىرىڭىزغا مانا مەن ئىگە !
ھە راست ، ئۇ مەكتەپ مۇدىرى بىلەن ئىلگىرى ھېچقانداق ئاداۋىتىم يوق ئىدى .  ئەگەر ئۇ ئۆز ئاكام بولغان بولسىمۇ شۇنداق قىلغان بولاتتىم .
   مېنىڭ مەقسىتىم ، شۇ جاپاكەش ئوقۇتقۇچىلارمۇ جەمئىيەتتىكى ھەر ساھە كىشىلىرىدەك ئۆزىنىڭ قانۇنىي ھوقۇق مەنپەئەتىدىن تولۇق بەھر ئالسا ھەم ئۆز مەجبۇرىيىتىنى تولۇق ئادا قىلسا ، دىگەن ئۈمىددىمەن .
    ئەنگىلىيىنىڭ سابىق باش ۋەزىرى چېرچىل __ « رەزىللىك شۇنىڭ ئۈچۈن كۈچىيەلەيدۇكى ، ئۇ ئۆزىنىڭ كۈچلۈك بولغانلىقىدىن ئەمەس ،بەلكى  ياخشى ئادەملەرنىڭ ئورنىدىن دەس تۇرۇپ  ئۇنىڭغا قارشى تۇرمىغانلىقىدىن كۈچىيەلەيدۇ » __ دىگەنىكەن .
    دۆلىتىمىزنىڭ قۇرغۇچىسى ۋە دۆلەت رەئىسى بولغان ماۋزېدوڭمۇ __ « ئىشلەشنىلا بىلىپ ، دەم ئېلىشنى بىلمىگەن كادىر ياخشى كادىر ئەمەس »  __ دىگەنىكەن .
شۇنداق بولغاندا ناتوغرا ئىشلارغا سىزمۇ ، مەنمۇ ، باشقىلارمۇ ساداسىز « خوش » دەۋەرسەك ، قانۇننىڭ نوپوزىنى جۈملىدىن ئۆزىمىزنىڭ ھوقۇق مەنپەئەتىنى كىم قوغدايدۇ ؟!
   « بەدەنگە چىققان كۆيدۈرگىنى ئاغرىيدۇ دىمەي ۋاقتىدا داغلىۋېتىش لازىم ، بولمىسا ئۇ جانغا زامىن بولىدۇ » ( ئويغانغان زېمىن رومانىدىن ئېلىندى )
   شۇڭا بىزنىڭ شۇنداق قىلىشىمىز ، ئەمەلىيەتتە شۇ چىرىكلىشىش چىگرىسىغا بېرىپ قالغان قىسمەن ئەمەلدارلارغا بېرىلگەن ئاگاھ سېگنالىدۇر .
   قېنى قېرىندىشىم « ناخشا » ! مىنىڭ يۇقۇرىدا تىلغا ئالغان يازمىلىرىمنى ئاغدۇرالىغۇدەك دەلىلىڭىز بولسا ئوتتۇرىغا قويۇڭ . مەن ھەر قاچان تەييار !
   ( ئوقۇرمەنلەر سەمىگە : شۇ كۈننىڭ ئەتىسىدىن باشلاپ مەن مەكتەپتە ئويلاپ باقمىغان ئىشلارغا يولۇقتۇم . قالغىنىنى قېرىنداشلار بىلەن بارا - بارا ئورتاقلىشارمىز . كەچۈرگەيسىلەر ، قىممەتلىك ۋاقتىڭلارنى ئېپ قويدۇم .

شەخسەن مەن مەكتەپ مۇدىرىنىڭ، مەمۇرىيىتىنىڭ ئىلمىي باشقۇرۇشنى يولغا قويۇپ، ئوقۇتۇش سۈپىتىنى كۆزلىگەن ئاساستا ئىلمىلىق بىلەن تۈزۈلگەن تۈزۈملەرنى يولغا قويۇپ، جانلىق باشقۇرۇپ، ئەمەلىي ئىشلىگەنلەرنى مۇكاپاتلاپ، كۈن ئۆتكۈزۈش يولىغا كىرىۋالغان، ئەۋلات، مىللەتنىڭ كەلگۈسىنى ئويلىمايدىغان، ئىش ئاقسىتىدىغان بىكار تەلەپلەرنى خىزمەت قىلىشقا مەجبۇر قىلىدىغان تۈزۈملەرنىڭ يولغا قويىشىنى، شۇنداقلا ئوقۇتقۇچىلارنىڭ يوللۇق تەلىپىنى قوبۇل قىلىشنى، ئوقۇتقۇچىلارنىڭ قانۇنلۇق مەنپەئەتىنى قوغدايدىغان بولىشىنى ، ۋە قوغدىغۇچىلارنى دۈشمەن سۈپىتىدە كۆرمەسلىكىنى، ئۆچ ئالماسلىقىنى ئۈمۈد قىلىمەن . لىكىن سىز قانداق مەن بىلمىدىم،   لىكىم بىزنىڭ مۇدىر (مۇشۇ تېمىدىكى مۇدىر مېنىڭمۇ مۇدىرىم) ۋە باشقا ئاتالمىش مۇدىرلار داۋاملىق تازلىق، باش كىيىم (دوپپىنى ئەلۋەتتە) كەيمەسلىك، ئوقۇغۇچىلارنىڭ كاللىسىنى سىياسىي ئىددىيە بىلەن تولدۇرۇش.... قاتارلىق   شەكىلۋازلىق بىلەن ئادەمنىڭ ئۆچلۈكىنى كەلتۈرىدۇ. ئۇلارنىڭ ئۇيغۇر مىللىتى بىلەن ئوقۇغۇچىلارنى يىراقلاشتۇرۇش ئۈچۈن تۈزگەن، قىلغان گەپ سۆزلىرى سەل ۋىژدانى بار كىشىلەرنىڭ ئوغىسىنى قاينىىتىدۇ. بەالكىم سىزمۇ بىلەرسىز. توغرا ئۇيغۇردا مەكتەپ مۇدىرى بولماق ھەقىقەتەن تەس. يۇقىرىدىنمۇ بېسىم كۆپ، مەكتەپتىكى قاملاشمىغان، تۈزەشكە تىگىشلىك ئىشلار ھەم بىر تالاي. لىكىن چوقۇم ھەققانىي، مىللەتسۆيەر بولىشى كېرەك. بولمىسا نېمە دىيىشىڭىزدىن قەتئىنەزەر مەن مۇناپىق دەيمەن خالاس.

مەنمۇ ئەسلى ئۇقۇتقۇچى. ئۇقۇتقۇچى بۇلۇشنىڭ تەسلىگىنى مەن ناھايىتى ئۇبدان بىلىمەن. بۇ ھەقتە باشقىلارمۇ چۈشەندۇرۇپ ئۆتتى. مەن قايتا تەكرارلاپ ئولتۇرماي. ھازىر مەكتەپ مۇدىرلىق قىلىۋاتىمەن. گەپنىڭ راستىنى ئېيتقاندا مەكتەپ مۇدىرى بولماق ئۇمۇ ئاسان ئەمەسكەن. ھازىر مەكتەپنىڭ خىزمەتلىرىنى تەكشۈرۈشكە چارۋا مال دۇختۇرلۇق ئۇرۇنلىرىدىن باشقا ھەممە ئۇرۇن دىگۈدەك مەكتەپ خىزمىتىنى تەكشۈرۇپ خىزمەتلەرگە باھا بىرىۋاتىدۇ. گەرچە ھەرقايسى خىزمەتلەرگە مەسئۇل خادىملار بولسىمۇ ئەمما مەكتەپ مۇدىرى 1-مەسئۇل كىشى بولغاچقا يەنىلا مەكتەپ مۇدىرى جاۋاپ بىرىش كىرەككەن. بۇ ھەقتىمۇ كۆپ توختالماي. پۇل تۇتۇشتىن باشقا ئامال يوقمۇ؟ تۇتۇشنى بىرىشكە ئۆزگەرتسە بولماسمۇ  دىگەن سۇئاللار ھىقىقىتەن ياخشى سۇئاللار بولدى. بۇ سۇئال مىنىڭ كۆڭلۈمدىكىدەك سۇئال بولدى. مەۋسۇم ئاخىردىكى مۇكاپات پۇللىرى؛  مەكتەپ فورمىسىنىڭ مەكتەپ قۇشۇپ بەرگەن پۇللىرى؛ ئارتۇق دەرىس ئۆتكەن ئۇقۇتقۇچىلارغا بەرگەن ھەر سائەتلىك پۇللار؛ خىزمەتتىن سىرت دەرىس تەكرارلاشقا بارگەن ھەقلەر؛ ھەر ئايدا ئۇقۇغۇچىلارنى دەرىسكە قاتناشتۇرۇش نىسپىتى بەلگۈلەنگەن ئۆلچەمگە يەتكەن سىنىپ مۇدىرلىرىغا بەرگەن مۇكاپات پۇللار (سىنىپ مۇدىرلىق ياردەم پۇلى بۇنىڭ سىرتىدا)؛ ھەرخىل مۇسابىقىلارغا قاتنىشىپ نەتىجىگە ئىرىشكەن ئۇقۇتقۇچىلارنى مەكتەپ ئىلھام بىرىش يۈزىسىدىن بىرىلگەن مۇكاپات پۇللار؛ ئۇقۇغۇچىسى يۇقۇرى دەرىجىلىك ئۇرۇنلار تەشكىللىگەن ھەر خىل پائالىيەتتە نەتىجىگە ئىرىشكەن ئۇقۇتقۇچى ھەم  ئۇقۇغۇچىغا بەرگەن مۇكاپات پۇللىرى؛ ھېيت-بايراملاردىكى تەمىنلەشلەر؛ يۇقۇرى ئۆرلەپ ئۇقۇشقا (ئىچكىرى تۇلۇق ئوتتۇرۇغا)ئۇقۇغۇچى يەتكۈزۈپ بىرىشكە ھەسسە قوشقانلارنى مۇكاپاتلاش؛ سۈپەت تەكشۈرۈشتە نەتىجىسى گەۋدىلىك بولغان ئۇقۇتقۇچىلارنى مۇكاپاتلاش.... قاتارلىق ئۇسۇللار بىلەن ‹‹بىرىش›› مۇ ھەر ھالدا ئەمەلگە ئېشىۋاتىدۇ. (بەرسەڭ ئۈزۈڭنىڭكىنى بەردىڭمۇ دەيدىغانلىغىڭلارنى بىلىمەن)بۇ لارنى ئۇقۇتقۇچىلارنىڭ تارتقان ئەجىرىگە ھەرگەز تەڭ قىلغىلى بولمايدۇ. چارى ئامال مۇشۇنچىلىك. لېكىن ھازىرغىچە مەكتەپتىكى پۇللارنى بارلىق ئۇقۇتقۇچىلارغا تەخسىم قىلىپ بىرىڭ دىگەن ھۆججەت ئەسلا كىلىپ باقمىدى. مەكتەپتە پۇل كۆپ لېكىن ھىساپ مائارىپ ئىدارىسىدا بۇلۇپ جۈجاڭ تەستىقلىمىسا مەكتەپ بىر پۇڭ چىقىم قىلغىلى بولمايدۇ. قالدىسقا نەشپۈتنىڭ تەمىنى بىلەي دىسەڭ تېتىپ باق دەپتىكەن...  يەنا بىر ئىش سىز مۇزىكا ئۇقۇتقۇچىسى ئىكەنسىز. بۇنداق ئۇقۇتقۇچىلار بىزنىڭ مەكتەپتىمۇ بار بۇ كەسىپ بەكمۇ ئەركىن كەسىپ. سىز قىلغاننى باشقىلار قىلالمايدۇ. سىز مەكتەپ مۇدىرىنى چۈشەنمەيدىكەنسىز. ئۇبدان دىققەت قىلسىڭىز مەكتەپ مۇدىرى ھەممىدىن سىزگە ئوخشاش ئۇقۇتقۇچىغا بەك ئامىراق. چۈنكى سىز ئالغان شان-شەرەپ ماھىيەتتە مەكتەپنىڭ شان-شەرىپى شۇنداقلا مەكتەپ مۇدىرىنىڭ شان شەرىپى. مەسلىھەتىم : مەكتەپ مۇدىرى بىلەن بىر قېتىم دوستلارچە پىكىرلىشىپ بېقىڭ. ئاخردا چۈشۈنەلەيسىز. ئەڭ ئاخىردا مەنمۇ سىزگە ئوخشاش ‹‹ئۇقۇتقۇچىلار قانۇنى››نىڭ ئەمىلى ئىجىرا قىلىنىشىنى تەقەززالىق بىلەن كۈتۈۋاتىمەن.

    ئەي مەكتەپ مۇدىرى قېرىندىشىم ! مەن قاچان سىزگە مەكتەپ مۇدىرى ماڭا ، ياكى مەن مەكتەپ مۇدىرىغا ئۆچ دەپ باقتىم ؟ « ئىسنىڭ ئاچچىقىنى مورا بىلىدۇ » نى ئىنكار قىلماقچىمۇسىز ؟ مۇشۇ ۋەقە يۈز بەرگەنگە قەدەر مەن مەكتەپ مۇدىرى بىلەن 3 قېتىم سائەت - سائەتلەپ سۆھبەتلەشتىم .ئۇ چاغدا سىز بارمىتىڭىز؟ مەن بۇ مەكتەپ مۇدىرىنى تولۇق لاياقەتسىز قاتارىغا چىقارغىنىم يوق ، ھەم ئۇنداق ھوقوقىممۇ يوق . قالغىنىغا يەنى مەكتەپ مۇدىرىنىڭ ياكى مىنىڭ قىلغىنىمنىڭ توغرا ياكى خاتالىقىغا كەلگۈسىدە تارىخ ناھايىتى ئادىل ھۆكۈم چىقىرىدۇ .

    مىنىڭ ، سىزنىڭ قانداق قىلىپ بىر مەكتەپنىڭ مۇدىرى بولۇپ قالغىنىڭىزغا گۇمانىم بار . سىز بىردەم  مىنىڭ قىلغىنىمدىن ئاغرىنىپ مۇدىرىمىزنى قوغدايمەن دەيسىز ، ئەمدى بولسا  « مەنمۇ سىزگە ئوخشاش ‹‹ئۇقۇتقۇچىلار قانۇنى››نىڭ ئەمىلى ئىجىرا قىلىنىشىنى تەقەززالىق بىلەن كۈتۈۋاتىمەن.» دەپ ئۆز سۆزىڭىزنى ئىنكار قىلىسىز ؟! ئۇنداق ئەمەسمۇيا ؟

مائاللىم دىگەن مانا مۇشۇنداق ئاسان ئاچچىقلىنىدۇ ئەمە ...

سىزنىڭ قانداق بولىشىڭىز بىلەن مېنىڭ كارىم يوق. باشقىلارغا كايىپسىز. ھازىرقى مالىملار مائاش ئېلىشتىن باشقا نېمىنى بىلىدۇ؟ باشقىلار مائاشىنى تۇتسا ئەرز قىلمايدۇ. شەنبە- يەكشەنبە ئىش ئورۇنلاشتۇرسا ئات - ئىشەكتەك قىلىدۇ. بۇنىغۇ ئۇلارنىڭ ئىنتىزامچانلىقى تۈپەيلىىن قىلغاندۇ دەپ چۈشىنەيلى (كۆپلىگەن ئوقۇتقۇچىلار مائاش سەۋەبىدىن قىلىدۇ. بۇنى ئىنكار قىلالمايسىز).
ئوقۇتقۇچىلار ئوقۇغۇچىلارنى ئەخلاقلىق تەربىيىلىشى كېرەك. ئەخلاقلىق ئوقۇتقۇچىلار، جەمئىيەتتە ئەخلاق سەۋەبىدىن كىشىلەر تەرىپىدىن ھۆرمەتلىنىدىغان ئوقۇتقۇچىلار قانچىلىك؟
ئوقۇتقۇچىلار ئوقۇغۇچىلارنى ئۆزى ئىزدىنىدىغان، ئۆزى تىرىشىدىغان قىلىپ تەربىيىلىشى كېرەك. بۇنىڭغا ئۆلگە بولالايدىغان ئوقۇتقۇچىلار قانچىلىك؟ ئوقۇتقۇچىلار سەۋرچان بولۇشى كېرەك. بۇنداق ئوقۇتقۇچىلار قانچىلىك؟ ئوقۇتقۇچىلارنى يېغىندا مۇدىر «بەزىلەر» دەپ ھەرخىل ئىبارىلەر تەنقىدلىسە يېغىندا ئولتۇرغانلار «ئۇنداقلارنى ئايرىم چاقىرىپ كۆرسەتمە بېرىڭ، بىزنىڭ ۋاقتىمىزنى كاشىلا قىلماڭ، بىز ئوقۇ-ئوقۇتۇش قىلىدىغان ئالتۇن ۋاقتىمىزنى ئىسراپ قىلماڭ» دىيەلەيدىغان قانچىلىك؟ (مۇنبەردىغۇ بىرەرسىمىزنىڭ دىتىغا ياقمايدىغان سۆزلەرنى كۆرسەك بەتلەرنى رەددىيە بىلەن تولدۇرىۋېتىمز). ئوقۇتقۇچىلار ئارىسىدا تۈگىمەس-پۈتمەس يېزىق ئالۋانلىرىنى ئاددىيلاشتۇرۇپ، كۆپلىگەن ۋاقتىمىزنى ئوقۇ-ئوقۇتۇش تەتقىقاتى بىلەن شۇغۇللىنىپ باققانلار قانچىلىك؟ سىنىپتا ئويلاپ باقمىغان ئۆگىتىش ھادىسىلىرى، ئوخشاش بولمىغان قىلمىشلار يۈز بېرىدۇ. مۇشۇنداق مەسىلىلەرگە قارىتا ئەڭ ئۈنۈملۈك ئامال ئۈستىدە ئىزدىنىپ باققان ئوقۇتقۇچىلار قانچىلىك؟ مەكتەپ مۇدىرلىرى ناچار بولسا، مەكتەپ مۇدىرى ئورنىدا ئىلمىي، ئوقۇ-ئوقۇتۇشقا پايدىلىق تاكتىكىلار ئۈستىدە ئىزدىنىپ باققانلار قانچىلىك؟ (پۇلنى يەمچۈك قىلىپ تاپقان تاكتىكىلار ھېساب ئەمەس). ئۆزىنىڭ ھوقۇق- مەنپەئەتىنى قوغداشنى بىلىدىغان ئوقۇتقۇچىلار قانچىلىك؟ ئادەم تەربىيىلەيدىغان كىشىلەر ئۆزىنىڭ ھوقۇق-مەنپەئەتىنى قوغداشنى بىلمىسە، ئۇلارنىڭ قولىدا تەربىيىلەنگەنلەر ئۆزىنىڭ ھوقۇق مەنپەئەتىنى قانچىلىك قوغدار؟ سەن قانداق ئادەم؟ سەن كىم؟ نېمە قىلماقچى؟ دېگەندەك ئىبارىلەر ئارقىلىق سوراپمۇ؟
