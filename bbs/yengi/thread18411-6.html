ئەخمەتجان قاسىممۇ ئۇيغۇرمۇ ؟خۇجىنىياز ھاجىنىڭ ئاچكۆزلۈكى ،تەكەببۇرلۇقى ۋە ئىشەكچىلىك ئەقىلىنىڭ يوقلۇقى بىلەن مەشھۇر زات دىيىلسە سىلىچە قانداق بۇلار ؟
______________________________________________________________
سەن ئاۋال ئۇيغۇر بول ،ئاندىن بۇرۇن تارخى   رومانلىرمىزنى يانچە ئۇسۇرۇپ يېتىپ ئۇقۇغان بولساڭ ، راۋرۇس تۇرۇپ ھەر بەتتە بىر توختاپ ھەزىم قىلىپ ئۇقۇ ،تۇرۇپلا پەيدا بوپ قالدىغان كۆپ چىكىتلەرگە دېھقەت قىل، مانداقلا ئاخېرلىشىپ قالغان جۈمۈللەرنى ئەقلىڭ بىلەن تىڭشاب كۆر ...چۇنكى كىتابنى ئاپتورلار مىڭ تەستە يىغىپ يازاي دىگەندە ھۆكۈمەت قايلاپ تۇغان  ،ئامال يوق مەزمۇنلارغا يۇشرۇن  سېڭدرۋەتكەن ،ياكى كۆپ چىكىت ئۇرۇپ ساڭا چۇشنۈشكە ھاۋالە قىلغان  (يەن كۆپ قىسمىنى  تەھررىرلەر  دىگەن كىشلەر <<قىسقارتىۋەتكەن>> )ئەگەر يېزىب سالسا ئاپتۇرلا ئەمەس ئۇشبۇ كىتابلارمۇ ‹‹سىر›› بۇلۇپ قالاتتى !!! شۇڭا ۋەزيەتكە باققان ...  شىنجاڭنىڭ يەرلىك تارخى دىگەن كىتابدا ‹‹ ھۆكۈمۈتىمىز ›› بىر قىسىم ‹‹ئۇنتۇلغۇسىز›› كىشلىرمىزنى ئۇچۇقتىن ئۇچۇق ‹‹ ۋەتەن ساتقۇچ›› دەپ ئىنىقلىما بەرگەن ، شۇڭا ئامال يوقتە........ يەنە بىر قىسىم ۋەقەلەرگە ئاپتور ھىچقانداق ئىپادە بەرمىگەن ،مەسسلەن : ئانا يۇرتنىڭ ئاخېرقى بەتلىرنى 3 قېتىم ئاستا ئۇقۇپ چىق! ئىشەك كاللاڭ ئىشلەپ قالا ...يەنە مەسلەن ھېس قىلساڭ كىتاپتا قايسى بىر ‹‹قىزىل›› كىتابلاردەك ‹‹ۋەتەن پەرۋەر›› دىگەندەك  ئاتالغۇ يوق دىيەرلىك ،سەۋەبى ئاپتور كىيىنكىلەرنىڭ ھۆكۈم قىلالايدىغانلىقىغا ئىشەنگەن...ئەمدى سەندەك كۆپ قىسىم چالا ساۋادلارنىڭ ئۇسرۇپ يېتىپ ،بەت ئاتلاپ ئۇقۇۋالغان  نەچچە ‹‹ئۆزلەشتۈرمە›› جۈملسى دوقمۇشتىكى چالا كەيىپ  ماڭقۇرتلارنىڭ گېپى بىلەن   يۇغرلۇپ چوتكا كۆرمىگەن ئېغىز بىلەن قاشاڭ ياغاچ قۇلاقلاردىن ئۈتۈپ تۇلا ئىستىمال قىلنىپ مانا ئەمدى ئىىتتەك قاۋاپ ،شىردەك بەھەيۋەت كىشلەر نامنى دەپسەندە  قىلىپ ........سەندەكلەرنى تولا كۈرۈپ ھۆ بۇلۇپ كەتتىم .ئۈزەڭلەرنى بىلىملىك  پەردازلاش ئ!ۈچۈن قالايماقان بېلجىرلاب ئاپتۇرلارنىڭ قان تەرنى 0 قىلسىشسەن، توۋا !!!!!سەندەكلەر نەچچە يىل ماتېريال توپلىغانلارنىڭ يازمىسىغا ئىشەنمەي نەدىكى دوقمۇش پارىڭنى رادىيو ئاڭلىغاندەك ئاڭلاپ كتىشسەن ، شۇڭا ئايالېرمىز ‹‹ ھازېرقى ئەرلەر غەيۋەتخور›› دەپ ئاسسلىق سۆزلەيدىغان بوپ كېتىشتى....
  قالغان مەزمۇنلارنى مەن بۇيەردە تەپسىلى يىشىپ ئولتۇمايمەن ، شۇنداق كۈنلەر كىلدۇ ،ئاشۇ كىتابلارنى خەتمۇ خەت  تەھلىل قىغىلى...ھەي!!!!!!!!!!ئاچىقىمدا يېرلاي دىدىم مەن ھازېر........
