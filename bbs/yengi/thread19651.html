   كونا كادىر (ھېكايە)





ئازاد شاڭجاڭ كۆتۈرۈلۈپ كەتتى، مۇئاۋىنلىقتىن باش شاڭجاڭلىققا ئۆستى، بويى گويا بىر غېرىچ ئۆستى، ئىشتىھاسى ئېچىلىپ كەتتى، قورساق سالدى، زۇۋانى چوڭايدى، مەجلىس ئاچسا مەيدىسىنى كېرىپ، ياقىسىنىڭ تۈگمىسىنى كەڭ ئېچىۋېتىدىغان بولدى. ئۇنى ئەمدى ھېچكىم قوڭالتاق يېزا كادىرى ياكى مۇئاۋىن غالچا دېيەلمەيتتى. خەقنى مۇشۇنداق دېگۈزمەسلىك ئۈچۈن قانچىلىك بەدەللەرنى تۆلىگەنلىكىنى بىر ئۆزى بىلىدۇ، جاھاندا ئۇنداق ئاسان ئۆسىدىغان ئىش نەدە بار؟ شۇڭا ئۇ ئادىتى بويىچە ھەر كۈنى سەھەر تۇرۇپ خەنزۇچە مەدھىيە سۆزلىرىنى، مەنىسى قايمايدىغان، شۇجىلارغا ھەسەلدەك ياقىدىغان تۇراقلىق ئىبارىلەرنى يادلايتتى، تەلەپپۇزىنى توغرىلايتتى، ئاندىن خەنزۇچە گېزىتلەرنى كۆرۈپ كۈندىلىك سىياسەت ۋە يۈزلىنىشلەردىن، يېڭى سىياسەت، يېڭى ئۇقۇم، يېڭى شوئارلاردىن خەۋەردار بولۇپ تۇراتتى. ئاندىن جاڭ شۇجىنىڭ ئىشخانىسىغا كىرىپ، بۈگۈنكى خىزمەتلەر توغرۇلۇق يوليورۇق ئالاتتى. بىكار ۋاقىتلىرىدا جاھاندارچىلىق سەنئىتى ياكى ئىگىلىك باشقۇرۇش كەسپىگە ئائىت خەنزۇچە كىتابلارنى كۆرۈپ، ئوقۇغان بىلگەنلىرىنى جاڭ شۇجىنىڭ تۇرمۇش ئۇسۇلى، سورۇندارچىلىقى، ئەمەلدارلىق ئەنئەنىسى، گەپ-سۆزگە، مۇناسىۋەتكە دىققەت قىلىش ئەدەبلىرىگە تەتبىقلايتتى. خەنزۇچە فىلىم كۆرسە نەدىكى ۋەقەلىك ياكى يىغلاڭغۇ مۇھەببەتكە ئەمەس، بەلكى ھوقۇق ۋە پۇل كۈرىشىدە ئۇتۇپ چىققان قەھرىمانلارنىڭ ئىش-پائالىيەتلىرىگە، ئۈستۈنلۈك قازىنىشقا يارايدىغان ماھارەتلىرىگە دىققەت قىلاتتى، نامى چىققان چولپانلارغا ئەمەس، چوڭ ئەمەلدارلارغا ھەۋەسلىنەتتى. قۇرۇلتاي، سايلام، يىغىن، تەكشۈرۈش… دېگەندەك ئىشلار ئاسماندا چاقماق چاققاندەك، يەر تەۋرىگەندەك، كەلكۈن كەلگەندەك دەھشەتلىك تۇيۇلۇپ كېتەتتى. يۇقىرىدىن چوڭ باشلىق كېلىش ئالدىدا نەچچە كېچە ئۇخلىيالمايتتى، ئۇ باشلىقنىڭ ئۆزى بىلەن كۆرۈشكەندە بولسا ھاياجاندىن تىترەپ، پۇتلىرى بوشاپ، ئاغزى گەپكە كەلمەي قالاتتى، شۇنداقتىمۇ بۇ قىممەتلىك پۇرسەتنى ئۆتكۈزۈۋەتمەسلىك ئۈچۈن ئۆزىنى تۇتىۋېلىپ، ئىككى ئېغىز مەدھىيە ۋە رەسمىيەت سۆزىنى قىلىۋالاتتى. چوڭ باشلىق: ”جاپا تارتتىڭلار!“ دېسە، ”ياق، باشلىق، سىز جاپا تارتتىڭىز“ دەيتتى. چوڭ باشلىق: ”خىزمەتلەر قانداقراق؟ “ دېسە، ”سىزدەك رەھبىرىمىزنىڭ يوليورۇقى بىلەن ھەممە خىزمەتلەر ئوڭۇشلۇق كېتىپ بارىدۇ“ دەيتتى. چوڭ باشلىق: ”مۇقىملىق خىزمىتىنى چىڭ تۇتۇۋاتامسىلەر؟“ دېسە، ” شۇنداق، بۇ خىزمەتنى قاتتىق چىڭ تۇتۇۋاتىمىز، شۇجى بىلەن ئىككىمىز بۇ خىزمەتكە بىۋاسىتە مەسئۇل. مۇشۇ ۋاققىچە يېزىمىزدىن بۆلگۈنچى ئۇنسۇرلار چىقىپ باقمىدى“ دەيتتى. چوڭ باشلىق: ”لېكىن ھۇشيارلىقنى ئۆستۈرمىسەڭلار بولمايدۇ“ دېسە، ”شۇنداق، بىز نازارەتنى، باشقۇرۇشنى قىلچە بوشاشتۇرۇپ قويمىدۇق، بۇنىڭدىن كېيىنمۇ بۇ خىزمەتنى ھەممىدىن مۇھىم بىلىپ، جان پىدالىق بىلەن ئىشلەيمىز“ دەيتتى. چوڭ باشلىق مەمنۇنىيەت بىلەن كۈلۈمسىرەپ قويسا، ئۇمۇ جاڭ شۇجى بىلەن بىرلىكتە كۈلۈمسىرەيتتى، ئەتىلا ئۆسۈپ ناھىيىگە ھاكىم بولىدىغاندەك زور ئۈمىدلەرگە تولاتتى.

بىر كۈنى ۋىلايەتلىك پارتكومنىڭ مۇئاۋىن شۇجىسى كېلىپ خىزمەتلەرنى كۆزدىن كەچۈرۈش جەريانىدا ئا شاڭجاڭدىن يۇقىرىقىدەك سوئاللارنى سورىدى، ئا شاڭجاڭمۇ پىشىپ كەتكەن جاۋابلىرى بىلەن ئۇ شۇجىنى قايىل قىلدى. شۇجىنىڭ بۇرۇنقى باشلىقلارغا ئوخشىمايدىغان بىر سۆزىگىمۇ مۇۋاپىق جاۋاب بېرىپ رازى قىلدى. چوڭ باشلىق كەتكەندىن كېيىن جاڭ شۇجى ئۇنى ئىشخانىسىغا چاقىرىپ:

ـــ باياتىن ۋۇ شۇجى ”يې تالىپۇ“ دېگەن بىر سۆزنى قىلدى، نېمە گەپ ئۇ؟ – دەپ سورىدى.

ـــ ياۋا تالىپ دېگەن گەپ،ـــ دېدى ئا شاڭجاڭ ئۆزىمۇ سەزمىگەن بىر كۆرەڭلىك بىلەن ، ــ يەنى رەسمىي دىنىي مەكتەپتە ئوقۇماي ئۆزى دىن ئوقۇپ موللا بولۇۋالغانلارنى ياۋا تالىپ دەيمىز. باياتىن ۋۇ شۇجى مۇشۇنى سورىۋىدى، مەن بۇنداق تالىپلارنى تىزىملاپ قويدۇق، ھەر ۋاقىت نازارەت قىلىپ تۇرۋاتىمىز، دېدىم.

ــ ياخشى دەپسىز، بۇنىڭدىن كېيىن رەھبەرلەر مىللەت ۋە دىنغا ئائىت ھەر قانداق گەپ سوراپ قالسا سىزلا جاۋاب بېرىڭ.

ــ بولىدۇ شۇجى، ھەر قانداق ئىشتا يۇقىرى-تۆۋەننى ماسلاشتۇرۇپ، ماسلىشىپ ئىشلەيمىز-دە!

ــ يېزىنىڭ ئامانلىق ساقلاش پۇلىنى يىغىدىغان ئىش قانداق بولدى؟

ــ 9-كەنتنىڭلا قالغان، بۈگۈن بېرىپ يىغىپ كېلەي.

ئا شاڭجاڭ شوپۇرىنى چاقىرىپ 9-كەنتكە ماڭدى. شوپۇر بالا يولدا بىر ھازا گەپ قىلماي ئاندىن تەستە ئېغىز ئاچتى:

ـــ شاڭجاڭ، يەكشەنبە كۈنى ۋاقىتلىرى چىقارمۇ؟

ـــ نېمىگە؟

ـــ تويۇم بولماقچىدى…

ـــ يەكشەنبە كۈنى ناھىيىگە يىغىنغا بارىمەن دەۋاتسام، توي دەيسەنغۇ؟

ـــ باغاقلارنى تارقىتىپ بوپتىكەنمىز.

ــــــ شوپۇر دېگەن باشلىقنىڭ كۈنتەرتىپىگە قاراپ ئىش قىلمامدۇ؟ نېمانداق كاللاڭ ئىشلىمەيدىغان نېمەسەن؟

ـــ ئۆزلىرىگە دەي دېگەن، لېكىن يۇقىرىدىن رەھبەرلەر كېلىدۇ دەپ بەك ئالدىراش بولۇپ كەتتىلە. قىزنىڭ ئۆيىدىكىلەر ئالدىرىتىپ…

ـــ بولمايدۇ، بۇنداق ئۆز بېشىمچىلىق قىلساڭ، ئاستا باشقا ئىش قىلساڭ بولىدۇ.

ـــ ئەمىسە، قاچان ۋاقىتلىرى چىقار، توينى شۇ كۈنگە بېكىتسەكمۇ بولىدۇ.

ـــ مەن ۋاقتىمغا قاراپ بېقىپ بىر نەرسە دەي.

شوپۇر بالا سالپىيىپ كەتتى، رولنى تۇتقان قوللىرى بوش تىترەيتتى، كاللىسى نەلەرگىدۇ كەتكەن بولغاچقىمۇ، ئالدىدىن كېلىۋاتقان ماشىنىغا ئۈسۈۋەتكىلى تاس قالدى.

ـــ چۈشە گۇي ماشىنىدىن!ــ دېدى شاڭجاڭ غەزەپ بىلەن ۋارقىراپ.

شوپۇر بالا ماشىنىنى يولنىڭ چېتىگە توختىتىپ، ماشىنىدىن ئاستا چۈشتى. ئا شاڭجاڭ ماشىنىدىن چۈشكەن پېتى ئۇنىڭ ئالدىغا كېلىپ تەستەك سالدى:

ـــ مېنى سوقاي دەمسەن ھېجىقىز؟ !

ـــ ….

ـــ چىقە ئەمدى، كۆزۈڭگە قاراپ ھەيدە، بولمىسا خوتۇن ئالالمايلا مىجىلىپ كېتىسەن!

شوپۇر بالىنىڭ زاڭاقلىرىدىن ئوت ياناتتى، كۆزلىرىدىن غەزەپلىك ئۇچقۇنلار چاقنايتتى.

ئا شاڭجاڭ 9 كەنت قورۇسىغا كىرىپلا كەنت باشلىقىنى چاقىرىپ سوراققا تارتىشقا باشلىدى:

ـــ بۇ كەنتنىڭ ئامانلىق ساقلاش پۇلى نېمىشقا يىغىلمايدۇ؟

ـــ دېھقانلارنىڭ بۇ پۇلغا پىكرى بار، مېنىڭمۇ پىكرىم بار،ـــ دېدى كەنت باشلىقى ئاق سانجىغان كوتمەك ساقاللىرىنى سىلاپ،- دۆلەت ئامانلىق ساقلاش ئۈچۈن ھەربىيدىن تارتىپ قوراللىق قىسىم، بىڭتۇەندىن تارتىپ ساقچىلارغىچە مائاش بېرىۋاتسا، يەنە نېمىشقا دېھقانلاردىن پۇل يىغىدۇ؟

ـــ بۇ دېگەن سىياسەت، ئۇنىڭغا ھۆرۆ-پۆرۆ دەيدىغان گەپ يوق.

ـــ شاڭجاڭ، ئويلاپ باقماملا، شۇنچە چوڭ ئارمىيە، ساقچىلار ئامانلىق ساقلاۋاتسا، يەنە نېمىشقا بۇ بىر نەچچە بىكارتەلەپ بالىلارنى ئامانلىق قوغدىغۇچى قىلىپ قويىسىلەر؟

ـــ ئارمىيە دېگەن چېگرانى قوغدايدۇ، ساقچى دېگەن ئوغرى، قاتىلنى تۇتىدۇ. مۇقىملىقنى ساقلاش ئۈچۈن ئاممىۋى كۈچلەر بولمىسا بولمايدۇ، ساقچىلار ھەممە ئۆينى تىڭ-تىڭلاپ بولالمايدۇ، شۇڭا ئاشۇنداق يەرلىك ئادەملەرنى ئىشقا سالىدىغان گەپ. مۇقىملىق ھەممىنى بېسىپ چۈشىدۇ، مۇقىملىق ئۈچۈن ھەممىنى قىلىدىغان گەپ. ئاقسىمۇ قىلىمىز، ئاقمىسىمۇ قىلىمىز، شۇڭا ماڭا تولا داۋلى سۆزلىمەي، ھازىرنىڭ ئۆزىدە پۇلنى تاپشۇرۇڭ.

ـــ ئۇلار كۈنبويى دادۈيدىن چىقماي قەرت ئوينايدۇ، ئىچىدۇ، چېكىدۇ، ئۇخلايدۇ. بۇنى كۆرگەن ئەزالار مۇشۇنداق گۇيلارغىمۇ بىز پۇل بېرىمىزمۇ دەپ تۇرۇۋالسا، مەن قانداق قىلىمەن؟ دېمىسىمۇ شۇ ئەمەسمۇ؟ ئۇلارنى بىز نېمىشقا باقىدىكەنمىز؟ ئۇلار بىزنىڭ نېمىمىزنى قوغداپتۇ؟

ـــ ئىمامنىياز شۇجى!ــ دېدى ئا شاڭجاڭ جىلە بولۇپ،ـــ ساقالغا ھۆرمەت قىلغاننى بىلىڭ. يۇقىرىنىڭ سىياسىتى بىلەن ئويناشماڭ! ئارتۇق گەپنى قويۇپ پۇلنى تاپشۇرۇڭ.

ـــ بۇ پۇلنى يىغقىلى بولمىدى، بېرىپ جاڭ شۇجىغىمۇ دەپ قويۇڭ، بۇ پۇلنى يىغىشقا بىزنىڭ ئەزالار قارشى تۇرىدىكەن.

ـــ باشقا كەنتتىن يىغىلغان پۇل نېمىشقا بۇ كەنتتىن يىغىلمايدۇ؟! گەپ سىزدە، سىز بۇ ئىشنى خەقلەرگە توغرا چۈشەندۈرمىگەن گەپ، پۇلنى قەستەن يىغمىغان گەپ! بۇ قىلمىشىڭىز ھۆكۈمەتنىڭ سىياسىتىگە قارشى چىققانلىق!

ـــ ئۇكام، بەك چاپچىپ كەتمەڭ!

ـــ سىزمۇ ھەددىڭىزدىن ئاشماڭ!

ـــ سىز تېخى يېڭى كەلدىڭىز ئۇكام، بۇ كەنتنىڭ ئەھۋالىنى ئۇقمايسىز.

ـــ قېنى دەڭا، بۇ كەنت قانداق كەنت؟ باشقا كەنتكە نېمىسى ئوخشىمايدۇ؟ ئادەملىرى شۇنداق يامانمۇ؟

ـــ ئۇكام، سىز نېمىشقا ھە دېسىلا ئىشنى يامانغا بۇرايسىز؟ بۇ يەرنىڭ ئەھۋالىنى جاڭ شۇجى ئوبدان بىلىدۇ، كەلسە شۇ ئادەم كەلسۇن، مەن ئۆزەم دېيىشىمەن.

ــــ جاڭ شۇجى بۇ ئىشقا مېنى مەسئۇل قىلدى!

ــــ شۇنداقتىمۇ دەي ئۇكام، سىز تېخى بەزى ئىشلارغا خام ئىكەنسىز.

ـــ ماڭەۋە ئوغرى!ـــ دېدى ئا شاڭجاڭ بىردىنلا سەكرەپ ـــ سېنىڭ ئۈستۈڭدىكى شاڭجاڭمەن، پېشقەدەم شۇجى دەپ ھۆرمىتىڭنى قىلغاننى بىل! پۇلنى تاپشۇرمىساڭ ۋەزىپەڭدىن دەرھال ئېلىپ تاشلايمەن!

ـــ كېتەۋە شۇم! ــ دېدى ئىمامنىياز شۇجىمۇ غەزەبلىنىپ،ــ تۈنۈگۈن كەلگەن سويمىنىڭ گېپىنى قارا، سەن تۇغۇلماستا پارتىيە ئەزاسى بولغان ئادەممەن! ماڭقاڭنى ئېيتىۋېتىپ گەپ قىل!

بۇ گەپكە چىداپ تۇرالمىغان ئا شاڭجاڭ ئىمامنىياز شۇجىنىڭ يۈزىگە بىر تەستەك سالدى. بالىسىدەك بىر ئادەمدىن تاياق يېگەن پېشقەدەم شۇجىمۇ غەزىۋىگە پايلىماي ئا شاڭجاڭنى بىر مۇشت بىلەن يەرگە يېقىتىپ، ئۈستىگە چىقىۋېلىپ قوناق سوققاندەك ئۇرۇپ كەتتى. شوپۇر بالا بىر ياقتىن ”ھاي ئۇنداق قىلمىسىلا شۇجى“ دەپ ئىمامنىياز شۇجىنى تارتىسا، بىر ياقتىن خوشلۇقتىن دىرىلداپ تىترەيتتى.

ئا شاڭجاڭنىڭ پولو يۈزى خاپاندەك ئىششىپ چىققانىدى، بۇرنىدىن قاننى تەستە توختاتتى، قان ئارىلاش غەزەپ بىلەن تۈكۈرۈپ:

ـــ ھۇ بۆلگۈنچى ئۇنسۇر، پارتىيە ئىچىگە سوقۇنۇپ كىرىۋالغان ئەكسىيەتچى ئوغرى، قاراپ تۇر، بۇ جىنايىتىڭ ئۈچۈن تۈرمىدە ياتىسەن!!!ــ دەپ ۋاقىرىدى.

ـــ ئاتاڭغا مۇشت ئاتساڭ كۆرگۈلۈڭنى كۆرىسەن ھاراملىق!ــ دېدى ئىمامنىياز شۇجىمۇ يەرگە تۈكۈرۈپ.

شوپۇر بالا ئا شاڭجاڭنى ماشىنىغا چىقىرىپ يېزىغا ئېلىپ كەتتى. شاڭجاڭ يېزىلىق ھۆكۈمەت قورۇسىغا كىرىپلا يۈز-كۆزىدىكى قاننىمۇ ئېيتماستىن جاڭ شۇجىنىڭ ئىشخانىسىغا كىرىپ دادلاشقا باشلىدى:

ـــ ھازىرلا پەيسۇغا تېلېفون قىلسىلا شۇجى، پارتىيەلىك تۇرۇپ پارتىيەگە قارشى چىققان بۇ بۆلگۈنچىنى قاتتىق جازالاش كېرەك!

ـــ نېمە ئىش بولدى، ئالدىرىماي دەڭ!

جاڭ شۇجى ئۇنىڭ سۆزىنى ئاڭلاپ بولۇپ، قاپىقىنى تۈرۈپ ئويلىنىپ كەتتى.

ــــ مۇشۇنداق ئۆز بېشىمچى ئوغرىغا ئۇ كەنتنى تۇتقۇزۇپ قويساق ئىش چاتاق! ئۇ بەلكىم ھازىرنىڭ ئۆزىدە ئەزالىرىنى كۈشكۈرتۈپ ئىش تېرىماقچى بولۇشى مۇمكىن!

جاڭ شۇجىنىڭ جىم تۇرغىنىغا قاراپ ئا شاڭجاڭنىڭ ئەرۋاھى ئۇچتى، بۇ باشلىقلار خۇددى سىرلىق بۇتقا ئوخشايدىكەن دەپ ئويلىدى.

ـــ ماڭا راستىنى ئېيتىڭ، بولمىسا ئۆزەم بېرىپ تەكشۈرىمەن،ــ دېدى جاڭ شۇجى،ـــ ئۇ نېمىشقا سىزنى بۇنچە ئۇردى؟

ـــ مەن باشتا بىر تەستەك سالغان.

ـــ نېمىشقا؟

ـــ مېنى كىچىك كۆرۈپ كەمسىتتى، ھاقارەتلىدى.

ـــ شۇنداقتىمۇ ئۇنداق پېشقەدەم كەنت كادىرىنى ئۇرسىڭىز توغرا بولامدۇ؟ نېمە گەپ بولسا چىرايلىق دېيىشسەڭلار بولمامدۇ؟

ـــ ياخشى گەپ قىلدىم، سىياسەتنى چۈشەندۈردۈم، لېكىن ئۇنىمىدى. بولمىسا بۇنداق ئاقىۋەت كېلىپ چىقمايتتى. سىياسەتكە قارشى چىققان بۇنداق بۆلگۈنچى ئوغرىنى ۋەزىپىسىدىن ئېلىپ تاشلاپ قاتتىق جازالاش كېرەك.

ـــ ۋەزىپىسىدىن ئېلىپ تاشلىساق بولىدۇ، لېكىن باشقا جازا بەرگىلى بولمايدۇ.

ـــ نېمىشقا؟

ـــ چۈنكى بۇ ئادەمنىڭ پارتىيەلىك تارىخى ئۇزۇن، مەن كېلىشتىن بۇرۇنلا نەچچە قېتىم مۇنەۋۋەر پارتىيە ئەزاسى، ئىلغار كەنت كادىرى بولۇپ باھالانغانىكەن. مەندىن بۇرۇنقى شۇجىمۇ ئۇ ئادەمنى ماڭا ئالاھىدە تونۇشتۇرغان، بۇ يېزىنى تەرەققىي قىلدۇرماقچى بولساڭ شۇ ئادەمدىن مەسلىھەت ئال دېگەن. ئۇنىڭ كەنتى يېزا بويىچە ئۈلگىلىك كەنت. كەنتتە بايلار كۆپ، داشۆدە ئوقۇيدىغان بالىلار كۆپ، ساۋاتسىزلىق ئاساسەن يوق. كەنتتە چوڭ بىر قىرائەتخانا بار، يەسلى ۋە باشلانغۇچ مەكتەپنىڭ كۆلىمى، شارائىتىمۇ يېزا بويىچە ئەڭ ئالدىدا. كەنت ئەزالىرى ئۇ ئادەمنى بەك ھۆرمەتلەيدۇ، ھېمايە قىلىدۇ، نەچچە قېتىم يېزىغا رەسمىي كادىر قىلىپ يۆتكەپ ئەكىلەيلى، ھەتتا مۇئاۋىن شاڭجاڭ قىلايلى دېسەكمۇ شۇ كەنتتىكىلەر ئۇنىمىغان… بۇ ئادەمگە ئېغىر جازا بەرسەك، سىزنىڭ ئۇ ئادەمگە باشتا قول تەگكۈزگىنىڭىزنى بىلىپ قالسا، كەنتتىكىلەر قارشى چىقىدۇ، مۇقىملىققا تەسىر يېتىدۇ. ئىشنى باشتا بۇزغان ئادەم سىز بولۇپ قالىسىز. بۇنى ئوبدان ئويلاڭ…

جاڭ شۇجىنىڭ ئاقلاشلىرىدىن قورسىقى كۆپكەن ئا شاڭجاڭ ئۆيىگە كىرىپ كەتكىنىچە چىقمىدى، ئەتىسى دوختۇرخانىنىڭ قەغىزىنى تاشلاپ بېرىپ، نەچچە كۈن قېيداپ يېتىۋالدى. يەكشەنبە كۈنىدىكى يىغىنغىمۇ بارمىدى. شۇ باھانىدا شوپۇر بالا تويىنى قىلىۋالدى. تويغا ئىمامنىياز شۇجىنىمۇ تەكلىپ قىلدى. ئىمامنىياز شۇجى گەۋدىلىك بەستى، تەمكىنلىكى، مەردانە سۆزلىرى، خۇش چاقچاقلىرى بىلەن سورۇنغا باشقىچە ھۆسن قوشاتتى. شوپۇر بالا ھۆكۈمەت نېمىشقا مۇشۇنداق ئادەمنى شاڭجاڭ قىلمايدىغاندۇ دەپ ئويلاپ قالدى. ئىمامنىياز شۇجى كېتىدىغان چاغدا شوپۇر بالىنى چاقىرىپ ئۇنىڭغا 200 يۈەن تۇتقۇزۇپ تويىنى مۇبارەكلىدى، ئا شاڭجاڭنىڭ ئەھۋالىنى سورىدى.

ـــ ئىككى-ئۈچ كۈن بولۇپ قالدى، ئىشخانىسىغا كەلمىدى،- دېدى شوپۇر بالا.

ـــ جاڭ شۇجىچۇ؟

ـــ ناھىيەگە ئىككى كۈنلۈك يىغىنغا كەتكەن، ئەتە كېلەمدىكىن.

ـــ مېنى جازالايدىغاننىڭ پارىڭى يوقمۇ؟

ـــ ئۇنى ئۇقمىدىم.

ـــ بوپتۇ، قانداق قىلسا قىلار. خىزمەت دېگەننى شۇنچىلىك قىلساق بولار.

ـــ ئەسلىدە سىلە شاڭجاڭ بولسىلا بولاتتىكەن.

ـــ ئۇمۇ يىراق ئەمەس ئۇكا، تاغامنىڭ ئوغلى كەلپىن ناھىيەسىنىڭ ھاكىمى ئىدى، يېقىندا ۋىلايەتكە تەشكىلات بۆلۈمىنىڭ باشلىقى بولۇپ ئۆسكىدەك.

شوپۇر بالا ئىمامنىياز شۇجىنى ئۇزىتىپ قويۇپ، ئەتىسىلا بۇ خەۋەرنى يېزا كادىرلىرى ئارىسىغا تارقىتىۋەتتى. نەتىجىدە ئا شاڭجاڭنىڭ يۈزىدىكى يارىمۇ، يۈرەك يارىسىمۇ تېزدىن ساقىيىپ، گەپ-سۆزسىزلا خىزمەتكە كىرىشىپ كەتتى، جاڭ شۇجىمۇ بۇ مەسىلىنىڭ ئۆزلۈكىدىن ھەل بولغىنىغا خوش بولۇپ، 9-كەنتنىڭ ئامانلىق پۇلىنى يىغىدىغان ئىشنى توختاتتى، ئامانلىق ساقلىغۇچىلارنىڭ مائاشىنى يېزا مالىيەسىدىن ھەل قىلدى.

ـــ مەن خىزمەت تەكشۈرگىلى 9-كەنتكە بارماقچى،ــ دېدى جاڭ شۇجى بىر كۈنى ئا شاڭجاڭنىڭ ئىشخانىسىغا كىرىپ، ــ ئىمامنىياز شۇجى بىلەن يارىشىۋالغۇڭىز بولسا بىللە بېرىپ كېلەيلى.

ـــ بىللە بارايلى،ــ دېدى ئا شاڭجاڭ ئورنىدىن توپتەك قاڭقىپ.

يولدا كېتىۋېتىپ جاڭ شۇجى سورىدى:

ـــ سىز دەپ بېقىڭە ئا شاڭجاڭ، سىزچە مۇشۇ ئامانلىق قوغدىغۇچىلىرىغا يېزىدىن مائاش بەرسەك بولامدۇ؟

ـــ بولىدۇ شۇجى، ئالاھىدە ئەھۋالدا شۇنداق قىلساق بولار دەيمەن.

ـــ مەسىلەن، قانداق؟

ـــ مەسىلەن، 9-كەنتتىكىدەك ئەھۋال بولسا، زىددىيەتنى چوڭايتماسلىق ئۈچۈن يېزىدىن ئازراق پۇل چىقارساق ھېچ نېمە بولمايدۇ. بۇنىڭ ھەممىسى مۇقىملىق ئۈچۈن. يەنە كېلىپ ئۇنىڭغا كېتىدىغان پۇلمۇ چاغلىق.

ـــ توغرا دېدىڭىز، لېكىن بۇ يەردە مەن چۈشەنمەيدىغان بىر ئىش بار ئا شاڭجاڭ. يەرلىك كىشىلەر نېمىشقا بۇ ئامانلىق قوغدىغۇچىلارنى ياخشى كۆرمەيدۇ، نېمىشقا ئۇلارنى ئېشەك ساقچىلىرى دېگەن سەت نام بىلەن ئاتايدۇ؟

ـــ بۇ مەسىلىنى بەك ئويلاپ كەتمەيلى شۇجى، يەنە كېلىپ بىزدە ئېشەك دېگەن گەپ بەك سەت گەپمۇ ئەمەس، ئېشەك بىز ئۇيغۇرلارغا ئەڭ يېقىن ھايۋان، ئۇ بىزگە نۇرغۇن ئىشلارنى قىلىپ بېرىدۇ.

ـــ ھە… مۇنداق دەڭ، چۈشەندىم.
