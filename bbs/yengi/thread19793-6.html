--بۇ سوزنى مەنمۇ كورگەن،لېكىن قايتا ئويلاپ باقساق،كىشىنىڭ گۇمانىنى قوزغايدىغان بىر سۇز بار،كىتاپنىڭ راۋاقلىرىغا ھاسەل سۇركەش ئۇچۇن كىتاپ يىمەكلىكلەردەك پاكىز بولۇشى كېرەك،ھەسەل سۇركەلگەندىن كېيىن ئۇ ئاسان چاپلىشىپ قالىدۇ،كىلەركى قېتىم ئۇنى ئاجىرتىشقا توغرا كەلسە،بەتلەرنىڭ يىرتىلىپ كېتىشىدىن ساخلانغىلى بولمايدۇ،بىز ئادەتتە باقمىچىلىق بىلەن شۇغۇللانسىق،يۇغان-يۇغان{يەنى قوش مۇشتۇمچىلىق  }يوغانلىقتىكى تۇزنى يەم-خەشەكنىڭ ئىچىگە قويۇپ،قوي ،ئات ،كالا قاتارلىق ھايۋانلارنىڭ بەدىنىنىڭ تەركىۋىدىكى تۇز مىختارىنى تەڭەيمىز،بۇ يەردىكى سوزدىن يەھۇدىلارنى كەمسىتىدىغان مەنا كېلىپ چىقامدۇ قانداق؟يەنى كىتاپلارنىڭ بېتىگە ھەسەل سۇركىمىسە،كىتاپ ئوقۇمايدىغان،يەنە كېلىپ يەھۇدىلەرنىڭ بالىلىرىمۇ ئۇيغۇر بالىلىرىغا ئوخشايدۇ،كىتاپنىڭ سانىمۇ شۇنچىلىك كوپ،كورگەنلا كىتاپنى ئوقۇپ چۇشەنگىلى بولسا،ئۇلار تۇغۇلۇشىدىنلا ئالىم،فېرااېسسور،دوكتۇر،ئاكادىمىك بولۇپ قالمامدۇ، يەھۇدى بالىلىرى چۇشەنگىلى بولمىغان كىتاپنى ھەر بىر بەتنى كورگەندە بىر قۇتىدىن نارات ھەسىلى يىگۇزۇپ قويسىمۇ كورۇشى ناتايىنغۇ؟

    -ئەركىن مۇئەللىم،نىسپىلىك نەزىريەسى نيوتوننىڭ نەزىريەسىدىكى{مۇتلەق  ماددا،ماكان،زامان }خاتالىقنى كورسۇتۇپ بېرىش ئۇچۇن ئوتتۇرغا چىققان،يەنى نۇرنىڭ ئۈزى فوتون شەكلىدە تارقايدۇ،فوتون جىم تۇرۇپ باققان ئەمەس،ئۇ نۇر تېزلىگىدە ھەركەت قىلىپلا تۇرۇدۇ،كېيىن ئېينىشتىيىن {ئادەم نۇر تېزلىگىدە ھەركەت قىلسا قانداق بولار} دەپ ئويلاپ،ئاندىن ئادەمنىڭ نۇر تېزلىگىدە ھەركەت قىلغاندا ئۇزۇن ئۇمۇر كۇرۇدىغانلىغىنى،يەنى بەدىنىدىكى بىولوگىيەلىك سائەتنىڭ ئاستا ئۇتۇدىغانلىغىنى ئوتتۇرغا قويغان،بۇ ھازىر {ۋاقىتقا ساياھەت قىلىش} نەزىريەسىنى بارلىققا كەلتۇردى،لېكىن بۇ نەزىريەنى 1905  يىلى ئېينىشتيىن ئوتتۇرغا قويۇپ  بولۇپلا ئاغدۇرۇپ تاشلىغان،يەنى نىسپىلىك نەزىريەسىنىڭ 2 مەزمونى {ھەركەت جەريانىدا جىسىم ماسسىسىنىڭ ئېشىپ بارىدىغانلىغىنى ئوتتۇرغا قويغان}،يەنى 109يىل جەريانىدا،بۇ نەزىريە  پۇتۇن دۇنياغا كەڭ تارقالغان،بىر -بىرىنى ئىنكار قىلىدىغان بۇ سۇزلەرنىڭ پۇتۇن دۇنياغا ئېلىپ كېلىدىغان ئىقتىسادى زىيىنى قانچە دوللار بولغىيىتتى؟،سىتىفون خاۋكىڭ بۇنى تەرەقى قىلدۇرىمەن دەپ،يەر شارىنىڭ تېمپۇراتۇرسىنى 500مىليون دەپ پەرەز قىلغان،بۇنى ماتىماتىكا جەھەتتىن ھىساپلاش بەكلا ئاددى،نيوتون بولسا ماتىماتىكا جەھەتتىن 4004+1685= 0دەپ تۇرۇپ، تەڭرىنى چاقىرىپ،ئالدى بىلەن ئالەمنى ھەركەتلەندۇرۇپ،ئاندىن ئۇچ قانۇننى ئوتتۇرغا قويغان،تاشپۇلات روزى  مۇئەللىم بۇنىڭ قانداق ئىش ئىكلەنلىگىنى ناھايىتى ئاسان ھىساپلاپ چىقالايدۇ،ئەلۋەتتە،بۇ توردىكى تورداشلارلا ئەمەس،ھەر بىر ئادەم دىگۇدەك بۇنى ھىساپلاپ چىقالايدۇ،بۇ سوز بىلەن {نۇرنىڭ تۇرغۇن ماسسىسى نول بولىدۇ}دىيىش بىر مەناغا بېرىپ قالىدىكەن،يەنى نيوتون،ئېينىشتىيىن  ۋە سىزنىڭ سوزىڭئىزدە دەل مۇشۇ نوختىدا ئورتاقلىق بار،بۇنى قايتا ئويلاپ بېقىشقا بولارمۇ؟ داۋامى بار

تېخىمۇ مويىم، بىر ئىش،بىز خېلى يىللاردىن بۇيان نيوتون قانۇنلىرىدىكى بايا دىگەن ئالدىنقى شەرت 4004+1685=0نى ئاساس قىلىپ {بۇ ساننىڭ ماتىماتىكىلىق ئىپادىسىنى يېقىندا بايقىدۇق}،نيوتون قانۇنلىرىنى تۇزگەنلىگىنى بىلىپ، نيوتون قانۇنلىرىنى قايتا تۇزۇشنىڭ زورورلىگىنى بايقاپ، باش قاتتۇرۇپ يۇرگەندە،ئۇچۇرۇغان  قىيىن مەسىلە،{تاشقى پلانىت ئادەملىرى} مەسىلىنىڭ قىيىن نوختىسى بولغان،تەخنىنەن 15يىل بولغان بولسا كېرەك،سىز بىر قېتىم ھازىرقى خىزمەت ئورنىڭئىزدىن يەنى ئامىرىكىدىن شىنجاڭغا بىر قېتىم كېلىپتىكەنسىز،دوس بۇرادەرلىرىڭئىز سىزنى بىر ۋاخلىق تاماققا چاقىرىپ چاقىرغاچ ئۇزلىرى تەتقىقاتتا يۇلۇقتۇرغان بەزى قىيىين مەسىلىلەرنى سىزدىن سوراپتىكەن،لېكىن ئۇرۇن مەسىلىسى سەۋەبىدىن 20لا ئادەم بۇ سورۇنغا قاتنىشالىپتىكەن،دەل شۇ ۋاقىتتا بىرەيلەن {تاشقى پىلانت ئادەملىرى} ھەققىدە بىر سۇئال سوراپتىكەن،بىرقانچە يىلدىن كېيىن بىز سىزنىڭ بۇ مەسىلىدىكى جاۋابىڭئىزنى  سورۇنغا قاتنىشالمىغان بىرەيلەننىڭ يەتكۇزگەن سوزىدىن ئاڭلاپ ئاندىن نيوتون قانۇنلىرىنى قايتا تۇزۇپ چىققان،ھازىر قارىسام سىز ئىمكانىيەتنىڭ بېرىچە تورداشلارغا يېڭى مەزمونلارنى يەتكۇزۇپ تۇرۇۋاتىسىز،شۇ بانىدا پىكىرلىشىش ئىمكانىيىتى تۇغۇلدى،يەنى سىزنىڭ ئەينى ۋاقىتتا دىگىنىڭئىز  {تاشقى پلانىت ئادەملىرى مەۋجۇت ئەمەسكەن}دىگەن سوزىڭئىز ،بىزنى مويىم يىپ ئۇچى بىلەن تەمىنلىگەن،بۇگۇنكى كۇنلۇكتە،پۇل مۇئامىلە كەسپىنىڭ نوختۇسۇدا تۇرۇپ يېزىلغان بۇ ماقالە بىرقانچە خەلقئارالىق ئىلمى ژورنالنىڭ دىققىتىنى قوزغاپ قالدى،{ماقالە قاچان پۇتۇدۇ}ھەققىدىكى خەتلەر ھەر كۇنى كېلىپ تۇرۇدۇ،دىمەكچى بولغىنىم،ھازىرقى بازار ئىگىلىگىدە،ئىلمىلىگى بولمىغان ماقالىنى خەلقئارالىق ژورناللارمۇ قۇبۇل قىلمايدۇ،لېكىن خەلقئارالىق ژورناللارنىڭ تۇرى بەكلا كوپ ئىكەن،بۇلارنىڭ پەرقىنى بىلگىلى بولمايدىكەن،سىز تۇنۇشۇدىغان،بىلىدىغان،مۇشۇ ساھادىكى ئالىملىرىمىز بارمىدۇ،بۇلار تۇنۇشتۇرۇلغان ماقالىلەر بولسا  تېخىمۇ ياخشى بولغان بولاتتى،ئەلۋەتتە،بىزمۇ سۇپىتى توۋەن ماقالىلارنى يېزىپ بۇ كىشىلىرىمىزنى ئاۋارە قىلمايمىز،
