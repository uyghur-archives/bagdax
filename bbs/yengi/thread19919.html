  ۋاڭ خۇڭۋېن جۇڭنەنخەيگە قانداق كېرىپ قالغان ؟

        لىن بياۋنىڭ ئۆزلىكىدىن پارتىلاپ كىتىشى ، چېن بودانى تەخىتتىن چۈشۈشى بىلەن جاڭ چۈنچاۋنىڭ ئالدىدىكى ئىككى كىشىنىڭ ئىسمى ئۈچىرىۋېتىلىپ ، جاڭ چۈنچاۋ  ماۋزېدۇڭ ، جۇئېنلەي ، كاڭشېڭ ، جياڭ چىڭدىن كېيىن تۇرىدىغان 5 - نومۇرلۇق شەخس بولۇپ قالدى.
       ئەينى ۋاقىتتا لىن بياۋ ج ك پ مەركىزى كومتېتىنىڭ بىرىنچى مۇئاۋىن رەئىسى ئىدى . ئۇ موللاق ئاتقاندىن كېيىن مۇئاۋېن  رەئىسلىك ئورنى بوش قالدى .  جاڭ چۈنچاۋنىڭ كۆزىمۇ ئاشۇ بوش ئورۇنغا مىقتەك قادالدى .  بولۇپمۇ ماۋزېدۇڭ ج ك پ 10 - قۇرۇلتىيىنى ئېچىشقا تەييارلىق كۆرۈش بۇيرۇقىنى چۈشۈرگەندىن كېيىن ،  جاڭ چۈنچاۋ پۇرسەتنىڭ يېتىپ كەلگەنلىكىنى پەملەپ ، قانداق قىلىپ پارتىيە مەركىزى كومتېتىنىڭ مۇئاۋېن رەئىسلىك ھوقۇقىنى قولغا چۈشۈرىۋېلىش ھەققىدە باش قاتۇرۇشقا باشلىدى .
       بىر كۈنى جۇڭنەنخەيدە خىزمەت توغرىسىدىكى سۆزلەر ئاخىرلاشقاندىن كېيىن ، ماۋ زېدۇڭ جاڭ چۈنچاۋدىن تۇيۇقسىزلا : " ۋاڭ خۇڭۋېن ماقالە يازالامدۇ " ؟ دەپ سۇراپ قالدى .
       ماۋ زېدۇڭنىڭ بۇ سۆزى جاڭ چۈنچاۋنى ساراسىمگە سېلىپ قويدى .
       ئۇ ماۋ زېدۇڭنىڭ سۆزىگە : " ۋاڭ خۇڭۋېن ماقالىنى ئانچە - مۇنچە يېزىپ قويىدۇ ، ئەمما ياۋ ۋېنيۇەنگە يېققىنمۇ كېلەلمەيدۇ" دەپ جاۋاب بېرىدۇ .
      جاڭ چۈنچاۋ دياۋيۈيتەيگە قايتقاندىن كېيىن ماۋ زېدۇڭنىڭ نېمە ئۈچۈن ۋاڭ خۇڭۋېننى سۇراپ قالغانلىقى ھەققىدە ئويلىنىشقا باشلىدى ھەمدە ماۋزېدۇڭ ج ك پ 10 - قۇرۇلتىيىدا ۋەزىپىگە قويۇلىدىغان كادىرلار ئۈستىدە ئويلىشىۋاتقان بولۇشى مۇمكىن ، ئۇ توساتتىنلا ۋاڭ خۇڭۋېننى سۇراپ قالدى ، شۇنىڭغا قارىغاندا ۋاڭ خۇڭۋېننى ياقتۇرۇپ قالدىمۇ - نېمە ؟ دەپ پەرەز قىلدى .
       ئۇ ماۋ زېدۇڭدىن بۇ ئىشنى سۇراپ بېقىشقا ئەلۋەتتە پېتىنالمايىتتى . شۇڭا جياڭ چىڭدىن ياردەم سۇراشقا مەجبۇر بولدى .
      يەنە كېلىپ جياڭ چىڭ بىلەن جاڭ چۈنچاۋ ئوتتۇرىسىدا ئۇزۇن يىللىق " جەڭگىۋار دوستلۇق" مۇناسىۋىتى بولغاچقا ، ئىككىسىنىڭ دېيىشمەيدىغان سۆزى يوق ئىدى .
      دېگەندەكلا جياڭ چىڭ ئۇنىڭغا : نەچچە كۈن بۇرۇن كاڭ شېڭ ماۋجۇشى بىلەن سۆزلەشكەندە ماۋجۇشىغا : ئۈزەم ئىلگىرى ئىزچىل تۈردە ئىشچىلار ھەرىكىتى بىلەن شوغۇللانغاچقا ، " مەدەنىيەت زور ئىنقىلابىدا ھەر قايسى جايلاردا بارلىققا كەلگەن " ئىشچىلار داھىلىرى"غا  ناھايىتى ئېتىبار بىلەن قارايمەن ، دېگەنلىكىنى ، شۇنداقلا يەنە : شاڭخەيدىكى ئىشچىلار ھەرىكىتى خېلى يامان ئەمەس ئېلىپ بېرىلدى ، ۋاڭ خۇڭۋېن دىققەت - ئېتىبارىمىزنى قوزغاچقا ئەرزىيدىغان ئادەم . . .  دېگەنلىكىنى ئېيتىپ بەردى .
        بۇنى ئاڭلىغان جاڭ چۈنچاۋنىڭ چىرايى بىردىنلا تاتىرىپ كەتتى .  ئۇ كاڭ شېڭنىڭ پەردە ئارقىسىدا تۇرۇپ ھېيلە - نەيرەڭ ئىشلىتىپ ئۆزىنى ۋە جياڭ چىڭنى ئەخمەق قىلىۋاتقانلىقى ، ئۆز ھوقۇقىنىڭ تېخىمۇ چوڭلىشىنى ئامالنىڭ بارىچە چەكلەۋاتقانلىقىنى چۈشۈنۇپ يەتتى .
        دېگەندەكلا بىر نەچچە كۈندىن كېيىن ، ماۋ زېدۇڭ جاڭ چۈنچاۋدىن ۋاڭ خۇڭۋېننىڭ ئەھۋالىنى تەپسىلى سۇرىدى ، ئەمما بۇنىڭدىكى مۇدداسىنى ئېيتمىدى .
        بۇ ئىش بولۇپ ئۇزۇن ئۆتمەيلا ، ماۋ زېدۇڭ ۋاڭ خۇڭۋېننى بېيجىڭغا يۆتكەپ كېلىشنى ئېنىق ئوتتۇرىغا قويدى .
       1972 - يىلى 9 - ئاينىڭ 7 - كۈنى ۋاڭ خۇڭۋېن بۇيرۇققا بىنائەن بېيجىڭغا كەلگەن بولسىمۇ ، بېيجىڭغا كېلىشنىڭ نېمىدىن دېرەك بېرىدىغانلىقىنى ھەتتا ئۆزىمۇ بېلەلمىدى .
       ماۋ زېدۇڭنىڭ ئۇنى قوبۇل قىلىشى ۋاڭ خۇڭۋېن ئۈچۈن كۈتۈلمىگەن ئەھۋال بولۇپ تۇيۇلدى : ماۋزېدۇڭ قىرائەتخانىسىدا 37 ياشلىق بۇ ياشنىڭ قولىنى چىڭ سىقىپ تۇراتتى .
       79 - ياشلىق ماۋزېدۇڭ  "  يېقىن  سەپدىشى " لىن بياۋ ئۆزىگە ئاسىيلىق قىلغاندىن كېيىن قاتتىق روھىي زەربىگە ئۇچىرىدى ، ئۇنىڭ چىرايىدىكى قېرىلىق ئالامەتلىرى روشەن كۆپەيدى .
       جاڭ چۈنچاۋ قوبۇل قىلىشتا بولىۋاتقان ئەھۋالنى ئېنىق كۆرۈپ ، دېيىلىۋاتقان  گەپ - سۆزلەرنى تۇلۇق ئاڭلاپ يان تەرەپتە ئولتۇراتتى .
      ماۋزېدۇڭ ناھايىتى قىزغىنلىق بىلەن ۋاڭ خۇڭۋېندىن ئۇنى - بۇنى سۇرىدى ، ئۇنىڭدىن قانچە يىل ئوقۇغانلىقى ، قانچە يىل ھەربىيدە تۇرغانلىقى قاتارلىقلارنى سۇرىغاندىن كېيىن
، ۋاڭ خۇڭۋېننى : " پۈتۈن ۋۇجۇدىڭىزدىن  ئىشچى ، دېھقان ، ئەسكەر پۇرىقى چىقىپ تۇرىدىكەن - ئەسكەرمۇ ، دېھقانمۇ ، ئىشچىمۇ بولغان ئىكەنسىز " دەپ ماختىدى .
      ماۋزېدۇڭ خۇددى ئىمتىھان ئالغۇچىدەك ئۇنىڭدىن مەسىللەرنى ئارقا - ئارقىدىن سۇرىدى ، ۋاڭ خۇڭۋېنغا ماركىسزىم - لېنىنزىم كىتابلىرىنى كۆپرەك كۆرۈش ، تۈرلۈك يىغىندىكى پىكىرلەرنى كۆپرەك ئاڭلاشنى تاپىلىدى .
        ماۋ زېدۇڭ ۋاڭ خۇڭۋېننى بېيجىڭغا يۆتكەپ كېلىشتىكى مۇددىئاسىنى ئېيىتمىغانلىقتىن يەنە كېلىپ جاڭ چۈنچاۋنىڭ بۇنى دەپ قويۇشىغا بولمىغانلىقتىن ، ۋاڭ خۇڭۋېن بېيجىڭدا تۇرۇپ قالغانلىقىدىن زىرىكىپ تىت - تىت بولدى . ئۇ ماركىس ، ئېنگىلىس ، لېنىنلارنىڭ قېلىن توملىرىنى بىر قانچە رەت ۋاراقلاپ قويۇپلا ناھايىتى زېرىكىشلىك ھېس قىلدى . يېنىدا پو ئېتىشىپ بېرىدىغان بۇرادەرلىرى بولمىغانلىقتىن ئىچى ناھايىتى پۇشتى .
        ئۇ جوزىدىكى تېلىفۇننى بىۋاستە شاڭخەيگە ئۇلاتقۇزۇپ ئىچ پۇشۇغىنى چىقىرىشقا باشلىدى .
        ماتيەنشۈي خاتىرىسىدە شۇ مەزگىلدىكى ۋاڭخۇنۋېننىڭ تېلىفۇندا سۆزلىگەن گەپلىرى مۇنداق خاتىرىلەنگەن :  " بېيجىڭغا كەلگەندىن كېيىن ئىچىم بەك پۇشۇپ كەتتى . بەزىدە بىر كۈن كەچكىچە يىغىنغا  قاتنىشىپ بەك چارچاپ كېتىمەن . بىر قانچە كۈندىن بۇيان ئودا يەتتىنچى ماشىنسازلىق سانائىتى مىنىستىرلىكىنىڭ يىغىنىغا يەنە خېنەن ، خۇنەن ئۆلكىلىرىنىڭ يىغىنىغا قاتناشتىم ، چۈشتىن كېيىن سائەت 3 تە چىقىپ كەتسەك  يېرىم كېچە سائەت 3 تە قايتىپ كېلىمىز . كىتاب كۆۈرگۈممۇ كەلمەيدىغان بولۇپ قالدى .  يوقىرىغا يۆتكىلىپ كەلگىنىم بىلەن زادىلا ئادەتلىنالمايۋاتىمەن .  شاڭخەيگە بالدۇرراق قايتىپ كېتىشكە ئىنتىزارمەن . . . "
       دەل مۇشۇ پەيىتتە ، جاڭ چۈنچاۋ ماۋزېدۇڭغا ۋاڭ خۇڭۋېننىڭ شاڭخەيگە قايتىپ كېتىش نىيىتى بارلىقىنى ئىنكاس قىلدى .   
       - ئۇ شاڭخەيگە قايتىپ كەتسە قانداق بولىدۇ ، - دېدى ماۋزېدۇڭ كۈلۈپ كېتىپ :
      - مەن تېخى ئۇنى مۇئاۋېن رەئىس قىلىش توغرىسىدا تەكلىپ بېرەي دەۋاتىمەن . ئەمما ، بۇ مېنىڭ شەخسى كۆز قارىشىم ، سىياسىي بيۇرونىڭ مۇزاكىرسىدىن تېخى ئۆتكۈزمىدۇق ، باشقىلارغا دەپ يۈرمەڭ ، ۋاڭ خۇڭۋېننىڭ ئۆزىگىمۇ ئېيىتماڭ ، - دېدى .
       جاڭ چۈنچاۋنىڭ ئەگىتىپ ئېيىتقان سۆزى نەتىجىگە ئېرىشتى . ئۇ ئاخىرى ماۋزېدۇڭنىڭ ئاغزىدىن ئۇنىڭ مۇددىئاسىنى ئەپچىللىك بىلەن بىلىۋالدى .
       جاڭ چۈنچاۋنىڭ قەلبى زىددىيەتكە تولدى :  ئۇ ۋاڭ  خۇڭۋېندىن ئىبارەت بۇ دۆلەت ئىگىلىكىدىكى شاڭخەي پاختا توقۇمچىلىق 17 - فابركىسى قوغداش بۆلۈمىنىڭ كادىرىنى " ئەنتىڭ ۋەقەسى" دە ئۆز قولى بىلەن يۆلەپ چىقارغان ھەمدە شاڭخەي شەھەرلىك " ئىنقىلابىي كومتېت"نىڭ مۇئاۋىن مۇدىرى ، ج ك پ 9 - نۆۋەتلىك مەركىزى كومتېتنىڭ ئەزاسى قىلغان . مانا ئەندى ۋاڭ خۇڭۋېن " راكىتا"غا ئولتۇرۇپ ئۆزىنىڭ قول ئاستىدىكى ئادىمىدىن بىراقلا باشلىقى بولىۋېلىپ ، ئەسلىدە جەزمەن ئۇنىڭغا تەئەللۇق بولغان مۇئاۋىن رەئىسلىك ئورۇننى تارتىۋالدى . . .   بۇ ئىش ئۇنى قاتتىق بىئارام قىلدى . ئەلۋەتتە ، ئۇ ۋاڭ خۇڭۋېننىڭ نېمىلا قىلغان بىلەن يەنىلا ئۆزىنىڭ ئادىمى ئىكەنلىكى ، ۋاڭ خۇڭۋېن مەركىزى كومتېتقا كەلسە ئۆزىنىڭ كۈچى زورىيىدىغانلىقىنى ئويلىغاندا كوڭلى خېلى ئارام تېپىپمۇ  قالدى . چۈنكى ، قانداقلا موللاق ئاتقىنى بىلەن يەنىلا ئۇنىڭ ئالقىنىدىن چىقىپ كېتەلمەيىتتى . . .
       بىر مەزگىل " پىراكتىكا قىلىش" تىن كېيىن ۋاڭ خۇڭۋېن بېيجىڭدا دەسلەپكى قەدەمدە كۆزگە كۆرۈنۈشكە باشلىدى .
       1973 - 5 - ئايدا ئېچىلغان مەركىزى كومتېت يىغىنىدا ماۋزېدۇڭنىڭ تەكلىپىگە بىنائەن ۋاڭ خۇڭۋېن ، خۇا گوفېڭ ، ۋۇدې قاتارلىق 3 كىشىنىڭ مەركىزى كومتېت سىياسىي بيۇروسىنىڭ يىغىنىغا سىرتتىن قاتنىشىپ ، مەركىزى كومتېتنىڭ خىزمەتلىرىگە قاتنىشىشنى قارار قىلدى .
       ئۇنىڭدىن باشقا يەنە ۋاڭ خۇڭۋېننى پارتىيە نىزامنامىسىغا تۈزىتىش كىرگۈزۈش گۇرۇپپىسىغا مەسئۇل قىلىش ،  جاڭ چۈنچاۋ بىلەن ياۋۋېنيۇەننى پارتىيە مەركىزى كومتېتىنىڭ 10 - قۇرۇلتايدا بېرىلىدىغان سىياسىي دوكىلاتىنى يېزىشقا مەسئۇل قىلىش قارار قىلىندى .
       ج ك پ 10 - نۆۋەتلىك قۇرۇلتىيىدىكى سايلامغا تەييارلىق قىلىش ھەيئىتى 1973 - يىلى 8 - ئاينىڭ 20 - كۈنى بېيجىڭدا قۇرۇلدى .  ئۇ 104 نەپەر ھەيئەت ئەزاسىدىن تەركىپ تەپقان بولۇپ ، ۋاڭ خۇڭۋېن داغدۇغا بىلەن مۇدىرلىققا تەيىنلەندى . ھالبۇكى ، جۇئېنلەي ، كاڭ شېڭ ،يىې جىيەنيىڭ ، جياڭ چىڭ ، جاڭ چۈنچاۋ ، لى دېشىڭلار مۇئاۋىن مۇدىر بولدى . بۇ ئىسىملىك ماۋزېدۇڭنىڭ تەكلىپى بىلەن بېكىتەلگەنىدى .
            ج ك پ 10 - نۆۋەتلىك قۇرۇلتىيى  1973 - يىلى 8 - ئاينىڭ 24- كۈنىدىن 28 - كۈنىگىچە بېيجىڭدا ئېچىلدى . يىغىندا جۇئېنلەينىڭ سىياسىي دوكىلاد بېرىشى كىشىلەرنىڭ پەرىزىدىكى ئىش ئىدى .
        چېچىنى ئىككى يانغا ئايرىپ ياسىتىۋالغان ۋاڭ خۇڭۋېن رەئىش مۇنبىرىگە چىقىپ ، پارتىيە نىزامنامىسىگە تۈزىتىش كىرگۈزۈش دوكلاتىنى بېرىشكە باشلىغاندا دۆلەت ئىچى ۋە سىرتىدىكىلەر بۇ ئىشتىن ھاڭ - تاڭ قېلىشتى .
       چەتئەل مۇخبىرلىرى سەزگۈرلۈك بىلەن :
      - ماۋزېدۇڭ شاڭخەي ئىشچىلىرىنىڭ ياش سەردارى ۋاڭ خۇڭۋېننى ئىز باسارلىققا تاللاپتۇ ، دەپ ئوبزور ئېلان قىلدى .  ئۇلار ئۇ دەسلىپىدە ليۇ شاۋچىنى ئىز باسارلىققا تاللىغانىدى ، ليۇ شاۋچى " مەدەنىيەت زور ئىنقىلابىي" دا ئاغدۇرىۋېتىلدى . ئۇ يەنە لىن بياۋنى ئىز باسارلىققا تاللىغان ، بىراق لىن بياۋ ئۆزلىكىدىن ھالاك بولدى ، بۇ قېتىم ئۇ ۋاڭ خۇڭۋېننى ئىز باسار قىلىپ بېكىتتى . بۇ ياش ئىز باسارنىڭ كېيىن قانداق بولىدىغانلىقىنى كىم بىلسۇن ؟ " دېيىشتى .
       8 - ئاينىڭ 30 كۈنى ج ك پ 10 - نۆۋەتلىك مەركىزى كومتېتىنىڭ 1 - ئۇمۇمىي يىغىنى سايلام نەتىجىسىنى ئېلان قىلدى . ماۋ زېدۇڭ جۇڭگو كوممۇنىستىك پارتىيەسى مەركىزى كومتېتىنىڭ رەئىسى ، جۇئېنلەي ، ۋاڭ خۇڭۋېن ، كاڭ شېڭ ، يىې جيەنيىڭ ، لى دېشىڭلار مەركىزى كومتېتنىڭ مۇئاۋىن رەئىسلىرى بولدى .
       مۇشۇنداق قىلىپ بىر زاۋۇتتىكى قوغداش بۆلۈمىنىڭ باشلىقى بىمەنە يىللاردىكى راكىتاغا ئولتۇرۇپ ، بىراقلا پارتىيىنىڭ ئەڭ يوقىرى رەھبەرلىك قاتلىمىغا كىرىپ قالدى .

            " ئەدەبىي تەرجىمىلەر" دىن ئېلىندى . " باغداش تورى" غا غاس .
