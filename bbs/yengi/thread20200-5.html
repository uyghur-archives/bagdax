موڭغۇللار بىلەن ئۇيغۇرلارنىڭ مۇناسىۋىتى بەلكىم ئىمپىراتور چىڭگىزخاندىن باش؛اپ قويۇقلاشقان بولسا كىرەك،مەزكۇر ئىمپىراتور موڭغۇل چەۋەندازلىرىنى باشلاپ ئۇچىرغاننى سۇر توقاي قىلىپ ھازىرقى شىنجاڭ رايۇنىغا كەلگىنىدە ئىدىقۇت ئۇيغۇر خانى ئۇرۇشسىز چىكىنىپ زىمىننى قوش قوللاپ موڭغۇللارغا سۇنىدىكەن(بۇنىڭغا چىچىلماي تۇرغايسىلەر بىر كىتاپتا شۇنداق ئوقۇغانتىم،ئىسمىنى ئۇنتۇپ قاپتىمەن،لىكىن ئۇزۇنغا قالماي پاكىتنى كۆرسىتىمەن ئىنشائاللاھ)شۇنىڭدىن كىيىن شىنجاڭ زىمىنىغا بىر قانچە ئەۋلات موڭغۇللار ئىگىدارچىلىق قىلغان،موڭغۇللار ئەلۋەتتە بۇ زىمىنغا بىر پادىشاھ بىر قانچە خىزمەتكار بىلەنلا كەلمەيدۇدە،نەچچە تۇمەن سەركەردە سەرۋزلىرى بىلەن كەلگەن چاغاتايخان ۋە ئۇنىڭ ئەۋلاتلىرى بىر قانچە ئەسىر ياشاش جەريانىدا ئۇيغۇرلار بىلەن چاڭلىشىپ كەتكەن بولىشى مۇمكىن،موغۇلىستان خانلىقىغا كەلسەك ئىسىمدە قىلىشچە سەمەرقەنت شەھرىدە قۇرۇلغانغۇ دەيمەن،باھادىرنامە رومانىدا بولسا سۇلتان شىرمۇھەممەتخان زامانىسىدىن باشلاپ تەسۋىرلەنگەچكە ئەينى ۋاقىتتىكى پايتەخىت بولمىش تۇرپان شەھرىدىن باشلاپ تەسۋىرلىنىدۇ،كىتاپتا تىلغا ئىلىنغان يەر جاي ناملىرى ھازىرقى يەر جاي ناملىرى بىلەن ئانچە پەرىقلەنمەيدىغۇ دەيمەن،مەسلەن تۇرپان،قۇمۇل،قەشقەر،يەكەن،ئاقسۇ ،خوتەن دىگەن شەھەرلەرنىڭ ھازىرقى نامىمۇ شۇ پىتىم،لىكىن كىتاپتا تىلغا ئىلىنغان تۇركىستان دۆلىتىنىڭ قايسى رايۇنلارغا توغرا كەلگىنىنى مەنمۇ ئىنىق بىلەلمىدىم تۇرپان تەرەپكە ئانچە يىراق بولمىغان بىر زىمىنلار بولسا كىرەك،غولجىغا دەپنە قىلىنغان سۇلتان ئۈۋەيىسخان دەل شۇ تۇركىستان دۆلىتىدە پاناھلىنىدۇ ۋە تۇركىستاننىڭ يالغۇز مەلىكىسىگە ئۆيلىنىدۇ.
