ئاتالمىش تەرەققىيات نۇرغۇنلىغان ئاجىز توپلارنىڭ كۈنىمىزدىكى مەنپەئەتىنى ۋەيران قىلىش، كەلگۈسى ئەۋلادلارنىڭ نىسىۋىسىنى تالان-تاراج قىلىش، مۇھىتنى ئېغىر دەرىجىدە بۇلغاش بەدىلىگە يۈرۈشمەكتە. چىرىكلىك مەسىلىسى ئىنتايىن ئېغىرلاشماقتا. چىرىكلەشكەنلەرنى جازالىدۇق دېگىنى بىلەن مىڭنىڭ بىرەرى پاش بولۇپ قالغاندىلا ئۇلارنى جازالاپ، ئاممىنىڭ ئېغىزىنى ئېتىش تاكتىكىسىدىن باشقا نەرسە ئەمەس. باي-نامراتلىق پەرقى ئىنتايىن چوڭايماقتا. %1 كە يەتمەيدىغان ئادەم %70 بايلىقنى كونتىرۇل قىلىۋالغان. ئىشقا ئورۇنلاشتۇرۇش بېسىمىمۇ تاغدەك ئېغىرلاشماقتا.ئەتراپتىكى دۆلەتلەر بىلەن بولۇۋاتقان زېمىن تالاش-تارتىشلىرىنى ياخشى ھەل قىلالماسلىقمۇ بىر خەۋىپلىك مەسىلىلەرنىڭ بىرى. دېمەك دەرەخ جىم تۇرىمەن دېسىمۇ شامال ئۇنى جىم تۇرغىلى قويمايدۇ.

جۇڭگۇ نەگە قۇنغۇسى كەلسە قۇنالىغۇدەك قۇنداق،نىمىنى ئالغۇسى كەلسە يۇلۇپ ئالالىغۇدەك ئۆتكۈر تىرناققا ئىرىشىمەن دىسە ئاۋۋال ئۆزىنى قايتىدىن قاراپ بىر ئىش قىلغىنى تۈزۈك،گەرچە دۇنيادىكى ئىككىنجى ئىخدىسادى دۆلەتكە ئايلاندۇق،دۇنيانىڭ زاۋۇتىغا ئايلاندۇق دەپ يۈرمەي ئاۋۋال مۇھىت،يىمەك ئىچمەك،ئىچىملىك سۇ ،تۇرالغۇ ئۆي،ئىشقا ئۇرۇنلىشىش قاتارلىق جۇڭخۇا خەلىق جۇمھۇرىيتىنىڭ ئاساسى قانۇنىدا بەلگىلەنگەن پۇخرا ھۇقۇقنى ۋە مەمۇرى ھۇقۇقىنى ھەقىقى ئىشقا ئاشۇرالىسا ئاندىن خەلىق بىر يەڭدىن قۇل بىر ياقىدىن باش چىقىرىپ دۈشمەننىڭ ھەيۋىسىنى ئۇرالىغۇدەك بولىدۇ.
دۇختۇرخانىلاردا سەۋۋەبىنى بىلىپ بولمايدىغان كەلسەل كۆپ.ئاياللار بۆلىمىدە ئەرلەر كۆرىۋاتقان.كوچىلاردا ئەخلەت تاشلاپ مىڭىۋاتقان مانچە يىللىق مەدىنىيەنىڭ ئەۋلاتلىرى تىخىچە بۇرنىنى كولاپ يۇقىرى ئاۋازدا تۇۋلىشىپ يۈرسە،پۇل ۋە ئەمەلگە ئىرىشىش ئۈچۈن ئارقا ئىشىك دىگەننى قىزىۋەرسە ئاقىۋەت نىمە بولىدۇ.دۆلىتىمىز بىلەن ياپۇن ئۇتتۇرسىدىكى ئارال ماجىراسىدا ئۇرۇش قاش كىرپىك ئارسىدا قالغىنىدا بىر يۇقىرى دەرىجىلىك ھەربىي گىنىرال نەچچە ئون ئاشنىسى بىلەن ئۇيناپ يۈرگەنلىكى ئىلان قىلىندى.يۇقىرى رەھبەرنىڭ بۇنداق قىلمىشى بار يەردە تۆۋەندىكىلەر نىمە قىلماقچىدى،يىلاننىڭ بىشى قاياققا ماڭسا قۇيرۇقىمۇ شۇنىڭغا ماىڭىدۇ.چىرىكلىك تۈگىمەي تۇرۇپ قۇيرۇق كۆرسىتىپ ئۆپكە سىتىشنى تۇختاتقىلى بولمايدۇ.بۇ سىلەرچە خەلىقئارالىق ئىناۋەتمىدۇ؟دىياۋيۈ ئارال ماجىراسىدا يۇقىرى دەرىجىلىق بىر رەھبەر ئۇرۇش ئىلان قىلىشنى سۆزلىۋىدى ئەتىسىلا ئامرىكا نىۋ يورىك ۋاقىت گىزىتى ئۇبىزۇرىدا ئەگەر ئۇرۇش بولۇپلا قالسا جۇڭگۇلۇق بىر مىلىيون سەككىز يۈزمىڭ جۇڭگۇلۇق ئەمەلدارنىڭ چەتئەلدىكى بايلىقىنى ئاشكارلىۋىتىمىز دەپ تەھدىت سىلىشىغا ئەتىسى ئۇرۇشنىڭ گىپىنى قىلغان ئەمەلدار سۇغا چۈشكەن مۆشۈكتەك يۇقاپ كەتمىدىمۇ ھىلىقى ئۇبزۇرچى ئىككى ئىغىز سۆزلەپلا بولدى قىلمىدىمۇ.شۇڭا چىرىكلىك تۈگىمەي تۇرۇپ خەلىقنىڭ كۆڭلىنى مايىل قىلىش تەس بىر ئىش.

ئۇرۇش قىلىش ئۈچۈن ئەڭ مۇھىم ئىككى تەلەپنى ئۇرۇنلىمىسا بولمايدۇ.بىرى،يىتەرلىك ئۇرۇش مالىيە چىقىمى ،يەنە بىر مۇھىم تەرىپى بولسا دۆلەت ئىچىدىكى خەلىقنىڭ ئۇرۇشنى قۇللاش ساداسى.بۇ نۇقتا بەك مۇھىم نۇقتا.جۇڭگۇدا پۇل تىپىپ بىيىغان جۇڭگۇلۇقلار پۇللىرىنى بىر ئاماللار بىلەن چەتئەلگە يۆتكەپ يۈرسە قانداق بولغىنى.ياكى مۇشۇنداق چەتئەلگە يۈزلىنىۋاتقان خەقلەر راسىتلا ساراڭمىدۇ.
