                             سەپەر ھىسسىياتلىرى - قەيسەر مىجىت




        ئىلاۋە: بىر نەچچە يىل بولدى، ئامىرىكىلىقلارنىڭ "مىننەتدارلىق بايرىمى" كىلىشى بىلەن مەن ئائىلەمنى ئىلىپ ئۇ يەر، بۇ يەرگە سەپەر قىلىدىغان بولغىلى. ئەلۋەتتە مىنىڭ سەپىرىم نىشانسىز ئەمەس. مەن باشقا ئىشتاتلارغا يەر كۆرگىلى ئەمەس، بەلكى دوستلىرىم بىلەن كۆرۈشكىلى، ئۇلارنىڭ يىغىلىپ قالغان چاقچاقلىرىنى ئاڭلىغىلى بارىمەن. بۇلتۇرمۇ شۇنداق قىلغان، بۇ يىلمۇ شۇنداق قىلدىم، ئاللاھ بۇيرىسا كىلەر يىلىمۇ شۇنداق قىلىمەن. بوستۇن بۇ مەزگىللەردە سوۋۇپ كىتىدۇ، بۇ ئىككى يىلدا مەن "جەنۇپقا، ئىسسىق جايغا سەپەر قىلدىم" دەرھەقىقەت، قىرىنداشلارنىڭ،   دوستلارنىڭ  مىھماندوستلۇغى ئادەمنى تەسىرلەندۈرسە، ئۇلارنىڭ ئاغزىدىن زاۋۇتتىن چىققاندەك تۆكۈلۇۋاتقان چاقچاقلىرى كۆپچىلىككە، جۈملىدىن ماڭا زور خوشاللىق ئاتا قىلدى. ئۆز قەۋمىڭ بار يەردە ھايات بار، ھايات بار يەردە -ئۈمىت بار، ئۈمىت بار يەردە چاقچاق بار، شۇنداقلا يەنە چاقچاقلارغا كىلىپ تۇرىدىغان ئادەملەر ۋە ئىشلار بار. بۇلتۇر بىر سورۇندا  مۇنۇ چاقچاقنى سورۇندىكى بىرسىگە قىلىۋىدىم، كۆپچىلىك كۈلۈشۈپ كىتىشتى. قىنى كۆپچىلىك سىلەرمۇ ئوقۇپ بىقىڭلار، بۇ  چاقچاق كەپتىمۇ، كەلمەپتىمۇ؟ كەلگەن بولسا كۈلۈپ قويۇڭلار، مىننەتدارمەن. كەلمىگەن بولسا، ئەپۇ قىلىڭلار، ئۆزرىلىكمەن، ئاستا-ئاستا ئۈگۈنۈۋاتىمەن، "غەيرەتنىڭ پايپىقىنى ئەسئەتكە كەيگۈزۈپ" دىگەندەك .
****************************************************************************
       ئالى مەكتەپتىن تارتىپ تونۇيدىغان بىر  تونۇشۇم بار ئىدى، ئۆزئارا ياخشى تونۇشاتتۇق، ئىككىمىز كۆرۈشمىگەنگە ئون بەش يىل بوپتىكەن، رەڭگى روھىمىزدە ئۆزگىرىشمۇ بوپتۇكەن . ئۇ بۇلتۇر بىر سورۇنغا خۇددى ئاسماندىن مۆلدۈر تۆكۈلگەندەكلا توكۇلداپ كىرىپ كەلدى. مەن ئۇنى دەرھاللا تونۇدۇم ۋە قىزغىن سالام بەردىم.
--سالام ساڭا ھوي غازى.
    ئۇزۇن تۇرا كەلگەن بۇ غازى خۇددى يوچۇن ئادەمنى كۆرگەندەك ئىپادىسىز قاراپ تۇرىدۇ، ئەتراپتا كىشىلەر ھەم بىزگە زەڭ سىلىپ تۇرىدۇ.، "ئىش چاتاق" دىدىم ۋە ئۆزۈمنى تونۇشتۇردۇم.
-ھوي غازى، مەن ھىلىقى سىنىڭ بالا  دوستۇڭ ھوشۇر قارى.
     مۇساپىرلىق تۇرمۇش ئۇنىڭغا چاققانلىقنى ئۈگۈتۈۋەتكەن چىغى، ئۇمۇ دەرھاللا چىرايىغا كۈلكە يۈگەرتىپ خۇددى "كونا دوستى ھوشۇر قارى بىلەن قايتا تىپىشقىنىغا خوش بولغاندەك قىزغىنلىق بىلەن"ۋاي ھوشۇر قارى ئاغىنەم دەپ" قۇچىغىنى ئاچتى. شۇ ئارىدا كۆزۈم ياندىكى مىنىڭ سۆزۈمنى ئاڭلاپ تىڭىرقاپ قالغان بىر نەچچە كىشىگە چۈشتى.
     مەن دەرھال ئۇنىڭ مەيدىسىگە يىنىك ئۇرۇپ دىدىم كۈلۈپ كىتىپ:
-مەن ھوشۇر قارى ئەمەس، مەن قەيسەر مىجىت.  ئىسىت سىنىڭ خاتىرەڭگە.
-ۋاي، ۋاي، ئاپلا، كەچۈر دەپ كەتتى ئۇ.
بۇنى كۆرۈپ تۇرغانلار كۈلۈشۈپ كىتىشتى.
*********************************************************************************************************************
      يەكشەنبە يىرىم كىچىدە سەپەردىن قايتىپ كەلدىم، سەپەر تەسىراتلىرى يىڭى بولغاچقا، ھىسلار مىنى دۈشەنبە كۈنى ئارام ئالغۇزمىدى. شۇڭا ئۇنى تۆكۈپ قويغۇم كەلدى.
**********************************************************************************************************************

مەن بىر يەرگە بارار چىغىمدا،
تاقىتىم -تاق بولۇپ كىتىدۇ.
بىر يەرلەرنى كۆرۈشتىن بۇرۇن،
سىماسىنى دوست-يارەنلەرنىڭ،
گەپ-سۆزىنى، چاقچاقلىرىنى،
ئۇسۇلى ۋە قىلىقلىرىنى-
بەكلا كۆرگۈم كىلىپ كىتىدۇ.
بەزىلەرگە نىسىپ بولمىغان،
مۇھەببەتلىك كۆڭۈلگە ئىگە،
ئۇيغۇرلۇغۇم تۇتۇپ كىتىدۇ.
مەن مۇشۇنداق توپنىڭ ئىچىدە،
دومىلىسا دومىلايمەن تەڭ،
توختىغاندا توختاپ قالىمەن.
گەرچە ئۆزەم شاشراق بولساممۇ،
ئەمما توپقا ھۆرمەت قىلىمەن.
ئۇلار كۈلسە كۈلۈمەن مەنمۇ،
ئاچچىق يۇتسە مەنمۇ دەرتمەن.  
"ئادىمىلىك- ئادەملەر بىلەن"،
نىمە دىگەن ئۇلۇغ سۆزلەر بۇ،
ئەجدادلارغا قايىل بولىمەن.  
  شۇ ھىسلارنىڭ ئەللەيلىشەدە،
جاپاسىنى ئۇزۇن سەپەرنىڭ-
بىر شىرىن چۈش كەبى كۆرىمەن.


مەن نىشانغا يەتكەن چىغىمدا،
مەندىن بۇرۇن كىرىپ كىتىدۇ،
مەن ئۇزاقتىن پەپىلەپ كەلگەن،
قەلبىمدىكى دوستلۇق سۆيگۈسى!
كۈلكىلەرگە خۇرۇچ بولىدۇ،
سىنى ئۇزۇن كۆرمىگەن بىرسى-
خاتا تونۇپ تاشلىسا سىنى!
ئۇندا ئەمەس ئەيپ ئەسلىدە،
ماغدۇرۇم بار دىگىنىڭ بىلەن،
چىراي-شەكلىڭ سەن ئەمەس بىرسى.
لۇتپۇللانىڭ ئىيتقىنى توغرا-
يىللار دىگەن ۋاقىت ئوغرىسى!
ئۆكۈنۈشنىڭ ئورنى يوق، شۇڭا-
ئاغدۇرىسەن نەزەر كۆزۈڭنى،
ھىسلىرىڭغا يانداش ھىسلارغا.
قۇچاقلىشار تەبئىتىمىز،
پارتىلايدۇ روھىيىتمىز،
ھارغۇنلۇقنىڭ شەيتانلىرىنى،
چاقچاق قىلىپ ھارغۇزىۋىتىمىز.
شۇ دەقىقە شۇ مىنۇتلارنى،
"چەكسىز بولسا" دەپمۇ كىتىمىز....


مەن ئۆيۈمگە قايتار چىغىمدا،
كەتكۈم كەلمەي قايتىپ كىلىمەن.
خىيالىمدا- كۆڭۈل تۆرۈمدە،
چاقچاقلارنىڭ ئەڭ پەيزىسىنى،
خاقان قىلىپ ئولتۇرغۇزىمەن.
مەن ئۆزۈمغۇ چاقچاقچى ئەمەس،
ئەمما پىشقان "چاقچاق ئوغرىسى" .
كاللام ساقتۇر -نىيىتىم دۇرۇس،
مەن "ئوغۇرلاپ" كەلگەن چاقچاقتىن،
"باشقىلارمۇ ھوزۇرلانغۇسى".
كۈلكىدە بار ئۈمىت ۋە ئىشەنچ،
ئۇندىن ياغار مىھىر-مۇھەببەت.
بۇ دۇنيانىڭ نامەرتلىگىنى،
ئىشەنچ بىلەن جايلايمىز پەقەت!


مەن ئويۈمگە كەلگەن چىغىمدا،
ھىس قىلىمەن مەۋجۇتلىغىنى-
پىچەتلىرى ھىچ بۇزۇلمىغان،
چاقچاقلارنىڭ سەپ تارتقىنىنى.
ۋە ئىيتىمەن ئۇلارغا شۇنداق:
"ئىيتىلمىغان ئەي چاقچاقلىرىم،
شۇنداق كۈنلەر كىلىدۇ بىر كۈن،
سۆزلىنىدۇ- سۆزلەنمىگەنلەر،
كۈيلىنىدۇ -كۆمۈلمىگەنلەر،
ئۇيغۇرلۇغۇم- قىزغىنلىق بىلەن،
ئۇيغۇرلۇغۇم - ھەر قايسىڭ بىلەن.
سىلەر بىلەن چىڭىيدۇ دوستلۇق،
روھلار ئالار سىلەردىن ئوزۇق.
دوستلار مىنى يوقلاپ كەلگۈچە،
بۇ كۆڭلۈمنى تۇتۇڭلار بۈيۈك"!


شەھرى بوستون، يىزىلدى ئىككىنجى سىنتەبىر

ئۇيغۇر ئائىللىرىدە يۇمۇرىستىك تۇيغۇ كام بولسا بولمىغىدەك ، باللىرىمىزنىڭ يۇمۇرۇسىتىك چوڭ بولۇشى ئۇچۇن مەخسوس ئويلانساق بولغۇدەك ،

بۇرۇنلاردا ئۇيغۇرلاردا يۇمۇرستىك تۇيغۇكام ئەمەس ، مەھەللە قارىشىنىڭ ئاستا ئاستا يىمىرلىشى بىلەن شەھەرلەردە تۇرمۇش رىېتىمىنىڭ تىزلىىششى بىلەن بۇ تۇيغۇلارغا ئازدۇر كۆپتۇر توپا قونمايمۇ قالمىدى.

يەنە كىلىپ ھەر قايسى شەھەرلەردە يۇمۇرلار ئوخشىمىغان خاسلىقلار بىلەن ئوتتۇرغا چىقىدىكەن ، غۇلجىدا بىۋاستە چاقچاق بىلەن ناھايىتى يوقۇرى كەيپيىاتتا باشلانسا ، قومۇلدا سىلىق سىپايە گېزى كەلگەندە ھەزىل قوشاق بىلەن سەھنىلەشكەن چىراي مەيدانغا كىلىدىكەن ، قەشقەردە ئاڭلاپ ئۆيگە بىرىپ بولغاندا چۇشۇنۋالىدىغان مارافۇنچە  يۇمۇر  بولسا خوتەندە نەق مەيداندا بىرنى بىر دەپ تۇگتىىدىغان چاقچاقلار مودا ئىكەن ،
