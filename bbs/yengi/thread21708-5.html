مەنچۇ قارا بۇ پىراگراممىغا ئازراق بولسىمۇ پايدىسى تىگىپ قالامدىكىن دەپ يازدىم. ياخشى ئوقۇ ...پوقۋاش...مەن ئۇنىڭدىن ئىۋەن ئىزدىمىدىم.بەلكى تەكلىپ بەردىم تەكلىپ. مەن سەندەك نىمىلا قويۇپ بەرسە ھە دەپ..... ئاغزىغا چىۋىن كىرىپ كەتكىنى بىلمەي ،ماڭقىسىنى شۇمەپ تىلۋۇزۇر كوردىغانلاردىن ئەمەس.ياخشى نىيەت بىلەن يېزىلغان بىر تەكلىپنى نىمە يامىنىغا بۇرايسەن؟ ئەقىل ئىگىسى بولساڭ ئادەمنىڭ شەخسىيتىگە تىگىپ ئىنكاس يازما....پوق يەپ ،ئاغزىڭنى سېستىپ يۇرگىچە...ئەل -خەلق ئۇچۇن پايدىسى تەگمەيدىغان پارازىت قۇرۇتتەك ياشايدىغانلىقىڭنى بىلگەن بولسا ،ئاپاڭمۇ سېنى تۇغماستىن ،بىرەرتۇپ كۆچەت تۇۋىگە چىچىپ قويغان بولسا ،ئوغۇت بولسىمۇ بولاتتىدە...ئىسىت !!!

ئەسسالامۇ  ئەلەيكۇم  ،‹‹ئىستىقبال سەھنىسى››گە پىكىر -تەكلىپ بىرىپ مۇشۇ پرگۇراممىنىڭ تېخىمۇ يۈكسىلىشى ھەمدە كەڭ ئۇيغۇر ياشلىرى ئۈچۈن يېڭىچە مۇكەممەل يول ھازىرلىشى ئۈچۈن ئۆسكەن ئۆسمەكچى بولاۋاتقان پۇتاقلارنى قىرماقچى بولغىنىڭىزنى ھېس قىلدىم چۈنكى يېڭى نوتىغا باغۋەن كېرەك بولىدۇ ئەمەسمۇ ؟ ..   سىزنىڭ شۇنچە كۆڭۈل بۆلىگىنىڭىزگە ئەستايىدىل پىكىر مۇلاھىزە ئېلپ بارغىنىڭىزگە تەشەككۈر ئيېتمەن .. ئەمما شۇنى ھەم تۆۋەنچىلىك بىلەن ئەسكەرتىپ ئۆتۈشنى توغرا تاپتىم   ،بىرىنچى پىكىرىڭىزدە 1.قاتناشقان شىركەتلەرنىڭ سانى ۋە دائىرسى چەكلىك بولۇپ قاپتۇ. دەپسىز  ...  سىز ئۇيغۇر مىللىي كارخانىچىلىرىمىزنىڭ ھازىرقى ئورنى ۋە تەرەققىياتىغا ئىستىقبال سەھنىسىگە دىققەت ئىتبارىڭىزنى يۈرگۈزگەندەك يۈرگۈزۈپ بېقىڭ .. ئىگىلىك تىكلىگەن كارىخانىلىرىمىز قانچىلىك ؟ بۇ يەرگە كېرىش سالاھىيتىگە ئىگە بولالايدىغانلىرى قانچىلىك .. سانىنى كۆپەيتىمىز دەپ سۈپەتكە ئېتبار بەرمىسە رىستۇران خوجايىنلىرىدىن تارتىپ كىرسە ئەلۋەتتە سىزدەك مەندەكلەرنىڭ قاتناشقۇسى كەلمەيدۇ .... ئالىي مەكتەپ ئوقۇغۇچىلىرى كارخانىچىلىرىمىزنىڭ ئىگلىك تىكلەش ئۇسۇلىنى ئۆگىنىپ  تېخىمۇ تەرەققىي قىلغان مىللىي كارخانىلارنى قۇرۇپ سىزنىڭ بىزنىڭ بۇ ئارزۇلىرىمىزنى ئىشقا ئاشۇرۇرشنى چىن دىلىمدىن تىلەيمەن  ....  ئىككىنچى پىكرىڭىز 2.دۆلەت ئىگىدارچىلىقىدىكى ۋە باشقا رايونىمىزدا ماكانلاشقان نۇرغۇن شىركەت گۇرۇھلارمۇ قاتناشقان بولسا   .... دۆلەت ئىگدارچىلىقىدىكى باشقا رايۇنلارمۇ كېرسە دېگەن پىكرىڭىز مېنى تولىمۇ ھاياجانغا سالدى ئەمما ئويلاپ بېقىڭ رايۇنىمىزدىكى كىچىك ماڭقا شىركەنلەرمۇ خىزمەتكە قوبۇل قىلىشتا بىزنى چەكلىمىگە ئۇچراتسا ئۇ شىركەتلەر قىزىقامدۇ ؟ ھەم ئۇلارنىڭ كۆپ قىسمى دۆلەتلىك ئىمتىھان شەكلى بىلەن ئادەم قوبۇل قىلىدۇ  ... بۇنىڭغا دۆلەتنىڭ ماس قائىدە -قانۇنلىرى بىلەن بىردەكللىككە كېلەمدۇ بىزنىڭ يېڭىلا قۇرۇلغان پرگۇراممىزغا قاتنىشىقا رازىلىقى بولامدۇ شۇنى ئويلىشىپ كۆرشىڭىزنى ئۈمىد قىلىمەن .. ئەڭ ياخشىسى ئۇنىڭدىن ئۈمىد كۈتمەي ھەربىرىمىز بەلنى باغلاپ مىللىي كارخانىلارنى كۆپرەك قۇرۇپ ئۆزىمىزگە يول ئېچىشقا تىرىشايلى  .... ئۈچىنچىن پىكرىڭىز 3.ۋاقىت سەل ئۇزارتىلسا.. سىزنىڭ بۇ پىرگۇراممىنى ناھايىتى قىزقىپ كۆرەۋاتقانلىقىڭىزنى بۇ گېپىڭىزدىن كۆرىۋىلىش تەس ئەمەس سىز ئەمەس بىزمۇ ھەم شۇنداق ئۈمىدتە ئەمما پىرگۇرامما ۋاقتى 90مىنۇت بۇنىڭدىن ئۇزۇن بولۇپ كەتسە بەلكىم بۇ پىرگۇراممىنى ياقتۇرمايدىغان ئۈچۈن 90مىنۇت 90يىل تۇيۇلار ئۇغۇ مەيلى ئەمما پىرگۇرامما قويۇشنىڭ مەلۇم بىر قائىدە قانۇنيەتلىرى بولىدۇ  ئائىلىدە ئائىلە تۈزۈمى دۆلەتتە دۆلەت قانۇنى بولغاندەك ....  بۇنىڭغا مېنىڭچە ئۇلارمۇ ئامالسىز  .. ئاخىرقى پىكرىڭىز 4.ماۋۇ شېرىكەت ۋەكىللىرىمۇ ئۆزىنىڭ كۆزقاراشلىرىنى ،ئويلىغانلىرىنى...دادىل ئوتتۇرغا تاشلىسا .  بايىدىن ئىنكاسلارغا بىرقاتار كۆز يۈگۈرتۈپ چىقتىم بىر ئەزىز قېرىندىشىم مىراسنىڭ لىدىرنى بەك كۆپ سۆزلەمدۇ نېمە دەپ ئاپتۇ ؟ .. ئادەتتە بۇ پىرگۇراممىدا ئوچۇقراق كۆپرەك سۆزلەيدىغاندىن شۇلا بار ئوخشايدۇ .. باشقىلارمۇ كۆپ سۆزلىمەيدۇ .. مېنىڭچە باشقا شىركەتتىكىلەر سۆزلىسىمۇ ۋاقىت مۇناسىۋىتىنى ئويلاشقان بولىشى ھەمدە قاتناشقان ئادەمنىڭ كەيپىياتنى ئاياپ قويغان بولىشى مۇمكىن ...  يەنە باشقىلار دادىل سۆزلىسە مىراسنى كۆپ سۆزلەيدۇ دېگىنىدەك ئۈندىدارنى ئۈندەللال قىلۋەتكەن بىز ئۇلارنىمۇ ۋىتىلداق دىيىشتىنمۇ يانمايدىغاندەك تۇرىمىز .. ئەمما ۋاقىت ئۈنۈم مەسىلىسىنى ئويلاشقانلىقىنى نەزەردىن ساقىت قىلماڭ   . ...  بوۋاق ئەمدى مېڭىشنى ئۈگەنگەندە ئۇنى قامچا بىلەن ساۋىماي بەلكى چىن مېھىر بىلەن توغرا مېڭىشنى كۆرسىتىشىمىز كېرەك  ... قامچا بىلەن ساۋۇسا يېڭىلا ماڭغان بوۋاق يەنە مېڭىشتا قورقۇمسراپ قالىدۇ ...  بىز ئۈچۈن پايدىلىق بۇ پىرگۇراممىغا ھەممىدىن يىقىندىن مەدەتتە بولايلى ....
