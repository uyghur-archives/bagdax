ئەسسالامۇ ئەلەيكۇم قېرىنداشلىرىم! مىلادىيە يېڭى يىلىغا مەرھابا!  مانا 21-ئەسىردە ياشاۋاتقىنىمىزغىمۇ 13يىل بوپتۇ، ئۆتكەن ئەسىردە(50 يىل بۇرۇن) قەۋمىمىزنى ھەر ۋاقىت يۇتۇۋېلىشنىڭ پېيىدا يۈرگۇچىلەر ۋە ئۆزىنىڭ (ئۇيغۇر)قەۋمىدىن ئۈمۈدىنى ئۈزگەن ئاز بىر قىسىم ئازغۇنلار، «ئۇيغۇر قەۋمى يېرىم ئەسىردىن كېيىن يەنە مەۋجۇت بولۇپ تۇرالامدۇ؟.......» دەپ پىتنە تارقىتىپ يۈرەتتى، ھالبۇكى مانا تارىخ 2014-يىلىغا قەدەم قويۇش ھارپىسىدىمۇ « يىقىلغاندا باش ئەگمىگەن، يىقىتقاندا دەسسىمىگەن ئەركەك بىر مىللەتنىڭ پەرزەنتى»لىرى نەچچە مىڭ يىللار ئىلگىرىكى ئەجداتلىرىنىڭ ئىزىنى داۋاملاشتۇرۇپ ئانا يۇرتى يەرشارىنىڭ ھەممە يەرلىرىگە يۇلتۇزدەك تارىلىپ «مەن ئۇيغۇر» دەپ ياشاۋاتماقتا! مەرھابا  مىلادىيە يېڭى يىلى !مەرھابا ئۇيغۇر!
«بالىلىرىم قۇلۇقۇڭلاردا دائىم جاراڭلاپ تۇرسۇنكى: ئىنسان ئۈمۈت بىلەن كۆكلەيدۇ، ئۈمۈتسىز ئىنسان ئاسانلا نىشانىدىن ئادىشىپ باشقىلارنىڭ ئولجىسىغا ئايلىنىدۇ؛ ئاينىڭ ئون بەشى ئايدىڭ ئونبەشى قاراڭغۇ »
ئادەمنىڭ چوڭ مېڭىسى بىرتۈگىمەس بايلىق، ئۇيغۇرنىڭ چوڭ مېڭىسى ئەمدى يەنە قايتىدىن ئاچىلىۋاتىدۇ! قاراڭلار 2013-يىلىكى دۇنيا سەھنىسىگە، نەچچە يۈزلىگەن «ئۇيغۇر چەۋانداز»لىرى دۇنيا ئىلىم ئەھلىنى ھەيرەتتە قالدۇرۇپ يېڭىدىن يېڭى ئۇتۇقلارغا ئېرىشمەكتە، دۇلدۇلىنىڭ تۇياقلىرىدىن ئوت چاقنىتىپ ، دۇنياغا ئۇيغارلىقىنى جاكارلىماقتا! چاپقىن  چەۋاندازلىرىم چاپقىن ! ئىلمۇ -ئىرپان داۋانلىرىدىن ئاشقىن،پەلەكىيات ساھىلىغا يەتكىن! 21-ئەسىر سېنىڭ باغاشلىشىڭنى كۈتۈپ ئىللىق قوينىنى كەڭ ئاچماقتا!

  ساپانىڭ تۇۋەنلىگى ئىنساننى ئوز قەدىرقىممىتىنى بىلىش،غۇرۇر بىلەن ياشاش ئىستىگىدىن مەھرۇم قىلىدۇ.كىشىنىڭ ساپاسى ئىمان،ئەخلاق،بىلىم دەرجىسى بىلەن بەلگىلىنىدۇ،بۇ ئۇچى بىرگەۋدە،كۇپ ۋاقىتلاردا بىز ئىمان،ئەخلاق،بىلىمنى ئايرىپ ۋە تەرتىبىنى ئالماشتۇرۇپ چۇشەندۇرۇپ،كۇپ خاتالارگە يول قويدۇق ۋەبۇ خاتالار داۋاملىشىۋاتىدۇ،بۇ خاتالار تۇزۇتۇلمىسە، بىز ئۇزىمىزنى،قىسمىتىمىزنى ئۇزگەرتەلمەيمىز. شېئىرىڭ كۇڭلۇمگە ياقتى،شېئىرىڭدىن پارچە ئېلىپ،كەينىگە ئىككى كۇپلېت قوشاق قوشۇپ،يېڭى يىلدا ھەمرا بولدۇم:

«تىلەپ تۇرماي ياخشى كۈنلەرنى،
بىز ئۆزىمىز بولايلى ئىشچان.
توسالمايدۇ پاسىق دۈشمەنلەر،
پۇختا بولساق،- بولساق تىرىشچان.
ھىچكىم سەنچۈن قاتۇرمايدۇ باش،
بولدى دەس تۇر، ئۇيقۇڭدىن ئويغان.
ماڭماق بولساڭ- كۆزۈڭ ئوچۇق ماڭ،
كاللاڭ سەزگۈر بولسۇن ھەر زامان.
قاراڭغۇدا قالساڭ مۇبادا،
يۇرۇقلۇقنى ئىزدە ئۆزۈڭدىن،
كىرسىن سوراپ بولما ئاۋارە،
ئۈمىت كۈتمە نامەرت قوشناڭدىن
ۋايساپ كەتمە، جاپالار چەكسەڭ.
جاپالارنى كىملەر چەكمىگەن؟.
بۇ ھاياتتا ئۇتىدۇ ئاخىر،
مۇشكۈلاتقا بويۇن ئەگمىگەن.

«بىلىم، ئىمان ۋە ئەدەپ بىلەن،»

كۆپ كىشىلەر مەقسەتكە يەتكەن.
قەدىرلەپ قوي مۇشۇ سۆزلەرنى،
يىراقتىكى قەيسەر ئەۋەتكەن»

قەيسەر ئەۋەتكەن سالامنى ئالدىم،
يېڭى يىلغا بولۇپتۇ مەشئەل.
يىللار داۋام ئەتكەن ئىبرەتلەر،
قىلىنىپتۇ ئەلگە قەتلىئەم.
خاتالاردىن كىلىدۇ ئەقىل،
ئاسانلىقچەئالمىدۇق بىزتەن.
كۆپ كىشىلەر مەقسەتكە يەتكەن،

«ئىمان،ئەدەپ ۋەبىلىم بىلەن.»
