ئۈندىدارغا ھەمبەھىرلەش ياردەم
يوللىغان ۋاقتى 2014-1-22 20:08:13 |ھەممە قەۋەتنى كۆرۈش

              چىراق123  قىرىندىشمىزنىڭ تېمىسىدا  يىزىلغىندەك  تازغۇن ھەق ئېلىش پونكىتىنىڭ  ھەققى بەكلا  يۇقىرى،،يىڭسارنىڭ ماشىنلىرى يۇقىرى سۇرئەتلىك تاشيولدا تەخمىنەن45 كىلومىتىرئەتراپىداماڭىمىز،،لىكىن يەكەندىن كەلگەن ماشىنلار بىلەن ئوخشاش ھەق تۆلەيمىز،،بۇنداق ئوشۇق تۆلىشىمىزنىڭ  سەۋەبى نىمە؟ نىمە ئۈچۈن پەرىقلەنمەيدۇ؟بىز شوپۇرلار مۇناسىۋەتلىك  ئورۇنلارغا ئەھۋالنى ئىنكاس قىلساق ھەق پونكىتىنىڭ  پۇلىنى ماشىنا  بىلتىگە قىتىپ بەردۇق،،،،يەنە نىمە دىيشسەن  دەپ ئاغزىمىزنى  ئاچۇرمايدۇ.....نەتجىدە  قانداق بولدى  ؟كەڭ خەلق ئاممسىغا،،،ئاۋام پۇخرالارغا،،،،،ئاران جىنىنى بىقىۋاتقان دىھقانلىرىمىزغا،،،شۇنداقلا بىز شوپۇرلارغىمۇ زىيان بولىۋاتىدۇ،،،،،خوش شوپۇرلارغا قانداق زىيان بولىدۇ دەپ سورىشىڭلار مۇمكىن ،،،مەن  ئەھۋالىمىزنى قىسقىچە دەپ ئۆتەي،
         يىڭساردىن--قەشقەرگە قاتنايدىغان قانۇنلۇق يول تاختىسى  ھۆكۈمەت دەپتىرىدە بار چوڭ  تىپتكى ماشىنلاردىنلا 59 ماشىنابار،،،،بۇ ماشىنلارنىڭ ئىچىدە 39 كىشلىك ماشىندىن 28 دانە،،،،18 كىشلىك ماشىندىن 31 دانىسى بار ،،، كىچىككنە يىڭسار ناھىيسىدە بۇ ماشىنلارنىمانچە كۆپ،،تىخى 4 كىشلىك كىچىك ماشىندىن 120چەماشىنا،،،،،يىڭسارنىڭ يىزلىرىدىن بىۋاستە قەشقەرگە ماڭىدىغان  چوڭ تىپتىكى ئاپتۇبۇستىن14 سى بار،،،،بۇماشىنلار نىمانچە كوپ ھە!!!!  ھەممە ماشىنا يىڭساردىن---قەشقەرگە   قاتنايدۇ  ھە!
            بۇندىن سىرت ھەممىدىن قولاي،، ئەرزان،،چىقسىڭىزلا  قەشقەرنىڭ ھەرقايسى كوچىلرىغا،بازارلاغىچە،،ئۆيىڭىزنىڭ ئشكىگىچە ئاپىرىپ قويىدىغان پائالىيەتچان  تىنىم تاپمايدىغان،،ئېزا تارتىشنى قەتئى بىلمەيدىغان50-60چە قارا ماشىنلار بار..خوش قارا ماشىنلارنى تۇتىدىغانلار يوقمۇ؟جۇڭگۇ كومۇنىستىك پارتىيسىنىڭ ياخشى سىياسەتلىرى قەشقەرگەكەلمەمدۇ ۋەياكى ئارلىق يىراق كەلگىچە ھىرىپ قىلىپ كۈچى قالمامدۇ بىلمەيلا قالدۇق،،،،، قاتناش تىرانسىپورىت باشقۇرۇش،نازارەت قىلىش ئىدارىلرى،،،ئاشۇ قاتناش باشقۇرۇش تەرتىپكە سىلىش خادىملىرى،،خىزمەتچىلرى نىمە ئۈچۈن ئۇ قارا ماشىنلارنى تۇتۇپ بىر تەرەپ قىلمايدۇ؟مەن بۇلارنىمۇ چۈشەندۈرۈپ ئۆتەي......ئۇ قاتناش باشقۇرۇش خادىملىرىنىڭ ئاپتۇبۇسى يوق ئۇلار ئىشخانسىدا ئولتۇرۇپلا كىم  پارابەرسە،،كىم سوۋغاتنى كوپرەك بەرسە يول تاختا لىنيەسىنى تەستىقلاۋىرىپ يىڭسار--قەشقەر لىنىيسىگە شۇنچىلىك كۆپ ماشىنلارنى تەستىقلاۋەرگەن،،ئۇلارنىڭ شوپۇرلارپۇل تاپالامدۇ-تاپالمامدۇ كارى يوق . يول لىنيە تاختىسنى تەستىقلاپ شىركەتلەر ئۇلارنىڭ تويماس نەفسىگە ئۇرغان پۇللارنى يەپلا تەستىقلاۋىرىپ ماشىنلار بايا سانىغاندەك زىيادە كۆپ،،،لىكىن ھەر ھالدا2013-يىلدىن بۇرۇن ئاللا بەرگەن رىسقىمىز بىلەن سىرىقتال كۈن ئىلىپ كىلىۋاتقان ئىدۇق،،2012-يىلى يۇقىرى سۇرئەتلىك يول پۈتۇپ كەڭ خەلق ئاممىسى  ۋە بىز شوپۇرلاربىشىمىز ئاسمانغا يەتكىدەك خوش بولغان ئىدۇق،ھەتتا يولدا كەلگىچە بەزى دىھقانلار پارتىيگە رەخمەت ئىيتىپ دۇئالارقىلغان.. ھالبۇكى 2012 -يىلى11-ئاينىڭ 15-كۈنى يۇقىرى سۈرئەتلىك يول ھەققنى ئىلشنى باشلغاندىن تارتىپلا چوڭ- كىچىك ماشىنلارنىڭ بىلەت باھاسىغا 2 ھەسسە قىتىلىپ كەتتى.شۇنىڭدىن كىيىن قەشقەر كىلىپ كىتىدىغان كەڭ خەلقىمىز،،دىھقانلىرىمىز،،تىجارەتچلىرىمىز قۇمغا سۇ سىڭگەندەكلا شىپپىدە يوقاپلا كەتتى،،،ئاپتۇبۇسلارنىڭ بىلەت باھاسى ئۆسكەن بىلەن تەڭلا قارا ماشىنلار   (4كىشلىك ) يامغۇردىن كىيىنكى بامبۇك نوتىلىرىدەك ئاۋۇپلا كەتتى .ئۇلار ھۆكۈمەتكە ۋەياكى باشقا ئورۇنلارغاھىچقانداق ھەق تۆلىمگەچكە  خىرىدارلارنى ئاپتۇز بىلەت باھاسى بويىچەۋەياكى5يۈەن پەرىقلەندۇرۇپلا توشىۋەرگەچكە خىرىدارلىرىمىزنى تەبئىلا تارتىپ كەتتى مەسلەن:  <<ئاپتۇزنىڭ بىلەت باھاسى19 يۈەن بولسا ،،قارا ماشىنلار20-25 يۈەن ئىلىپلامىڭىپ كىتىپ بارىدۇ >>ئەمدى بۇلارنى تۇتىدىغان قاتناش باشقۇرۇش خادىملىرى،،،بىكەتتىكى مۇناسىۋەتلىك خادىملارغا كەلسەك ئۇلار بايا يازغىنىمدەك نەفسىنىڭ قۇللىرىدۇريەنى قارا ماشىنلارنىڭمۇ ئۆزى تەيىنلىگەن باشلىقلىرى بولۇپ شۇ باشلىقلرى  ئايلىق باشقۇرۇش ھەققى يىغىپ قاتناش باشقۇرۇش تارماقلىرىنىڭ ،بىكەت خادىملىرىنىڭ گىلىغا يىگۈزىدۇ،،شۇڭا قارا ماشىنلارنىڭ يولى بەكمۇ داغدام ئوچۇق
              بۇ ئىشلارغا ئىسپات بارمۇ دەپ سورىشىڭلار مۇمكىن،،،ئەلۋەتتە ئىسپات تولۇق،،يەنى قارا ماشىنلارنىڭ يىڭسار ناھىيلىك يولۇچى توشۇش بىكىتىدىن 5-6 يۈز مىتىر يۇقىرسىدىلا ( مائارىپ ئىدارىسىنىڭ ئالدىدا)تاكسىداكەلگەن يولۇچىلارنى توسۇپ،،توۋلاپ يۈرۈپ 4كىشلىك ماشىنلىرىنى توشقۇزۇپ مىڭىشى ئەڭ ياخشى ئىسپات،قەشقەر ۋىلايەتلىك يولۇچى توشۇش بىكىتنىڭ تۆت ئەتراپىدا قارا ماشىنلار قورشاغلىق شۇ ئەراپتا ئۆيى بار كىشلەرمۇشۇ ئەتراپقا كەلسىلا قارا ماشىنا شوپۇرلىرى يىىڭسارغا بارامسەن؟يىڭسارغا بىرلا ئادىمىىم (خىرىدارىم)كەم دەپ ئالدىنى توسۇپ ئۇلارنى بىزار قىلىۋىتىدۇ،،ئۇلار كەچتە يىرىم كىچىلەردىمۇبار ئىشەنمىسىڭىڭىز  شۇ ئەتراپقا كىلىپ سىناپ بىقىڭ ئۇلارنى كۈزىتىپ باقسىڭىز كۈلۈپ كىتىسىز.......تىخى كەچقۇرۇن سائەت5لەر بىلەنلا يىڭسارناھىيلىك بىكەتنىڭ ئۇدۇلىغا كىلىپلا خىرىدارلارنى ماشىنغا سىلىشى بار . شۇ ئىشلارغا شوپۇرلار پىكىر قىلىپ يۈرۈپ بىكەتتىكى خادىملاردىن3-2 لەننى شۇلارنى تۇتۇشقا ئاجراتقانمۇ بولدى لىكىن ئۇلارمۇ ھەپتە ئون كۈن ئىشلەپلا يەنە يوقاپلا كەتتى،،،،بۇلار دائىم شۇنداق،پىكىر قىلىپ داد ئەھۋالىمىزنى ئېيىتساق ھەپتە ياكى ئىشىپ كەتسە ئون كۈن ئىشلەپلا يوقايدۇ.شۇنداق ئشلەش جەريانىدا قارا ماشىنا شوپۇرلىرى  ئۇلارنى سوقۇۋىتىپ قىچىپ كەتتى،ئۇلار بىلەن جىدەللىشىپ-ئۇرۇشۇپ،،شوپۇرلار ياردەمگە بارساق كىيىن بىزدىن ئۆچىنى ئىلىپ دىگەندەك قاراجەمىيەت تەشكىلى تۈسى بار،<<مال ئىگىسىدىن ئوغرى كۈچلۈك>> ھالەت شەكىللەندى.ئۇلارنىڭ گىپى كۈچلۈك،،ئۇلارنىڭ گىپى توغرىدەك ھالەت شەكىللەندى.قىسقسقى قارا ماشىنا شوپۇرلىرى  قاتناش باشقۇرۇش تارماقلرىنىڭ خادىملىرىنىمۇ،بىكەت خادىملىرىنىمۇ،بىز شوپۇرلارنىمۇ قورقۇتتى. قاتناش باشقۇرۇش ،بىكەت خادىملىرىنىڭمۇ ۋە شوپۇرلارنىڭمۇ ئەنسىرەيدىغان ئۆيى،ئائىلسى باردە .قىسقىسى قارا ماشىنا شوپۇرلىرى يىڭىپ چىقتى.خوش ئۇلاردىن ئاشقان خىرىدارلارنى بىكەتنىڭ ئالدىدا تۇرىدىغان4 كىشلىك كىچىك ماشىنلار بۇلاڭ-تالاڭ قىلغاندەك ماشىنسىغا كۆتۈرۈپ ئالغىدەك  قىلىپ قولتۇقىدىن تارتىپ يۈرۈپ ،ئۇنىڭغا ئۇنىمسا ئاپتۇبۇسقا چىقساڭ 2سائەت ئولتۇرىسەن،،يىرىم كۈن ئولتۇرىسەن دەپ يۈرۈپ سىلۋالىدۇ .بىزنىڭ ئاپتۇبۇسلار باغلاقتىكى ئىتتەك بىكەت قورۇسىنىڭ ئىچىدە تۇرىمىز،كىچىك ماشىنلار تىشىدا تۇرىدۇ شۇڭا ساددا دىھقانلارنى بايامقىدەك سۆزلەر بىلەن ئالداپ يۈرۈپ سىلۋالىدۇ .ئابتۇبۇسنىمۇ تىشىغا چىقىپ خىرىدار سالايلى  دىسەك  ئېلىپ كەلگەن يۈك تاقنى تەكشۈرەيمىز،سالاھىيەت كىنشكىسىنى تەكشۈرىمىزدەپ تىشىغاچىققلى قويمايدۇ، ،ئىچىگەكىرىدىغان خىرىدارلارنىڭ قوغۇن تىلىپ يەيدىغان قەلەمتۇرۇچى بولسىلا بىشىنى تاشقا ئۇرسىمۇ كىرىپ ئاپتۇبۇسقا چىقالىشىمۇمكىن ئەمەس،تىشىدىكى كىچىك ماشىنلارغا ئۇنداق تەكشۇرۇشلە يوق خىرىدار كىلىپ ماشىىنغا چىقسىلا يىندا پارتىلاتقۇچ بومبا بولسىمۇ تەكشۈرەيدىغان ئادەم يوق، ماشىنغا تۆت خىرىدارنى سالسىلا ھىچنىمگە قارىماي كىتىۋىردۇ.بىلەت باھاسىمۇ مۇقىم ئەمەس ئەسلى بىلەت باھاسى30 يۈەن ،لىكىن خىرىداركۆپ چاغلاردا40-30يۈەن ،خىرىدارئاز چاغلاردا25-20يۈەن قلىپ مڭۋىردۇ... خىرىدارلار كىچىك ماشىنا شوپۇرلىرىنىڭ مۇشۇنداق قاقتى-سوقتى قىلشىغا كۆپ نارازى بولسىمۇ لىكىن كىچىك ماشنا قولاي بولغاچقا قاقشاپ يۈرۈپ چىقۋىردىغان بولۇپ كەتتى،،خوش كىچىك ماشىنانىمە ئۈچۈن بۇنچە ئەركىن؟ ئۇلار بىكەتتكىلەرنىڭ ئۆزبالىسى چوڭ ماشىنلار ئۆگەي بالىسىمۇ؟بۇ يەردىكى مەسلىنىمۇ چۈشەندۈرۈپ ئۆتەي،،،،،چوڭ ماشىنلاربىكەت بىلەن ئايدا بىر ھىساپ بوغىمىز بىكەت بىزگەتەخمىنەن 20 پىرسەنىت ئەتراپىدارەسمىيەت ھەققنى ئايرىۋىلىپ ئاشقان پۇلمىزنى بىرىدۇ.كىچىك ماشىننڭىچۇ ؟ ئوخشىمايدۇ ،ئۇلار بىر يولىغا 7يۈەن ھەق تۆلەپلا خالىغان بىلەت باھاسىدا ماڭالايدۇ،،،بۇ نىمە ئۈچۈن؟؟؟؟؟؟؟؟؟؟نىمە ئۈچۈن ئوخشاش باشقۇرمايدۇ؟ھەققىتەن بىز بىكەتنىڭ ئۆگەي بالىسى بۇنىڭدىكى سەۋەپ كىچىك ماشىنلاردىن ئالدىغان پۇل ئۇدۇل بىكەتكە ۋە ياكى بىكەت خادىملىرىنىڭ بىۋاستە يانچۇغىغا كىرىدۇ ،چوڭ ماشىنلارنىڭ كىرىمى بولسا بىكەتكە،ۋىلايەتلىك بىكەتكەۋە ئاپتۇنۇم رايۇنغا ئەۋەتىلدۇ.شۇنىڭ بىلەن قانداق نەتىجە كىلىپ چىقتى دىسەك بايا مەن سۆزلەپ بەرگەن ئەگلەكلەردىن ئۆتۈپ بىكەتنىڭ ئىچىگە كىرىپ ئاپتۇبۇستا ماڭىدىغان خىرىدارلار يوق دىيەرلىك،بەكمۇ ئاز،كۈنسىرى ئاز قىلىۋاتىدۇ.بايا سانىغان 59 دانە ئاپتۇبۇس ۋە ئارىبۇسلاريىڭسار ناھىيلىك بىكەتنىڭ ئىچىدە <<ۋولتىرنىڭ سارايۋونى قوغدىشى  >> فىلمدىكى يىقلغۇ يوق توختاپ قالغان تانكىلاردەك قاتار تىزىلىپ ياتماقتا....
