                                 ئامېرىكىلىقلار جۇڭگۇلۇقلارنى زادىلا چۈشەنمەيدۇ             

     
         «جۇڭگونى دۇنياغا تونۇتايلى،دۇنيانى جۇڭگو غا تونۇتايلى»دېگەن شۇئارنى بېيجىڭنىڭ قەيەرىدە كۆرگىنىم ئېسىمدە يوق .جۇڭگولۇقلارنىڭ چەت ئەل تىلى ئۆگۈنۈش قىزغىنلىقى ھەمدە ھەريىلى چەتكە چىقىۋاتقان ئادەم سانىدىنلا جۇڭگونىڭ ۋە جۇڭگولۇقلارنىڭ دۇنيانى بىلىشكە قانچىلىك تىرىشۋاتقانلىقىنى بىلگىلى بولىدۇ .ئەمما مەن ئامىرىكىغا كەلگەندىن كېيىنلا ئاندىن جۇڭگونى دۇنياغا تونۇتۇشنىڭمۇ ئوخشاشلا ئىنتايىن مۇھىم ۋە تەخىرسىز ئىكەنلىگىنى ھېس قىلدىم.ئالدى بىلەن مەن پۇردۇ ئۇنۋېرسىتېتى مۇناسىۋەتشۇناسلىق فاكۇلتېتىنڭ ئاسپارانىتلىرى سورىغان بىر قانچە مەسىلىنى مىسال كەلتۇرۇپ،ئامىرىكىلىقلارنىڭ جۇڭگو ھەققىدە بىلدىغانلىرىنىڭ بەك ئازلىقىنى چۈشەندۈرۈپ ئۆتەي :
    ___بېيجىڭ قەيەردە ؟ ئۇ دېڭىز بويدىكى شەھەرمۇ ؟
    ___سىلەر نىمىشقا بىرلا پەرزەنىتلىك بولۇش سىياسىتىنى يولغا قويىسىلەر ؟ئۇنىڭ جىنايەتئىكەنلىگىنى بىلمەمسىلەر ؟
    ___يەن ئەپەندى،بىر سوئال سورىسام كۆڭلىڭىزگە كەلمىسۇن، جۇڭگلۇقلار ھېلىھەم نامىراتلىق ۋە زۇلۇم ئاستىدا ياشاۋاتامدۇ؟
    ___دىنغا ئىشىندىغانلار جۇڭگودا زىيانكەشلىككە ئۇچىرايدۇ دېگەن گەپلەر راسىتمۇ؟
         ئامىرىكىلىق ساۋاقداشلىرىمنىڭ مېنى ئەجەبلەندۈرگەن مۇنداق سوئاللىرىنى مىسال كەلتۈرسەم بەكمۇ تولا. ئەمدى مەن ئۇلارنى ھەيران قالدۇرغان ئىشلىرىمىزنىمۇ سۆزلەپ ئۆتۈشنىمۇ توغرا تاپتىم :
      تۇنجى قېتىملىق دوكتۇرلۇق دەرىسىدە ئۆزىمىزنى تونۇشتۇرىدىغان ئىش بار. مەن ئۆزۇمنى تونۇشتۇرغاندا، ئۇنۋان ئېلىش ئوقۇشىنى بالدۇرراق تاماملاپ ،دۆلىتىمگە قايتىشنىئويلايدىغانلىقىمنى ئېيتىۋىدىم ،ئامىرىكىلىق ئوقۇغۇچىلار ھەيرانلىق ئىچىدە ئىشەنگۈسى كەلمىگەندەك ماڭا قاراشتى .ئۇلارنىڭ قارىشىچە ،ئامىرىكىغا كەلگەن ھەممىلا كىشى(ئەلۋەتتە جۇڭگولۇقلارمۇ شۇنىڭ ئىچىدە )پۈتۈن كۈچى بىلەن بۇيەردە قېلىشقا ،تېخىمۇ ياخشىراق كۈنگە ئېرىشىشكەتىرىشاتتى .ئۇلارنىڭ قارىشىچە ،ئامىرىكا ياخشى تۇرمۇشقا ئېرىشكىلى بولدىغان بىردىنبىر ماكانئىدى .
        مەن يېڭى ماشىنا سېتىۋېلىش ھەققىدە بىر ئىشخانىدىكى ئايال خىزمەتدېشىمدىن مەسلىھەتسورىغىنىمدا،ئۇ ئىشەنمىگەندەك ماڭا چەكچىيىپ :
___ نېمە يېڭى ماشىنا ؟__دەپ سورىدى .كېيىن بىلسەم ،ئۇنىڭ كاللىسىدا جۇڭگولۇقئوقۇغۇچىلارنى ئومۇمەن ئىنتايىن نامرات ياكى بولمىىسا بەكلا ئىختىسادچان دېگەن قاراشبار ئىكەن .
        بىر ئامىرىكىلىق ئاسسىستېنت سىستېنت مەندىن تولۇق كۇرۇس ئوقۇغۇچىلىرىغا «11سىنتەبىر »ۋەقەسى-دە تەسىراتىمنى بايان  قىلغانلىقىمنى سورىۋىدى ، مەن جاۋابىمنى قايتىلاپ :
        مەن ئىنتايىن قايغۇردۇم ھەم غەزەپلەندىم .قايغۇرۇشۇم ،مەن ئۆز تۇققانلىرىدىن ئايرىلىپقېلىشنىڭ قانداق بولدىغانلىقىنى بىلىمەن،مەن يېقىندىلا قېيناتامدىن ئايرىلدىم .غەزەپلىنىشىم ،مەن كىشىلەرنىڭ ۋەتىنى باشقىلارنىڭ باشقىلارنىڭ ھۇجۇمىغا ئۇچۇرىغاندىكى ھېسىياتىنى بىلىمەن . بىزنىڭ يۇگۇسلاۋىيەدىكى ئەلچىخانىمىز1999-يىلى شىمالى ئاتلانتىك ئەھدى تەشكىلاتى تەرپىدىن بومباردىمان قىلىندى .شۈبىھسىزكى،بۇ مېنىڭ ۋەتىنىمگە قىلىنغان بىر قېتىىملىق ھۇجۇم ،-دېدىم.
       ئۇھەيرانلىق ۋە ئىشەنمەسلىك بىلەن بېشىنى چايقىدى .مەن ئۇ ئايالنىڭ نېمەئويلاۋاتقىنىنى بىلىمەن . ئۇ : سەن بىر جۇڭگولۇق ئاسسىستېنىت،شۇنچە قاتتىق پىكىرىڭنى ،بولۇپمۇ ئامىرىكىلىق ئوقۇغۇچىلارنىڭ نارازىلىقىنى قوزغاش مۇمكىنچىىلىكى شۇنداق پىكىرنى بايان قىلىشقا قانداق پېتىندىڭ ؟ دېمەكچى. ئۇ ئايالمۇ دەل باشقائامىرىكىلىقلارغا ئوخشاش جۇڭگولۇقئوقۇغۇچىلارنىڭ ئېھتىياتچان ھەم سۇكۇت قىلىش پوزىتسىيىىسىگە ئادەتلىنىپ قالغان .
      بىز جۇڭگولۇق ئوقۇغۇچىلار بىرىنجى قېتىم فاكۇلتىتنىڭ ئايدا بىر قېتىم ئۆتكۈزۈلدىغان ئويۇن كۆرسۈتۈش پالىيىتىگە قاتنىشىپ،جۇڭگونىڭ بىر نەچچە ئەنەنىۋى ناخشىلىرنى ئېيتقان ھەمدە   ناخشا ئېيتىشتىن بۇرۇن ئىنتايىن يۇمۇرلۇق قىلىپ قىسقىچە چۈشەندۇرۈش بەرگەن چېغىمىزدا ،مەيداندىكى ئوقۇتقۇچى-ئوقۇغۇچىلار ھاڭ-تاڭ قالد ىئۇلارنىڭ قارىشىچە جۇڭگۇلۇق ئوقۇغۇچىلار ئىنتايىن سۇرلۇك ،ئۇياتچان ھەتتاباشقىلاردىن ئۆزىنى قاچۇرىدۇ . مۇنداقئالاھىدىلىك مەلۇم دەرىجىدە جۇڭگولۇق ئوقۇغۇچىلارنىڭ باشقىلارنىڭ نەزىرىدە مەۋجۇت بولماسلىقىغا سەۋەب بولغان .
    مەن ئۆز كەچۇرمىشلىرىمدىكى بەزى مىساللار بىلەنلا ئامىرىكىىلىقلارنىڭ جۇڭگوغا بولغان چۇشەنچىسىنىڭ كەملىكى ھەمدە جۇڭگولۇقلارغا بىرتەرەپلىمە قارىشىنى ئىسپاتلاپ بېرەلمەيدىغانلىقىىمنى بىلىمەن.ئەمما ،ئۇلارنىڭ ھەممىسىنىڭ مۇناسىۋەتشۇناسلىق كەسپىنىڭ دوكتۇر ياكى
ماگىستىرلىقىتا ئوقۇۋاتقان ئوقۇغۇچىلىرى ئكەنلىكىنى نەزەردە تۇتقاندا ،بىر كىشى ئادەتتىكى ئامىرىكىلىقلارنىڭ جۇڭگۇغا بولغان تونۇشى ھەم ئۇلارنىڭ نەزىرىدىكى جۇڭگولۇقلارنىڭ ئوبرازى ھەققىدە يەنە قانچىلىك ئۈمۇدۋار بولالايدۇ ؟
   جۇڭگۇنى دۇنيا ئۇنتۇپ قالمسۇن .خاتا چۈشەنمىسۇن .

      ھازىرقى زاماندا، كىشىلەرنىڭ دۇنيا ھەققىدىكى بىلىشى ئاساسەن دېگۈدەك مېدىيە ۋاستىلىرىنىڭ تارقاتقان خەۋىرىگە تايىنىدىغان بولۇپ كەتتى. ئېنتىرنېت تورىنىڭ دۇنيا مىقياسىدا ئومۇملىشىشىغا ئەگىشىپ ئامېرىكىغا بولغان چۈشەنچىمىزمۇ بارغانسېرى جىقلاۋاتىدۇ. ئەمما مېنىڭ بايقىشىمچە، قىزىق يېرى، ئامېرىكا تارقاتقان خەۋەرلەر بىلەن كىشىلىرىمىز ئۆزى بېرىپ كۆرگەن ئىشلار ئىككى ئىش بولۇپ قالىدىكەن. مەن چەتئەلگە چىقىپ باقمىدىم، ئامېرىكىغا بولغان چۈشەنچەممۇ جىق ئەمەس. ئەمما ئاخبارات ۋاستىلىرىنىڭ تەسىرىنى سەل چاغلىغىلى بولمايدىغانلىقىنى خېلى چوڭقۇر بىلىپ يەتتىم. مەسىلەن، ئىچكىرىدە ئوقۇش جەريانىدا ئىچكىرىدىكى بەزىلەر شىنجاڭ ھەققىدە خۇددى يۇقىرىدىكى يازمىدا يېزىلغان، سوئال سورىغان چەتئەللىكلەردەك سوئاللارنى سورىغانلار بولغانىدى. بۇ مىسالنى ئېلىشىمدىكى سەۋەب، ئامېرىكا چوڭ بىر دۆلەت، جۇڭگومۇ ھەم چوڭ بىر دۆلەت. ئامېرىكىلىقلارنىڭ جۇڭگو ھەققىدىكى بىلىشىنىڭ ئاز بولۇشى بىلەن جۇڭگودىكى ئىچكىرى ئۆلكىلەردە ياشاۋاتقان خەنزۇلارنىڭ بىز ھەققىدىكى بىلىشىمۇ ئوخشىشىپ قالىدىكەن.
       يېقىنقى يىللاردىن بېرى شىنجاڭنى تونۇيدىغانلار خېلىلا جىقلاپ قالدى ( شىنجاڭ ھەققىدە سەلبىي خەۋەرلەرنىڭ كۆپ تارقىلىشى سەۋەبىمۇ بار)، ئەمما بىز ھەققىدە ئەمەلىيەتكە ئۇيغۇن، توغرا، ئەتراپلىق چۈشەنچىگە ئىگە بولغانلىرى قانچىلىك بۇنى مەن بىلمەيمەن. ئەمما مېنىڭ ھېس قىلىشىمدا بىزمۇ خۇددى مۇشۇ يازمىدا يېزىلغاندەك ئۆزىمىزنى جۇڭگوغا قانداق تونۇتۇش ھەققىدە ئويلىنىۋاتىمىز. بەزىلىرىمىز دۆلەت قانۇنىدىن چىقماسلىق پىرىنسىپى بويىچە ئۆزىمىزنىڭ قانداقلىقى ھەققىدە چۈشەنچە بېرىۋاتىمىز، بەزىلىرىمىز باشقا بىر خىل ئۇسۇللاردىن پايدىلىنىپ ئۆزىمىزنىڭ ئويلىغانلىرىنى بىلدۈرۈۋاتىمىز.
       مېنىڭچە، بىز ئۆزىمىزنى باشقىلارغا تونۇتۇشتىن بۇرۇن ئۆزىمىز ھەققىدە ئويلىنىشىمىز كېرەك. بىز قانداق ئادەملەر؟ بىز ئۆزىمىزگە قانداق باھا بېرىمىز؟ بىزنىڭ ئارزۇ-خىيالىمىز نېمە؟ بىز نېمىگە ئىنتىلىمىز؟... مېنىڭچە بىز ئۆزىمىزنى ياخشى چۈشىنەلىسەك، ئاندىن باشقىلارغا ئۆزىمىزنى توغرا چۈشەندۈرەلەيمىز. خەق بىزنى توغرا چۈشەنمىدى دەپ قاقشىغاننىڭ نېمە پايدىسى، ئۆزىمىزنىڭ كىملىكى ھەققىدە تۈزۈك بىر ئويلىنىپ باقماي تۇرۇپلا؟!
      ئەمەلىيەت ھامان شەپقەتسىزدۇر، رىياللىق ئۆزىمىز ھەققىدە سوغۇققانلىق بىلەن ئويلىنىشقا مەجبۇر قىلماقتا. مېنىڭچە، بىز ئالدى بىلەن ئۆزىمىزگە توغرا باھا بېرەيلى، ئۆزىمىزنىڭ ياشام مىزانى بويىچە ياشايلى. ھەرگىزمۇ «مەن ئوغرى ئەمەس، قولۇم خەقنىڭ يانچۇقىدىن يىراق ئەمەس» دېگەندەك ياشىساق، دىلىمىز بىلەن تىلىمىز بىر بولمىسا، يەنە خەقتىن رەنجىشنىڭ زۆرۈرىيىتى بارمۇ؟ بىزدە مۇنداق بىر ئوخشىتىش بار: ئالتۇن ھامان ئالتۇن، تېزەك ھامان تېزەك. ئەگەر بىز ئۆزىمىزنى ئالتۇن دېسەك، ئالتۇنغا تۇشلۇق قەدىر-قىممىتىمىز بولسۇن، شۇندىلا ھەر قانداق چاغدا ھەقىقىي قىممىتىمىزنى ھەر ۋاقىت نامايان قىلىپ تۇرالايمىز.
       مۇشۇ تېمىنى مۇنبەرگە يوللاپ، ماڭا تەپەككۇر قىلىش پۇرسىتىنى بەرگىنىڭىزگە رەھمەت، قېرىندىشىم!
