                                              پۇتىن سەن خاتا قىلدىڭ
                                               
                                                        كامرانبەگ يوللانمىسى


           روسىيىنىڭ تۇنجى زوڭتۇڭى يېلىتسىن ھاكىمىيەت  بېشىدا تۇرىۋاتقاندا نۇرغۇن كىشىلەر يىلىتسىن «ھوقوقنى جېنىدىنمۇ ئارتۇق كۆرىدۇ »،ئۇنىڭ ھاكىمىيەتسىز تۇرمۇشىنى تەسەۋۋۇر قىلغىلى بولمايدۇ ،دىيىشكەنىدى .يېقىندا يېلىتسىن كومسومولكايا پىراۋدا گېزىتى مۇخبىرىنىڭ مەخسۇس زىيارىتىنى قۇبۇل قىلىپ ،ئاخبارات ساھاسىدىكىلەرگە ئۆزىنىڭ پېنسىيىگە چىققان بىر يىلدەك ۋاقىتتىن بۇيانقى خىزمەت ۋە خىزمەت ئەھۋالىنى ئاشكارلىدى ھەم ئۆز ۋەزىپىسىنى ئۆتكۈزىۋالغان پۇتىنغا ۋە ئۇ يۈرگۈزىۋاتقان سىياسەتكە باھا بەردى .
                                                   
                                      شۇنداق ،تۇرمۇش ئۆزگەردى   
           
       مەن خىزمىتىمدىن ئىستىپا بەرگەنگە پۇشايمان قىلمايمەن ،بۇ توغرا تاللاش بولدى . شۇنداق،تۇرمۇش پۈتۈنلەي ئۆزگەردى .ئۆز خىزمەت ئورنىدىن ئايرىلغان ھەربىر كىشى ، بۇلۇپمۇ دۆلەتنىڭ بىرىنچى نۇمۇرلۇق ئورنىدىن ئايرىلغان كىشى :داۋاملىق ئوتتۇرىغا چىقىپ نۇرغۇن كىشىلەردەك ئوتتۇرىغا چىقىپ تۇشمۇ تۇشتا دوكلات بېرەيمۇ ياكى ۋاقتىمنى ئۆزۈم ياخىشى كۆرىدىغان كىشىلەرگە بېرەيمۇ ؟ دېگەندەك تاللاشقا دۇچ كېلىدۇ .مەن ئىككىنچى يولنى تاللىۋېلىپ دەم ئالدىم ،قىزىمغا ھەم نەۋرىلىرىمگە ھەمراھ بولدۇم .
         ھازىر مەن كىچىك قىزىمنىڭ ئۆيىدىكىلەر بىلەن «گوركى 9- نۇمۇرلۇق » سارايدا تۇرىۋاتىمەن ، كېلەر يىلى باشقا بىر جايدىكى دۆلەتلىك داچا «بارۋىكخا 4-نۇمۇر »غا كۆچۈپ كەتمەكچى بولىۋاتىمىز   
         بەزى گېزىتلەر ھۆكۈمەتنىڭ بىر مىليون 400مىڭ دوللار خەجلەپ داچا سېلىپ بەرمەكچى بولىۋاتقانلىقىنى يېزىپتۇ ،بۇ پۈتۈنلەي تۇخۇمدىن تۈك ئۈندۈرگەنلىك ،چېرنومىلكىن :كىمكى ئۇنىڭ مىليونلىغان دوللىرىنى تېپىۋالسا ،شۇ ئېلىپ كەتسۇن دەپ ياخشى ئېيىتقان .
         بۇرۇنقى قىزىقىشىممۇ ئۆزگەرمىدى ،مەسىلەن ،ئوۋچىلىقتىن ۋاز كەچمىدىم .ئالدىنقى بىر مەزەگىلدە بىر ئوۋچىلىق مەيدانىغا بېرىپ ياۋا ئۆردەك ئېتىۋالدىم بۇرۇنقىغا ئوخشاشلا ،ھەر قېتىم ئوۋدىن يېنىپ
كەلگەندە كەيپىياتىم ئالاھىدە ياخشىلىنىپ قالىدۇ .
        تېلىۋىزورنى ھەمىشە كۆرۈپ تۇرىمەن .ئۆز ئارا سېلىشتۇرغاندا ،سىياسى تۈس ئالمىغان پىروگراممىلارنى تېخىمۇ ياخشى كۆرىمەن ،بۇلۇپمۇ پۇتبول ،ۋالىبول ،چويلا توپ ،بوكىس دېگەندەك تەنتەربىيە پىروگراممىلىرىنى كۆرۈشنى ياخشى كۆرىمەن .بۇيىل ئولىمپىك تەنتەربىيە مەزگىلىدە ئۇنىڭغا پۈتۈنلەي مەستانە بولۇپ كەتتىم .
        بىر مەھەل نەۋرەم بىلەن بىللە كومپىيوتېر  تورىنى ئۈگۈنىپ بېقىشنى ئويلىغانىدىم ،ئەمما، راس گەپنى قىلسام ،كومپىيوتېر كۆزنى بەك ھارغۇزۋېتىدىكەن ،سالامەتلىككە زىيانلىقكەن ،كومپىيوتېردا ئىشلىيەلمىگىدەمەن .مەن ئۆزۈمنىڭ تور بېتىنى قۇرماقچىمۇ ئەمەسمەن ،چۈنكى تور بېتىمنى قۇرۇپلا قالسام ،تورچىلارنىڭ سوئاللىرىغا كېچىنى –كۈندۈزگە ئۇلاپ جاۋاپ بەرسەم بولغۇدەك ،مەن بۇنىڭغا چىدىمايمەن .لېكىن يېلىتسىن فونىت جەمىيىتى نامىدا تور بېتى قۇرۇپ قالسام ،بۇباشقا گەپ.
                             
                                           پۇتىن سەن خاتا قىلدىڭ      
      
       مەن پۇتىن بىلەن ئېيىغا ئاز دىگەندىمۇ بىر –ئىككى قېتىم كۆرۈشۈپ تۇرىمەن ،ئىككىمىزنىڭ ئۆيى يىراق ئەمەس .بىزنىڭ سۆھبىتىمىزدىكى ئادەتتىكى پىرىنسىپى مۇنداق بولىدۇ : پۇتىن ئالدى بىلەن ماڭا ئۆزىنىڭ زىيارەتكە چىقىش ، كۆرۈشۈش ،سۆھبەتلىشىش نەتىجىسىنى تونۇشتۇرىدۇ ، مەن ئاڭلاپ بولغاندىن كېيىن ئۆزۈمنىڭ پىكىرىنى بايان قىلىپ ،ئۇنداق ياكى مۇنداق تەلەپلەرنى ئوتتۇرىغا قويىمەن .ئەمەلىيەتتە ماڭا شۇنىسى ئېنىقكى پۇتىن ئۆزىنىڭ پىلانى بويىچە ھەرىكەت قىلىدۇ ،ئىچكى مۇستەقىللىق ۋە دەل خوجايىنلىق دەل پۇتىننىڭ خۇسۇسىيىتى ،مەن دەل مۇشۇ ئالاھىدىلىكى ئارقىلىق .ئۇنى ئىزباسارلىققا تاللىغان ھازىر قارىسام مېنىڭ تاللىشىم توغرىكەن ،ئۇمېنىڭ ئىشەنچىمنى  يەردە قويمىدى  ،مەن پۇتىننىڭ خىزمەتتىكى خاتالىقىلىرىنى سەمىيمىيلىك بىلەمەن تۈزىتىپ بېرەلمىسەممۇ ،ئەمما ئارىمىزدا پىرىنسىپانلىق ئىختىلاپ يوق .
        «كۇرىسكى»ناملىق يادرو سۇ ئاستى پاراخوتى ھادىسىسىگە ئۇچرىغاندىن كېيىن ،مەن پۇتىنغا توغرىدىن –توغرىغىلا : ۋىلادىمىر ،سەن خاتا قىلدىڭ ،سەن سۇچىدىكى دەم ئېلىشنى توختۇتۇپ ،خەلىقنىڭ ئالدىغا چىقىپ ئەھۋالنى چۈشەندۈرمىدىڭ ، ھېسداشلىقىڭنى بىلدۈمىدىڭ ،شۇ چاغدىكى ئەھۋالدا كىشىلەر دۆلەت باشلىقىدىن قانداقتۇر تېخنىكىلىق تەكلىپ پىكىرلەرنى ئەمەس ،بەلكى ئىنسانى ھېسداشلىقىنى كۈتكەنىدى زۇڭتۇڭ بەخىتسىزلىككە ئۇچرىغان دېڭىزچىلارنىڭ بالا چاقىلىرىنىڭ جاراھەتلەنگەن قەلبىگە تەسەللى بېرىش ،جەمىيەتتىكى جىددى ۋەزىيەتنى يومۇشتۇرىشى كېرەك ئىدى . دېگەنىدىم .ئەممامەن بەزى تەشۋىقات ۋاستىلىرىنىڭ پۇتىننىبۇ ۋەقەنىڭ جاۋاپكارى دېگەنلىكىگە قوشۇلمايمەن .
                        
                   مەن سوۋىت ئىتپاقى دۆلەت شېئىرىنى ئىشلىتىشكە قارشى
   
              دۆلەت شېئىرى مەسىلسىدە ،مەن سابىق سوۋىت ئىتپاقى دۆلەت شېئىرىنىڭ مىلودىيىسىنى روسىيە دۆلەت شېئىرى قىلىپ بېكىتىشكە قەتتى قارشى تۇرىمەن .نۆۋەتتە ،روسىيە جەمىيىتىدە يەنىلا نۇرغۇن كىشىلەر، بولۇپمۇ ئاتا ـ بوۋىلار سابىق سوۋىت دۆلەت شېئىرىنى ياقتۇرىدۇ .ياق ،بۇچاقچاق قىلىدىغان ئىش ئەمەس .مەن شۇنىڭغا ئىشىنىمەنكى ،ياش ئەۋلادلار سابىق سوۋىت ئىتپاقى دۆلەت شېئىرىنى ياقتۇرمايدۇ ،چۈنكى
ئۇلار كەينىگە ئەمەس ئالدىغا قارايدۇ .چۈبەيىس بۇ مەسىلىدە ،زۇڭتۇڭ ئەل رايىن ىقارىغۇلارچە قەدىرلەپ كەتمەسلىكى ،بەلكى ئەل رايىغا پئال تەسىر كۆسىتىشى كېرەك ، دەپ ناھايىتى ياخشى ئېيىتقان .
     
               لېنىننى باشقا جايغا يۆتكەپ دەپنە قىلىش كېرەك
   
      لېنىننىڭ جەسىدىنى لېنىن مازولېيدىن يۆتكەپ چىقىپ كەڭ زىمىنغا تاپشۇرۇش ئەلۋەتتە قىلىشقا تىگىشلىك ئىش .
     مېنىڭ قاتتىق تەدبىر قوللۇنۇشۇمغا نېمە تەسىر بولدى ؟ ئاساسلىقى شۇچاغدىكى جەمىيەت تېخى بۇنىڭغا روھى ھازىرلىق كۆرۈپ  بولمىغانىدى .مەن يەنە جىددى ۋەزىيەت كەلتۈرۈپ چىقماسلىق ئۈچۈن قارار چىقىرىشنى قايتا قايتا كېچىكۈرۈپ قويدۇم . بۇلار بىركۈنى بولمىسا بىر كۈنى ھەل بولىدۇ .ناھايىتى ئەپسۇسكى ،مەن بۇ ۋەزىپىنى ئورۇنداشنى يېى زۇڭتۇڭغا ئىتتىرىپ قويدۇم .
     يېڭى يىل يېتىپ ككېلىش ئالدىدا تۇرىدۇ ، مەن روسىيىنىڭ  يېڭى بىر دەۋىردە تۇرىۋاتقانلىقىنى خوشاللىق بىلەن كۆردۈم ،كىشىلەر ئاخىرى دېمكۇراتىيە ،بازار ئىگىلىكى ئۋە ئەركىنلىك شارائىتىدا ياشاشقا ئادەتلەندى  .مەن خەلقىمگە ئىشىنىمەن .
