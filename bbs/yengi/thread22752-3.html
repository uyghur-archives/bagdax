يەتمەكچى بولغان نىشان تاق بولغاندا <بارلىقىنى ئىشقا سېلىش، دىققەتنى مەركەزلەشتۈرۈش > ياخشى ئۈنۈم بېرىدۇ. مەسىلەن جەڭ، ئۇرۇش. مەقسەت تاق ھەم ئېنىق، قارشى تەرەپنىڭ كۈچىنى يوقىتىپ غەلىبىنى قولغا كەلتۈرۈش. بۇ جەرياندا سىزنىڭ ئىشلەتكەن ئۇسۇلىڭىز ئانچە مۇھىم ئەمەس. 36 تەدبىرنىڭ ھەممىسىنى ئىشلەتسىڭىز بولىدۇ. ئالىي مەكتەپكە ئۆتۈش، تەتقىقاتتا نەتىجە يارىتىش ھەممىسىدە بارلىق زېھنىڭىزنى ئىشقا سىلىشىڭىز لازىم. لېكىن بىزلەر ئىنتايىن ئارزۇ قىلغان <روناق تېپىش، يۈكسىلىش، تەرەققىي قىلىش> لار سۆز ئاددىي بولغان بىلەن ئۆز ئىچىگە ئالغان مەزمۇن ئىنتايىن كۆپ. ئادالەتنى تۇرغۇزىمىز، باراۋەرلىكنى ئەمەلگە ئاشۇرىمىز، جىنايەتكە زەربە بېرىمىز، چىرىكلىشىشنى توسايمىز، ئاجىزلارنى قوغدايمىز..... ۋەھاۋاكازالار. بۇ نۇقتىلارنىڭ ھەممىسى بىزنىڭ ئارزۇلىرىمىزنىڭ مەزمۇنى، كەم بولسا بولمايدۇ، ھەم چوقۇم تەڭ ئېلىپ بېرىلىشى كېرەك. ئالدى بىلەن كۈچنى مەركەزلەشتۈرۈپ بىرىنچىسىنى قىلىلى، ئاندىن ئىككىنچىسىنى قىلىلى دېگىلى بولمايدۇ. يېڭى جۇڭگو قۇرۇلغاندىن كېيىنكى تۇتقان يول كۈچنى مەركەزلەشتۈرۈپ بىرنەچچە ئىشنى قىلىش بولدى.چوڭ سەكرىدۇق، ئاتوم بومبىسى ياسىدۇق. لېكىن خەلقىنىڭ تۇرمۇشىچۇ؟ بۇ خىل يولنى ئىسلاھات باشلانغاندا يەنە باستۇق. ئىقتىساد ئاشتى، مۇھىتچۇ؟ پارىخورلۇقچۇ؟ باي- نامرات پەرقىچۇ؟ توردا ئېيتىلغان سۆزلەر بار<تاجاۋۇزچىلىققا ئۇچرىغاندا دۆلىتىمىز يوق بولدى، لېكىن تاغ- دەريالىرىمىز بار ئىدى، ئەمدى دۆلىتىمىز بار، ئەكسىچە تاغ- دەريالىرىمىز يوق بولدى>. مانا ھازىر چاۋشيەن، ئىران خەلقىنىڭ ئېھتىياجى بىلەن كارى يوق يەنە كۆچنى مەركەزلەشتۈرۈپ قورال ياساش يولىدا مېڭىۋاتىدۇ. تەرەققىي قىلىش بىر جاپالىق، مۇرەككەپ جەريان، ئاددىي قىلىپ بىر جۈملە سۆز بەلەنلا شەرھلەپ بولغىلى بولمايدۇ. ئىللەت تۈزەلسىلا، ئىللەت بولمىسىلا قورساق تۇيىدىغان، كىيىم پۈتۈن بولىدىغان ئىش يوق. قورساققا قورساقنىڭ سەۋەبىنى، كىيىمگە كىيىمنىڭ سەۋەبىنى قىلىشىمىز كېرەك. ئىللەت بىلەن تەرەققىيات ئاددىيلا تارازىنىڭ ئىككى تەرىپى ئەمەس. ئىللەت بولمىسىلا تەرەققىي قىلىدىغان، يۈكسىلىدىغان ئىش يوق. ئەگەر ئوخشىتىشقا توغرا كەلسە تەرەققىيات ئۈچۈن ئاكتىپ ئامىل بىلەن پاسسىپ ئامىلنى ئارغامچىنىڭ ئىككى ئۇچىدىكى يۆلىنىشى قارشى كۈچكە ئوخشىتىشقا بولىدۇ. پاسسىپ كۈچ نۆل ھالەتتە ئاكتىپ كۈچمۇ نۆل بولسا تەرەققىياتنىڭ تىزلىنىشى نۆل، تىنچ ھازىرقى ھالەت ساقلىنىدۇ. ئەكسىچە پاسسىپ كۈچ نۆل بولمىسىمۇ، قىممىتى خېلى چوڭ بولسىمۇ، ئاكتىپ كۈچ ھامان پاسسىپ كۆچتىن چوڭ بولىدىكەن، بىزدە ئۈمىد بار، تەرەققىي قىلىشىمىز، يۈكسىلىشىمىز پەقەت ۋاقىت مەسىلىسى. بۇ ماقالىدە ھەم ئالدىنقى ماقالىدە تىلغان ئېلىنغان ياپۇنلاردىكى، يەھۇدىيلاردىكى بىزنىڭ ئۆلچىمىمىزدە سەت، ئىللەت، ناچار بىلىنگەن ئىشلار ئۇلارنىڭ ئاكتىپ كۈچلىرىنىڭ ئالدىدا ھېچنېمە ئەمەس. <قانداق قىلىش كېرەك؟> بۇنىڭغا بەك جىق زېھنىمىزنى سەرپ قىلىش لازىم ئەمەس. ھەر قېتىم كاللىڭىزغا بۇ سوئال كەلگەندە ئۇلاپلا<باشقىلار قانداق قىلىپتىكەن؟> دەپ سوئال سوراڭ. نېمە قىلىشنى بىلەلمىسەكمۇ باشقىلارنىڭ تەجرىبىلىرىدىن، بىلىملىرىدىن پايدىلىنىشنى، ئىزدەپ تېپىشنى ئۆگىنىۋالساق بولىدۇ. يېڭى ئىشلارغا يولۇققاندا باشقىلارنى دوراشتىن باشلىساق بولىدۇ. سەۋىيىمىز باشقىلاردىن ئاشقاندا، ياكى تەڭ بولغاندا ئاندىن ئىشەنچىمىز بىلەن باشقىلار مېڭىپ باقمىغان يېڭى يولغا تەۋەككۈل قىلساق بولىدۇ. <چەتئەلگە چىقىپ ئوقۇماقچى ئىدىم، قانداق قىلىشىم كېرەك؟>. چەتئەلدە ئوقۇغان، ئوقۇۋاتقانلاردىن مەسلىھەت سوراڭ. <ناسا دا خىزمەت قىلماقچى ئىدىم. قانداق قىلىمەن؟> ئەركىن سىدىق ئاكىدىن مەسلىھەت قىلىڭ. <بارلىقىمنى بىئولوگىيە تەتقىقاتىغا ئاتىماقچى ئىدىم>، ئەمىسە پەخىرلىك ئالىمىمىز شۆھرەت مۇتەللىپ ئاكىدىن مەسلىھەت سوراپ كۆرۈڭ. گەپ چەتنەپ كەتتى. ئەمدى ياپونلارغا كېلىلى.
سامۇرايلاردا بار بولغان ياخشى خىسلەتلەرنىڭ ئىچىدە مەن ئۈچۈن ئەڭ ئېسىلى <مەسئۇلىيەت، ئارنۇمۇس، شان شەرەپ>. مەسئۇلىيەت ئائىلىگە بولغان، جەمەتكە بولغان، ئاۋامغا بولغان، دۆلەتكە بولغان، پادىشاھقا بولغان، يۇقۇردىكىلەرگە بولغان مەسئۇلىيەتنى ئۆز ئىچىگە ئالغان بولىدۇ. بۇنىڭ ياخشى تەرىپى مەسىلە چىققاندا كىمنى ئىزدەش، كىمنى سۈرۈشتۈرۈش ئاسان بولىدۇ. بۇ يەردە ياپوننىڭ بىر كونا كىنوسىنى تەۋسىيە قىلىمەن <يەتتە سامۇراي>. باك كونا، بەك ئۇزۇن فىلىم بولغاچقا ئامېرىكا فىلىملىرىنى كۆرۈپ كۆنۈپ قالغانلارغا ئاخىرغىچە كۆرۈپ بولۇش تەس بولۇشى مۇمكىن. ئاساسىي مەزمۇنىنى دەپ بېرەي. يېزا دېھقانلىرى باندىتلارنىڭ زۇلۇمىغا ئۇچرايدۇ، زۇلۇمدىن قۇتۇلۇش ئۈچۈن سامۇرايلارغا مۇراجىئەت قىلىدۇ. 7 سامۇراي بىرلىشىپ، دېھقانلارنى يېتەكلەپ باندىتلارنى يوقىتىدۇ. بۇ يەردىكى ئېسىل ئىش دېھقانلار كىمگە مۇراجىئەت قىلىش، كىمنى ئىزدەشنى بىلىدۇ. ياپون جەمئىيىتىدە بۇ ئېنىق. ھەممىسى سامۇرايلارنىڭ ۋەزىپىسى، مەجبۇرىيىتى ھەم شان شەرىپى. تاجاۋۇزچىلار كەلدى، ئالدىن ئاتلانغانلار سامۇرايلار. سامۇرايلار باشقىلارنى ساقلاپ تۇرمايدۇ، ۋەزىيەتنى كۆزەتمەيدۇ. ئىش بولسا ئالدىغا ئېتىلىپ تۇرىدۇ. جەڭگە ئاتلىنىشتىن بۇرۇن باشقا مەدەنىيەت توپىدىكىلەر كۆپرەك جەڭدە زەپەر قۇچۇشنى تىلەيدۇ. لېكىن سامۇرايلار باتۇرلارچە ئۈلۈشنى تىلەيدۇ. چەتئەل كۈچلىرى بىرىنچى قېتىم ياپونغا قەدەم باسقاندا، جۇڭگوغا ئوخشاش ئۇلارنى ئۆلتۈرىدىغان ئىشلار بولغان. ئوخشىمايدىغىنى جۇڭگودا بۇنى سادىر قىلغانلار كۆپىنچىسى ئاۋام خەلق، ئۆلتۈرۈپ بولۇپ يوقاپ كېتىدۇ. ياپۇندا بۇنى قىلغانلار ھەممىسى سامۇرايلار، چەتئەل كۈچلىرى ئۆلۈم-يېتىمنى سەۋەب قىلىپ ياپون ھۆكۈمىتىگە، پادىشاھقا بىسىم سالغاندا، ئىشنى قىلغان سامۇرايلار چەتئەل ئەلچىخانىلىرىنىڭ ئالدىغا بېرىپ قارنىنى يېرىپ ئۆلۈپ بېرىپ خۇن تۆلىگەن. 19945 يىلى ئامېرىكا ياپون قۇرۇقلۇقىغا قەدەم بېسىپ، كىملەرنى سورتقا تارتىش، پادىشاھنى سوراق قىلىش كېرەكمۇ؟ دېگەنلەرنى ئويلاشقاندا نۇرغۇن ياپۇنلۇقلار، ئاياللارمۇ بار، ئامېرىكالىقلارنىڭ ئالدىغا بېرىپ ئۆزىنى ئۆلتۈرۈۋېلىپ پادىشاھنىڭ سوراققا تارتىلماسلىقىنى تىلىگەن. ئەمدى ئۆزىمىزگە نەزەر سالايلى؟ مەسىلە كۆرۈلسە؟ كىم بار؟ دەردىمىزنى كىمگە ئېيتىمىز؟ دېھقانلارنىڭ دەردىنى كىم ئاڭلايدۇ؟ كىم قۇلاق سالىدۇ؟ كىم ئاجىزلار ئۈچۈن سۆزلەيدۇ؟ كىم ھوقوقلىرىمىزنىڭ سورىغىنى قىلىدۇ؟ بىزدە، بىزنىڭ تەربىيىمىزدە ئاۋامنىڭ زارىغا داۋا بولۇشنى ئۆزىنىڭ بۇرچى قىلغان كىشىلەر تۇپى،ياكى تەبىقە يوق، شەكىللەنمىدى. ئاجىزلارغا ئىچ ئاغرىتىمىز، كۆڭۈل بۆلۈشكە تىرىشىمىز، لېكىن بىر قەدەم ئىلگىرىلەپ ئۇلارنىڭ ھوقۇقلىرىنىڭ سورىغىنى قىلىشقا كەلگەندە قەدەملىرىمىز ئاستىلاپ قالىدۇ. چىقىدىغانلار پەقەت ساناقلىق بىرنەچچەيلەن. بۇلارنى زىيالىيلار قىلىشى كېرەك دەيمىز، ئۇلارنىڭ قىلىشىنى ئۈمىد قىلىمىز، لېكىن تاپالمايمىز. چۈنكى زىيالىيلارنىڭ ئېنىقلىمىسى ئېنىق ئەمەس. بەزىلەر دەيدۇ، خەلق ئۈچۈن ھەقىقىي كۆيۈنگەنلەرنى زىيالىي دەيمىز دەپ. ئاڭلىماققا توغرا، لېكىن نەتىجىدە ھېچ كىمنى تاپالمايسىز، ھەممە زىيالىي بولۇشنى رەت قىلىدۇ.<خاپا بولماڭ مەن زىيالىي ئەمەس، ھەقىقىي زىيالىيلارنى ئىزدەڭ>، ئەڭ ئاخىرىدا قالغانلىرى گۈلەن، ئەركىن سىدىق، ئىلھام توختى، قانۇن يولى مۇشۇ بىرنەچچە ئادەم. بۇلار راست زىيالىي. زىيالىيلارنىڭ ئۈلگىسى بولۇشقا لايىق. لېكىن ھەرگىز ئۇلارنى ئۆلچەم قىلىشقا بولمايدۇ. ئۆلچەم قىلسىڭىز مۇشۇ بىرنەچچەيلەندىن باشقا ھېچكىمنى تاپالمايسىز. مەنچە بارلىق ئالىي مەكتەپلەرنى پۈتتۈرگەنلەر ھەممىسى زىيالىي. مىللەتنىڭ ھازىرقى ئەھۋالىغا، كەلگۈسىدە قانداق بولۇشىغا ئاساسلىق مۇشۇلار مەسئۇل. مۇشۇلارنىڭ مەسئۇلىيىتى. ئېتىز بۇيىدىكى دېھقاننىڭ مەسئۇلىيىتى ئەمەس. دېھقان ئېتىزدىكى زىرائەتكە مەسئۇل بولسا بولدى.
ھەممە سامۇراي ماددىي جەھەتتىن باياشات ئۆتكەن ئەمەس، غۇربەتچىلىكتە ئۆتكەنلەر نۇرغۇن. لېكىن مەسئۇلىيەت، ئارنۇمۇس، شان-شەرەپنى تاشلىغىنى يوق. شان شەرەپ توۋەن ئاۋامدىن، دۆلەتتىن، باشقا سامۇرايلارنىڭ مەدھىيىسىدىن كۈلىدۇ. بۇنداق شان –شەرەپ ھەممىگە كېلىۋەرمەيدۇ. سامۇراي بولغانلىق شان-شەرەپ ھەممىگە نېسىپ بولمايدۇ. ئاز بولغاچقا ئۇنىڭ رىغبەتلەندۈرۈش كۈچى زور بولىدۇ. ھەممە سامۇراي بولۇشنى تاما ياكى ئارزۇ قىلىدۇ. ئادەتتىكى كىشىلەر ئۈچۈن پەقەت جەڭدە باتۇرلۇق كۆرسىتىش بىلەن ئېرىشىش ئىمكانىيىتى بولىدۇ. پۇل-مال بىلەن سېتىۋالغىلى بولمايدۇ. سامۇرايلارمۇ سامۇرايلىق ئىززەت ھۆرمەتنىڭ كۇرسىنى بال بايلىق بىلەن چۈشۈرمەيدۇ. سامۇراي كىشىلەرگە بىر ئالدىغا قاراپ ئىنتىلىش ئىستىكىنى يارىتىپ باردى. خۇلاسىلەپ كۆرىلى. ئۆزىمىزدە يېقىن 500 يىلدا مۇشۇ نەرسىلەر كەم بولدىمۇ؟ يەنى ئاۋامغا باش بولىدىغان مەسئۇلىيەتچان تەبىقە ياكى كىشىلەر تۇپى، كىشىلەرنى ئالدىغا ئىلگىرىلەشكە ئۈندەيدىغان سەۋەبلەر،قىزىقتۇرۇشلار.

ياپۇنلارنىڭ تىلغا ئىلىشقا ئەرزىيدىغان يەنە بىر خىسلىتى ئۇلاردا بار بولغان ھۈنەرۋەنچىلىك روھى. بۇ روھ ياپۇنلارنىڭ مەسئۇلىيەتچان، ئىزدىنىش روھى بىلەن بىرىكىپ كەتكەن. مىسال ئالسام ئاسانلا چۈشىنىۋالالايسىز. بىر يىلى پۇرسەت بولۇپ جەنۇبنىڭ يېزا، ناھىيە بازارلىرىنى ئارىلايدىغان پۇرسەت بولدى. ياغاچ ساندۇق ياسايدىغانلارغا دىققەت قىلىپ باقتىم. ساندۇقنىڭ شەكلى، ماتېرىيال، ئىشلەش جەريانى دادام ئاپاملارنىڭ دەۋرىدىكى ياغاچ ساندۇق بىلەن ئوخشاش.،پەقەت نەقىش بىرئاز پەرقلىق. بىزنىڭ ئۆيدىكى ساندۇقتا ئۈچ يېتەرسىزلىك بار. تۇتقۇچ قول ئاغرىتىدۇ. نەقىشلەر قولنى ئىلىدۇ. ئىشلەتكەن ياغاچ يېرىلىپ كېتىدۇ. يېرىم ئەسىر كىيىن بۇ كەمچىلىكلەر يەنىلا بار. ياپونغا كېلىشتىن بۇرۇن بىرنەچچە كىچىك قەلەمتىراچ سېتىۋالدىم. سوغا قىلىش ئۈچۈن. يېڭىساردا ئىشلەنگىنىنى تاللىدىم. بىرنى سوغا قىلدىم. قالغىنىغا پۇرسەت بولمىدى. كىيىن قارىسام ھەممىسى قارىداپ كېتىپتۇ. بۇنىڭغا بەلكىم ياپوننىڭ نەم ھاۋاسى سەۋەب بولغان بۇلىشى مۇمكىن. ياغاچ قاپلىرى بۇزۇلۇپ كەتتى. قاپتىكى خەتلەر ئۆچۈپ كەتتى. ئىچىپ يېپىش ئانچە قۇلايلىق ئەمەس. مەن ئالغان قەلەمتىراچنى ياسىغان ھۈنەرۋەننىڭ ئۇستىسىمۇ بەلكىم يەنىلا مۇشۇنداق. بۇندىن باشقا خوتەننىڭ قولدا توقۇلغان ساپ يۇڭ گىلەم ئالدۇق. يۇمشاق، نۇسخىسى، رەڭگى ياخشى يارىدى. لېكىن يۇڭى چىقىدىكەن. ئويلاپ قالدىم بۇرۇننىڭ بۇرۇنىسىدا بىزنىڭ ئويدىمۇ شۇنداق كىيىملەرگە يۇڭى چاپلىشىدىغان گىلەم بار ئىدى. مەن ئۆزۈم تۇتۇپ باققان تۈركىيە، ئىران، ياپون گىلەملىرىدە بۇ مەسىلە يوق.

ماقالىدە تىلغا ئىلىنغان ياپۇنلارنىڭ قىلىچىنىڭ سۈپىتىدە مەسىلە يوق. بۇنىڭغا ئىسپات قىلىپ توردىن ئىزدەپ كۆرسەڭلار بولىدۇ. ياپون جۇڭگو ئۇرۇشىدا جۇڭگو تەرەپ كۆتۈرۈۋالغان چوڭ قىلىچلار ھەرگىز ياپوننىڭ ئىنچىكە سامۇراي قىلىچلىرىنىڭ خىلى ئەمەس. بىر چېپىشتا جۇڭگو ئەسكەرلىرىنىڭ قۇلىدىكى قىلىچلارنى پارە-پارە قىلىۋەتكەن ۋەقەلىكلەر كۆپ خاتىرىلەنگەن. ئىككىنچى قېتىملىق دۇنيا ئۇرۇشىدىمۇ ئارىلىق ئەڭ يىراق، ئوقنىڭ ئېغىشى ئەڭ كىچىك قورالنىمۇ ياپۇنلار ئىشلەتكەن. ھەتتا ھازىرمۇ ياۋروپادىكى نۇرغۇن چېركاۋلار ياپۇندا ئىشلەتكەن <داڭ>، قوڭغۇراقلارنى ئىشلىتىدىكەن. بۇمۇ تېلېۋىزوردا تونۇشتۇرۇلغان. ئادەتتىكى تۇرمۇش بۇيۇملىرىنى ساتىدىغان دۇكاندا بىر جۈپ ئىككى مىڭ ياپون يېنىغا سېتىلىدىغان چوكا كۆرگەنمەن. جۇڭگو پۇلىغا يۇزنەچچە سوم بولىدۇ. ساتىدىغىنى چوكا ئەمەس، ھۈنەردە. بۇنداق مىساللار بەك كۆپ. يىغىنچاقلىسىڭىز باشقا تەرەققىي قىلغان دۆلەتلەردىمۇ بۇ ئەنئەنە بار. ھۈنەر قىلىدىغانلار، كەسىپ قىلىدىغانلار توختىماي ئويلايدىغىنى قانداق قىلىپ ئۆز قۇلىدىن چىققان نەرسىلەرنىڭ سۈپىتىنى تېخىمۇ ياخشىلار. ئەگەردە پۇرسەت بار بولسا ئەتراپىڭلاردىكى، ياكى ئۇچرىغان ھۈنەرۋەنلەردىن سوراپ كۆرۈڭ. تېخىمۇ ياخشى قىلىش دېگەنلەرنى ئويلىشىپ بېقىپتىمۇ؟

ھەرقانداق ئىشنى ۋۇجۇدقا چىقىرىمىز دەيدىكەنمىز، باشقا شارائىتلار بىلەن ھېسابلاشمىغان ئەھۋالدىمۇ، چوقۇم ۋاقىت بىلەن زېھىن سەرپ قىلىمىز. ياپۇنغا كەلگەندە چوڭلار ئېيتقان <ياپۇندا ۋاقىت بەك تىز ئۈتىدۇ> دەپ. ئۇزۇن قالماي بۇنى ھېس قىلىشقا باشلىدۇق. بىر نەچچەيلەن بىر يەرگە كەلسەك دەردىمىز ئوخشاش، قانداقلا قىلساق ۋاقىت يەتمەيۋاتقان. بىرىمىز دەيمىز ئادەم نېمىشقا ئۇخلايدىغان بولۇپ يارىتىلغاندۇ؟ ئۇخلىمايدىغان، ياكى ئاز ئۇخلايدىغان بولساق ياخشى بۇلار ئىدى. بىرىمىز دەيمىز يەرشارى ئاستىراق ئايلانسا بولماسمۇ؟ مەنمۇ قوشۇپ قۇيىمەن بىر كۈن 48 سائەت ياكى 72 سائەت بولسا تېخىمۇ ياخشى بۇلار ئىدى دەپ. ھازىرقى دەۋردە ئىشتىن سىرت ۋاقىتلىرىدا، ئويناپ يۈرۈپ نەتىجە يارىتىدىغان كۈنلەر كەلمەسكە كەتتى. تۇتۇپ تۇرۇپ ئاسانلا نەتىجە قازانغانلارنى ئاڭلاپ قىلىش ئېھتىماللىق بار. لېكىن جىق ئەمەس، قولدا سانىغۇدەكلا. <تالات 99% تىرىشچانلىق بىلەن 1% ئىلھامنىڭ يىغىندىسى>. بۇ ھەممەيلەنگە تونۇش بولغان ئىدىسوننىڭ ئەقلىيە سۆزى. ئەپسۇن تەرجىمە تۇلۇق بولمىغان. كەينى <1% ئىلھام 99% تىرىشچانلىقتىن مۇھىم>. قىيىن مەسىلىگە دۇچ كەلگەندە ھەل قىلىش ئىلھامىنى تېپىش ئۈچۈن 99 % تىرىشىڭىز يەنى زېھنىڭىزنىڭ 99%، ۋاقتىڭىزنىڭ 99% نى سەراپ قىلىشىڭىز،باشقا ھەممە ئىشتىن سىرتقى ئىشلارغا ئايلىنىشى كېرەك. تاماق يېيىشىڭىز، ئۇخلىشىڭىز، ئائىلە، دوست بۇرادەرلەر، ئۇرۇق تۇغقانلار ھەممە يولنىڭ چېتىگە ئۆتۈپ تۇرۇشى كېرەك. تاشپولات ئاكا تونۇشتۇرغان ياپۇنلۇق قىزمۇ ئىلھامىنىڭ كېلىشىنى تىلغا ئالدى. شۆھرەت مۇتەللىپ ئاكىنىڭ 2 نەتىجىسىمۇ ئىلھام بولغانغا بارلىققا كەلگەن. بۇنىڭغا ئوخشاش ئىلھامغا مىسال نۇرغۇن نۇرغۇن. شۇ نەتىجە قازانغانلارنى زىيارەت قىلسىڭىز ئىلھام كېلىشنىڭ ئالدىدا تىرىشقانلىرىنى مەن يازغاندەك بايان قىلىدۇ. ياپون ئوقۇغۇچىلارمۇ مۇشۇنداق، نەتىجە چىقىشتىن بۇرۇن دەم ئىلىش، شەنبە، يەكشەنبە، قىزىل كۈن يوق. نەتىجە چىقىپ دوكلاتنى يېزىپ بولۇپ بىرەر كۈن دەم ئېلىپ، يەنە يېڭى نەتىجە ئۈچۈن ئوخشاشلا دەم ئېلىش يوق ھالەتتە تىرىشىدۇ. يۈكسىلىمىز دەيدىكەنمىز ياپۇنلاردىمۇ بەك تىرىشماي بولمايدۇ. ھەممە كۈندە 8 سائەت تىرىشقانغا مىللەت تەرەققىي قىلمايدۇ. 8سائەت تىرىشالمايدىغانلار بار، باشقىلارنىڭ تىرىشلىرىنى نۆل قىلىۋېتىدىغان ناچارلار بار. چوقۇم شارائىت بارلار چوقۇم كۆپرەك تىرىشىشى كېرەك. تېخىمۇ ئېنىق گەپنى قىلساق ئوقۇغانلار بەكرەك تىرىشىمىز كېرەك. بۇ مىللەتكە بولغان مەسئۇلىيەت بولۇپلا قالماي، ئېتىقادىمىز ئاللاھ قا بولغان جاۋابىمىز. چۈنكى بەرگەن نەرسىنىڭ ھەممىسىنىڭ سورىقى بولىدۇ. بۇ نەرسىلەر ماددىي نەرسىلا بولماستىن بىلىم، پۇرسەت، سالامەت، ئەقىل، تالات، ياخشى خىسلەت ھەممىنى ئۆز ئىچىگە ئالىدۇ. مىللەتنىڭ كەلگۈسىدە ھەممىنىڭ مەسئۇلىيىتى بار. لېكىن ئاساسلىقى شۇ مىللەتنىڭ سەرخىللىرى بەلگىلەيدۇ. سەرخىللەرنىڭ ئويلىغان، قىلغان ئىشىدىن، باسقان قەدىمىدىن كەلگۈسىگە ھۆكۈم قىلغىلى بولىدۇ.

ھىندونېزىيە بېرىپ ئالىي مەكتەپتە بىر مەزگىل دەرس سۆزلىگەن ئوقۇتقۇچىم <يۈكسىلىش بۇ دۆلەتتىن بەك يىراقتا> دەپ ئېيتقان. ئوقۇتقۇچىمنىڭ ئالغان بىر مىسالى، دەرس سۆزلەۋاتقاندا ئوقۇغۇچىلار توختاشنى تەلەپ قىلغان، سەۋەبى ناماز ۋاقتى بولغان. ياپۇندا بىر قېتىملىق دەرس 90 مىنۇت بولىدۇ. ئارىلىقتا ئوقۇغۇچىلار ھاجەتلەرگە چىقسا بولىدۇ، لېكىن توختاشنى تەلەپ قىلىش تەسەۋۋۇر قىلغۇسىز بىر ئىش. يېقىنقى 500 يىلنى ئويلايلى؟ زېھنىمىزنىڭ، ۋاقتىمىزنىڭ قانچە پىرسەنتى تەرەققىي قىلىشقا، يۈكسىلىشكە؟ قانچە پىرسەنتى دىن ۋە باشقا تالاش تارتىشلارغا سەرپ بولدى؟

2006-يىلى بىرىنچى قېتىم ياپونغا قەدەم باسقاندا تاشپولات ئاكا تىلغا ئالغان ياپۇنلۇق باينى كۆرۈش پۇرسىتى بولغان. شۇ چاغدا كۆرگەندىمۇ بۇ باي رەسىمدىكىدەك كۆڭلەك كىيىۋېلىپتىكەن. ئوقۇتقۇچىمدىن . ئۇ كىشىنىڭ تەرىپىنى ئاڭلاپ شۇنداق ھۆرمىتىم بولغان. تەپسىلاتلار جىق ئىدى، مەن پەقەت بۇ تېما بىلەن مۇناسىۋەتلىك بىر ئىشنى مىسال قىلاي. ئۇ باي بىلەن كۆرۈشكەندەك قېشىدا ئۇكىسىمۇ بار ئىدى. بىرنەچچە كۈن بىزگە ھەمراھ بولدى. ئۇستازىمدىن سورۇدۇم <ئىنىسى نېمە ئىش قىلىدۇ؟> . <ئاكىسىغا ئىشلەيدۇ>. ھەيران بولدۇم. دادىسىدىن ئىنىسىغا بايلىق ئايرىلمىغانما؟. ياپونلاردا ئائىلىگە چوڭ ئوغۇل ۋارىسلىق قىلىدۇ. باشقىلارغا ھېچنېمە يوق. (ھازىر ئانچە مۇنچە ئۆزگىرىش بولغان بولۇشى مۇمكىن). ئاڭلىغاندا بىرئاز رەھىمسىز بىلىندى. لېكىن بۇنىڭ ياخشى تەرەپلىرىمۇ بارىكەن. بىرى بايلىق چېچىلمايدۇ، چېچىلىپ يوقاپ كېتىشنىڭ ئالدىنى ئالغىلى بولىدۇ. بايلىق بولىدىكەن ئىش قىلىشقا پۇرسەتلەر بولىدۇ. بىر تۇغقىنىمىزنىڭ ئۆيى چىقىلغاندا نۇرغۇن تۆلەم پۇل كەلدى. بىر ئوغۇل مەسلىھەت قىلدى. بۇ پۇلنى دەسمايە قىلىپ بىر ئىش قىلىلى. باشقا قېرىنداشلار قوشۇلمىدى. ئوغۇل يەنە دېدى مەن پۇلنى قەرز ئالغان بۇلاي كىيىن ھەممىڭلارغا ئاشۇرۇپ قايتۇرار. يەنە قوشۇلمىدى. 10 بالا پۇلنى تەڭ بولۇشتى. بۆلۈشكەن پۇل ئۇلارنىڭ بىر ئېھتىياجىدىن چىققان بولسىمۇ، تەقدىرىنى ئۆزگەرتىشكە قادىر بولمىدى. بۇرۇن قانداق بولغان بولسا ھازىرمۇ شۇنداق. بۇندىن كىيىن ئۇلارغا بۇنداق پۇرسەت يوق. ياۋروپامۇ بۇ ئەنئەنىنى ساقلىغان. ياپون قۇدرەت تاپتى، ياۋروپا يۈكسەلدى. بۇنىڭدا ئۇلار قوللانغان مىراس بولۇش ئۇسۇلىنىڭ ئاكتىپ روللىرى بارمىكىن دەپ ئويلايمەن. يەنە بىر تەرىپى باشقا باللارغا مىراس بولمىغاچقا چوقۇم بەك تىرىشىپ ئۆزىنىڭ ئىگىلىكىنى تىكلىشى لازىم. بەك ئوقۇشى كېرەك، جەڭلەردە باتۇرلۇق كۆرسىتىپ سامۇراي بولۇشى كېرەك. زېمىن تېپىشى كېرەك. دادامنىڭ بىز كىچىك ۋاقتىمىزدا دەيدىغان گېپى <كۆمۈرلۈكتىن چىقىپ كىتىڭلار>. بالىلار ئۈچۈن خېلى تىرىشتى لېكىن 3 ئاكام چۆگلەپ-چۆگلەپ يەنە كۆمۈرلۈككە قايتىپ كەلدى. ئاسپىرانتلىق ئوقۇشۇم تۈگىگەندە ئاكىلىرىم كۆمۈرلۈككە قايتىشنى تەۋسىيە قىلدى، قىزىقتۇردى. نېفىتلىكلەر بىلەن ئوخشاش بولغان كەلگۈسىنى تەسۋىرلەپ بەرگەن. لېكىن دادام كەسكىن رەت قىلغان. رازى بولمايمەن دەپ گەپنىڭ ئېغىرىنى قىلغان. ئەتراپىمدا 3ئەۋلاد كۆمۈرلۈكتە خىزمەت قىلغانلار بار. باشقا يۆلىنىشتە تىرىشىپ بېقىشنى ئويلىغانلار ساناقلىق. ئاخىرى كىتابلىرىمىزدا ئوقۇغان، ئاڭلىق دادىلار تىلغا ئالغان ئىشلار يۈز بېرىشكە باشلىدى. كۆمۈر تۈگىدى، كۆمۈرلۈك تاقالدى. يېشى چوڭلار دەم ئېلىشقا چىقتى، ياشلار ئائىلىدىن يىراق كۆمۈرلۈكلەرگە تارقىلىپ كەتتى. كېيىنكىلەر بىرىنچى قېتىم ئەمدى نېمە قىلىمىز؟ دېگەننى ئويلاشقا باشلىدى. ئاسمان ئۆرۈلمىدى، يوللار ئېتىلگىنى يوق، رىزىق تۈگىگىنى يوق. ھەممىسى يول تاپتى، بەزىلەر شوپۇر بولدى، بەزىلەر رېمونتچى بولدى، بەزىلەر ھۈنەر قىلدى. بۇ خىل قىسمەت كۆمۈرلۈكتىلا ئەمەس، نېفىتىمۇ بار، زاۋۇتلاردىمۇ بار، ھەتتا مەمۇرلاردىمۇ بار. گۈلەن يازغان ھۈنەر قىلىلى، ئەركىن سىدىق ئاكىمۇ يەھۇدىيلارنى تونۇشتۇرغاندىمۇ جاپا، كرىزىسنىڭ ئاكتىپ رولىنى تىلغا ئالغان. شۇ سانجىشلار بولمىسا مەن ياپونغا كەلمەس بولغىيتتىم، باشقىلار باشقا قەدەملەرنى باسماس بولغىيتتى.
دوكتورلۇق ئوقۇش ۋاقتىمدا مەكتەپتە بىر توپ ياپۇنلۇق قېرىلاردىن تەشكىللەنگەن چەتئەللىك ئوقۇغۇچىلارغا خالىسانە ياپون تىلى، ياپون مەدەنىيىتى ئۆگىتىدىغان كۇرس بار ئىدى. بىر بوۋاي سۆز قىلغان <ياش ۋاقتىمدا ئالدىراشچىلىقتىن بىرئۆيدەتۇرۇم بىر نەچچە ھەپتە بالىلىرىمنى كۆرمىگەن ۋاقىتلار بولغان>، ئەتىگىنى بالىلاردىن بۇرۇن قوپۇپ ئىشقا ماڭىدۇ، كەچتە بالىلار ئۇخلاپ بولغاندىن كىيىن قايتىپ كۈلىدۇ. كوچىلاردا بالىلارنى ئوينىتىۋاتقانلار ھەممىسى ئانىلار، دادىلار قېنى؟ مۇنچە ئالدىراش بولۇپ نېمىگە ئېرىشمەكچى؟ ھۇزۇرلىنىشنى بىلمەي ئۇنچە ئۇزاق ئۆمۈر كۆرگەننىڭ نېمە ئەھمىيىتى؟ ئوخشىمىغان ياپونلاردا مەن سورىغان. مېڭىسى يۇيۇلۇپ كەتكەندەك ھەممىسىنىڭ جاۋابى ئوخشاش. <بىرنېمىنى بىرنېمە قىلىمىز دەپ ئەمەس، بۇنداق قىلمىساق دۇنيادا ياپۇنلار مەۋجۇت بولۇپ تۇرالمايدۇ>. چۈشەنگەندەك بولدۇم. ياپۇنلارنىڭ ھازىرقى تەرەققىياتى <مەۋجۇتلۇقىمىز ئۈچۈن> دېگەن ئاساسلىق نىشاننىڭ قوشۇمچە مەھسۇلاتى ئىكەن. باشقا ئىقلىمدىكى قېرىلارغا ئوخشاش ياپون قېرىلىرى ياشلاردىن رازى ئەمەس. قېرىلاردا بۇنىڭغا دەسمايە بار. مېيجى يىلىنى ئەسلەڭ، ئۇرۇشتىن كېيىنكى ئەھۋالىنى ئەسلەڭ. قېرىلارنىڭ تىرىشچانلىقىدا ئامېرىكادىن ئېشىپ كېتىشكە يېقىنلاشقان. لېكىن ياشلار جەمئىيەتنى قولغا ئالغاندىن كىيىن داۋاملاشتۇرالمىدى، يىقىلىپ كەتتى. ياشلار قېرىلارنى رازى قىلالمايدۇ. لېكىن ياپون جەمئىيىتى ئاستا بولسىمۇ يەنىلا ئالدىغا قاراپ تەرەققىي قىلىۋاتىدۇ. بىر زاۋۇتتا، شىركەتتە ئاستىدىكى ئىشچىلار ئىچىدە كۈندە پەقەت 8 سائەت ئىشلەشنى ئارزۇ قىلىدىغانلار بار بولسا چوڭ گەپ ئەمەس، لېكىن يۇقىرىدىكى، رەھبىرى ئورۇندىكىلەر چوقۇم كۆپ خىزمەت قىلىشى لازىم. ئەگەر ئۇلارمۇ 8 سائەت ئىشلىسە زاۋۇت، شىركەت تاقىلىدۇ. ياپون جەمئىيىتىدە پەقەت 8 سائەت ئىشلەشنى مەقسەت قىلغان ئادەم يۇقۇرغا ئۆرلىيەلمەيدۇ. قېرىلارمۇ بۇلارنى ئۆرلىگىلى قويمايدۇ. پەقەت ياپونلارنىڭ روھىغا، مەنپەئىتىگە ۋارىسلىق قىلغان ھەم بۇ روھنى داۋاملاشتۇرالايدىغان ياشلارلا مۇھىم ئورۇنلاردا خىزمەت قىلالايدۇ. ياپۇنلار نەتىجە ياراتقانلارنى ھۆرمەت قىلىدۇ. تەقدىرلەيدۇ. ياپون روھىنى ئەكس ئەتتۈرگەنلەرنىمۇ بەك ھۆرمەت قىلىدۇ. ئوقۇتقۇچىم ياپونلار ئاددىي ئىشلارنى خاتالىق چىقارماي داۋاملاشتۇرالىغانلارنىمۇ بەك ھۆرمەت قىلىدۇ دەپ بىر پوچتالىيوننىڭ ھېكايىسىنى دەپ بەرگەن. بۇ پوچتىكەش 25 يىل خىزمەت جەريانىدا بىر قېتىم خاتالىق ئۆتكۈزگىنى يوق، بىر قېتىم خىزمەتكە چىقمىغىنى يوق، رۇخسەت سورىغىنى يوق. ئاخىرى دەم ئىلىشقا چىققاندا ياپون پادىشاھىنىڭ قوبۇل قىلىپ تەقدىرلىشىگە مۇيەسسەر بولدى. ئەكس ئەتكىنى ياپونلار چوقۇنىدىغان مەسئۇلىيەت روھى!

90 ياشتىن ئاشقان بوۋام بار. بىر ئۆمۈر تەقۋادار ياشىدى. ھازىر سالامەتلىكى نامازنى، روزىنى ئادا قىلىشقا يار بەرمەيدۇ. لېكىن يەنىلا داۋاملاشتۇرۇۋاتىدۇ. بوۋام بىلىدۇ، سانىنى ئازايتسا، ياكى ئادا قىلمىسىمۇ گۇناھ بولمايدىغانلىقىنى، لېكىن بىر ئۆمۈر داۋاملىشىپ قېنى قېنىغا، ھۈجەيرىسىگە، بەدىنىدىكى ھەر بىر مالىكولارغا سىڭىپ كەتكەن تەقۋادارلىق، ئېتىقاد، ئادەت بوۋامنى ئۈندەيدۇ، جىم تۇرغۇزمايدۇ ئۇخلاتمايدۇ. بىر قېتىم ئادا قىلالمىسا شۇنچىلىك ئازابلىنىدۇ، مۇشۇ بىر قېتىملىقتىن جاۋاب قىلالماسلىقتىن ئەنسىرەيدۇ. تارىخقا نەزەر سالساقمۇ نۇرغۇن يۈكسەك تەرەققىي قىلغان مەدەنىيەتلەر ھەممىسى ئازدىن كۆپتىن بولسۇن ئېتىقادىنىڭ مەھسۇلى، ئىلاھلىرىغا بولغان مەدھىيىسىنىڭ مەھسۇلى. گىرېتسىيە، رىم ھەيكەلتىراشلىق سەنئىتى ئارقىلىق ئىلاھقا چوقۇنۇشىنى ئىپادىلىدى. مىسىر، مايا چوڭ قۇرۇلۇشلارنى بارلىققا كەلتۈرۈپ ئىلاھلىرىغا سەجدە قىلدى. .. ياپۇنلار مەۋجۇتلۇقى ئۈچۈن تىرىشتى. بىزلەرمۇ، مۇسۇلمانلارمۇ بىر كۈنى ئەركىن پىكىر قىلىش، توختىماي تىرىشىش، ئىلىم ئۆگىنىش، تەرەققىي قىلىش، يۈكسىلىشنى ناماز، روزىغا ئوخشاش ئېتىقادىمىز دەرىجىسىگە كۆتۈرۈپ، تەقۋادار بوۋامدەك بىر كۈن بىر قېتىم پىكىر قىلالمىغىنىمىزدىن، ئىلىم ئۈگۈنۈلمىگەنلىكىمىزدىن، تىرىشالمىغانلىقىمىزدىن ناماز، روزىنى ئادا قىلالمىغاندەك ئازابلانغان، ياراتقۇچىنىڭ شاپائىتىگە جاۋاب قايتۇرالمىغاندەك ئازابلانغان كۈننىڭ ئۆگۈنىسى بىزنىڭ، مۇسۇلمانلارنىڭ قايتا يۈكسەلگەن كۈنىمىز.
