                                          جۇڭگوچىلىقتا ھاقارەتلەشنىڭ بەدىلى نىمە ؟

                                                               توخوتۇر


     1. مەن دەبدەبىلىك ئەسئەسىلىك قانۇن نىزاملاردىكى ھاقارەتلەش جىنايىتىگە قانداق جازا بىرىلدىغانلىقىنى ئۇقمايمەن. پەقەت قانۇنغا يېزىپ قويۇلغان بىلەن باشقىلار تەرىپىدىن ھوقۇقىمىز دەپسەندە بولۇپ ھاقارەتلەنسەك ياكى تىللانساق يا ساقچىنى يا سوتنى ئىزدىگەننىڭ پايدىسى يوقلىقىنى ئوبدان بىلىمەن . ئىزدىسىڭىزمۇ شۇ ئۇنداق بۇنداق دەپ ئىشنى تۇگۇتىدۇ ، بۇ تېخى ھاقارەتكە ئۇچۇرغۇچىغا ئىككىنچى قېتىملىق ھاقارەت ، ھاقارەتلەپ تىللىگۇچىىگە ئىلھام بولۇشى مۇمكىن .

    2. ئاشۇ بىر ئۇچۇم بەدبەخىت باشقىلارنى ھاقارەتلەپ تىللاپ ئۆز كۆڭلىنى شات ئىتىدىغانلارغا شۇ نەرسە ئايانكى بۇ ئىشلارنى قانۇن سوراپ كەتمەيدۇ ، بەك تولا بولسا  شۇ نەسىھەت ياكى ئانچە مۇنچە جەرىمانە تۆلەپ  قويىدۇ .

    3. شۇنداق بەزىدە باشقىلارنى تىللاپ ھاقارەتلەپ كۆنۇپ قالغانلار ئۆزىدىنمۇ يامانلارغا ئۇچراپمۇ قالىدۇ ، بۇ چاغدا ئىش يۇغاناپ يىغىشتۇرۋالاغۇسىز  ئېچىنىشلىق ھالەتكە قالىدۇ .  مانا شۇ چاغدا ساقچىلار ئارلاشسا كېرەك .ئەپسۇس

    4.ئەمما  ساقچىخانىلار ئادەتتە ما ئادەم مېنى ھاقارەت قىلدى ،ماۋۇ ئادەم مېنى تىللدى دەپ ئەرىز قىلىپ بارسىڭىز ، ۋاي ئۇكام تىللىغانغا نىمە بولاتتى ، بولدى سەنمۇ تىللاپلا ۋال دىگەندەك  پاراڭلارنى سەپ قويۇپ ئولتۇردىكەن . ئشىقىلىپ بۇ ئشىنىڭ ھەل قىلىنىشى ئازراقمۇ كۆڭۇلدىكىدەك ئەمەس .

    5. مەن دائىم ئويلاپ قالىمەن ، قانۇنىمىز نىمىشقا ئۇرۇش جىدەلنىڭ پىلتىسى بولغان باشقىلارنى ھاقارەتلەش باشقىلارنى تىللاشقا ئېرەڭشىمەيدىغاندۇ . ناۋادا  ئاشۇ باشقىلارنى ھاقارەتلىگەن باشقىلارنى تىللاش خۇمۇرى بارلارغا ناھايىتى ياخشى قانۇنى جازا بىرىلسە ، شۇنداق جازا بىرىلىشى كېرەككى ئاشۇ ھاقارەت قىلغۇچىلار ئىككىنچى قېتىم باشقىلارنى ھاقارەت قىلىشقا پېتىنالمايدىغان ھالەتتە بولسا ئاندىن  قانۇن جايىنى تاپقان بولاتتى . ھازىر خەلق ئارىسدا شۇنداق بىر ئورتاق قاراش باركى ھە بۇ تىللاش ھاقارەتلەش دىگەننى قانۇن سورىمايدۇ ، قانۇن سورىمىسا قانداق قىلىش كېرەك ، ئۆزىمىز ھەل قىلۋېلىشىمىز كېرەكمۇ ، ئۆزىمىز ھەل قىلىپ بىر ئېچىنىشلىق دەرىجىگە بارغاندا ئاندىن قانۇن بۇ ئىشنى سورامدۇ  ؟

    6. جۇڭگوچىلىقتا موشۇ ئەمەلدارلار نىڭ تۆۋەندىكىلەرنى ھاقارەتلىشى تىللىشىمۇ ناھايىتى نورما ل بولۇپ قالغاندەك ئەھۋاللارمۇ بار ، بولۇپمۇ دۆلەت ئىگىلىكىدىكى چوڭ كارخانىلاردىكى ھەر دەرىجىلىك باشلىقلار  ئۆز بەگ ئۆز خان ھالەتتە بولۇپ قول ئاستىدىكىلەرنى ھاقارەت قىلىشنى كارخانا باشقۇرۇشنىڭ بىر تۇرلۇك ئەپچىل ئۇسۇلى قىلۋالىدىغان ھالەت ساقلانغان ، بۇ خىل ناچار خاھىش شىنجاڭ ئۇيغۇر ئاپتۇنۇم رايونىدىكى دۆلەت كارخانىلىرىدا ھەرمىللەت ئاممىسى ئىشلەيدىغان  شارائىتلاردىمۇ ئۆزلۇكسىز يۇز بەرمەكتە .

    7.ئۇمۈدىم  دۆلىتىمىز قانۇنچىلىق ئىشلىرىنىڭ مۇكەممەللىشىشىنى پۇقرارلارنىڭ ھوقۇقىنى تېخىمۇ مۇكەممەل قوغدىشىنى ئۇمۇت قىلىش بولۇپمۇ موشۇنداق ھەر خىل سورۇنلاردا باشقىلارنى ھاقارەتلەيدىغان تىللەيدىغانلار غا قارىتا كارغا كەلگۇدەك قانۇن نىزام بەلگىلىمە چىقىرىپ كەڭ خەلق ئاممىسنىڭ ھاقارەتلىگەننى تىللىغاننى ساقچى قانۇن سۇرىمايدىكەن دىگەن  مۇدھىش قارىشىنى ئاغدۇرۇپ تاشلىشىنى ئۇمۇت قىلىمەن . بۇ جەميەتنىڭ مۇقۇملىقىنى يەنىمۇ ئىلگىرلىگەن ھالدا قوغداپ ، جىدەل ماجرالارنى بىخ ھالتىدە بېسىقتۇرۇشقا پايدىلىقمىكىن دەپ ئويلايمەن .

    8.بۇ ئەھۋال شىنجاڭ ئۇيغۇر ئاپتۇنۇم رايۇنىنىڭ دۆلەت كارخانىلىرىدىمۇ ھاقارەت ۋە تىللاشلار گەۋدىلىك بولغاچقا كارخانىلاردىكى تىللاش ۋە ھاقارەتلەشلەرگە قارىتا يەنە ئايرىم قانۇن نىزام تۇرغۇزۇپ شىنجاڭ ئۇيغۇر ئاپتۇنۇم رايۇنى تەۋەسىدىكى دۆلەت كارخانىلىرىدا ئشىلەۋاتقان ھەرمىللەت ئىشچىلىرنىڭ جۇملىدىن ئۇيغۇر ئىشچىلىرىنىڭ قانۇن ھوقۇق مەنپەئەتى ، غۇرۇرى  ، كشىىلىك قەدىر قىممىتىگە ھەر ۋاقىت كاپالەتلىك قىلالايدىغان بولۇشىنى چىن دىلىمدىن ئۇمۇت قىلىمەن .

قانۇن-تۈزۈم ياۋرۇپا دۆلەتلىرىدىكىدەك مۇكەممەل بولغاندا ئاندىن تۈگەرمىكى بۇ ئەھۋاللار.ساقچىلارنىڭ قاراپ تۇرۇپ: تىللىغانغا نىمە بولاتتى ،ئۇ تىللىسا سەنمۇ تىللا دىگىنى خۇددى ئت كىلپ چشلىۋالسا سەنمۇ چشلە دىگەندەك ئش  بوپتۇ.منىڭ خەنچە بىر ژۇرنالدىن ئوقىغان،ئامىرىكىدا بولغان كىچىككىنە بىر ۋەقەنى قىستۇرۇپ قويغۇم كەلدى.بىر جۇڭگۇلۇق ئايال بەش ياشلارغا كىرگەن بالسىنى ئېلىپ بىر نەچچە يىلدىن بۇيان ئامىرىكىدا ئشلەۋاتقان ئىرىنى يوقلىماقچى بۇلۇپ يولغا چىقىپتۇ،ئايرۇپلاندىن چۇشۇپ ئايرۇدۇرۇمدا ئىرىنى ساقلاپ تۇرۇپتۇ،شۇ ئارلقتىا ھلقى بەش ياشلىق بالسى ھەرتەرەپكە يۇگۇرۇپ ئويناپتۇ،بالسنىڭ قالايمىقان مىڭىپ يۇرسە يۇقاپ كىتشىدىن ئەنسىرىگەن ئايال بالسىغا تەنبىھ بېرىپ جازالىماقچى بوپتۇۋە بالنىڭ قايتا كەپسىزلىك قىلماسلقى ئۈچۈن بالسنىڭ كاسسغا ئۇرۇپ ئەدەپلەپتۇ.شۇ ئارىدا ئايرۇپورىت ساقچىسى پەيدا بوپ ئۇ ئايالغا ئنگلىسچە :بالنى ئۇرسا قانۇنغا خىلاپ قلمش ئىكەنلىكىنى شۇڭا 300دوللار جەرىمانە قويدىغانلىقىنى دەپتۇ،بىراق ئۇ ئايال تىل بىلمىگەپ ھچنىمنى چۈشەنمەپتۇ،شۇ ئارىدا يولدىشى يتىپ كەپتۇ ۋە بولغان ئەھۋالدىن خەۋەر تىپىپ ساقچىغا ئايالىنىڭ ئامىركىغا تخى ئەمىدىلا كەلگنىنى بۇ دۆلەتنىڭ قانۇن تۈزۈمنى پەقەت بىلمەيدىغانلقىنى كەڭچىلك قلشىنى ئۆتۈنۈپ دەپتۇ،ئۇ ساقچى ئەمىلى ئەھۋالنى بىلگەندىن كىين بۇندىن كىينى بۇنداق ئەھۋالنى سادىر قلماسلىقنى ئېيتپ كىتىپ قاپتۇ،ئاندىن ھلقى ئايال ئىرى ئارقىلق ئشنىڭ تىگىگە يتىپتۇ،لىكىن يەنىلا كاللسدىن ئوتمەپتۇ،نىمشقا ئۆز بالامنى ئۇرساممۇ جازاغا ئۇچرايمەن؟دەپ تۇرۋاپتۇ.ئىرى چۇشەندۇرۇش ئارقىلىق ئاران قايىل قىپتۇ.خۇددى مۇشۇ ۋەقەدىكىگە ئوخشاش تەرەققى قىلغان ئەللەرنىڭ ھەممە نەرسسى تەڭ تەرەققى قىلدىغان گەپكەن.
