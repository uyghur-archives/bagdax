يەنە بىر جەھەتتىن ئىقتسادى بىسىم سەۋەبىدىنمۇ كىتاپ سىتۋىلىپ ئۇقۇيدىغانلار ئازلاپ كەتكەن . ..............................

بىر چەتئەللىك بولسىڭىزمۇ ئۇيغۇرۇم ھەققىدىكى بۇ تىمىڭىزنىڭ چىنلىقىدىن ،بىزدە بار بولغان بىر قىسىم يىتەرسىزلىكەرنى يورۇتۇپ بەرگىنىڭزدىن ،ئاناتىلىمىزدا شۇنچە راۋان ،شۇنچە توغرا ئەسەريېزىپ بىزگە سۇنغىنىڭزدىن سۆيۈندۈم . سىزنىڭ بۇنىڭدىن كىيىنمۇ بۇ خىل ئىزدىنىشلىرىڭىزنىڭ توختاپ قالماسلىقىنى ئۈمىد قىلمەن .

   ئەمدى يازمىڭىزدىكى بىر نەچچە تەرەپكە قارىتا ئۆز پىكىرىمنى بېرىپ ئۆتمەكچىمەن .
   بىرى ، «ئۇيغۇرلار ئۇيغۇر تلىنى پۇختا ئۈگەنمەيدىكەن » بۇ پىكىرىڭىزنى مەنمۇ قوللايمەن ،لىكىن بۇنى ئومۇمىيلىق دەرىجىسگە كۆتۈرىۋىتىشكە بولمايدۇ ،چۈنكى ئارىمىزدا ھازىر ئاناتىلنىڭ موھىملىقىنى قايتىدىن تونۇپ يېتىۋاتقان ،شارائىت ھەر قانچە ناچار بولسىمۇ ، ئامال قىلىپ ئاناتىلىمىزنى ئۆگىنىۋاتقان ۋە ئۆگىتىۋاتقانلار كۆپىيىۋاتىدۇ ،ھازىرقىدەك بىر موھىتتا بىزنىڭ يەنىمۇ تىرىشىشىمىزغا توغرا كېلىدۇ ،بۇ ھەقتە يازغان ئىكەنسىز ،يەنىمۇ چوڭقۇرلاپ ئاناتىلنى پۇختا ئۈگەنمەسلىكتىكى سەۋەپ ھەققىدىمۇ ئىزدەنگەن بولسىڭىز كۆپ نەرسىلەرنى چۈشەنگەن بولاتتىڭىز ...

   يەنەبىرى ،كۈتۈپخانا بىلەن كىتاپخانىنىڭ پەرقىنى بىلەلمەسلىك ۋە كىتاپ ئوقۇشقا قىزىقماسلىق مەسىلىسى ،بۇ مەسىلىمۇ ھەقىقەتەن بىزدە ماۋجۇت ،كىتابسىز ئۆي جانسىز تەن ،دېگەندەك  ئەسلى ھەر بىر ئۇيغۇر ئۆيىدە بىردىن كىتاپ ئىشكابى بولىشى شۇنداقلا بىھۇدە ئۆتۈپ كىتىۋاتقان ۋاقتىمىزنى كىتاپ ئوقۇشقا سەرپ قىلىشىمىز كىرەك ئىدى ،لىكىن ئۇنداق بولماي قالدى ، بۇمۇ ئوخشاشلا ئايرىم ساندا . ئاللا خالىسا بۇنىڭدىن كىيىن خەلقىمىز چوقۇم كىتاپنىڭ ،ئوقۇشنىڭ ئۆزىلىرىنى قۇتقۇزىدىغان موھىم ئامىل ئىكەنلىگىنى تونۇپ يېتەلەيدۇ .
  كۈتۈپخانا بىلەن كىتاپخانا مەسىلىسىگە كەلسەك ،سىز بەلكىم ھىس قىلغانسىز دىيارىمىزدا زادى قانچىلىك كۈتۈپخانا بار ؟ قانچىلىك كىتاپخانابار ؟ بۇنى ئويلاپ بېقىشقا ئەرزىيدۇ ، مەن ياشايدىغان 40مىڭ نوپۇس بار بىر يېزىدا شەخسى ئاچقان 40 كۇادىرات مىتىر  كەلمەيدىغان بىر كىتاپخانا ۋە يېزىلىق ھۆكۈمەت ئاچقان قايسى كۈنى ئېچىلىدىغانلىقىنى بىلگىلى بولمايدىغان بىر كۈتۈپخانا بار ، بار بولغان كىتاپ زادى قانداق كىتاپ ؟ بۇمۇ موھىم بىر مەسىلە ، كۆپىنىچىللىرى (كۈتۈپخانىدىكى ) ئالۋاڭغا ئەكىلىنگەن تۆكمە كىتاپلار ،ئەلنىڭ قەلبىدىكى ،ئازۇسىدىكى كىتاپلارنى تېپىش قىيىن ...

  گەپ قىلساق گەپ تېخمۇ ئۇزىرايدۇ ،ئەگر راستىنلا ئۇيغۇر مەدىنىيىتىگە ،ئۇيغۇر خەلقىگە قىزىققان بولسىڭىز يەنىمۇ چوڭقۇرلاپ ئىزدىنىشىڭىزگە ،بىز ھەققىدىكى ھەر قايسى ئەللەردىكى تارىخنى بىلىشىڭزگە توغرا كېلىدۇ ،ئاخىرىدا سىزنىڭ بۇ قەدەملىرىڭىزنىڭ ئۈزۈلمەي داۋاملىشىشىنى ،تېخىمۇ ئەتراپىلىق نەتىجىگە ئېرىشىشىڭىزنى ئۈمىد قىلىمەن .

     يۇقارقى ماقالە بىرقەدەر ئوبىكتىپ يېزىلىپتۇ . ئاپتورغا رەھمەت .
    مېنىڭچە  بەزى مەسىلىلەر بىۋاستە سىياسەتنىڭ تەسىرى ۋە بېسىمىدىن پەيدا بولغان دەپ قارايمەن . مەسىلەن بىزدە بۇرۇندىن تارتىپ كىنو-فىلىم خەلقنى سوتىسيالىستىك جەمىئەيتنىڭ نەتىجىلىرىنى تەشۋىق قىلىدىغان ۋاستە بولۇپ كەلگەن . باشقا تاراتقۇلارمۇ پەقەتلا سىياسەتنىڭ تۆۋەندىكى جاكارچىسى خالاس . شۇڭا ئاددى ئاۋامدا باشقا دۆلەتلەرنىڭ مەدىنىيەت ھادىسىلىرىنىمۇ شۇ دۆلەتنىڭ مەدىنىيىتى ۋە سىياسىتى بىلەن باغلاپ چۈشۈنۈش ئادىتى شەكىللەنگەن . ھەتتا بىزدە ئامېرىكادىكى ئالما شېركىتىنى ئامېرىكا ھۆكىمىتى باشقۇرىدۇ دەپ قارايدىغان نادان ئادەملەرمۇ ئاز ئەمەس . دېمەك بەزى ئىشلار رايوندىكى سىياسەتنىڭ خەلقنىڭ ئېڭىدىكى ئىپادىسىدىن ئايرىلالمايدۇ .
   كىتاپ ئوقۇماسلىق مەسىلىسىگە كەلسەك . بۇ ھەقىقەتەن ئېغىر ئەھۋال . بىزدە كىتاپ ئوقۇيدىغانلارنى ‹‹ كىتاپ خالتىسى ››دەپ مەسخىرە قىلىدىغان ئادەتمۇ شۇ كىتاپ ئوقۇماسلىقنى نورمال ئىش ھېساپلايدىغان پىسخىكىدىن كەلگەن .
   چەتئەللىكلەرنى ياكى ئۇيغۇر ئەمەسلەرنى يات كۆرۈش ھادىسىسى ئۇزۇن مەزگىل بېكىنمە ھالەتتە ياشاپ سىرتقى دۇنيانى چۈشۈنۈش ئېمكانىيىتىگە ئېرىشەلمىگەن مىللەتلەردە بولىدىغان ئومۇميۈزلۈك ئەھۋال بولسا كېرەك . بۇ ئىشلار ئۇيغۇر بىلەن دۇنيانىڭ ئۇچرىشىشنىڭ كۆپىيىشى بىلەن ھەل بولغۇسى !
