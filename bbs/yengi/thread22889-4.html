ئەھمىيەتلىك لېكسىيە سورۇنىنىڭ ھازىرلىنىش جەريانىغا شاھىت بولغانلىقىم ئۈچۈنمىكىن، جەمئىيىتىمىزدىكى ھەر بىر قېرىندىشىمنىڭ خۇددى رەختكە كەشتە تىككەندەك بەزىدە يىپ ئۈزۈلۈپ قالىدىغان بەزىدە يىپقا تۈگۈم سالىدىغان ، بەزىدە تىكىشتىن زېرىكىپ «بولدىلا!» -دەپ تاشلاپ قويسىمۇ شۇ كەشتىچىلىككە بولغان ئىشتىياقنىڭ قىستىشى بىلەن يەنى قولىغا يىڭنە - يىپ ئېلىپ كەشتىلەشنى باشلايدىغان جەرياندەك جاپاغا چىداش روھىنى ، نازۇكلۇق ، ئىنچىكىلىك، سەمىيمىلىك تەلەپ قىلىدىغان ئىلمى سورۇن تەييارلاش جەريانىنى ھېس قىلىشىنى ،ھەممە تەييارلىق پۈتۈپ سەھنە ئېچىلغاندا ئادەم ئۆزىمۇ بىلىپ - بىلمەيدىغان سېھىرلىك مۇھىتنىڭ ئىلكىدە مۇتەخەسسسىسلەرنىڭ دۇردانىلىرىدىن ئوزۇق ئالىدىغان، بۇ كاتتا يىغىلىشقا مىڭلىغان ،ئونمىڭلىغان قېرىنداشلىرىمنىڭ كېلىشىنى ئۈمىد قىلىمەن!
بۇ ئۈمىدىمنىڭ ئىچىدە ئابلاجان مۇئەللىمنىڭ باسقان جاپالىق قەدەملىرىگە بولغان قايىللىق،  كۇتۇپخانىنىڭ ئۇزاق مۇددەت تەرەقىياتىغا بولغان ئۈمىد يەنى ئائىلە ئۈستىدە ئىزدىنىپ مەجروھ ئائىلىلەرنى قۇتقۇزۇش ۋە قۇتقۇزۋېلىشقا بولغان ئىلتىجايىم سىڭگەن ! جەمئىيەتنىڭ كىچىك ھۈجەيرىسى بولغان ئائىلە كىشلىك قەدىر -قىممىتىمىزنىڭ يېتىلىش ئوۋىسى ! بۇ ئوۋا مېھرى - مۇھەببەت بىلەن گۈزەل ،مەسئولىيەت بىلەن ئۇلۇغ! ئۇ ھەققىدىكى ئىزدىنىشلىرىمىز ئۆزىمىز ھەققىدىكى ئىزدىنىش ۋە تەمتىرەشلىرىمىز، بۆسۈش ۋە تەرەقىياتىمىز بىلەن چەمبەرچەس باغلانغان ! ئۇ ھەقتىكى ئىلمى مۇنازىرىلەرگە ئىشتىراك قىلىش ،ئائىلە ئالدىدىكى مەجبۇريىتىمىزنى ئادا قىلىش ئۈچۈن بېسىلغان قەدەملىرىمىزدىن بەرىكەت تامچىلىرى تامچىغاي!  قىسقىسى بۇ قېتىمقى پائالىيەتنىڭمۇ ئالدىنقى قېتىمقى پائالىيەتتەك مۇۋاپىقىيەتلىك ئۆتىشىگە تىلەكداشمەن!
