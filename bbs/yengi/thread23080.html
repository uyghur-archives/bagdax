نەزىر ئۆتكۈزۈش قورساق توقلاش ئەمەس!


        نەزىر توغرىسىدا ئىككى ئېغىز سۆز قىلاي .ئەسكەرتىش تۆۋەندىكى مەزمونلار پەتىۋا ئەمەس ،بەلكى كۆز قاراش .مەن پەتىۋا چىقىرالايدىغان سەۋىيەدە ئەمەس .ئەگەر خاتا كەتكەن يېرى بولسا ئۇستازلارنىڭ كۆرسىتىپ بىرىشىنى سورايمەن .
       نەزىر سۆزى قۇرئاندا ئايسا سەللەللاھۇ ئەلەيھى ۋەسسەللەم توغرىسىدىكى ئايەتتە كېلىدۇ .نەزىر خۇدا يولىدا ئاتاش ،خەيىر ئىھسان قىلىش ،ئەڭ ياخشى نەرسىسىنى خۇدا يولىدا بەخشەندە قىلىش ،ئەڭ ياخشى تىلەكلەرنى ئۈممەتنىڭ كۈچىيىشى ئۈچۈن بىرىش ....قاتارلىق ئىسلام ئۈچۈن بېغىشلاش مەنسىدە دەپ چۈشەندىم .نەزىرنىڭ مەزمونى بەك كەڭرى .
      ئاجىز ئورۇقلارغا ياردەم قىلىش نەزىرنىڭ بىر تارماق مەزمونى .ھازىر بىزنىڭ قۇرسىقىمىز ئازراق تويۇپ قېلىپ ،نەزىر مۇراسىملىرىنى ئاز ئۆتكۈزۈشنى تەۋسىيە قىلىۋاتىمىز .ھەمدە بۇ مۇراسىملارنى نەزىر ئەمەس چاي دەپ ئاتاشنى ئۈمۈد قىلىۋاتىمىز .ئەمەلىيەتتە بۇ بىر ياخشى تىلەك ئارزۇ .لېكىن بۇ مۇراسىملارنىڭ نەزىر ئاتىلىپ قېلىپ چوڭلارنىڭ قەلىپ تۆرىدە چوڭقۇر يەر ئېلىش ۋە چاي دەپ ئاتىلىشقا قارشى كەيپىياتىنى توغرا چۈشىنىشكە بولىدۇ .تۇرمۇشىمىز ياخشىلىنىشتىن سەل ئىلگىرىكى ۋاقىتلاردا مەسىلەن 40-يىللاردىن ئىلگىرى ھەتتا مەدەنىيەت ئىنقىلابىي ،80-يىللاردا بىر مەھەللىنىڭ ياكى بىر يۇرتنىڭ ئاھالىسى ئاساسەن نامرات ئۆتەتتى .ئۇلار ھەقىقەتەن يوقسۇللار ،مىسكىنلار ئىدى .ئۇلارنىڭ  ھالى بۈگۈن تاپسا ئەتىگە يوق ،چۈشلۈكىنى غەملىسە كەچلىكى ئاچ ياتقان ئەھۋالدا ئىدى .بەلكىم ھازىرقى بالىلار بۇنى بىلمەسلىكى مۇمكىن .چوڭلىرىمىزدىن سورىساڭلار نامراتلىقنىڭ نىمىلىكىنى بىلىسىلەر .ئاللاھنىڭ ھىممىتى بىلەن شۇكرىكى شۇ چاغلاردىكى دەۋەتچى موللىرىمىزنىڭ كۈچىشى ۋە دەۋىتى بىلەن قول ئىلكىدە بار بايلار ،يۇرت كاتتىلىرى دائىم يۇرتقا ئاش تارتىپ تۇراتتى .نەزىر بىرىپ تۇراتتى .خەلىق نامرات بولغاچقا ئۇلار قىلغان بۇ مۇراسىملار ھەقىقەتەن نەزىر سانىلاتتى .مانا ھازىر سەل قۇرسىقىمىز تويۇپ قېلىپ بىز قىلغان مۇرسىمغا قاتنىشىدىغانلار ئىچىدە غېرىپ غۇرۋالار ئاز بولغاچقا ئۆتكۈزىۋاتقان بۇ مۇراسىملىرىمىزنىڭ نەزىر ياكى ئەمەسلىكىدىن گۇنانلىنىپ قالدۇق .
       ئازاتلىقتىن ئىلگىرى ،تەبلىغ سورۇنلىرى ئۇيۇشتۇرۇلۇپ تۇرۇلاتتى .ھۆكۈمدارلار،قول ئىلكىدە بارلار ،ئالىملار بىر يەرگە كېلىپ دىنى سۆھبەت بولۇنۇپ ،قۇرئانلار ئوقۇپ تۇرۇلاتتى  ،ئىلىم تەھسىل قىلىشنى ،بىرلىك ئىتىپاقلىقنى كۆزلەپ قۇرۇلغان بۇ سورۇن ئەلبەتتە نەزىر ئاتىلىدۇ .چۈنكى بۇنداق سورۇنغا قاتناشقانلار ھاللىقلار بولسىمۇ لېكىن ئىلىم ئالماشتۇرۇشنى مەقسەت قىلغان بولغاچقا سورۇن ئىگىسىنىڭ بۇ چىقىمى ئەلبەتتە ئاللاھ يولىدا نەزىر ئاتىلىدۇ دە .
لېكىن ھازىرقى نەزىرلىرىمىزدە قوساق توقلاش ۋە ئاخىرىدا ئازراق قىرائەت قىلىش بار .بۇندىن باشقا ئىش يوق .ئاللىكىملەردىن قورقۇپ تەۋھىد ئەقىدىسىنى سۆزلەش يوق دىيەرلىك .سۆزلىگەنلەرنى ئاپرىپ چېقىۋاتقان .شۇڭا توق قۇرساققا بىرىلگەن ۋە ئىلىم سۆزلەنمەي ،مۆلۆت ئوقۇپ پۇل تېپىش ئاتىلىپ قالغان بۇ سورۇنلاردىن ئىمانى كۈچلۈكرەكلەر ،ئىسلامنىڭ ساپلىقىغا ئىنتىلدىغان خەلق بىزار بولۇشقا باشلىدى .ۋە بۇ سورۇننىڭ باشقا بىر مەنىدە ئاتىلىشىنى نەزىر ئاتالماسلىقىنى ئۈمۈد قىلدى .بۇ ئۈمۈدلەردىن بىر مۇرەككەپ ھىسياتنى بايقىۋالالايمىز .بۇ ھىسيات شۇكى گەرچە بۇ سۇرۇنلارنىڭ نەزىرلىكىدىن گۇمان قىلىپ قالغان بولساقمۇ لېكىن دىننى تەشۋىق قىلىدىغان مۇھىت ۋە پۇرسەتتنىڭ يىلدىن يىلغا قىسىلىۋاتقانلىقى ئۈچۈن بۇ سۇرۇنلارنىڭ يوقۇلۇپ كېتىشىنى ئۈمۈد قىلمىدۇق .شۇڭا چاي دەپ ئاتىدۇق ياكى بۇ سورۇنلاردا قۇرئان تامام قىلىپ سورۇن مەزمونىنى نەزىرگە يېقىنلاشتۇرۇشقا تىرىشتۇق .بەزى كالتە پەم ۋە يىراقنى كۆزلىمەيدىغان لېكىن نىيىتى ياخشى ئاتالمىش ۋاھابىي سۈپەت دىنى قېرىنداشلىرىمىز بۇنداق مۇراسىملارنى تۈپ يىلتىزىدىن يوقىلىپ كېتىشنى ئۈمۈد قىلدى .بۇ ھازىرقى سىياسىي ھالىتىمىزگە ماس كەلمەيدىغان قاراش دەپ قارايمەن .چۈنكى ئالىملارنىڭ پەتىۋالرىنىڭ ئىجرا قىلىنىشى زامان ۋە ماكاننىڭ ئۆزگىرىشىگە قارىتا تۈزىتىلىش بولىدۇ .نەزىر ئۆتكۈزۈش ئۆتكۈزۈلمەسلىك پەتىۋاسىنى قايتىدىن بىر ئىسلاھات قىلىپ ،ئەمەلىيتىمىزگە ماس ھالدا تەتقىق قىلىپ بېقىش زورۇر ۋە جىددى مەسىلە دەپ قارايمەن .
         بەزى قېرىنداشلىرىمىزدەك  ھازىرقى نەزىرسىمان مۇراسىملارنى نەزىر دەپ ئاتىۋېلىپ ،ئۇنىڭغا دۈم چۈشۈش خاتا شۇنىڭ بىلەن بىللە بىتئەتنى يوقىتىمەن دەپ ۋاستىدە خاتالىق سادىر قىلىپ قالغان بەزى سۈننەتچى  قېرىنداشلىرىمىزدەك  بۇ مۇراسىملارنى ئىنكار قىلىپ يوقىتىۋېتىش ھەم خاتا .بۇ مۇراسىملارنى چوقۇم داۋاملاشتۇرىشىمىز ۋە ئىسلامى مەزمونلىرىنى موللاشتۇرشىمىز كىرەك .نەزىر مەنىسگە يېقىنلاشتۇرۇشقا تىرىشىشىمىز لازىم .غېرىپ يوقسۇللارنىڭ نسىۋىسىنى يەۋەلمايلى !بالاھىي ئاپەتكە ئۇچىرىغانلارنى،قەرىزگە بوغۇلۇپ ئامالسىز جازانە يولىغا مېڭىش ئالدىدا تۇرغانلارنى،شەيتاننىڭ زىيانكەشلىكىگە ئۇچىراپ قازا قىلغانلارنىڭ ئائىلىلىرىنى ،ئىسلامى ئاقارتىش قىلىش يولىدا زەربە يىگەنلەرنى ئۇنتۇپ قالمايلى .
       ئەگەر بەزىلەردەك مۇشۇنداق نەزىرنى چەكلىمىسىز ھالدا تەشۋىق قىلىدىغان بولساق يوقىرقى قېرىنداشلىرىمىزنىڭ نىسىۋىسى توق قۇرساقلارغا ۋە ئىسراپچىلىقلارغا كىرىپ كېتىدۇ .

نەزىر قانداق ئىش؟جاۋاب – نەزىر، ئاللاھ ئەمر قىلمىغان بىر ياخشى ئىشنى قىلىمەن، دەپ ئۆز ئۈستىگە ئېلىش دېگەن بولىدۇ. مەسىلەن: ئاللاھنىڭ رازىلىقى ئۈچۈن ئۆمرە قىلىشنى ياكى بىر قانچە كۈن نەپلە روزا تۇتۇشنى ياكى بىر قانچە رەكئەت نەپلە ناماز ئوقۇشنى ۋە ياكى بىر مىقدار پۇلنى ئاللاھ يولىدا چىقىم قىلىشنى ئۆزىگە ۋەزىپە قىلىش… قاتارلىقلارنىڭ ھەر بىرى بىرەر نەزىردۇر. بۇنداق نەزىرلەرنى ئورۇنلاشنىڭ، جەننەتكە كىرىدىغان كىشىلەرنىڭ سۈپەتلىرىدىن  بىرى ئىكەنلىكى ئىنسان سۈرىسىنىڭ 7 – ئايىتىدە بايان قىلىنىدۇ. ئاللاھنىڭ دىنىدىكى «نەزىر» مانا مۇشۇنداق نەزىرلەردۇر.ئەمما مېيىت چىققان ئۆيلەردە ئۆتكۈزۈلىدىغان نەزىرلەرنىڭ دىنىمىزدا ھېچقانداق ئاساسى يوق. كۈن توختىتىپ «بۇ كۈندە ياغ پۇرىتىلمىسا مېيتنىڭ روھى خاپا بولۇپ قالىدۇ» دېگەن ئېتىقادلار ئىنايىن خەتەرلىكتۇر، ھەتتا شېرىككە يېقىندۇر.ئۇيغۇر مۇسۇلمانلارنىڭ ئارىسىدا مېيىتنىڭ ئارقىسىدىن «بامدىتى»، «ئۈچى»، «يەتتىسى»، «يىگىرمىسى»، «قىرىقى» ۋە «يىلى» دەپ ئۆتكۈزۈلىدىغان نەزىرلەر بار. بۇلار نەزىر دەپ ئاتالغان بولسىمۇ ئەمەلىيەتتە دىنىمىزدا كۆرسىتىلگەن نەزىرلەردىن ھېساپلانمايدۇ. بۇنىڭدىن قاتتىق ساقلىنىش لازىم. مەن ئوقۇغۇچى ۋاقتىمدا ئەھلى تەۋھىد بىر ئالىمنىڭ: «يەتتە، قىرىقى، يىگىرمە دېمىدۈك، ماتەم ئۆيىدە نەزرە يېمىدۈك» دېگەن سۆزىنى ئاڭلىغان ئىدىم. مانا بۇ سۆز، ئۇ ئالىمغا «ئەھلى تەۋھىد» دېيىلىشنىڭ نەقەدەر توغرا ئىكەنلىكىگە دالالەت قىلىدۇ.（مۇسۇلمانلار تورىدىن كۆچۈرۈلدى）
توي تۆكۈن ھايالىق بولۇشنى تەكىتلەپ مىھمان چاقىرسا،ئىسراپ قىلماي تاماق تارتسا ،ئەخلاقنى كۈيلەپ مۇۋاپىق ناخشا مۇزىكىلار بولۇنسا شەرىئەتكە بىز ئۇيغۇرلار ئۆرپ ئادىتىگە ئۇيغۇن،（ھەرقايسى پەتىۋاا تورلىرىدىن كۆرىۋىلىڭ）
ھەممىنى يوقىتىش ئۇيغۇرلارنى مەنىۋى جەھەتتىن بىسىش ماددى جەھەتتىن شۈلۈش سىزدەك ئاز بىر قىسىم كىشىلەرنىڭ خىيالى ،لىكىن جانابى ھەق ئەمەلگە ئاشۇرمىسا سىلەرنىڭ كۈچىگىنىڭلار بىكا！
