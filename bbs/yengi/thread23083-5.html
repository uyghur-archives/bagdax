بىزنىڭ تىلىۋىزىيە قاناللرىمىزدە ناخشا ئۇسۇلدىن باشقا يەنە نىمىمىز بار دەيسىلەر،كۆڭەلگە ياققۇدەك بىرەر ئاللىمىلىرىمىزنى تونۇشتىرىدىغان ،يىڭى پەن تىخنىكا بىلىملىرىدىن خەۋەر قىلىدىغان قاناللىرىمىز يوق دىيەرلىك .شۇڭا قاناللارنى ئسلاھ قىلىپ ناخشا -ئۇسۇل بىرىش سانىنى ئازايتىپ ،يىڭى قاناللارنى يولغا قويسا ،بىزنىڭ ئۆتمىش ،بۈگۈنىمىز ،ئەتىمىز توغرىسىدا مۇلاھىزە قىلىدىغان ئاللىملىرىمىزنى ،تەتقىقاتچىلىرىمىزنى تەكىلىپ قىلىدىغان پىروگىراممىنى يولغا قويسا ،ئۈچ مەزمۇن توغرىسىدا مۇلاھىزە قىلساق ،شۇ دەۋىردە مۇشۇ دەۋىردە ئۆتكەن تەتقىقاتچىلىرىمىزنى تونۇشتۇرسا باللىرىمىز ،ياشلىرىمىز كۆرسە ئىدىيەدە ئۆزگىرىش بولىدۇ .ھازىر سۈنئى ھەمرا قاناللىرىنى دىھقان چارۋىچىلىرىمىز ،شەھەر-يىزىدىكى ئادەملىرىمىز تولۇق كۆرەلەيدۇ ،دۆلەت يىزىلارغا سۈنئى ھەمرا قوبۇل قىلغۇچ (قازان) ئورنىتىپ بەردى شۇڭا دىھقانلارمۇ بەھىرىنىلەيدۇ..بىز ئۇيغۇرلار ئۆتمىشتە تارىخنى ياراتتۇق مەدىنيەت ياراتتۇق ،يىڭى كەشپىياتلارنى بارلىققا كەلتۈردۇق ،ھازىرچۇ ھاراق تاماكا نەشە ،خىروئىن ،پايىشۋازلىقنى ،ئەيدىس كىسىلىنى ئۆزىمىزگە دوس قىلۋاتىمىز .....شۇڭا ئۆتمۈشنى ئۇنتۇپ قالمايلى بۈگۈنىمىزنى ،ۋە ئەتىمىزنى ئۇنتۇپ قالمايلى

مىنڭ ئويلايدىغنىم بولسا مۇشۇنداق تىمدىكى ماقالىلەرنى رادىئۇ ئستانسسى، تىلۋىزىيە ئستانسىسى دىكى قرىنداشلىرمىز كۆپىرەك كۆرسىكەن دەيمەن، بىلمىدىم ئۇلار كۆرسىمۇ كۆرمەسكە سىلپ، تەپەككۇ ر قىلماي  مىنڭ قولۇمدىن نىمە كىلەتتى دەپ ئايلىق مۇئاشنى ئىلپ بىشنى ئچىگە تىقىپ يۈرۋاتامدىغاندۇ؟
