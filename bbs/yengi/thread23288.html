                                                         توك توخۇسىنىڭ تۇغۇت تويى                               
                                          (سۆز ئوينى)  
                                         
    تورداشلار،تۈنۈگۈن توك توخۇسىنىڭ تۇغۇت تويى تەبەسسۇم تانسىخانىسىدا تەنتەنىلىك تەبرىكلەندى.تورداشلارنى تويغا توۋلىيالمىدۇق.توۋلىيالماسلىقتا؛تىلفۇن توكىنىڭ تۈگىشى،توي تەرتىپلىرىنى توغرا تۇرغۇزالماسلىقىمىز،توك توخۇسىغا تاۋۇشداش تاماششىبىنلارنىڭ تولىلىقى،تويغا تەييارلانغان تانسىخانىنىڭ تارلىقىدەك تۈرلۈك تەتۈر توسالغۇلار توساپ تۇرۇۋالدى.
     توك توخۇسىنىڭ تۈنۈگۈنكى تۇغۇت تويىغا تورداشلىرىمىزنى توۋلىيالمىغاچقا،توينىڭ تەپسىلاتىنى تورداشلىرىمىزغا ت تاۋۇشلۇق تىلفۇننىڭ تاتلىق تىلىدا تەپسىلى تونۇشتۇرىمىز.
     توك توخۇسىنىڭ تۇغۇت تويىغا تەئەللۇق تەپسىلاتلارنى تەجىرىبىلىك تورداشلىرىمىز تولۇق تاپشۇرۋىلىپ،توي تەرتىپلىرىنى تەپسىلى تەكشۈرۈپ تەھرىرلەپ،تۈزۈتىشكە تىگىشلىك تەرەپلەرگە تەلەپ-تەكلىپلىرىڭلارنى تۇراقلاشتۇرۇپ،توك توخۇسىنىڭ تۇغۇت تويىنى تەستىقلىغان تاپاۋەت تەشكىلاتىغا تاپشۇرۇشىڭلارنى تەشنالىق تۇيغۇسىدا تىلەيمىز.
    تورداشلار،تىلىمىزنىڭ تەملىك ت تاۋۇشىدا تونۇشتۇرۇۋاتقان توك توخۇسى،توختىماي تۇغۇۋاتقان توخۇلاردىن تەسىرلىنىپ،تۇغۇشقا تەييارلىنىپ،تۇغۇشنىڭ تەرتىپلىرى توغرىسىدا تالانتلىق تېۋىپلاردا تەرتىپلىك تەربىيلىنىپ،تۇغماس توخۇلارنىڭ تۇغۇشىنى تىزىملايدىغان تاپاۋەت تەشكىلاتىغا تىزىملىتىپ،تەڭگە-تىللالىنى تۆلەپ تالۇن تۇرغۇزۇپ،تۆمۈرتامغىدا تەستىقلىتىپ،تولىمۇتەستە تۇققان.توك توخۇسىنىڭ تۇغۇت تەپسىلاتى تۆۋەندىكىدەك؛
      توك توخۇسىنىڭ تولغىقى تۈنۈگۈن تۈن-تەڭدە تۇتقاچقا،توخۇلارنى تۇغدۇردىغان تەرىسا تېۋىپلارنى تاپالماي،تۇغماس تۇققانلىرىنىڭ تېتىقسىز تللاشلىرىنىڭ تەسىرىدە تەستە تۇققان.توك توخۇسىنىڭ تۇغۇشىنىڭ تەسكە توختىشى؛-تۇغماس توخۇلار تەۋەلىكىدىكى توڭلاتقۇدا تۇرغاچقا، تەجىرىبىلىك توڭلاتقۇ توك توخۇسىنىڭ تەرتىپلىك تۇغۇدىغان تۇڭلۇك تۆشۈكچىللىرىنى توڭلۇتۇپ توسىۋەتكەن،تەلەيلىك توك توخۇسى تۈرلۈك توسالغۇلار توسسىمۇ تەمتىرەپ تۇرماي،تولغاق تۇتماستىنلا تەتۇر تولغۇنۇپ،تۆۋەن تەرەپتىكى تۇغۇدىغان تۇڭلۇكقا تەلمۈرۈپ تۇرماي،تۇمشۇقى تەرەپتىكى تىنىدىغان تۆشۈكتىنلا توككىدىلا تۇغۇۋەتكەن.
     توك توخۇسىنىڭ تولغاق تۇتمىسىمۇ تەمتىرەپ تۇرماي تەتۈر تولغۇنۇپلا تۇغۇدىغان تەجىرىبىسىدىن تەسرلەنگەن تۇققانلىرى توك توخۇسىنىڭ تۇنجى تۇغۇتىنى توي تەرتىۋىدە تەبرىكلەشنى تاپاۋەت تەشكىلاتىدىن تىلىگەن،تاپاۋەت تەشكىلاتى توخۇ تۆلىگەن تەڭگە-تىللانىڭ تاتلىقلىقىدا تىلىنى توختۇتىۋالالماي توك توخۇسىنىڭ تۇنجى تۇغۇت تويىنى تەنتەنىلىك تەبرىكلەشنى تەستىقلىغان،توك توخۇسىنىڭ تۇققانلىرى تەستىقنى تاپشۇرىۋىلىپ،توينى تۆۋەندىكىدەك تەرتىپتە تەشكىللىگەن.
    توك توخۇسىنىڭ تاۋۇشداش تونۇشلىرىنى تىلفۇندا توۋلاپ تانسىخانىغا توپلاش،توينىڭ تاماقلىرىنى تەييارلاش.تاماششىبىنلارنىڭ توينى تەبرىكلەشكە تەييارلىغان تەڭگە-تاۋارلىرىنى تولوقتاپشۇرۇۋىلىش.تاماششىبىنلار توينىڭ تەمىناتىغا تويۇپ تەرەتكە تەمشەلگەندە توينى تاماملاش.
    تاماششىبىنلارنى تىلفۇندا توۋلاشقا تۇمۇچۇقنى،تاماقلارنى تەخسىملەپ تارقىتىشقا تارىختىكى تالانتلىق تەتۈر تەخسىماتچى تۈلكىنى،تاماقنى تەييارلاشقا تىرىشچان توغاينى،توينى تەبرىكلەشكە تاماششىبىنلار تەييارلىغان تاۋارلارنى تاپشۇرۋىلىشقا تەرىسا تىمساقنى.توپلانغان تاۋارلارنى توختاتماي توشۇشقا تۇمشۇقىدىن تۆشۈك تىشىپ،تانىدا تارتمىسا تەۋەلىكىنى تاپالمايدىغان تاختاي تاپانلىق تۆگىنى.تاماششىبىنلار تاشلىۋەتكەن تاماقلارنى تىيىپ-تازىلاشقا تارتىنماي توققۇزنى تۇغۇدىغان توڭگۇزنى.توي تۈگىگەندە تاماششىبىنلارنى تارقىتىشقا تاغىسىنىمۇ
تىپىپ-تۈتۈپ تىللايدىغان تايغان تەلەتلىك توپاقنى تەيىنلىگەن.
     توك توخۇسىنىڭ تۇغۇت تويىنى تەبرىكلەش توغرىسىدىكى تىلفۇننى تاپشۇرۋالغان تاۋۇشداش تاماششىبىنلاردىن؛-تۈلكە،تۇلپار،توپاقلار.تۆگە،تايغان،تىمساقلار.توخو،توشقان،توغايلار.
تۈلكە تۇماق تۇققانلار تەتىللىك تاپشۇرۇقلىرىنى توختۇتۇپ،تاق تاۋۇشنىڭ تىزلىكىدە تاغ تەرەپتىكى
تەبەسسۇم تانسىخانىسىغا توپلاشتى.
     تەرىسا تىمساق تاماششىبىنلارنىڭ توينى تەبرىكلەشكە تەييارلىغان تاۋارلىرىنى،تارتىنىپ   تەمتىرەپ تۇرماي تەپسىلى تەكشۈرۈپ تەمى تاتلىق تاۋۇزلاردىن تارتىپ توغرالمىغان تىرەكتوغراقلار،تۇز تەملىك تاش -  توقماقلار.تەڭلە،تەشتەك تاۋاقلار.تازلار تالاشقان تۆمۈر تاغاقلار.توڭلاتقۇ،تىلۋۇزۇر تىلفۇنلار.تەخسىلەرگە تىزىلغان تەڭگە-تىللار،تېرىقلار.تۇپا توشۇيدىغان تاق توچەيلىك
تەرەكتۇر،تانا-توقماق تىشاڭلارنى توينىڭ تارازىسىدا تارتىپ تەڭپۇڭلاشتۇرۇپ تىزلىكتە تاپشۇرۋىلىپ تۆگىگەتوشۇتۇپلا توك توخۇسىنىڭ تويدا تاۋار توپلايدىغان تۆمۈر توڭلاتقۇسىنىڭ تەتۈر تاناسىپلىق تونۇرىدا توڭلىتىۋەتتى.
     تۇمۇچۇق تاق تارىلىق تەمبۇرنىڭ تاماششىبىنلارنى تانسىغا توۋلايدىغان تاۋۇشىنى تەڭشەۋاتقاندا،تانسىخانىنىڭ تۆرىدە تۇرىۋالغان تۆت تۇغماس تەرىسا توخىنىڭ تويى توغرىسىدا تەتۈرتەشۋىق تارقىتىپ،تاماششىبىنلارنى تانسىخانىدا تۇرغۇزمىدى.


   -توۋا،توۋا،توڭلاتقۇدىكى توك توخۇسىمۇ تۇخامدۇ؟
   -تۇغماس تېۋىپلار تەربىيلىگەچكە تۇغالىدى.
   -توۋا،توۋا،تۇغماس تېۋىپ توك توخۇسىنى تەربىيلەپ تۇغدۇرالايدۇ، تۇغالمايدۇ.
   - تاپاۋەت تەشكىلات تۇغۇشنى تەستىقلىسا توك توخۇسى تۈگۈل،تاغدىكى تاشلارمۇ تاۋۇز تۇغۇدۇ،تونوردىن تارقىغان تۈتۈنلەرمۇ تۇخۇم تۇغۇدۇ.
   -تاماششىبىنلار تولغۇما تانسىنى توختاتمىسا،تەرىتىنى تۇتالماي تانسىخانىنى تولدۇرۋىتىدۇ، تاماششىبىنلارغا توينىڭ تامىقىنى تۈلكە تېخىچە تارقاتمىدى.تاماششىبىنلار تۈتۈن تەرىتىنى تارقىتىشقا تەمشىلىۋاتقاندا،توساتتىنلا تەلۋىلەشكەن توپاق تاققا-تۇققا توۋلاپ؛-تامششىبىنلار!تانسىنى توختۇتۇڭلار،تىلۋىزىيە تىلفۇنىدىن تارقالغان تەشۋىقاتتا تەبەيىمىزنىڭ تىمپىراتۇرسى تۆۋەنلەيدىكەن،تۇرالغۇللىرىمىزغا تىزدىن تارقالمىساق تانسىخانىنىڭ تونۇرىدا تۇرساقمۇ توڭلاپ تۈگۈشىمىز.تەمتىرەپ تۇرماي تىزدىن تارقىلايلى.توينىڭ تاماقتا تەمىللەيدىغان تۈرى توختۇتىلدى.
توي تامام.
      تاماششىبىنلار تاماق تەييارلاشقا تەيىنلەنگەن توغاينى،تاماق تەخسىملەشكە تەيىنلەنگەن تۈلكىنى،تويىغا توۋلىغان توك توخۇسىنى تېتىقسىز تىللاردا تىللاپ تويغۇزىۋەتتى.تىمساقتىن تاۋارلىرىنى تارتىۋىلىشقا تەمشىلىۋاتاتتى، تىمساقتىن تەپ-تارتىمۇتاڭ ،توختاپ توختاپ تىنىپتونوردىن تارقىغان تۈتۈندەك تەتۈر تولغۇنۇپ تەۋەلىكلىرىگە تارقالدى،تەبەسسۇم تانسىخانىسىمۇ تاراققىدا تاقالدى.
   
       ئاپتۇرى؛مەخەت ئابدۇرېھىم.

تەبرىكلەيمىز

تۇنجى توك توخويىڭىزنىڭ تۇلۇم تاشتەك تۇخۇملارنى تۇغقانلىقىنى تىللىرىمىزنى تامشىتىپ تۇرۇپ، تۇمشاقلارنى تۈرۈپ تۇرۇپ تەبرىكلەيمىز.توخۇلىرىڭىز تۇققان تۇخۇملار تەغرىتاغ تاغلىرىدەك توپلانغاي.
تىللا تۇغالمىسىمۇ، تۇخۇم تۇغۇپ تۇمشۇقلىرىڭىزنى تومپايتىپ تۇرۇدىغان توخۇلىرىڭىزنى تىللاپ تۇرىۋەرمەڭ. تۇخۇم تۇغقاندا تۇخۇملىرىنى توپىغا تەككۈزۈپ تاشلىماڭ.
تارىم تۇخۇمىدەك تۇخۇم تۇغسىمۇ تۇغسا، تۇغالىسا توكسىزمۇ تۇغالايدىغان توخۇلارنى تەربىيلەپ،توك تۇخۇمى تۇغۇپ قۇيۇپ تىرنىقىنى تاتلاپ تۇرىۋالدىغان تاكاراڭ توخۇلارنى تەشكىلگە تاپشۇرغايسىز. تەشكىلنىڭ تەپسىرىدىمۇ تەربىيلەنمەي، تاپشۇرۇقلارنى توغرا تاپشۇرمىغان توخۇلارنى تىمساھقا تالىتىمىز،تىكەنگە تاشلايمىز، تاكى تەربىيلەنگۈچە تالىبان تەرەپتىكى تاغلارغا تاشلايمىز.

توخۇلىرىڭىزغا تەشەككۈرىمىزنى تاپشۇرغايسىز، تەشكىل تاپشۇرغان تىلىۋەتلەرنى تاشلىۋەتمەي تەكشۈرۈشكە تەييارلانغاي.
