--بۇ سۇزلەر بىر-بىرىگە زىت بولۇپ قالدىمۇ نىمە؟توت تادۇ تەرەقىيەتقا قارشى مەيداندا تۇرۇپ پىكىر قىلىدىكەن،ھەم بۇ ئاسماندىن چۇشۇپتۇ،{قۇرئان كەرىم-ئۇنلۇك ئوقۇماق دىگەن مەنىدە}نى ئاسماندىن چۇشكەن دەپ قارايمىز،بۇ 1400يىل بۇرۇنقى گەپ، لېكىن ئۇ تەرەقىياتنى تەشەببۇس قىلىدۇ،توت تادۇ قاچان ئاسماندىن چۇشكەن بولغىيىتتى؟بىز ئاسماندا ئاللاھلا بار دەپ قارايمىز،ئۇيغۇرلارنىڭ تارىخى بىلەن ئىسلامنىڭ تارىخى بىر گەپ ئەمەستۇ،ئۇيغۇرلارنىڭ تارىخى ئىسلامنىڭ تاررىخىدىن كوپ ئۇزۇنغۇ دەيمەن،توت تادۇ بالدۇر چۇشكەنمۇ ياكى {قۇرئانى كەرىم }باشتا چۇشكەنمۇ؟ئۇنى بىر ئاللاھ چۇشۇرگەنمۇ ياكى ئوخشىمىغان .......چۇشۇرگەنمۇ؟ماقالە يېزىلىشتا پايدىلانغان ماتىرياللارنىڭ ئىچىدە {قۇرئان كەرىم} باركەن،ئىلاھى كىتاپتا تۇت تادۇنىڭ قاچان چۇشكەنلىگى ھەققىدە بىرەر بىشارەت بارمۇ ياكى سىز بۇنى بىرەر ماتىريالدىن پايدىلىنىپ تاپتىڭئىزمۇ؟ماتىريالنىڭ كېلىش مەنبەسىنى ،بەت سانىنى ئەسكەرتكىلى بولارمۇ؟مەن ئاڭلىغان گەپلەردە {قۇرئان كەرىم}ئىنتايىن مۇكەممەل يېزىلغان كىتاپ،مەن 25يىلدىن بۇيان ئىمكەنىيەتنىڭ بېرىچە بۇنى تەتقىق قىلىۋاتىمەن،سىز دىگەن سۇزلەرنى مۇقەددەس كىتاپنىڭ نەرىدىن كۇرگەنلىگىمنى ھىچ بىلەلمەيۋاتىمەن يا؟سىزنى ئاۋارە قىلىدىغان بولدۇم؟توت تادۇ تەرەقىيارقا قارشى تۇرۇدىكەن،بۇنى سىز دىدىڭئىز،بۇ {قۇرئان كەرىم}بىلەن قارمۇ قارشى ئىكەن،توت تادۇ راستىنلا ئاسماندىن چۇشۇپتىما؟بۇنى يۇسۇپ خاس ھاجىپ بوۋىمىز كۇرۇپتىما؟ياكى سىز كودۇڭىزمىيا؟

ھەكىم.         تەرەققىيات نوختىنەزىرى دىگەنلىك ، ماددا يەنى  بىر جىسىمغا بىرسىرىتقى كۇچ تەسىر قىلسا، تۇزسىزىقلىق تەكشى ھەركەت ھالىتىنى ساقلايدۇ، يەنى ماددىغا ماددا سۇقۇلۇپ تەدىرجى  راكسىيەلىشىپ، سۇ پەيدا بۇلۇپ، اندىن يەنە تەرەققىي قىلىپ  سۇنىڭ دولقۇنى بىر بىرىگە سۇركۇشۇپ، دەسلەپكى ىپتىدايى جانلىقلار پەيدا بۇلۇپ، ماددىنىڭ ايلىنىش قانۇنى دەمدۇ، بۇجانلىقلار تەدىرجى ىنسان تاكاممۇلىقىغا يەتكەن دىگەندەك مۇشۇ يۇسۇندىكى گەپلەرنى  تەرەققىيات نوختى نەزىرى دەيمىز. يۇقارقىسى بۇنىڭ دۇنياقارشى،ەمدى مىتۇت يەنى پىكىر ۇسۇلى دىگەن گەپ بار. تۆت تادۇ بۇيىچە مەسىللەرگە باھا بەرگەندە ، ۇخىل پىكىر ۇسۇلىدا تادلىمايمىز،بۇنىڭ مىتۇدى ماتىريال ىلمى بۇيىچە بولمايدۇ . تۆت تادۇ زىدىيەت مۇناسىۋىتىنى چىقىش قىلىدۇ. نۇرغۇن ىدىيال قاراشلار ھازىر دۇنياقاراشتا ىلاھنى چىقىش قىلسىمۇ،بىراق پىكىر ۇسۇلىدا شۇ تەدىرجىتەرەققىيات نوختىنەزىرى  مىتۇدىنى قوللىنىۋاتىدۇ. مەسلەن، ەرەپلەرنىڭ جاھىلىيەت دەۋرىدىكى ەھۋالى اندىن ىسلام دىنغا كىرگەندىكى ەھۋالى ، اندىن رەسۇلىللادىن كىيىنكى ەھۋال دىگەندەك، ۇسەۋەپ بۇسەۋەپ مۇنداق بولغان دىگەندەك،تۆپىدىكى گەپ بىلەن ماۋۇ گەپ تىكىستى وخشىمىسىمۇ ، اھاڭى وخشاش .  ھەتتا پەتىۋا مۇھىم قارالمىلاردىمۇ شۇ مىتوت. مىتوت  جايىغا كەلمىسە، اخىرقى ھۇكۇمدە زور پەرىق بۇلىدۇ.  ىدىيال قاراشلارغا ۇزىنىڭ مىتۇدى بولغان تۇززۇك.                                  سۇرە ھىجىردە ىنساننىڭ تۇپراقتىن يارتىلغانلىقى ، شەيتان جىن وت يالقۇنى، وتتىن يارتىلغانلىقى، سۇرە نۇردا بەزىسى ىككى پۇتلاپ، بەزىسى قورسىقىدا ماڭىدۇ دىگەن يەردە بىز ۇلارنى سۇدىن ياراتتۇق ،يەنى ھەيۋانلارنىڭ سۇدىن يارتىلغانلىقى قەيىت قىلىنىدۇ. بۇ تۆت تادۇنىڭ اسماندىنچۇشكەنلىكىگە پاكىت بۇلالماسمۇ. ىلاھا مەيلى ماكان بولسۇن مەيلى زامان،ىنسان قۇرانغىچە تۆت تادۇنى مۇجەسسەم قىلغان. ھەممىنى تۆت تادۇ بىلەن خەلىق ەتكەن. ھەممىسىدىن تۆت تادۇ چىقىدۇ.
