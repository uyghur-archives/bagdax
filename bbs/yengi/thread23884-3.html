مەزكۇر ئەسەردە ئاپتۇر خېلى ئەتراپلىق كۆزىتىپتۇ ۋە خېلى ئىزدىنىپ تەھلىل قىلىپتۇ. ئەمما مەسىلىنىڭ نىگىزى يەنىلا شۇ ئۇيغۇر ئىگىلىكىنىڭ نېمە ئۈچۈن ساغلام، تېز تەرەققىي قىلمايدۇ؟ دېگەن سوئالغا بېرىپ تاقىشىدىغان بولغاچقا، بۇ مەسىلىنى تۈپ نىگىزىدىن ھەل قىلىش كۆپ تەرەپكە بېرىپ تاقىشىدۇ.
بىرىنچىدىن، ئۇيغۇر كارخانىچىلىرى ئالدى بىلەن ئەنئەنىۋىي سودا سېتىق رامكىسىدىن ھالقىپ چىقىشى كېرەك. گەرچە يېقىندىن بۇيان ئۇيغۇر كارخانىچىلىرى خېلى كۆپلەپ، شىركەتلەرمۇ ئارقا-ئارقىدىن بارلىققا كېلىپ، خېلى خۇشاللىنارلىق نەتىجىلەرنى قولغا كەلتۈرگەن بولسىمۇ، ئەمما كارخانىچىلىرىمىزنىڭ كۆپ ساندىكىلىرى گەرچە شىركەت قۇرۇپ، سىستېمىلىشىشنىڭ، قېلىپلىشىشنىڭ دەسلەپكى قەدىمىنى باسقان بولسىمۇ، ئەمما يەنىلا ئەنئەنىۋىي ئېلىم-بېرىم رامكىسىدىن ھالقىيالماسلىق ئەھۋالى بەك ئېغىر.
ئىككىنچىدىن، مىللىي كارخانىچىلارنىڭ بىلىم قۇرۇلمىسى ۋە ساپاسى ئۆستۈرۈلۈشى كېرەك. كارخانىچىلار ئۆزىدە نېمىنىڭ باشقۇرۇش، نېمىنىڭ باشقۇرۇلۇش، نېمىنىڭ ئىلمى باشقۇرۇش، نېمىنىڭ ئادەم كۈچى بايلىقى، نېمىنىڭ بازار ئىگىلىكى (ھازىرقى زامان بازار ئىگىلىكى)، نېمىنىڭ ئېلانشۇناسلىق، نېمىنىڭ مۇلازىمەت، نېمىنىڭ شىركەت مەدەنىيىتى، نېمىنىڭ شىركەت ئوبرازى بولىدىغانلىقىدەك نۇرغۇن ئۇقۇملارنى ھەقىقىي ماھىيەت جەھەتتىن چۈشۈنۈپ يېتىشى تولىمۇ زۆرۈر.
يۇقىرىدىكى ئۇقۇملار ھەققىدە بىلەمسەن دېسە بىلىمەن دەيدىغانلار كۆپ، ئەمما ھەقىقىي ماھىيەت جەھەتتىن چۈشۈنۈپ يېتەلەيدىغانلار ھەم ئۇنى ئەمەلىيەتكە باغلاپ قوللىنىپ، ئۇنىڭ ئۈنۈمىنى كۆرەلەيدىغانلار تېخى دېگەندەك كۆپ ئەمەس. بۇ ئۇقۇملارنى تېگى-تەكتىدىن چۈشۈنۈپ، ھازىرقى زامان بازار ئىگىلىكىگە ياخشى ماسلىشىپ، ئۆزىنىڭ ئىگىلىكىنى مۇستەھكەم، ساغلام تەرەققىياتلارغا ئېرىشتۈرۈش شۇ كارخانىچىدىن ناھايىتى زور كەسپىي بىلىم، كەسپىي ساپا تەلەپ قىلىدۇ. ئەمما بىزنىڭ ھازىرقى كارخانىچىلىرىمىزدا بۇ تەلەپكە يەتكەنلەرنى پەقەت بارماق بىلەن سانىۋېلىش مۇمكىن.
ئۈچىنچىدىن، كارخانىچى ۋە ئىختىساس ئىگىسى ئوتتۇرىسىدا كۆرۈنمەس توساق مەۋجۇت. ھەر قانداق بىر شىركەتكە نىسبەتەن شىركەت مەدەنىيىتى تولىمۇ مۇھىم. ئەمما شىركەت مەدەنىيىتى شۇ شىركەتنىڭ قۇرغۇچىسىنىڭ ئىدىيەسى تەرىپىدىن بەلگىلىنىدۇ. ئەگەر شۇ شىركەتنى قۇرغۇچى ئىلغار ئىدىيەگە ئىگە بولسا، ئۇ ھالدا بۇ شىركەتتە ناھايىتى زور بولغان يوشۇرۇن كۈچ بولۇشى مۇمكىن. شىركەتلەر ئوزاق مۇددەت تەرەققىي قىلىمەن دەيدىكەن، چوقۇم ئىلغار بولغان شىركەت مەدەنىيىتىنى بەرپا قىلىشى كېرەك. پەقەت ئىلغار بولغان شىركەت مەدەنىيىتىنى بەرپا قىلغاندىلا، ئاندىن ھەقىقىي ئىختىساس ئىگىلىرىنى تۇتۇپ تۇرغىلى، شىركەتكە ۋاقتىدا قان تولۇقلاپ شىركەتنى ھاياتى كۈچكە ئىگە قىلغىلى بولىدۇ. ئەمما بىزنىڭ نۇرغۇن كارخانىچىلىرىمىزدا تېخى شىركەت مەدەنىيىتى ئۇقۇمى يوق دېيەرلىك ياكى شەكىللەنمىدى. شۇ سەۋەبتىن كارخانىچىلاردا ھەقىقىي ئىختىساس ئىگىسىنى قەدىرلىمەسلىك ياكى بايقىيالماسلىق، بايقىسىمۇ تۇتۇپ تۇرالماسلىقتەك ئەھۋاللار ئومۇميۈزلۈك مەۋجۇت.
ھازىرقى كارخانىچىلىرىمىزنىڭ كۆپىنچىسى 80-يىللاردىن بۇرۇنقىلار بولغاچقا، دەۋرنىڭ تەرەققىيات تېزلىكى بىلەن ئۇلارنىڭ بىلىم قۇرۇلمىسىنىڭ يېڭىلىنىش تېزلىكى گارمونىيەلىك مۇناسىۋەتنى ساقلىماستىن، بەلكى كارخانىچىلارنىڭ بىلىم قۇرۇلمىسى، ساپاسى قاتارلىق جەھەتلەردە دەۋرنىڭ ئېھتىياجىنى قاندۇرالماسلىق ھەم يېتىشىپ چىقىۋاتقان ئىختىساس ئىگىلىرىنى كىچىك چاغلاپ ئۇلارنى ھە دېسە نەزەرىيەچى قاتارغا چىقىرىۋېتىپ، ئۇلارنىڭ ئىلمىي كۆز قاراشلىرىنى، تەكلىپ پىكىرلىرىنى قوبۇل قىلماسلىق، ھازىرقى زامان بازار ئىگىلىكىنىڭ تەرەققىيات يۆنىلىشىگە توغرا دىئاگنۇز قويالماي، پۇرسەتنى قولدىن بېرىپ قويۇش قاتارلىق ئەھۋاللاردىن ساقلىنالماسلىق ئەھۋالىمۇ خېلى ئېغىر. ئەلۋەتتە، يېقىندىن بىرى مەيلى كارخانىچى بولسۇن ياكى شىركەتتە ئىشلەۋاتقانلار بولسۇن ياخشىلىنىش خېلى بولىۋاتىدۇ. نۇرغۇن ئىشلارغا ۋاقىت، جەريان كېتىدۇ. بازار ئىگىلىكىدىكى كەسكىن رىقابەت كارخانىچىلىرىمىزدا خېلى ئويغىنىش، سەگەكلىشىش قەدىمىنى تېزلەتتى ھەم تېزلىتىۋاتىدۇ. شۇڭا بازارنىڭ كۈنسېرى تەرەققىي قىلىشىغا ئەگىشىپ بۇ جەھەتتىكى مەسىلىلەرمۇ تەدرىجى ھەل بولۇپ كەتكۈسى.
تۆتىنچى، كارخانىچىلاردا توغرا بولغان ئېلانشۇناسلىق بىلىملىرى كەمچىل. ھازىر بىزنىڭ كارخانىچىلىرىمىزدا ئېلان ۋە ئۇنىڭ ئۈنۈمى ھەققىدىمۇ ياخشى چۈشەنچە يوق. ئۇلار پەقەت تېلېۋېزىيە ئېلانى، گېزىت-ژۇرنال ئېلانى ۋە ئۇلىنىش تورىدىكى ئېلان بىلەنلا چەكلىنىپ قېلىۋاتىدۇ. يەنى كارخانىچىلار پەقەت قاتتىق ئېلان شەكلىنىلا قوللىنىپ كېلىۋاتىدۇ، ئەمما ھەقىقىي ئۈنۈم بېرىدىغان يۇمشاق ئېلان ئۇقۇمى ھەققىدە چۈشەنچە يوق. يۇمشاق ئېلاننىڭ كۈچىگە ھەرگىز سەل قاراشقا بولمايدۇ. ئەمما بىزنىڭ كارخانىچىلىرىمىزدا تېخى يۇمشاق ئېلان ئۇقۇمى بولمىغاچقا، ئۇلار يۇمشاق ئېلاننىڭ رولى، ئۈنۈمى جەھەتلەردە قاتتىق ئېلاندىن نەچچە ھەسسە ياخشى ئىكەنلىكىنى ھېس قىلىپ بولالمىدى.
يۇمشاق ئېلان مەسىلىسىگە كەلگەندە، بىزنىڭ كارخانىچىلىرىمىز چوقۇم يىراقنى كۆرەلەيدىغان، سەۋرچان بولۇشى كېرەك. چۈنكى يۇمشاق ئېلان قاتتىق ئېلاندەك تېز ئۈنۈم بەرمەيدۇ، ئەمما يۇمشاق ئېلان ئۈنۈمىنى كۆرسەتكەن ۋاقىتتا بولسا، قاتتىق ئېلاننىڭ ئۈنۈمىنى يۇمشاق ئېلان بىلەن سېلىشتۇرۇش ئەسلا مۇمكىن ئەمەس. ئەلۋەتتە، بۇ دېگەنلىك پەقەت يۇمشاق ئېلاننىلا بېرىپ، قاتتىق ئېلان بەرمەسلىك كېرەك دېگەنلىك ئەمەس، چوقۇم قاتتىق ئېلان بىلەن يۇمشاق ئېلاننىڭ نىسبىتىنى ياخشى تەڭشەش، ھەرگىزمۇ قاتتىق ئېلانغىلا كۈچەپ يۇمشاق ئېلانغا سەل قاراشقا بولمايدۇ. ئەڭ مۇھىمى بىزنىڭ كارخانىچىلىرىمىزدا ئەڭ ئالدى بىلەن يۇمشاق ئېلان ئۇقۇمى شەكىللىنىشى، ئاندىن ئۇنى قانداق بېرىش، قەيەردە بېرىش، كىم ئارقىلىق بېرىش دېگەن مەسىلىلەر ھەققىدە ئەستايىدىل ئويلىنىشى كېرەك. بۇ يەردە يۇمشاق ئېلاننىڭ رولى ۋە ئەھمىيىتى ھەققىدە تولۇق سۆزلەپ بولالماسلىقىمىز مۇمكىن. ئۆزۈمنىڭ يۇمشاق ئېلان جەھەتتىكى ئۇزاق ۋاقىتلىق ئىزدىنىشىنىڭ ھاسىلاتى بولغان يۇمشاق ئېلان ھەققىدە تەپسىلىي، كونكرېت، چۈشىنىشلىك بولغان يازمىلىرىم پات ئارىدا تەييارلانسا بەلكىم تورداشلارنىڭ ھوزۇرىغا سۇنۇشۇم مۇمكىن.

مەزكۇر يازمىدا بىر قىسىم ئۇقۇملار ھەققىدە توختالدۇق، تۆۋەندە بۇ ئۇقۇملار ھەققىدە بەزىبىر چۈشەنچىلەر ئۇلىنىش قىلىپ قالدۇرۇلدى، تورداشلار ئەگەر قىزىقسا كۆپرەك كۆرۈپ، چۈشىنىپ باقسا بولىدۇ.
