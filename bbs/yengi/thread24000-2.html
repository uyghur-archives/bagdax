چەتئەلدە ئوقۇيدىغان كەسىپنى تاللىغاندا «تەمىنات» ئەھۋالىغا قانداق قاراش كېرەك


(بېيجىڭ ئىمتىھان گېزىتى خەۋىرى) يېقىندا ئۆتكۈزۈلگەن چەتئەل سەنئەت كۆرگەزمىسىدە، ئوقۇغۇچىلارنىڭ ئاتا-ئانىسى مەلۇم كەسپنى سۈرۈشتە قىلغاندا، كۆپىنچىلىرىنىڭ ئالدى بىلەن سورايدىغىنى بالىلارنىڭ ئىشقا ئورۇنلىشىش ئىستىقبالى جەھەتتىكى مەسىلىلەر بولدى. كەسپىي خادىملارنىڭ ئېيتىشىچە، ھازىر ئاتا-ئانىلار ئىشقا ئورۇنلىشىشنى ئالدىنقى ئورۇنغا قويىدىغان بولۇپ، كەلگۈسىدىكى ئىشقا ئورۇنلىشىش ئەھۋالى، كەسىپىي كىرىم ئەھۋالى قاتارلىقلارنى ئالدىنئالا كۆزىتىدىكەن. بۇنداق قىلغاندا چەتئەلدىكى ئوقۇشنى تۈگەتكەندىن كېيىنكى ئىشقا ئورۇنلىشىش خېيىم-خەتىرىنى تۆۋەنلەتكىلى بولىدىكەن.
ئوخشاش بىر كەسىپنىڭ ئىشقا ئورۇنلىشىش جەھەتتىكى ئىستىقبالى دۆلەت ئىچى ۋە سىرتىدا ئوخشاش بولمايدۇ. مەسىلەن، لايىھىلەش كەسپىدە ئوقۇغان ئوقۇغۇچى بەلكىم ئەنگىلىيىگە بەكرەك مۇۋاپىق كېلىشى مۇمكىن. ئەنگىلىيەنىڭ لاھىيلەش كەسپى،  بولۇپمۇ ھازىرقى زامان سەنئىتى، ھازىرقى زامان لايىھىلەش كەسپى ئىنتايىن تەرەققىي قىلغان. ناۋادا ئوقۇغۇچىلارنىڭ ئىقتىدارى كۈچلۈك بولسا، ناھايىتى ئاسانلا خىزمەتكە ئېرىشەلەيدۇ. ئەنگىلىيەدىكى مەركىزى سان-مارتېن سەنئەت ۋە لايىھىلەش ئىنىستىتۇتىنىڭ تەسۋىر يەتكۈزۈشنى لايىھىلەش كەسپىنى پۈتتۈرگەن دىڭ شياۋيۈەننىڭ تەجرىبىسى بويىچە بولغاندا، بەزى سەنئەت مەكتەپلىرى ئوقۇۋاتقان ئوقۇغۇچىلارغا پراكتىكا قىلىش پۇرسىتى يارىتىپ بېرىدىكەن. يەنى، داڭلىق كارخانىلار پراكتىكانتلارنى قوبۇل قىلىپ ئىشلىتىدىكەن.  ئەگەر ئوقۇغۇچىلار كىيىم-كېچەك لايھىلەش ياكى زىننەتلىك لايىھىلەش، ۋە ياكى قىستۇرما رەسىم كەسپىدە ئوقۇغان بولسىمۇ، ئوقۇش پۈتتۈرگەندىن كېيىن يەرلىكتە خىزمەت تېپىشى بىر قەدەر ئاسانغا چۈشىدىكەن، ھەمدە «تەمىناتى»مۇ يامان ئەمەس بولىدىكەن.
ئوقۇغۇچىلارنىڭ تاللىغان كەسپى كەلگۈسىدىكى ئىشقا ئورۇنلىشىش يۆنىلىشىنى بىۋاستە بەلگىلەيدۇ. چەتئەلدىكى سەنئەت مەكتەپلىرى ئەمەلىي قوللىنىشچانلىققا ئېتىبار بېرىدۇ. تولۇق كۇرس ئوقۇغۇچىلىرى كارخانىغا كىرگەن تەقدىردىمۇ، ئىشقا چىققان تۇنجى كۈنىلا نىمە ئىش قىلىشىنى بىلىشى كېرەك. مانا بۇ كۆپ ساندىكى قوللىنىشچان گۈزەل-سەنئەت مەكتەپلىرى ئوقۇغۇچىلارنىڭ ئەمەلىي ماھارىتىنى يېتىلدۈرۈشتە ئېتىبار بېرىدىغان تەرىپى. كانادادا ئوقۇيدىغان چەتئەللىك ئوقۇغۇچى ليۇ فېيرەن ئۆز ساۋاقداشلىرىنىڭ ئىچىدە ٪90نىڭ ئۆگەنگەن پەنلىرىگە ئائىت كەسىپلەر بىلەن شۇغۇللىنىۋاتقانلىقىنى ئېيتقان.
كەلگۈسىدىكى ئىشقا ئورۇنلىشىش يۆنىلىشىنى تاللىغان كەسىپ بەلگىلىسىمۇ، ئەمما ساھە ئوخشىمىغانلىقتىن، مائاش كىرىمىدىمۇ زور پەرق مەۋجۇت بولىدۇ. ئوقۇغۇچىلارنىڭ ئاتا-ئانىسى كەسىپ تاللاشتىن ئاۋۋال بەزى سانلىق مەلۇماتلاردىن پايدىلانسا بولىدۇ. ئالايلۇق، ئامېرىكا مائاش كىرىمى سىتاتىستېكىسىغا دائىر نوپۇزلۇق تور بېكەت Payscale.com تىن ئايان بولۇشىچە, جۇڭگودا، پروگرامما تۈزگۈچىلەرنىڭ يىللىق ئوتتۇرىچە كىرىمى 149مىڭ يۈەن خەلق پۇلىغا؛ يۇمشاق دېتال ئېنژىنېرىنىڭ تەخمىنەن 119مىڭ 500يۈەن خەلق پۇلىغا؛ قۇرۇلۇش ئېنژىنېرىنىڭ بولسا 203مىڭ 400يۈەن خەلق پۇلىغا توغرا كېلىدىكەن. بەزى دۆلەتلەر يۇقىرى مائاشلىق يوشۇرۇن كۈچكە ئېگە كەسىپلەرنى ئېلان قىلغان. مەسىلەن، ئامېرىكىنىڭ ئالاقىدار ئورگانلىرى كۆرسەتكەن يۇقىرى مائاشلىق كەسپلەر نېفىت قۇرۇلۇشى، ئاۋىئاتسىيە قۇرۇلۇشى، ئىنچىكە ھېسابلاش ماتىماتېكىسى، خىمىيە قۇرۇلۇشى، كومپيۇتېر قۇرۇلۇشى قاتارلىقلارنى ئۆز ئىچىگە ئالىدىكەن.
كەسپتىن خەۋەردار زاتلار بۇنىڭغا قارىتا ئەسكەرتىش بىرىپ مۇنداق كۆرسەتكەن: مائاشنىڭ يۇقىرى-تۆۋەنلىكى كۆپ خىل ئامىلغا مۇناسىۋەتلىك بولىدۇ. ئۇلارنىڭ ئىچىدە كەسىپ تاللاش ھالقىلىق ئامىل بولسىمۇ، ئوقۇغۇچىلار ئوقۇش پۈتتۈرگەندىن كېيىنكى ئەمەلىي خىزمەت ئىقتىدارى ۋە تەجرىبىسى، ئىشقا ئورۇنلاشقان شەھەر، تاللىغان كارخانىنىڭ كۆلىمى، شۇنداقلا تاللىغان دۆلەت قاتارلىقلارمۇ كىرىمنىڭ يۇقىرى-تۆۋەنلىكىگە تەسىر كۆرسىتىدۇ. چەتئەلدە ئوقۇش ئىشلىرى مۇتەخەسسىسلىرى ئوقۇغۇچىلارنى كەسپ تاللىغاندا ئىلمىي تونۇپ، كۆپ تەرەپتىن ئويلىنىشقا دەۋەت قىلىدۇ.
شۇڭلاشقا، ئوقۇغۇچىلار كەسىپ تاللىغاندا مەكتەپنىڭ جۇغراپىيىلىك ئورنىنى ئوبدان ئويلىشىشى كېرەك. مەسىلەن، بوغالتىرلىق كەسپىدە ئوقۇغانلارغا نىسبەتەن، ئۇلارنىڭ كۆڭۈلدىكىدەك ئىشلەتكۈچىلىرى دۆلەت ۋە رايون ھالقىغان چوڭ تىپتىكى بوغالتىرلىق ئىشلىرى ئورنى، ئۇنۋېرستېتلار ياكى ئالاقىدار مائارىپ مەسلىھەتچىلىك مۇلازىمىتى ئورگانلىرى ۋە ئىجتىمائىي مۇلازىمەت ئورگانلىرى ياكى تەشكىلاتلار بولىدۇ. بۇ خىلدىكى ئورگانلارنىڭ كۆپىنچىسى چوڭ شەھەرلەر ياكى ئىقتىسادى تەرەققىي قىلغان رايونلارغا توپلاشقان بولىدۇ. شۇڭا، ئىقتىسادى تەرەققىي قىلغان رايونلاردا مەيلى كەسپى بىلىمنى يۇقىرى كۆتۈرۈش جەھەتتىن بولسۇن، مەيلى كەلگۈسىدىكى ئىشقا ئورۇنلىشىش پۇرسىتىنى تاللاش جەھەتتە بولسۇن ھەممىسىدىلا ئەۋزەللىككە ئېگە.
ئوقۇغۇچىلار كەسىپ تاللىغاندا يەنە مەكتەپ، كەسىپلەرنىڭ تۇتقان ئورنى، دەرسلىك ئورۇنلاشتۇرۇلۇشى، ئېلىنىدىغان ھەقلەر قاتارلىق ئامىللارنىمۇ ئوبدان كۆزىتىشى كېرەك. سانائەت پەنلەر مەكتەپلىرىنى تاللىغاندا يەنە كەسىپنى تەتقىق قىلىش ئەھۋالىغىمۇ قاراش كېرەك. بۇنىڭدىن باشقا، كەسىپ تاللىغاندا، پراكتىكا پۇرسىتى، ئالاھىدە ئوقۇتۇش ياكى تەتقىقات ئورۇنلاشتۇرۇلۇشى، دەرستىن سىرت قوشۇمچە ئىشلەش پۇرسىتى، مەكتەپنىڭ كۆلىمى ۋە قاتتىق-يۇمشاق دېتال ئەسلىھەلىرى، ئوقۇش پۈتتۈرۈش ۋە ئىشقا ئورۇنلىشىش نىسبىتى قاتارلىقلارمۇ ئوقۇغۇچىلار ئەستايىدىل ئويلىنىشقا تېگىشلىك ئامىللاردۇر.

(بېيجىڭ ئىمتىھان گېزىتى خەۋىرى) يېقىندا ئۆتكۈزۈلگەن چەتئەل سەنئەت كۆرگەزمىسىدە، ئوقۇغۇچىلارنىڭ ئاتا-ئانىسى مەلۇم كەسپنى سۈرۈشتە قىلغاندا، كۆپىنچىلىرىنىڭ ئالدى بىلەن سورايدىغىنى بالىلارنىڭ ئىشقا ئورۇنلىشىش ئىستىقبالى جەھەتتىكى مەسىلىلەر بولدى. كەسپىي خادىملارنىڭ ئېيتىشىچە، ھازىر ئاتا-ئانىلار ئىشقا ئورۇنلىشىشنى ئالدىنقى ئورۇنغا قويىدىغان بولۇپ، كەلگۈسىدىكى ئىشقا ئورۇنلىشىش ئەھۋالى، كەسىپىي كىرىم ئەھۋالى قاتارلىقلارنى ئالدىنئالا كۆزىتىدىكەن. بۇنداق قىلغاندا چەتئەلدىكى ئوقۇشنى تۈگەتكەندىن كېيىنكى ئىشقا ئورۇنلىشىش خېيىم-خەتىرىنى تۆۋەنلەتكىلى بولىدىكەن.
ئوخشاش بىر كەسىپنىڭ ئىشقا ئورۇنلىشىش جەھەتتىكى ئىستىقبالى دۆلەت ئىچى ۋە سىرتىدا ئوخشاش بولمايدۇ. مەسىلەن، لايىھىلەش كەسپىدە ئوقۇغان ئوقۇغۇچى بەلكىم ئەنگىلىيىگە بەكرەك مۇۋاپىق كېلىشى مۇمكىن. ئەنگىلىيەنىڭ لاھىيلەش كەسپى،  بولۇپمۇ ھازىرقى زامان سەنئىتى، ھازىرقى زامان لايىھىلەش كەسپى ئىنتايىن تەرەققىي قىلغان. ناۋادا ئوقۇغۇچىلارنىڭ ئىقتىدارى كۈچلۈك بولسا، ناھايىتى ئاسانلا خىزمەتكە ئېرىشەلەيدۇ. ئەنگىلىيەدىكى مەركىزى سان-مارتېن سەنئەت ۋە لايىھىلەش ئىنىستىتۇتىنىڭ تەسۋىر يەتكۈزۈشنى لايىھىلەش كەسپىنى پۈتتۈرگەن دىڭ شياۋيۈەننىڭ تەجرىبىسى بويىچە بولغاندا، بەزى سەنئەت مەكتەپلىرى ئوقۇۋاتقان ئوقۇغۇچىلارغا پراكتىكا قىلىش پۇرسىتى يارىتىپ بېرىدىكەن. يەنى، داڭلىق كارخانىلار پراكتىكانتلارنى قوبۇل قىلىپ ئىشلىتىدىكەن.  ئەگەر ئوقۇغۇچىلار كىيىم-كېچەك لايھىلەش ياكى زىننەتلىك لايىھىلەش، ۋە ياكى قىستۇرما رەسىم كەسپىدە ئوقۇغان بولسىمۇ، ئوقۇش پۈتتۈرگەندىن كېيىن يەرلىكتە خىزمەت تېپىشى بىر قەدەر ئاسانغا چۈشىدىكەن، ھەمدە «تەمىناتى»مۇ يامان ئەمەس بولىدىكەن.
ئوقۇغۇچىلارنىڭ تاللىغان كەسپى كەلگۈسىدىكى ئىشقا ئورۇنلىشىش يۆنىلىشىنى بىۋاستە بەلگىلەيدۇ. چەتئەلدىكى سەنئەت مەكتەپلىرى ئەمەلىي قوللىنىشچانلىققا ئېتىبار بېرىدۇ. تولۇق كۇرس ئوقۇغۇچىلىرى كارخانىغا كىرگەن تەقدىردىمۇ، ئىشقا چىققان تۇنجى كۈنىلا نىمە ئىش قىلىشىنى بىلىشى كېرەك. مانا بۇ كۆپ ساندىكى قوللىنىشچان گۈزەل-سەنئەت مەكتەپلىرى ئوقۇغۇچىلارنىڭ ئەمەلىي ماھارىتىنى يېتىلدۈرۈشتە ئېتىبار بېرىدىغان تەرىپى. كانادادا ئوقۇيدىغان چەتئەللىك ئوقۇغۇچى ليۇ فېيرەن ئۆز ساۋاقداشلىرىنىڭ ئىچىدە ٪90نىڭ ئۆگەنگەن پەنلىرىگە ئائىت كەسىپلەر بىلەن شۇغۇللىنىۋاتقانلىقىنى ئېيتقان.
كەلگۈسىدىكى ئىشقا ئورۇنلىشىش يۆنىلىشىنى تاللىغان كەسىپ بەلگىلىسىمۇ، ئەمما ساھە ئوخشىمىغانلىقتىن، مائاش كىرىمىدىمۇ زور پەرق مەۋجۇت بولىدۇ. ئوقۇغۇچىلارنىڭ ئاتا-ئانىسى كەسىپ تاللاشتىن ئاۋۋال بەزى سانلىق مەلۇماتلاردىن پايدىلانسا بولىدۇ. ئالايلۇق، ئامېرىكا مائاش كىرىمى سىتاتىستېكىسىغا دائىر نوپۇزلۇق تور بېكەت Payscale.com تىن ئايان بولۇشىچە, جۇڭگودا، پروگرامما تۈزگۈچىلەرنىڭ يىللىق ئوتتۇرىچە كىرىمى 149مىڭ يۈەن خەلق پۇلىغا؛ يۇمشاق دېتال ئېنژىنېرىنىڭ تەخمىنەن 119مىڭ 500يۈەن خەلق پۇلىغا؛ قۇرۇلۇش ئېنژىنېرىنىڭ بولسا 203مىڭ 400يۈەن خەلق پۇلىغا توغرا كېلىدىكەن. بەزى دۆلەتلەر يۇقىرى مائاشلىق يوشۇرۇن كۈچكە ئېگە كەسىپلەرنى ئېلان قىلغان. مەسىلەن، ئامېرىكىنىڭ ئالاقىدار ئورگانلىرى كۆرسەتكەن يۇقىرى مائاشلىق كەسپلەر نېفىت قۇرۇلۇشى، ئاۋىئاتسىيە قۇرۇلۇشى، ئىنچىكە ھېسابلاش ماتىماتېكىسى، خىمىيە قۇرۇلۇشى، كومپيۇتېر قۇرۇلۇشى قاتارلىقلارنى ئۆز ئىچىگە ئالىدىكەن.
كەسپتىن خەۋەردار زاتلار بۇنىڭغا قارىتا ئەسكەرتىش بىرىپ مۇنداق كۆرسەتكەن: مائاشنىڭ يۇقىرى-تۆۋەنلىكى كۆپ خىل ئامىلغا مۇناسىۋەتلىك بولىدۇ. ئۇلارنىڭ ئىچىدە كەسىپ تاللاش ھالقىلىق ئامىل بولسىمۇ، ئوقۇغۇچىلار ئوقۇش پۈتتۈرگەندىن كېيىنكى ئەمەلىي خىزمەت ئىقتىدارى ۋە تەجرىبىسى، ئىشقا ئورۇنلاشقان شەھەر، تاللىغان كارخانىنىڭ كۆلىمى، شۇنداقلا تاللىغان دۆلەت قاتارلىقلارمۇ كىرىمنىڭ يۇقىرى-تۆۋەنلىكىگە تەسىر كۆرسىتىدۇ. چەتئەلدە ئوقۇش ئىشلىرى مۇتەخەسسىسلىرى ئوقۇغۇچىلارنى كەسپ تاللىغاندا ئىلمىي تونۇپ، كۆپ تەرەپتىن ئويلىنىشقا دەۋەت قىلىدۇ.
شۇڭلاشقا، ئوقۇغۇچىلار كەسىپ تاللىغاندا مەكتەپنىڭ جۇغراپىيىلىك ئورنىنى ئوبدان ئويلىشىشى كېرەك. مەسىلەن، بوغالتىرلىق كەسپىدە ئوقۇغانلارغا نىسبەتەن، ئۇلارنىڭ كۆڭۈلدىكىدەك ئىشلەتكۈچىلىرى دۆلەت ۋە رايون ھالقىغان چوڭ تىپتىكى بوغالتىرلىق ئىشلىرى ئورنى، ئۇنۋېرستېتلار ياكى ئالاقىدار مائارىپ مەسلىھەتچىلىك مۇلازىمىتى ئورگانلىرى ۋە ئىجتىمائىي مۇلازىمەت ئورگانلىرى ياكى تەشكىلاتلار بولىدۇ. بۇ خىلدىكى ئورگانلارنىڭ كۆپىنچىسى چوڭ شەھەرلەر ياكى ئىقتىسادى تەرەققىي قىلغان رايونلارغا توپلاشقان بولىدۇ. شۇڭا، ئىقتىسادى تەرەققىي قىلغان رايونلاردا مەيلى كەسپى بىلىمنى يۇقىرى كۆتۈرۈش جەھەتتىن بولسۇن، مەيلى كەلگۈسىدىكى ئىشقا ئورۇنلىشىش پۇرسىتىنى تاللاش جەھەتتە بولسۇن ھەممىسىدىلا ئەۋزەللىككە ئېگە.
ئوقۇغۇچىلار كەسىپ تاللىغاندا يەنە مەكتەپ، كەسىپلەرنىڭ تۇتقان ئورنى، دەرسلىك ئورۇنلاشتۇرۇلۇشى، ئېلىنىدىغان ھەقلەر قاتارلىق ئامىللارنىمۇ ئوبدان كۆزىتىشى كېرەك. سانائەت پەنلەر مەكتەپلىرىنى تاللىغاندا يەنە كەسىپنى تەتقىق قىلىش ئەھۋالىغىمۇ قاراش كېرەك. بۇنىڭدىن باشقا، كەسىپ تاللىغاندا، پراكتىكا پۇرسىتى، ئالاھىدە ئوقۇتۇش ياكى تەتقىقات ئورۇنلاشتۇرۇلۇشى، دەرستىن سىرت قوشۇمچە ئىشلەش پۇرسىتى، مەكتەپنىڭ كۆلىمى ۋە قاتتىق-يۇمشاق دېتال ئەسلىھەلىرى، ئوقۇش پۈتتۈرۈش ۋە ئىشقا ئورۇنلىشىش نىسبىتى قاتارلىقلارمۇ ئوقۇغۇچىلار ئەستايىدىل ئويلىنىشقا تېگىشلىك ئامىللاردۇر.

2014-يىلى كانادانىڭ ئوقۇش ۋىزىسى 980يۈەنگە تۆۋەنلىدى

كانادادا ئوقۇشقا ئىلتىماس قىلماقچى بولغان ساۋاقداشلار ۋىزىنى مۇۋەپپەقىيەتلىك ئېلىش ئۈچۈن، ۋىزا ئىلتىماس قىلىش ئۇچۇرىغا ھەر ۋاقت دىققەت قىلىشى زۆرۈر.


جۇڭگو مائارىپ ۋە پەن تەتقىقات تورى خەۋىرى: ئىگىلىشىمىزچە، كانادادا ئوقۇش ۋىزىسىنىڭ تەستىقلىنىش باھاسى 980يۈەنگە تەڭشەلگەن. ئەمما ئەنگىلىيەنىڭ ۋىزا ھەققى ئەسلىدىكى 2980يۈەندىن 3129يۈەنگە ئۆستۈرۈلگەن.
ئەنگىلىيىگە بېرىپ تولۇق كۇرس ياكى ماگىستېر ئاسپىرانتلىقتا ئوقۇماقچى بولغانلار تۆتىنچى دەرىجىدىكى ئادەتتىكى ئوقۇغۇچىلار ۋىزىسىغا ئىلتىماس قىلىدۇ. ئۇنىڭ ئېنگىلىزچە ئاتىلىشى «Tier4GeneralStudentVisa» بولۇپ، باھاسى 300 فوندېستېرلىڭغا تەڭ كېلىدۇ. يېقىندا، فوندېستېرىلىڭىنىڭ خەلق پۇلىغا ئايرىش نىسبىتى داۋاملىق ئۆسكەنلىكتىن، تۆتىنجى دەرىجىدىكى ئادەتتىكى ئوقۇغۇچىلار ۋىزىسىدىمۇ ئۆسۈش كۆرۈلگەن.
تۆتىنجى دەرىجىدىكى ئادەتتىكى ئوقۇغۇچىلار ۋىزىسى باھاسىدا ئۆسۈش كۆرۈلگەندىن باشقا، قالغان تۈردىكى ۋىزىلاردىمۇ ئۆسۈش يۈزلىنىشى كۆرۈلگەن: ئوقۇغۇچىلار زىيارەت ۋىزىسىغا ئىلتىماس قىلىش ھەققى 840يۈەن؛ ئوقۇغۇچىلار زىيارەت ۋىزا ھەققى 1512يۈەن؛ بالىلار زىيارەت ۋىزىسى 840يۈەن؛ تۇغقان يوقلاش ۋىزىسى 840يۈەن بولغان.
كانادانىڭ جۇڭگودا تۇرۇشلۇق ئەلچىخانىسى يېقىندا ئېلان قىلغان ئۇچۇردىن ئايان بولۇشىچە، كانادادا ئوقۇش ۋىزىسىنى تەستىقلاش ھەققى ئەسلىدىكى 1040يۈەندىن 980يۈەنگە،  كۆپ قېتىم بېرىپ-كىلىشنى تەستىقلاش ھەققى ئەسلىدىكى 1200يۈەندىن 1130يۈەنگە تەڭشەلگەن.  بۇنىڭدىن باشقا، كانادا ۋىزا ئىلتىماس قىلىش مەركىزى نۆۋەتتە پەقەت خەلق پۇلى ئارقىلىق تۆلەش شەكلىنىلا قوبۇل قىلىدىغان بولۇپ، ئۇمۇمىي پۇل ۋىزا مەركىزىنىڭ 250يۈەن مۇلازىمەت ھەققىنىمۇ ئۆز ئىچىگە ئالىدىكەن.
كانادادا ئوقۇش ۋىزىسى ئىلتىماس قىلغان ئوقۇغۇچىلاردىن كاپالەت پۇلى كۆرسىتىش، شۇنداقلا يەنە چوقۇم 12ئاي ساقلانغانلىق تارىخى بولۇشى  تەلەپ قىلىنىدىكەن. قىسقا مۇددەتلىك زور سوممىدىكى ئامانەت ياكى ساقلىنىش تارىخى بولمىغان ئامانەت سوممىسى ئەلچىخانا تەرىپىدىن ئىقتىسادىي كىرىمى يېتىشمەيدۇ ياكى مۇقىم ئەمەس دەپ قارىلىدىكەن. نورمال ئەھۋالدا، ئوقۇغۇچىلارنىڭ ئاتا-ئانىسى ئالدىنئالا پىلان تۈزۈپ، ئاندىن مەبلەغ غەملەپ، ئامانەت ئىسپاتى ۋە ھىسابات تالونى چىقىرىپ تەييارلىق قىلىشى كېرەك ئىكەن.

چەتئەلدىن گىرمانىيىگە ئوقۇشقا چىقىش قىزغىنلىقى يۇقىرىلىماقتا

گىرمانىيە چەتئەللىك ئوقۇغۇچىلارنىڭ بارغانسېرى ياقتۇرۇشىغا ئېرىشىۋاتقان بولۇپ، ھازىر 200 مىڭدىن كۆپرەك يات ئەللەر ئوقۇغۇچىلىرى گىرمانىيە ئالىي مەكتەپلىرىدە ئىلىم تەھسىل قىلماقتا ئىكەن.


بېيجىڭ ئىمتىھان گېزىتى خەۋىرى: گىرمانىيە ئالىي مەكتەپ ۋە ئىلىم-پەن تەتقىقات مەركىزى يېقىندا ئېلان قىلغان بىر تۈرلۈك تەكشۈرۈش دوكلاتىدا كۆرسىتىلىشىچە، نۆۋەتتە گىرمانىيىدە ئوقۇش قىزغىنلىقى يۇقىرىلاشقا قاراپ يۈزلىنىۋاتقان بولۇپ، چەتئەللىك ئوقۇغۇچىلارنى دۆلەت بويىچە تەرتىپكە تىزغاندا، گىرمانىيىگە 30مىڭدىن كۆپرەك ئوقۇغۇچى چىقارغان جۇڭگو ئەڭ ئالدىغا تىزىلغان. شەرقىي ياۋروپا دۆلەتلىرى جۇڭگونىڭ كەينىدىن ئورۇن ئالغان.
تەكشۈرۈش دوكلاتىدىن ئايان بولۇشىچە، بۇ ئوقۇغۇچىلارنىڭ گىرمانىيەنى تاللىشى ھەرگىزمۇ تاسادىپىيلىقتىن كېلىپ چىقمىغانىكەن. زىيارەت قىلىنغان ئۈچتىن ئىككى قىسىم (٪61) ئوقۇغۇچىنىڭ بىلدۈرۈشىچە، گىرمانىيە ئۇلارنىڭ ئالدىنقى تاللىشى ئىكەن. گىرمانىيىدە ئوقۇۋاتقان ٪82 چەتئەللىك ئوقۇغۇچى بۇ يەردە ئوقۇش ئارقىلىق تېخىمۇ ياخشى خىزمەت پۇرسىتىگە ئېرىشىشنى ئارزۇ قىلىدىكەن. ئۇلارنىڭ ئىچىدە ٪25 ئوقۇغۇچى قۇرۇلۇش تۈرىدىكى كەسىپلەرنى تاللىغان بولۇپ، بۇنداق بولۇشىدا قۇرۇلۇش تۈرىدىكى كەسىپلەرنىڭ جەلپ قىلىش كۈچى ئەڭ يۇقىرى ئىكەن. ئۇنىڭدىن قالسا مەدەنىيەت ۋە تىل تۈرىدىكى كەسىپلەر كۆپرەك سالماقنى ئىگىلەيدىكەن.
بۇنىڭدىن باشقا، چەتئەللىك ئوقۇغۇچىلارنىڭ دەرستىن سىرت ئىشلەش ھادىسىسى ئىنتايىن ئومۇملاشقان. تەخمىنەن يېرىم نىسبەتتىكى ئوقۇغۇچىلار مۇشۇ ئۇسۇلغا تايىنىپ ئوقۇشىنى قامدايدۇ.. بۇ جەھەتتە، بويتاق ئوقۇغۇچىلارنىڭ ھەر ئايلىق ئوتتۇرىچە چىقىمى 749 ياۋرودىن ئاشىدىكەن. كۆپ ساندىكى ئوقۇغۇچىلار قوشۇمچە ئىشلەپ ئوقۇشىنى قامدىغاندىن باشقا يەنە تۇرالغۇ ئىزدەش بىلەنمۇ ئالدىراش بولۇپ كېتىدىكەن.  ٪42 كىشى بۇ مەسىلىنى چوڭ بىر مەسىلە دەپ قارايدىكەن. بۇنىڭدىن باشقا، ئۈچتىن بىر قىسىم  زىيارەت قىلىنغۇچىلارنىڭ ئېيتىشىچە، ئۇلار ئۆزىگە قويۇلغان ئۆگىنىش نەتىجىسىگە يىتىش تەلىپىنى تاماملاشتا خېلى قىينىلىدىكەن. گىرمانىيە ئالىي مەكتەپلىرى بۇ مەسىلىنى تونۇپ يەتكەن بولۇپ، ئۇلار  تىل دەرسلىكلىرىنى تەسىس قىلىش، ئوقۇش ئىشلىرىنى سۈرۈشتۈرۈش مۇلازىمىتى بىلەن تەمىنلەش ۋە ئوقۇغۇچىلارغا ياردەملىشىپ مەمۇرىي جەھەتتىكى رەسمىيەتلەرنى بېجىرىپ بېرىش قاتارلىق كۆپ خىل مۇلازىمەتنى يولغا قويۇش ئارقىلىق، بىر قىسىم مەسىلىلەرنى ھەل قىلماقچى بولغان. تەكشۈرۈش نەتىجىسىدە، بۇ تەدبىرلەرنىڭ ئىنتايىن ئۈنۈملۈك بولغانلىقى مەلۇم بولغان، يەنى، ٪70كە يېقىن چەتئەللىك ئوقۇغۇچى، تۇرۇش ھوقۇقى ياكى ئىقتىسادىي ياردەمگە ئېرىشىش قاتارلىق ئۇچۇرلارغا ئېگە بولۇش جەھەتلەردىن ئىنتايىن رازى ئىكەنلىكنى بىلدۈرۈشكەن.
گىرمانىيە چەتئەللىك ئوقۇغۇچىلارنىڭ بارغانسېرى ياقتۇرۇشىغا ئېرىشىۋاتقان بولۇپ، ھازىر 200 مىڭدىن كۆپرەك يات ئەللەر ئوقۇغۇچىلىرى گىرمانىيە ئالىي مەكتەپلىرىدە ئىلىم تەھسىل قىلماقتا ئىكەن.
