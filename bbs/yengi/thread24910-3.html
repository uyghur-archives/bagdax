ماۋۇ ئەزھەرىيلەر ئاڭلاڭلار ، ئىسلام دىنىدا مەزھەب ۋە قان تۆكۈشتىن باشقا نەرسە يوقما ؟ مۇشۇنداق مەزھەبچىلىك توغرىسىدىكى نەرسە يوللىغۇچە ،خەلقنى تىنچلىققا ،مەرىپەتكە ئۈندەيدىغان تېمىلارنى يوللىساڭلار بولمامدۇ؟ ئۇيغۇرلار سۇننىيمۇ ،ۋاھابىيمۇ ،شىئەمۇ ئەمەس ، كۆپىنچىسى بۇنداق نەرسىلەرنى چۈشەنمەيدىغان چۈشىنىشنىمۇ خالىمايدىغان مەزھەبسىز مۇسۇلمانلار . مانداق مەزھەپ پەزھەب دىگەنلەر ماڭا ئوخشاش ئىسلامغا گۇمانىي ۋە تەنقىدىي نەزەر بىلەن قارايدىغانلارنى تېخىمۇ ئىسلامغا بولغان گۇماننى چوڭقۇرلاشتۇرىۋېتىدىكەن  ، شىئە بولسا نىمە بوپتۇ ،سۇننىي بولسىچۇ ؟ناسارا بولسا نىمە بوپتۇ ياكى يەھۇدىي ،بۇددىسىت ،دىنسىز بولسىچۇ ؟ ھەممىمىز ئىنسان ، بىر بىرمىزنىڭ جىسمانىي ھاياتىغا ،مال مۈلكىگە ،ئىنسانىي ئابرويىغا دەخلى تەرۈز قىلماي ،باراۋەر دوستانە ئۆتكەنگە نىمە يەتمەيدۇ ؟
ھازىرقى دۇنيادا ئىنسانپەرۋەرلىكنى تەكىتلەش ھەممىدىن مۇھىم ....مۇشۇنداق ئۇ ئېقىم بۇ مەزھەب دەيدىغان ئوتتۇرا شەرقنىڭ مەينەت ئىدىيىلىرىنى دىيارىمىزغا ئىمپورت قىلماڭلار ، بولدى بىر ئۆمۈر مۇشۇنداق مەزھەبچىلىكنى بىلمەي چۈشەنمەي ئۆتۈپ كېتىلى ....مۇشۇنداق نەرسىلەرنى بىلمەي ، گۇناھدىن يىراق يۈرۈپ ، بەش ۋاق نامىزىنى ئوقۇپ ،خەققە زىيان سالماي مۇسۇلمان بولغىلى بولمامدۇيا؟
