مەن ئەمۋاينى سىزدىن ياخشى چۈشەنگىنىم ئۈچۈن مۇنازىرىگە سۆز قىستۇرۇپ قويدۇم.
ئەمۋاي شىركىتىنىڭ تۈزۈلمىسى نوبىل مۇكاپاتىغا تەۋسىيە قىلىنغان،
ئۇنىڭ ، ئادىل ، قانۇنلۇق ،ئېچىۋېتىلگەن ئىقتىسادىي تۈزۈلمىسىنى نۇرغۇن شىركەتلەر ئەندىزە قىلىۋاتىدۇ.
ئەمۋاي باشقىلارغا ياردەم قىلش بەدىلىگە باشقىلارنىمۇ  ئۆزىنىمۇ كىرىمگە ئېرىشتۈرىدىغان ،ئىنسانىيەتكە ساغلاملىق تەقدىم قىلىدىغان  بىر كەسىپ ،
ئەمۋاي قىلغۇچىلارمۇ ھەرخىل بولىدۇ ، بەزىلىرى شىركەت تۈزۈلمىسىگە بوي سۇنۇپ
ئىستىمالچىلارنىڭ جانىجان مەنپەتىنى ئاساس قىلىپ سەمىيمىلىك بىلەن ئىشلەيدۇ ،
ھەم باشقىلار بىلەن ھەمكارلىشىپ تەڭ باراۋەر مەنپەتلىنىدۇ.
ئاز ئىشلىگەنلەر ئاز ئالىدۇ  ،كۆپ ئىشلىگەنلەر كۆپ ئالىدۇ ،
بەزى ئەمۋايچىلار ئۆزىنىڭلا مەنپەتىنى ئويلىغاچقا ،ياكى شۇنداق يىتەكچىلەرگە ئۇچراپ قېلىپ
سىزگە ئوخشاش يېرىم يولدا چۈشۈپ قالىدۇ ،ئادەمنى بەك تاللايدىغان كەسىپ بۇ ،
ئادىمىيلىگى بولمىغانلار بۇ كەسىپكە لايىق كەلمەيدۇ...
ئەمۋايدا ئىستىمالچى بولغانلار شۇغۇللۇنغۇچىمۇ بولالايدۇ،
بۇ كەسىپكە دىپلوم كەتمەيدۇ ئاددى كىشىلەرمۇ  كىرىپ قىلسا بولىۋېرىدۇ ، پەقەت ساۋادى يوقلارمۇ باشقىلارنىڭ ياردىمىدە ئۆزلۈكسىز ئۆگۈنۈپ بۇ كەسىپتە نەتىجە قازىنالايدۇ....
بۇرۇن كىرگەنلەر كېيىنكىلەرنى يېتەكلەپ ماڭىدۇ ،تەجرىبىلىرىنى ئۆگىتىدۇ ،خېىدارلارغا مۇلازىمەت قىلىشتا دۇنيادا ھىچقانداق شىركەت ئەمۋايغا يەتمەيدۇ .
بىرسىنى قويۇپ مال ئالغۇزۇپ پۇل تېپىش ئەمۋاي پىرىنسىپىدا يوق ،شۇنداقلار بايقىلىپ قالسا
(باشقىلارنى زىيانغا ئۇچراتقانلار) شىركەت بىلەن تۈزىگەن توختام بىكار قىلىنىپ مۇئاشى توختايدۇ ھەم شىركەتكە مەڭگۈ كىرەلمەيدۇ ....
پىرامىدا ، شاخلىتىش دىگەنلەر ئەمۋايغا ماس كەلمەيدۇ.
مەن بۇ كەسىپ بىلەن شۇغۇللىنىۋاتقىلى 4 يىل بولدى،
مېنى تونۇشتۇرۇپ بۇ شىركەتكە ئېلىپ كىرگەن ئوقۇتقۇچۇم ھازىر مەنچىلىك مۇئاش ئالالمايدۇ
ئەگەر پىرامىدا بولسا ئۇ مەندىن يۇققۇرى مۇئاش ئالغان بولاتتى.
شاخلىتىش كەسپىدىمۇ بۇرۇن كىرگەنلەر كېيىنكىلەردىن كۆپ ئالىدۇ ...
ئەمما ئەمۋايدا ئۇنداق ئەمەس .
مۇھىت (سۇ، ھاۋا،تۇپراق، يىمەك -ئىچمەك) بۇلغانغان بۈگۈنكى كۈندە كىشىلەر نۇرغۇن كېسەللەرگە گىرىپتار بولماقتا ، دوختۇرخانىلار بارغانچە تەرەققى قىلىنىپ بىنالىرى ئىگىزلەپ
ئالىدىغان پۇللىرىمۇ ئىگىزلەپ ، تەرەققى قىلىپ كەتتى ،ئەمما كېسەللەر ئازايماي كۆپىيىپ كېتىۋاتىدۇ.
ئاددى بىر ۋىتامىننىڭ كاملىقى تۈپەيلىدىن دىئابىت كېسەللىگىگە گىرىپتار بولىدىغىنىنى بىلمەيدىغانلار نۇرغۇنغۇ بىزدە.
خۇلاسەم :

مەن بىر مۆئمىن مۇسۇلمانمەن ،يىگانە  ئاللاھقا  ، ئاللاھنىڭ پەرىشتىلىرىگە ،پەيغەمبەرلىرىگە جۈملىدىن پەيغەمبىرىمىز مۇھەممەت ئەلەيھىسسالامغا  ، ئاللاھنىڭ  كىتاپلىرىغا (قۇرئانغا)  ئىمان كەلتۈرىمەن .
قۇرئاندا ھارام قىلىنغانلارنى بىزگە ھارام دەپ بىلىمەن.
ئۆزۈم شۇغۇللىنىۋاتقان كەسپىمنى ئىنسانىيەتنىڭ ساغلاملىقىغا پايدىلىق بولغانلىقى ئۈچۈن قىلغانمەن .
پەتىۋانى ئەلۋەتتە ئالىملار چىقىرىدۇ ...لېكىن قۇرئانى كېرىمدە مۇنداق سۈرىلەرمۇ  بار .

ئاللاھ تەئالا مۇنداق دەيدۇ:
«تىللىرىڭلار توقۇغان يالغانغا ئاساسلىنىپ: ‹ئۇ ھالال، بۇ ھارام› دېمەڭلار. بولمىسا ئاللاھقا بوھتان چاپلىغان بولىسىلەر. ئاللاھقا بوھتان چاپلىغانلار ھەرگىزمۇ مۇرادىغا يېتەلمەيدۇ»- نەھل سۈرىسى 116- ئايەت.

ئاللاھ تەئالا مۇنداق دەيدۇ:
«ئى مۇئمىنلار! ئۆزۈڭلار قىلمايدىغان ئىشنى نېمە ئۈچۈن سۆزلەيسىلەر؟ ئۆزۈڭلار قىلمايدىغان ئىشنى سۆزلىشىڭلار ئاللاھنىڭ قاتتىق غەزىپىنى قوزغايدۇ»- سەف، 61/2-3.

ئاللاھ تەئالا مۇنداق دەيدۇ:
«ئى مۇئمىنلار! ئۆزۈڭلارنى گۇناھتىن ساقلاڭلار، سىلەر قاچانىكى توغرا يولدا بولساڭلار، باشقىلارنىڭ ئاداشقىنى سىلەرگە زىيان يەتكۈزەلمەيدۇ. سىلەر ھەممىڭلار ئاللاھنىڭ دەرگاھىغا قايتىسىلەر، ئاللاھ قىلمىشىڭلارنى سىلەرگە ئېيتىپ بېرىدۇ» –مائىدە، 5/105.

ئەمۋاينىڭ قىسقىچە تارىخى

ئەمۋاي شىركىتىنىڭ باش شىتابى-ئامىرىكا دۆلىتىنڭ مىچىگان شىتاتىنىڭ ئادا شەھرىدە

شىركەت ئىسمى ئىنگىلىزچە amway،خەنزۇچە 安利

ئەمۋاي شىركىتى  ئامىرىكا دۆلىتىنڭ مىچىگان شىتاتىنىڭ ئادا شەھرىدە قۇرۇلغان بولۇپ، ئەمۋاينى قۇرغۇچىلار جىك ۋىن ئەنلۇ (Jay Van Ansel)بىلەن رىچارد دېۋىس(Rich Devos) لار سەمىمى ھەمكارلىق ئاساسىدا شىرىكلىشىپ قۇرغان،ئۇلار ھەر ئىككىلىسى گوللاندىيەدىن كۆچۈپ كەلگەن كۆچمەنلەرنىڭ ئۇرۇق ئەۋلادىدىن بولۇپ،كىچىكىدىن تارتىپ بىرلىكتە چوڭ بولغان.ئوخشاش ئۆسۈپ يىتىلىش تارىخى ۋە گوللاندىيەلىكنىڭ ئۇرۇق ئەۋلادىدىن بولۇشتىن ئىبارەت شەخسىي خاراكتىرى كيىنچە ئۇلارنىڭ ھەمكارلىشپ شىرىكلىشىغا ئاساس سالغان. جىك ۋىن ئەنلۇ  (Van Ansel Jay) ئەپەندى 2004-يىلى ۋاپات بولغان.ھازىر ئەمۋاي شىركىتىنى ئۇلارنىڭ پەرزەنىتلىرى باشقۇرىۋاتىدۇ.شىتىف ۋىن ئەنلۇ ئەپەندى ئامىرىكا ئەمۋاي شىركىتىنىڭ مۇدىرىيەت رەئىسى.دى دىۋىس ئەپەندى ئامىرىكا ئەمۋاي شىركىتىنىڭ باش ئىش بىجىرگۈچىسى.

ئەمۋاي شىركىتى كۆپ قاتلاملىق ھەق ھىساپلاش تۈزىمىدىكى دۇنياۋىي تىجارەت ئەندىسىزىسىنى ئىجاد قلىغان بولۇپ،دۇنيادىكى ھەر خىل چوڭ تىجارەت شىركەتلىرى ئۈگىنىدىغان ئەندىزىسگە ئايلانغان،شۇنىڭ بىلەن بىرگە جۇڭگو خەلق ئونۋىرسىتىتى تەرىپىدىن  MBAدەرىسلىكىگە كىرگۈزۈلگەن.

ئەمۋاينىڭ يىلنامىسى



1959-يىلى ياش جىك ۋىن ئەنلۇ (Jay Van Ansel)بىلەن رىچارد دېۋىس(Rich Devos) ئامىرىكا دۆلىتىنڭ مىچىگان شىتاتىنىڭ ئادا شەھرىدىكى بىر ئۆينىڭ يەر ئاستى ئۆيىدە قول تۇتۇشۇپ ئەمۋاي شىركىتىنى قۇرۇپ چىقتى ھەمدە مۇھىت ئاسىرايدىغان ئۈنىمى ئالاھىدە يۇقۇرىي بولغان كۆپ خىل ئىقتىدارلىق قۇيۇقلاندۇرۇلغان تازىلاش سۇيۇقلىقىنى ئىشلەپچىقىرىپ مۇۋەپىقىيەتلىك ھالدا بازارغا سالدى.

1962-يىلى تىجارىتىنى كاناداغا كىڭەيىتتى،1971-يىلى شىمالىي ئامرىكىدىن باشقا تۇنجى شۆبە شىركەت ئاۋىستىرالىيەدە تىجارەت باشلىدى.

1972-يىلى نۈچۈرلەت ئوزۇقلۇق تولۇقلاش يىمەكلىك شىركىتى رەسمىي ئەمۋاي شىركىتىنىڭ تارماق ئاپپاراتىغا ئايلاندى.

1980-يىلى يىللىق تىجارەت سوممىسى بىر مىليارىد دوللارلىق چوڭ ئۆتكەلنى ئاشۇرۇپ ئورۇنلىدى.

1989-يىلى بىرلەشكەن دۆلەتلەر تەشكىلاتىنىڭ 《مۇھىت ئاسراش مۇۋەپىقىيەت مۇكاپاتى》 غا ئىرىشىپ،ئىككىنىچى ئالاھىدە مۇكاپاتقا ئىرىشكەن خۇسۇسىي شىركەت بولۇپ قالدى.

1995-يىلى گۇاڭجو ئىقتىسادىي تىخنىكا تەرەققىيات رايۇنىغا رەسمىي زاۋۇت قۇرۇپ مەبلەغ سىلىپ،ئەمۋاي(جۇڭگو)شىركىتىنى قۇردى ھەمدە شۇ يىلى 4-ئايدا تىجارەت باشلىدى.

1996-يىلى ئامىرىكا خۇسۇسىي كارخانىلارنىڭ22رەت تەرتىۋىدىن ئورۇن ئالدى.

1997-يىلى نۈچۈرلەت دۇنيادىكى ئەڭ چوڭ ۋىتامىن ۋە مىنرالنى مادىللار تولۇقلاش يىمەكلىكلىرىنىڭ داڭلىق ماركىسنىڭ بىرىگە ئايلاندى.

1998-يىلى 7-ئايدادۆلىتىمىزنىڭ تاشقى ئىقتىساد سودا مىنىستىرلىقى،دۆلەت سودا سانائەت مەمۇرىي باشقۇرۇش ئىدارىسى ،دۆلەت ئىچكى سودا ئىدارىسىنىڭ تەستىقلىشى بىلەن ،ئەمۋاي(جۇڭگو) دۇككان تەسىس  قىلدى ھەمدە مال ساتقۇچى ياللاپ سىتىش شەكلى مۇۋەپىقىيەتلىك ھالدا ئىگىلىك باشقۇرۇش شەكلىگە ئۆزگەردى.

2000-يىلى ،ئەمۋاي(AMWAY)،كىيۇستار(QUIXTAR) ۋە ئاككىس بوسنىس(ACCESS BUSINESS GROUP) لار بىرلەشتۈرۈلۈپ،ئالىكتور(ALTIKOR)شىركىتى قىلىپ قۇرۇپ چىقتى. ئالىكتور(ALTIKOR)شىركىتىنىڭ 13مىڭدىن ئارتۇق خىزمەتچىسى ۋە 4مىليوندىن ئارتۇق سىتىش خادىمىگە ئىگە بولدى.

2003-يىلى 8-ئايدا ئەمۋاي (جۇڭگو)نىڭ تىجارەت سوممىسى دۇنيا بويىچە بىرىنچى ئورۇنغا ئۆتتى.

2004-يىلى ئەمۋاي (جۇڭگو)نىڭ تىجارەت سوممىسى 17مىليارد يۇەنگە يىتىپ،دۆلەتكە تاپشۇرغان بىجى 370مىليون يۇەنگە يەتتى.

ھازىر ئەمۋاي ئۇنىڭ كەسپى بەش قىتئە 100نەچچە دۆلەت ۋە رايۇنغا تارقىلىپ،  AMWAY  توۋار ماركىسىدىكى 5چوڭ تۈر 450خىلدىن ئارتۇق مەسۇلاتى بولۇپ،بۇنىڭ ئىچىدە 380 خىل مەھسۇلات پاتىنىتقا ئىرىشكەن.يەنە 430خىل مەھسۇلاتقا پاتىنىت ئىلتىماسى قىلىۋاتىدۇ.

قېرىنداش يارى!  سەل سەۋىرچانلىق بىلەن سۆزلەڭ، بولامدۇ؟

بۇ مەزمۇن 8قەۋەتتىكى yariدە2011-07-03 04:17يوللىغان يازمىسىغا نەقىل :
[quote]بۇ مەزمۇن 9قەۋەتتىكى yariدە2011-07-03 04:35يوللىغان يازمىسىغا نەقىل :
.لېكىن قۇرئانى كېرىمدە مۇنداق سۈرىلەرمۇ  بار .

ئاللاھ تەئالا مۇنداق دەيدۇ:
«تىللىرىڭلار توقۇغان يالغانغا ئاساسلىنىپ: ‹ئۇ ھالال، بۇ ھارام› دېمەڭلار. بولمىسا ئاللاھقا بوھتان چاپلىغان بولىسىلەر. ئاللاھقا بوھتان چاپلىغانلار ھەرگىزمۇ مۇرادىغا يېتەلمەيدۇ»- نەھل سۈرىسى 116- ئايەت.

ئاللاھ تەئالا مۇنداق دەيدۇ:
«ئى مۇئمىنلار! ئۆزۈڭلار قىلمايدىغان ئىشنى نېمە ئۈچۈن سۆزلەيسىلەر؟ ئۆزۈڭلار قىلمايدىغان ئىشنى سۆزلىشىڭلار ئاللاھنىڭ قاتتىق غەزىپىنى قوزغايدۇ»- سەف، 61/2-3.

ئاللاھ تەئالا مۇنداق دەيدۇ:
«ئى مۇئمىنلار! ئۆزۈڭلارنى گۇناھتىن ساقلاڭلار، سىلەر قاچانىكى توغرا يولدا بولساڭلار، باشقىلارنىڭ ئاداشقىنى سىلەرگە زىيان يەتكۈزەلمەيدۇ. سىلەر ھەممىڭلار ئاللاھنىڭ دەرگاھىغا قايتىسىلەر، ئاللاھ قىلمىشىڭلارنى سىلەرگە ئېيتىپ بېرىدۇ» –مائىدە، 5/105.
....... 


«قۇرئانى كېرىم» دېگەن قانداق نەرسە؟  بۇ ئاتىشىڭىز خاتا.
 سۈرە دېگەن قانداق بولىدۇ؟  يۇقارقىنى سۈرە دەپ ئاتىشىڭىز خاتا، ئۇ دېگەن ئايەت.
شۇنىمۇ بىلمىگەن ئەھۋالىڭىزغا ئايەتنى مىسال كەلتۈرۈپسىز.ئاشۇ ئايەتلەر يۇقارقى پەتىۋاغا ماس كېلەرمۇ؟
كاللا دېسە پاقالچاق، دېمەڭ.  يۇقارقى پەتىۋا ئۇ كىشىنىڭ  تىلى بىلەن توقۇغان يالغان سۆزىمىكەن؟  ئېغزىڭىزنى يۇمۇڭ! 
 يۇقارقى پەتىۋانى تولۇق ئوقۇدىڭىزمۇ؟ نۇرغۇن ئايەت، ھەدىسلەر دەلىل كەلتۈرۈلگەن، دۇنياغا داڭلىق ئالىملرنىڭ بايانلىرى نەقىل كەلتۈرۈلگەن. سىز پەتىۋا بەرگۈچىگە بوھتان چاپلىدىڭىز!
نوچى بولسىڭىز يۇقارقى شىركەتنىڭ تىجارەت تۈزۈمىنىڭ ئىسلام تىجارەت تۈزۈمىگە ماس كېلىدىغانلىقىنى قۇرئان- ھەدىسلەر ئارقىلىق ئىسپاتلاڭ، ھەرگىزمۇ ئوقۇتقۇچىڭىزنىڭ سۆزلىرى، كىتابىڭىزنىڭ بايانلىرى بىلەن ئەمەس!
ئاۋۇ يىلنامىلەرنى خېلى ئوبدان ئۆگىنىپ چىققىنىڭىزنى بىلىمەن. مەھسۇلاتنىڭ ئالاھىدىلىكلىرىنى يادقا دەپ بېرەلەيدىغانلىقىڭىزنىمۇ بىلىمەن. پەيغەمبىرىمىزنىڭ تارىخىنى، ئىسلام يىلنامىسىنى شۇنچىلىك بىلەمسىز؟
ئەمدى بىر ئوبدانراق مائاش ئالىدىغان بولغاندا مائاشنى تاشلىۋېتىش ئېغىر كېلىۋاتىدۇغۇ، دەيمەن.
  قانچە ئۇزاق تۇرغانسىرى شۇنچە چۆكۈپ كېتىسىز، ئۇ چاغدا 100% ھاراملىقىنى بىلگەن تەقدىردىمۇ يەنىلا چېكىنىپ چىقىش ئېغىر كېلىدۇ.مۇھەممەدتۇرغۇن، پەرىدە مۇئەللىمىڭلار بۇ تىجارەتنىڭ ھاراملىقىنى 100% بىلسىمۇ، چېكىنىپ چىقمايدىغانلىقىغا ئىشىنىمەن. نىمىشقا دەمسىز؟ ئۇ كىشى ھاراملىقى 100% ئېنىق بولغان ھاراقنى ئىچىدۇ، ھاراملىقى 100% ئېنىق بولغان قىمارنى ئوينايدۇ. ئايالىمۇ ئەڭ كەم دېگەندە ھاراملىقى 100% ئېنىق بولغان ئەۋرىتىنى ياپماسلىق ئىشىنى قىلىدۇ. 

ئاللاھ مۇنداق دېگەن:« ئى مۆمىنلەر! ھاراق ئىچىش،قىمار ئويناش، بۇتلار (يەنى چوقۇنۇش ئۈچۈن تىكلەنگەن تاشلار)غا چوقۇنۇش، پال ئوقلىرى بىلەن پال سېلىش شەيتاننىڭ ئىشى، پاسكىنا قىلىقلاردۇر، بەختكە ئېرىشىشىڭلار ئۈچۈن شەيتاننىڭ ئىشىدىن يىراق بولۇڭلار. شەيتان ھاراق، قىمار ئارقىلىق ئاراڭلاردا دۈشمەنلىك، ئاداۋەت تۇغدۇرماقچى، سىلەرنى نامازدىن ۋە اﷲ نى ياد ئېتىشتىن توسماقچى بولىدۇ، سىلەر ئەمدى (ھاراقتىن،قىماردىن) يانمامسىلەر؟ اﷲ غا ۋە پەيغەمبەرگە ئىتائەت قىلىڭلار، (ئۇلارغا مۇخالىپەتچىلىك قىلىشتىن) ھەزەر قىلىڭلار، ئەگەر (ئىتائەت قىلشتىن) باش تارتساڭلار،
بىلىڭلاركى، رەسۇلىمىزنىڭ مەسئۇلىيىتى پەقەت ئوچۇق تەبلىغ قىلىشتۇر» (مائىدە سۈرىسى 90- 92- ئايەت)

ئۇلار  بۇ ئىشلارنى قىلسا ئۆزلىرىگە قىلچە پايدا ئەكەلمەيدىغانلىقىنى، بەلكى زىيان ئېلىپ كېلىدىغانلىقىنى بىلىپ تۇرۇپ قىلىدۇ.  شۇنداق ئادەملەرنىڭ ئايدا نەچچە تۈمەن پايدا ئەكىلىۋاتقان بىر سودىنى تاشلىيالىشى مۇمكىنمۇ؟  ئۇلار پەرزلىكى 100% ئېنىق بولغان نامازنى ئوقۇمايدۇ. يۇقارقىلارغا ئىسپاتىم بار.  ئەمدى سىزگە دەيدىغان گېپىم: ئۇلارغا ئۇنچىۋۇلا ئىشىنىپ كەتمەڭ! 
قۇرئان كەرىم سۈرە ھۇجۇراتنىڭ 6- ئايىتىدە : «ئى مۆمىنلەر! ئەگەر سىلەرگە بىر پاسىق ئادەم بىرەر خەۋەر ئېلىپ كەلسە، (ئىشنىڭ ھەقىقىتىنى) بىلمەستىن بىرەر قەۋمنى رەنجىتىپ قويۇپ، قىلمىشىڭلارغا پۇشايمان قىلىپ قالماسلىقىڭلار ئۈچۈن، (ئۇ خەۋەرنى) ئېنىقلاپ كۆرۈڭلار» دېيىلگەن. 
 بىز شۇ سەۋەبتىن شىركەت خوجايىنلىرىنىڭ، مۇھەممەد تۇرغۇننىڭ، باشقا ئەمۋايچىلارنىڭ، ئەمۋاي تونۇشتۇرۇلغان كىتابلارنىڭ، ئېلىپ كەلگەن خەۋەرلىرىنى ئېنىقلاپ كۆردۇق.
ئىسلامغا بويسۇنامسىز، بويسۇنمامسىز، ئۇ ئۆزىڭىزنىڭ ئىشى.  ھەممىمىز ئۆز ئىشىمىزنىڭ ئاقىۋىتىنى كۆرىمىز.  يەنە شۇنى ئۇنتۇماڭكى،  ئاللاھ ئاتا قىلغان رىزقنى كۆپەيتەلمەيسىز ياكى ئازلىتالمايسىز. سىز پەقەت ھەرىكەت قىلغۇچى، ئاللاھ ئاتا قىلغۇچى.  توغرىلىقىدا شەك بولمىغان ئىشنى قىلسىڭىزمۇ رىزقىڭىز ئازلاپ كەتمەيدۇ.سەگەكلىك بىلەن ئويلىنىپ بىر ئىش قىلىڭ.  شۇنى بىلىپ تۇرىۋاتىمەنكى، ھازىر كاللىڭىزدا نۇرغۇن چىگىش خىياللار پەيدا بولۇپ، تىجارەتكە بولغان قىزغىنلىقىڭىزغا سۇ سېپىلدى.  كەچتە دەرسكە قاتناشسىڭىز يەنە روھلىنىپ كېتىسىز. ئەمدى گېپىم تۈگىدى. ئەمدى نېمە قىلسىڭىز ئۆزىڭىزنىڭ ئىشى.

«اﷲ خالىغان ئادەمنى گۇمراھ قىلىدۇ، خالىغان ئادەمنى ھىدايەت قىلىدۇ (يەنى پەيغەمبەرلەرنىڭ ۋەزىپىسى پەقەت تەبلىغ قىلىش بولۇپ، ھىدايەت قىلىش ئىشى اﷲ نىڭ ئىلگىدىدۇر). اﷲ غالىبتۇر ھېكمەت بىلەن ئىش قىلغۇچىدۇر» سۈرە ئىبراھىم 4- ئايەت.

«سەن پەقەت (ئۇلارنى اﷲ نىڭ ئازابىدىن) ئاگاھلاندۇرغۇچىسىمەن، اﷲ ھەممە نەرسىنى تەسەررۇپ قىلغۇچىدۇر» سۈرە ھۇد 12- ئايەت

ئۇلار:« بىزنىڭ مەسئۇليىتىمىز پەقەت (روشەن دەلىللەر بىلەن) چۈشۈنۈشلۈك قىلىپ تەبلىغ قىلىشتۇر» دېدى. سۈرە ياسىن 17- ئايەت

«اﷲ غا يالغاننى چاپلىغان ياكى اﷲ نىڭ ئايەتلىرىنى ئىنكار قىلغان ئادەمنىمۇ زالىم ئادەم بارمۇ؟ گۇناھكارلار ھەرگىز بەختگە ئېرىشەلمەيدۇ» سۈرە يۇنۇس17- ئايەت.

مەن نەچچە كۇننىڭ ئالدىدا بىر نەچچە ئوغۇل - قىز بىلەن سەپەرداش بولۇپ قالدىم ، ئۇلارنىڭ ئىچىدىكى بىر قىز ماڭا ئۆزلىرىنىڭ ئۇرۇمچىگە بىر يىغىنغا ماڭغانلىقىنى <نيۇسكىن >(ئېنگىلىزچە يېزىلىشى ئېسىمدە قالماپتۇ )دەپ بىر خەلقىئارالىق بىر شىركەتنىڭ بارلىقىنى ئۇنىڭ مەھسۇلاتلىرنىڭ يەنى ئوزۇقلۇقلۇق ، ئاياللار گىرىم بويۇملىرىنىڭ خەلقىئارا بويىچە ئالدىنقى قاتاردا ئىكەنلىگىنى ، ئۆزلىرىنىڭ مۇشۇ شىركەتكە ئەزا بولۇپ ، مەھسۇلاتلىرىنى ئىستىمال قىلىۋاتقانلىقىنى ھەم سېتىۋاتقانلىقىنى دىدى . مەن ئىلگىرى ئەمۋاينىڭ مەھسۇلاتلىرىنى ئىشلەتكەن ھەم بۇ ھەقتە  ئازدۇر - كوپتۇر چۇشەنچىگە ئىگە بولغانلىقىم ئۇچۇن ئۇ قىزدىن سىزنىڭ بۇ شىركىتىڭىزنىڭ خاراكتىرى ئەمۋاينىڭكىگە ئوخشاشكەندە ، دىسەم ئۇ بىر مۇنچە چۇشەنچىلەر ھەم ئۇقۇتقۇچىلىرىنىڭ تەجىرىبە قىلىپ بەرگەنلىرىنى سۆزلەپ ئوخشاش ئەمەس دىدى ، ھەم بۇ <نيۇسكىن>نىڭ مەھسۇلاتلىرى ئىسلامغا ئۇيغۇن دەپ بىر مۇنچە دەلىل - ئىسپاتلار بىلەن مېنىمۇ قىزىقتۇردى ، مەن ئىلگىرى بۇ شىركەت ھەققىدە پەقەت ئاڭلاپ باقماپتىكەنمەن ، سەپەردىن قايتىپ كەلگەندىن كېيىن قوشنا ئىدارىگە يېڭىدىن كەلگەن بىر قىزغا بۇ ئىشلارنى سۆزلەپ بەرسەم ئۇ قىز <نيۇسكىن>نىڭ مەھسۇلاتلىرى ھەقىقەتەنمۇ ياخشى ، ئەمۋاينىڭكىدىندىنمۇ ئېشىپ چۇشىدۇ ، دەپ ئۆزىنىڭ تۇققانلىرىنىڭ ئىچىدە ئۇنىڭ ئوزۇقلۇقلىرىنى يەپ ساقايغان كىشىلەرنى سۆزلەپ بەردى .
