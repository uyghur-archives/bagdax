ئالما ئاچچىقسۈيىنى نۇرغۇنلىغان ناھىيەلەردىن تاپماق بىرقەدەر تەسكەن، لېكىن مەن توردىن ئىزدەپ ئالما ئاچچقسۈيىنى ئۆيدە ئچىتىش ئۇسۇلىنى تېپىپ ئۆيدە ئۆزەم ئېچتتىم، ئۇسۇلى ئىنتايىن ئاددىكەن بىراق شۇ ئېچىتىش ۋاقتى سەل ئۇزۇنكەن.
ئالما ئاچچىقسۈيى ئېچتىش ئۇسۇلىنى ئۆيدە ئېچتىش ئۇسۇلى:
يېڭى ئالما 500گىرام،500گىراملىق ئاق ئاچچىقسۇ بىر بوتۇلكا( يەنى گۈرۈچتە ئېچتىلغان ھېچقانداق ئارلاشما ماددا قۇشۇلمىغان ساپ ئاچچىقسۇنى تاللاڭ، رەڭگى سەل سارغۇچ كېلىدۇ)، ناۋات 100گىرام ئەتىراپىدا. بىر كلولۇق ئەينەك كونسرۋا قۇتىسى بىر دانە.
ئاۋۋال ئالمىنى تۇز سۈيىگە چىلاپ بىرنەچچە قېتىم پاكىز يۇيۋالمىز ئاندىن پاكز لاتا بىلەن سۈرتىمىز، ئالمىنى ياپلاق توغرايمز ئاندىن يۇقۇرى تېپمىراتۇرىدا دېزىنفىكىسيە قىلىنغان ئەينەك قۇتىغا بىر قۇر يېيتىپ ئۈستىگە ناۋاتنى سالمىز، ئاندىن ئالمىنى يەنە بىر قۇر يېيتىمىز ئۈستىگە ناۋاتنى سالمىز، مۇشۇ ئۇسۇل بۇيىچە ئەينەك قۇتا ئىچگە ئالما توشقاندىن كېيىن ئۈستىگە ئاق ئاچچىقسۇنى قۇيىمىز. (ئاچچىقسۇنى چۇقۇم ئالما پۈتۈنلەي چۆككىدەك يەنى قۇتىنىڭ ئاغىزدىن تېشىپ چىققۇچە قۇيمىز بولمىسا ئۈستى تەرىپىدىكى ئاچچىقسۇغا چۆكمەي قالغان ئالما سېسىپ پاختىلىشىپ قېلىشى مۇمكىن).
ئاندىن قۇتىنىڭ ئاغىزىغا بىر قەۋەت يالتىراق يېپىپ قۇتىنىڭ ئاغىزىنى مەھكەم ئېتىۋىتىمىز. ئانىدىن كېيىن قۇتىنى سالقىن كۈنگەي جايغا قۇيۇپ ئۈچ ئاي ئېچىتمىز. ئۈچ ئايدىن كېيىن قۇتىنىڭ ئاغىزىنى ئچىپ ئىچىدىكى ئالمىنى چۇكا بىلەن ئېلۋىتىپ، بىر پىيالنىڭ ئۈستىگە داكا قۇيۇپ ئالما ئاچچقىسۈينى ئۈستىگە قۇيۇپ سۈزۈلدۈرمىز. شۇنىڭ بىلەن تەبىئىي ۋە شەكسىز ئالما ئاچچقسۈيى تەييار بولىدۇ. ئۇرۇقلىماقچى بولغانلار ھەر كۈنى چۈشلۈك تاماق ۋە كەچلىك تاماقتىن كېيىن بىر قۇشۇق ئىچىپ بەرسىڭىز ئۈنۈمنى كۆرسىز، ( ئاچچىقسۇ ئېچىتىدىغان كونسىرۋا قۇتسىى ئوبدان قۇرۇتۇلغان بۇلىشى كېرەك بولىمىسا ئاچچىقسۇنىڭ سۈپىتى ئۆزگۈرۈپ كېتىدۇ.)
