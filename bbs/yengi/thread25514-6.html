گازنى ئائىلىلەرگە يەتكۈزۈش ھەرگىز ئۇنداق ئاسان ئىش ئەمەس ، قىرىنداشلار .
ئاۋۋال يىلىغا 1-2 مىليارد كۇپ گاز چىقىدىغان بىر گازلىق بولىشى ، ئاندىن شۇگازلىقىتىن چىققان گازنى تۇرۇبا لىنىيەسى ئارقىلىق ئادەم بار رايۇنلار ئاپىرىشى كىرەك ، ئۇنىڭ ئۈستىگە يەرنىڭ ئاستىدىن چىققان گازنى بىۋاستە ئائىلەرگە ئىشلەتكىلى بولمايدۇ ، ئۇنى بىر تەرەپ قىلىپ ، تەركىبىدىكى ھىدرو سولفىت دەيدىغان زەھەرلىك گازنىڭ مىقىدارىنى تۆۋەنلەتكەندىن كىيىن ، ئاندىن ئۇنىڭغا زەھەرسىز ئەمما پۇراقلىق بولغان بىر گازنى قاتىدۇ ، چۈنكى يەر ئاستدىن چىققان تەبىي گاز رەڭىسىز ، پۇراقسىز بولۇش بىلەن بىللە ئۆتكۈر زەھەرلىك بولۇپ ، ئائىلەردە گاز قىچىپ قالسا ، بۇنداق پۇراقسىز ، رەڭسىز گاز ئاسانلا بىزنى زەھەرلەپ قويشى مۇمكىن ، تۇرۇبا ئارقىلق گازنى شەھەرگە ئىلىپ كەلگەندىن كىيىن ، ئاندىن بىسىم تەڭشەش پونكىتى قۇرۇش لازىم ، بولمىسا تۆۋەن بىسىملىق گاز بىلەن تاماق ئەتكىلى بولمايدۇ . ئاندىن تەڭشەش پونكىتىدىن ھەرقايسى ئاھالىلەر رايونىغا گاز تۇرۇببىسى تارتىش كىرەك  ، ئەگەر بىنا ئۆي بولسا ئىش ئاسان ، بىر تۇرۇبىنى ياتقۇزسىلا ئىش ھەل بولىدۇ ، چىققىمۇ ئاز كىتىدۇ ، لىكىن يىزىلاردىكى ئۆيلەر بىر قەدەر تارقاق جايلاشقان بولغاچقا ، ھەر بىر ئۆيگە تۇرۇبا تارتىشقا توغرا كىلىدۇ ، چىقىم كۆپ كىتىدۇ ، گەپنىڭ راستىنى قىلغاندا ، پۈتۈنلەي زىيانغا ماڭىدۇ ، چۈنكى بىر ئائىلە يلىغا ناھايىتى 200-300 كويلۇق گاز ئىشلىدۇ ، مۇشۇ بىر ئائىلە ئۈچۈن تۇرۇبا تارتىشقىلا 50 مىڭ يۈەندىن ئارتۇق پۇل كىتىدۇ ، يەنە ئاسراش ، رىمۇنت خىراجىتى ، خادىملارنىڭ ئىش ھەققىنى تارقتىش قاتارلىقلار . مانا بۇ يىزىلارغا تەبىي گاز يەتكۈزۈشنىڭ تەس بولىشىدىكى ئاساسلىق سەۋەپ ، دۆلىتىمىزدە پۈتۈنلەي تەبىي گاز ئارقىلىق ئىسنىشنى ئىشقا ئاشۇرۇغان بىر شەھەر بار ، ئۇ شەھەرقايسى شەھەردۇ ؟ ئۇدەل خوتەن شەھىرى (خوتەن ۋىلايىتى ئەمەس جۇمۇڭلار). ئۇنىڭ ئۈستىگە بۇ تەبىي گاز ، نېفىت دىگەن شوۋا گەدەنلەر ساپلا ئادەم بارمايدىغان چۆل جەزىرىلەردىن چىقىدىكەن ، گاز قۇدۇقى قىزىشقىمۇ ناھايىتى كۆپ پۇل كىتىدىغان ئوخشايدۇ ، بۇرادەرلەردىن سوراپ باقسىام ، بىر قۇدۇق قىزىشقا 200 مىليۇندىن ئارتۇق پۇل كىتىدۇ ، 2 يىلدا ئاران قىزىپ بولىمىز دەيدۇ . كەمىنە پىقىر يۈسۈپقارىمۇ قەشقەر دىيارىدىن ، شۇڭا بۇ ئىشلارغا ئانچە مۇنچە كۆڭۈل بۆلۈپ قويقانلىق سەۋەپلىك ئىرىشكەن ئۇچۇرلارنى كۆپچىلىككە يەتكۈزۈپ قويدۇم ، خاتا يەرلىرى بولسا كەچۈرەرسىلەر ، يەنە بىر ئىش ، ما قەشقەردە بۇ يىل ياز ئىسسىق دەستىدىن تۇنۇر كاۋىپىغا ئايلىنىپ قالىدىغان چىغىمىزۋا ..........

قارىغاندا گىپىمنى خاتا چۈشۈنۈپ قاپسىزھە ، قېرىندىشىم ، مەن قانداقتۇر بىرەر گاز شىركىتىنىڭ خىزمەتچىسى ئەمەس ، يۈسۈپقارى دىگەن ئىسىمىمگە قاراپلا ئۇلار مىنى ھەرگىز ئالماس ھەقاچان . ئەمدى مەخسەتكە كەلسەك قېرىندىشىم ، جەنۇبىي شىنجاڭدا ئۈچ جايدىن چىقىدۇ ، بىرى قاغىلىق ناھىيىسىنىڭ ئۇششارباش يىزىسىدىكى كۆكيار گازلىقى ، يەنە بىر ئۇلۇغچاتتىكى ئاقموم گازلىقى ، يەنەبىرى خوتەن ۋىلايىتىدىكى خوتەن دەرياسى گازلىقى ، بۇ گازلىقلارنىڭ ھەممىسى مەركىزى شەھەردىن يىراق ، ئۇنىڭ ئۈسىتىگە شىنجاڭنىڭ زىمىنى كەڭ ، يىزىلار تارتاق جايلاشقان بولغاچقا ، يىزىلارغا گاز يەتكۈزۈش ئۈچۈن تۇرۇبا تارتىشقا كىتەرلىك مەبلەغنى چىقىرىشقا مەركەز كارخانىلىرى تازا ئۇناپ كەتمەيدۇ ، مەسلەن جوڭگو نېفىتى ، جوڭگو نېفىت خىمىيىسى قاتارلىقلار ، جەنۇبىي شىنجاڭنى گازلاشتۇرۇش دىگەن بىر قۇرۇلۇش باركەن ، گىزتتىن ئوقۇسام ئۈچ يىلدا 6 مىليارد يۈەندىن ئارتۇق مەبلەغ سىلىنىپتۇ ، ئاخىرىدا جەنۇبىي شىنجاڭدىكى ئىككى ۋىلايەت بىر ئوبلاستىنى 80 %  تۇرۇبا ئارقىلىق قالغان 20% تىنى پىرىسلانغان تەبىي گاز ماشىسنى ئارقىلق گازتۇشۇپ ئاپىرىپ ھەل قىلىمىز دەپتۇ . بۇنى خەلق گىزىتىنىڭ 2- بىتىدىن كۆردۈم ، شۇڭا چوقۇم ئەمەلىيلىشەر دەپ ئۇيلايمەن . بۇنىڭدا يەنە زور بىر قىسىم سىياسىي ئامىللار باردەك تۇرىدۇ . قالغىنىنى ھۆكىمەتكام بىلمىسە بىزدەك پۇقرالار بىلەيدىكەنمىز ، بىلىدىغىنىمنى دەپ باقتىم ، خاتا چۈشەنچە بىرىپ قويغانلىقىغا ئۆزرە ئىيتىمەن قىرىندىشىم .
