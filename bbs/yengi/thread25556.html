     كونىلار نامرات بولساڭ كۆچۈپ باق دەپتىكەن شۇنداق قىلىپ ئۈرۈمچىگە كەلگەن بۇ بەش يىل ئىچىدە قانچە قېتىم كۆچكىنىمنى بىلمەيمەن. لىكىن بەشىنجى يىلى كۆچۈش دىگەن نىمە بەكمۇ ئېنىق بىلىۋالدىم.مەن بۇيىل ئالى مەكتەپنىڭ تولۇق كۇرسنى پۈتتۈرۈپ يېڭى ئوقۇش مەۋسۇمىدىن باشلاپ ئاسپىرانتىلقتا ئوقۇماقچىتىم.بەلكىم تەلەيلىك بولسام كىرەك تۆتىنچى يىللىقتا مەلۇم بىر شىركەت بىلەن توختاملاشتىم ھەم بىر تەرەپتىن ئوقۇغاچ بىر تەرەپتىن ئىشلەيدىغانغا دىيىشتىم.ئىشلىرىم ھەر ھالدا ياخشى كىتىۋاتاتتى.بېشىم قېتىۋاتقان بىردىن بىر ئىش ئوقۇش باشلىغىچەتۇرىدىغان جاي تېپىش مەسلىسى ئىدى.
    ئوقۇش پۈتتۈرىدىغان يىلى بەكمۇ ئالدىراش بولۇپ كىتىدىغان گەپكەن.ھېلى ئۇ ئىش دىگەن ،ھېلى بۇ ئىش دىگەن.قېرىشقاندەك دەم ئېلىش كۈنىگە ئورۇنلاشتۇرۇپ تىزىملاتقان دەرسىمنىڭ ئىمتىھانىسى بىلەن مەكتەپتىكى ئىمتىھانىم بىر كۈنگە توغرا كېلىپ قالسا بولىدۇ.ئەتىگەندىن كەچكىچە ئىمتىھان بىرىپ ئۇھ دىسەم يىراق بىر تۇغقىنىمىز تۇغۇلىشىدن مېيىپ تۇغۇلۇپ قالغان قىزىنى ئوپراتسىيىگە ئېلىپ كەلسە بولىدۇ.ماڭا زادى ئارامچىلىق بولمىغۇدەك. بوپتۇلا بۇ بەش يىل جەرياندا مۇساپىرچىلىقنىڭ دەردىنى مەنمۇ ئاز تارتمىدىم.ھەممىسى يەنە ئۆتۈپ كىتىدىكەن.شۇنداق بىر تالاي ئىشلاردىن كىيىن بىر ئىشخانىدىكى خىزمەتدىشىم بىلەن بىللە بىر ئۆينى ئىجارىگە ئېلىپ بىللە تۇرىدىغان بولدۇق.ئۆينىڭ شارائىتىمۇ يامان ئەمەس ئىدى.ئەڭ مۇھىمى كۈندە بىللە ئىشقا بېرىپ بىللە ئۆيگە قايتىدىغانغا ،چۈشتەبىللە تاماق يەيدىغانغا ھەمراھىم بار بولغانىدى.
    مەن ئۆيدىنمۇ بەكرەك بۇنىڭغا خوش ئىدىم.ئىشتىن چۈشۈپ ئۆيگە قايتىپ بىللە تاماق يەيتۇق.يولدا قايتقۇچە خىزمەتدىشىم بىلەن چېقىشىپ كۈلۈشۈپ ئۆيگە قانداق قايتىپ كەپ قالغىنىمىزنى بىلمەيلا قالاتتۇق.بىركۈنى ئىشچانلىقىم تۇتۇپ شىرىن خىياللار ئىلكىدە ئاشخانا ئۆينى تازلاپ ھەتتا ئەخلەتچىلىكىنىڭ ئىچى-سىرتىنىمۇ سۈرتۈپ چىققان ئىدىم .لىكىن بۇ كۈنلەر ئۇزاققا بارمىدى.ئىشتىن چۈشۈپ بىر يىشىك سۈت ئالدىم ھەم ئەتە ئىشقا مېڭىشتىن بۇرۇن ئەتكەن چاي قىلىشنى ئويلىشىپ ئۆيگە قايتتىم .بۇ شىرىن خىياللرىمىڭ بۇنچە تىز  يوققا چىقىشىنى ئويلىمىغان ئىكەنمەن.ئەتكەن چايمۇيوق،تازلاپ سۈرتۈپ پارقىرىتىۋەتكەن ئاشخانىدا سەي قىڭرىقىمنى تاقىلدىتىپ بىرەقېتىم سەي توغراپ باققان بولسامچۇ كاشكى. ئۆينى بىز بىلەن بىللە ئىجارىگە ئالغان قىز ماڭا  ئۆزىنىڭ ئاكىسىنىڭ كىلىپ بىللەتۇرىدىغانلىقىنى ھەم ماڭا باشقا جايدىن ئۆي تېپىپ ئىككى كۈن ئىچىدە كۆچۈشنى ئېيىتتى.بۇ خۇددى ماڭا چۈشۇمدىكى چاقچاقتەك تۇيۇلدى.چۈنكى مەن بۇ ئۆيگە كۆچۈپ كىرگىلى تېخى تولۇق بىر ھەپتىمۇ بولمىغان ،تاكى  قولۇمدىكى ئۈچ ئايلىق ئىجارە ھەققى بىلەن زاكالەت پۇلى يەرگە چېچىلىپ ئاياغلىرىم ئاستىدا دەسسەلگەندىلا ئاندىن بۇنىڭ چۈش ئەمەس راسلىقىنى بىلدىم.        كۆزلىرىمدىن ياشلار تاراملاپ چۈشمەكتە،نىمانداق تاۋى نازۇك ئۇششۇق بوپ كەتتىم ؟نىمىگە يىغلىدىم نەدىن كەلگەن ياش...ھېچنىمىنى بىلمەيتىم.      
     كاللامداپەقەتلا بىرلا سۇئال نىمىشقا؟ئەمدى نەدىن ئۆي ئىزدەمەن؟ئەگەر تۇرىدىغان ئۆي تاپالمىسامچۇ؟بۇ كۆز ياشلار توختايدىغاندەك ئەمەس ئىدى.توۋا ئادەم يىغلىساكۆز ھەم بۇرنىمۇ تەڭ يىغلايدىكەن.كۆزدىن ئېقىۋاتقاندىكىن بولمىدىمۇ،كۆڭلۇم شۇنداق يېرىم،كىمگە نىمە دەپ ئىچىمنى بوشاتسام بولا.قېرىشقاندەك مۇشۇ كۈنى مىنىڭ يېقىن دوسلىرىمنىڭ ھەممىسىنىڭ ئىشى با.بىرى ئاغرىق،بىرسى بېيجىڭغان كەتكەن،يەنە بىرسى ئۆيگە كەتكەن.ئۇلاردىنمۇ بەكرەك يېقىنلىرىم يوق ئىدى.ئۆيگە تىلفۇن قىلشقا تېخىمۇبولمايتى.كىم مىنى ئاتا-ئاناڭنىڭ پۇلىنى ئىككى ئىنىڭ ئۈچۈن تىجە، سىنىڭمۇ بەش ئەزايىڭ بىجىرىم بولغاندىكىن ئۆزەڭ تاپ دەپ ئۈگەتكەن ؟بىر چاغدا قارىسام ياش ئېيىتقان قەغەز قىپقىزىل قان  تۇراتتى،ئۆزۈممۇ قورقۇپ كەتتىم،بۇرنۇمدىن قان ئېقىۋاتاتتى.شۇنداق تەسلىكتە بۇرنۇمدىكى قاننى ئېيتىۋاتىمەن.يەنە بىر تەرەپتىن مەن بىلىدىغان تونۇيدىغان تونۇشلىرىمنىڭ  ھەممىسىدىن ئۆي سوراۋاتىمەن.سائەتنىڭ نەچچەبولغىنى كۆزۈمگە كۆرۈنمەيتى.ناماز ۋاقتى بولاي دەپ قاپتۇ،ھەم مەن يىغلاپ ئۇخلىمىغان بىلەن باشقىلار ئۇخلايدۇ دە.يىغىدىن كۆزلىرىم قاپاقلىرىم ئىششىپ كەتكەنىدى.
     نامازنى ئوقۇپ بولۇپ پۇتۇمنىڭ ئۇچىدا دەسسەپ سومكام ۋە كومپىيۇتىرىمنى كۆتۈرگىنىمچەئىشقا ماڭدىم.ياخشى يېرى بۈگۈن مەن ئىشخانا نۆۋەتچىسى .ئىشخانىنىڭ ئاچقۇچىمۇ بار.شۇنداق قىلىپ ھەممىدىن بۇرۇن كىلىپ ئىشخانا تازلىقىنى قىلدىم.لىكىن يەنىلائىشقا چۈشۈشتىن بىر سائەتتەك بۇرۇن ئىشىم تۈگىدى.ئىش ئۈستىلىمگە بېشىمنى قويغىنىمچە مۈگدەپ قاپتىمەن.كۈپكۈندۈزدە چۈش كۆرۈۋاتاتتىم.مەن نەرسە كىرەكلىرىمنى كۆتۈرۈپ ئۆي ئىزدەۋاتاتتىم، لىكىن خۇددى ھەممە مىنى مەسخىرە قىلىپ كۈلىۋاتقاندەك.ھېي،سەن چوڭ ئالدىراش خېنىم ئەمەسمىدىڭ،تېخى كۈچلۈك ئايال.سىنىڭ كۈچلۈكلىكىڭدىن ئۆرگىلەي.مۇشۇھالىڭغىما؟!ھا ھاھاھاھا.بىرى چۈشۈمدە مىنى مۇشۇنداق مەسخىرى قىلىپ كۈلىۋاتاتتى.چىكەمدىن ئاققان سوغۇق تەر،چۆچۈپ ئويغىنىپ ئۆزۈمنىڭ ئىشخانىدا ئىكەنلىكىمنى بىلدىم.توۋاكۈپكۈندۈزدە ....مىنى دائىم ساخو دەپ قويىدىغان ئاكاش كىلىپ گىزىت كۆرۈپ ئولتۇراتتى. ھوي ساخو .بېشىمنى كۆتۈرۈپ ئىتتىك كۆزەينىكىمنى تاقاپ ئۈلگۈردۈم(ئىششىپ كەتكەن قاپاقلىرىمنى كۆرۈپ باشقىچە ئويلاپ قالمىسۇن دىدىم).كومپىيۇتىرنى ئاچتىم ۋەئۆي ئىزدەشكە باشلىدىم.بىر نەچچە ئۆينىڭ ئورنى ۋە شارائىتىنى مۇۋاپىق كۆرسەم پۇلى قىممەت،پۇلى مۇۋاپىقنىڭ ئورنى چەت ياكى بەكمۇ يىراق.شۇنداق قىلىپ چۈشتىن كىيىن بولغاندا شىركەتتىن رۇخسەت سوراپ ئۆي كۆرگىلى چىقتىم.پۈتۈن چۈشتىن كىيىن يەتتەسەككىز يەردىن ئۆي كۆرۈپتىمەن،لىكىن مۇۋاپىقراقى يوق.بېشىمنى ساڭگىلاتقىنىمچە يەنەئاشۇ{ئۆيۈم}گە ماڭدىم.تۇيۇقسىز ئالدىنقى يىلى بىللە تۇرغان دوستۇمنى ئويلاپ قالدىم.بولمىسانەچچە كۈن شۇنىڭ بىلەن بىللە تۇرۇپ تۇرايمۇ يا.تىلفۇن ئۇلاندى بەكمۇ خوش بوپكەتتىم .شۇنىڭ بىلەن كەينىمگە ياندىم دە ئۇنىڭ قېشىغا باردىم.ئۇنىڭ ئۆيىدە ئۇيالغۇز ئىدى.كۆرشۈپلا يەنە ئازراق يىغلاپ سالدىم.تاغدىن باغدىن دەپ يۈرۈپ ئۇدوستۇمنىڭ قېشىدا قونۇپ قالدىم.ھەم ئۇنىڭ يېنىدا بىر نەچچە كۈن تۇرىدىغان يا بىللەتۇرىدىغانغا ئىجارە ۋە باشقا چىقىملارنى بىللە تۆلەيدىغانغا دىيىشىپ ئاچقۇچنى ئېلىپ ئىشقا ماڭدىم.
      بۇ دوستۇم بىلەن ئىزدەشسەك ھالىمىز يوق ،ئالدىراش بوپ كەتسەك نەلەردە يۈرگىنىمىزنىمۇ بىلمەي قالاتتۇق.لىكىن ياردەم كىرەك بولغاندا تېپىشالايتۇق.شۇنداق قىلىپ  مەن ئۆي كۆچىدىغان بولدۇم.مىنىڭ كىيىم-كىچەكلىرىمدىنمۇبەكرەك بىر نەچچە يىشىك كىتاپلىرىم بار ئىدى.ئاشۇ كىتاپلار سەل ئېغىر ئىدى.ئوقۇغۇچىنىڭ يەنە نىمىسى بولسۇن .ھەي بىچارە ئىنىم يەنە جاپا تارتىدىغان بولدى دە.بىرىنجى قېتىم كۆچكەندىمۇ ئىنىم كۆچۈشۈپ بەرگەن .ئالتىنچى قەۋەتتىن يەنە تۆتىنچى قەۋەتكە.ئەمدى تۆتىنچى قەۋەتتىن كۆچىمەن ،ھەر ھالدا يېڭى ماكانىم بىرىنجى قەۋەتتە ئىدى.يۈك-تاقلارنى توشۇيدىغانغا شىركەتنىڭ توشۇش بۆلمىدىكىلەر ياردەم قىلدى .مەن نەرسە كىرەكلىرمنى يۆتكەپ چىقىپ كەتكۈچە بېشىمدا ساقچى بولدى.نىمە بولسا بولسۇن،ئەمدى بۇ ئۆيگە ئاياغ باسماسلىققا قەسەم قىلىپ چىقىپ كەتتىم.شۇنداق قىلىپ كوچۈۋالدىم .ھەي ،بولسا مۇشۇئۆيدە تۇرۇپ قالسام،ئوقۇش باشلاشقىمۇ ئۇزۇن قالمىدى.كىيىنكى قېتىم مەكتەپكە كۆچۈپ كەتسەم يەنە كۆچمەن بولۇپمۇ يۈرمەسمەن.شۇنداق قىلىپ نۇرغۇن جاپالاردا  ئىككىنچى قېتىم كۆچتۈم.ئۆي كۆچۈپ نەرسە يۆتكەپ ھارغان بولساممۇ لىكىن كۆڭلۈم توق بولغاچقا ھارغىنچىلىقمۇ بىلىنمىدى بولغاي، ئۆينى رەتلەپ تازلاپ شۇنداق ئادەمنىڭ مەسلىكى كەلگۈدەك قىلۋەتتىم.ئۇھ دەپ تۇرسام مەن  بىلەن بىللە تۇرىدىغان قىز كەلدى.بىز چىرايلىق كۆرۈشتۇق،مەن قويغان نەرسىلەرنىڭ ئورنىنى ئۇنىڭغان دەپ بەردىم.
     لىكىن ئويلىماپتىكەنمەن،بۇ ئۆي تۇنجى كۈنىلا ماڭا خاتىرجەم ئۇخلاشقىمۇ نىسىپ قىلمىدى.مىنىڭ بۇ يېڭى ھەمراھىمنىڭ يېزىنىسى ھەم ئاچىسى ئۈرۈمچىگە كىسەل كۆرسەتكىنى كىلىدىكەن.ئۇنىڭ دىيىشىچە ئۇ بۇئۆينى ئۆيىدىكىلەر كەلسە شۇلارنى كەڭ- تاشا تۇرامدىكى دەپ ئاغانمىش.توۋا بۇ ئىجارەئۆي تۇرسا،ئۇنىڭ ئۈستىگە ئۇ پۈتۈن بىر يۈرۈش ئۆينى ئىجارىگە ئالغان بولسىكىن.ئىچىمدەتوۋا بۇ ئۆيگە ئايغىم يارىشىپ قالسا بولاتتى دەپ ئويلايتىم.ۋە تىلفۇنغا قاراپ ئۇيان بۇيان ئاختۇرۇپ،يا بولمىسا بىرە مىھمان كۈتۈشكە ئورۇنلاشتۇرامدۇق ،ھېرىپ كەلگەندە كەڭ-تاشا ئۇخلاپ ئارام ئالسادىگىنىمنى بىلىمەن.....گىپىمنىڭ ئاخىرى تېخى تۈگمەستىنلا گەپ قىلغانغا تويدۇم.ماڭا قالغىنى ئىشىكنى جالاقشىتىپ تۇمشۇقىنى ئۇچلاش بولدى. شۇنداق ئاچچىغىم كەلگەن بولسىمۇ،بىرىنجى كۈن تۇرسا ، ئەتە ياخشى بوپقالا.شۇنداق قىلىپ ئىشكنى تاراقشىتىپ ۋاڭ چۇڭ بىلەن بىزنىڭ
