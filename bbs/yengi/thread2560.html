             ئەسسالامۇ ئەلەيكۇم باغداش مۇنبىرىدىكى بارلىق قېرىنداشلار!مەن بۇ مۇنبەرگە ئەزا بولغىلى تېخى نەچچە كۈنلا بولدى.مانا بۈگۈن مۇنبەردىكى قېرىنداشلار بىلەن مۇشۇ تۇنجى تېمام ئارقىلىق ئۆزىمىزنىڭ كىملىكى،ئىچكىردە ئوقۇۋاتقان ئوقۇغۇچىلار يولۇقىۋاتقان ۋە سەل قاراۋاتقان مەسىلىلەر،باشقىلارنىڭ ئىچكىردىكى قېرىنداشلارغا بولغان قارىشى ۋە شۇنداقلا بىزنىڭ يوقىتىپ قويىۋاتقان ۋە ئېرىشىۋاتقان نەرسىلىرىمىز توغرىسىدا پىكىر ئالماشتۇرۇشنى مەقسەت قىلىپ مۇنۇ تېمىنى يېزىپ ئولتۇرۇپتىمەن.قېرىنداشلارنىڭ ئاكتىپ ئاۋاز قوشۇپ بىرلىكتە پىكىر قاتناشتۇرىشىنى ئۈمىد قىلىمەن.

        بىز كىندىك قېنىمىز تۆكۈلگەن ئاشۇ مىللى ۋە دىنىي پۇراق چىقىپ تۇرغان ئانا ماكانىمىزدىن،مېھرىبان ئاتا-ئانىمىزدىن،جان-جىگەر قېرىنداشلاردىن،كىچىكىمىزدىن بىللە ئويناپ چوڭ بولغان ئاغىنىلەردىن،ئاشۇ ھەممە نەرسە تونۇش بولغان گۆھەر زېمىندىن ئايرىلىپ،مانا مۇشۇنداق دۆلىتىمىزنىڭ ھەر قايسى جايلىرىدىكى ئۆرۈپ-ئادىتى،دىنىي ئېتىقادى ۋە تۇرمۇش ئۇسۇلى،ھەتتا ياشاش نىشانى ئوخشاش بولمىغان كىشىلەر توپى ئارىسىغا ئىلىم ئېلىشتەك بۈيۈك ئارزۇ-ئارمانلارنى قەلبىمىزگە پۈكۈپ كەلدۇق ۋە شۇ نىشان ئۈچۈن كۈرەش قىلماقتىمىز.بۇ جەرياندا تۈرلۈك ئىشلارغا يولۇقتۇق ۋە يولۇقىۋاتىمىز.

           1.تۈگىمەس سورۇن.
      بەلكىم بۇ ئىش ھەممە ئالىي مەكتەپلەردە بولۇشى مۇمكىن.ئالىي مەكتەپكە قەدەم باسقان دەسلەپكى كۈنلەردە ئاتالمىش يىڭى ئوقۇغۇچىلارنى كۈتىۋېلىش نامى بىلەن ئۆتكۈزۈلگەن سورۇندىن باشلىنىپ،تاكى مەكتەپ پۈتتۈرىدىغان ئوقۇغۇچىلارنى ئۇزاتقان ئولتۇرۇشقىچە نۇرغۇن سورۇن تۈزۈلۈپ بولىدۇ.يىڭى ئوقۇغۇچىلارنى كۈتىۋېلىش،روزا ھېيتلىق ئولتۇرۇش،قۇربان ھيتلىق ئولتۇرۇش،يىڭى يىللىق پائالىيەت،خانىم-قىزلار بايرىمى مۇناسىۋىتى بىلەن قىزلارغا پائالىيەت قىلىپ بېرىش،نورۇز،يەنە ھەرقايسى يىللىقلارنىڭ ئىچىدە ئۆتكۈزۈلىدىغان چوڭ كىچىك سورۇنلار،يەنە ھەر قايسى يۇرتلاردىن كەلگەنلەر ئۆتكۈزىدىغان يۇرتداشلار يىغىلىشى،مەكتەپ پۈتتۈرگەنلەرنى ئۇزىتىش پائالىيىتى.......
        مانا مۇشۇنداق سورۇنلار ئۈزۈلمەي ئۆتكۈزۈلۈپ تۇرىدۇ.ئۇنىڭ ئۈستىگە ھەممە سورۇندا دېگۈدەك ھاراق بار.ئەسلى بىز مۇسۇلمان ئەۋلادى ئىدۇق،بۇ دۇنياغا مۇسۇلمان بولۇپ يارىتىلغان ئىدۇق.ھاراق قۇرئان كەرىمدە ئېنىق،كەسكىن چەكلەنگەن ھارام نەرسە ئىدى،بىراق پەرۋا قىلمىدۇق.ھاراق ئىچىش بۇتقا چوقىنىشتىن يامان ئىدى.بۇنى بىلمىدۇق.ئەنە شۇنداق سورۇنلاردا قىزلار بىلەن بىللە ئولتۇرۇپ ئويناپ كەتتۇق.سورۇننىڭ مەقسىدى قىزلار ۋە ھاراق بولۇپ قالدى.ئاغزىدىن مىللەت چۈشمەيدىغان يىگىتلىرىمىز ئەنە قىزلارنى قولتۇقلاپ ئۇسۇل-تانسا ئوينىماقتا.ۋىجدان-غورۇردىن گەپ ئاچىدىغان يىگىتلىرىمىز،شەرمى-ھايانى تەكىتلەيدىغان قىزلىرىمىز ئۆزلىرىنى مۇشۇ سورۇنلاردا نامايەن قىلماقتا.مانا مىللەتنىڭ كەلگۈسى ئۈمىدى بولغان ئىلىم ئىگىلىرى!قىزلار بۇزۇلدى،قىزلار ئۇنداق قىزلار مۇنداق دەپ يۈرىمىز،قىزلار شۇ سىز بىلەن بىزنىڭ قولىمىزدا بۇزىلىدۇ،ئەنە شۇنداق سورۇنلاردا بۇزىلىدۇ.قىزلارنى قوغداش ئەسلى بىزدەك ئەركەكلەرنىڭ ئىشى ئىدى.بىراق ئۆزىمىز قوغدىماق تۈگۈل بۇزىۋاتىمىز.يۈرىكىڭىزنى تۇتۇپ ئويلاپ بېقىڭ،سىزنىڭ شۇنداق سورۇندىن كەلمەيدىغان قىز بىلەن توي قىلغۇڭىز كېلەمدۇ؟بىر ئۆمۈر بىللە  ياشاشنى خالامسىز؟ھېچكىم مۇنداق بولۇشنى خالىمايدۇ،بىراق ئۆزىنىڭ نەپسى-خاھىشى ئۈچۈن بۇ تۇيۇق يولدىن يانالمايدۇ.ئېسىڭىزدە بولسۇنكى،مىللەتنىڭ ئانىلىرى تۈزەلمەي تۇرۇپ مىللەت تۈزەلمەيدۇ،مىللەتنىڭ ئانىلىرى پەقەت ئىمانى مۇستەھكەم،غورۇرى كۈچلۈك،شىجائىتى بار ئەركەكلەرنىڭ قولىدىلا ئاندىن تۈزىلىدۇ.ئۇنداق قۇرۇق سورۇندىن ئۆزىمىزنى تارتايلى،قانداقتۇر ناخشا-ئۇسۇل ئويناشنى مىللەتنى سۆيۈشنىڭ بەلگىسى قىلىۋالغان نادان قېرىنداشلار ئاۋۋال ئۆزىڭىزنى سۆيۈشنى بىلىڭ،شۇندىلا باشقىدىن ئېغىز ئاچقىلى بولىدۇ.
          2.مىللىي كىملىك.
       ئىچكىردە ئوقۇۋاتقان ھەممە قېرىنداشلارغا ئايان.ئىچكىردىكى نۇرغۇن خەنزۇلار ئۇيغۇرنى چۈشەنمەيدۇ،پەقەت تېلىۋوزوردىن ئانچە-مۇنچە ئاڭلىۋالغاندىن باشقا ئۇيغۇر توغرىسىدا بىلىدىغىنى پەقەت ئوغرى ،كاۋاپ ۋە ناخشا.چۈنكى شىنجاڭ ناخشا-ئۇسۇل ماكانى دەپ پۈتۈن مەملىكەتكە داڭلىق،ھېلىمۇ مۇشۇ نام بىلەن تا ھازىرغىچە غەپلەت ئۇيقۇسىدا ئۇخلاۋاتقان قېرىنداشلار ساماندەك.يا بۇ ناخشا ئۇسۇل بىزنى بىر يەرگە ئېلىپ كەتمىدى.شۇ پېتى تۇرىمىز يەنە.كاۋاپقا كەلسەك نۇرغۇن شەھەرلەردە كاۋاپچى قېرىنداشلار بار.مەيلى ئىختىيارى بولسۇن ياكى مەجبۇرىي بولسۇن ئوغرىلىق قىلىۋاتقانلار ھەم بار.شۇڭا بۇ ئۈچ نەرسىنى خەنزۇلار بەك ئېنىق بىلىدۇ.چۈنكى ئۇلارنىڭ كۆرگىنى مۇشۇ.باشقىسىنى بىلمەيدۇ.ئۇيغۇرنى كۆرسە كۆزىگە ئوغرى كۆرىنىدۇ.ياتاقمۇ بەرمەيدۇ ھازىر ئالدىراپ.سەۋەب زادى نېمىدە؟ زادى نېمىدە؟ئۇلاردىمۇ ياكى بىزدىمۇ ؟سەۋەب يەنىلا بىزدە،ھېچ بولمىسا كۆپ قىسمى بىزدە.چۈنكى ئۇلارنىڭ كۆرگىنى ئوغرى ياكى كاۋاپچى،شۇنداق بولغاندىن ئۇلار ئۇيغۇرنى كۆرسە ئەلۋەتتە شۇنداق قارايدۇ.چۈنكى تۇنجى تەسىر بەك مۇھىم.ئۇلار دەسلەپ شۇنداق قېرىنداشلارنى كۆرگەچكە،بىزدىن قورقۇپ كەتكەن.كۆزىگە ئوغرى كۆرىنىمىز.ئەگەر ئاشۇ قېرىنداشلارمۇ ھەقىقىي بىر مۇسۇلماندەك يۈرگەن بولسا،ھەقىقىي بىر مۇسۇلماننىڭ ئوبرازىنى تىكلىگەن بولسا،بەلكىم ئۇلارمۇ بىزگە ئۇنداق قارىماس بولار ئىدى.(ئەلۋەتتە بۇ ئىشلارنىڭ يەنە بىرقىسىم ئارقا كۆرىنىشى بار).دېمەك بىز بۇ يەردە ئۆزىمىزگىلا ۋەكىللىك قىلمايمىز،بەلكى مىللەتكە ۋەكىللىك قىلىپ قالىمىز.چۈنكى ئۇلار بىز ئارقىلىق مىللەتكە باھا بېرىدۇ.بىزنىڭ ھەركىتىمىز ئارقىلىق پۈتۈن ئۇيغۇرنىڭ ساپاسىغا باھا بېرىدۇ.چۈنكى ئۇلار بىزنىڭ ئۇيغۇرنىڭ كەلگۈسى ئۈمىدى ۋە ئىستىقبالى دەپ بىلىدۇ.شۇڭا ھەر بىر ھەرىكىتىمىز ئۆزىمىزگىلا ۋەكىللىك قىلمايدۇ.

       3.ئېتىقاد.
      ئىچكىردىكى نۇرغۇن ئالىي مەكتەپلەردە چەكلىمە يوق دىيەرلىك.بىراق نۇرغۇن قېرىندالار چەكلىمىنى ئۆزى قويىۋېلىپ،نامازدىن باش تارتىدۇ،روزىنى ئادا قىلمايدۇ.ھالبۇكى ناماز ۋە روزا ئىسلامدىكى بەش ئاساس ئىچىدە بىز ئوقۇغۇچى بارلىلار قىلالايدىغان ئىبادەت ئىدى.ئۇنىمۇ قىلماي تاشلاپ قويغانلار ساماندەك.يەنە قايسى يۈزىمىز بىلەن ئۆزىمىزنى مۇسۇلمان دەيمىز؟ئەجىبا مۇسۇلماننىڭ قانداق بولىدىغانلىقىنى بىلمەمدۇق؟چوشقا گۆشى يە دېسە ئاچچىقتىن ئۆلۈپ بېرىسىز،بىراق ھاراقنى گۈپۈلدەپ ئىچسىڭىز ھۇزۇرلىنىپ كېتىسىز،نىمە ئۈچۈن؟ياتاقتىكى خەنزۇلار مەندىن كۆپ قېتىم سورىغان،نۇرغۇن قېرىنداشلار ھەم مۇنداق ئەھۋالغا يولۇققان بولۇشى مۇمكىن.
-سىلەر دىنغا ئىشىنەمسىلەر؟
-ھەئە،بىز مۇسۇلمان،ئىسلام دىنىغا ئېتىقاد قىلىمىز.
-ئالىي مەكتەپ ئوقۇغۇچىلىرى دىنغا ئىشەنسە بولمايدىغۇ؟
-دۆلىتىمىز پۇقرالىرىنىڭ دىنىي ئەركىنلىك ھوقۇقى بارلىقىنى ئۆگەنمىگەنما.
-ئۇنداقتا مۇسۇلمانلار چوقۇم ناماز ئوقۇشى،روزا تۇتىشى كېرەكمۇ؟
-شۇنداق.
-ھاراق ئىچسەڭلار،تاماكا چەكسەڭلار،چوشقا گۆشى يىسەڭلار بولمايدۇ،شۇنداقمۇ؟
-شۇنداق،بۇلارنى ساڭا بىر ئىككى ئىغىز گەپ بىلەن چۈشەندۈرۈپ بولغىلى بولمايدۇ.
-ئۇنداقتا سېنىڭ باشقا يۇرتداشلىرىڭ ناماز ئوقۇمايدىغۇ؟يەنە ھاراق ئىچىپ،تاماكا چېكىپ يۈرىدىغانلىرىمۇ كۆپ،ئۇلار مۇسۇلمان ئەمەسما؟
-بىلمەيمەن.بەلكىم ئۇلارغا خوشياقمىسا كېرەك.ھەممە ئادەم ئوخشاش بولمايدۇ-دە.
         نۇرغۇن قېرىنداشلار دەسلەپ ياتىقىدىكى خەنزۇلارغا ئانداق قىلما،مانداق قىلما دەيدۇ،ئۇزۇن ئۆتمەي ئۆزى مەس قايتىپ كېلىدۇ.بۇنىڭغا نىمە دەيمىز.ئۆزىمىزنىڭ دىنىنى ئۆزىمىز مۇشۇنداق ۋەيران قىلىۋاتساق،دەپسەندە قىلىۋاتساق يەنە قايسى يۈزىمىز بىلەن باشقىلاردىن ئاغرىنىمىز؟ھېلىمۇ ئاخبارات ۋاستىلىرنىڭ تەتۈر تەشۋىقاتى نەتىجىسىدە مۇسۇلمان دىسە خەقنىڭ كاللىسىغا تېرورچى كېلىدىغان بولدى،بىزنىڭ ئارىمىزدىمۇ بار مۇنداق ئەخمەقلەر تېخى.شۇڭا بىزنىڭ نۆۋەتتكى ۋەزىپىمىز قانداقتۇر ئۇلارنى بوزەك قىلىش،ئۆزىمىزنى كۆرسىتىپ قويۇش ئەمەس،ئەمەلىي ھەركىتىمىز بىلەن ھەقىقىي بىر مۇسۇلمان ئوبرازىنى تىكلەش.ئۇلارغا ھەقىقىي ئىسلامنىڭ قانداق بولىدىغانلىقىنى ئۆزىمىزنىڭ ھەرىكىتى بىلەن كۆرسىتىپ قويۇش.بىزنىڭ ئەڭ ياخشى ئۈلگىمىز بولغان مۇھەممەد سەللەللاھۇ ئەلەيسى ۋەسسەللامنىڭ دەۋەت ئۇسۇلىغا نەزەر سالساق قانداق قىلىشنى بىلىۋېلىش تەس ئەمەس.

4.تاماق مەسىلىسى.
5.دىيارىمىزدىكى كىشىلەرنىڭ بىزگە بولغان قارىشى
 
4-،5- تارماقلىرىنى قېرىنداشلار تولۇقلاپ قويسا بولىدۇ.قالغان گەپنى ئىنكاسلاردا ئالدىرىماي دىيىشىلارمىز ئەمىسە.ئۇزۇن بوپتىكەن مۇنداق كۆپ خەت باسمىغىلى.ئەمىسە قېنى ئويلىغانلىرىڭلارنى دەپ بېقىڭلار قېرىنداشلار~~~~~

يوللىغان ۋاقتى 2011-7-24 23:55:55 |ھەممە قەۋەتنى كۆرۈش
بۇ بىر  سىر،بەلكىم قىيەمەتتە يىشلىپ قالار،ئەمما شۇنى ئىنىق بىلشىمىز كىرەككى دىندا چىىڭ تۇرۇش شۇ دىنغا ئىتقاد قىلغۇچىنىڭ دىنغا بولغان سادىقلىقىنى بىلدۈردۇ.دىنغا بولغان ئىتقادى كۈچەيگەنسىرى دىنسىزلاردىن پەرقى كۈپيۈپ ھەممە ئىشلارغا دىنى ئىتقاد بىلەن قارايدىغان بۇلۇدۇ،قۇرئاندىكى قانۇن تۈزۈملەرنىڭ ۋە ئىزدىمائىيەتتىكى قانۇن تۈزۈم بىلەن بولغان تۈپ پەرقى قۇرئان ئىلاھى تۈزۈمدۇر،ئىزدىمائى تۈزۈم بولسا داۋاملىق يىڭىلنىپ  ،ئۆزگۈرۈپ ،گاھى قارشى ئىلىنسا قاھى تۈزتىلپ گاھى قارشىلىققا ئۇچراپ تۇرىدۇ.ئەمما قۇرئان ئۇنداق ئەمەس
1400 يىل بولدى ئىسلاھات ئەمەس بىر ھەرىپىمۇ ئۆزگەرتىلگىنى يوق ،باشقىلارنىڭ سەۋەپ كۆرسىتىشىگە ئۇرۇن يو ق شۇنداق مۇكەممەل،تارىختىن بۇيان ھۆرمەتلىنىپ ئىززەتلىنىپ كىلۋاتىدۇ، ئمانىڭىز قانچە يۇقۇرلىسا شۇنىڭغا ئەگىشىپ ھەرخىل سىناقلارغا دۇچ كىلسىز ،
بۇرۇنقى كىچىك بىلىنگەن ئىشلار ئەمدىلىكتە چوڭ بىلنىشكە،ئىشلارنى تاماكىدىن باشلاپ ئۈزگەرتىشكە قىزلىرمىز مودا قىلۋالغان ئىپپەتلىرنى يۇشرۇشقا باشلايدۇ،باشقىلار ئالدىدا ئىزا تارتىپ ياشاشنى بىلدىغان،ئۇغىرى قاراقچىلىرمىز ھچ بولمىسا تاپقىننڭ بىر ھەقسسىز ئىئانە قىلىدۇ،  ئىنساپىنى تىلەپ ،شۇ قاراقچىلىق كۇچسىدىن بىر قۇتلۇشنى ئارزرۇ قىلىدۇ،
بىر دىنغا ئىتقاد قىلىش خۇددى ياخشى بىر ئەسەر يىزىش جەريانىغا ئوخشايدۇ،
ئەسەرگە تەييارلىغان ماتىرياللىرڭىز گاھى سىناقلاردىن ئۈتەلمىگەندەك ھەممە ئىشلار ئۇنداق ئاسان بۇلۇۋەرمەيدۇ،سىزنى قايمۇقتۇردۇ ،ئەسەر يىزىشتىن ۋاز كىچىشكە زورلايدۇ،ئىرادىڭىزنى سىنايدۇ،بىر قىتىملىق مەغلۇبىيەت ئالدىدا باش ئەگمەسلىكنى ئىستەيدۇ.دىنمۇ شۇنداق سىزنىڭ بىر قىتىملىق ئىچىپ قۇيغان ھارقىڭىز سىزنىڭ مەڭگۈگە مۇسۇلمان بۇلالمايدىغانلىقىڭىزنى ئىپادىيەلمەيدۇ،قىسىقىغىنا ئۆزگىرشىڭىزگە قاراپمۇ سىزنىڭ مۇسۇلمان ئىكەنلىكىىزنى ئىپادىلەپ بەرمەيدۇ،ھايات تۇلمۇ ئۇزۇن ۋە تۇلمۇ بەك قىسقا ئىمان سىزنىڭ قىسقا ھاياتتا  مەڭگۈلۈڭ ھاياتقا بولغان ئىنتىلشىىڭزنى كۈچەيتىدۇ،
بۇ دۇنيادىكى بىز تالشىۋاتقان پۇل مال دۇنيا،ۋە ھاكازالار ھەممسى ئاللانىڭ دەرگاھىدا ھىچنىمگە ئەرزىمەيدۇ،ئەمما رىرئال تۇرمۇشتا ئۇنداق ئەمەس،
پۇل ئۈچۈن ۋەتىننى .ئانسىنى ،بالسىنى ،ئىشقىلىپ ساتمىغان نەرسىللىرمىز قالمىدى ،
سىتىپ بۇلۇپ پۇشمان قىلغانلىرمزمۇ،ۋە تەلۋىلەرچە داۋاملاشتۇرۋاتمىز.
ئىمان بىزدىن ئاخرىقى تىنقىمىز قالغۇچە چىڭ تۇرشىمىزنى بىر ئىككى قىتىم مۇش يەپ كەينىگە داجىشنى ،بىر ئىككى قىلىپ قۇيغان خاتالىق بىلەن ئۈزنى بىر ئۈمۈر ھاقارەتلمەسلىكىنى ،ئاخرقى بىر تىنقىدا بولسىمۇ ئىمانغا كىرسە بولدىغانلىقىنى ،ئاللانىڭ ھەر كىشگە بەرگەن باھالىرنىڭمۇ ئۇخشاش بولمايدىغانلىقىتىمۇ،شەيتان دىگەن مەلئۇن داۋاملىق ئىنسانلارنى ئازدۇرۇشقا تاكى قىيامەتكىچە رۇخسەت ئالغانلىقى قىيامەتتە ئەگەشگۈچىللىرى بىلەن بىللە جازانى قۇبول قىلشقا تەييار ئىكەنلىكىنى ئىستەپتۇ،شەيتاننىڭ قۇلدن قۇتلۇشقا تىرشىۋاتقانلارمۇ ،ۋاز كەچكەنلەرمۇ،تىركىشىپ غەلبە قىلغانلارمۇ بار .
