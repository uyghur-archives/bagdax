    جىياڭ جيېشى «شىئەندىكى يېرىم ئايلىق ئىشلار خاتىرسى»دا ئۆزىنىتۇتقان سۇن مىڭجيۈنى تىلغا ئالىدۇ.1936ـ يىلى دېكابىر تاڭ سەھەردە يۈز بەرگەنھەربىي ئۆزگىرىشتە ،جىياڭ جيېشى «تامدىن سەكرەپ سەكرەپ قېچىپ » ،«قاراڭغۇدا پەرىق ئېتەلمەي تام سېرتىدىكى ئازگالغا چۈشۈپ كېتىپ »،«مىڭبىر تەسلىكتە ئازگالدىن ئۆمىلەپ چىقىدۇ »،ئاندىن ،ئۇ دىققەتسىزلىكتىن تېيىلىپ كېتىپ ئازگالغا چۈشۈپ كېتىدۇ.ئۇ ئاشۇ غارداتۇرۇپ «يۈزبەرگەن ئۆزگىرىشنى كۆزىتىدۇ .»
    ــ ھەربىي ئۆزگىرىش قىلغان قىسىمدىكىلەر ھەممە جاينى ئاختۇرىدى .  ئۇلار مەن تۇرغان غارنىڭ ئالدىدىن ئىككى نۆۋەت ئۆتكەن بولسىمۇ ،مېنى غاردا ئىكەنلىكىمنى بايقىمىدى.شۇ ئەسنادا ،مەن تۇرغان يەردىن تۆت ،بەش ماڭدام يىراق يەردە ئۆزگىرىش قىلغان قىسمىدىكىلەر بىلەن بىرەيلەننىڭ گەپ تالىشىۋاتقان ئاۋازىنى ئاڭلاپ قالدىم .كۆزىتىش ئارقىلىق ئۇنىڭ جىياڭ شىياۋچىڭ ئىكەنلىكىنى بىلدىم .ھەربىي ئۆزگىرىش قىلغان قىسىمدىكىلەرمۇ ئاختۇرۇشنى جىددىيلەشتۇردى .ئۇلارنىڭ غار ئۈسىتىدە قىلىشقان : ــ غار ئىچىدەپۇقراچە كىيىنگەن بىرەيلەن تۇرىدۇ ،ــ ئالدى بىلەن بىر پاي ئوق چىقىرىپ ئاندىن بىرنىمە دىمەمدۇق .ـ دېدى .يەنە بىر ئەسكەر ئۇلارنى توسۇپ :ــ كەلسە كەلمەس گەپلىرىڭنى قوي ــ دېدى .مەن ئۇلارغا قارشېلىق كۆرسۈتۈپ : مەن جياڭ ۋېييۇەنجاڭغاھەرقايسىڭنىڭ يولسىزلىق قىلىشقا ھەددىڭ يوق ! ئەگەر سىلەر مېنى ئەسىرگە چۈشۈرمىزدېيىشىدىكەنسىەن ،ئۇھالدا ئېتىۋەتسەڭ دەرھال ئېتىۋەتكىنكى ،بىراق قىلچىمۇ خورلۇشۇڭغايول قويمايمەن ،ــ  دېدىم .ھەربىي ئۆزگىرىش قىلغان قىسىمدىكى ئەسكەرلەر ئۇنداق قىلمايدىغانلىقىنى ئېيتتتى .ئۇلار ئاسمانغا ئۈچ پاي ئوق ئېتىپ :
ــ جياڭ ۋېييۈەنجاڭ بۇيەردە ئىكەن !ــ دەپ ۋارقىرۋىدى.
يىڭجاڭ سۇن مىڭجيۇ ئالدىغا كېلىپ تەزىم قىلىپ يىغلاپ تۇرۇپ:
ـــ  ۋېييۈەنجاڭ ،ئۆزلىرىنىڭ تاغدىن چۈشۈشلىرىنى ئۆتىنىمىز ،ــ دېدى .مەن شىڭيۈەنگە قورشاپ ھۇجۇم قىلغانلارنىڭ جاڭ شۆلياڭنىڭ مۇھاپىزەت ئەترىتىنىڭ ئىككىنجى يىڭى ئىكەنلىكىنى بىلدىم .سۇن مىڭجيۇ مېنى مۇھاپىزەت قىلىپ تاغدىن ئېلىپ چۈشتى .خۇا چىڭ كۆلچىكى يېنىدىكى شىڭ يۈەنگە كەلگەندە،مەن ئۆيگە كىرىپ بىرپەس ھاردۇق ئېلىۋالاي ،دېدىم .سۇن مىڭجيۇ ئۆيدە نەرسىلەنىڭ قالايمىقان تاشلانغانلىقىنى ،بىر نەچچە جەسەتنىڭ ئوڭغۇل دوڭغۇل ياتقىنىنى كۆرۈپ:
ــ ۋېييۇەنجاڭ ،ئۆيلىرىنى بەك قالايمىقان قىلۋېتىپتۇ،باشلىقىم ئۆزلىرىنى شەھەرگە ئېلىپ كىرىشىمنى بۇيرىدى ،ـــ دېدى .مەن سۇنمىڭ جيۇغا: ــ مۇئاۋىن قوماندانىڭنى تېپىپ كەل!ــ دەپ بۇيرۇق قىلدىم .
ــ مۇئاۋىن قوماندان ئۆزلىرىنى شىئەن شەھىرىدە كۈتىۋالىدۇ .پېقىر باشلىقىمغا ئاسىيلىق قىلالمايمەن ،دۆلەت ئىشىنى كۆزدە تۇتۇپ ئۆزلىرىگە يۈزتۇرانە مەزكۇر ئىلتىماسمنى دېدىم .ۋېييۈەنجاڭنىڭ پېقىرنىڭ ئىلتىماسىنى قوبۇل قىلىشىنى ئۆتىنىمەن ،ــ دېدى سۇن مىڭجيۇ.مەن ئۇنىڭغا ئاچچىقلاپ :
ــ شۇنچىۋالا يۈزسىزلىك قىلغان بارمۇ ! تولا كاپشىماي،مېنى ئۆلتۈرگۈڭ بولسا تېزدىن ئۆلتۈرگىن !ــ دېدىم .سۇڭ مىڭجيۇ بىلەن 105 ـ شى ئىككىنجى لۈينىڭ باشلىقى تاڭجۇنياۋ ماڭا يەنە ھەربىيچە چاس بېرىپ مېنى ماشىنىغا ئولتۇرغۇزۇپ شەھەرگە كېرىشىمنى ئۆتۈندى.مەن خەنچىڭ (جاڭ شۆلياڭنىڭ يەنە بىرئىسمى)بىلەن كۆرۈشۈپ ،ئىشنىڭ تىگى تەكتىنى سۈرۈشتۈرۈش ئۈچۈن ماشىندا يولغا چىقتىم .
    سۇڭمىڭ جيۇ بىلەن لۈيجاڭ تاڭ جۇنياۋ مېنى قولتۇقۇمدىن يۆلەپ ماشىنىغا چېقىرىپ ،ئىككىسى ئىككى تەرىپىمدە ئولتۇردى؛ يەنە بىر كانىۋاي يەنە جاڭ خەنچىڭ (جاڭ شۆلياڭ) نىڭ ئەڭ ئىشەنچىلىك مۇھاپىزەتچىسى تەن خەي ماشىنىنىڭ ئالدىدا ئولتۇردى .ماشىنا شىئەنگەقاراپ ئۇچقاندەك ماڭدى،شەرقىي قوۋۇقتىن ئۆتكەندە تاڭ لۇيجاڭ جاڭ خەنچىڭ ماشىنىسىنى يىراقتىن كۆرۈپ :
    ــ مۇئاۋىن قوماندان كەلدى!ــ دېدى .يېقىن كەلگەندىن كېيىن، قارىسام ئۇ جاڭ شۆلياڭ بولماستىن بەلكى مېنىڭ قەيەرگە ئېلىپ باردىغانلىقىمنى ئېيتقىلى كەلگەن كىشى ئىكەن.تاڭ لۈجاڭ كانىۋاي تەن خەيدىن :
    ــ ۋېييۇنجاڭنى نەگە ئاپىرىمىز ؟ـــ دەپ سورىدى .كانىۋاي :ــ يېڭى شەھەردىكى چوڭ بىناغا ـــ دەپ جاۋاپ بەردى .يېڭى شەھەردىكى چوڭ بىنا يەنى شىئەن سۇيچىڭ مەھكىمىسى ياڭ خۇچىڭ تۇرۇشلۇق جاي ئىدى .بۇ گەپلەرنى ئاڭلىغاندىن كېيىن مەندە قورشاپ ھۇجۇم قىلغۇچىلارشەرقىي شىمال ئارمىيەسى تۇرسا ،مېىنى نىمىشقا يەنە ياڭ(خۇچىڭ )تۇرۇشلۇق جايغا ئاپىرىپ قويىدىغاندۇ؟دېگەن گۇمان پەيدا بولدى . بۇچاغدا ماشىناشەرقىي دەرۋازىغا يېققىنلاپ قالغانىدى ،قارىسام ئەسكەرلەر «17ـ ئارمىيە »دىگەن يەڭ بەلگىسىنى تاقىۋالغانكەن .،بۇنى كۆرۈپ تېخىمۇ چۆچۈدۈم ئۇنىڭغا ئۇلاپلا تۆنۈگۈن ئاخشامقى زىياپەتكە تەكلىپ قىلىنغان گېنراللارنىڭ كۆزئالدىمغا كەلتۈردۈم ،ياڭخۇچىڭنىڭ زىياپەتكە كەلمىگەنلىكىگە قاراپ ،ئۇنى ئالدىن جاڭ (شۆلياڭ) نىڭ زىياپىتگەبېرىپ ،جاڭ شۆلىياڭ تەرپىدىن تىزگىنلەنگەن ۋە تۇتۇپ قېلىنغان ئوخشايدۇ ،دەپ ئويلىغانىدىم .مەركەزنىڭ شىئەندىكى يوقىرى دەرىجىلىك گېنرالى ئۇلار تەرپىدىن بىراقلا ئوجۇقتۇرۋېتىلمىگەندۇ،دەپ بەك تەشۋىش قىلغانىدىم .بايىقى بىلەكلەرگە«17-ئارمىيە »بەلگىسىنى تاقىۋالغان ئەسكەرلەر 17-ئارمىيەنىڭ شىئەندىكى قالدۇق قىسىملىرى تەرىپىدىن قورالسىزلاندۇرۇلغاندىن كېيىن، باشقىلارنى قايمۇق تۇرۇش ئۈچۈن ئەسلىدىكى ھەربىي كىيىملىرىنى سېلىپ تاشلاپ ،شەرقىي شىمال ئارمىيەسىنىڭ كېيىملىرىنى كىيىشكە بۇيرۇق قىلىنغان ئەسكەرلەرمىدۇ ،ــ دەپ گۇمان قىلدىم .ياڭخۇچېڭنىڭ ئىنقىلاپقا قاتناشقىنىغا ئۇزۇن بولغانلىقى ،شۇنداقلا ئۇنىڭ گومىنداڭنىڭ پىشقەدەم ئەربابى ئىكەنلىكىنى كۆزئالدىمغا كەلتۈرۈپ ،ئۇنىڭ ھەربىي ئۆزگىرىش قىلغاندا دوست تارتىشىپ بەرمەيدىغانلىقىغا ئىشىنەتتىم .شەھەرگە كىرەي دەپ قالغاندا،تاڭ جۈنياۋ ماڭا قاراپ ھەسرەت بىلەن : ــ ۋېييۈەنجاڭ ،چېكىلىرىدىكى ئاقلاربارغانسېرى كۆپىيىشكە باشلاپتۇ ،ئىككى يىلنىڭ ئالدىدا لۈشەندە ئۆزلىرىدىن تەلىم ئالغان ۋاقتىمگىكىگە قارغاندا بەكلا قېرىپ كېتىپلا !راستىنى ئېيتقاندا دۆلەت ۋېييۇەنجاڭسىز بىر كۈنمۇ مەۋجۇت بولۇپ تۇرالمايدۇ ! مانا ھازىر شىئەن شەھرىنىڭ ئەھۋالىنى بۇرۇنقى ئىككى يىل ئالدىدىكى ئاۋاتچىلىققا زادىلا ئوخشاتقىلى بولمايدۇ.ۋېييۇەنجاڭدىن بەلەك ئادەم غەربىي شىمال قۇرلۇشىغا رىياسەتچىلىك قىلسا ،بۇخىل ھالەتنى ئۆگەرتەلەمدۇ ؟ شۇ ۋەجىدىن ۋېييۈەنجاڭ ئۆزلىرىنى بەك ئاسرىغايلا،ــ دېدى.مەن بۇنىڭغا جاۋاپ بېرىشكە ئۈلگۈرمەيلا سائەت ئوندا يېڭى شەھەردىكى چوڭ بىناغايىتىپ كەلدۇق .
     سۇيچىڭ مەھكىمىسىگە كىردىم ،بىراق ياڭ خۇچىڭ كۆرۈنمىدى ،بىر پەس ئۆتكەندىن كېيىن ،مەھكىمە«ئالاھىدە يىڭ» نىڭ باشلىقى سۇڭ ۋېنىمېي كىردى ،سۇڭ مىڭجيۇ قوغداش مەجبۇرىيىتى جەھەتتىكى ئىشلارنى سۇڭ ۋېنمېيغا تاپشۇرۇپ قويۇپ چىقىپ كەتتى ...
     ئەتىسى13- دىكابىر كۈنى جاڭ (شۆلياڭ) مېنى يوقلاپ كەينى كەينىدىن تۆت قېتىم كەلدى.ئۇنىڭ چىرايى ئېلگىرىكىدىن غەمكىن ئىدى .شۇكۈنى كېچىسى ھەربىي كىيىملەربىلەن مېنى يەنەيوقلاپ كەلدى .ئىشىكنى ئېچىپ ،ئۇخلاپ قالغانلىقىمنى كۆرۈپ :ــ ۋېييۇنجاڭ ئۇخلاپتۇ،ئويغاتماي! ـــ دەپ چىقىپ كەتتى .ئۇزالغا  چىققاندا ،بىرنەچچەيلەن بىر ئىش ئۈستىدەمەسلىھەتلىشىۋاتقاندەك قىلدى.ئۇلارنىڭ كىملەر ئىكەنلىكىنى ئاۋازىدىن پەرىق ئەتكىلى بولمايتتى .ئۇلار خەلىقنىڭ سوت قىلىشىغا تاپشۇرۇپ بىرەيلى دېگەندەك گەپلەرنى قىلىشىۋاتقاندەك قىلدى .كېچە سائەت 12 يېرىم بولغاندا سۇڭ ۋېنمېي:
      ـــ سۇن مېڭجيۇ كۆرۈشكىلى كەپتۇــ دېدى ،ئۇنىڭغا : ــ ئۇخلايمەن ،ــ دېسەم ،ئۇ يەنە :ــ كىرىپ كۆرۈشىمەن ،ۋېييۇەنجاڭنى بۇ يەردىن يۆتكەپ كەتكىلى كەلدىم ــ دەيدۇ، ــدېدى .شۇ ئەسنادا سۇن مىڭجيۇ يېنىغا تاپانچا ئاسقان ھالدا مەن بىلەن كۆرۈشكىلى كىرىپ:
    ــ بۈگۈن ئاخشام ۋېييۇەنجاڭنى بۇ يەردىن يۆتكەيمىز ،ــ دېدى.مەن ئۇنىڭغا :
    ــ بۇ يەر قەبرىگاھىم بولسۇن ،ئۆلسەم ئۆلىمەنكى ھەرگىز يۆتكەلمەيمەن .مەن ئۆلگەندىن كىيىن ،سەن ئىككىڭ مۇشۇ ئۆينىڭ سىرتىدىكى چوڭزالنى مېنىڭ قەبرىگاھىم قىلىشقا بۇيرۇق چۈشۈرسەڭ بولىدۇ.سېنىڭ ئۆيۈمگە قوراللىق كىرگىنىڭدىن قارىغاندا تەھدىت سېلىۋاتقاندەك قىلىسەن ،بىلىپ قويغىنكى ،بۇيەردەمەندە قورال بولمىغىنى بىلەن ھەققانىي جاسارەت بار ،ئۆلتۈرسەڭ مېنى ئۆلتۈرۈش،ئەمما بۇيەردىن ھەرگىز يۆتكەلمەيمەن ،ــ دېدىم .
     سۇن مىڭجيۇئۆزىنى بىر ئاز تۈزۈۋېلىپ ،گىپىنى سىلىقلاشتۇرۇپ ،كەينى كەينىدىن كۆچۈرۈشكە دەۋەت قىلىپ ،سائەت ئىككىگىچە چىقىپ كەتمىدى .مېنىڭ قاتتىق ئاچچىقىم كېلىپ :ـــ تۈن نىسپىي بولغاندىمۇ قوراللىق كۆزۈمگە كىرىۋېلىپ ئارام بەرمىدىڭ ،بۇ زادى نىمە ئۈچۈن؟مەن سېنىڭ باشلىقىڭ بولۇش سۈپىتىم بىلەن بۇيرۇق قىلىمەنكى سەن بۇيەردىن دەرھال چىقىپ كەت ،بۇيرۇققا بىنائەن چىقىپ كېتىشىڭ لازىم دېدىم .سۇن مىڭجيۇ ئاندىن چىقىپ كەتتى.
    جىياڭ جيېشىنىڭ يۇقىرىقى گەپلىرىنى سۇن مىڭجيۇنىڭ «جياڭ جيېشىنى لىنتۇڭدا تۇتۇپ قېلىش » دېگەن ئەسلىمىسى بىلەن سېلىشتۇرىدىغان بولساق ئەھۋال مۇنداق 1ـ جياڭ جيېشى : ئاكامنىڭ ئوغلى جياڭ شىياۋجۇنىڭ ھەربىي ئۆزگۈرۈش قىلغان قىسىمدىكىلەر بىلەن قاتتىق تاكاللىشىۋاتقانلىقىنى ئاڭلىدىم ،دېگەن .بۇ ئۇنىڭ جىياڭ شىياجىڭىنى پەردازلاپ كۆرسەتكىنىدىن باشقا نەرسە ئەمەس .ئەمىليەتتە بولسا ،سۇن مىڭجيۇ ئەينى چاغدىكى ئەھۋال ئۈستىدە توختۇلۇپ :ــ جياڭ شىياجىڭ دېگەن «بۇ بىر نېمىنىڭ قورققىنىدىن چىرايىدا ئادەم سىياقى قالمىغاندى »،گەپ قىلغاندا «تىلىنى چايناپ» كەتكەن تۇرسا،«قاتتىق تاكاللىشىدىغان»ئىش نەدە بولسۇن ؟دەيدۇ .جىياڭ شىياۋجىڭ جياڭ جيېشىنىڭ دېرىكىنى بىلمەيدىغانلىقىنى ئېيتقاندىن كېيىن ،سۇن مىڭجيۇ ئۇنىڭغا تاپانچىسىنى تەڭلەپ تۇرۇپ : ـــ راست گەپ قىلمىساڭ ،ھازىرلا ئېتىپ تاشلايمەن !ــ دېگەن .شۇنىڭ بىلەن جياڭ شىياۋچىڭ كۆزى بىلەن تاغ تەرەپنى ئىما قىلغان».سۇن مىڭجيۇ ئۇ«ئىماقىلغان نىشان بويىچە قىسىمنىڭ تاغقا چىقىپ ئاختۇرىشىغا قوماندانلىق »قىلغان .(2) جىياڭ جىيىشى :ئۇلار مېنى تېپىۋالغاندا ئۇلارغائېتىراز بىلدۈرۈپ :
ـــ مەن جياڭ ۋېييۇەنجاڭ بولىمەن ،سەنلەرنىڭ ئەدەپسىزلىك قىلىشىڭغا يول قويۇلمايدۇ!دېدىم،ـــ دەيدۇ.بىراق سۇن مىڭجۇنىڭ ئەسلىمىسىدە ـــ ھېلىقى نىشان بويىچە ئاختۇرۇپ يوغان تاشنىڭ يىنىغابارغاندا ،كۆزىمىزگە غاردا بىر ئادەمنىڭ تۈگۈلۈپ يېتىۋالغانلىقى كۆزگە چېلىقتى،شۇنىڭ بىلەن :
ـــ ۋېييۇەنجاڭمۇ سەن ،ئەمەسمۇ ؟دەرھال بۇياققا چىق،چىقمىساڭ ئېتىپ تاشلايمىز! ــــ دەپ ۋارقىرىدۇق .ئۇ غارئىچىدىن :
ـــ مەن ۋېييۇەنجاڭ ،ئوق چىقارماڭلار ،ئوق چىقارماڭلار !دېگىنىچە ئۆمىلەپ چىقتى ،ـــ دەيدۇ.
(3) جىياڭ جيىشى سۇن مىڭجيۇ مېنى كۆرۈپلا «ئۇزۇنغىچەيۈكۈنۈپ تۇرۇپ يىغلاپ كەتتى » دەيدۇ.بىراق سۇن مىڭجۇنىڭ ئەسلىمىسى بولسا : ـــ جياڭ جيېشىنىڭ قېشىغا كەلگىنىمدە ئۇ توڭلاپ تىترەپ كېتىپتۇ.ئۇ بېشىنى كۆتۈرۈپ ماڭغا قاراپ ئۆزىنى مەندىن قاچۇرۇپ :
ـــ مېنى ئېتىۋەت ... ـــ دېدى  مەن : سېنى ئېتىپ ئۆلتۈرمەيمىز ،ياپۇنغا قارشىتۇرسۇن دەيمىز ،ـــ دېدىم .شۇ چاغدا چىرايى تاتارغان ،پۇتى يالاڭ ئاياق ،ئۈستىگەقوڭۇر رەڭ تاۋار پەشمەت ،ئاستىغا ئاق ئۇخلاش تامبىلى كىيىۋالغان بولۇپ ،ئۈستى- بېشىنى چاڭ توزان قاپلىغانىدى .ئۇ:
ـــ سىلەر قەيەردىن كەلدىڭلار ؟ دەپ سورىدى .مەن :ـــ شەرقىي شىمال قىسىمدىن كەلدۇق ،مۇئاۋىن قوماندان جاڭ شۆلياڭ بىزنى سەن جياڭ ۋېييۇنجاڭنى قوغداشقا ئەۋەتتى .ئۇسېنى شەھەرگە كىرىپ ،ياپۇنغا قارشى تۇرشىمىزغا،شەرقىي شىمالنى قايتۇرۋېلىشىمىزغا رەھپەرلىك قىلىشقا تەكلىپ قىلدى ،ــ دەپ جاۋاپبەردىم .
ـــ ھەئە سەن سۇن يىڭجاڭ ،سۇن مىڭجيۇ بولامسەن ؟
ـــ ھەئە !سەن ئىسمىمنى قانداق بىلدىڭ ؟
ـــ بىلىمەن ،بىرەيلەن دەپ بەرگەنىدى .
ـــ جياڭ جيېشى مېنىڭ ئۇنىڭغا زىيان زەخمەت يەتكۈزمەيدىغانلىقىمنى بىلگەندىن كىيىنلا:
ـــ سەن ياخشى يىگىت ...مېنى ئېتىۋەتكىن مېنى ئېتىۋەت،دېدىم .
ـــ مۇئاۋىن قوماندان ۋېييۇەنجاڭنى ياپونغاقارشى تۇرىشىمىزغارەھبەرلىك قىلسۇن دەۋاتىدۇ ،بىزگە ۋېييۇەنجاڭنى ئېتىۋېتىڭلار دىگىنى يوق.ۋېييۇەنجاڭ تاغدىن دەرھال چۈشۈپ شەھەرگە كىرگىن ،مۇئاۋىن قوماندان سېنى شەھەردەكۈتۈپ تۇرۇۋاتىدۇ،دەپ چۈشەندۈردۈم ھەم دەۋەت قىلدىم ،ـــ دەيدۇ .
(4) جياڭ جيېشى :
ــ مۇئاۋىن قوماندانىڭنى تېپىپ كەل ! دېدىم ،ــ دەيدۇ.سۇن مىڭجيۇنىڭ ئەسلىمىسىدە بولسا :
ـــ جياڭ دوسمىيىپ يەردىلا ئولتۇرۋالدى ،چالۋاقاپ تۇرۇپ :
ـــ مۇئاۋىن قوماندانىڭلارنى باشلاپ كېلىڭلار ! بېلىم ئاغرىيدۇ ،ماڭالمايمەن !ـــدېدى.جياڭ جيېشىنىڭ بېلىنىڭ ئاغرىغانلىقى راست ئىدى ،ئۇ ئۆيدىن سىرتقا قېچىپ ئارقا تامدىن سەكرىگىنىدە يىقىلىپ چۈشكەن ،ـــ دەيدۇ.
5 ـ جياڭ جيېشى :
ـــ ماشىنىغا چىقىشقا ئۇنىمىدىم ،ـــ دەيدۇ.
ـــ سۇن مىڭجيۇنىڭ ئەسلىمىسىدە بولسا :
ــــ جياڭ جيېشىنىڭ ماڭغىلى ئۇنىمىغانلىقىنى كۆرۈپ :«بۇيەر تىنىچ ئەمەس ،ۋېييۇەنجاڭ،تاغدىن دەرھال چۈشكىن  ،بېلىڭ ئاغرىساسېنى يۈدۈپ ئەپ چۈشەيلى،ـــ دەپ نەسىھەت قىلدىم .ئۇيەنە قىمىر قىلمىدى ھەمدە : ــ ئاتقا مىنىپ ماڭىمەن ،ــ دېدى ،مەن ئەتراپىمدىكى مۇھاپىزەتچىلەرگە جياڭ جيېشىنى قولتۇقىدىن يۆلەپ ئورنىدىن قوپۇرۇپ تاغدىن ئېلىپ چۈشىشنى ئىشارەت قىلدىم .خۇاچىڭ كۆلچىكىگە كەلگەندە ،ئۇشەھەرگە  كىرگىلى ئۇنىماي تۇرۋالدى .مەن بىرقانچە مۇھاپىزەتچى بىلەن بىللە ئۇنى يۆلەپ ئىتتىرىپ يۈرۈپ پىكاپكا ئەپچىقتۇق .پىكاپقا مەنمۇ چىقتىم ،ـــ دەيدۇ.
     (6) جياڭ جيېشى ماشىنىغا چىققاندىن كېيىن : ــ سۇڭ مىڭجيۇ بىلەن لۈيجاڭ تاڭ جۈنياۋ ئىككسى قولتۇقۇمدىن يۆلەپ پىكاپقا چىقاردى ھەمدەمېنى ئوتتۇرغا ئېلىپ ئىككى تەرپىمدە ئولتۇردى ،يەنە بىر كانىۋاي  ـــ جاڭ خەنچىڭنىڭ سادىق مۇھاپىزەتچىسى تەن خەيپىكاپنىڭ ئالدى ئورۇندۇقىدا ئولتۇردى ،ـــ دەيدۇ .سۇن مىڭجيۇنىڭ ئەسلىمىسىدە بولسا: ـــ پىكاپتا مەن جىياڭ جيېشىنىڭ سول تەرىپىدە ،تاڭ جۈنياۋ ئوڭ تەرپىدە ئولتۇردۇق: ئالدى ئورۇندۇقتا كانىۋاي تەن خەي بىلەن شوپۇر ئولتۇردى .جياڭ جيېشى قاپىقىنى تۈرۈپ :
     ـــ بەك قىستىلىپ كەتتىم دېدىيۇ ،گەپنىڭ ئاخىرىنى قىلماي ئىچىگە يۈتىۋەتتى ،پىكاپ شىئەنگە قاراپ ئىلدام ماڭماقتا ئىدى .جياڭ جيېشى ئىزچىل تۈردە كۆكرىكىنى قولى بىلەن تۇتۇپ ،كۆزىنى چىڭ يۇمۇپ ،پۇش ،پۇش تىناتتى .مەن ئۇنىڭدىن :ـــ بۇگۈندىن ئېتىبارەن ئۆتكەن ئىشلارغا سالىۋات ،بۇندىن كىيىن قانداق قىلماقچى بولىۋاتىسەن ؟ ـــ دەپ سورىدىم .جياڭ جيېشى جاۋاپ بېرىپ : ــ مۇئاۋىن قوماندانىڭلار ئامالى بار ،ــ دېدى .
ـــ مۇئاۋىن قوماندانىمىز جاڭ جيېشىنىڭ ياپونغا قارشى تۇرۇشنى ھىمايە قىلىدۇ ،دېدىم مەن مېنىڭمۇ ياپونغا ياخشى قارشى تۇمىغىنىم يوققۇ ،ــ دېدى جاڭ جيېشى بوش ئاۋاز بىلەن،ــ دېدى جاڭ جيېشى بوش ئاۋاز بىلەن ــ گۇڭچەنداڭغاقارشى تۇرۇش ــ دۆلەتنىڭ سىياسىتى ،خاتا ئەمەس بۇنى مەن بەلگىلىگەن .
     جاڭ جيېشىنىڭ ئاچچىقلىنىۋاتقىنى مانا مەن چىقىپ تۇراتتى.مەن ئۇنىڭغا قائىدىلىك رەدىيە قايتۇردۇم .ئۇ: ــ سەن !... مەن دۆلەتنىڭ داھىسى ،دۆلەتنىڭ ئەڭ ئالىي باش يىتەكچىسى ،مەن بەلگىلىگەن دۆلەت تەدبىرى توغرا !سەن بۇنى چۈشەنمەيۋاتىسەن !ــ دېدى .
     ئاشۇ چاغدا جياڭ جيېشىنىڭ كەيپىياتىدا ئۆزگىرىش بولدى،چىرايىدىن ئاچچىقلىنىش ئالامىتى كۆرۈلدى .ئۇ:ـــ سەن ئەمدى ماڭا قايتىدىن گەپ قىلغۇچى بولما ،ماڭاگەپ قىلما ،ـــ  دېدى دە ،كۆزىنى يۇمىۋالدى.ياچياۋ كۆۋرىكى ئەتراپىغا كەلگەندە ،يولدا ھەربىيلەرناھايىتى كۆپىيىپ كەتكەنلىكىنى كۆردۇق .(كېيىن ئۇقسام جاڭ شۆلياڭ ئۇلارنىمۇھاپىزەتچىلەر يىڭىنىڭ جياڭ جيېشىنى تۇتۇشىغا ياردەمگە ئەۋەتكەنىكەن ) ،پىكاپ بىردەم توختاپ ،بىردەم ماڭاتتى ،جياڭ جيېشى سىرتقا قارىدى دە ،بىردىنلا مەندىن بۇلار قايسى قىسىمدىن ؟ــ دەپ سورىدى .شەرقىي شىمال قىسىمدىن ، ــ دەپ قىسقىلا جاۋاپ بەردىم.پىكاپ شىئەنىڭ شەھەر دەرۋازىسىغا كەلدى .دەرۋازىدا قاراۋۇل تۇراتتى .جياڭ جيېشى يەنە :ـــ بۇلار قايسى قىسىمدىن ؟ــ دەپ سورىدى .
     17ـ ئارمىيدىن ، ـــ دەپ جاۋاپ بەردىم .شەھەردەرۋازىسىدىن كىرگەندىن كىيىن پىكاپ ئۇدۇللا يېڭى شەھەر چوڭ بىناسىدىكى سۈيچىڭ مەھكىمىسىگە قاراپ ماڭدى، مەن جياڭ جيېشى بىلەن بىللە ئاللىبۇرۇن تەييارلاپ قويۇلغان قونالغۇمىزغا كىردۇق،ـــ دەيدۇ .
   (7) جياڭ جيېشى :
ـــ يېڭى شەھەردىكى چوڭ بىناغا كەلگەندىن كېيىن سۇنمىڭجيۇ مۇھاپىزەت قىلىش مەجبۇرىيىتىنى سۇڭ ۋېينمېيغا تاپشۇرۇپ چىقىپ كەتتى ،ـــ دەيدۇ؛ بىراق سۇن مىڭجيۇنىڭ ئەسلىمىسىدە بولسا :
ـــ مەن جياڭ جيېشى بىلەن بىللە قونالغۇغا كىرگەندىن كېيىن ،ئۇنى يۆلەنچۈكۈك ئورۇندۇقتا ئولتۇرغۇزۇپ قويۇپ ،ئۆزۈم ئۇنىڭ كەينىدىن شىرەگە يۆلۈنۈپ ئۆرە تۇردۇم .جياڭ جيېشى ھېلىمۇ كۆكرىكىنى تۇتقىنىچە ئۈندىمەي ئولتۇراتتى .ئارىدىن بىرپەس ئۆتكەندىن كېيىن ،زۇۋانغا كېلىپ مېنى ئولتۇرۇشقا بىرنەچچە رەت دەۋەت قىلدى ھەمدى يېنىش ـ يېنىشلاپ :
ـــ سەن ياخشى يىگت،ـــ دېدى .شۇ ئەسنادا بىرەيلەن چاي ئېلىپ كىردى ،ئارقىدىنلا يەنە بىرەيلەن جاڭ شۆلياڭنىڭ پەلتۇسىنى ئېلىپ كىردى،ئارقىدىنلا يەنە بىرەيلەن جاڭ شۆلياڭنىڭ پەلتوسىنى ئېلىپ كىردى ،ئاندىن كېيىن يەنە سۈت چاي ئېلىپ كىردى .جياڭ جيېشى چاينى ئىچمىدى ،پەلتونى كەيمىدى .ئۇمەندىن :
