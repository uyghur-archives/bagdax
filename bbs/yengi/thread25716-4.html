نېمە دېگەن ئېچىنىشلىق تۇرمۇش -ھە! بەزىلەرنىڭ سۆزلىرى شۇنداق سىپايە ،لېكىن كىشىلەرنى ئادەم ئورنىدا كۆرمەي ،ئۇلارنى خارلاشلىرى ......ئادەمنى  نەپرەتلەندۈرىدۇ. دەيلا. تۆغرا  بۇنى پەقەت ھازىرقى خۇجايدىنلام كۆرگىلى بولمايدۇ .ھازىرقى خۇجايىنلار دىنگە ئىتقات قىلغان بىلەن ئاللاھقا ئىتقات قىلمايدۇ. شۈڭە خۈجايىنلەر ئىشلىتىۋاتقان قىزلارنى مىنىڭ چۆرىلىرىم دەپ خالغانچە خارلايدىغان. ئازار بىردىغان ئىشلار كىلىپ چىقۋاتىدۇ. بۇ خۈجايىنلەر مۇشۇ سەبى قىزلارنى پەقەتلام ئىلىپ ساتالمىغان بىلەن دىنلار تەكىتلەۋاتقان چورلەرگە مۈئامىلە قىلغاندەك مۇئامىلە قىلدۇ. ھەر ھالدا كوم پارتىيە ئادەم ئىلىپ ساتقانلارنى ۋە خارلىغانلارنىڭ جاجىسنى بىرۋاتقان بولسىمۇ ، يەنىيلام قانۇن تۆردىن قىچىپ قۇتۇلغانلار بار .بۈ  يىتىم بالار ،مۇشنىڭغا ئۆخشاش سەبىلەرنى ،خارلىغانلار ئاللاھنىڭ قانۇندىن قىچىپ قۇتلالمايدۇ. بۇ خۇجايىنلارنىڭمۇ بالا چاقىلىرى بار . ئاللاھ بۇلار قىلغاندىنمۇ نەچە ھەسە كۆرسىتىدۇ.

بىزنىڭ قوشىنىمىزنىڭ قىزىمۇ  ئۈرۈمچىگە بالا باققىلى كەتكەنىتى ، ئاڭلسام ئۇ خەقلەر بالىمىز يەسلىگە كىرگىدەك بوغىچە بېقىپ بەرسە ئۆزىمىز دىپلوم ئەپ ئىشقا ئۇرۇنلاشتۇرۇپ قۇيىمىز دەپ ئېلىپ كىتىپتىكەنتۇق ، ئاشۇ خەقلەگىمۇ مەھەللىمىزمىڭ ئۇقۇمۇشلۇق ھىساپلىندىغان بىر ئۇقۇتقۇچى تۇنۇشتۇرۇپتىكەن ،  ئۇ قىز بىر يىلدا بىر كىلدۇ ، ھەر قېتىم بارمايمەن دەپ تۇرۋالسا ئاتا-ئانسى سىنى كىيىچە ئىشقا ئۇرۇنلاشتۇرۇپ قۇيىدۇ دەپ زۇرلاپ ئەۋەتىدىكەن ، ئۈچ يىلىدىن جىقراق تۇردىغۇ دەيمەن ، كىيىن بامىدى ، ئۇ قىز  ئۆزى بەك چىرايلىق ھايالىق قىز بالىتتى ،  كىيىن كەپتۇ كۆرەشتۇق ، گەپ سۆزلىردە ئۆزگىرىش ، يۈرۈش -تۇرۇشىدىكى ئۆزگىرىشلەر شۇنداق چوڭ ، ھەيرانلا قالدىم ، كىيىن ئۇلارمۇ ئىشقا ئۇرۇنلاشتۇرمىدى ، ئاڭلىسام ئازغىنە پۇل بىرىپتۇدەك ،  كىيىن قۇشنا مەھەللىدىكى بىر بالىغا تۇيىنى قىپ قۇيۇپتۇ ،  17 ياشتىن سەل ئاشقان قىزتى ، ئاڭلسام ئىككى ئۈچ  ئايمۇ ئۆي تۇتالماي ئاجىرىشىپ كىتىپتۇ ، قوشىنىمىزنىمۇ خېلى گەپكە قويغان ئوخشايدۇ ، ئويلاپ قالدىم ، بىر بالىنى ئۆز ئاتا-ئانسى بىر قوللۇق تەربىيلىمەي ، باشقىلار ئۇنداق قىپ بىردىكەن ، ئازغىنە پۇل بىردىكەن دەپ ناتۇنۇش ئادەملەرگە تاشلاپ بەرسە ، ئۇ قىز ئاتا -ئانا مىھىرى ، ئائىلە مىھرى دىگەندەكلەنى بۇلۇپمۇ شۇ خەقلەنىڭ قاش -كۈزىگە قاراپ ئولتۇرۇپ ، ئەڭ مۇھىم بولغان ئىمان -ئېتقادىمىزدىن يىراقىلىشىپ كەتمەسمۇ ، ئاتا-ئانىلارغا شۇنداق تەكلىۋىم ئاللاھ بىر پەرزەنىتىنى بەرگەنكەن ئۇنىڭ رىزقىنىمۇ قۇشۇپ بىردۇ ،شۇڭا يۈرەك پارىمىز بولغان پەرزەنتىلەرنى ئۆز تەربىيمىزدە ئاتا-ئانىلىق مىھىر -مۇھەبىتتى بىلەن بېقىپ چوڭ قىلشىنى ئۈمىد قىلمەن
