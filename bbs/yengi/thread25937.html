11- نۆۋەتلىك قومۇل قوغۇن بايرىمىدا «ئاۋام»دىكىلەرقومۇلغا كېلىپ ئاۋام
ئارىسىغا چوڭقۇر   چۆكتى


       يىلدا بىرقېتىم ئۆتكۈزۈلۈپ، قومۇل شەھىرىنىڭ سەيلى- ساياھەتچىلىك تەرەققىياتىنى ئىلگىرى سۈرۈپ، قومۇل شەھىرىنىڭ «قوغۇن بايرىمى»دىن ئىبارەت بۇ مەدەنىيەت سەھنىسى ئارقىلىق قومۇلنى تونۇتۇش، قومۇلنى تەشۋىق قىلىش سالمىقىنى يەنىمۇ يۇقىرى كۆتۈرۈپ، شەھەرنىڭ مەدەنىيەت ئوبرازىنى تىكلەشتە مۇھىم رولىنى جاراي قىلدۇرۇپ كېلىۋاتقان «جۇڭگو.قومۇل ‹شېرىن ساياھەت› 11- نۆۋەتلىك قوغۇن بايرىمى بۈگۈن قومۇل شەھىرىنىڭ قاراتال قوغۇن باغچىسىدا داغدۇغىلىق باشلاندى.
      بۇ قېتىمقى بايرامنىڭ باشلىنىش مۇراسىمى،بايرام تەرتىپى قاتارلىق ھالقىلىرىدا ئىلگىركى نۆۋەتلىك بايراملارغا قارىغاندا زورئۆزگىرىش ۋە يېڭىلىقلار ئوتتۇرىغا چىقىپ، بايرامنىڭ بارلىق تەرتىپ، ئۇسلۇپلىرىداپۈتۈنلەي شۇ جايدىكى دېھقانلار ۋە كەڭ خەلق ئاممىسىنىڭ ئارزۇسى ئەكىس ئەتتۈرۈلۈپ،پائالىيەت جەريانى ئەل راھى، ئەل مەنپەئەتى، ئەل كۆڭلى بويىچە ئورۇنلاشتۇرۇلۇپ،خەلق ئاممىسىنىڭ ئەنئەنىۋىي بايرام ئۆتكۈزۈش ئارزۇسى تولۇق نامايەن قىلدۇرۇلدى.شۇڭا بۈگۈنكى بايرام شادلىقىغا جۆر بولۇپ، تىلنى يارغۇدەك «قەنت، شېكەر، ناۋات»قوغۇنلاردىن قانغۇدەك يەپ، قوغۇننىڭ تاتلىقى، دىلنىڭ شادلىقى ئۈچۈن تەرەپ-تۇشتىنكەلگەن يەرلىك ھەر مىللەت ئاممىسى، دېھقانلار، ئوقۇغۇچىلار ۋە ئىچكىرى ئۆلكىلەردىن كەلگەن ساياھەتچىلەر بولۇپ 30 مىڭدىن ئاشتى. قوغۇن بايرىمىغا ئورۇنلاشتۇرۇلغان يۈرۈشلۈك پائالىيەت تۈرىمۇ ئىلگىرىكىدىن كۆپەيگەن بولۇپ،قوغۇن بايرىمىغاقاتناشقىلى كەلگەن ھەر مىللەت ئامما، قوغۇن مۇسابىقىسى، ھەرقايسى  يېزىلارنىڭ ئالاھىدە يەرلىك  مەھسۇلاتلىرى، كىچىك تىپتىكى يىمەك  ئىچمەك  پىششىقلاپ  ئىشلەش كارخانىلىرىنىڭ ئالاھىدە مەھسۇلاتلىرى،دېھقانلار ئۆزلىرى ئەتكەن يەرلىك مېزىلىك تائاملىرى، قوغۇن موزىيى، تۆت پەسىلدەقوغۇن تېرىپ يىگىلى بولىدىغان غەربىي-شىمال بويىچە ئەڭ چوڭ زامانىۋىي يۇقىرىتېخنىكىلىق قوغۇن ئۆستۈرۈش پارنىكى قاتارلىقلارنى قانغۇدەك تاماشا قىلىپ، ھوزۇرىنى سۈرىۋاتقاندا بىر يايما ئورنى نۇرغۇنلىغان ئاممىنىڭ دىققىتىنى تارتتى، ئۇ بولسىمۇشىنجاڭ «ئاۋام» ئادۋۇكاتخانىسىنىڭ تۆت نەپەر ئادۋۇكاتىدىن تەشكىللەنگە شىنجاڭ‹ئاۋام› ئادۋۇكاتخانىسىنىڭ ئائىلە، جەمئىيەت، مۇقىملىق ۋە قانۇن باش تېما قىلىنغان دېھقانلارغا قانۇندىن ھەقسىز مەسلىھەت بېرىش، مۇلازىمەت قىلىش ئورنى ئىدى.ئۇلاربىر تەرەپتىن بايرام شادلىقىدىن ھوزۇرلانسا، بىر تەرەپتىن دېھقانلارغا قانۇن ۋەيەرلىك نىزاملارنىڭ يېزىلاردىكى دېھقانلارغا مۇناسىۋەتلىك بولغان مۇھىم مەزمۇنلىرى ئاساس قىلىنغان تەشۋىقات ۋاراقچىلىرى، چاتما رەسىملەر ئارقىلىق چۈشەندۈرۈلگەن قانۇنغا رىئايە قىلىپ، ئىجتىمائىي مۇقىملىقنى ساقلاش ھەققىدىكى كىچىك قوللانماقاتارلىق ماتېرىياللارنى تارقىتىپ، ھەم چۈشەندۈرۈپ، بىرقىسىم كىشىلەرنىڭئىدىيىسىدە ئېنىق بولمىغان بىر قىسىم ئائىلىۋىي زىدىيەتلەردىن ۋە ئىجتىمائىيمەسىللەردىن تولۇق ھەم چۈشىنىشلىك مەسلىھەتلەرنى بېرىپ، بايراملىق پائالىيەت مەزمۇنىغا يېڭى بىر مەزمۇن قوشتى.  بۇنىڭدىن خەۋەر تاپقان قومۇل ۋىلايەتلىك پارتكومنىڭ شۇجىسى لىيۇ جىيەن، خەلق قۇرۇلتىيى ۋىلايەتلىك خىزمەت كومىتېتىنىڭ مۇدىرى مۇھەممەت نامان، ۋىلايەتلىك پارتكومنىڭ ئەزاسى، شەھەرلىك پارتكومنىڭ شۇجىسى خۇاڭ جىيەنكۈن، شەھەر باشلىقى مۇھەممەت نىيازقاتارلىق رەھبەرلەر كېلىپ، «ئاۋام» ئادۋۇكاتلىرى بىلەن قىزغىن قول ئېلىشىپ كۆرۈشتى ۋە ئۇلارنىڭ بۇ قېتىم ئۈرۈمچىدىن ئاتايىتەن مۇشۇ پائالىيەتكە كېلىپ، مۇشۇنداق بىرچوڭ ئاممىۋىي يىغىلىشتىن پايدىلىنىپ، شەھىرىمىزدىكى ھەر مىللەت ئاممىسىغا قانۇندىن مەسلىھەت بېرىپ، كۆپلىگەن تەشۋىقات ماتېرىياللىرىنى تارقىتىپ، تولىمۇ ئەھمىيەتلىك ئىشلارنى قىلغانلىقىدىن مەمنۇن بولغانلىقىنى بىلدۈرۈشتى.
     بىر قىسىم دېھقانلار ئادۋۇكاتلار بىلەن قىزغىن كۆرۈشۈتى ۋە  ئۇلارغا چاخچاق قىلىپ،ــــ « ھازىر پارتىيىمىزنىڭ سىياسىتى بەك ياخشى بولۇپ كېتىۋاتىدۇ، بولمىسا بىزدېھقانلار ئادۋۇكات دېگەننى چۈشۈمىزدىمۇ كۆرەلمەيمىز، بىرەر ئىشقا يولۇقۇپ قېلىپ ئادۋۇكات ئىزدىمەكچى بولساقمۇ نەدىن، كىمنى ئىزدىشىمىزنى بىلمەيمىز، مانا بۈگۈن ئادۋۇكاتنىڭ نوچىلىرى ئۆزى ئالدىمىزغا كەپتۇ-ئەمەسمۇ؟» دەپ ئەتىراپتىكىلەرنى كۈلدۈرىۋەتتى...

پائالىيەت جەريانى ئەل راھى، ئەل مەنپەئەتى، ئەل كۆڭلى بويىچە ئورۇنلاشتۇرۇلۇپ،خەلق ئاممىسىنىڭ ئەنئەنىۋىي بايرام ئۆتكۈزۈش ئارزۇسى تولۇق نامايەن قىلدۇرۇلدى.شۇڭا بۈگۈنكى بايرام شادلىقىغا جۆر بولۇپ، تىلنى يارغۇدەك «قەنت، شېكەر، ناۋات»قوغۇنلاردىن قانغۇدەك يەپ، قوغۇننىڭ تاتلىقى، دىلنىڭ شادلىقى ئۈچۈن تەرەپ-تۇشتىنكەلگەن يەرلىك ھەر مىللەت ئاممىسى، دېھقانلار، ئوقۇغۇچىلار ۋە ئىچكىرى ئۆلكىلەردىن كەلگەن ساياھەتچىلەر بولۇپ 30 مىڭدىن ئاشتى.
---
خوش، پېقىرنىڭمۇ بۇ قېتىملىق قوغۇن بايرىمىدىكى يېڭىلىقلارنى ئوقۇرمەنلەر بىلەن ئورتاقلاشقۇم كېلىۋاتىدۇ. ئەل رايى زادى نىمىگە مايىل بولىشى كېرەك؟ ئەلنىڭ كۆڭلى نىمىنى خالايدۇ؟! قومۇلدا مۇسۇلمانلارنىڭ نوپۇسى قانچە؟! پۈتۈن دۇنيا مۇسۇلمانلىرى روزا تۇتىۋاتقان ئۇلۇغ روزا ئايلىرىدا بۇ قېتىملىق پائالىيەتنىڭ ئۆتكۈزىلىشى قانداقمۇ ئەل رايى قۇرۇلۇشى بولالىسۇن؟! ئاپئاق ساقال بوۋىلار، پەرىشتە سۈپەت مومىلار « شېرىن قوغۇندىن قانغۇچە يىدىمۇ؟! ياكى دەردى ئىچىدىمۇ؟»
بۇ قېتىمقى پائالىيەتتە تىل-يېزىققا ئىنتايىن ئەھمىيەت بېرىلىپ ئۇيغۇرچە، خەنزۇچە يېزىق ھەر ھالدا ئۆلچەملىك ئىشلىتىلىپتۇ، بىراق ياغاچ لەپمىنىڭ (كارىدور) كىرىش، چىقىش ئېغىزىدىكى ئاساسلىق لوزۇنكا يەنىلا بىر خىل يېزىقتا يېزىلغان.
قوغۇنلارنىڭ ئەسلى ئۇيغۇرچە ئاتىلىشى بار تۇرۇپ بەزىلىرى خەنزۇ ئاتتا يېزىلىپ قالغان.
مەيدان كىرىش ئېغىزىدا چاڭچىلىكەش مۇزىكىسى چېلىنغان، ئۇيغۇر نوپۇسى 100% نى ئىگىلىگەن كەنت ئەسلى يەرلىك ئۇيغۇر ئەنئەنىسىنى جەۋلان قىلدۇرۇشى كېرەك، بۇ يەردە چاڭچىلىكەشنىڭ لازىمى يوق، مىللىي، يەرلىك ساپ مەدەنىيەتنى ساياھەتچىلەرمۇ كۆرۈشنى خالايدۇ، ھەرگىزمۇ ئىچكىرىسىدىكى قۇرۇلۇش ۋە سەنئەتنى شىنجاڭغا چىقسىمۇ كۆرۈپ يۈرۈشنى خالىمايدۇ، ئۇنداق بولمىسا شىنجاڭ ئۇيغۇر ئاپتونوم رايونىغا خاس مەدەنىيەتنىڭ يەنە  نىمە خاسلىقى بولسۇن؟!
ئېلان، تەشۋىقات ۋەرەقلىرىنى تارقىتىپ يۈرگەن پارتىزانلار ياخشى باشقۇرۇلمىغان، قارتال كەنتى ئەخلەت دېڭىزىغا ئايلىنىپ كەتكەن.
تاماق، يىمەك-ئىچمەك تازلىق ئۆتكىلى چىڭ تۇتۇلمىغان...

ئەڭ ھالقىلىق تەرپى: مەركەز ۋە ئاپتونوم رايون مىللەتلەرنىڭ دىننىي ئېتىقادىغا، دىنىي بايراملىرىغا ھۆرمەت قىلىدۇ. ئاساسى قانۇندىمۇ بۇ بەلگىلىمىلەر بار، لېكىن، روزا-رەمزاندا بۇنداق يىمەك-ئىچمەك بايرىمنى ئۆتكۈزۈشكە بىرەرمۇ ۋەتەنپەرۋەر دىنىي زات ۋە ياكى ئۇيغۇر رەھبەرلەر باشقىچە پىكىر تاشلىيالمىغانلىقىغا ئەقلىم ھەيران! قوغۇن ئىلگىرىكىلەردىكىدەك 8-ئايغىچە تۇرالمىسىمۇ مۇشۇ يىللاردا ئورتاق تەرەققىي قىلىپ، مەركەزنىڭ توغرا رەھبەرلىكىدە جاپاغا چىداپ ئالغا ئىلگىرلەپ توڭ چىللىدىمۇ قوغۇن، تاۋۇز يەيدىغان تەرەققىياتلارغا ئېرىشىپ كەتتۇق، شۇنىڭ ئۈچۈن «قوغۇن بايرىمى»نى كەينىگە 10 نەچچە كۈن سۈرگەن بولسا تېخىمۇ بەلەن بولاتتى. ئۇچاغدا چىلانمۇ پىشىدۇ، ئاقتاش، باركۆللەرنىڭ تازا كۆگەرگەن ئىسىل پەسلى كېلىدۇ، ساياھەتچىلىكمۇ ئۆز رايونىنىڭ ئەمەلىيتىنى چىقىش قىلىپ تەرەققىي قىلدۇرۇلمىسا ئەلنىڭ رايىدىن، قوللىشىدىن مەھرۇم قالىدۇ!...
