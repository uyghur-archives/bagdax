     بۈگۈن تېلۋۇزۇردىن ئىلى تېلۋىزىيە ئىستانسىسىنىڭ ئۈمۈد غۇنچىلىرى》پرۇگراممىسنى كۆرۈپ قالدىم، بۇ بالىلار پرۇگراممىسدا بۈگۈن ئۆزگىچە بىر ھېكايە سۆزلىنىۋاتاتتى، چوڭ زالدا ئىكران، تۆۋەندە لىق تاماشىبىن، پۇتۈن تاماشىبنلارنىڭ دىققىتى ئىكراندا، باشتا بىر ئۈسمۈر مۇخبىرنىڭ مازار يېزىسدىكى بىر ئائىلنىڭ تۇرمۇش ئەھۋالنى زىيارەت قىلغان كۆرۈنۈشى قويۇلۇشقا باشلاندى، ئاسىيە ئىسمىلىك بۇ 13ياشلىق كىچىك دۇستنىڭ بۇرۇنقى ئۆيى ئۆرۆلۈپ كېتەي دېگەچكە، دوڭمازار بازىرىدىن بىر ئىغىز ئۆينى ئىجارە ئىلىپ ئولتۇرۇۋاتاتتى، ئائىلىدە ئاسيەنىڭ ئاپسى، ئاكىسى، مۇمسى، ئاسىيە تۆت جان بولۇپ پۈتۈن ئىقتساد ئاپسنىڭ يول سۈپۈرۈپ تاپقان 300سوم پۇلدىن كىلەتتى، ئۆي ئېجارىسگە پۇل يوق، ئاسىيەنىڭ ئاپسى مەيلى نېمە بولسا بولسۇن، خەقنىڭ ئالدىدا قىزارغىچە ئۆي ئۆرۆلۈپ كەتسمۇ ئۆز ئۆيىمىزدە تۇرايلى دەپ يېنىپ كەلدى، بۇنىڭدىن خەۋەر تاپقان ئاسىيەنىڭ مۇئەللىمى دەرھال ساۋاقداشلارنى ھەركەتلەندۈرۈپ ئاسىيەلەرنى دۇڭمازار بازىرىغا ياندۇرۇپ كەلدى، ئاسىيە بىلەن ئاكىسى، ئاپسغا ھەمدەم بولۇش ئۈچۈن شەنبە يەكشەنبە كۈنى ئاشخانىلاردا قاچا يۇيدى، لېكىن بۇ ئازغىنە پۇل ھېچنىمىگە دال بولالمىدى،ئاسىيە مەكتەپتىن قايتىۋاتقاندا يولدا ياتقان بىر ئىت ئېتىلىپ كىلىپ ئۇنى چىشلىۋالدى، ۋاكسنىغا پۇل يوق، ئاسىيە ھازىرغىچە ئاخشىمى قىزىپ، چۆچۈپ كىلۋاتىدۇ، بىچارە موماي بىر ئاي بۇرۇن ئاغرىپ قالدى، ئۇنىڭغىمۇ پۇل يوق، ھەركۈنى يۇتقانغا پۈركىنىپ يىتىۋاتىدۇ، رىياسەتچى روشەنگۈل ھەدە ئىكراندىكى باش پىرسۇناژ ئاسىيە بىلەن ئاپىسنى سەھنىگە تەكلىپ قىلدى، بۇ ئاسىيەنىڭ تۇنجى قېتىم غۇلجا شەھىرىگە كىرىشى ئېدى.
       -ئاسىيە، پۇتىڭىزدىكى ئاياغنى كىيىۋاتقىلى قانچىلىك بولدى؟                     
-ئۈچ يىل.
ئۇ ئاشۇ ئاياغ بىلەن ئۈچ قىش، ئۈچ يازنى چىقاردى.
       ئاپسنىڭ 36ياشقا كىرگەنلىكنى دېگىنىدە ھېچكىم ئىشەنمەيلا قالدى، چۈنكى سەھنىدە تۇرغان ئايالنى كۆرگەن ھەر قانداق كىشى، ئۇ ئايالنى كەم دېگەندە 45ياشتىن كەم ئەمەس دىيەلمەيدۇ، مانا بۇ جاپا مۇشەققەتنىڭ ئاشۇ ئانىغا بەرگەن ھەسرەتلىك ئىزلىرى.
  2-ھېكايە باشلاندى. 12ياشلىق قازاق قىزى نادىيە تۇرمۇش كارتىنىسى ئىكراندا پەيدا بولدى، نادىيەنىڭ ئاتا ئانىسى كىچىكىدە ئاجرىشىپ كەتكەن، شۇنىڭدىن باشلاپ ئۆينىڭ تاماق، كىر يۇيۇش، تازلاش ۋەزىپىلىرى نادىيەنىڭ زىممىسگە يۈكلەندى، <<يىقىلغان تام ئۆلۈمتۈكنى بىسپتۇ>> دىگەندەك نادىيەنىڭ دادىسىنىڭ پۇتى سۇنۇپ كېتىپ پۇل بولمىغاچ داۋالىتالماي ئورنىدا يىتىپ قالدى، ئىقتسادى كىرىم يوق بۇ ئائىلگە نىسبەتەن نېمە تاماق ئېتىشمۇ چوڭ مەسلە ھىسابلىناتتى.
  -سىلەر لەغمەنگە گۆش توغرىمامسىلە؟
  -ياق،گۆشكە پۇل يوق، گۆشسز يەپ كۆنۈپ قاپتۇق، ئاپامنىڭ بىزدىن ئايرىلىپ كېتىشى ماڭا بەك ئېغىر كەلدى، ئېنىمغۇ مەكتەپتىن كىلىپلا، ئوينىغلى چىقىپ كېتىدۇ، لېكىن مەن بۈگۈن نېمە ئېتەرمەن، ئەتە تاماق ئەتكۈدەك ئۇن، ماي بارمىدۇ دېگەن غەمدە يۈرىمەن، سىرتقا چىقىپ ئوينغۇدەك ۋاقىتمۇ يوق، ئاغرىق دادامنىڭ خىزمىتىنى قىلمەن، ھازىر ھەممە تاماقنى ئىتەلەيدىغان بولدۇم.
   3-ھېكايە. 12ياشلىق ئوماق قىز شەھنازنىڭ تۇرمۇش كارتىنسمۇ ئىكراندا پەيدا بولىدى، شەھناز كىچىكدىنلا ئاتا ئانسدىن يېتىم قالغان بولۇپ مومىسى بېقىپ چوڭ قىلغان ئىدى، مومسنىڭ سالامەتلىكى ياخشى بولمىغاچقا، ئۆينىڭ بارلىق خىزمەتلىرى شەھنازنىڭ زىممىسىگە يۈكلەندى، ئۇنىڭ پۇتىدىكى يىرتىق ئاياغقىمۇ بىر يىلدىن ئاشتى، تاماقنىڭ تۈرلىرى دېمەيلا قويۇڭ، ئۇ ھازىر ئۇستا ئاشپەز، لېكىن ئۇ بالىلىقتىكى گۈزەل چاغلاردىن مەھرۇم قالدى، دەم ئېلىش كۈنلىرى ئۇ تاغقا چىقىپ ياۋا كۆكتاتلارنى تېرىپ كېلىپ مۇمىسىغا تاماق ئېتىپ بېرىدۇ، ئۇ مەكتەپتىكى ئەڭ ئەلاچى قىز، ئۇ كېچىسى چۈشلىرىدە ئەزەلدىن كۆرۈپ باقمىغان ئاپىسى بىلەن خىيالەن سىردىشىدۇ.
   زالدا ئولتۇرغان، تېلىۋۇزۇر ئالدىدا ئولتۇرغان بارلىق ئىلى خەلقى بۈگۈن يۈرۈكىدىن يىغلىدى.
   رىياسەتچى روشەنگۈل ھەدە ئۈچ قىزچاقنى سەھنىگە تەكلىپ قىلدى، بىر يىگىت ئورنىدىن تۇرۇپ يىغلاپ تۇرۇپ: نادىيەنىڭ تۈنۈگىنى ئەمدى تارىخقا ئايلىندۇ، مانا مەن ئۇنىڭغا پۈتۈن كۈچۈم بىلەن ياردەم قىلمەن،- دېدى.
   كەينىدىنلا بىر ياشانغان ئانا ئورنىدىن تۇرۇپ: شاھنازغا بۈگۈندىن باشلاپ مانا مەن ئاپا بولىمەن، مەن ئۇنىڭ ھازىردىن تارتىپ ئالىي مەكتەپ پۈتتۈرگىچە بولغان بارلىق چىقىمنى مانا ئۆزەم كۆتۈرىمەن.
   نەق مەيداندا ئىئانە توپلاش پائالىيتى ئىلىپ بىرىلىپ40مىڭ سوم توپلاندى، ئەلۋەتتە، ئۇلارنىڭ بانكا كارتىسىنىڭ نۇمۇرىمۇ جەمىيەتكە ئىلان قىلىندى،تاماشىبنلار قاتارىدىكى بىر موئەللىمە مۇنداق دېدى: مانا بۇ شەھەرلىك ئوقۇغۇچىلار بىلەن يېزىدىكى ئوقۇغۇچىلارنىڭ پەرقى، شەھەرلىك ئوقۇغۇچىلارنىڭ قولىدا تېلفۇن، ئاغزىدا يىمسىمۇ بولىدىغان يىمەكلىكلەر، يېزىدىكى ئاشۇ بالىلاردىچۇ، ئۇلارنىڭ ھەتتا تېلۋۇزۇر كۆرۈش پۇرسىتىمۇ بولمايۋاتىدۇ، سەھرالارغا بارايلى، كۆرۈپ كىلەيلى.
   ئەڭ ئاخىردا ئىلى پىداگوگىكا ئېنستىتۇنىڭ داڭلىق ئوقۇتقۇچىسى ئابىدە مۇئەللىم مۇنداق دېدى: بۇلارنىڭ ھېكايىسى بىزنىڭ قەلبىمىزنى ئوخشاش بىر رىئاللىق بىلەن چەكتى، يەنى ئۇلارنىڭ ئاتا ئانىسى يېنىدا ئەمەس، باللىرىنى تىرىك يېتىم قىلغان ئاتا ئانىلارنىڭ باللىرىدىن ئاغرىنىدىغان ھوقۇقى يوق، مانا بۇ ئۈچ قىزىمىز بالىلىقىدىن، خۇشاللىقدىن بالدۇرلا مەھرۇم بولدى، بىز ھېلھەم بۇلارنى ئۇلارغا قايتۇرۇپ كىلەلەيمىز.
   ئىلى تېلۋىزىيە ئىستانسى بالىلار پرۇگراممىسى ئارقىلىق مانا مۇشۇنداق ئالەم مېھىر مۇھەببەتكە تولسۇن دېگەن تېمىسدىكى پائالىيەتنىڭ ئۈسمۈر مۇخىبىرلارئارقىلىق ئىلىپ بېرىشى بالىلارنىڭ قەلبىگە مېھىر ئۇرۇقنى چېچىشىدا ناھايىتى ياخشى ئۈنۈمنى قولغا كەلتۈرگەن، بىز بارلىق ئىلى خەلقى ئىلى تېلۋىزىيە ئىستانسغا چىن كۆڭلىمىزدىن رەھمەت ئېيتىمىز، ھەم سىلەرنىڭ مۇشۇنىڭغا ئوخشاش پرۇگرامما ۋە زىيارەتلەرنى كۆپلەپ ئىلىپ بېرشىڭلانى چىن قەلبىمىزدىن ئۈمۈد قىلىمىز.
