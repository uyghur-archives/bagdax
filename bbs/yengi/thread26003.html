يوللىغان ۋاقتى 2014-7-24 00:29:15 |ھەممە قەۋەتنى كۆرۈش
    _قاچانلا قارىسا پەقەت سەندىنلا ئىش چىقىدۇ،ھەجەپ زىرىكتۇردىڭ ئادەمنى؟  
    _ۋاي بولدى قىلە ئاكا،ئاچامغا گەپ قىلساقمۇ پايدىسى يوق،يەنە ئۆزى بىلگەن ئىشنى قىلىدۇ.
    _ھەي...نۇمۇس ئەمەسمۇ؟بىرسى سىنىڭ زاكات ئالغىنىڭنى ئاڭلاپ قالسا ماڭا ۋە بۇ ئۆيدىكى باشقىلارغا نۇمۇس ئەمەسمۇ؟ نىمىشقا بىزنى ئازراق بولسىمۇ ئويلاپ قويمايسەن،بىزنى ياشىغىلى قويساڭ بولمامدۇ-ھەي...بەزدەكلا ئولتۇرىدىغۇ بۇ،ئاپا گەپ قىلمامسەن بۇنىڭغا؟
   _مەن نىمە دەي ئەمدى،ئاچ قىلىپ ئۆلگەن ئوخشايدۇ بۇ ،بىزنى ئويلاپ قويمايلا باشقىلار بەرگەن زاكاتنى ئاپتۇ،«زاكات» دىگەن بىر ئوت،بۇنى قانداق ئادەم ئىلىپ خەجلەيدۇ،بۇنى بىلمەمەدىغان، بىزگە ئىش تىرىپلا تۇرغان بوپتۇ بالىلىرىم بۇنى يوق دەپ بىلىڭلا مەن ناماقۇل خاپا بولماڭلا ھەممىڭلا روزا ئىغىز مالال بولۇپ قالىسىلەر.
  _ئاپا مىنى يوق دەپ بىلىڭلار دەپ قىرىنداشلىرىمغا تەۋسىيە قىلىۋاتامسەن،مىنى يوق دەپ بىلمەكچىمۇ؟ئۇنداقتا مەن كىم؟مەن بۇلارنىڭ ئاچىسى ئەمەسمۇ؟مەن بالىلارنىڭ چوڭى ئەمەسمۇ؟توۋا....توۋا.....مەن سىلەردىن بىر سۇئال سوراپ باقاي،سىلەر بۇ يۇرۇتنى تاشلاپ باشقا بىر يۇرۇتقا بىرىپ ئۆلىنەلمەيۋاتان بىرسىگە تويلۇق پۇل بىرىپ ئۆلەپ قويدىڭلار،يەنە كىملەرگە......كىملەرگە پۇللارنى بەردىڭلار...مەسچىتلەرگە گىلەملەرنى سىلىپ بەردىڭلار،مۇشۇنداق قىلىش ئارقىلىق كىمنى يۆلىمەكچى،كىمگە ياردەم قىلماقچى؟
_ كىمنى بولاتتى،ئىگە-چاقىسىز،بىچارە،يىتىم بالىلارنى يۆلىمەكچى.
_يىتىم،ئىگە_چاقىسىز،بىچارە.....ئۇنداقتە مىنىڭ ئۆيۇمدىكى ئىككى ئەردىن بولغان ئىككى بالىچۇ؟ بۇلار يىتىم ئەمەسما؟بۇلار بىچارە ئەمەسما؟بۇلار يالغۇز ئانا بىقىۋاتقان بىچارىلەر ئەمەسما،بۇلارنىڭ مەزمۇت مۇكەممەل بىر ئائىلىسى يوق،شۇنداق بولسىمۇ ئىجارە ئۆيلەردە ئولتۇرغۇزماي ھىچ بولمىغاندا ئەزالار تولۇق بولمىسىمۇ بىر ھەقىقى ماكان مۇشۇلار مەنسۇپ بولسۇن دەپ قەرىز ئىلىپ ئۆي ئالدىم،مۇئاشىمنىڭ تەڭدىن تولىسى ھەر ئايلىق بانكا قەرزىگە كىتىدۇ،ئايدا قالغان ئازراقلا پۇلغا خىزمەت ئورنۇمدا باشقىلاردىن چىنىپ قالماسلىق ئۈچۈن،بىر تال بولسىمۇ يىڭى پايپاق كىيىشكە،كىسەل،توي....ۋاكازا باردى_كەلدىلەرگە،توخدەك تاتىلىساممۇ ئۆينىڭ ئۇلگۇرۇپ تۇگەۋاتقان كەم_كوتىلىرىغا،توك،سۇ،گوش،سەي،نان،ئۇن،ماي،قەغەز، تۇرلۇك سوپۇن،تازلىق پۇلى،ئامانلىق پۇلى،باشقۇرۇش ھەققى،تىخى ھەممىدىن مىنى تەمتىرتىدىغان يىل ئاخىرىدىكى پار پۇلى......مانا مۇشۇنداق تۇرلۇك ئىقتىسات قىسىنچىلىقىتا ياشاپ بالا بىقىۋاتقان بىر ئانا بىچارە ئەمەسمۇ،تىخى مۇشۇ يەردە دىيىۋالايئاللاغا مىڭ مەرتىۋە رەخمەت«بالىلىرىم ۋە شۇنداقلا مىنىڭ سالامەتلىگىمىز خىلى ياخشى ئاللا خالىسا ئايدىن-ئايغا ئاغرىپ قالمايمىز،تاسادىپى ئاغرىپ قالساقمۇ ئاغرىققا چىداپ ئائىلىدە دورا ۋە تاماق ئارقىلىق كىسىلىمىزنى ساقايتىمىز دوختۇرغا بارمايمىز چۇنكى دوختۇرخانىنى كۆتۇرگىدەك پۇلىمىز يوق.....'بىراق بالىلىرىمغا كىيىم ئىلىشقا توغرا كەلسە پەقەت پىشىقلىق قىلمايمەن،بۇلار مىنىۋى جەھەتتىن كەمتۇك،ئەمما ماددى جەھەتتىن كەمتۇك بولۇپ باشقىلارنىڭ ئالدىدا بويۇن قىسىپ قىلىشىنى خالىمايمەن،ئۆزۇممۇ ھەر كۇنى ئىشدىن يىنىپ باشقا جايلاردا ئىشلەيمەن،باشقىلار ئۇچۇن كىچىسى ئۇخلىماي ماتىريال كۆرۇپ ماقالە يازىمەن،توختاملارنى ئۇرىمەن،ئائىلە ئوقۇتقۇچىسى بولىمەن،خىزمەت ئورنۇمدىمۇ باشقىلار ئۈچۈن كۆپرەك ئىش قىلىمەن،كۆپرەك خىزمىتىمدىن ئارتۇق ئىسمىنا ئىشلەيمەن،چۇنكى ئازراق بولسىمۇ مەن پۇلغا مۇھتاج، «سىلەر ھەر قىتىم جىنىڭغا ئىچىڭ ئاغرىمامدۇ؟ ھارماي پالاقشىپلا يۇرۇيدىكەنسەن تاپقىنىڭغا شۇكرى قىلىپ تىنىچ ياشىساڭ بولمامادۇ دەيسىلەر،بىراق بىر ئاي ئىشلىسەم قولۇمغا نىمە قالىدۇ مىنىڭ،قايسى تاپقىنىمغا شۇكرى قىلىمەن ئاي ئاخىرىغا كەلگەندە قانچىلىك ئىچىنىشلىق كۇنلەرگە قالىمەن بىلەمسىلەر؟ ياق سىلەر بىلمەيسىلەر،مەن ھەتتا نامۇ ئالاممايمەن بىلەمسىلەر بۇنى....ئادەتتە چوڭ ئۆيدە  ھەق ئارىيمەن دەپ كۆپ ئىشقا ئارلىشىمەن ،ئارتۇق ئىشقا ئارلىشىدۇ ۋات-ۋات دەپ بىلىسىلەر،كۆپ چارچىغاچقا ئۆزۇم بەك سەپرا كۆپ ھاللاردا چىچىلىمەن،مىنى بىشەم ئۇرۇشقاق دەپ بىلىسىلەر،سىرىتتا ئازراق پۇل ئۈچۈن قاتلاممۇ-قاتلام ئىشلەپ ھارغاچقا ئانامنىڭ ئۆيىگە چىقسام بىر ئاز بولساممۇ يىتىۋالىمەن،مۇشۇ يەردە ماڭا ھاردۇق يىتىدۇ، بىراق سىلەر بۇ بالىنىڭ چوڭى بولغان بىلەن ھورۇن،ئانىنىڭ ئىشىنىمۇ قىلمامدۇ،ھەممە ئىشنى بىز قىلىمىز بۇ بولسا ياتىدۇ دەپ بىلىسىلەر،ئىچىمدىكى دەرتلەرنى دەي دەپ گىپىمنىڭ بىشىنى باشلاپ كۆزلىرىمنى ياشلاپ بولغىچە ھەممە ئادەمنىڭ تۇرمۇشدا خاپىلىقى بولىدۇ بىرسىمىز ئەرنىڭ،بىرسىمىز بالىنىڭ يەنە بىرلىرىمىز بار-يوقنىڭ خاپىلىقىنى تارتىمىز دەرت كۆتۇرۇپ كىلىپ ئاپامنىڭ بىشىنى ئاغرىتما بىزگە پەقەت مۇشۇ بىرلا ئانا قالدى،ئانامنىڭ خاپىلىق كۆتۇرگۇچىلىكى يوق دەپ ئىچىمنى بوشۇتۇشقىمۇ يول قويمايسىلەر،ئاخىرى دەيدىغان يەر تاپالمىغاندا دادامنىڭ بىشىغا چىقىپ جىمىنى دەپ تۆكۇلۇپ كىرسەم ئەقلىڭ يوق سىنىڭ تۇپراق بىشىغا چىقىپ يىغلاپ يۇرگىچە جەينىمازدا ئولتۇرۇپ ئاللاغا يىغلىساڭ بولمامدۇ دەپ مىنىڭ قىلغان ئىشلىرىمنى لەت قىلىسىلەر،مىنىڭ ئىچىمنى بوشۇتۇپ ھۆڭرەپ يىغلىغۇدەك،پاناھلانغىدەك چوڭ ئەمەس كىچىككىنە ئىللىق بىر بۇرجىگىممۇ يوق،شۇنىڭ بىلەن سىلەرگە بىر خىل ئۆچلۇك نەزىرى بىلەن قاراپ كەلدىم،تىخى «نونى» ساغلاملىق دورىسى چىقىپتۇ ئۆتكەن ئاي مەن ئالدىم ئەمدى ئۆزەڭلار سوراپ بىرىڭلار ئەپچىقارسىلە،ئاپامنىڭ دورىسى تۈگىدى دىسەڭلار يەنە بىرىڭلار قوپۇپ «سىئەمدى»ھەممىدىن ياخشىكەن دىسەڭلار،يەنە بىرىڭلار«پىپدىت»ھەممىدىن ياخشىكەن شۇنى ئالسۇن دەپ مۇلاھىزە قىلىشلىرىڭلار مىنى بۇ ئائىلىدە پاسسىپ ئورۇنغا چۇشۇرۇپ قويىدۇ،بۇچاغلاردا ئۆزەمنى قويغىدەك ئورۇنمۇ تاپالمايمەن،بەزىدە ئولتۇرۇپلا يىغلاپ كىتىمەن،ھەي..... بۇ نىمى بولدى بۇ قاچاندا قارىسا ساراڭدەك يىغلايدىكەن دىيىشىپ كىتىسىلەر،ھەتتا بەزىدە مەندىن بىزارمۇ بولىسىلەر........
        مىنىڭمۇ ئانامغا مۇنداق ساغلانلىق دورىلىرىنى ئىلىپ بەرگىم يوقمۇ،بىراق ئۆزۇمگە نانمۇ ئالاممايۋاتسام.....يەنە دىگىم كەلدى مەنمۇ 40 ياشقا ئاز قالدىم،مەنمۇ كۇتۇنۇشكە مۇھتاج،لىكىن  مۇقۇم،تىنىچ ئۆتىدىغان بىر كۇنۇم يوق،تۇرمۇش تىخچە ياقامدىن تۇتىۋىلىپ سۆرىگىنىچە مىنى قويۇپ بەرمىدى........بىر ئىقتىسادى كاپالەتكە ئىگە بولاي دەپ بىر ئەھلى مۇسۇلمان بىلەن توي قىلدىم ئەپسۇس تويدىن كىيىن شۇنى چۇشەندىمكى بۇ كىشىمنىڭ يات يەرگە مەبلەغ سالغىسى كەلمىدى،ئىقتىسادى جەھەتتە كاپالەتكە ئىگە بولاممايلا قالماي ئەكسىچە بالىلىرىمنى ئۈگەي دادىنىڭ ئالدىدا پاتۇراممىدىم،بالىلىرىمنىڭ ھىچقانداق قىلىقى بۇ دادىغا تىتىمىدى،كۆزىگە سىغمىدى،بالىرىمدىن بىزار بولدى،خۇيى تۇتتى،تولا ۋاقىراپ چىچىلىپ ئائىلەمنى دوزاققا،ئوچۇق گۆرگە ئايلاندۇردى،تىخىمۇ بىچارە،تىخمۇ يالغۇز قالدىم.......بۇ كۇنلىرىمنى كۆرگەن ياتلار ماڭا ئىچ ئاغرىتىپتۇ،مەن شۇلارنىڭ قولىدىن «زاكات»ئالغاندا ۋىژدان جەھەتتە مەن ئازاپلانمامدىم،ئازاپلاندىم،مەن سىلەرنى ئويلاپ نىمە قىلىمەن مەن يالغان گەپ قىلمىدىم مەن سىلەرنىڭ نىمە بولۇشىڭلار،قانداق ئويلاپ قىلىشىڭلار،ئازاپلىنىپ كىتىشىڭلارنى ئەزەلدىن ئويلۇمىدىم بىراق ئۆزەمنىڭ بۇ كۇنلىرىگە ئىچىندىم،شۇ كۇنى كۇندۇزدىن تارتىپ تاكى ئۇخلاپ قالغاچە ساراڭدەك يىغلاپ يۇردۇم،ئەمىليەتتە ئەڭ ئازاپلىنىدىغىنى ۋە ئازاپلانغىنى پەقەت مەن......بالىلىرىم ئۈچۈن يىنىمدىكى ئىككى نامەرت ئەرلەر تاشلاپ كەتكەن ئىككى پەرزەنتىم ئۈچۈن،قارا يىتىملىرىم ئۈچۈن مەن «زاكات»ئالدىم.سىلەر مىنى چۇشۇنەمتىڭلار،سىلەرنىڭ توققۇزىڭلار تەل،قولۇڭلارنى نەگە ئۇزاتساڭلار شۇ يەرگە يىتىدۇ،بىراڭ مىنىڭ ئىھتىياجىم بىر تال يۇز يۇەن پۇلغا چۇشۇدۇ،مىنىڭ شۇنىڭغىمۇ قۇربۇم ئاران يىتىدۇ ۋە بەزىدە يەتمەيدۇ....سىلەر مىنى بىلەمتىڭلار،سىلەر ماڭا ئىچ ئاغرىتامتىڭلار،مىنىڭ يىتىملىرىمنى سىلەرنىڭ كۆزۇڭلار كۆرەمتى ياق ياق........ياق........مەندىن نۇمۇس قىلماڭلار قىرىنداشلىرىم،مەندىن تىنىپمۇ تىنىپ بولاممايسىلەر،بىز ھەممىمىز مەرھۇم دادام ۋە ھازىرقى جىنىڭلاردىن ئىزىز كۆرۇپ قوغداۋاتقان،ئاسراۋاتقان ۋاپادارلىق يەتكۇزۇۋاتقان ئانىمىزنىڭ پەرزەنىتلىرى بىز......مۇشۇ يەردە سىلەرگە دەۋالاي مىنىڭ ئورنۇمدا ئانىمىزغا يەتكۇزىۋاتقان ۋاپادارلىقىڭلارغا كۆپ تەشەككۇر،سىلەرگە ئىگىلىپ تازىم،سىلەرگە كۆپ رەخمەت.......مەندىن نۇمۇس قىلماڭلار قىرىنداشلىرىم......مەنمۇ ھاردىم چارچىدىم ماڭىمۇ ھاردۇق يەتتى..........
