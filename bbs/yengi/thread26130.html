نىلقا ناھىيىسى تەڭرىتېغىنىڭ قۇچىقىغا جايلاشقان بولۇپ ،ئۇنىڭ شىمالىدا پولۇكنۇ تېغى بار ، شەرقىدە ئاۋرال تېغى ،ئوتتۇرسىدا قاش دەرياسى شەرىقتىن غەرىبكە قاراپ ئېقىپ ئىلى دەرياسىغا قوشۇلىدۇ ، شىمال تەرەپتە ئاچال تاغ ئېغىزىدىن ئۆتۈپ جىڭ ناھىيىسىگە بارغىلى بولىدۇ ، شەرق تەرىپى كۈنەسكە تۇتىشىدۇ ، نارات يايلىقىدىن بايىنبۇلاق (يۇلتۇز دەپمۇ ئاتىلىدۇ ) ۋە خېجىڭ قاتارلىق جايلارغىمۇ بارغىلى بولىدۇ . يەر تۈزۈلۈشى خەتەرلىك بولۇپ ،ساقلىماق ئاسان ،ھۇجۇم قىلىپ ئېلىش تەس .بۇ يەرنىڭ ئاساسلىق ئاھالىسى چارۋىچىلىق بىلەن شۇغۇللىنىدىغان قازاقلار ، ئۇنڭدىن قالسا دېھقانچىلىق ، قول -ھۈنۈرۋەنچىلىك ۋە سودىگەرچىلىك بىلەن شۇغۇللىنىدىغان ئۇيغۇر ،خەنزۇ ،ئۆزبېك ،،تاتار ،رۇس قاتارلىق مىللەتلەر بولۇپ ، 1944-يىلىدىكى نوپۇسى 45 مىڭدىن كۆپرەك ئىدى.
شېڭ شىسەي ۋە گومىنداڭ ھۆكۈمىتىنىڭ زالىم ھۆكۈمرانلىقى ۋە دەھشەتلىك مىللىي زۇلۇمى ، شۇنىڭدەك ئاز سانلىق مىللەتلەر ئىچىدىكى فېئودال سىنىپلارنىڭ ۋەھشىيانە ئىېكىسپىلاتاتسىيىسى ۋە زۇلۇمى تۈپەيلىدىن ، ھەر مىللەت خەلقى نامراتلىق ۋە قالاقلىق ئىچىدە ئېچىنىشلىق تۇرمۇش كەچۈرۋاتاتتى . 1943-يىلى 3- ئايدا ،شېڭ شىسەي ھۆكۈمىتى ھەرمىللەت خەلقىنى 10 مىڭ توياق ھەربىي ئات ئىئانە قىلىشقا زورلىدى . ئات ئىئانە قىلىشقا قۇربى يەتمىگەنلەر بازار نەرخىدىن بىر ھەسسە كۆپ نەق پۇل تاپشۇرۇشقا توغرا كېلەتتى ، ئۇنداق قىلمىسا ،ھۆكۈمەت تەرىپىدىن تۈرمىگە تاشلىنىپ ،تاكى نەق پۇلنى تولۇق تاپشۇرغاندا ،ئاندىن قويۇپ بېرىلەتتى . بۇ سېلىق ئاساسلىقى ئىلى ،تارباغاتاي ،ئالتاي ۋىلايەتلىرىدىكى دېھقان -چارۋىچىلارنىڭ زىممىسىگە يۈكلەنگەنلىكتىن ، بۇ ئۈچ ۋىلايەتتىكى كەڭ دېھقان - چارۋىچىلارنىڭ تېخىمۇ كۈچلۈك نارازىلىقىنى قوزغىدى .
1944- يىلى 4 -ئايدا ، غۇلجا ئازادلىق تەشكىلاتى قۇرۇلغاندىن كېيىن ،ھەر مىللەت خەلقى ئارىسىدا تەشۋىق ۋە سەپەرۋەرلىك ئېلىپ بېرىپ ،گومىنداڭنىڭ ئەكسىيەتچىل ھۆكۈمرانلىقىغا قارشى قوراللىق قوزغىلاڭ كۆتۈرۈشكە تەييارلىق كۆردى . 7- ئاينىڭ ئاخىرلىرى ئوۋ مىلتىقىنى تاپشۇرۇشنى رەت قىلغانلىقى ئۈچۈن ،نىلقا ناھىيىلىك ساقچى ئىدارىسى تەرىپىدىن قامالغان ئاكا -ئۇكا قازاق چارۋىچى ئەكبەر بىلەن سېيىت تۈرمىدىن چىققاندىن كېيىن ، ئۆز-ئارا كېڭىشىپ ،بەزى كىشىلەر بىلەن ئالاقە باغلاپ ،شېڭ شىسەي ۋە گومىنداڭنىڭ ئەكسىيەتچىل ھۆكۈمرانلىقىغا قوراللىق قارشى تۇرۇش نىيىتىگە كەلدى .شۇنىڭ ئۈچۈن ، ئۇلار 8- ئاينىڭ باشلىرىدا قورال -ياراغ تېپىپ كېلىش ئۈچۈن سوۋېت ئىتتىپاقىغا بېرىپ ،جۇڭگو -سوۋېت چېگرىسىدە سابىق نىلقا ناھىيىلىك يەرلىك مەسھۇلاتلار شىركىتىنىڭ مۇئاۋىن دىرېكتورى فاتىخ موسلىموپ بىلەن ئۇچراشتى . موسلىموپ قاتارلىقلار 1943-يىلى يازدىن باشلاپ ، سوۋېت ئىتتىپاقىنىڭ ئالمۇتا قاتارلىق جايلىرىدا سوۋېت ئىتتىپاقى تەرىپىدىن ھەربىي ۋە سىياسى تەلىم -تەربىيە ئالغان ،تەلىم -تەربىيە ئاياغلاشقاندىن كېيىن ،1944- يىلى 8- ئايدا قوراللىق قوزغىلاڭ كۆتۈرۈش ئۈچۈن نىلقا ناھىيىسىگە ئەۋەتىلگەن. ئەكبەر قاتارلىقلار فاتىخ موسلىموپ بىلەن بىرلەشكەندىن كېيىن ،قورال- ياراغ ،ئوق-دورىلارنى ئېلىپ قورغاس ناھىيىسىنىڭ شىمالىي چېگرىسىغا جايلاشقان باش كۆنچى ئىگىز تاغ يايلىقى ئارقىلىق چېگرىدىن يوشۇرۇنچە ئۆتۈپ ،داشىگۇنىڭ شىمالىدىكى تاغلاردىن ئېشىپ غۇلجا ناھىيىسىدىكى جېلىليۈزىگە كېلىپ ،يەنە ئۇ يەردىن كۆكساي ئارقىلىق نىلقا ناھىيىسىنىڭ ئۇلاستاي تاغلىق رايۇنىغا كېلىپ پارتىزانلار ئەترىتىنى تەشكىللىدى . پارتىزانلار ئەترىتى دەسلەپتە فاتىخ موسلىموپ ئەكەلگەن سوۋېت ئىتتىپاقىدا بىللە تەربىيلەنگەن رەپىق بايچۇرىن ،خەمىت موسلىموپ ،قۇربان بۇرھان ،ئوسمان ئىبراھىم ،نۇرۇم ئوبۇل ،ھوشۇر مامۇت ۋە ئەكبەر ، سېيىت ،غېنى ئابدۇللا قاتارلىقلاردىن تەركىب تاپقانىدى ،كېيىن پارتىزانلار ئەترىتى جالىنغۇل ،جارتۇغان ، كۆكساي قاتارلىق جايلارغا ئادەم ئەۋەتىپ ، تەشۋىقات ئېلىپ باردى ، شېڭ شىسەي ۋە گومىنداڭ ھۆكۈمىتىنىڭ ھەر مىللەت خەلق ئاممىسىغا زۇلۇم سالغان ،ئىېكىسپىلاتاتسيە قىلغان ۋە زىيانكەشلىك قىلغانلىقىدەك جىنايى قىلمىشلىرىنى پاش قىلىش ئارقىلىق خەلقنى قوزغىلىپ قوراللىق كۈرەش ئېلىپ بېرىشقا سەپەرۋەر قىلدى ، شۇ جايدىكى ئابرويلۇق زاتلار ۋە قەبىلە ئاقساقاللىرىنى ، نامرات چارۋىچىلارنى شۇنىڭدەك گومىنداڭ ھۆكۈمىتى تەرىپىدىن تۇتقۇن قىلىنماقچى بولغانلارنى پارتىزانلار ئەترىتىگە قوبۇل قىلدى . بۇنىڭ بىلەن پارتىزانلار ئەترىتى ناھايىتى تېز تەرەققى قىلدى ھەمدە ئادەم ئەۋەتىپ نىلقا ناھىيىسىدىن مازارغا تۇتۇشىدىغان توك سىمىنى ئۈزۈۋەتتى ،كۆۋرۈك ،يوللارنى بۇزۇۋەتتى ، گومىنداڭنىڭ ھاربىي ،ساقچىلىرىغا ئۇشتۇمتۇت زەربە بەردى .
8-ئاينىڭ 17 -كۈنى فاتىخ قاتارلىق كىشىلەر ئۇلاستايدا ئاممىنى قوزغاپ قوزغىلاڭ كۆتۈردى . نىلقا ناھىيىلىك ساقچى ئىدارىسىنىڭ باشلىقى فېڭ ۋېيجۈن 50نەچچە ساقچىنى باشلاپ كېلىپ توسۇماقچى بولدى ، قايتىش يولىدا ئۇلاستاي پارتىزانلىرىنىڭ يوشۇرۇنۇپ زەربە بېرىشىگە ئۇچرىدى ، ئەتراپتىكى ئامما قوللىرىغا تۈرلۈك قوراللارنى ئېلىپ جەڭگە قاتناشتى ، خېلى كۆپ ساقچىنى ئۆلتۈردى ۋە يارىدار قىلدى . بەزى قورال-ياراغلارنى ئولجا ئالدى .مانا بۇ ـــــــــــــ نىلقا قوزغىلىڭىنىڭ باشلىنىشى ،شۇنىڭدىن ئېتبارەن ئۇلاستاي پارتىزانلىرىنىڭ ھەيۋىسى ئېشىپ كەتتى ، كېيىن يەنە گومىنداڭ ئارمىيىسى ۋە ساقچىلىرىغا بىر نەچچە قېتىم ئۇشتۇمتۇت زەربە بەردى ، گومىنداڭ ئارمىيىسى ۋە ساقچىلىرىنىڭ ھۇجۇمىنى چېكىندۈردى .
ئۇلاستاي پارتىزانلىرىنىڭ ھەركىتى گومىنداڭ شىنجاڭ ئۆلكىلىك ھۆكۈمىتىنىڭ دىققىتىنى قوزغىدى . ئۆلكىلىك ھۆكۈمەت ئۆلكىلىك ساقچى ئىشلار باشقارمىسىنىڭ مۇئاۋىن باشلىقى ليۇ بىڭدېنى ئەھۋال ئىگەللەش ،تاقابىل تۇرۇش تەدبىرىنى تەتقىق قىلىش ئۈچۈن غۇلجىغا ئەۋەتتى . ليۇ بىڭدې غۇلجىلىق خۇيزۇ بولۇپ ، ئۇ غۇلجىنىڭ ئادەملىرى ،جۇغراپىيىلىك ئەھۋالى بىلەن تونۇشلۇق بولۇش ھەم ئاز سانلىق مىللەتلەرنىڭ تىلىنى بىلىشتەك قولايلىق شارائىتىدىن پايدىلىنىپ ،ئىلىدىكى ئەھۋاللارنى ئىگەلىگەندىن كېيىن ،بىر تەرەپتىن ،ئۆلكىلىك ھۆكۈمەتكە ئىلى ۋىلايىتىنىڭ ۋالىيسىنى بالدۇرراق تەيىنلەش (ئەينى ۋاقىتتا ۋالىي چياۋ گېن شېڭ شىسەي تەرىپىدىن ئۈرۈمچىدىكى تۈرمىگە تاشلىغانىدى )،قوزغىلاڭنى باستۇرۇش ئۈچۈن ئىلىغا قوشۇن يۆتكەش ،ئات ئىئانە قىلىشنى توختىىتىش ، يەرلىك ئاتاقلىق زاتلارنى تەسەللى بېرىشكە ئەۋەتىش ھەققىدە تەكلىپ بەرسە ، يەنە بىر تەرەپتىن ، ساقچى ،ئىشپىيونلارنى ئورۇنلاشتۇرۇپ ،ئامانلىق ساقلاشنى كۈچەيتتى ، بىر مۇنچە ئىنقىلابىي ئامما ۋە تەرەققىيپەرۋەر زاتلارنى قولغا ئالدى ، بۇنىڭ بىلەن غۇلجا ئازادلىق تەشكىلاتنىڭ پائالىيەتلىرىگە مۇئەييەن قىيىنچىلىقلارنى تۇغدۇرغانىدى . لېكىن ئەينى چاغ شېڭ شىسەي چوڭچىڭغا بېرىپ مىللىي ھۆكۈمەتنىڭ دېھقانچىلىق -ئورمانچىلىق مىنىستىرى بولۇشقا تەييارلنىىۋاتقان، ئۆلكىلىك ھۆكۈمەتنىڭ يېڭىدىن تەيىنلەنگەن رەئىسى ۋۇ جۇڭشىن تېخى ۋەزىپىگە ئولتۇرمىغان پەيت بولغاچقا ،ئۇلاستاي پارتىزانلىرىنىڭ ھەركىتى ۋە ليۇ بىڭدېنىڭ پىلانى گومىنداڭ شىنجاڭ دائىرلىرىنىڭ يېتەرلىك ئەھمىيەت بېرىشىنى قوزغۇيالمىدى . پەقەت 9-ئاينىڭ 2- كۈنى ،زاپاس 7 -شىسىنىڭ مۇئاۋىن شىجاڭى دۇدېفۇنى ئايرۇپىلان بىلەن غۇلجىغا كېلىپ ئىلى ۋىلايىتىدە تۇرۇشلۇق بارلىق قىسىملار ۋە ئامانلىق ساقلاش تۈەن -دۈيلىرىگە بىر تۇتاش قوماندانلىق قىلىش ۋە قوزغىلاڭچى ئاممىنى باستۇرۇش ئۈچۈن ئەۋەتتى . ئەينى چاغدا ، گومىنداڭنىڭ ئىلىدا ئاران 2500دىن كۆپرەك ئەسكىرىي كۈچى بار ئىدى ،ئۇنىڭ ئۈستىگە بۇلار ئاساسلىقى شېڭ شىسەيدىن قالغان كونا ئۆلكە ئارمىيىسى بولۇپ ،جەڭگىۋارلىغى ئاجىز ئىدى، بۇ نىلغا قوزغىلىڭىنىڭ تەرەققي قىلىشى ۋە غۇلجا قوزغىلىڭىنىڭ پارتىلىشى ئۈچۈن پايدىلىق پۇرسەت يارىتىپ بەردى .
9- ئاينىڭ ئوتتۇرلىرىدا ، ئىلى ۋىلايەتلىك ساقچى ئىدارىسىنىڭ باشلىقى ،مۇۋەققەت ۋالىيسى قوشۇمچە ئامانلىق ساقلاش قىسىملىرىنىڭ باش قوماندانى گاۋۋېي ۋىلايەتلىك ساقچى ئىدارىسىنىڭ مۇئاۋىن باشلىقى جاۋجۇڭمىڭ قاتارلىقلارنى 128- شىنىڭ 3- يىڭىدىن بىر لىيەن ئەسكەر ۋە بىر قىسىم ساقچىنى باشلاپ قوزغىلاڭنى باستۇرۇش ئۈچۈن ئۇلاستايغا ئەۋەتتى . 9-ئاينىڭ 20-كۈنى ،300نەچچە ھەربىي -ساقچى جاۋجۇڭمىڭنىڭ قوماندانلىقىدا ئۈچ يولغا بۆلۈنۈپ ئۇلاستاي پارتىزانلىرىنى «يوقاتماقچى »بولدى . 21-كۈنى ئۇلاستاينىڭ جەنۇبىدىكى يۇمۇلاق تاغقا ھۇجۇم قىلغان گومىنداڭنىڭ ھەربىي- ساقچىلىرى پارتىزانلار ۋە ئاممىنىڭ قورشاۋىدا قېلىپ ، خېلى كۆپ ئادىمى ئۆلدى ۋە يارىدار بولدى ،9-ئاينىڭ 24-كۈنى مازار جىلغىسىدىن ئاچال ئېغىزىغا كىرىپ ،پارتىزانلارنىڭ شىمالغا يۈرۈش قىلىشىنى توسۇماقچى بولغان گومىنداڭنىڭ يەنە بىر يول ھەربىي- ساقچى قوشۇنىمۇ پارتىزانلارنىڭ ئۇشتۇمتۇت زەربىسىگە ئۇچراپ ،خېلى كۆپ چىقىم تارتتى .ئارقىدىنلا پارتىزانلار مازار جىلغىسىنى ئىشغال قىلىپ ، غۇلجا ناھىيە بازىرىدىن نىلقا ناھىيە بازىرىغا بارىدىغان ئىستىراتېگىيىلىك مۇھىم جاينى كونترول قىلدى ھەمدە نىلقا ناھىيە بازىرىغا ھۇجۇم قىلىشقا تەييارلاندى .
10-ئاينىڭ 5- كۈنى ، پارتىزانلار ئەترىتىنىڭ باش قوماندانى فاتىخ موسلىموپ ئۇلاستايدىكى باش شىتابىدا نىلقا ناھىيە بازىرىغا ھۇجۇم قىلىش بۇيرۇقى ۋە پىلانىنى جاكارلىدى . شۇ كۈنى چۈشتىن كېيىن ، پارتىزانلار ئۈچ يولغا بۆلۈنۈپ ، نىلقا ناھىيە بازىرىغا يۈرۈش قىلدى . 10-ئاينىڭ 6-كۈنى سەھەردە ، ئەكبەرنىڭ باش قوماندانلىقىدا ،سېيىت باشچىلىقىدىكى 1- چوڭ ئەترەت شەرق تەرەپتىن ،غېنى ، ئوسمان باشچىلىقىدىكى 2-چوڭ ئەترەت جەنۇب تەرەپتىن ، بايچۇرىن ،شوتۇپ ئېۋان باشچىلىقىدىكى 3- چوڭ ئەترەت غەرب تەرەپتىن ناھىيە بازىرىغا شىددەتلىك ھۇجۇم قوزغىدى .ناھىيە بازىرىنىڭ ئىچى - سىرتىدىكى ئاز سانلىق مىللەت ئاممىسىمۇ ئۆزلۈكىدىن پارتىزانلارغا ماسلاشتى . كەسكىن جەڭ قىلىش نەتىجىسىدە ، ناھىيە بازىرىنى ساقلاۋاتقان گومىنداڭ ھەربىي - ساقچىلىرىدىن 200نەچچە ئادەم يوقىتىلدى ، 7-كۈنى چۈشتىن بۇرۇن ناھىيە بازىرى ئىشغال قىلىندى ، ئىلى ۋىلايەتلىك ساقچى ئىدارىسىنىڭ مۇئاۋىن باشلىقى جاۋ جۇڭمىڭ قاتارلىقلار ئۆلۈمگە مەھكۇم قىلىندى ،چوبان قەبىلىسىدىن بولغان ئوسمان نىلقا ناھىيىسىنىڭ ۋاقىتلىق مەسئۇلى بولۇپ تەيىنلەندى .
10-ئاينىڭ 9- كۈنى ، ئۇلاستاي پارتىزانلىرى نىلقا ناھىيىسىدە ئىككىنچى قېتىم تەرتىپكە سېلىندى . ئەكبەر بىلەن سېيىت ئاساسلىقى قازاقلاردىن تەركىب تاپقان 1- چوڭ ئەترەتنىڭ باشلىقى ۋە مۇئاۋىن باشلىقى بولدى ؛ غېنى ئاساسلىقى ئۇيغۇرلاردىن تەركىب تاپقان 2- چوڭ ئەترەتنىڭ باشلىقى بولدى ؛شۇتۇپ ئېۋان ئاساسلىقى رۇسلاردىن تەركىب تاپقان 3- چوڭ ئەترەتنىڭ باشلىقلىقىنى ئۈستىگە ئالدى . پارتىزانلارنىڭ ئومۇنىي ئادەم سانى تەخمىنەن 800نەچچە كىشىگە يەتتى . شۇنىڭدىن ئېتىبارەن ،بۇ ئەترەت رەسمىي رەۋىشتە نىلقا پارتىزانلىرى دەپ ئاتالدى . نىلقا پارتىزانلىرى مازار جىلغىسى ،سوپتاي ،قاراسۇ قاتارلىق جايلاردا كۆپ قېتىم گومىنداڭ قوشۇنلىرىنىڭ ھۇجۇمىنى چېكىندۈردى . پارتىزانلارنىڭ كۈچى تېز زورىيىپ ، تەسىرى تېز سۈرئەتتە كۈنەس ،توققۇزتارا ،تېكەس ،غۇلجا قاتارلىق جايلارغىچە كېڭەيدى . غۇلجا ئازادلىق تەشكىلاتىمۇ ھەركىتىنى تېزلەتتى ، غۇلجىدا قوراللىق قوزغىلاڭ كۆتۈرۈشكە تەييارلىق كۆردى .
نىلقا قوزغىلىڭى پارتىلغاندىن كېيىن ، گومىنداڭنىڭ شىنجاڭدىكى ئارمىيە ،ھۆكۈمەت دائىرلىرى قاتتىق ساراسىمىگە چۈشتى . 1944-يىلى 10-ئاينىڭ 4-كۈنى ۋۇجۇڭشىن ئۈرۈمچىگە يېتىپ كېلىپ ، شىنجاڭ ئۆلكىلىك ھۆكۈمەتنىڭ رەئىسلىكىنى قولىغا ئالدى . 10- ئاينىڭ 6-كۈنى ،ھەرقايسى نازارەت ،كوممۇتېتلارنىڭ مەسئۇللىرى يىغىنى چاقىرىپ ،جايلاردىكى خەلق قوزغىلاڭچى كۈچلىرىگە « قورشاپ يوقىتىش بىلەن تەسەللى بېرىشنى تەڭ ئېلىپ بېرىش » فاڭجىنىنى قوللىنىش ،ھەربىي ئىشلار جەھەتتە جايلاردىكى ھۆكۈمەتلەر ۋىلايەت ،ناھىيىلەردە « قوغدىنىش ئەترىتى» قۇرۇپ ،ھەربىي قىسىملار ۋە ساقچىلارنىڭ باستۇرۇش ھەركىتىگە يېقىندىن ماسلىشىشنى بەلگىلىدى. سىياسىي جەھەتتە، شېڭ شىسەي تەرىپىدىن قامالغان ھەر مىللەتتىن بولغان ئابرويلۇق زاتلاردىن 270كىشىنى قويۇپ بېرىش ، كېيىن يەنە نەچچە يۈز ئادەمنى ئارقا -ئارقىدىن قويۇپ بېرىشنى قارار قىلدى . قاماقتا تۇرۇۋاتقان ئاز سانلىق مىللەت ۋەكىللىككە ئىگە زاتلىرى ، مەسىلەن ، بۇرھان شاھىدى ،ئالتايدىن ئەلەن ۋاڭ ، شىبەدىن گۇاڭ لۇ قاتارلىقلارنى ۋۇجۇڭشىن ئىسمىنى ئاتاپ قويۇپ بەردى ، ئۇلار بىلەن شەخسەن ئۆزى دىدار كۆرۈشۈپ ، ئۇلارنىڭ بەزىلىرىنى مۇھىم ۋەزىپىگە قويدى ، بەزىلىرىنى پۇل بېرىپ يۇرتىغا ئاپىرىپ قويدى .
نىلقا ناھىيە بازىرىنى پارتىزانلار ئىشغال قىلغاندىن كېيىن ، گومىنداڭ دائىرلىرى ئىلى ۋىلايىتىنىڭ ۋەزىيەت تەرەققىياتىنىڭ كۇنسايىن كەسكىنلىشىپ كېتىۋاتقانلىقىنى كۆزدە تۇتۇپ ،10-ئاينىڭ 9-كۈنى زاپاس 7- شىنىڭ مۇئاۋىن شىجاڭى دۇدېفۇ غۇلجىدىن ئايرۇپىلان بىلەن ئۈرۈمچىگە كېلىپ ، جۇ شاۋلياڭ ( بۇ چاغدا گومىنداڭ ھۆكۈمىتى 8-ئۇرۇش رايۇنىنىڭ قوماندانى ئىدى) غا ئىلىنىڭ ئەھۋالىنى دوكلات قىلدى ھەمدە ئەسكەر كۆپەيتىپ بېرىشنى تەلەپ قىلدى . 10- ئاينىڭ 12- كۈنى، جۇشاۋلياڭ زاپاس 7-شى 19- تۈەنىنىڭ تۈەنجاڭى پېڭ جۈنيېغا ئەسكەر تارتىپ ئىلىغا ياردەم بېرىش ھەمدە ئىلى ۋىلايىتىدە ( جىڭ ، بورتالا ،ئارشاڭ ناھىيىلىرىدىن باشقا )تۇرۇشلۇق گومىنداڭ قوشۇنى ۋە ئامانلىق ساقلاش قىسىملىرىغا بىر تۇتاش قوماندانلىق قىلىش توغرىسىدا بۇيرۇق چۈشۈردى .پېڭ جۈنيې قوشۇن تارتىپ ، ئاپتوموبىل بىلەن 10-ئاينىڭ 16- كۈنى كەچتە غۇلجىغا يېتىپ كەلدى . 10-ئاينىڭ 18- كۈنى ، جۇشاۋلياڭ پىلانىنى ئۆزگەرتىپ ،زاپاس 7- شىنىڭ سەنمۇجاڭى ساۋ رىلىڭنى ۋاقتىنچە ئىلى ۋىلايىتىنىڭ مۇۋەققەت تىنچىتىش قوماندانلىقىغا بەلگىلەپ ،ئىلىدا تۇرۇشلۇق قوشۇنلار ۋە ئامانلىق ساقلاش قىسىملىرىغا بىر تۇتاش قوماندانلىق قىلىپ ، نىلقا پارتىزانلىرىغا ھەربىي ھۇجۇم قوزغاشقا ئەۋەتتى .
گومىنداڭ ھۆكۈمىتى پۈتۈن شىنجاڭغا ، بولۇپمۇ ئىلى ۋىلايىتىگە سىياسى جەھەتتىن تەسەللى بېرىش ئىشىنى تېزلەتتى . 10-ئاينىڭ 21- كۈنى شىنجاڭ ئۆلكىلىك ھۆكۈمەت ئەۋەتكەن ئۆلكىلىك خەلق ئىشلىرى نازارىتىنىڭ نازىرى دېڭ شياڭخەي باشچىلىقىدىكى گۇاڭ لۇ ،ساراچۇن قاتارلىقلاردىن تەركىب تاپقان « ئىلى جارچى -سىلاۋچىلىق ئۆمىكى » غۇلجىغا يېتىپ كېلىپ ،ھاكىمبەگ غوجا قاتارلىق ھەر مىللەتتىن بولغان ئابرويلۇق زاتلار ۋە سوۋېت ئىتتىپاقىنىڭ غۇلجىدا تۇرۇشلۇق كونسۇلخانىسىنى زىيارەت قەلدى ھەمدە ھەر مىللەت مۆتىۋەرلىرى قاتناشقان سۆھبەت يىغىنى ئېچىپ : « نىلقا ۋەقەسى شېڭ شىسەي قالدۇرۇپ كەتكەن جاراھەت ،بىز ئۇنى داۋالىماقچى» ( « شىنجاڭ ئۈچ ۋىلايەت ئىنقىلابىغا دائىر چوڭ ئىشلار خاتىرىسى »،شىنجاڭ خەلق نەشىرياتى 1995-يىلى 8-ئاي .ئۇيغۇرچە 1- نەشىرى) دەپ خۇپسەنلىك قىلدى ھەمدە « مەھبۇسلار» نى قويۇپ بېرىش ، ئۆلگەنلەرنىڭ جەسىتىنى ۋە مۇسادىرە قىلىنغان مال -مۈلۈكلەرنى قايتۇرۇپ بېرىشكە ۋەدە بېرىپ ،خالقنىڭ قارشلىق كۆرسۈتىش غەزىپىنى بېسىشقا ئورۇندى .
گومىنداڭنىڭ شىنجاڭدىكى ئارمىيە ،ھۆكۈمەت دائىرلىرى قوللانغان بۇ تەدبىرلەر ھەرمىللەت خەلقىنىڭ بارغانسېرى ئۇلغۇيۋاتقان ئىنقىلابىي كۈرەش ئوتىنى ئۆچۈرەلمىدى ھەم ئۆچۈرەلىشىمۇ مۈمكىن ئەمەس ئىدى .
10-ئاينىڭ 19- كۈنى ،ساۋرىلىڭ ئايرۇپىلان بىلەن غۇلجىغا يېتىپ بارغاندىن كېيىن ، « نىلقا ئەتراپىدىكى باندىتلارنى تازىلاش پىلانى» غا تۈزىتىش كىرگۈزدى .ئۇنىڭ كونكىرېت ئورۇنلاشتۇرۇشى مۇنداق :(1) زاپاس 7- شى 19- تۈەنىنىڭ 1- يىڭى مازاردىن يولغا چىقىپ قاراسۇ ئارقىلىق نىلقىغا ئۇدۇلدىن ھۇجۇم قىلىش ؛(2) 19-تۈەننىڭ 2- يىڭى توققۇزتارادىن يولغا چىقىپ نىلقىغا ياندىن ھۇجۇم قىلىش ؛(3) 19- تۈەننىڭ 2-لىيەنى ۋە 128-شى 383-تۈەننىڭ 3- يىڭى ياماتۇدىن يولغا چىقىپ بالاشىق داۋىنى ئارقىلىق نىلقىنىڭ ئوڭ تەرىپىگە ھۇجۇم قىلىش ؛(4) 19-تۈەننىڭ 6- لىيەنى كۈنەستىن يولغا چىقىپ نىلقىغا كەينى تەرەپتىن ھۇجۇم قىلىش ؛(5)زاپاس 7- شى 21-تۈەنىنىڭ 3- يىڭى ئاچالدىن يولغا چىقىپ ، سۇپتاي ئارقىلىق نىلقىنىڭ ئوڭ تەرىپىدىن ھۇجۇم قىلىش .
ساۋ رىلىڭ يەنە كۇچادا تۇرۇشلۇق ئاتلىق ئەسكەرلەر 4- تۈەنىگە مۇزات داۋىنىدىن ئېشىپ تېكەس ناھىيىسىگە كېلىپ ، جەنۇبتىن شىمالغا قاراپ « قورشاپ يوقىتىش» قا ئۆتۈش توغرىسىدا بۇيرۇق چۈشۈردى . بۇ ئۇرۇنلاشتۇرۇش 11- ئاينىڭ 1- كۈنىدىن باشلاپ يولغا قويۇلدى . كەسكىن ئېلىشىش ئارقىلىق ، زاپاس 7- شى 19 -تۈەنىنىڭ 2- يىڭى 11- ئاينىڭ 2- كۈنى نىلقا ناھىيە بازىرىنى ئىشغال قىلدى .ئارقىدىنلا ، پارتىزانلار قاراسۇدا دۈشمەننىڭ ھۇجۇمىنى توسۇش جېڭىدە ، ئەكبەر ئېغىر يارىلىنىپ ،11- ئاينىڭ10- كۈنى قۇربان بولدى .11- ئاينىڭ 3- كۈنى پارتىزانلار جالىنغۇلدا ئۇلاستايغا ھۇجۇم قىلىشقا كەلگەن گومىنداڭ ئارمىيىسى بىلەن كەسكىن جەڭ قىلدى ، قايتۇرما زەربىگە ئۇچرىغان گومىنداڭ قوشۇنى قاراسۇ ئارقىلىق مازار جىلغىسىغا چېكىندى . نىلقا پارتىزانلىرىنىڭ ئاساسلىق كۈچىمۇ پۈتۈنلەي ئۇلاستاي تاغلىق رايۇنىغا قايتىپ كەلدى . بۇ چاغدا نىلقا ئەتراپىدا پارتىزانلار تەرىپىدىن ئىسكەنجىگە ئېلىنغان گومىنداڭ قوشۇنلىرىنىڭ ئادەم سانى 1600 گە يېقىنلىشىپ قالغانىدى ، ئەمما گومىنداڭنىڭ غۇلجىدا قالدۇرۇلغان ھەربىي -ساقچىلىرى 1000 غا يەتمەيتتى ، ئۇنىڭ ئۈستىگە بىر نەچچە مۇھىم تايانچ بازىلار بۆلۈۋېتىلگەنىدى . زاپاس 7- شى 19-تۈەن 1- يىڭىنىڭ 1-لىيەنى ۋالىي مەھكىمىسىنى ؛ 19-تۈەننىڭ 1- يىڭى (1-لىيىەن بىلەن 2- لىيەن كەم ) ۋىلايەتلىك ساقچى ئىدارىسىنى ؛دۇبەن مەھكىمىسىنىڭ يېڭى ئەسكەرلەر بىلەن تولۇقلانغان مۇستەقىل يىڭى بىلەن ئالاھىدە ۋەزىپە ئۆتەش تۈەنىدىكى قىسمەن ئوفىتسېر -ئەسكەرلەر سانائەت شىركىتى ۋە ھەربىي قوماندانلىق شىتابى ، ھاۋا ئارمىيە جاۋداۋدۈيى قاتارلىق جايلارنى ساقلاشقا قويۇلغانىدى .بۇ يەردە تۇرۇشلۇق گومىنداڭ قوشۇنلىرىنىڭ ئەسكىرىي كۈچى ئاجىز ،قورال-ياراغلىرى بىر قەدەر ناچار بولغانلىقى ئىنقىلابىي ئاممىنىڭ قوراللىق قوزغىلاڭ كۆتۈرىشى ئۈچۈن ئىنتايىن ياخشى پۇرسەت بولدى . سوۋېت ئىتتىپاقىنىڭ غۇلجىدا تۇرۇشلۇق كونسۇلخانىسى مۇشۇ ئەھۋالنى كۆزدە تۇتۇپ ، ئىچكى -تاشقى جەھەتتىن ماسلىشىپ ، غۇلجا شەھرىنى ئىگەللەشكە قولايلىق بولۇش ئۈچۈن ، بىر تەرەپتىن «غۇلجا ئازادلىق تەشكىلاتى » ئارقىلىق غۇلجا شەھرى ئىچىدە قوراللىق قوزغىلاڭ كۆتۈرۈشكە جىددى تەييارلىق كۆرسە ،يەنە بىر تەرەپتىن ،11-ئاينىڭ 4- كۈنى ئۇلاستاي تاغلىق رايۇنىغا ئادەم ئەۋەتىپ نىلقا پارتىزانلىرىنىڭ مازار جىلغىسى ئەتراپىدا گومىنداڭ قوشۇنلىرى بىلەن جەڭ قىلىش پىلانىنى ئۆزگەرتىپ ، باشقا يول بىلەن غۇلجىغا ھۇجۇم قوزغىشىنى تەلەپ قىلدى.
