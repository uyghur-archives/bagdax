مەن سىزنى سەل مەنمەنچىدەك ھېس قىلىپ قالدىمغۇ ياكى ئۇن ...

بۇنى مۇنداق چۇشۇنۇش كېرەك،2007يىلى نوبېل مۇكاپاتى ئالغان ئالىملار ئەنگىلىيە پەننى ئومومىلاشتۇرۇش يازغۇچىلىرىدىن يوھان فاندۇن باشچىلىغىكىلەر يازغان {ئىنسانىيەت تارىخىدىكى ئەڭ ئۇلۇق ئالىملار}دىگان كىتاپنى پۇتكۇل ئىنسانىيەتكە كۇچلۇك تەۋسىيە قىلغان،بىر قەدەم ئىلگىرلەپ چۇشەنسەك،ئالىملار بۇ كىتاپنىڭ ئوبىراز ۋەكىللىگىنى ئوستۇگە ئالغان،بۇنى مۇنۇ توردىن كۇرەلەيسىز،{http://baike.baidu.com/view/5955558.htm?fr=aladdin}،بۇنىڭ نيوتون تۇنۇشتۇرۇلغان قىسمىدا،يەنى 74-بېتىدە،{ئەگەر ئالەم غايەت زور سائەتكە ئوخشىسا،ئۇ بىز بىلەلەيدىغان مۇئەييەن بىر قانۇنيەتكە بويسۇنىدۇ،بۇنىڭدا نيوتون قانۇنلىرى بۇنى كونتۇرول قىلىدىغان بىرىنچى ئامىل ياكى كونتۇرول قىلغۇچى}،بۇ قاراشنى قايتا قايتا تەھلىل قىلسىڭئىز ئالىمنىڭ سائەتكە ئوخشايدىغانلىغى مۇئەييەنلەشتۇرۇلگەن.يەنى ئالەم غايەت زور سائەتكە ئوخشايدۇ دىگەن قاراشنى نيوتون تەسەۋۇر قىلغان،ھەم مىلادى 1685يىللارغىچە  بۇ سائەتنىڭ تېخىچىلا ھەرىكەت قىلمىغانلىغىنى{يەنى بۇ سائەئەتكە ئوخشايدىغان ئالەمنىڭ ھەركەت قىلمىغانلىغىنى  بايقاپ،تەڭرىنى چاقىرىپ،بۇ سائەتنى ھەركەتلەندۇرگەن،{ئەمىليەتتە ھەرىكەت قىلغىنى ئالەم}،ئىنجىلدا كورسۇتۇلۇشىچە،ئالەم مىلادىدىن بۇرۇن 4004يىلى پايدا بولغان،نيوتون بۇ يىللارنى تامامەن قوللۇغان،{نيوتون ئۇمۇرىنىڭ ئاخىرىدىمۇ پۇتۇن كۇچى بىلەن ئىنجىلنى تەتقىق قىلغان،}بۇ يىلغا نيوتون تەڭرىنى چاقىرغان مىلادى 1685يىلنى قوشسىڭئىز 5689يىل بولىدۇ،بۇ يىل دەل ئاشۇ چاغدا نيوتون پادىلانغان ماتىريال،نيوتون 1685يىلى تەڭرىنى چاقىرغان بولسا،بۇ ۋاقىتتا ئالەم ھەرىكەت قىلغىلى 5689يىل بولماستىن بەلكى 0بولغان بولىدۇ،چۇنكى ئالەم نيوتوننىڭ تەڭرىنى چاقىرىشى سەۋەبىدىن ھەرىكەت قىلغان بولىدۇ،بۇ يەنە بىر تەرەپتىن نيوتون تەڭرىنى چاقىرىشتىن بۇرۇن ئالەمدە ئالەملىك تارتىشىش كۇچىنىڭ بولماسلىغى سەۋەبىدىن ئالەم ھەرىكەت قىلمىغان،شۇ سەۋەپتىن نيوتون {ئەگەر ئىسلام دىنىمىزنىڭ قايدىسى بويىچە ئاللاھ بىر دەپ قارىسىڭئىز،نيوتون ئىسلام دىنىدىكى ئاللاھنى چاقىرغان،يەنى ئالاھ بىلمىگەن ئىشنى نيوتون بىلگەن،بۇنىڭغا يەنە بىر قارىسىڭئىز {قۇرئان كەرىم}نىمۇ نيوتون يازغاندەك قىلامدىمۇ نىمە؟چۇنكى ئاللاھ نيوتون بىلگەنچىلىك ئىشلارنىمۇ بىلمەيدىغان تۇرسا،قانداق قىلىپ نەچچە يۇز مىليون مۇسۇلمانلارنىڭ مۇقەددەس كىتاۋىنى يېزىپ چىقالايدۇ}،ئەسلى گەپكە قايتىپ كەلسەك،2007نوبېل مۇكاپاتى ئالغان ئالىملار ئالەمنىڭ سائەتكە ئوخشايدىغانلىغىنى مۇئەييەنلەشتۇرگەن،بۇ ئەمىليەتتە ،نيوتون خۇدانى چاقىرىشتىن بۇرۇن ئالەمدە ئالەملىك تارتىشىش كۇچىنىڭ يوقلىغىنى مۇئەييەنلەشتۇرگەن بولىدۇ،نيوتوبمۇ ئالەمنى ھەركەتلەندۇرگەن كۇچ يوق دەپ قاراپ تەڭرىنى چاقىرغان ئەمەسمۇ،نيوتون بىلەن 2007يىلى نوبېل مۇكاپاتى ئالغان ئالىملارنىڭ قارىشى ئوخشاش،بۇ دىمەك،مىلادى 1685يىلىدىن بۇرۇن ئالەمدە تارتىشىش كۇچى يوق دىگەن لوگىكىنى دۇنيادىكى ئەڭ نوپوزلۇق ئالىملار قوللۇغان،بۇنى ئۈزلىرى بايقىمىغاندەك قىلامدىمۇ نىمە؟بۇ تەھلىل قانداقراق بولدىكىن،بولمىسا يەنىمۇ دەلىللىسەم بولىدۇ؟

بىر مويىم ئىشنى ئۇنتۇپ قاپتىمەن،نيوتون،ئېينىشتىيىنغا مۇناسىۋەتلىك بۇ تىمىنى شىنجاڭ ئىجتىمايى پەنلەر ئاكادىمىيىسىنىڭ مۇئاۋىن مۇدىرى ئابدۇروزاق سايىم ئەپەندىم بىلەن شىنجاڭ ھازىرقى زامان كەسپى تېخنىكا ئىنىستىتوتىنىڭ مۇدىريەت باشلىغى غۇپۇر يۇنۇس ئەپەندىممۇ ھەر كۇنلۇق تەھلىلدىن كۇرۇپ تۇرۇۋاتىدۇ، يېنىدا تور بولمىغاندا،مەن ئۇچۇر يوللاپ تۇرىمەن، دىمەك مەن يالغۇز ئەمەس،تۇنۇگۇن ۋە بۇرناكۇن ئاپتونوم رايون 8قەۋەتتە چوڭ يىغىن ئاچقان،ئابدۇروزاق ئاكىمىز يىىغىنغا قاتناشقان،ئەركىن مۇئەللىم بىلەن بولغان تور سوھبىتىنىڭ مويىم مەزمونىنى قىسقا ئۇچۇر ئارقىلىق يوللاپ بەردىم،چۇنكى چوڭ يىغىندا تېلىفون قۇبۇل قىلىشقا بولمايۇ،يىغىن تۇگۇگەندە،تېلىفونلاشتۇق،چۇنكى بىزنىڭ يەنىمۇ ئىزدىنىش ئارقىلىق ئايدىڭلاشتۇرۇدىغان مەسىلىلىرىمىز خېلى بار،يەنە كېلىپ بۇنداق ماقالىنى يالغۇزلا يېزىپ چىققىلى بولمايدۇ،{يالغۇز ئاتنىڭ دېڭى چىقماس،دېڭى چىقسىمۇ چېڭى چىقماس}بۇ سوزنى بەلكىم بىزنىڭ ئاتا -بوۋىلىرىمىز ئېيتقان،قاراڭ،بىز ھەر كۇنى 80-90دولەتتىكى ئىلىم ساھاسىدىكىلەرگە خەت يېزىپ تۇرىمىز،بەزىدە ئادەم سانى 2000دىن ئېشىپ كېتىدۇ،لېكىن بۇلاردىن ھەممىسىلا بىزنىڭ كۇتكەن يېرىمىزدىن چىقىپ كېتەلمەيدۇ،بۇنداق قىلىشتىن مەخسەت،بۇ كىشىلەر نيوتون قانۇنلىرى بىلەن نىسپىلىك نەزىريەسىگە قانداق قارايدىغاندۇ دىگەن مويىم مەسىلىنى بىلىش،بۇ جاھاندا ھەممىمىز { قېنىي سىلىدىن كەلسۇن.قېنى تورگە چىقسىلا،ئاز ئىشلەپ،كوپ مۇئاش ئالىدىغان ئىشلار ئۈزلىرىدىن كەلسۇن،بىزلەر ئۇمۇر بويى سىلىنىڭ خىزمەتلىرىگە تەييار}دەپ يۇرسەك بولماسمىكىن،چۇنكى ئالاھ بىزنى تەڭ دەرىجىدە ئەقىل بەرگەن،{ھەددىدىن ئاشقان كەمتەرلىك مەغلۇبىيەت}بولۇپ قالۇرمىكىن،مويىم بىر گەپ،ئەركىن مۇئەللىم بىلەن ئۇستاز ئەركىن مۇئەللىم بىر گەپ،سىزمۇ شۇنداق قارامسىزكىن؟
