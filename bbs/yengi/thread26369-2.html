نىمە دىگۈلۈك بۇ ئىشللارغا،بۇنداق ئەھۋاللار بەلكىم يوق ئەمەس بار بولىشى مۇمكىن،
مېنڭ بېشىمغا كەلگەن بىر ئىش .. .بۇلتۇر قىشتا 12-ئاي مەزگىلىدە مەن ئۈرۈمچىگە ماركامنى ئەنگە ئالدۇرۇش ھەم ئاپتۇنۇم رايۇنلۇق تىل تەتقىقات مەركىزىدىن مۇناسىۋەتلىك ئىسپاتلارنى ئېلىش ئۈچۈن بېرىپ 10 كۈن تۇرۇپ قالدىم،سىڭلىمنىڭ ئىجارە ئالغان ئۆيىدە بىرگە تۇرۇش جەريانىدا ئۇ ئۆيدە چۇسا بار ئىكەندۇق،ئاخشىمى توكنى ئۈچۈرۈپ بىز ئۇخلىغان ۋاقتىدا مىنى چۇسا چېقىۋىلىپ جىق تەسىرى بولمىغاندەك قىلغان ئىدى،پويىزدا يولغا چىققاندىمۇ ساپ-ساق ئىدىم كىيىن قانداق بولدى بۈگۈنى كەچ سائەت 7:30 دا پويىز ئۈرۈمچىدىن يولغا چىققان ئىدى،ئەتىسى كۈندۈزى سەئەت 10 ئەتراپىدا پۈتۈن بەدىنىم قىچىشىپ،قىزىرىپ،ھالىم بەكلا خاراپ بولۇپ كەتتى، پويىزدا ئولتۇرۇپ خەقنىڭ قاراپ قېلىشى بىلەن كارىم يوق تىزىمنىڭ تۆۋەق قىسىملىرىنى قاشلاپ تۇرالمايلا قالدىم،شۇ ئارىدا مىنى كۆرگەن بىر خەنزۇ يولداش مەندە بىر سىقماق بار،شۇنى سۈرتۈپ قوياي،ئاغرىق توختايدۇ دەپ سۈرتۈپ قويماقچى بولىۋىدى،مەن ئۆزەم سۈرتىۋالاي ماڭا بىرىپ تۇرۇڭ دەپ ئۆزەم سۈرتىۋالغاندىن كىيىن سەل ئارام تاپقاندەك بولدۇم،شۇنداق قىلىپ ئاقسۇغىمۇ كەپ قالدۇق، ئۆيگە كىرىپ ئۆيدىكىلەر ئەتكەن تاماقنى يەپلا بالدۇرلا ئۇخلىۋالدىم،تۈن تەڭ بولغاندا ئاغرىق يەنە قوزغۇلۇپ ئۇيقۇممۇ بۇزۇلۇپ تاڭ ئاتقۇچە ئۇخلىماي تاڭنى ئاتقۇزدۇم،ئەتكەن سەھەردىلا ئۆزىمىز  تۇرۇشلۇق يېزىلىق دوختۇرخانىغا باېرىپ كۆرسەتسەم ، دوختۇرلار بۇ سىزدىكى كىسەللىك ئەھۋالىنى بىز پەقەت ئۇچرۇتۇپ باقمىغان،بىز قەتئىيلا ئامال قىلالمايمىز،ناھىيەگە بېرىڭ دىگەن ئىدى،دەرھاللا ناھىيەگە بېرىپ ناھىيلىك خەلىق دوختۇرخانىسىغا كىرەيمۇ ياكى ناھىيلىك مىللىي تىبابەت دوختۇرخانىسىغا كىرەيمۇ دىگەن تاللاشقا دۇچ كېلىپ ، ناھىيلىك دوختۇرخانىغا كىرسەم ئوكۇل ئاسىمىز دەپ تۇرۇۋالسا،مەن ئوقۇل ئېسىشقا ئۆچ تۇرسام،ئەڭ ياخشىسى مىللىي تىبابەت دوختۇرخانىسىغا كىرىپ كۆرسىتىپ باقاي،نىمە دەپ ھۆكۈم چىقىرىدۇ،ئەھۋالنى ئۇقۇپ بىر ئىش قىلارمەن دەپ ئويلاپ شۇ بۇيىنچە كۆرسەتكەن ئىدىم،ئۇ يەردىكى دوختۇرلار بۇ‹‹ تارقىلىشچان قىزىل ئەسۋىيسى›› ئىكەن دەپ ھۆكۈم چىقىرىپ بىز داۋالاپ كىتەلەيمىز ياتاقتا ياتسىڭىز ياخشى ئىدى دىگەن ئىدى،مەن ئۇلارغا مەن ياتاقتا ياتالمەيمەن،دۇكىنىم سىلەر بىلەن قوشنا بولغاندىكىن مەن چىقىپ كىرىپ داۋالىناي دىگەن تەكلىپنى بەرسەم، سىزنى دورا قاينىتىپ بەدىنىڭىزدىكى ئىششىپ كەتكەن جايلارنى يۇيۇپ ھورداتمىساق ياخشى بولالمايدۇ،چۇقۇم ياتاقتا يېتىڭ دەپ تۇرۇۋالدى، مەن ماقۇل بولىۋىدىم،1000يۈەن بالنىستتا يىتىش رەسمىيتى تۆلەپ كارۋات كۆرسىتىپ بەرسەك دوختۇرخانىمىزدا ياتسىڭىز داۋالاشنى باشلايمىز دەپ ئورنۇمنى كۆرسىتىپ بىرىپ مەسئۇل دوختۇرنى بەلگىلەپ بىرىپ ئە ئەتىگەندە تاماق يىمەي كىلىسىز تەكشۈرۈش قىلىمىز دىدى، ئەتىسى مەسئۇل دوختۇر كىسەل تارىخىمنى يىزىش جەريانىدا دوختۇرخانىغا كەلگەن چاغدىكى ھالىتىمنى تەسۋىرلەپ يازغىلى تۇرۇپتۇ قارا دىگەندەكلا قاراپ قاپتىمەن، دوختۇر بىمارنىڭ بېشى قايىدۇ، كۆڭلى ئېلىشىدۇ،مېڭىش ھالىتىمۇ ناھايىتى جانسىزپۈتۈن بەدىنى قىزىرىپ ئىششىپ،پىلدىرلاپ سۈيى چىقىدۇ،تاماق سىڭمەيدۇ،قۇسۇش ئالامىتى بار .....قاتارلىق بىر تالاي ئەمەلىيەتچالىقى يوق يالغان كۆپتۈرمە دىئاگنوزلار توختىماي يېزىلىۋاتىدۇ،قان ئېلىش ،پۈتۈن بەدەن بىجاۋ، سىتى دىگەنلەردە تەكشۈرۈش ئېلىپ بېرىش . . تەكشۈرۈش نەتىنجىسى بىماردا ھىچقانداق ئالامەت يوق ساق دىگەن دىئاگنوز نەتىنجىسى چىققان ، ياتاقتا بىر كۈنمۇ يېتىپ باقمىدىم..قىنى مېنى ھوردىتىمىز،تازلايمىز يۇيىمىز، ھەممىسى يالغان بولۇپ چىقتى ، بەرگىنى خالاس ئىسسىق قوقاقنى ياندۇرىدىغان دورىلار ، پۇتۇپنى چىلاپ ئولتۇرىدىغان سۇيۇق دورا  شۇنچىلىكلا، ئەمەلىەتتە ياتاقتا يېتىشىمنىڭ پۈتۈن بەدەن تەكشۈرۈش ئېلىپ بېرىشىمنىڭ قىلچە رۆرۈريىتى يوقلا بىر ئىش ئىكەن ، دورىنىمۇ بەك كۆپ بەرگەن ئىدى، 5 كۈنلا ئىستىمال قىلىپ پۈتۈنلەي ساقىيىپ كەتتىم،شۇ چاغدا بەرگەن دورىلار تاكى ھازىرغا قەدەر تۇرىدۇ،يەنە شۇنداق كىسەللىك كۆرۈنۈپ قالسا ئىستىمال قىلارمەن دەپ تاشلاپ قويدۇم،يېزىلىق دوختۇرخانىدىكىلەر كىسىلىڭىز ئېغىركەن دەپ بىزنى قورقۇتۇپ ناھىيگە يوللاپ، يۇقۇرىدىكىلەر تېخىمۇ قورقۇتۇپ بەكلا ئېغىركەن دەپ بىزنى تېخىمۇ قورقۇتۇپ دورا بىلەن داۋالاشقا مۇمكىن بولىدىغان كىسەللىكلەرنىمۇ بالنىستتا ياتقۇزۇپ كۆپلەپ پۇل تېپىشنى مەقسەت قىلىدىكەن،ئۈستىڭلاردىن ئەرىش قىلىمەن ،كىسەل تارىخىمنى بىرىڭلار دەپ خېلى قورقۇتتۇم،ناھىيلىك فاڭىزەنگىمۇ ئىنكاس قىلىپ قويدۇم ، بۇ قېتىم بوپتۇ ، مەندىن باشقا قانچىلىغان نامرات كىشلەرنى يەنە شۇنداق ئالداشلار يۈز بەرمىسۇن دىمەكچىمەن،دىگەننى فاڭىزەنگە ئۇختۇرۇپ قويدۇم، شۇنىڭ بىلەن ئىشنى ئاخىرلاشتۇردۇم .
