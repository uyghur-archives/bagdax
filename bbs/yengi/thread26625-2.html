     ئەمۋايچىلار قايسى تورداشقا مەھسۇلاتىنى زورلاپ ساتقاندۇ؟ ھازىر بازار ئىگىلىگى دەۋرى. كىمنىڭ مەھسۇلاتى سۇپەتلىك بولسا شۇنىڭكىنى ئىشلىتىدىغان گەپ. مەھسۇلات سۇپەتلىك، باھاسى مۇۋاپىق بولسىلا يەھۇدىي ئەمەس پەھۇدىينىڭ بولسىمۇ بولىۋېرىدۇ. ھازىر ھەم قانۇن ئارقىلىق ئىدارە قىلىنىدىغان دەۋر. قايسى كارخانا ساختىلىق قىلسا پاش قىلىپ قانۇننىڭ ھەيۋىسىنى كۆرسەتسەك بولىدۇ. يەنە كېلىپ جاي جايلاردىكى سۇپەت تەكشۇرۇش، بازار باشقۇرۇش ئورۇنلىرىنىڭ ھەممىسى ئىشتىن توختىغىنى يوق. ئەمۋاي مەھسۇلاتىنى ياقتۇرمايدىغانلار ئۇ مەھسۇلاتنى ئالمىسىلا بولىدۇغۇ. مەھسۇلات سۇپۇتى ۋە بازار ئىچىش ئۇسۇلى جەھەتتىكى «ئالدامچى»لىقى قانۇن ئورگانلىرىنىڭ ئىشى.
     ئۆزەمنى ئىلىپ ئېيتسام نەچچە يىلدىن بىرى ئۇلارنىڭ بىر قانچە خىل مەھسۇلاتىنى ئىشلىتىپ كېلىۋاتىمەن. مەسىلەن: 47 يۇەنگە بىر بوتۇلكا يۇيۇش سۇيۇقلىقىنى ئالسام بىر يىلدىن كۆپرەك يېتىدۇ. ئىلگىرى 3 - 4 يۇەنلىك شۇ خىل مەھسۇلاتتىن ئوخشاش ۋاقتتا 80 - 90 يۇەنلىك ئىشلىتەتتىم. بۇ يۇيۇش سۇيۇقلىقى بىلەن يۇيۇلغان قازان - قومۇش سۇلىرىنى مال چارۋېلىرىمغا خاتىرجەم بېرىمەن.
     مەن ھەرگىزمۇ «ئەمۋايچى»لاردىن ئەمەس، پەقەتلا بىر ئادەتتىكى خېرىدارى، خالاس. شۇڭا «ئەمۋايچى»لارنى كۈنلەيدىغان تورداشلارنىڭ مېنىمۇ شۇ سەپكە قوشۇپ ئەدەپلەپ كەتمەسلىكىڭلارنى ئەسكەرتىمەن.

ئەمۋايچىلار مەھسۇلات ساتقاندا ئۆزىگە چۇشلۇق سېتىش ئۇسلۇبى بار . مال تونۇشتۇرۇش جەريانىدا ئۇ ھەرگىزمۇ سىزگە مەھسۇلاتىنى زورلاپ ساتمايدۇ . پەقەت مەھسۇلاتىنىڭ ئارتۇقچىلىقىنى كۆرسىتىپلا قويىدۇ . بۇ تەرىپىنى مەن كۆرگەن .
ئەمۋاينىڭ لېكسىيەسىنىمۇ جىق ئاڭلىدىم . سالامەتلىكنى ئاسراش توغرىسىدا سۆزلەيدۇ .
ئاڭلاپ قويسا پايدىسى باركى ھەرگىزمۇ زىيىنى يوق . سىزنىڭ قىلىدىغىنىڭىز پەقەت بىرلا ، ئۇ بولسىمۇ لېكسىيەنى ئاڭلاپلا قويۇش ، بەھىرلىنىش ، ئەمۋايغا بولسا قاتناشماسلىق .
مەھسۇلاتىنى ئىشلىتىپ كۆرۈپ ئاندىن بىر نىمە دەڭ .
تېما ئىگىسىنىڭ بەزى سۆزلىرىنىڭ پاكىت ئاساسى يوقتەك بىلىندى . مۇمكىن بولسا ئۆزى ئاخىرىدا مىسال ئېلىپ كۆرسەتكەن مەھسۇلاتلارنىڭ ئاسمان زىمىن باھا پەرقىنى ئەمىلىي ئۇلىنىشلاردىن مىسال ئېلىپ كۆرسىتىپ قويۇشىنى سورايمەن .
