تېما ئىگىسىگە كۆپ تەشككۇرلەر بولسۇن ، بۇ تېما ھازىرقى ئومومىي ئۇيغۇر جەمئىيىتىدىكى يوقۇرى پەللىگە كۆتۈرۈلگەن تۈركىيە قىزغىنلىقىغا نىسبەتەن ھەر جەھەتتىن تولۇق ئومومىي چۈشەنچە بولدى . بۇ تۈردىكى تېمىلارنىڭ كۆپىيىشى بىلەن تورداشلار ۋە كەڭ ئۇيغۇر جەمئىيىتىگە نىسبەتەن تۈركىيە ۋە باشقا ئوتتۇرا شەرق ، غەرب دۆلەتلىرىگە قارىتا چۇشەنچە جەھەتتىن يۈزەكىلىكتىن چوڭقۇرلۇققا ئۆتۈپ بارا-بارا ھەر جەھەتتىن سېلىشتۇرما جەھەتكە بېرىپ يەتتى ، بوپمۇ تۈركىيە تەسىراتلىرىمدىن باشلاپ مېنىڭ بۇ جەھەتتىكى قىزغىنلىقىم ھەسسىلەپ ئاشتى . چۈنكى ، ئوخشاش دىنىي ئېتىقاد ۋە ئوخشاش بىر تۇركىي تىلدا سۆزلەيدىغان ، ئورتاق تارىخىي يىلتىزداشلىققا ئىگە بىر مىللەتنى ، بىر دۆلەتنى ۋە بىر رايوننى چۈشىنىش ، ئۇنىڭغا بەزىدە سەۋدايىلارچە تەلپۈنۈش ، چوقۇنۇشتەك بىزدىكى خاھىش ئاخىرى بېرىپ ۋىجدانى ئۇيغاق قېرىنداشلىرمىزنى بۇ توغرىسىدا توختىماي تەپەككۇر پىكىر يۈرگۈزۈشكە ، ھەم تولۇق توغرا چۈشەنچە بېرىشكە قەلەم تەۋرىتىشكە مەجبۇر بولدى . ئادەم ئەلۋەتتە ئاڭ ئىگىسى بولغاندىكىن ئۆزى قىزىققان نەرسىگە قارىغۇلارچە ئەگەشمەستىن ئۇنى ئالدى بىلەن بېشىدىن تاپىنىغىچە تولۇق چۈشىنىشى كېرەك ، بولمىسا ھەممىنى قارا قويۇق قوبۇل قىلىپ ئۆزىدىكى خاسلىقنى ئالاھىدىلىنى توغرا جارى قىلدۇرالمايدۇ . بۇ جەھەتتە يەنە داڭلىق ئەدىبلىرىمىزدىن يالقۇن روزى قاتارلىقلارنىڭ مەدەنىيەت بوستانىمۇ يە چىن يۈرەكتىن چىن سۆزلەردىمۇ ئىشقىلىپ . ‹ تۈركىيە قىزغىنلىقى بىزگە نېمىلەرنى ئېلىپ كەلدى › ناملىق مۇنازىرە لېكسىيىسى ھېلىمۇ توردا باركەن ، ئىزدىسە ھېلىمۇ چىقامدىكىن . يەنە دېسەم مۇشۇنداق ئومۇمنىڭ قىزغىنلىقى مەنپەئىتى ئۈچۈن جان پىدالىق بىلەن كېيىنكى چىقىش يولىمىزنى مەشئەلدەك يورۇتۇپ بېرىۋاتقان كىشىلىرىمىزگە كۆپتىنكۆپ تەشەككۇر ھەم مىننەتدارمىز . ئۆزۈڭىزگە جاپا ئالماي ، ئېغىر ئالماي ئېرىنمەستىن مۇشۇ تېمىڭىز ئارقىلىق بىزنى نۇرغۇن يېڭى مۇھىم مەزمۇنلار بىلەن ۋاقىپلاندۇرغىنىڭىزدىن مەنچە سىزدىن ھەممىمىز مىننەتدار . بولسا ئاۋام ئۈچۈن مۇشۇنداق ئېسىل تېمىلىرىڭىز ئۈزۈلۈپ قالمىغاي ، قولۇڭىزغا دەرت كەلمىسۇن ............

شەرقىي رىم ئىمپېرىيىسى ئىچىدىكى ناسارالاردىكى ئىككى چوڭ مەزھەپ غەربتىكى كاتولىكلار بىلەن شەرقتىكى پرۇتېسانتلار ئوتتۇرىسىدىكى زىددىيەتتىن ئۈنۈملۈك پايدىلىنىپ ئۇرۇش قوزغاپ كونسىتانتىنوپولنى ئىشغال قىلىپ بۇ شەھەرنى ئىسلاملاشتۇرۇش نىشانى بويىچە تۈرۈكچە ئىسلام ماكانى ئىسلام زىمىنى بول دېگەن مەنىدە بۇ شەھەر نامىنى ئىسلامبول دەپ قويۇپتىكەن . بۇ توغرىسىدا ئەينى ۋاقىتتا پەيغەمبىرىمىز رەسۇلۇللامۇ بۇ شەھەرنى ئىسلاملاشتۇرۇش ھەققىدە ئۇچۇر بېرىپتىكەن ، بۇ خەلىپىلىك دەل ئوغۇز تۈرۈكلىرىنىڭ قولىغا ئۆتكىلى ئۇزۇن بولغاچقا بۇ پەيغەمبىرىمىز ئاغزىدىن چىققان ۋەزىپىنى ئوسمانىيە سۇلتانلىرىنىڭ ئەمەلگە ئاشۇرۇش نىيەت مەقسەتلىرى بۇرۇنلا كۆڭلىگە پۈكۈلۈپ بولغانىدى . جۇمھۇرىيەت قۇرۇلغاندىن كېيىن ھاكىمىيەت بىلەن دىن ئايرىۋېتىلگەن بولغاچقا ، ئۇنىڭ ئۈستىگە ، ھەر خىل ئىشلار سەۋەبلىك تۈرۈك مىللەت ئېڭىنىڭ شەكىللىنىشى بىلەن مىللىي دۆلەت قۇرۇش نىشانى بويىچە ئىستانبول يەنى زىمىن ماكان بول دېگەن مەنىدە ئۆزگەرگەن . چۈنكى بۇ نام ئىلگىرىكى نامىغا ئېيتىلىش جەھەتتىن يېقىن ھەم ئۆزگىچە شارائىتقا نىسبەتەن يېڭىلىق ۋە ئۆزگىچە . بۇنى مەن ‹ ئىسلامبولنىڭ فەتىھى قىلىنىشى › دېگەن فىلىمدىن كۆرىشىم . ئۇنىڭ ئۈستىگە سالجۇق ئورخۇن ئۇيغۇر خانلىقى يىمىرىلىپ قارا خانىيلار قۇرۇلۇشتىن بۇرۇنقى ئۇيغۇرلار ئىچىدىكى بىر مىڭ بېشىنىڭ نامى ، كېيىنچە ئۇ بۆلۈنمە كۈچكە ئايلىنىپ ئاستا-ئاستا كۈچىيىپ چىقىپ دۇنياغا مەشھۇر تارىختىكى ئەڭ كۈچلۈك ئىمپىرىيىلەردىن بىرى سالجۇقىيلار ئىمپېرىيىسىنى قۇرۇپ چىققان . كۈچىيىش مەزگىلىدە ئەينى ۋاقىتتىكى ئۇرۇش مالىمانچىلىق ، بۆلۈنۈش تۈپەيلى بىر قىسىم پارىسلار قولىغا ئۆتۈپ كەتكەن تۈرۈك زىمىنلىرىنى قايتۇرۇۋېلىشتا سالجۇقىيلار ئالدى بىلەن قېرىنداش غەربىي قارا خانىيلار بىلەن بولغان مۇناسىۋەتنى ئۆز-ئارا ياخشىلاپ يول ئېچىپ شۇ ئارقىلىق ھەتتا غەربىي ئاسىياغىمۇ بېسىپ كىرىپ چوڭ ئىمپېرىيىگە ئايلانغان . غەربىي قارا خانىيلارغىمۇ ئىگىدارچىلىق قىلغان بولۇپ ، قارا قىتان ۋە مۇڭغۇللار بىلەن بولغان سۈركىلىش مۇناسىۋەتلەردە ھەممىسى چڭگىز ئىستېلاسىغا بويسۇنۇپ ئوتتۇرا ئاسىيادىكى ھۆكۈمرۈنلىقى ئاخىرلىشىپ پەقەت غەربىي ئاسىيا بىلەنلا چەكلىنىپ قالغان . شۇڭا مۇڭغۇللار قىپچاق تۈرۈكلىرىنى ئالدىغا سېلىپ بېسىپ كىرگەچكە ئوغۇز دالاسى دېگەن ئۇقۇم تەدرىجىي قىپچاق يايلىقىغا نۆۋەتلەشكەن ، چۈنكى تۈرۈكلەر ئوغۇز ۋە قىپچاق دەپ بۆلىنىدۇ. ئۇلارنىڭ ئىسلامنى قوبۇل قىلىشى غەربكە سۈرۈلۈشتىن بۇرۇن يەنى دەسلەپ ئۆز جايىدا يېڭى قەدەمدە كۈچ بولۇپ ئۇيۇشقان چاغدىلا باشلانغان بولۇپ ، پارىس ئەرەبلەرنى بويسۇندۇرغاندىن كېيىن ھەربىي ئىقتىسادىي كۈچىنىڭ يوقۇرىلىقى ھەم ئورتاقلاشقان دىنىي ئېتىقادنىڭ تۈرتكىسىدە تەدرىجىي ئىسلامدىكى خەلىپىلىك ھوقوقى ئەرەبلەنىڭ قولىدىن ئوغۇز تۈرۈكلىرىنىڭ قولىغا ئۆتكەن . بىر مەزگىل ئۆتۈپ چىڭگىز ئىستېلاسىنىڭ تۆمۈر تاپىنى غەربكىمۇ سۈرۈلۈپ مۇڭغۇللار سالجۇقىيلار ئۈستىدىن غەلىبە قىلدى . بۇ چاغدا مۇڭغۇل قوشۇنلىرىدا ئۇيغۇرلارمۇ بار بولۇپ ، كېينچە تۈركىيىدە ھازىرغىچە ماكانلىشىپ قالغان . شۇ چاغدا خوتەندىن چىققان ئۇيغۇرلار ئاناتولىيە ۋادىسىغا خوتەندىكى پەرھات-شېرىن داستانىنى تۈركىيىگە ئېلىپ چىقىپ تۈركىيىدىمۇ كەڭ تارقىلىشىغا تۈرتكە بولغان . مۇڭغۇل ھاكىمىيىتى ئاجىزلىشىپ سالجۇق ئىمپېرىيىسى قايتا كۈچەيگەن ، ھەتتا تۆمۈرلەڭ ئىمپېرىيىسى كۈچەيگەن چاغلاردا بۇ ئىككى ئوتتۇرىدىمۇ سۈركىلىشلەر بولغان . كېينكى ھاكىمىيەت ئۆزگىرىشىدە ئوسمان ناملىق سۇلتاننىڭ تەختكە چىقىشى بىلەن بۇرۇنقى سالجۇقنىڭ نامىغا ئوخشاش بۇ كۈچلۈك ئىمپېرىيىنىڭ نامى يېڭى تەختكە چىققان سۇلتاننىڭ نامى بىلەن ئاتىلىپ ئوسمان تۈرۈك ئىپېرىيىسى دەپ ئاتالغان شۇڭا ئۇيغۇرلار بىلەن باشقا تۈركىي مىللەتلەرنىڭ مۇناسىۋىتى مۇشۇ ئاساستا بولىدۇ . شۇڭلاشقىمىكىن ، تۈركىيىدىكى ئۇيغۇرلار نوپوسىغا نىسبەتەن گۇمانىم بار ، ئۇيغۇرلار تۈركىيىدە قوشنىلىرى بىلەن ئارىلىشىپ ياشىغىلى خېلى ئۇزۇن زامانلار بوپتۇ ، ئارانلا تېمىدا ئەسكەرتكەندەك 300 مىڭ بولارمۇ ، يەنە بەزىلەر مىليوندىن ئېشىپ بارىدۇ دەيدۇ  .
