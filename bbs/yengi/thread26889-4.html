يا قۇشقاچ بولالماي يا بۈركۈت بولالماي ئوتتۇردا قالغان  ...

شۇ دېمىسىمۇ ئۆزۈمىزنىڭ مەھسۇلاتىنى ئىشلەتمەي ، خەقنىڭ مەھسۇلاتىغا قارىغۇلارچە چېپىپ بېكاردىن بېكار خەققە پايدا مەنپەئەت يەتكۈزۈۋاتىمىز ، خەقنىڭ بازىرى كېڭىيىپلا كېتىۋاتىدۇ ، ئۆزۈمىزنىڭ بولسا ئۆز رايونىمىز ئىچىدىمۇ خەۋپ ئىچىدە ، ئۆز مەھسۇلاتىمىزنىڭ تەرەققىياتى ھەم شۇ يولدا تەۋەككۈلچىلىك ئىلكىدە مەغرۇر قەدەم بېسىۋاتقان قېرىنداشلىرىمىزغا چىقىش يولى بولسىمۇ قالدۇرۇپ قويغۇلۇق . بىر چاغلاردا تۈكىيە يېمەكلىك مەھسۇلاتلىرى دىيارىمىزنى قاپلاپ كەتكەنىدى ، كەينىدىنلا ئارمان ئىخلاس ئامىنە مەھسۇلاتلىمىزمۇ بوش كەلمىدى ، دېمەك بىز ئەلۋەتتە قىلالايمىز ، لېكىن بۇ بازار رىقابىتىدە يەنىلا باشقىلارنىڭ مەھسۇلاتلىرىنىڭ بازىرى يەنىلا ئىتتىك ، تۈرىمۇ كۆپ بولماقتا ، شۇڭا قېرىنداشلار ھەر جەھەتتىن ئويلىغاندا يەنىلا ئۇيغۇرچە مەھسۇلاتلار ياخشى . ھازىر ئۇيغۇرچە مەھسۇلاتلارنىڭ تۈرىمۇ خىلىمۇ كۆپەيدى ، لېكىن ‹‹ ئادەم جىق بولسا ئىش ھارام ›› دېگەندەك يەنە بىر قىسىم يېڭى چىققان بوپمۇ يىمەك-ئىچمەككە دائىر مەھسۇلاتلاردا خېلى گۇمانلىق نۇقسانلار ، يەنى مەھسۇلات خۇرۇچى ئىشلەپچىقىرىلغان ئورۇن شىكەت نامى قاتارلىق جەھەتلەردە ئېنىق گۇمانلىق نۇقتىلار بولغاچقا  بۇ جەھەتتە  جەمئىيەتتە تۈرلۈك سۆز-چۆچەك بولماقتا ، م : ئۇيغۇرچە يېڭى چىققان ئەرزان باھالىق ئالمىخان كەمپۈتلىرى ، قازاقلارنىڭ داستارقان قاتارلىقلار . شۇڭىمۇ ھازىر كىشىلىرىمىز قىممەت بولسىمۇ ئەرەب پاكىستان تۈركىيە دۈبەي مەھسۇلاتلىرىغا ئەڭ ئىشەنچىلىك پاكىز دەپ يۈگرەيدۇ ، شۇڭا ئۆزىمىزنىڭ بازىرى ئانچە بەك روناق تاپالمايۋاتىدۇ ، ئەمەلىيەتتە بولسا ئۆزۈمىزنىڭ تېخمۇ ئەرزان باھاسىمۇ مۇۋاپىق . ئەمدى بايىقى مەن مىسالغا ئالغاندەك نەدىن كەلگەنلىكى نائېنىق بولغان ئىككى توك-توك مەھسۇلاتنىڭ ھېلىقى ‹‹ بىر تال ئالما بىر سېۋەت ئالمىنى سېسىتىپتۇ  ›› دېگەندەك ئەسلىدە بار بولغان يېڭىدىن چىقىۋاتقان ھەم چىقماقچى بولغان ئىشەنچىلىك مەھسۇلاتلارنىڭ بازىرىنى ئۆلتۈرۈپ قويمىسىلا بولاستىغۇ . يىراقتىكى موللام داموللام ، يېقىندىكى موللام جا موللام دېگەندەك بۇرۇنلاردا ئوخشاش بىر نەرسە بولسىمۇ خەنزۇنىڭ ياخشى دەپ خەنزۇدىن ئالىدىغان ئىشلارمۇ جەمئىيەتتە ئاز ساندا بار ئىدى ، لېكىن ھازىر بولسا يوقۇرى ئىشەنچ تۇرغۇزۇلغان ، خەلق رايىغا باققان مەھسۇلاتلىمىزنىڭ تۈرى تۈرى بىلەن بارغانچە كۆپلەپ مەيدانغا كېلىشى بىلەن نۇرغۇنلىغان كىشىلىرىمىز ئۆزىمىزنىڭ مەھسۇلاتىغا چاپىدىغان ياخشى ۋەزىيەت بارلىققا كەلدى . بۇرۇنلاردا گەرچە كىچىك بولساممۇ ئۇيغۇر ئالدامچى سۈپىتى تۆۋەننى ساتىدۇ خەنزۇنىڭ بولسا سۈپىتى بەك ياخشى ئادەم ئالدىمايدۇ دېگەننى ئاڭلىسام بەك ئاچچىقىمدىن يېرىلىپ كېتەتتىم ، يېقىندا بەزى بىر ئادەمنى چۆچۈتكىدەك ئۇچۇرلارنى ھەر يەرلەردىن بىلىپ يېتىپ ئەل-ئاغىنىلەر ئارىسىداخەنچە يېمەك-ئىچمەك دېسە ھۆ دەيدىغان بولۇپ كەتتۇق ، چۈنكى بىلىشىمچە جۇڭگودا ئىشلەپچىقىرىلغان ئىچىملىكلەرنىڭ ھالاللىق نىسبىتى 40 پىرسەنتكە يەتمەيدىكەن . لېكىن ئارمانغا چۇشلۇق دەرمان يوق دېگەندەك بەزى بىر ئۇيغۇرچە ئىچىملىكلەرنىڭ پاكىزلىق تەرىپى گەرچە ئۆلچەمگە يەتسىمۇ لېكىن قانداقتۇ مەبلەغ يېتىشمەي مەبلەغ سېلىشىمىز يېتەرسىزمۇ بىلمىدىم ئىشقىلىپ تەمى باشقىچىلا ئادەمگە خۇشياقمايدىغان دەرىجىدە تېتىيدىكەن ، شۇڭلاشقىمۇ بىرنىڭ ئۈستىگە بىرى چىقسىمۇ سۈپەت ھەر جەھەتتىن ئۆلچەمگە يەتمىسە بىكاردە ، شۇڭا نۇرغۇن مەھسۇكاتلىرىمىز چىقتى لېكىن ناھايىتى ئاز ساندىكىسىلا قۇدرەت تاپالىدى ، م : ئابىدە سۇدا سۈيى ، ماڭگو شەربىتى . نېمە ئۈچۈن بۇلارنى مىسالغا ئالدىڭ دېگەندە تېخى يېقىندا ئالدىنقى ھەپتە ئاغىنىلەر بىلەن مەكتەپ ئالدىدىكى ئۇيغۇرنىڭ دۇككىنىغا كىرىپ ئىچىملىك ئالدۇق ئاغىنە بالىلار سۇدا سۈيى ئالدى ، مەن قاراپ باقسام يېڭى چىققان ئۇيغۇرچە ئىچىملىكنىڭ تۈرى خىلكەن ، شۇڭا سۇدا سۈيى ئىچىۋەرمەي باشقىسىنىمۇ ئىچىپ باقايلى دەپ جىيەن لى باۋ ئېلىپ ئىچسەم بەك لاۋزا تېتىپ كەتتى خۇددى سۇ ئارىلاشتۇرغاندەك ، چىقىمنى دەپ بەلكىم ئارىلاشتۇرغاندۇ ، بۇ قانداق بولغىنى ئەمدى ، بۇنداق بولسا قانداقمۇ بازار تېپىپ خەق بىلەن رىقابەتلىشىپ ماڭالايدۇ ، تۇرغىچىلا كىچىك ۋاقلىرىمدا ئاڭلايدىغان ئۇيغۇرنىڭ مەھسۇلاتى ساختا سۈپىتى ناچار دېگەننى ئېسىمگە ئالدىم ، لېكىن بۇنداق گەپ تۇراقلىشىپ مېڭىۋەرسە ئەسلا بولمايتتى ، يەنە ئەڭ بۇرۇن چىققان ھەم ھازىرمۇ تەسىر كۈچىنى يوقاتمىغان قاراقات بىلەن سېلىشتۇرسام گەرچە كونا بولسىمۇ شۇ قاراقات بۇنىڭدىن مىڭ ياخشىكەن دەپ قالدىم . ئۆزىمىزنى ئۆزىمىز يۆلىمىسەك باشقىلار بىزنى يۆلىمەيدۇ ، گەرچە مەھسۇلاتلىرىمىزدا سۈپەت مەسىلىسى كۆرۈلسىمۇ ئۆزىمىزنىڭكىنى ئىشلىتىش ،قوللاپ قۇۋەتلەش ئارقىلىق مەھسۇلاتلىرىمىزنىڭ سۈپىتىنى ئۆستۈرۈشكە تېخىمۇ كەڭ بازار تېپىشىغا ، تۇرمۇشتا قولايلىق يارىتىشقا تېخىمۇ كۆپ مەبلەغ پۇرسەت شارائىت يارىتىپ بېرەيلى ، شۇنداق قىلساقلا ھەر بىر كارخانىچىمىز ھەتتا ھەر بىر ئادىمىمىز خەلقتىن ئېلىپ خەلققە قايتۇرۇشتەك ئېسىل خىسلەتنى دەۋر ئىنكاسىنى مەنچە ھەرگىز ئۇنتۇپ قالمايدۇ ، شۇنداق بولغاچقىمۇ م : مەشرەپ مەھسۇلاتلىرىمىز ئىشەنچىلىك خەلق رايىغا باققانلىقى ئۈچۈن ھەممىمىزنىڭ ئۆيىدە مەشرەپ بولسۇن دەپ ئېلانلار بېرىلىۋاتىدۇ ، خەلقمۇ كۆڭلىنى توختىتىپ بەس-بەستە سېۋتىۋېلىپ ئۆي-ئۆيلىرىنى ئۇنىڭ بىلەن توشقۇزۇۋاتىدۇ ، بازىرىمۇ بولۇشىغا قايناۋاتىدۇ ، شۇڭا ئۆۈمىزنىڭ ئۆزۈمىزگىلا ياخشى ، يەنىلا ئۆزۈمىزنىڭ ئۆزۈمىزگە قايتىدۇ ئەمەسمۇ .

يوللىغان ۋاقتى 2014-11-9 16:22:00 |ھەممە قەۋەتنى كۆرۈش
كوپچىلىك ھەرقانداق ئىشنى ھاۋالە قىلىپ ياسىتىپ بازارغا سالدۇ،مەسلەن ،ئىچىملىك،يىمەكلىك،سۇت،ئىلىكتىر ئۇسكۇنلىرى،ئائىلە سايمانلىرى،قاتارلىق مەھسۇلاتلارنىڭ ھەممىسىنى ئىچكىرى ئولكىلەردە ھاۋالە قىلپ ياسسىتىپ ئۇيغۇرچە ماركا بېسىپ بىزنىڭ ئويغۇر بازىردا ساتىدۇ،كارخانچىلارنى ئالدامچى قاتارىغا چىقارساق بولمايدۇ،قايسى بىر كارخانا قۇرۇلۇپ ئوزى مۇستەقىل مەھسۇلاتىنى بازارغا سالغان،ھەرقانداق تۇنجى قۇرۇلغان شىركەت،كارخانلار ئاۋال ھاۋالە قىلىپ مەھسۇلاتنى بازارغا سېلىپ تەرەقى قىلىپ مەلۇم ئىقسادى ئۇنۇم ياراتقاندىكىن ئاندىن ئوزى مۇستەقىل زاۋات قۇرۇپ تەرەقى قىلدۇ.
  1. پەقەت مەھسۇلات نامى ئۇيغۇر يېزىقىدا بېسىلغان،باشقا ئۇچۇرلىرى خەنزۇ يېزىقىدا، بۇلار%100 ئىچكىرى ئۆلكىلەردە ئىشلەنگەن، كۆپىنچە ئەرزان باھالىق بالىلار يىمەكلىكى قاتارلىقلار.....
   2.مەھسۇلات نامى،ماركىسى،شىركەت نامى(xxسودا شىركىتى) ئۇيغۇر تىل-يېزىقىدا بېسىلغان،ئىشلەنگەن ئورنى ئادرېسى بولسا خەنزۇ يېزىقىدا،  شىنجاڭ ئىچى ياكى ئىچكىرى ئۆلكىلەردىكى باشقا شىركەت زاۋۇتلاردا ئىشلەنگەن؛ بۇ تۈردىكى مەھسۇلات ھەر خىل ساھەلەردە كۆپ ئۇچرايدۇ؛
   3.مەھسۇلات نامى،ماركىسى،شىركەت نامى، ئىشلەنگەن ئورنى ئادرېسى ئۇيغۇر تىل-يېزىقى بىلەن خەنزۇ يېزىقىدائوخشاش مەزمۇندا بېسىلغان،پەقەت مۇشۇ تۈردىكى مەھسۇلاتلار ھەقىقى ئۆز مەھسۇلاتىمىزدۇر، سىز دېگەن يۇقارقى ئىشلار ھەقىقەتەن مەۋجۇت،بەزى شىركەتلەر خەلىقنى ئالدىمايمەن دەپ ھەممە ئۇچۇرلىرىنى تۇلۇق يازدۇ.
مەسىلەن شىنجاڭ دىننار بئىو-پەن تېخنىكا تەرەقىيات چەكلىك شىركىتى بازارغا سالغان مەھسۇلاتلىرىنىڭ سىز دېگەندەك ئەھۋال مەۋجۇت ئەمەس.ئىشەنمىسىڭز كۇرۇپ باقسىڭز بۇلدۇ.بۇ شىركەت شىنجاڭ بازىرغا تۇنجى ئاقسىلنى ئەكىرگەن شىركەت،ھازىر تەرەقى قىلدى ،ئەمدى بۇ شىركەت ئاقسۇ توقسۇ ناھىيىسىگە ئاقسىل زاۋۇتىنى ئوزى قۇردى.
شۇڭا نۇقىل ھالدا ھەممە شىركەتلەرنى سىز دېگەن يۇقۇرقى 3 ئولچەمگە سالسىڭز بولمايدۇ.

   1. پەقەت مەھسۇلات نامى ئۇيغۇر يېزىقىدا بېسىلغان،باشقا ئۇچۇرلىرى خەنزۇ يېزىقىدا، بۇلار%100 ئىچكىرى ئۆلكىلەردە ئىشلەنگەن، كۆپىنچە ئەرزان باھالىق بالىلار يىمەكلىكى قاتارلىقلار.....
   2.مەھسۇلات نامى،ماركىسى،شىركەت نامى(xxسودا شىركىتى) ئۇيغۇر تىل-يېزىقىدا بېسىلغان،ئىشلەنگەن ئورنى ئادرېسى بولسا خەنزۇ يېزىقىدا،  شىنجاڭ ئىچى ياكى ئىچكىرى ئۆلكىلەردىكى باشقا شىركەت زاۋۇتلاردا ئىشلەنگەن؛ بۇ تۈردىكى مەھسۇلات ھەر خىل ساھەلەردە كۆپ ئۇچرايدۇ؛
