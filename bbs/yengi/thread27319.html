ئەسسالامۇئەلەيكۇم قېرىنداشلار! ئوبدان، ئامان-ئېسەن تۇرۇۋاتامسىلەر؟
     مەن نوپۇس ئىشىمدىن كۆپچىلىكتىن مەسلىھەت سوراپ باقاي دېگەنتىم. كۆپچىلىكنىڭ بەلەن مەسلىھەت-پىكىرلىرىنى ئايىماسلىقىنى ئۈمىد قىلىمەن.
ئىش مۇنداق: مەن ئوقۇش پۈتتۈرگەندىن بۇيان ئۈرۈمچىدىكى مەلۇم بىر ئىدارىدە ئىشلەۋاتىمەن. نوپۇسۇم ئىدارىنىڭ كوللىكتىپ نوپۇسىدا، ئايالىم بولسا مەلۇم بىر شەخسى دوختۇرخانىدا ۋاقىتلىق ئىشلەۋاتىدۇ، نوپۇسى بولسا ئەسلىدىكى يۇرتىدا (يەنى جەنۇبدا). ئىدارە يېڭى شەھەر رايونىدا بولغانلىقىتىن، قولايلىقلىقنى كۆزلەپ، يېڭى شەھەر تەرەپتىن يەنى ئىدارىگە يېقىن جايدىن بىر يۈرۈش ئۆينى ئىجارە ئالغان. ھازىر شۇ جايدا ئولتۇرۇۋاتىمىز. ئايالىم ھازىر ھامىلدار. بۇ بىزنىڭ قىسقىچە ئەھۋالىمىز.
    شۇڭا مەن بالا توغۇلغاندىن كېيىن نوپۇس ئىشىدا ئاۋارىچىلىق بولمىسۇن دەپ ئايالىم ئىككىمىزنىڭ نوپۇسىنى بىر يەرگە ئەكىلىۋالايلى دېگەن. شۇ مەقسەتتە ئىدارىنىڭ نوپۇس ئىشلىرىغا مەسئۇل خادىمنىڭ قېشىغا كىرسەم «بۇنى مەن بىلمەيدىكەنمەن، ساقچىخانىغا چىقىپ باققىن» دەپ يولغا سالدى.   مەن ساقچىخانىغا چىقسام ئۇلار «پەقەت ئۆي ئالغاندىلا بۇ مەسلىنى ھەل قىلغىلى بولمايدۇ» دەيدۇ. مەن «ئۇنداقتا بۇ مەسلىنى قانداق ھەل قىلىمەن؟» دەپ سورىسام، ئۇلار «باشقا ئامال يوق، ئايالىڭنىڭ نوپۇسى تۇرۇشلۇق جاي بىلەن ئالاقىلشىپ باق» دەيدۇ. شۇنىڭ بىلەن  يەنە ئامال بولۇپ قالار دەپ ھازىر ئىجارىگە ئولتۇرۇۋاتقان ئۆيىمىز تەۋە بولغان ئاھالىلەر كومىېتىغا چىقسام «بۇ بىزنىڭ خىزمەت دائىرىمىزدىكى ئىش ئەمەسكەن» دەپ تۇرۇۋالدى. مەن يەنە «ئەمسە نېمە ئامال بار؟ بالا توغۇلسا، نوپۇسىنى قانداق قېلىمەن؟» دېسەم، ئۇلارمۇ ساقچىلار ئېيىتقان «ئۆي ئېلىش ياكى ئەسلى نوپۇس تۇرۇشلۇق بىلەن ئالاقىلىشىپ بىر تەرەپ قىلىش» نى ئېيتىپ تۇرۇۋالدى. مۇئامىلسىمۇ ئانچە ياخشى ئەمەس بۇ ھۆكۈمەت مەپىسىدە ئولتۇرۇدىغان خوجاملارنىڭ....
  ھازىر ھەر قايسى يۇرتلاردا ئوقۇپ، ئىشلەپ قالىدىغان ئادەملەر بەك كۆپ بولغانلىقتىن، مېنىڭچە بۇنداق ھادىسەلەر ناھايىتى نۇرغۇن، ھەم كۆپ قىسىم ئادەملەر مۇشۇنداق ئىشلارغا يولۇقۇپ باققان، دەپ قارايمەن. ئايالىم ئىككىمىزنىڭ نوپۇسنى بىر يەرگە ئەكلىش ئۈچۈن، ئىككىمىزنىڭ نامىدا چۇقۇم «ئۆي بولۇش» دەيدىغان پىرىنسىپ پۇت دەسسەپ تۇرالامدۇ؟   شۇڭا كۆپچىلىكتىن سوراپ باقسام:
(1) پەرزەنتىمىز توغۇلغاندىن كېيىن ئۇنىڭ نوپۇس ئىشلىرىنىڭ تېخىمۇ مۇرەككەپ بولۇپ كەتمەسلىكى ئۈچۈن، ھازىر ئۆز ئۆيىمىز بولمىغان شارائىتتا قانداق قىلسام، ئايالىم ئىككىمىزنىڭ نوپۇسىنى بىر دەپتەرگە يازغۇزالايمەن؟
(2) يەنە ئەگەر ئايالىمنىڭ نوپۇسىنى يۆتكەش زادىلا مۇمكىن بولمىغان شارائىتتا، تۇغۇلغۇسى بالىمىزنىڭ تۇغۇش رەسمىيىتىنى، نوپۇس ئىشلىرىنى قانداق ھەل قىلىمەن؟
(3) ئايالىم ئىككىمىزنىڭ نوپۇسى ئىككى جايدا بولغا شارائىت ئاستىدا، ئايالىمنىڭ نوپۇسى تۇرۇشلۇق ئورۇندىن بىزگە توغۇش رەسمىيىتى چىقىرىپ بېرىشى مۇمكىنمۇ؟
  مۇشۇ مەسىلىلەرنى ئويلاپ يۈرگىلى بىر قانچە كۈن بولدى، سورايدىغان يەرلەردىن سوراپ ئېنىق جاۋاب ئالالماي كەڭ تور ئەھلىدىن بۇ مەسلىنى سوراپ باقاي دەپ بۇ ئىلتىماسىمنى يازدىم. بۇنداق مەسىلىگە يولۇققان مەنلا بولماسلىقىم مۇمكىن، شۇڭا ھەممەيلەننىڭ پايدىلىنىشى ئۈچۈن، بۇ توغرىسىدا بىلگىسى بار قېرىنداشلارنىڭ سەمىمىي تەكلىپ-پىكىرلىرىنى ئايىماسلىقىنى ئۈمىد قىلىمەن.

سىرتتىن  كەلگەنلەر ھەرقانداق بىر شەھەرگە نوپۇسىنى ئورۇنلاشتۇرۇشتا چوقۇم شۇ شەھەردە  تۇرغان ۋاقتى چوقۇم بەش يىلدىن ئاشقان بولىشى ، مۇقىم ئىقتىسادى كىرىمى ۋە شۇ شەھەردە چوقۇم ئۈيى بولىشى كىرەك ، ساقچىغانا سىزگە توغرا چۈشەندۈرۈش بېرىپتۇ ، يوقۇرىقى 3 شەرت ھازىرلانغادا نوپوس ئىشىڭىزنى بېجىرىش تەس ئەمەس ،
ئۈزىڭىزمۇ ئويلاپ كۈرۈڭ ئۈيىڭىزبولمىسا ئايالىڭىز بىلەن ئۈزىڭزىنىڭ نوپوسىنى قەيەردىكى  ئادىرسقا تىزملىتىسىز
            نوپۇس ئىشى ئەلۋەتتە ئاھالىلەر كومىتتى بىلەن موناسىۋەتسىز ،ئۇلاردىن رەنجىڭىزنىڭ ئاساسى يوق ،  سىز ئالدى بىلەن ئۈي ئېلىڭ ، ئەگەر ئۈي ئېلىشقا قۇربىڭىز يەتمىسە قاچان ئۈي ئالغۇچە نوپۇسىڭىز ھازىرقى ھالەتتە تۇرىۋەرسىمۇ ھېچقانداق ئىشىڭىزغا تەسىرى بولمايدۇ ،   
بىلمىدىم ئايالىڭىز ئۈرۈمچىدە ئولتۇراقلىشى كىنىشكىسى 居住证  بېجىرگەنمۇ يوق ، ئولتۇراقلىشىش كىنىشكىسى بولمىسا بالىڭىز تۇغۇلغاندىن كىيىن بالىغا ئەملەش ئوكۇللىرنى ئۇرمايدۇ ، بۇ كىنشكنى ئۈزىڭزى تۇرۇشلۇش جايدىكى ساقچىغانىنىڭ ساقچى ئىشلىرى ئىشغانسدا بېجرسىز ، ھازىر ئولتۇرىۋاتقان ئۈينىڭ ئىگىسى بىلەن ئالاقىلىشىپ ئۇنىڭ كىملكىنى بىر نۇسغا كۈپەيتىپ ساقچى ئىشلىرى ئىشغانىسغا تاپشۇرۇپ بېرىسىز ، ئايالىڭىز چوقۇم ئۈزى بېرىش كىرەك ، رەسىمگە تارتىپ كىنىشكىنى 1ھەپتە ئىچدە چىقىرپ بېرىدۇ
بالىڭىزنىڭ نوپۇسىغا كەلسەك بالىڭىز تۇغۇلغاندىن كىيىن ئۈزىڭزى تۇرۇشلۇق جايدىكى ساقچى ئىشلار ئىشغانىسغا بېرىپ بالللىق بولغانلىقىڭىزنى ئېيتىپ ئۇلارنى خەۋەرلەندۈرۈپ قويۇڭ ، پەقەپ ئۈيڭىز بولغاندىلا ئاندىن 3 ئادەمنىڭ نوپۇسىنى بىر ئادىرىسقا تىزملاتقىلى بولىدۇ ،   ئۈيڭىز بولمىغان شارائىىتا بالىنىڭ بوپۇسىنى قانداق قىلىش كىرەكلىكىنى مەن يەنە باشقىلاردىن سوراپ سىز گە جاۋابىنى بېرەي ،

يوللىغان ۋاقتى 2014-12-12 03:19:30 |ھەممە قەۋەتنى كۆرۈش
11-دېكابىردىن باشلاپ ئۈرۈمچىدە نوپۇس ۋە كىملىك بېجىرىش ۋاقىتلىق توختىتىلىدۇ
ئاسىيا كىندىكى تورىنىڭ 9-دېكابىر خەۋىرى، مۇخبىرنىڭ 8-دېكابىر ئۈرۈمچى شەھەرلىك جامائەت خەۋپسىزلىكى ئىدارىسىدىن ئىگىلىشىچە، 2014-يىلى 12-ئاينىڭ 11-كۈنىدىن 12-ئاينىڭ 26-كۈنىگىچە، ئۈرۈمچى شەھەرلىك جامائەت خەۋپسزلىك ئورگانلىرى نوپۇس ۋە كىملىك بېجىرىش كەسپىنى توختىتىدىكەن.
8-دېكابىر، ئۈرۈمچى شەھەرلىك جامائەت خەۋپسىزلىكى ئىدارىسىدىكى مۇناسىۋەتلىك خادىمنىڭ تونۇشتۇرۇشىچە، نوپۇس ۋە كىملىك بېجىرىش كەسپىنى بېجىرىشنى توختىتىشتا ئاپتونوم رايوننىڭ بەلگىلىمىسىگە ئاساسلانغان بولۇپ، پۈتۈن شەھەرنىڭ 2014-يىللىق نوپۇس ئىستاتىستىكىسى خىزمىتىنى ئېلىپ بارىدىكەن.
كونكرېت نوپۇس ۋە كىملىك بېجىرىشنى توختىتىش ۋاقتى: 12-ئاينىڭ 10-كۈنى 24:00 دىن 12-ئاينىڭ 15-كۈنى 24:00 گىچە (بېيجىڭ ۋاقتى)، پۈتۈن شەھەردىكى ھەر دەرىجىلىك جامائەت خەۋپسىزلىكى ئورگانلىرىنىڭ ھەممىسى نوپۇس ۋە كىملىك بېجىرىش كەسپىنى توختىتىدۇ. ھەر قايسى ساقچىخانىلار بۇ مەزگىلدە، ھەر خىل نوپۇسقا ئالدۇرۇشقا ئىلتىماس قىلىش ماتېرىياللىىرنى قوبۇل قىلىشنى توختىتىدۇ، يەنە كىملىك بېجىرىشكە مۇناسىۋەتلىك كەسىپلەرمۇ توختىتىلىدۇ.
ئۈرۈمچى شەھىرى 12-ئاينىڭ 26-كۈنىدىن باشلاپ ھەر خىل نوپۇس ۋە كىملىك كەسپىنى بېجىرىشنى باشلايدۇ. جامائەت خەۋپسىزلىكى ئورگانلىرى بۇ كەسىپلەرنى بېجىرىشنى ۋاقىتلىق توختىتىش مەزگىلىدە پەيدا قىلغان قولايسىزلىقلارنى كەڭ خەلق ئاممىسنىڭ توغرا چۈشىنىشنى ئۈمىد قىلىدۇ.
