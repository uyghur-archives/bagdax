تۆۋەندىكىسى مېنىڭ مىسرانىمدىكى بىر قېرىندىشىمىزغا قايتۇرغان جاۋابىم بولۇپ، «دوكتۇر-ئاشتى» ھەققىدىكى ئەھۋالنى ئانچە ياخشى بىلمەيدىغان قېرىنداشلار ئۈچۈن، مەن ئۇنى بۇ يەرگىمۇ چاپلاپ قويدۇم.

سوئال(@izden):
ئۆتكەندە بىر توردا تۇنجى دوكتۇر ئاشتى دەپ بىر كىشىنى تونۇشتۇرۋاتىتى.سىز يازمىڭىزدا دوكتۇر ئاشتىلىق ئۇنىۋاننى ئېلىپ دەپ قالدىڭىز .بىز ئۇقماي قالدىمىزمۇ ياكى سىز كەمتەرلىك قىلىپ يازمىڭىزدا دوكتۇر ئاشتىلىق ئۇنىۋاننى تىلغا ئالمىدىڭىزمۇ ؟ ئۇيغۇرلاردىكى تۇنجى دوكتۇر ئاشتى زادى كىمۇ ؟

مېنىڭ جاۋابىم:
ئالىي مەكتەپلەر بېرىدىغان، ھەمدە باشقىلار ئېتىراپ قىلىدىغان ئەڭ يۇقىرى دەرىجىلىك ئۇنۋام «دوكتۇرلۇق ئۇنۋانى» بولۇپ، «دوكتۇر-ئاشتى» دېگەن بىر ئۇنۋان ئەمەس.  بىر قىسىم كىشىلەر ھەر خىل سەۋەبلەر تۈپەيلىدىن دوكتۇرلۇقنى ئېلىپ بولغاندىن كېيىن بىر قانچە يىل «دوكتۇر-ئاشتى» بولۇپ، مەخسۇس ئىلمىي تەتقىقاتلار بىلەن شۇغۇللىنىدۇ.  ئامېرىكىدا ياخشىراق ئالىي مەكتەپتە پروفېسسور بولماقچى بولسىڭىز، ئۇلار سىزدىن كەم دېگەندە 2 يىل دوكتۇر-ئاشتى بولغان بولۇشنى تەلەپ قىلىدۇ.  باشقا خىزمەتلەر ئادەتتە دوكتۇر-ئاشتىلىقنى تەلەپ قىلمايدۇ.  ئەمما، دوكتۇر-ئاشتى بولسىڭىز، سىزنىڭ خىزمەت تېپىشتىكى ۋە كېيىنكى خىزمەتلەردىكى رىقابەت كۈچىڭىز كۈچلۈكرەك بولىدۇ.

مەن 1995- ۋە 1996-يىللىرى ئامېرىكىدىكى ئەڭ داڭلىق تەتقىقات ئورۇنلىرىنىڭ بىرى بولغان ساندىيا دۆلەت تەجرىبىخانىسى (Sandia National Laboratory) دا بىر يېرىم يىل دوكتۇر-ئاشتى بولغان (دۇنيادا تۇنجى قېتىم سۇنئىي-ھەمرالارنى ئېتىپ چۈشۈرىۋېتىدىغان لازېر-نۇرى قورالىنى سىناق قىلغان ئورۇن مۇشۇ شۇ).  مېنىڭ شۇ چاغدىكى ئۇستازىم رىك ترېبىنو (Rick Trebino) ئوپتىكىنىڭ بىر ساھەسىدە بىر دۇنياغا داڭلىق ئالىم بولۇپ، ئۇ مېنى ئىنتايىن ياخشى كۆرۈپ كەتكەن ئىدى (ئۇ بىر قېتىم بىر تەتقىقاتچىلار يىغىلغان سورۇندا مۇنداق دېگەن ئىدى: «مەن ئەركىننىڭ قانداق ئىشلەيدىغانلىقىنى بىلمەيمەن.  ئىشقىلىپ مەن بەزىدە ئۇنىڭ ھەر بىر ھەپتە ئىچىدە چىقىرىدىغان تەتقىقات نەتىجىلىرىنى چۈشىنىشتە ئۇنىڭغا يېتىشىپ ماڭالماي قالىمەن»).  ئۇ كېيىن ئامېرىكىدىكى داڭلىق مەكتەپلەرنىڭ بىرى بولغان Georgia Institute of Technology نىڭ فىزىكا فاكۇلتىتىغا مۇدىر بولۇپ يۆتكەلدى، ھەمدە ئۆزى تۇرىۋاتقان ئاتلانتا شەھەردە بىر يۇقىرى تېخنولوگىيىلىك ئوپتىكا شىركىتى قۇردى.  2003-يىلى كۈزدە ئۇ مېنى ئاشۇ شىركەتكە «ئىنژېنېرلىققا مەسئۇل مۇئاۋىن رەئىس» (Vice-President of Engineering)  قىلىپ ئالغان، ھەمدە ماڭا شىركەت بويىچە ئەڭ يۇقىرى مائاش بەرگەن  ئىدى، مەن ماكان ۋە مائاش قاتارلىق ئامىللارنى كۆزدە تۇتۇپ، ئۇ خىزمەتنى قوبۇل قىلماي، نەچچە ئايدىن كېيىن ناسا غا خىزمەتكە كىردىم.  رىك مېنى ھازىرمۇ ئىنتايىن ياخشى كۆرىدۇ.  بىز ھەر بىر-ئىككى يىلدا بىر قېتىم خەلقئارالىق ئىلمىي دوكلات يىغىنلىرىدا كۆرۈشۈپ تۇرىمىز.

ئەسسالامۇ ئەلەيكۇم ھۆرمەتلىك بىلىمخۇمار ئەپەندى ، ئالدى بىلەن ئۆزى ئۈچۈن ۋە باشقىلار ئۈچۈن ئىلىم تەھسىل قىلغان ، باشقىلارغا ئىلىمنىڭ زاكىتىنى بىرىۋاتقان ياخشى كىشىلىرىمىز ئۇلۇغ تەڭرىدىن ياخشىلىق تىلەيمەن . سىزنىڭ مۇشۇ ماقالىڭىز ئارقىلىق ئىلگرىكى سورىغان سۇئاللىرىمغا جاۋاپ بەرگەنلىكىڭىزگە كۆپ تەشەككۈر ئىېيتىمەن . يېزا مائارىپىنى قانداق قىلىپ راۋاجلاندۇرۇش ، ئوقۇغۇچىلارنى دەرسكە قانداق قىلىپ قىزىقتۇرۇش ئەلسۆيەر مائارىپچىلارنىڭ كۆپ ئويلايدىغان خىيالىدۇر . ئوقۇغۇچىلارنى دەرسكە قىزىقتۇرۇشنىڭ ئۇسۇلى (مېنىڭچە) ئىلىم ئۆگىنىپ تەمىنى تىتىغان كىشىلەرنى ئۈلگە قىلىشتۇر . شۇڭلاشقا سىزدەك ۋە قەيسەر مىجىت،بىلگەجان،تاشپولات رۇزى ۋە باشقا بىز بىلمەيدىغان ئىلىم ساھىبلىرىنىڭ تەرجىمىھاللىرىنى ئوقۇغۇچىلارغا بىلدۈرۈش ، ئۆزىمىز كېيىنكى قەدەملەر ھەققىدە پۇختا پىلان تۈزۈپ (ئوقۇغۇچىلىرىمىزنى ئىلھاملاندۇرۇش يۈزىسىدىن )ئوقۇغۇچىلارغا ئۈلگە ئاساسىدا ئوقۇپ بېرىش ، ئوقۇغۇچىلارنىمۇ پىلان تۈزۈشكە يىتەكلەپ ، پىلانلارنى ئىجرا قىلىشقا ھەيدەكچىلىك قىلىش كېرەك دەپ ئويلايمەن . ئالىمرىمىزنىڭ سىنئالغۇلۇق كۆرۈنۈشلىرى بولمىسىمۇ ئۆزى ئىشلەۋاتقان ئورۇننىڭ سۈرەتلىرى بولسا ئازراق بولسىمۇ پايدىسى تىگەتتى .
