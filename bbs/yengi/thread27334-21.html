    ئالدى بىلەن دەۋالاي ، ماۋۇ ئۆركەش ئىسىملىك ئاداش ئادەم تىللىما ، ئىنكاس يازماي تۇرۇپ  قوقتۇم زۇۋانىڭدىن ... ئاغزىڭ بەك يامانكەن ...  بۇ بىرىنىڭ شەخسىي تەجىرىبىخانىسى بولمىغانكىن مەنمۇ ئىككى ئېغىز سۆزلەي...

      مەزكۇر تېمىنى 96 مىنۇت ئوقۇدۇم ، ئاخىرىدا كۆزۈم بەرداشلىق بېرەلمەي ، ئاخىرىغا يۆتكىسەم يەنە شۇ پاراڭ ...

      مەن بىر ساۋاتسىز ئادەم ، ئارانلا ئاقسۇ دارىلمۇئەللىمىن ئومۇمىي پەن كەسپىدە ۋە قەشقەر پىداگوگىكا ئۇنۋېرستېتى باشلانغۇچ تىل ئەدەبىيات كەسپىدە ئوقۇغان . قاسمىكام كىم بۇنىمۇ تۇنىمايمەن ، ياسىنجان ، شەيتانچاق .... ئۆركەشلەر كىم بۇنىمۇ بىلمەيمەن . ئاۋۇ مۇناسىۋەتسىز ئىنكاستىنلا كۆردىڭلار تەبئىي پەن بىلەن بەك خۇشام يوقلۇقىنى ... شۇڭا 29 - تېئورمىنىمۇ بىلمەيمەن ، نىمىشقا تالىشىۋېرىسىلەر ئوقۇۋاتقان ۋاقىتتا خەق ئىسپاتلاپ ئالگېبىرا ، گىئومېتىرىيە ، فىزىكا ، خېمىيە دەرىسلىكىمىزگە چۈشۈرۈپ بەرگەن نەرسىنى يەنە ئىسپاتلاپ چىقىمىز دەپ 1 دەرس سائىتى يېتىشمەي ، بىچارە ئوقۇتقۇچىمىز كىيىنكى سائەتلىك دەرس ۋاقتىنىمۇ ئېلىپ ئارام ئېلىشمۇ يوق ئۆتەتتى ... شۇنىڭغا قاراپ بۇنى ئىجاد قىلغانلار 40-50 يىل ۋاقىت سەرپ قىپتۇ دەپ ئويلايتۇق ، بۇ 29 - تېئورما توغۇرلۇق ئەمدى 3 يىل بوپتۇ (راست گەپ تەپسىلاتىنى ئوقۇمۇدۇم -- ئۆركەش بۇرادەر تىللاپ كەتمەي ئوقۇ-ھە؟!) ، بەزى گەپ - سۆزلىرىڭىزنىڭ ئاساسىمۇ باركەن ، ئەركىن سىدىق ئاكىمىز ، سۈركىلىش كۈچىنى ھېسابلىساق ئېغىش بولىدۇ لىكىن بۇ بەك كىچىك ....دەپتۇ . كىيىن بىر نەچچە تېپىلگەن پۇتبولدىن ئىلھام ئاپسىز ، ئەلۋەتتە باشقىلار ئالمىدىنمۇ ئىلھاملانغانغۇ ؟؟ لىكىن بۇ توغرۇلۇق 1852 - يىلىلا كۆز قاراش شەرھلەنگەن ئىكەن سىز بۇنى تېخىمۇ بېيىتىپسىز ، بۇ روھىڭىز ھەممىمىزدە ھۆرمەت تۇيغۇسى پەيدا قىلدى . نىمىشقا مۇشۇلارنى يىغىنچاقلاپ ئەمەلىي پاكىت مىساللار بىلەن بىر ئىلمىي ماقالە يېزىپ ئېلان قىلمايسىز ، نىيۇتوننىڭ قانۇنىيىتىنى ماھىيەت جەھەتتىن كېڭەيتىش ئۈچۈن چۇقۇم ئالدى بىلەن مەملىكەت ئىچىدە نۇپۇزلۇق ژۇرناللاردا ، ئاندىن چەتئەللەردىكى English تاراتقۇلاردا ئېلان قىلدۇرسىڭىز ، شۇندىلا بۇنىڭ ئەمەلىي قىممىتى بولاتتى . سىزنىڭ ماقالىڭىز ھەقىقەتەن بىر قارىماققا ناھايىتى ئورۇنلۇقكەن ، ئوقۇغانلىرىمنى تەپسىلىي ئوقۇدۇم . لىكىن ئەمەلىي تەجىرىبە جەريانى نەتىجىسى كېرەكتە . سىزنىڭ ئوتتۇرغا قويغان كۆپىنچە پاكىتلىرىڭىز توردىكى ئۇچۇرلاركەن ، ئۇنى ئىنكار قىلغىلى بولمىسىمۇ ، لىكىن 100% ئېتىراپ قىلغىلى بولمايدۇ -دە ؟ بۇ دىگىنىم مەن سىزدىن بەك بىلىمەن دېگەنلىك ئەمەس ، بەلكىم مەنمۇ ۋاقىت چىقىرىپ سىزنىڭ بۇ تېئورمىڭىزنىڭ تەجىرىبىسىنى ئىشلەپ كۆرەمەن ، ئەركىن سىدىق ئەپەندىنىڭ سىزگە بەرگەن جاۋابىنى كۆرمەي تۇرۇپلا مەن چويۇن توپتەك ئېغىر نەرسىلەرنى ئويلىغان ، ئالىمىمىز ئەركىن سىدىق ئەپەندىمۇ ئېغىر نەرسىنى ۋە سىلىق ھالىتى 100% بولسا ....... دېگەندەك قاراشلىرىنىمۇ ئوتتۇرغا قۇيۇپتۇ . دېمەك سىزنىڭ تەتقىقاتىڭىز ئالىمىمىزنىڭمۇ مۇئەييەن دەرىجىدىكى قوللىشىغا ئېرىشىپتۇ شۇنداق بولغانكىن مىنىڭ كۆز قارىشىم يەنىلا مۇشۇ پاكىت ، ئىسپات ، كۆز قاراشلىرىڭىزنى ئېلان قىلىدىغان ۋاقتى كەپ قالغان چېغى دەپ ئويلاپ قالدىم . ھەممەيلەننىڭ ئېتىراپ قىلىشىغا ئېرىشىتىڭىز ئۇ چاغدا ، نىيۇتون قانۇنىيىتىگە ئۆزگەرتىش ، كېڭەيتىش تەكلىپىڭىزمۇ شۇ ۋاقىتتىلا ئۆزلۈكىدىن ھەل بولاتتى . مۇنبەر ھەقىقەتەن مۇنازىرە قىلىدىغان ياخشى جاي لىكىن ، بۇ يەردە ئېلان قىلغان تەتقىقات مېۋىمىزنى ئۆزىمىزمۇ ئېتراپ قىلماي تۇرساق خەق قانداق چۈشىنىدۇ .
    خۇلاسە كالام ، مۇنازىرىدە ئىلمىي بولايلى ، ئىلمىي پاكىت بىلەن سۆزلەيلى ... ھېسىياتقا بېرىلمەيلى ... ئىلىم -پەن ھېسىياتقا بېرىلمەيدۇ ، يەنە كىلىپ سۈكۈتتە تۇرغانلىقنىڭ ھەممىسى كەمتەرلىك ئەمەس . كەمتەرلىك -يېرىم يالغانچىلىق ! سۈكۈتتە تۇرۇۋېلىشمۇ ياخشى ئىش ئەمەس ... ھەممە ئىشقا ئىگىسى ئىگە بولسۇن ، ماۋۇ ئۆركەش بۇرادەر تولا قىزىققانلىق قىلماڭ .... ئاغزىڭىز بەك يامانكەن ، ئىككى ئېغىز گەپ يوقكەن سىزنىڭ پۈتۈن ئىنكاسلىرىڭىزدا ، يا تىللاش ، يا ماختاش !!! سەۋەبى بولۇشى كېرەكتە ھەممە ئىشنىڭ ، مەن ياقتۇرغاننى سەن ياقتۇرمىدىڭ دەپمۇ تىللامدا ؟؟؟ مەنلا ئەمەس ، ھەممە بۇرادەرلەر بىلىدىكەن بىر ئېغىز ئىنكاس باشلىنىشىدا ‹ئۆركەش چىقىدۇ ....› دەيدىكەن دەرھاللا ئېتىلىدىكەنسىز ، ئۇنداق قىلماڭ ... ەمن ئەڭ يېڭى ئەزاكەنمان ، ئازراق كۈن كۈن ئەپ قالاي ، ماخما ؟؟
