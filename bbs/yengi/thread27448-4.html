مېنىڭچە ، بۇ ئىشلار جەزمەن ئوڭشىلىپ كېتىدۇ ، غوجىنىڭ غوجىسى بار دېگەندەك ئەلۋەتتە بۇ ئىشلارنى سۈرۈشتە قىلىدىغان ھەل قىلىدىغان ئورۇن ئاپپاراتلار مېنىڭچە بىر كۈنى بولمىسىمۇ بىر كۈنى ئۆز كۈچ -قۇدرىتىنى چوقۇم روياپقا چىقارغۇسى شۇ ئارقىلىق دۆلەت ھۆكۈمەت ۋە پارتىيىنىڭ خەلق قەلبىدىكى ئوبرازىنى تېخىمۇ يوقۇرى تىكلەنگۈسى . دېمىسىمۇ بۇ يىللاردا رەئىس شى جىنپىڭ تەختكە چىققاندىن ياقى ‹‹ ئىسراپخورلۇققا قارشى تۇرۇش ، پارىخورلۇق ۋە چىرىكلىيككە قارشى تۇرۇپ خەلقنى ئەمىن تاپقۇزۇش ، كەڭ تۈرلۈك ئەل رايى قۇرۇلۇشىنى كەڭ تۈردە قانات يايدۇرۇش ، شۇڭنىڭدەك ساھەلەر بويىچە ھالقىما تەرەققىيات باسقۇچىغا ئىچكىرىلەپ كىرىش ›› قاتارلىق دۆلەتنىڭ ئەمەلىي ئەھۋالىغا ئاساسەن خەلق ئەل رايىنى سىناشنى ئاساسىي مەقسەت قىلغان بىر تۈرلۈك خەلق تۇرمۇشى خەلق ئىگىلىكىنى ئاساسىي ۋاسىتە قىلغان ئىستېراتېگىيىلىك مۇھىم تەدبىرلەرنى قوللىنىش ئارقىلىق يوقۇرى تۆۋەندىكى ھەرقانداق پارىخور شەكىلۋاز رەھبەرلەر ۋە رھبەرلىك قاتلىمىنى قانۇن مەمۇرى ۋاسىتىلەرنىڭ ياردىمى بىلەن جازالاش ئېلىپ بېرىپ ماھىيەتتىن چەتنىگەن شەكىلۋازلار ھەم شەكىلۋازلىقلارنى تۈپ يىلتىزىدىن يوقىتىپ پارتىيە ھۆكۈمەتنىڭ خەلق قەلبىدىكى ئوبرازىنىڭ تېخىمۇ يوقۇرى تىكلىنىشىگە ئاساس سېلىپ بەردى ، قالغانلىرى بولسا مۇناسىۋەتلىك تارماقلار ئارقىلىق قاتتىق تەكشۈرۈلۈۋاتىدۇ ، شۇڭا ئارتۇق ئۈمىتسىزلىك روھىي مەغلۇبىيەتلكتىن باشقا نەرسە ئەمەس . شۇڭلاشقا ‹‹ غوجىنىڭ غوجىسى بار ، ئۆرۈكنىڭ غورىسى ›› دېگەندەك بۇ ئىشلارنىڭمۇ جەزمەن سۈرۈشتۈرۈلىدىغان ۋاقتى سائىتى بولىدۇ . ئەڭ مۇھىمى بىر نەرسىنى ئۇنتۇپ قالماڭلار : ‹‹ ئاللا ھەممىنى بىلىپ تۇرغۇچىدۇر ، ھەم ھېكمەت بىلەن .ئىش قىلغۇچىدۇر ›› .

ھەي نىمىسىنى دەي ،
بۇ ھەقتە مىنىڭ دەردىم مىڭ تاغار ، مىڭ تاغار . . . . .
مەن يېزا بازىرىدە دۇكان ئېچىپ تىجارەت قىلىۋاتقىلى 10 يىل بولدى ،
نۇپۇسۇمنى ، تېرىلغۇ يىرىمنى چوڭلاردىن ئايرىپ چىققىلىمۇ 7 يىل بولدى ،
ئىككىنچى ئىنىم قايسى كۈنى تۇغۇت كۆرسەتكۈچى ( زىپىيو ) ئېلىش ئۈچۈن مەھەللە كومىتىتىغا كىرىپتىكەن بېجىرىپ بەرمەپتۇ ،
سەۋەبى ؛
- ئاكاڭ دادۈيدىكى مەجىلىسكە قاتناشمىدى ، ( بىرھەپتىدە 4 كۈن ئاچىدۇ ، بىر كۈن قاتناشمىسا 50 يۈەن جەرىمانە ئالىدىغان قائىدىسى بار ئىكەن )
-  دادۈيگە كىلىپ جەرىمانىنىڭ ھەممىسىنى تۆلىمىسە بېجىرىپ بەرمەيمىز ، دەپ يولغا ساپتۇ . ئەمىلىيەتتە بىزمۇ ھەپتىدە بىر دىگەندەك يېزىنىڭ يىغىنىغا قاتنىشپ تۇراتتۇق ،
ئىنىم شۇ يەردىلا ئاچچىق يۈتىۋالغىلى تاسلا قاپتۇ ،
- ئۆيى ئايرىم تۇرسا ، مەن بىلەن نىمە ئالاقىسى ، دىسىمۇ ئۇنماپتۇ .  توۋا دەيمەندە ، بۇ ياكى بىراۋ گۇناھ ئۆتكۈزسە ھەممىسىنى جازالايدىغان چىڭ خانداقلىقى بولمىسا ،بۇ نەچچىنچى ئەسىر ، مۇشۇ ئەسىردىمۇ مۇشۇنداق تەڭسىزلىكنى كۆرىۋاتىمىز دەيمىنا .

بىرنەچچە كۈندىن كىيىن ئىنىم بۇ گەپنى ماڭا دىدى .
قاتتىق ئاچچىقىم كەلدى ، يەنىلا نىمە ئامال دەيسىز .
مۇشۇنداق ئاچچىق يۈتۈپلا ئۈتۈدىغان گەپ ئىكەن ،
بىزنىڭ يوللۇق بولسىمۇ كادىرلار بىلەن ئازراقلا تاكاللىشىپ قويساق ، ساقچىلار كىلىپ ھىچبىر گەپ -سۆز يوقلا بىرنەچچە كۈن سولاپ قۇيۇپ جەرىمانە ئېلىۋاتقان . . . . . . . .
