ئۇيغۇر مۇنبەرلىرىنىڭ  ئەخلەت ساندۇقى بىلەن بولغان
پەرقىنى كىم بەلگىلەيدۇ ؟

مۇھەممەدتۇرسۇن ئابدۇرېھىم ئوغلانى

         ئەسسالامۇ ئەلەيكۇم،  مۇنبەر ئەھلى،مەن بىر باشقۇرغۇچى ۋە ئاددىي ئىزدەنگۈچى بولۇش سۈپىتىم بىلەن ئادديغىنە ئويلىغانلىرىمنى تورداشلار بىلەن ئورتاقلىشىشنى ئۈمىد قىلىپ قەلەم تەۋرەتتىم . ماۋزۇنى كۆرۈپلا مۇنبەرنى ئەخلەت ساندۇقى بىلەن سېلىشتۇرۇپتۇ دەپ ئاچچىقىڭىز كەلدىمۇ ؟توغرا ، مۇنداق بىر چوڭ ئىلىم سورۇنىنى ئەخلەت ساندۇقى بىلەن سېلىشتۇرۇش بىر ئەخمەقلىق بولۇشى مومكىن ،  بەلكىم بەزىلەرنىڭ ئاچچىقى كەلگەندۇ ، كەلمىگەنلەرنىڭ ئۆزىنىڭ سەۋەبى باردۇر .
        ئەخلەت ساندۇقىنىڭ رولى قانداق ؟ ئۇنى ئاڭلىغان ئادەم چوقۇم سېسىقچىلىقنى ، مەينەتچىلىكنى ئويلاپ كۆڭلى ئېلىشدۇ ياكى ئەخلەت ساندۇقىنىڭ كەم بولسا بولمايدىغانلىقى، ئۇنىڭغا ئەخلەت سالىدىغانلىقى ھەققىدە مۇنازىرىگە چۈشىدۇ . ئەمەليەتتە مېنىڭ دىگىنىم باشقا بىر نوقتا ، ئەخلەت ساندۇقى  ئۇ رول ئۆتىمەيدۇ ، ئۇنى ئىنسانلار كەشىپ قىلىپ ، ئاشۇنداق رولنى ئۆتەيدىغان قىلىپ ئۇنىڭ خاراكتىرىنى بەلگىلگەن . دىمەك ئىنسانلار شۇنداق بىر ئەخلەت ساندۇقىنى ئۆزىنىڭ ئىھتىياجى بىلەن ياساپ ، ئۇنى ئەخلەت ساندۇقى دەپ ئاتىغان ،شۇڭا بۇنىڭ رولى ھەرگىزمۇ ئەخلەت ئېلىشلا ئەمەس ، ئەگەر زامان تەرەققىي قىلسا رولىمۇ باشقا بولۇپ ئۆزگىرىشى مومكىن .
      مەن نىمىشقا ئەخلەت ساندۇقى توغۇرلۇق توختىلىمەن ؟  
      بىرىنجىدىن ،تېما جەھەتتىن ئالغاندا ، ئەخلەت ساندۇقىغا كىشىلەر ئادەتتە ئىشىلتىپ بولغان ،لازىمى بولمىغان نەرسىلەرنى تاشلايدۇ ، ھېچكىممۇ ئۆزىنىڭ يېڭى ئالغان ئايىغىنى ، بىلەيزۈك ۋە بايلىقلىرىنى ئەخلەت ساندۇقى ئىچىگە تاشلىمايدۇ . ئەگەر موھىم،قىممەتلىك  نەرسىلەرنى ئەخلەت ساندۇقى ئىچىگە تاشلىدى دىسىمۇ ئەھۋال ئالاھىدە بولغاچقا تاشلىغان بولىدۇ . ئۇنداقتا مۇنبەرگە سىز ئەخلەت تاشلىدىڭىزمۇ ؟ مېنىڭ ئالدىنقى بىر مەزگىللىك ئازغىنە  تەجىربەمدىن قارىغاندا ، تېمىلاردا ئىملا خاتالىقىنى دىمەي تۇرساقمۇ ، كۆپ كىشىلەرنىڭ ئەۋەتكەن تېمىلىرى ياكى ئۆزىنىڭ قاملاشمىغان يازمىللىرى ، ياكى خالىغان بېشى بار ئايىغى يوق مەزمۇنسىز بىر ئابزاس گەپ،يەنە شېركەت ئىلانلىرى ، بىر قىسىم ئاساسسىر ئۆسەك سۆزلەرنى ئۆزى ئويلىنىشتىن ئۆتكۈزمەيلا كۆچۈرۈپ ئېلىپ كەلگەن ... قاتارلىق ئشىلار ئېغىر (يېقىندا ئازراق ياخشىلاندى لېكىن يەنە خېلى بار، ) . ئەگەر بۇ توغۇرلۇق گەپ بولغاندا بىر قەلەم ھەققى بەرمەيمىز ، شۇڭلاشقا بىزنىڭ سىزنىڭ ئەسىرىڭىزنى ئۇنداق مۇنداق دەيدىغان ھوقوقىمىز يوق. لېكىن بىزنىڭ مۇنبەرنىڭ ئەسەرلىرىنى سەرخللاشتۇرىدىغان ، مۇنبەرنى گۈللەندۈرىدىغان ھوقوقىمىز ۋە مەسئۇلىيتىمىز بار ، ئەگەر بىزنىڭچە بولغاندا ھەر بىر ئەسەرگە ئاپتورنىڭ جاپاسى ئۈچۈن مۇۋاپىق قەلەم ھەققى بەرسەك بولاتتى ، لېكىن ھازىرچە ئۇيغۇر مۇنبەرلىرىدە ھەرخىل سەۋەبلەر تۈپەيلى بۇنى ئەمەلگە ئاشۇرۇش تەس .گەپنىڭ پوسكاللىسىنى دىسەك ،باشقۇرغۇچىلار نوقتىسىدىن ئېلىپ ئېيتساق مۇنبەرگە نىمە تىما يوللىشىڭىزنى بىز باشقۇرالمايمىز ، پەقەت تەستىقلايمىز ، شۇڭلاشقا قازاندا نىمە بولسا چۆمۈچكە شۇ چىقىدۇ . مۇنبەرنىڭ خاراكتىرى مۇنبەرگە چىقىۋاتقان زىيالىلارنىڭ ساپاسى ، قوشقان تۆھپىسى شۇنداقلا مۇنبەرگە قانداق قارىشى تەرىپىدىن بەلگىلىنىدۇ . ئەگەر سىز مۇنبەرنى بىر بىلىم سورۇنى دەپ يېڭىلىقىڭىزنى ، ئەڭ ياخشى ئەسەرلىرىڭىزنى،يېڭى ئىددىيلىرىڭىزنى  سۇنسىڭىز مۇنبەر ئەخلەت ساندۇقى ئەمەس ئالتۇن ساندۇققا ئايلىنىدۇ ، بولمىسا ئەخلەت ساندۇق بىلەن پەرىقى بولمايدۇ . دىمەك كومپىيوتىرنىڭ ئالدىدا ئولتۇرغان مۇنبەرنىڭ ھەربىر ئەزاسى مۇنبەرنىڭ ئەخلەت ساندۇقى ياكى ئەمەسلىكىنى بەلگىلەيسىز ،ھەربىرىڭىز  باھالىغۇچىلىقتىن ئۇنى ياخىشى كۆرگۈچىگە ، ئۇنى گۈللەندۈرگۈچىگە ،ئۇنى سۆيگۈچىگە ئايلانسىڭىز مۇنبەر تېخىمۇ گۈللىنىدۇ، بۇ خۇددى بالىڭىزنى دۆت دەپ باھالاپ ئۇنى تىللاپ يۈرگەننىڭ ئورنىغا ئۇنىڭ ئۈچۈن ۋاقىت سەرىپ قىلىپ ، ئۇنى ياخشى تەربىيلىسىڭىز بالىنىڭ ئەقىللىق بولۇپ كەتكىنىدەك ئەھۋال . دىمەك مۇنبەرنىڭ ئەخلەت ساندۇقى بىلەن بولغان پەرقىنى سىز ۋە مەن بەلگىلەيمىز ، كىچىككىنە تۆھپىڭىزگە ۋە ئىزدىنىشىڭىزگە سەل قارىماڭ .
      ئىككىنجى ، ئەزا جەھەتتىن ئالغاندا ، مۇنبەرنىڭ ئىچىدە چوڭ -كىچىك ، قېرى ياش ، ئالىم سودىگەر، داڭلىق داڭسىز ھەممە كىشىلەر دىگۈدەك بار ، لېكىن كۆپىنچە كىشىلەرنىڭ ئۇنداق داڭلىق بولۇشى ناتايىن . بىزنىڭ قوشنا مەھەللىنىڭ  "قالماق مەھەللىسى" دەپ نامى بار ئىدى ،مەيلى كىم بولسۇن ئۇ مەھەللىدىكىلەر قالماق دەپ ئاتىلاتتى ، باللىرى ، بايلىرى ، كەمبەغەللىرى ھەممىسى "قالماق" لار ئىدى . كېيىن بىلسەم ئۇلارنىڭ بۇرۇنقى ئاتا بوۋىللىرى گەپ ئاڭلىماي ، باشقىلارغا يولسىزلىق قىلىپ ، ئۆزىنىڭ گېپىنى راست قىلىۋەرگەچكە ، ئۇلار قالماق بولۇپ ئاتىلىپتىكەن . تور بىر مەۋھۇم دۇنيا ،ئۇ سىزنىڭ ئىنكاسلىرىڭىزدىن  سىزنىڭ سالاھىيتىڭىزنى بىلمەيدۇ ، لېكىن سىزنىڭ ساپايىڭىزنى بىلىدۇ ، سىزنىڭ چىرايىڭىزنى بىلمەيدۇ ، لېكىن ئەخلاقىڭىزنى بىلىدۇ ، ئادەممۇ نۆۋىتى كەلگەندە ئۆزىنىڭ ساپاسى ، ئەخلاقى ،بىلىمىگە قاراپ كىشىلەرگە زىيانلىق ئەخلەت ئادەم ۋە ئۆزىنى قەدىرلەيدىغان قەدىرلىك ئادەم  دەپ ئايرىلىدۇ . ئالدىنقىسىدىن كىشىلەر يىرگىنىدۇ ،كىيىنكىسىنى ھۆرمەتلەيدۇ ، گەرچە سىزنى باشقىلار تونۇمىسىمۇ ، ئەگەر سىز مۇنبەرنى " قالماق“،مەھەللىسىگە ئايلاندۇرىدىكەنسىز ، مۇنبەر ئەخلەتخانىغا ئايلىنىدۇ ، ئۇ چاغدا سىزمۇ ئەخلەت ھېسابلىنىسىز . لېكىن قارشىسى بويىچە دىگەندە ، ئۆزىڭىزنى بېسىۋېلىپ ، مۇنبەردىن بىر نەرسە ئۈگۈنۈشنى مەخسەت قىلىپ ، ئىلمىي پىكىر يۈرگۈزسىڭىز ، مۇنبەرنىڭ ئەخلەت ساندۇقىدىن پەرقى بولىدۇ . ئۇ چاغدا ھەممىمىز بۇ مۇنبەرنىڭ بولغانلىقىدىن پەخىرلىنىمىز .
      قىسقىسى ، "ئويغانغان زىمىن"رومانىدىكى مۇڭغۇل سەردارىنىڭ سۆزلىگەن بىر ھېكايىسى بار ، بىر يىلاننىڭ بىر بېشى مىڭ تىنى بارلىقى ، يەنە بىر يىلاننىڭ مىڭ بېشى بىر تىنى بارلقى ، بىر كۈنى بىر يوغان مەخلۇق يولدا باستۇرۇپ كەلگەندە ، ئالدىنقى يىلاننىڭ ئۆزىنى چەتكە ئالغانلىقى ، كىيىنكىسىنىڭ باشلىرى بىر بىرىنىڭ گېپىنى ئاڭلىماي دەسسىلىپ ئۆلگەنلىكى  بايان قىلىنغان . بەلكىم بەزىلەر" يېڭى ئىددىيەلەرنىڭ بولغىنى ياخشى ، پىكىر ئوخشىماسلىقى بولىدىغان ئىش "دىيىشى مومكىن . مېنىڭ كۆزدە تۇتقان نوقتام بۇ ئەمەس ،بەلكى سىزنىڭ نىيتىڭىز ، ئەگەر نىيەت ياخشى بولسا، تور بىر بېشى بار يىلان بولىدۇ ، لېكىن نىيتىڭىز ھەرخىل بولسا ئۇ چاغدا ئاقىۋىتىنى ئۆزىڭىز بىلىسىز  ، نىيتىڭىز ياخشى ، ئەخلاقنى تاشلىۋەتمىگەن ھالدا قانچىلىك ئىلمى ۋە يېڭى پىكىر يۈرگۈزۈپ، ئوخشىمىغان پىكىرلەرنى ئوتتۇرىغا قويۇپ،يېڭى -يېڭى ئىددىيلەرنى مۇنبەرگە سۇنسىڭىز ، ئۇ چاغدا مۇنبەرنىڭمۇ ئەخلەت ساندۇقى بىلەن پەرىقى شۇنچىلىك چوڭ بولىدۇ ، كىممۇ ئەخلەت ساندۇقىنىڭ ئىچدە مۇنازىرە قىلىشنى خالايدۇ ؟
      « ئالتۇن كۈمۈش ئارىلى» قاتارلىق دۇنياۋى ئەسەرنىڭ يازغۇچىسى ، ئەنگىليىلىك داڭلىق يازغۇچى روبىرت لويس ستېۋىنسون(Rorbert Louis Stevenson)   ئاددىي قىلىپ مۇنداق دىگەن ئىدى :« ئۆزەڭنىڭ قانچىلىك مىۋە ئالغىنىڭنى سورىماي ، ئالدى بىلەن ئۆزەڭنىڭ قانچىلىك ئۇرۇق تېرىغانلىقىڭنى سورا » .مەن بەزى قاقشىغان سۆزلەرنى كۆپ كۆرىمەن ، باشقۇرغۇچىلارمۇ پۈتۈن كۈچىمىزنى چىقىرىمىز ،لېكىن ئاخىرى ياخشى ئۇرۇق تېرىپ مۇنبەرنىڭ ئەخلەت ساندۇقىدىن پەرقىنى چوڭايتىش مۇنبەردىكى ھەممە ئەزالاردىن ئايرىلالمايدۇ .
     ئەسلىدە بىر مەزگىل ئىزدىنىپ  ئۇيغۇر مۇنبەرلىرى توغۇرلۇق بىر چوڭ ئىلمى ماقالە يېزىپ چىقماقچى ئىدىم ، ۋاقىت ۋە شارائىت مۇناسىۋىتى بىلەن مۇشۇنچىلىك يېزىپ تۇرۇشقا  مەجبۇرمەن ،يېقىندىن بۇيانقى شۇنداقلا داۋاملىشىپ كىلىۋاتقان ھەرخىل مۇنبەرنىڭ سۈپىتىنى چۈشۈرىدىغان ئۆسەك سۆز ۋە قالايمىقان پىكىرلەرگە چىداپ ئولتۇرالمىدىم . كىيىنچە چوڭراق ئىلمىي ماقالىمۇ  بولۇپ قالار. كۆپچىلىكنىڭ ئىلمى پىكىرلەر بىلەن مۇنازىرە قىلىپ ئۇيغۇر مۇنبەرلىرىنى تېخىمۇ ئىسىل قىلىپ قۇرۇپ چىقىشىنى ئۈمىد قىلىمەن .رەھمەت .  
باغداشتا تېما ئاز بولسا بولسۇن، ساز  بولسۇن ، خەلق رازى بولسۇن !

مۇنبەر تەرەققىياتىنىڭ ئىجابىي يۆلنىشتە مىڭىشى مۇنبەردىكى تىمىلارنىڭ سۈپىتى بىلەنلا ئەمەس ئىنكاسنىڭ سۈپىتى بىلەنمۇ باغلىنىشلىق.
1.باغداش تورى زىيادە تىما قىسچىلىقى سەۋەپلىك  چىكىنىۋاتىدۇ.
يوللىغان تىمىلارنىڭ يىڭىلىنىشى ئىنتايىن تۆۋەن.چۈشكەن ئىنكاسلارنىڭ سەۋيىسى ئىنتايىن سۈپەتسىز.
ھەتتا بىر باشقۇرغۇچى شىئىر يىزىپ يوللاپ قويسام سەۋيىسى بەك يۇقۇرلاپ كىتىپ ياراتماي قالدىمۇ ئۆچۈرۇپ تاشلاپتۇ.
2.باغداش تورى باشقۇرغۇچىللىرىنىڭ تىمىدىكى ئەخلەت ئىنكاسقا بولغان قارىشى ۋە نازارىتى ئىنتايىن تۆۋەن.يىزىپ
ئەخلەت ئىنكاس پەقەت ۋە پەقەت چىكىدىن ئاشقان ھاقارەت سۆزلىرى ۋە قانۇنغا خىلاپ ئىنكاسلار بىلەن چەكلەنسە بولىدۇ.ئەمما باشقۇرغۇچىلار ئوخشىمىغان پىكىردىكىلەرنىڭ ئىنكاسىنى يۇيۇپ ئوچۇقتىن ئوچۇق تىللاشلارغا سورۇن ھازىرلاپ بىرىۋاتىدۇ.
4.چەتئەلدىكىلەرنىڭ ئىنكاس يوللاش ئىشلىرىدا توسالغۇ مەۋجۇت.

يۇقارقىدەك سەۋەپلەر تۈپەيلى باغداشتىكى ئىلغار ئەزالار قىچىپ بۇرۇنقىدەك ئىلمىي سورۇن پايپاق ساتىدىغانلارنىڭ غەيۋەت سورىنىغا،كوچا لۆكچەكلىرىنىڭ داۋخۇلىقىنى بازارغا سالىدىغان ئورنىغا ئوخشاپ قاپتۇ.
بىكەت باشقۇرغۇچىللىرىدىن تەرتىپ ئورنۇتۇپ بىكەتنى ئەسلىگە سىلىشىنى ئۆمۈد قىلىمەن.
كۆڭلۈمدىكىنى ئىتسام بۇرۇنقى باغداشنى سىغىندىم.
