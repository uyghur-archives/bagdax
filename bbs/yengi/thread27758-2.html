     يۇقىرىقى ئىنكاسلاردا ئېيتىلغاندەك، تېما ئىگىسى ئاكىمىز ئازراق ھاياجانلىنىپ قاپتۇ. ئەلى قۇرباننى ياخشى كۆرىمىز، ئۇنىڭ نۇتۇقلىرىدىن ئۆزىمىزگە مەنپەئەت ئالىمىز. شۇنىڭ بىلەن بىللە، ئەلى قۇرباننىڭ خاتالىشىپ قالىدىغانلىقى ياكى مەقسىتى ئالىيجاناپ بولسىمۇ، ئۇسۇلغا دىققەت قىلالماي، ئاڭلىغۇچىلارغا خاتا تەسىر بېرىپ قويىدىغانلىقىنى ئېتىراپ قىلىمىز. شۇڭا، ئەلى قۇرباننىڭ كەتكۈزۈپ قويغان يېرىنى كۆرسىتىپ بەرگەن زىۋىدە خانىمنىمۇ ھۆرمەت قىلىمىز، قوللايمىز. ئەلى قۇربان خاتالىقىنى كۆرسىتىپ بەرگىلى بولمايدىغان بۈگۈنكى چاۋشيەندىكى كىم جۇڭئۇن ياكى ئېلىمىزنىڭ مەدەنىيەت ئىنقىلابى يىللىرىدىكى ماۋ زېدۇڭ ئەمەس. شۇڭا، ئەلى قۇرباننىڭ خاتا قىلىپ قويغان يېرىنى كۆرسىتىپ بەرگەن كىشىنى ۋە ئۇنىڭ تەرىپىنى ئالغانلارنى ئىتقا ئوخشىتىشى ھاياجانلانغانلىقتىن بولغان.
      ئەگەر ئەلى قۇرباننى تەنقىدلىگەننى كۆتۈرەلمەي، بۇنى غاجاشقا، ھەسەتخورلۇققا، بۇنداقلارنى ئىتقا چىقىرىۋەتسەك، خەلقىمىزنىڭ ئومۇميۈزلۈك ھۆرمىتىگە ئېرىشكەن ئابدۇرېھىم ئۆتكۈر، زۇردۇن سابىردەك مۆھتەرەملىرىمىزنىڭ كىتابخانلار سۆيۈپ ئوقۇۋاتقان رومانلىرىدىن خاتالىقلارنى بايقاپ، يازغان ئوبزورلارنى نېمە دەيمىز؟ ئەدناسى، 9 مىليون 600 مىڭ كىۋادرات كىلومېتىر زېمىندىكى بىر مىليارد 300 مىليوندىن ئارتۇق خەلقنى بىر ئۆزى سوراۋاتقان كومپارتىيەنىڭمۇ «تەنقىد ۋە ئۆز-ئۆزىنى تەنقىد قىلىش» پىرىنسىپى بارغۇ؟

   بىز ياكى زىۋىدە خانىم ئەلى قۇرباننى خاتالىق ئۆتكۈزمىسۇن دىمىدۇق ، خاتالىق ئۆتكۈزدى ، زىۋىدە خانىم كۆرسىتىپ بەردى ، بىز ئەلىنىڭ خاتالىقىنى تۇنۇپ كىيىنكى نۇتۇق ھاياتىدا تېخىمۇ قالتىس بۇلۇپ كەتسۇن دەپ پىكىر بەردۇق . بۇنىڭ ھىچقانداق يامىنى يوق ، چۈنكى بىز ھاياتنىڭ ھىندىستاننىڭ كىنوسى ئەمەسلىكىنى ئوبدان بىلىمىز ، بىرىنى سەمىمى تەنقىد قىلىشنى ئۇنىڭغا ياردەم بىرىش دەپ قارايمىز ، قانداقتۇ بايىقى كىنودەك توغرا دىسەك ھەممە يېرىنى توغرا دىمىگەنگە ئوخشاش خاتا دىگىنىمىزمۇ ھەممە يېرىنى ئىنكار قىلىشتىن دىرەك بەرمەيدۇ . مىنىڭچە نۇرغۇن تورداش مۇشۇ نوقتىدا خاتالاشتى . مەرھۇم ئابدىشۈكۈر مۇھەممەتئىمىن بىزدىكى بۇ ئىللەتنى « ئۇ قۇتۇپتىن بۇ قۇتۇپقا ئېغىپ كىتىش » دەپ ئاتىغان ، بۇ قېتىملىق مۇنازىرىدە بىزدىكى بۇ ئىللەت يەتكۈچە ئىپادىلەندى . مەيلى قايسىلا قۇتۇپنىڭ چوققىسىدا تۇرۇپ پىكىر بىرىڭ ، سىز شەرتسىز خاتالاشقان بۇلىسىز ، بىلىڭكى ، ھەرقانداق شەيئىنىڭ ئوتتۇرىسىدا ئارىلىق بۇلىدۇ . تەڭرىم بىلەن ئىنسان ئوتتۇرىسىدا پەيغەمبەر بولغاندەك ..............
    ئاۋۇ يولداشنقا ئوخشاشلارنىڭ گىپىنىمۇ تەنقىدى قۇبۇل قىلىشنى بىلىۋېلىش كىرەك ، بىزدىكى ئۆملۈك ، قوللاش ،قىزغىنلىقنى بەزى مىللەتلەر 10يىللىق ج د پ سىنى  گۆرۆگە قۇيۇپمۇ سېتىۋالالمايدۇ . بۇنىڭدەك گەپلەرنى ئۇيغۇر قۇبۇل قىلماسلىقى كىرەك ، شەخىس قۇبۇل قىلسا ئۇنىڭ يۇلى باشقا . ئېنىقكى بۇنداق گەپلەر ئۆملۈكنى بۇزىدىغان پىتنىلەر بىلەن سۇغۇرۇلغان . باشقىلار بىلەن بىللە يۈرۈپ باققاندىلا ئاندىن ئۇيغۇرنىڭ قانچىلىك ئىسىل خەلق ئىكەنلىكىنى ئاندىن ھىس قىلالايسىز ، يۇقىرىقىدەك ئىللەتلەر بىزدە بار ئىللەت بولماستىن كۆپىنچىسى باشقىلار كاللا قاتۇرۇپ ياسىغان ئىللەتلەردۇر ، بۇنىڭدىن ئېھتىيات قىلىش كىرەك .
  بەلكىم ئەلى قۇرباننى بىز ئەشۇ ئالى مەكتەپ ئوقوغۇچىلىرىدىن بەكرەك قوغدايمىز ، ياخشى كۈرىمىز . بىراق ئەلىنىڭ لىكسىيە ئۇيۇشتۇرغۇچىلارنىڭ بىلەت باھاسىنى 150 ، 200 قىلىۋېلىپ تېخىمۇ كۆپ كىشىنىڭ ئىقتىسادىغا ماس كەلمەي ئاڭلىيالمىغانلىقىنى ئەلۋەتتە تەنقىد قىلىمىز ، ئاۋۇ لىكسىيىدىكى زىۋىدە خانىم تەنقىد قىلغان جايلىرىنى ئوخشاشلا تەنقىد قىلىمىز .
  ئاخىرقى مەقسەت ئەلىنى يىتەكلەش بولسا ڧىرىئودتىن ئېشىپ كىتىشىگە چاۋاك چېلىش ................
