ئېلى قۇرباننىڭ لىكسىيەسى تىڭىرقاش ئىچىدە ياشاۋاتقان ئۇيغۇر جەمىيتىگەچىقىش يۇلى تىپىپ بەرگەندەك ھىسسىياتقا كەلتۈرگىنى بىلەن ھازىرقى رىياللىققا ھەقىقى يۈزلەنگەندە بۇنداق قورۇق مەنتىقىنىڭ كىشىنىڭ شەخسى تۇرمۇشىغا ئازراق تەسىر كۆرسەتكىنىنى ھىسابقا ئالمىغاندا ھازىرقى مۇرەككەپ جەمىيەت ئەھۋالىنى قىلچەئۆزگەرتەلمەيدىغانلىقى  خۇددى ئاغرىق توختۇتۇش مەلھىمىدەك سۈركىگەندە ئازراق ھوزۇر ئاتاقىلغان بىلەن كىسەللىك داۋالاشتا قىلچە ئۈنۈم يوق تېخى كىسەللىكنى يۇشۇرۇپ ئېغىرلاشتۇرىۋېتىشتەك ئاقىۋەتلەرنى چىقىرىشىمۇ مۇمكىن.
............................................................
جاۋاب : ئەلىنىڭ لېكىسيەسىنى تولۇق چۈشىنىپ ئاڭلىغانلىقىڭىزغا گۇمان قىلىۋاتىمەن ، ئەلىنىڭ ئىنكاسلىرىدا جەمئىيەتنى ئۆزگەرتىش دەيدىغان گەپ ئازراقمۇ يوق ، ھەتتا خەلقمۇ مەۋھۈم ئۇقۇم ھالىتىدە دەپ كۆرسىتىلدى  ، ئەلى قايتا قايتا قىلىپ جەمئىيەتنى ئۆزگەرتىش ئەمەس ئۆزىنى ئۆزگەرتىش ، يەنى ئىنسان تاشقى مۇھىت تۈپەيلى ئۆزىنىڭ يوشۇرۇن ئېڭىدا پەيدا قىلىۋالغان چەكلىمىنى بۇزۇپ تاشلاشقا ئىلھام خاراكتىرىدىكى سۆزلەرنى قىلىۋاتىدۇ ، جەمئىيەت توغرىسىدا سۆزلىگەنلەرنىڭ نەتىجىسىنى كۆردۈق ،سىز ھازىرقى ئۇيغۇر جەمئىيتى توغرىلىق نېمىلەر ئاڭلاشنى خالايسىز ،؟مەسىلىنىڭ نېگىزىنى سۆزلەش ئىمكانىيتى بارمۇ ؟) يوق !!!سەۋەبىنى ئۆزىڭىز تېپىۋىلىڭ.( بىز ئۆزىمىزنى ئۆزگەرتىشىمىز كېرەك ، جەمئىيەت دەل مىڭلىغان "مەن" لەردىن تۈزۈلىدۇ ، ئۆزۈڭلارغا ئىشنىڭلار، يوشۇرۇن ئاڭدىكى تۈرلۈك توساقلارنى يېڭىپ چىقىڭلار دەپ ئۆزىنى مىسال قىلدى ، يەنى دادىسىدىن كىچىدە يېگەن بىر شاپىلاق تۈپەيلى "مۇۋەپپەقىيەتتىن قورقۇش " كېسىلىگە مۇناسىۋەتلىك مەزمۇنلار
.........................................................................


دىمەك ئېلى قۇرباندا ئۇيغۇرلارغا ھەقىقى يولباشچى بۇلالىغۇدەك ئۇلارنى ساغلام پىسخىكىغا ئىگە قىلالىغۇدەك ئەمەلى ئىقتىدار  يوق.پەقەت ئەركىن كۈچلۈك جەمىيەتنىڭ پەلسەپىسىنى ئاجىزلىقتا ئۆزىنىڭ مەدىنىيتى تۈگۈل ئۆزىنىڭ كىملىكىگىمۇ ئىگە بۇلالمايۋاتقان مۇشۇنداق بىچارە توپقا ھىكايە قىلىپ سۈزلەپ ئانچە مۇنچە ئىقتىسادى ئۈنۈم ھاسىل قىلماقچى خالاس!!ئېلىنىڭ تېخى مۇنداق تەسىرلىك گىپى بار‹‹سىلەرمىنى تاشلىساڭلارمۇ مەن سىلەرنى تاشلىمايمەن››شۇنداق ئېلىدىن بىزئايرىلساقمۇ بىزگەھېچگەپ ئەمەس ئەمما ئېلى  بىزدىن ئايرىلالمايدۇ.ئاخىرىدا دەيدىغىنىم يۇسۇپ خاس ھاجىپ   ،مەھمۇد قەشقەرى ئۇيغۇرنىڭ مەڭگۈلۈك ئابىدىسى شۇنداقلا پەخرى ئەمما ئېلى قۇربان بولسا قۇربان ئىسىملىك ئادەمنىڭ ئوغلى.
............................................................................
جاۋاب : ئەلى قۇرباننىڭ بىزگە يولباشچى بولۇش مەجبۇرىيتى ھەم ئىمكانىيتى يوق ، ئەلى قۇربان باشقىلارنى ساغلام پىسخكىغا ئىگە قىلالمايدۇ ،پەقەت پىسخىكىمىزدىكى بىر قىسىم توسالغۇلارنى يېڭىپ چىقىشىمىزغا تۈرتكە بولالايدۇ ،ئەلى قۇربان "نامراتلىق جىنايەت " دېگەن قاراشنى ئوتتۇرىغا قويدى ، دېمەك ئۇ پۇل تېپىشنى بىر قابىلىيەت دەپ قارايدۇ  ، نۇرغۇن كىشىلەرنىڭ ئىدىيەسىدىن ھەتتا بۇرۇن مېنىڭمۇ ئىدىيەمدىن ئۆتمىگەن يېرى شۇ ئىدى ، بىراق بۇ قېتىملىق لېكىسيەدىن ھېس قىلدىمكى ئەلى قۇرباننىڭ باشقىلارنىڭ يانچۇقىدىن ئۆزى خالاپ تۇرۇپ شۇ يۈر كوينى سۇغۇرۇپ بەرگۈزىدىغان كارامىتى بار ئىكەن
يۈسۈپ خاس ھاجىپ بىزنىڭ ئابىدىمىز ، لېكىن ئەڭ يۇقىرى ئىشىپ كېتىشىكە بولمايدىغان پەللىمىز  ئەمەس ،كەلگۈسىدە  مۇھەممەد ياكى قۇربان ۋە ياكى تۇرداخۇننىڭ ئوغلى ياكى قىزلىرى  شۇلاردىن ئېشىپ چۈشۈپ كېتىشى تامامەن مۇمكىن !!!!
مانا بۇ مېنىڭ چۈشەنچەمدىكى ئەلىنىڭ بىر قىسىم پىكىرلىرى
ئەلۋەتتە ئەلىنىڭ ھەممە پىكىرلىرىنى بىر بىرلەپ سۆزلەپ ئولتۇرۇش ئىمكانىيتىم يوق
قارشى پىكىر چۈشۈشى نورماللىق ، لېكىن قارشى پىكىر ئەلىنىڭ قىلىپ باقمىغان گەپلىرى ياكى سۆزلىرىنىڭ ئەكسىچە ھالەتتە  دەستەك قىلىپ تۇرۇپ سۆزلەنمىگەي
