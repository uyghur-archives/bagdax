بازىرى كاساتلىشىشقا يۈزلەنگەن تېمىنى نىمە قىلارسىز ئاۋۇتۇپ؟ئاۋۇ ئىلاھىڭىز ئەپتى-بەشىرىسىنى ئاشكارىلاپ بولدى،سىزنىڭچە كىم ئۇتقاندەك قىلىدۇ؟مۇزاينڭ يۈگىرىشى سامانلىققىچە.......ئۇيغۇرلار ھەرقانچە ئازاپ چەككەن بولسىمۇ،لىكىن ئۇنچىلىك دۆت ئەمەس،بىر نەچچە توك-توك ،مەجرۇھ،كاللىسىنى قاغا چۇقىۋالغان ستودىنتنىڭ مۇنبەرلەردىكى چاۋاكلىرىغا مەستخوش بوپ كەتمەڭ،ئەلنىڭ كۆزى ئەللىك،خەلقنىڭ تارازىسى يەنىلا ھەق،سىزدەك چاپانچىلارنىڭكى چاۋىكىنى ئەمەس،كىمنىڭ توغرا كىمنىڭ خاتالىقىنى ئايرىيالايدۇ،ئۇيغۇرلارغا قىلچە ھۆرمەت قىلماي،كۆزى كۆر بوپ كىتىدۇ،چىشى سۇنۇپ كىتىدۇ......مانا بۇ ئاۋۇ چۇقۇنغۇچىڭىزنىڭ سۆزلىرى،بۇ گەپلەرنى كىمگە دەۋاتىدۇ؟ئۇيغۇرغا دەۋاتىدۇ.....ئۇيغۇرغا........يېڭىلىش يېڭىلمەسلىكنى سىزدەك كىسەللەر بەلگۈلەيدىغان دەرىجىگە چۈشۈپ قالمىدى ئۇيغۇرلار تېخى......شۇڭا ۋاقتلىق بۇراندا خۇجايىنىڭىزغا ياخشىچاق بۇلۇمەن دەپ ئاۋارە بولماي،بېرىپ ئايىغىنى مايلاپ بېرىڭە .....ھازىر ئۆزىنىڭ كىسىلىنىمۇ ساقايتالماي دوختۇرخانىدا ئۇكۇل سالغۇزىۋاتقان ئاۋۇ مەبۇدىڭىزنىڭ......
