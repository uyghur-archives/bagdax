    تۇرمۇشتا ئۇچرىغان مەسىلىنى ئوتتۇرغا قۇيۇپ ، تورداشلاردىن مەسلىھەت سوراپ سىز. بۇمۇ سىزنىڭ ھازىرقى ھالىتىڭىزنىڭ ناچارلىقىدىن دىرەك بىرىدۇ.      
     ئەگەر سىزنىڭ مۇقۇم خىزمىتىڭىز، جاپالىق ئىشلەپ ،قىممەت يارتىدىغان ،مۇنداقچە ئىيتقاندا كۆپ پۇل تىپىپ بىر ئۆينى قامدايدىغان ، مۇقۇم ئىشىڭىز بولغان بولسا بۇ يەردە ۋاختىڭىزنى سەرپ قىلىپ ،مۇراجەت نامە يىزىپ مەسلىھەت سوراپ ئولتارماي ،ئۆز ئىشىڭىزنى قىلىۋىرەتتىڭىز. مەسلىھەت سورىغان ئىكەنسىز ،تورداشلارنىڭ نىمە دىيىشى بىلەن كارىڭىز بولماي ،تۆۋەنچىلىك بىلەن ئاڭلاڭ. ئىغىر بولسىمۇ كۆتۈرۈڭ. تىتىقسىز بولسىمۇ چىداڭ.
   مىنىڭمۇ ئىككى ئىغىز بىرنەرسە دىگۈم كەلدى . بىرسى سىز ئالى مەكتەپنى پۈتتۈرگەن بىلەن جەمىيەت مەكتىۋىنى تىخى پۈتتۈرمەپسىز. سىز تۇرمۇشتا خام بالىكەنسىز. نىمىشقا شۇنچە ئوقۇپ ،جاپادا ئىرىشكەن خىزمەتنى تاشلاپ كىلىسىز؟ سىلەر تونۇشقاندا ئىشلەيدىغان بولغىيتىڭىز؟ خىزمىتىڭىزنى تاشلىغانغا ئايالىڭىز نىمە دىگەن؟
    يەنە بىرسى ئايال كىشى ئارقا ئارقىدىن قۇرساق كۈتۈرۈپ (سىزنىڭ پۇشتىڭىزنى)چۈشۈپ كەتسە ،قانداق ھالەتتە بولىدىغانلىقىنى بىلمەمسىز؟مەيلى ھىسسىيات جەھەتتە بولسۇن، سالامەتلىك جەھەتتە بولسۇن، روھى جەھەتتە بولسۇن ئىغىر بىسىمغا ئۇچرايدۇ؟بۇنىمۇ چۈشەنمەمسىز؟
    مىنىڭ سىزگە بىرىدىغان تەكلىۋىم ئۆزىڭىزنى كۈچلەندۈرۈڭ. ئائىلىدە ئايالىڭىزغا گەپ ئۇختۇرالماي نامەرتلىك قىلغان ئادەم ،ئايالىڭىز ئالدىدا قانچىلىك ھۆرمەتكە سازاۋەر بولىسىز؟ ھىلىمۇ بالىڭىز يوق ئىكەن بالىڭىز بولسىمۇ سىزنى كۆزگە ئىلمايدۇ.بىرەر ئىشنىڭ پىشىنى تۇتۇپ مۇۋاپىقىيەت قازىنىڭ. كىشىنىڭ ھۆرمىتىگە ئىرىشى ،قايىللىقىنى كەلتۈرۈش ئۈچۈن باشقىلار بىر ھەسسە كۈچ چىقارسا ، سىز ئون ھەسسە كۈچ چىقىرىڭ. پەقەت پاكىتلا ھەممىنى قايىل قىلىدۇ. مۈشۈكمۇ ئاپتاپقا بىكارغا چەقمايدۇ. ھازىرقى ئاياللار چىرايلىق يالغان ياۋىداق گەپكە ئىرىپ كەتمەيدۇ، ھەم ھەيۋە ،ئۇرۇش تىللاشلاردىنمۇ قورقۇپ كەتمەيدۇ.
    چوڭلانىڭ يەنىلا ئۆيەڭلارنى تۇتۇڭلا دىگىنى بويىچە، ئۆي تۇتۇش مەخسىدىڭىز بولسا ، ئايالىڭىزنى ئەمىلىيەتتە قايىل قىلىڭ. كۆڭلىنى چىگىپ قۇيۇپسىز. كۆڭلىنى ئىلىڭ . ياكى مەلۇم مەزگىل ۋاقىت بىرىپ ،ئۆزىڭىزنى ئۆزگەرتىڭ.

كۆپ جاپا تارتىپسىزبۇرادەر ،مەنمۇ ئىككى  توي قىلغان ،بىرىنچى ئايالدىن بىر بالام بار ،يەنە كىلىپ ئۇنىڭ بىلەن بىر ئ‍وروندا خىزمەت قىلىمىز … مەن ئۇنىڭ بىلەن چوقۇم تۇرمۇشنى داۋاملاشتۇرىمەن دەپ بەك جىق جاپا مۇشەققەت تارتقان ،بالامنى تىرىك يىتىم قىلمايمەن دەپ قىلمىغىنىم قامىغان، ئۇنىڭغا يول قويوش،ئ‍ارقىمۇ ئ‍ارقا يول قويوش،كۆڭۈل بۆلۈش ،چوڭ كىچىك ئۇرۇق تۇققانلار ۋە تونوش بىلىشلەرنىڭ ھەممىسى دېگۈدەك نەسىھەتمۇ قىلغان ئ‍ەمما ئوخشىمايدىغان يېرى ئۇنىڭ ئ‍اجراشقۇسى يوق، ئلىكىن  باشقا جەھەتتىن سىزنىڭ ئ‍ايالىڭىز بىلەن بىلەن جىق پەرقلىق ئ‍ەمەس ،غۇرۇرۇمغا تىگىدىغان گەپ قىلمايدۇ ئەمما بۇقىنى جىمدىپ ئ‍ۆلتۈرىدىغانلار ىقاتارىدىن ،ئ‍ۆي ئىشى دىگەننى خىيالىغىمۇ كەلتۈرمەيدۇ،بالىمىز بىر ياشقا كىرگۈچە ئ‍ۆيدە تۇرغان ۋاقتىمۇ تايىنلىق ،قەدەمدە بىر كىتىۋالغان ،بىر قاچا ئ‍ەتكەن تامىقىمۇ يوق شۇ بالام بىر ياشقا كىرىپ بولغىچە توي قىلغاندىن تارتىپ بار تىرىشچانلىقنى كۆرسەتتىم بار ئ‍امالنى قىلدىم لىكىن ئ‍ۇ يا ئ‍اجراشمايدۇ يا ئ‍ۆي تۇتمايدۇ …ئ‍اخىر ئۆزۈم تەلەپ قىلىپ ئ‍اجراشتىم ،بالىنى دەسلەپتە تالاشتىم ،كىيىن بوپتۇ دەپ ئانىسى باقىدىغانغا قوشولدۇم
