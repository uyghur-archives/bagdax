تىمىڭىزنىڭ بېشىدا ؛ ئەسسالامۇئەلەيكۇم !  باغداش مۇنبىرىدىكى ئاكا-ئاچاشلار،قېرىنداشلار! دەپ باغداش مۇنبىردىكىلەردىن ئەھۋال سوراپ ؛ شۇڭا ئەلكۈيى مۇنبىرىدىكى قېرىنداشلارنىڭ ماڭا يول كۆرسىتىشىنى ئۆتۈنۈپ مۇنبەرگە بۇ تىمىنى يوللۇدۇم دەپ ئۆزىڭىزنىڭمۇ بىر ئىشلارغا تازا ئەستائىدىل مۇئامىلە قىلمايدىغان زىيالى ئىكەنلىكىڭىزنى ئاشكارلاپ قويۇپسىز . گەپنىڭ قىسقىسى ئايالىڭىز نىىمشقا سىزنىڭ غورۇرىڭىزغا تىگدىغان گەپلەرنى قىلىدۇ ؟ سىزدە خوتۇنىڭىزنىڭ گەپلىرىى ئاسانلا غورۇرىڭىزغا تىگدىغان ئاجىزلىق ھالقا زادى نەدە تازا بىر ئويلۇنۇپ بېقىڭ . ئەركىشى دىگەن سىرىتتا قىلدىغان ئىشمىز پەقەت تۈگەپ كەتكەندىمۇ ئۆيدە خوتۇن كىشىنىڭ ئالدىدا ئۇزۇن ئولتۇرۇپ قالساق بولمايدۇ ئىنىم . ئۆزىگە يېقىن ئەمەس كۆڭلىگە يېقىن تۇرۇشقا تىرشىڭ . يېنىدا بولمىسىڭىز ئەسلەپ تۇردىغان بولسۇن ، ۋاختىدا ئۆيگە قايتالمىسىڭىز تىلفۇن قىلىپ ئەھۋال سورايدىغان بولسۇن . كۈندىزى يىراق تۇرۇڭ كىچىسى ئالاھىدە يېقىن تۇرۇڭ . گەپلىرىڭىزنى  كاللامنىڭ يىتشىچە تەھلىل قىلسام ئائىلەڭلەردە ئىختىزادى قىيىنچىلىق دەيدىغان نەرسە مەۋجۈت ئەمەسكەن . دىمەك خوتۇنىڭىزنىڭ يىمەك- ئىچمەك كىيىم -كىچەك دىگەندەك ئىتىياجلىرى تولۇق كاپالەتكە ئىگە بوپتۇ. ئەمما  ئەڭ يامىنى ئىناق ئائىلنىڭ ئاساسلىق ئۇلى بولغان ھىسيات (ئەر -ئاياللىق تۇرمۇش ) ئىھتىياجىنى قاندۇرالماپسىز . ئايال كىشى دىگەننىڭ پىسخولوگىيسى قاتمال بولدۇ . ھىسيات - كۈينۈش . سىردىدىش دىگەندەك ئۆز جۈپتىدىن بەرھرىمان بولۇشقا تىگىشلىك ئىھتىياجلىرىنى قاندۇرالمىغاندا ئاسانلا تىركىدىغان ،چېچىلغاق بوپقالغاندىن سىرىت . يىلنىش ، ئۈتۈنۈش ، يول قويۇش دىگەنلەر كار قىلمايدىغان ئاشۇنداق جىدەلخور بوپ قالىدۇ . خۇلاسا كاللام سىز مۇشۇ ئەڭ چوڭ يەردىن كەتكۈزۈپ قويۇپسىز . كۆڭلىنى چىگپ قويۇپسىز . سىزگە بىردىغان مەسلەھەتىم ؛ئايالىڭىزنى يەنە بىر قېتىم قېشڭىزغا قايتۇرۇپ كىلىشكە كۆزىڭىز يەتسە ، قايتۇرۇپ كىلىپ بىر مەزگىل ئۆيدە ئارام ئېلىشقا ئۈندەپ سىرت بىلەن بولغان ئالاقىسنى ئۈزۈپ تاشلاپ كۈندىزى قەتئى كۆزگە كىرۋالماي . كۈندۈزلۈك خىزمەتلىرىڭىزنى تۈگۈتۈپ قېشىغا كىرگەندە كۆپرەك پىكىر ئالماشتۇرۇپ بىرىڭ .گەپ سۆزدە  ئىش ھەركەتتە بوش تۇرماي ئايالىڭىز ئالدىدىكى ئەرلىك ئورنىڭىزنى دەرھال ئەسلىگە كەلتۈرۈڭ . ئەمدى قېشىمغا كەلمەيدىكەن دىسىڭىز بۇ قېتىمقى نىكاھنى بىر قېتىملىق ئاچچىق ساۋاق ئورنىدا ئۆتكىزۋىتىپ ئۆزىڭىزنى ۋە ئايالىڭىزنى قىينىماي ۋاختىدا بىر- بىرىڭلارغا ئورۇن بوشتۇپ بىرىڭلار . شۇنىڭ بىلەن يەنە بىر تۇرمۇشلۇق بولغىچە يېرىم يىلدىن بىر يىلغىىچە ۋاقىت يالغۇزلۇقتا ئۆتۈپ كىتىدۇ شۇ چاغدىلا مەسلەھەت سورمايمۇ كىيىنكى تۇرمۇشقا داغدام قەدەم باسقىدەك بوپ كىتسىلەر . چۈنكى نىكاھتىن كىيىنكى يالغۇزلۇقتا مەيلى ئەر بولسۇن ياكى ئايال بولسۇن بىر -بىرىنىڭ قەدرى ئۆتلىدۇ . ئائىلنىڭ قەدرى ئۆتلىدۇ ئەنە شۇندىلا  تۇرمۇش ئىشىغا قانداق مۇئاملە قىلىشنى . خوتۇن ئەرگە قانداق مۇئامىلە قىلىشنى . ئەر ئائىلگە قانداق مەسئۇل بولۇشنى ، بەخىتلىك بىر ئائىلنىڭ قۇرغۇچىللىرى بولغان ئەر -ئايال بۇ ئىككى باش پۇرسۇناژنىڭ ئائىلدىكى ئورنى ، ئۆتەشكە تىگىشلىك مەجبۇريىتى نىملەرنى چوقۇم قىلىش نىملەرنى قەتئى قىلماسلىق دىگەندەك مۇھىم نوختىلارنى
ھىس قىلىپ يىتسىلەر .

   ئەسسالامۇ ئەلەيكۇم!
   ئائىلىنى ساقلاپ قېلىش ئۈچۈن ئىزدەنگەن قەدىمىڭىز ئۈچۈن ئەجرىڭىز زايا كەتمىسۇن،سىز چوڭلارنىڭ ياردىمىدە تۇققانلارغا تۇي قىلغان سەۋەپلىك،ئالدىراپ ئاجراشماڭ،تۇققانلار ياتلىشىپ جىدەل ئوينايدىغان كۈنگە قالسىڭىز، ئاتا-ئانىڭىزغا تېخىمۇ يۈز كېلەلمەيسىز.سىز تورداشلارنىڭ‹‹ كەسپ قىلىش ۋە ئۆزنى كۈچەيتىش›› توغرىسىدىكى نەسىھەتلىرىنى ئەمەلگە ئاشۇرۇڭ . ئەگەر ئايالىڭىز سىزنىڭ ئەركەكلىك غۇرۇرىڭىزغا تىگىدىغان گەپلەرنى قىلمىغان بولسا ، باشقا سۆزلىرىنى كۆچۈرۈڭ ، چۈنكى ، كۆچۈرۈش ئارقىلىق كۆچۈرۈمگە ئېرىشكىلى بولىدۇ . سىز سۆزىڭىزدە بەزى ئىللەتلەرنى تۈزەتكەنلىكىڭىزنى دەپسىز ، بۇمۇ ئائىلىنى قوراشتۇرۇشتىكى ياخشى قەدەم.  ئادەم ئۆزىنىڭ يۈزىدىكى قارىنى  تولۇق كۆرەلمەيدۇ ، يەنە  بەزى  كەمچىلىكلىرىڭىز بولىشى مۇمكىن . سىز  ئەڭ  ياخشىسى سىلەرنىمۇ  بىلىدىغان ، ئۇلارنىمۇ بىلىدىغان يۇرتتا ئىناۋىتى يوقۇرى ، ئىشنى  ئادىل بىر تەرەپ قىلىدىغان كىشىدىن  ئىككىنى  ئايالىڭىزنىڭ ئۆيىگە ئەۋەتىڭ ، ئايالىڭىزنىڭ ئۆيىدىكىلەرنىڭ ياراشتۇرغۇسى بولسا، سىزدىن نىمىلەرنى كۈتىدىغانلىقىنى جىكىلەپ ، ئەلچىدىن ئۈمۈدلىك سۆزلەرنى ئەۋەتىدۇ .قالغان كەمچىلىكىڭىزنى ئەلچىلەردىن ئاڭلاڭ ، ئۆزىڭىزنى تۈزەڭ، ئاچچىقنىڭ كەينىگە كېرىپ ئىشنى ئوڭشىغىلى بولمايدۇ.

       ئەسسالامۇ ئەلەيكۇم!
       نىكاھ _ ۋەدىدۇر.
      ئۇ پىشىپ يىتىلگەن بىر ئەرنىڭ نىكاھىغا ئالغان ئايالنى جىسمانى جەھەتتىن قوغداش؛ يات جىنىسلارنىڭ تاجاۋۇزىدىن قوغداش، تۇرمۇش لاۋازىمەتلىرىنى تەل قىلىپ بېرىش ۋەدىسىدۇر؛ ئايالىنى ۋە پەرزەنتلىرىنى  ئىزچىل مەنىۋىي جەھەتتىن تەربىيلەپ تۇرۇش ۋەدىسىدۇر؛ ئايالىدىن باشقا قىز _ ئاياللار بىلەن غەيرى مۇناسىۋەتتە بولماسلىق ۋەدىسىدۇر؛ چىرايلىقچە ئۆي تۇتۇش ۋەدىسىدۇر؛  ئايالىدىن سەھۋەنلىك، كەمچىلىك ، خاتالىق سادىر بولغاندا،قول تەككۈزمەي نەسىىھەت قىلىش، جايلىرىنى ۋاقىتلىق ئايرىش، قاتتىقراق تەنبىھ بېرىش، ياكى ئەر_ ئايال تەرەپتىن  ئىككى ئادىل گۇۋاھچىنى تەكلىپ قىلىپ ياراشتۇرۇپ قويۇش ياكى ئاجراشتۇرۇش تەكلىۋىنى بېرىش، زۇلۇم قىلماسلىق ۋەدىسىدۇر؛نىكاھلانغان ئايالنىڭ ئېرىنىڭ كورپىسىگە يات جىنىسلىقلارنى يولاتماسلىق، ئائىلىنىڭ  مال_ مۈلكىنى قوغداش، توغرا يۇسۇندا ئىرىدىن سوراپ سەرپ قىلىش، ئۆزىنىڭ قىلغان بارلىق ئىشلىرىدىن ئېرىنى خەۋەردار قىلىش ۋەدىسىدۇر؛ ئەر_ئايالنىڭ ئورتاق چوڭلار ۋە ئۇرۇغ _ تۇققانلار،خۇلۇم_خوشنىلار بىلەن ئىناق ئۆتۈش ۋەدىسىدۇر؛ ئىشلارنى كېڭىشىپ بىر تەرەپ قىلىش ۋەدىسىدۇر.   
      جامائەت گۇۋاھلىقى _ نىكاھنىڭ مۇستەھكەم ئاساسى.
      ئەر ياكى ئايال نىكاھىنى تىزىملاتقاندىن كىيىن ، ئۇرۇغ_تۇققان خۇلۇم _خوشنىلارنىڭ گۇۋاھلىقىدا توي قىلسا،بۇ ئائىلە بىلەن جامائەتنىڭ ۋەجەمىيەتنىڭ ئىناقلىقىغا  پايدىسى بولۇپلا قالماستىن ، ئۆزىنىڭ ھەمىشە شۇ گۇۋاھ بولغۇچىلارنىڭ كۆزىچە ياشايدىغانلىقى ئۇلار ئۈچۈن دىققەت بەلگىسىدۇر؛ ئەكسىچە بولغاندا بۇ ئۈنۈمگە يەتمەك مۈشكۈل.
