       ئەسسالامۇئەلەيكۇم !  باغداش مۇنبىرىدىكى ئاكا-ئاچاشلار،قېرىنداشلار!
      مەن تۇرمۇشۇمدا ئايالىم ئىككىمىز ئارىسىدا يۈز بەرگەن كۆڭۈلسىزلىك تۈپەيلىدىن باغداش مۇنبىرىدىكى ھەر بىر تورداشلاردىن مەسلىھەت سوراش ئۈچۈن بۇ تىمىنى يوللىدىم . تۇرمۇشنىڭ ئىسسىق-سوغۇقىنى بېشىدىن ئۆتكۈزگەن ئاكا-ئاچاشلارنىڭ ماڭا مەسلىھەت بىرىشىنى تۆۋەنچىلىك بىلەن ئۈمۈت قىلىمەن.
      ئالدى بىلەن ئۆزۈمنىڭ ئەھۋالىنى بايان قىلىپ ئۆتسەم.
      مەن 2007-يىلى 7-ئايدا قەشقەر پىداگوگىكا ئىنىستىتۇتىدا ئوقۇش پۈتتۈرۈپ  شۇ يىلى 9-ئايدا قەشقەر ۋىلايىتىنىڭ ئوقۇتقۇچى تولۇقلاش ئىمتاھانىغا قاتنىشىپ ئىمتاھاندىن ئۆتكەندىن كىيىن 10-ئاينىڭ ئاخىرى مارالبېشى ناھىيسىنىڭ ئاقساقمارال يىزىسىدىكى بىر توققۇز يىللىق تۈزۈمدىكى مەكتەپكە تەقسىم قىلىندىم . ئۇ يەرگە بارغاندىن كىيىن 2013-يىلى 11-ئايغىچە 6 يىل ئوقۇتقۇچىلىق خىزمىتى بىلەن شۇغۇللاندىم . يېشىمنىڭ چوڭىيشىغا ئەگىشىپ ئاتا-ئانام تۇرمۇشلۇق بولۇشۇم توغرىسىدا كۆپ نەسىھەتلەرنى قىلدى . 2012-يىلى ئۆيدىكىلەرنىڭ تونۇتۇرۇشى بىلەن ئانام تەرەپتىن ئۇرۇق-تۇققانچىلىق مۇناسىۋىتى بار بىر تۇققىنىمىزنىڭ قىزى بىلەن تونۇشۇپ ، بىر يىلدىن كۆپرەك ۋاقىت ئىزدىشىپ ئۆتتۈق . بۇ ئارلىقتا ھەر ئىككىمىزدە بىر-بىرىمىزنىڭ مىجەز-خاراكتىرىگە قارىتا ئاز بولمىغان چۈشەنچىلەر پەيدا بولدى . ئاتا-ئانىلىرىمىزنىڭ مەسلىھەتى ۋە ئىككىمىزنىڭ پىكىردە بىردەكلىككە كىلىشى بىلەن 2013-يىلى 9-ئايدا توي قىلدۇق . ئايالممۇ ئالىي مەكتەپنى پۈتتۈرگەن بولغاچقا ئۆز تىرىشچانلىقى بىلەن قەشقەر شەھەر ئىچىدە مەلۇم بىر ئورۇندا خىزمەت قىلدى . تويدىن كىيىن مىنىڭ خىزمىتىم مارالبېشىدا ، ئايالىمنىڭ خىزمىتى قەشقەر شەھەر ئىچىدە بولغاچقا تۇرمۇشىمىزغا قولايسىزلىقلارنى ئېلىپ كەلدى .
       شۇ سەۋەپتىن ئائىلە ۋە تۇرمۇش ئۈچۈن مەن خىزمىتىمدىن ئىستىپا بىرىپ يېنىپ كەلدىم ۋە بىر مەزگىل ئۆيدە بولدۇم . 2014-يىلى 3-ئايلاردا تۇنجى بالىمىز 2 ئايلىق بولغاندا ئايالىمنىڭ بويىدىن ئاجراپ كەتتى . ئارلقتا بىر مەزگىل ئۆتكەندىن كىيىن مەنمۇ قەشقەر شەھەر ئىچىدىكى بىر شىركەتتە ئىشلەشكە باشلىدىم . كەچتە ئىشتىن چۈشۈپ ئۆيگە كەلگەندىن كىيىن كومپىيۇتېردا ئانچە-مۇنچە ئويۇن ئوينىغاچ لاتارىيە سېتىۋالدىم . مىنىڭ لاتارىيە سېتىۋىلىشىمغا ئايالىم قارشى ئىدى . مۇشۇ تەرىقىدە ئارلىقتا بىر مەزگىل ئۆتكەندىن كىيىن 2014-يىلى 11-ئاينىڭ ئاخىرى ئايالىم دوختۇرخانىغا سالامەتلىكىنى تەكشۈرتكىلى بارسا ، دوختۇرلار تەكشۈرۈش ئارقىلىق ئايالىمنىڭ يەنە بويىدىن ئاجراپ كەتكەنلىكىنى ئېيتىپتۇ . شۇ ئىشتىن كىيىن ئايالىم ئاجرىشىش تەلىپىنى ئوتتۇرىغا قويدى . شۇنىڭدىن باشلاپ بۇرۇنقىغا ئوخشاش كومپىيۇتېردا ئويۇن ئوينايدىغان ، لاتارىيە سېتىۋالىدىغان ئىشلارنى توختاتتىم . ئايالىمنىڭ كەيپىياتى ياخشى بولمىغاچقا ئىشتىن چۈشۈپ ئۆيگە كىرسىلا مىنىڭ غورۇرۇمغا تىگىدىغان ، مىنى ھاقارەتلەيدىغان سۆزلەرنى كۆپ دەيدىغان بولۇۋالدى . شۇنداق بولسىمۇ ئايالىمنىڭ كەيپىياتىنىڭ ياخشى ئەمەسلىكىنى ئويلاپ شۇنداقلا ئۆزۈمنىڭ ئاجرىشىش خىيالىم بولمىغاچقا ، ئارلىقتا بىر مەزگىل ئۆتسە كەيپىياتى ياخشا بولۇپ كىتەر دىگەن ئۈمۈتتە سەۋىر تاقەت قىلىپ كەلدىم . لىكىن 12-ئاينىڭ ئوتتۇرلىرىدا ئايالىم ئاتا-ئانىسىنىڭ ئۆيىگە چىقىپ كىرىدىغانلىقىنى ئېيتىپ چىقىپ كەتكەنچە 2 ھەپتىدىن كۆپرەك ئاتا-ئانىسىنىڭ ئۆيىدە ئولتۇرىۋالدى . ( ئايالىمنىڭ خىزمەت ئورنى بىلەن ئۆيىمىز چوڭ يولنىڭ ئىككى تەرىپىدە قارىشىپلا تۇرىدۇ ) يېڭى يىل كىرگەندىن كىيىن ئايالىمنىڭ ئۆيىگە چىقىپ چوڭلاردىن ھەم ئايالىمدىن ئەپۇ سوراپ ئۆيگە ياندۇرۇپ كىردىم . ئايالىمنى ياندۇرۇپ كىرگەندىن كىيىن ئارىمىزدىكى سوغۇق مۇناسىۋەتنى تۈگۈتۈش ، ئائىلىمىزنى ساقلاپ قېلىش ئۈچۈن ئايالىمغا ياخشى مۇئامىلە قىلىپ كۆپ تىرىشچانىلق كۆرسەتكەن بولساممۇ ، ئايالىم داۋاملىق مىنى ھاقارەتلەيدىغان سۆزلەرنى قىلىۋەردى . شۇ سەۋەپلىك ئايالىمنىڭ دىگەن بىر ئېغىز سۆزىگە چىداپ تۇرالماي ئايالىمغا قول تەككۈزۈپ سالدىم .
شۇ ئىشتىن كىيىن ئايالىم ئەمدى سەن بىلەن ھەرگىز ئۆي تۇتمايمەن ، ئاجراشقىنىم ئاجراشقان دەپ ئۆيدىن چىقىپ كەتكەنچە ئۆيگە كىرمىدى .
    نىكاھتىن ئاجرىشىش رەسمىيتىنى بىجىردىغان ئورۇنغا بېرىپ ، ئەگەر نىكاھتىن ئاجراشماقچى بولسا ، ئەر-ئايال ئىككى تەرەپتىن بىرسى ئاجرىشىشقا قوشۇلمىسا يەنىلا نىكاھتىن ئاجرىشىش رەسمىيتىنى بىجىرگىلى بولامدۇ ؟ دەپ سورىسام چوقۇم ئەر-ئايال ئىككى تەرەپ قوشۇلغاندىلا ئاندىن نىكاھتىن ئاجرىشىش رەسمىيتىنى بىجىرگىلى بولىدۇ ، بىر تەرەپ قوشۇلمىسا ئاجرىشىش رەسمىيتىنى بىجىرگىلى بولمايدۇ دىدى . بۇ ئەھۋالنى ئايالىمغا تېلفۇندا دىسەم ئۇنداق بولسا سوتقا ئەرز قىلىمەن دەيدۇ .
    مىنىڭ ئايالىم بىلەن ئاجرىشىپ ئائىلىمىزنى بۇزۇش مەخسىتىم يوق ، ئايالىمنىڭ ئۆيىدىكىللىرىمۇ ئايالىمغا ئاجراشماسلىق توغرىسىدا كۆپ نەسىھەت قىلغان بولسىمۇ ھېچكىمنىڭ نەسىھىتىگە قۇلاق سالماپتۇ .
    تۇرمۇش تەجرىبەمنىڭ كەمچىل بولۇشى سەۋەبىدىن بۇ مەسلىنى قانداق ھەل قىلىشنى بىلەلمىدىم . شۇڭا ئەلكۈيى مۇنبىرىدىكى قېرىنداشلارنىڭ ماڭا يول كۆرسىتىشىنى ئۆتۈنۈپ مۇنبەرگە بۇ تىمىنى يوللىدىم . مۇنبەردىكى بارلىق ئاكا-ئاچا ، ئىنى-سىڭىل قېرىنداشلارنىڭ پايدىلىق مەسلىھەتى بولسا يول كۆرسىتىشىنى ئۈمۈت قىلىمەن .
كۆپنىڭ ئەقلى كۆپ بولىدۇ دىگەندەك ، مۇنبەردىكى قېرىنداشلارنىڭ ياخشى نىيەت بىلەن قىلىنغان نەسىھەتلىرىنىڭ ياردىمىدە تۇرمۇشۇمدىكى كۆڭۈلسىزلىكلەرنى ھەل قىلىپ كىتەلىسەم ، قېرىنداشلاردىن تولىمۇ مىننەتدار بولغان بولاتتىم .

بىزنىڭ قوشنىمىز ئۆمرىدە ئالتە قىتىم ئاجرىشىپ ، ھەر قىتىم يېنىشقاندا بىر بالىلىق بولۇپ دېگەندەك ئالتە بالىلىق بولغاندا ئاجراشتى. بالا دېگەن رىسقى بىلەن تۇغۇلغاچقا چوڭ بولىدىكەن. ھازىر بالىلىرىنىڭ ئالدى بىيجىڭدا ئىشلەۋاتىدۇ. ئەر ئايال ئاللىبۇرۇن ئاجراشتى. ئەرمۇ ئۆز يولىنى تىپىپ توي قىلۋالدى. ئايال بولسا ئۆيدە. .. سىزگە سەمىمىي تەكلىپ- ئىنسان دۇنياغا پەقەت ۋە پەقەت بىرلا قىتىم كېلىدۇ. ئادەم ئۆز خوشاللىقىنى قانداقتۇر ئەھمىيەتسىز ئىشلار ئۈچۈن قۇربان قىلىۋەتمەسلىكى كېرەك . ئايالىڭىز سىزنىڭ غۇرۇرىڭىزغا تېگىدىغان سۆزلەرنى قىلىۋاتقان بولسا سىزنىڭ ئايالىڭىز ئالدىدا ئەرلىك سالايىتىڭىز قالماپتۇ. ئەپلەپ -سەپلەپ ئۆي تۇتقان تەغدىردىمۇ ئەجىلىڭىزدىن بۇرۇن ئۆلىسىز. «ئادەم بېشى تاشتەك قاتتىق تامغا تاققىدە تەگمىگۈچە دۇنيادا توساق ۋە توسالغۇ بارلىقىنى بىلمەيدۇ» . ئايالىڭىزنى ئۆز ئەركىنلىكىگە قويۇپ بىرىڭ. بىر كۈنى چۈشىنىدۇ. بولمىسا بىرئۆمۈر يالۋۇرۇپ ئۆتىسىز.

سوتقا بارسىڭىزمۇ ئاسانلىقچە ئاجراشتۇرمايدۇ. ئاجراشتۇرماسلىق توغرىسىدا ھۆكۈم چىقىرىلغان بولسا 6ئايغىچە يەنە داۋاملىق ئۆي تۇتىسىلەر. ھىچكىمنىڭ ئاجرىشىمەن دەپ ئەرز قىلىشىغا بولمايدۇ. ئەمدى ئالتە ئايدىن كېيىن ئايالىڭىز يەنە ئاجرىشىمەن دەپ قالسا ئىش چاتاق.
سىلەرنىڭ نىكاھ تۇرمۇشىڭلاردا سىزنىڭ خىزمەت ئىشلىمەي ئۆيدە ئولتۇرۋالغىنىڭىز ئايالىڭىزنىڭ كاللىسىغا بەك تەسىر قىلىپ سىزدىن ئۈمىتسىزلىنىپ كەتكەنلىكىدىن مۇھاببەت مۇناسىۋىتى بۇزۇلغان بولۇشى مۈمكىن.
مېنىڭ تولا خەقلەرگە دەپ كۈلدۈرىدىغان بىر ئېغىز گېپىم بار ئىدى. « ئايال كىشى ئۆيدە بىكار ئولتۇرۇپ قالسا كاۋا مانتا ئېتىپ بولسىمۇ باشقىلارنى خوش قىلىدۇ. ئەگەر ئەر كىشى بىكار ئولتۇرۇپ قالسا ....» شۇنداق ئەمەسمۇ خىزمەتتىن چارچاپ قايتىپ كەلگەن ئايالىڭىز ئۆيگە كىرىپ قارىسا يوتقانلار جايىدا تۇرغان. سىز ئەتىگەنلىك چاينىمۇ ئىچمەي ئىچ كىيىم بىلەنلا توردا ئولتۇرغان.............. ئاياللار مۇشۇنداق مۇھىتتىن ئەڭ بىزار.
ئائىلەمنى ساقلاپ قالىمەن دىسىڭىز مۇشۇ 6 ئايدا مائاشى يۇقۇرى خىزمەتتىن  ياكى بولمىسا مۇقىم خىزمەتتىن بىرنى تىپىڭ ، ئاندىن ئايالىڭىزنىڭ سىزگە بولغان كۆزقارىشى ئۆزگىرىپ قېلىشى مۈمكىن.
