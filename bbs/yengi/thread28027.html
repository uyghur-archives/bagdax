مانا بۇنى ياشاش دەيمىز(بوكىس چىمپىيونى ئابدۇراخمان ئابلېكىم ۋاپاتىغا)
ئابدۇۋەلى تۇرسۇن
(بىڭتۇەن مىللىي پېداگوگىكا مەكتىپى)  


       يېقىنىدىن بۇيان ‹‹مەن ھازىرغىچە قانداق ياشىدىم، بۇندىن كېيىن قانداق ياشىسام بولىدۇ›› ھەققىدە ئويلايدىغان بولۇپ قالدىم،بۇنىڭغا نىمە سەۋەپچى بولغانلىقىنى بىلمەيمەن. ئەمما ئويلاپ باقسام كىشلەر ۋاي دېگۈدەك ئىشلارنى قىلالماپتىمەن، خىزمەتنى ياخشى قىلىۋاتىمەن ،بۇنڭدا گەپ يوق.ئەمما ئەسلىدىكى غايەم بويىچە ئۆلچىسەم دېگەنلىرىمنىڭ كۆپىنچىسى تېخى ئىشقا ئاشماپتۇ.بەزى ئىشلارغا تەن بەرمەي ئامالىم يوق،ئەمما يەنە بەزى ئىشلارغا پەقەت تەن بەرگۈم يوق.
      ئۈندىدار بارلىققا كەلگەندىن بۇيان ئۇچۇرنىڭ تارقىلىشىغا بولغان چەكلىمە  يەنىمۇ ئازلاپ، خالىغان يەردە خالىغان نەرسىنى كۆرۈش ئىكانىيىتىگە ئىگە بولدۇق.بۇنىڭ كىشىلەرنىڭ تۇرمۇشىغا بولغان تەسىرى ھەقىقەتەنمۇ بەك زور بولدى.كىشىلەر ئوتتۇرىسىدىكى  مۇناسىۋەتنى  تەڭشەپ ياكى يېقىنلاشتۇردى ياكى يىراقلاشتۇردى. بۇمۇ مۇقەررەر بولىدىغان ئىش. مەنمۇ بۇنىڭدىن داۋاملىق يېڭىلىقلارنى ئىگەللەپ تۇرىمەن.شۇڭا سۆزۈمدە  ئۈدىدارنى قەتئى ئەيپلەش مەنىسى يوق.
     ئۈندىداردا  كېلىدىغان ئۇچۇرلار ئىچىدە يەنە شۇنداق بىر خىل مەزمۇندىكى ئۇچرلارمۇ  بار، ئۇنىڭدا كىشىلەرگە نەسىھەت كۆپ.بولۇپمۇ بايلىق توغرىسىدا  نەسىھەتلەر خېلى جىق  كېلىدۇ.ئۇنىڭدا بۇ دۇنيانىڭ تۆت كۈنلۈك ئالەم ئىكەنلىكىنى، مال-دۇنيانىڭ قالىدىغانلىقىنى ،شۇڭا ئۇنىڭغا بەك ھېرىسمەن بولۇپ كەتمەسلىكنى ،  ئۇنىڭ ھەممىنى ھەل قىلىپ كېتەلمەيدىغانلىلىنى ، ئادىمىلىك بىلەن ياشاشنىڭ مۇھىلىقىنى سۆزلەيدۇ.بۇنى مەن ئىنسانلارنىڭ قانۇنىيەتكە خىلاپلىق قىلىپ ، ئۆزىنى بەزلەپ، ئۆز كۆڭلىنى ياساش،ئۆز -ئۆزىگە تەسەللى بېرىش  دەپ چۈشىنىمەن. چۈنكى دىققەت قىلىپ باقسام ئاشۇنداق دېگەنلەرنىڭ ئۆزىدە ناھايىتى كۆپ مال دۇنيا باركەن، دېمەك ئۇلار بۇ جەھەتە ئۆزى نەسىھەت قىلىۋاتقان كىشىلەرگە قارىغاندا كۆپ ئۈستۈن تۇرىدىكەن.شۇنداق  بولغان ئىكەن ،سەن شۇنچىۋالا مال دۇنيانى يىغىۋېلىپ نىمە قىلارسەن ،ھازىر توختىماي قلىۋاتقان ئىشىڭ يەنە نىمە ، باشقىلارغا نەسىھەت قىلىشنىڭ ئۇرنىغا تالاغا چىقىپ ، شۇ مال- دۇنيالىرىڭنى شۇلارغا تارقىتىپ بېرىۋەتسەڭ بولمامدۇ،مۇھەممەد ئەلەيسۇلاممۇ ۋاپات بولغان چاغدا ئۆيىدە بىر تال پالاس بىلەن بىر تونى قاپتىكەن ، ئۇنىڭدىن باشقا ھېچنىمە قالماپتىكەنغۇ، سەن شۇنداق دېگەندىكىن  نىمىشقا شۇنداق قىلالمايسەن دەپ قالىمەن ئىچىمدە . ۋە سېنىڭ نەسىھىتىڭگە قۇلاق سېلىشنىڭ ھاجىتى يوق.ئۆزەمنىڭ يولىدا تىرىشىۋېرىمەن، بۇ بەلكىم پۇل تېپىش يولىدۇر،ياكى يول تېپىش يولىدۇر،ياكى بالىلىرىمنى ياخشى تەربىيەلەش يولىدۇر. ئىشقىلپ مەن ئۈچۈن ئەھمىيەتلىك ئىشلاردۇر.
    مېنىڭچە ياشاشتىمۇ قانۇنىيەت بار، بۇ خۇددى فىزىكا ،ماتېماتىكا پەنلىرىدە قانۇن بولغاندەك ئۆزگەرمەيدۇ، بىز چوقۇم ئۇنىڭغا بوي سۇنۇشىمىز كېرەك.بولمىسا قانۇنىيەتكە خىالپلىق قىلغانلىقىمىز ئۈچۈن جازالىنىمىز.
    ئىنسان تۇغۇلغاندە قورساق بىلەن بىللە تۇغۇلىدۇ.كىيم كەيمىسە بولمايدۇ،بولمىسا توڭلاپ ئۆلۈپ قالىدۇ،ياشىغان ئىكەن ئۆي ئىچىدە تۇرۇشى كېرەك،تالادا كۈن ئۆتكۈزگىلى بولمايدۇ، يەنە ئىنساندا يۇرەك بار،ئۇنىڭ نازۇك ھېس -تۇيغۇلىرى بار. مېڭىسى بار ، ئۇ بىر ئۆمۈر پىكىر قىلىش ۋەزىپىسىنى ئۈستىگە ئالغان.
      فىرىئويىد(يەھۇدىلار دۇنياغا 3 مېڭە تەقدىم قىلدۇق دەپ قارايدۇ، ئۇنىڭ بىرى ماركىس،يەنە  بىرى ئېينىشتىيىن، ئۈچىنچىسى بولسا ھازىر بىز تىلغا ئالغان مۇشۇ ئادەمدۇر)  ئىنساننىڭ ئېھتىياج قاتلىمى نەزىرىيىسىنى ئوتتۇرىغا قويدى.مۇنداقچە ئېيتقاندا ياشاشتىكى قانۇنىيەتنى ئوتتۇرىغا قويدى.  بۇ بىز  ئېىراپ قىلىشقا ، بويسۇنۇشقا تېگىشلىك قانۇنىيەتتۇر. بۇنىڭدا بىر قاتلام بىر پەلەمپەيگە ئوخشايدۇ. بىز بىر پەلەمپەيگە دەسسەپ چىققاندىن كېيىن يەنە بىر بالداق ئۆرلەش ئېھتىياجى تۇغۇلىدۇ.چۈنكى ئۇنىڭ ئۈستىدە تېخىمۇ ياخشى نەرسىلەر بىزنى كۈتۈپ تۇرغان بولىدۇ.بۇ دۇنيانى تەرەققى قىلدۇرۇدىغان، ئايلاندۇرۇدىغان  بىر خىل كۆرۈنمەس كۈچتۇر. بۇ كۈچ ئىنسانلارنى توختىماي تىرىشىشقا ئۈندەيدۇ ھەم شۇنداق قىلدۇرىدۇ.
     مەن بۇنىڭدىن  بۇرۇنراق بىر ئىنگىلىزچە گېزىتتىن مۇنداق بىر تەتقىقات دوكلادىنى ئوقۇغان ئىدىم، ئۇنىڭدا مۇنداق يېزىلغان ئىكەن:  ئادەم كەمبىغەل بولسا پۇل ئۇنىڭغا خوشاللىق ئېلىپ كېلىدۇ.ئەمما ماددى ئېھتىياجى تولۇق قانغاندىن كېيىن يەنە داۋامىق خوشاللىق ئېلىپ كېلەلمەيدۇ. ئەمما ئادەمگە بىر خىل خاتىرجەملىك ئېلىپ كېلەلەيدۇ.
   شۇڭا مۇشۇنداق  قانۇنىيەتلەرگە خىلاپلىق قىلىپ ، بايلىققا بېرىلمەڭلار،بىر قاچا تاماققا قورساق تويىدۇ ، دېگەندەك سۆزلەرگە چىن پۈتمەيمەن.ئادەم ھاياتلا بولىدىكەن،تىرىشىشى كېرەك،ئۆزى ئۈچۈن،بالىلىرى ئۈچۈن،  ئەۋلاتلار ئۈچۈن تېخىمۇ كۆپ بايلىق يارىتىشى كېرەك دەپ قارايمەن. بۇ يەردىكى بايلىق ماددى بايلىق ۋە مەنىۋى بايلىق ھەممىسىنى ئۆز ئىچىگە ئالىدۇ . بايلىق بۇنى خاتا دېسا ئۇ ھالدا  ئىنسانلارنىڭ تىرىكچىلىك يولىدا  توختاۋسىز تىرىشىشنى چۈشەندۇرگەىلى بولمايدۇ.  يېزىۋېتىپ سۆزنىڭ يۆنۈلۈشىنىڭ باشقا ياققا قاراپ مېڭىۋاتقانلىقىنى  بايقاپ قالدىم. ئەمدى ئەسلى دېمەكچى بولغىنىمغا قايتىپ كېلەي.
    مېنىڭ ئەسلى دېمەكچى بولغۇنىم ئىنسان قانداق ياشىشى كېرەك دېگەن تېما ئىدى.بۇ ئوينىڭ پەيدا بولۇشىغا تۆنۈگۈن كەچتە ئۇندىدارنى قاپلىغان مۇسىبەت خەۋىرى يەنى داڭلىق تەنھەركەتچىمىز ، ئاسىيا بوكس شاھى ئابۇراخمان ئابلىكىمنىڭ ۋاپات بولۇشى خەۋرىدن پەيدا بولغان ئوي -پىكىر  ئىدى.
  ھەيھات ، ئەجەلگە ئامال يوق، ئۆلۇم دېگەندا ئۆچرەت بولمايدۇ،ئۇ ئادەم ئىلغىمايدۇ. بۇمۇ بۇ دۇنيانىڭ  تەڭلىكلرىدىن بىرى بولسا كېرەك.
   تۈنۈگۈنكى مۇسىبەت خەۋرى كۆپ ئادەمەلەرنىڭ ئچىنى سېرىۋەتتى.نىمىشقا دېگەندە بىرى بۇ يىگىت مىللىتىمىزگە شان -شەرەپ كەلتۈرگەن  بىر ئەزىمەت ئىدى.يەنە بىرى ئۇنىڭ  تېخى ناھايىتى ياش ئىكەنلىكى ئىدى.
    كۈندە نۇرغۇن ئادەم تۈرلۈك سەۋەپلەر تۈپەيلىدىن بۇ دۇنيادىن كېتىدۇ.ئۇلارنىڭ ھەممىسىنىڭ بارىدىغان يېرى ئوخشاش.ئەمما بىز مۇتلەق كۆپچىلىكىنىڭ كىملىكىمى بىلمەيمىز. مانا بۇ ياشاشتىكى پەرقتۇر.بەزىلەر كەتسە پۈتۈن دۇنيا تەۋرىشى مۈمكىن. يەنە بەزىلەرنىڭ كەتكىنىنى بايا دېگەندەك ھېچكىم بىلمەيدۇ.
    تۆنۈگۈن بۇ ياش بوكس شاھىمىز كېتىۋىدى مۇسىبەت خەۋىرى ئۇندىدار سالونلىرىنى ،تور دۇنياسىنى قاپلىدى. ھەممە ئادەم دۇئا قىلىدى.دېمەك ئۇ  37 يىل ياشىغان بولسىمۇ  ئەمما ئۆزىنى شۇنچىۋالا ئادەمگە بىلدۇرەلىدى. مانا بۇنى ياشاش دەيمىز.
    ئۆلۈمىنىڭ مەنىسى ئوخشىمايدۇ.   بۇنى بۇنىڭدىن ئىككى مىڭ يىلنىڭ  ئالدىدا جوڭگونىڭ بۇرۇنقى تارىخچىسى سى ماچىيەن دەپ بولغان ئىدى.  قىسقىسى بۇ ئەزىمەت فىرويىد دېگەن  ياشاش ئھتىياج قاتلىمىنىڭ  5-پەلەمپىيىگە يەنى باشقىلارنىڭ ئېتراپ قىلىشىغا ئىرىشىش، يەنە بىر تۈرلۈك قىلىپ ئېيتساق نام قالدۇرۇپ  ياشاش چوققىسىغا   چىقالىدى.مانا بۇنى ياشاش دەيمىز.  
    ‹‹ توشقاندەك يۈز يىل ياشىغىچە يولۋاستەك بىر كۈن ياشا ››  .توغرا ، ئۇ مۇشۇنداق ياشىيالىدى.ئۇ قىسقا بولسىمۇ يولۋاستەك ياشاپ كېتەلىدى.مانا بۇنى ياشاش دەيمىز.
     مۇشۇنىڭ ئۆزى بىز ئۈچۈن ئۈلگەدۇر.
   مۇشۇ كىچىككىنە يازمام بىلەن ئۇ ئەزىمەتنىڭ ۋاپاتىغا قايغۇلۇق تەزىيەدە ئىكەنلىكىمنى بىلدۇرۇمەن.

2015-يىلى 1-ئاينىڭ 22-كۈنى سەھەردە كاللامغا كەلگەن پىكىرلەر، ئورنۇمدىن تۇرۇپلا يېزىلدى.
