سىلەرنىڭ قىممەتلىك مەزمۇنلىق بىلىم بىلەن تەمىن ئەتكىنىڭلارغا رەخمەت مەن ئەركىن سىدىق مۇئەللىمنى ۋە بارلىق بىلىم ئەكلىنى ھورمەتلەيمەن  مىنىڭ بىلىم سەۋىيەن ئاجىز بىر ئاددى ھۇنەرۋەن لىكىن كىتاپ كۇرۇشكە مۇشۇنداق قىممەتلىك ئۇستازلىرىمىزنىڭ يازغان ئەسەرلىرىنى بىرىلىپ ئۇقۇشقا ئامتاق مۇشۇ يىقىننىڭ ياغى مىنىڭ كاللامغا مۇشۇ سۇئال كىرىۋالدى بۇ سۇئالغا ئەركىن سىدىق ئاكاملا قانائەتلىنىرلىك جاۋاپ بىرەلەيدۇ دەپ ئىشەنگەش بۇ سۇئالنى سوراشقا مەجبۇرمەن    مەندەك بىلىمى ئاز ئەمدى يول مىڭىشنى ئۇگۇنىۋاتقان بىلىم ھەۋەسكارىنىڭ بىلمەكچى بولغىنىنى كەڭ مۇنبەر داشلانىڭ ۋاقىت چىقىرىپ ئىنكاس شەكلىدە يەتكۇزدىغىنىغا ئىشىنىمەن  مىنىڭچە دوسلانىڭ يوللىغان ئۇچۇرى ماڭا نىسپەتەن مۇجىمەل بوپقالدى دەپ ئويلايمەن شۇڭا قايتا سورىسام كەچۇرۇڭلا  :  ‹‹يەرشارى سۇنئى ھەمياھى قانچە سائەتتە   قانچىلىك تىزلىكتە ئايشارى ئوربىتىغا يىتىپ بارالايدۇ ؟ ئايغىچۇ؟  مۇشۇ تىزلىك بىلەن قۇياشقا قانچىلىك تىزلىكتە بارالايدۇ ؟  ››   مۇشۇ سۇئال گەرچە ئاددى بولسىمۇ مەن ئۇچۇن مۇيىم بوپقالدى بۇ سۇئال مەن چۇشەنگۇدەك جاۋاپقا ئىرىشەلىسە ئاندىن ئەركىن سىدىق ئاكامدىن ھەقىقى مەن بىلمىكچى بولغان تۇگۇن يىشىلىدۇ دوسلا ئاۋارە قىلدىم كەچۇرۇڭلار

       1.يېقىن ئوربېتا
        يېقىن ئوربېتا ( ياكى تۆۋەن ئوربېتا ) بوشلۇق ئەسۋابلىرىنىڭ يەر يۈزىگە يېقىن ئوربېتىلىرىنى كۆسىتىدۇ . يېقىن ئوربېتىنىڭ ئومۇملاشقان ئېنىقلىمىسى يوق . ئادەتتە يەر يۈزىدىن 2000 كىلومېتىردىن تۆۋەن بولغان ، چەمبەرگە يېقىن بولغان ئوربېتىلار بۇ تۈرگە كىرىدۇ . يېقىن ئوربېتىدىكى سۈنئىي ھەمراھلار يەر يۈزىگە بىر قەدەر يېقىن بولغانلىقى ئۈچۈن ، بۇنداق سۈنئىي ھەمراھلارنىڭ كۆپىنچىسى كۆزىتىش سۈنئىي ھەمراھلىرى ، ئۆلچەش سۈنئىي ھەمراھلىرى ، بوشلۇق پونكىتلىرى ۋە بىر قىسىم يېڭى ئالاقە سۈنئىي ھەمراھلىرى بولىدۇ . تۆۋەندىكىسى بىر قانچە مىساللار :
        خەلقئارا بوشلۇق پونكىتى : يەرگە يېقىن نۇقتا 347 كىلومېتىر ، يەرگە يىراق نۇقتا 360 كىلومېتىر ، ئوتتۇرىچە تېزلىكى 7706.6 مېتىر ھەر سېكۇنت ( 27743.8 كىلومېتىر ھەر سائەت ) ، ئوربېتىدا ئايلىنىش دەۋرى 91 مىنۇت .
        خابىل بوشلۇق تېلېسكوپى : يەرگە يېقىن نۇقتا 551.4 كىلومېتىر ، يەرگە يىراق نۇقتا 555.6 كىلومېتىر ، ئوتتۇرىچە تېزلىكى 7106 مېتىر ھەر سېكۇنت ، ئوربېتىدا ئايلىنىش دەۋرى 95.6 مىنۇت .
        ئىردىي يۇلتۇزلىرى : ۋىكىپېدىيەدە ماتېرىيال يوق ئىكەن .
       2.ئوتتۇرا ئوربېتا
        ئوتتۇرا ئوربېتا ، تۆۋەن ئوربېتا بىلەن يۇقىرى ئوربېتا ئارلىقىدا بولۇپ ، يەر يۈزىدىن 2000 كىلومېتىردىن 35786 كىلومېتىرغىچە بولغان ئوربېتىنى كۆرسىتىدۇ . بۇ خىل ئوربېتىدىكى سۈنئىي ھەمراھلار ئاساسلىقى يول باشلاش سۈنئىي ھەمراھلىرى ، مەسىلەن يەر شارى ئورۇن بەلگىلەش سېستىمىسى ( 20200 كىلومېتىر ) ، گلوناس سېستىمىسى ( 19100 كىلومېتىر ) ، يەتتە قاراقچى سۈنئىي ھەمراھ يول باشلاش سېستىمىسى ( 21500 كىلومېتىر ) ۋە گالىلېي ئورۇن بەلگىلەش سېستىمىسى ( 23222 كىلومېتىر ) قاتارلىقلار . بىر قىسىم قۇتۇپ رايونىدىن ئۆتىدىغان ئالاقە سۈنئىي ھەمراھلىرىمۇ ئوتتۇرا ئوربېتىدا . ئوتتۇرا ئوربېتىدىكى سۈنئىي ھەمراھلارنىڭ ئايلىنىش دەۋرى 2 سائەتتىن 12 سائەتكىچە . تۆۋەندىكىسى بىر قانچە مىساللار :
        گلوناس : ئوربېتا ئايلىنىش دەۋرى 11 سائەت ئىتراپىدا .
       يەر شارى ئورۇن بەلگىلەش سېستىمىسى : دەۋرى 12 سائەت . GPS .
        يەتتە قاراقچى سۈنئىي ھەمراھلىرىدىن گ 6 : يەرگە يېقىن نۇقتا 35783 كىلومېتىر ، يەرگە يىراق نۇقتا 35803 كىلومېتىر .
        گالىلېي ئورۇن بەلگىلەش سېستىمىسى : دەۋرى تەخمىنەن 14 سائەت .
        3.ماس قەدەملىك ئوربېتا
        بۇ خىل ئوربېتىدىكى سۈنئىي ھەمراھلار ئېكۋاتوردىلا ئايلىنىدۇ ، ئوربېتىسى مۇتلەق چەمبەر ، ئوربېتىسىنىڭ يەر يۈزىدىن ئېگىزلىكى دەل 35786 كىلومېتىر ، ئوربېتىدا ئايلىنىش دەۋرى 23 سائەت 56 مىنۇت 4 سېكۇنت . بۇ خىل سۈنئىي ھەمراھلار يەر يۈزىگە نىسبەتەن پەقەت بىرلا ئورۇندا تۇرىدۇ .

        مېنىڭ ھېس قىلىشىمچە يەر شارى سۈنئىي ھەمراھى پەقەت يەر شارىنى ئايلىنىپ ھەرىكەت قىلىدۇ . ئايغا بارىدىغانلىرى تۆت بەش كۈنلەردە بارىدىغۇ دەيمەن ، توردىن ئاپوللو كېمىلىرىنى ۋە باشقىلارنى تەكشۈرۈپ باقسىڭىز بولىدۇ . قۇياشقا كەلسەك ، ھېس قىلىشىمچە بىرەر ئۈسكۈنە تېخى قۇياشقا تۈپتۈز بارمىدى ، قۇياشنى بىر ئىككىنى ئايلىنىش جەريانىدا ئاستا ئاستا يېقىنلىشىدىغۇ دەيمەن ، ھەقىچان ئايغىمۇ شۇنداق ، بۇنىڭ فىزىكىلىق سەۋەبى بولۇشى مۇمكىن . يەنە ، قۇياشنى تەكشۈرۈش ئۈسكۈنىسى قۇياشقا ئۇدەرىجىدە يېقىنلاشمايدۇ ، ئەڭ يېقىن بولغىنىمۇ مېركۇري بىلەن قۇياشنىڭ ئوتتۇرسىغىراق بارامدىكىن . بۇلارنىڭ بەزىلىرى پەرىزىم بولۇشى مۇمكىن ، كەچۈرۈڭلار *-_-*

قىرىندىشىم Jimgur0997  سىزنىڭ سەمىمى ئىنكاسىڭىزغا كۆپ رەھمەت. دەيدىغىنىم يەنىلا شۇ،خاپا بولماي ئۆزىڭىز ئىنتىرنىتتىن ئىزدىنىپ باقسىڭىز،بۇ سۇئالىڭىز ئۈچۈن ھۆرمەتلىك ئەركىن ئاكىمىزنىڭ قىممەتلىك ۋاقتىنى زايا قىلمىسىڭىز.ئەگەر كۆڭلىڭىزگە ئالمىسىڭىز ئەركىن ئاكام ئۈچۈن بەزى سۇئالىڭىزغا مەن جاۋاب بىرەي.
بىز ياشاۋاتقان يەر شارى ھەر سائەتتە 1670كىلومىتىر سۈرئەتتە ئايلىنىدۇ. قۇياشنى بولسا 108.000كىلومىتىر سۈرئەتتە ئايلىنىدۇ. قۇياش سىستىمىسىنىڭ گالاكسى ياكى سامان يولىسىستىمىسىدىكى ئايلىنىشى بولسا ھەر سائەتتە 720.000كىلومىتىر. سامان يولى سىسىتىمىسى ياكى گالاكسى ھەر سائەتتە 950.000كىلومىتىرلىق سۈرئەت بىلەن ئايلىنىدۇ. سۈنئى ھەمرا ھەر سىكوتتا 7مىڭ 900مىتىر سۈرئەت بىلەن يۇقىرىغا ئۆرلەيدۇ. بۇ 1-ئالەم تىزلىكى.ئالەم بوشلۇقىدا ئۇنىڭ سۈرئىتى سكوتىغا 11مىڭ200مىتىر بولىدۇ. يەر شارى بىلەن قۇياشنىڭ ئارلىقى تەخمىنەن 250.000.000كىلومىتىر.ئاي بولسا بىزدىن 386.694كىلومىتىر ئۇزاقتا.مۇشۇ سانلىق مەلۇمات بىلەن ئۆزىڭىز ئاددى بىر ھىسابلاش ئىلىپ بارسىڭىز بارلىق سۇئالىڭىزنىڭ جاۋابى چىقىدۇ. مەن بۇنى بەيدۇ تورىدىن ئىزدەپ تاپتىم.يەنى ئەركىن ئاكىمىزنى ئاۋارە قىلمايمۇ كۆپ نەرسىلەرنى تاپقۇلى بولىدىكەن. سۇ ئال سوراش ئىنتايىن ياخشى ئىش.ئەمما ئالدى بىلەن ئۆزىڭىز بىر ئىزدىنىپ باقسىڭىز. ئەركىن ئاكامنىڭ يازمىسىغا تەقەزا ئىنتايىن كۆپ ئادەملەر بار. ئۇلارنىڭ ۋە ئەركىن ئاكامنىڭ ۋاقتىغا زىيان سالمىسىڭىز. سىزدىن يەنە بىر قىتىم ئەپۇ سورايمەن. بىراق ئەركىن ئاكامنىڭ ۋاقتى ئۆزىمىزمۇ بىلەلەيدىغان ئىزدىسەك تاپالايدىغان نەرسىلەرگە ئىسراپ بولۇپ كەتسە بۇ پۈتۈن مىللەتنىڭ زىيىنى بولىدۇ.ئاخىرىدا ئەركىن ئاكامنىڭ تىنىگە سالامەتلىك ئىشىغا ئوڭۇشلۇق تىلەيمەن ۋە چۇڭقۇر رەھمەت ئىيتىمەن.
