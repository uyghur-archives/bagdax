ئەرەبلەرنىڭ ئۈممەتكە تۇتقان دۈشمەنلىگى يەھودىنىڭكىدىن قېلىشمايدۇ.

قېلىشمايدۇلام ئەمەس بەلىكى نەچچە ھەسسە ئېشىپ چۇشىدۇ
مۇسۇلمانلار ئەمىلىيەتتە دىنىم ئۇچۇن كۆرەش قىلىمەن دەپ مۇجاھىتلىق بايرىقىنى ئىگز كۇتۇرۇپ چىققان بىلەن  ئەمىلىيەتتە مۇسۇلمانلار ئارىسىدىكى كۆرەشلەر مەنپەئەت كۆرىشىدىن ئىبارەت ھاكىمىيەتتىكى مەنسەپ پەرەزلەرنىڭ ھۆقۆق تالىشىش كۆرىشى  .  چۆڭ جەھەتتىن ئېيىتساق نۆۋەتتىكى پارىسلىقلار يەنى شىئەلەر بىلەن  ئەرەپلەر يەنى سۇنئىلەرنىڭ كۆرىشى مىسال بۇلالايدۇ  .  بۇ ئىككى پىرقىلارنىڭ ھاكىمىيەت بىشىدىكى ئەمەلدارلىرى مەنپەئەت جەھەتتىن ئۆز ئارا كىلىشەلىگەن زامانلاردا بۇلارنىڭ پەتىۋاچىلىرى سۇنئى ۋە شىئە ھەممىمىز ئۆخشاش مۇسۇلماندۇرمىز بىزنىڭ قىلبىمىز بىر دىلىمىز بىر دىگەن شۇئارنى كۆتۇرۇپ چىقىپ خەلىقنى مۇشۇخىل پىكىر ئېقىمىغا ئەگىششىكە كۇچەپ تەشۋىق قىلىدۇ   شۇنداقلام بۇلار ئۆتتۇرىسىدىكى مەنپەئەتلەر بۇزۇلۇپ زىدىيەت باشلانغان ھامان سۇنئىلەر شىئەلەرنى شىئەلەر سۇنئىلەرنى مۇسۇلمانلىقتىن چىقىرىۋىپتۇرۇپ پەتىۋا تارقىتىپ بىر بىرىنى كاپىرغا چىقىرىپ تەشۋىقاتنى كۇچەيتىدۇ  ۋە بۇ ئارقىلىق خەلىقنى بىر بىرىگە دۇشمەن قىلىپ  جىدەلگە سېلىپ ئۇزىنىڭ مەنپەتىنى قۆغداپ قالماقچى بۇلىشىدۇ  .
مۇشۇنىڭدىن شۇنى كۇرىۋېلىش تەس ئەمەسكى مۇسۇلمان خەلىقتە ئاق قارىنى پەرىق ئەتكۇدەك  بىلىم ۋە ساپا يۆق  كىم نىمە دىسە قانداق تەشۋىقات يۇرگۇزسە شۇ بۇيۇنچە ئەگىشىپ كىتىۋېرىدۇ  شۇڭىمۇ بۇلارنىڭ جىدىلى 1000 يىلدىن بېرى تۆختاپ قالغىنى يۆق ھەم بۇندىن كىيىنمۇ خىلە ئۇزۇن بىر مەزگىللەرگىچە تۆختاپ قالمايدۇ .
كىچىك جەھەتتىن مەيلى سۇنئىلەر بولسۇن مەيلى شىئەلەر بولسۇن بۇلارنىڭ ئارىسىدىمۇ ھەرخىل  كىچىك مەنپەئەت گۆرۆھلىرىنىڭ زىدىيەتلىرى خېلە ئېغىر دەرىجىدە مەۋجۇت .
دىمەك ئىسلام دۇنياسى   خەلىقلەرنىڭ بىلىمسىزلىكى سەۋەبلىك پۇرسەت پەرەس ھۆقۆدارلارنىڭ ئۆز مەنپەئەتلىرىنى قۆغداش ئۇچۇن قۇزۇقتەك قېقىپ قۆيغان نۇرغۇن  زىدىيەتلەر بىلەن تۆلغان

تۆۋەندىكى  ماقالە ئارقىلىق مۇسۇلمانلارنىڭ ئارقىدا قېلىشىغا سەۋەپچى بولغان باشقىچە كۆز قاراشلارغىمۇ قاراپ باقايلى

مەقالەر مەنبەسى-:- ئەرەپ رادىۆسى ..

مۇسۇلمانلارنىڭ تەرەقىياتتا ئارقىدا قېلىشىنىڭ سەۋەپلىرى زادى نىمە ... ؟
مۇسۇلمانلارنىڭ دەستۇرى بولغان قۇرئان كىرىم مۇسۇلمانلارغا ھەر ساھە ئىلىملىرىنى ئۇگۇنۇشنى پەرھىز قىلغاننىڭ سىرتىدا ئۇلار ئۇچۇن نۇرغۇن كەشپىياتلاردىن يىپ ئۇچى بەرگەن تۇرسا ئۇلار يىنە نىمە ئۇچۇن ئىلىم  پەندە ۋە باشقا ھەر تۇرلۇك تەرەقىياتلاردا باشقىلاردىن ئارقىدا قالىدۇ ... ؟؟
مەرھۇم مۇتەپپەككۇر ئەلى تەنتاۋى بۇ ھەقتە جاۋاپ بېرىپ مۇنداق دىگەن ..
‹‹ دۇنيا مەدىنىيىتى 3 قەۋەتلىك بىر بىناغا ئوخشاش بۇلۇپ ئۇنىڭ 1- قەۋىتىنى ئاۋالقى ئەسىرلەردىكى فىرئەۋىنلەر ، ناپالىئۇنلار ، ھونلار  ، ئاندىن يۇنانلىقلار ، ئاندىن رىملىقلار بەرپا قىلغان بولسا .. ئۇنىڭ 2- قەۋىتىنى ئۆتتۇرا ئەسىرلەردىكى مۇسۇلمانلار ياراتقان .. 3- قەۋىتىنى بولسا ھازىرقى ياۋرۇپالىقلار بارپا قىلغان بولدى ››  دەپ كۆرسەتكەن ....
ئەمدى بىز نەزىرىمىزنى ئاغدۇرۇپ باخساق   دۇنيادا سائەت دىگەن نەرسە ئىجات قىلىنمىغان زامانلاردا  دەسلەپتە مۇسۇلمانلار قۇم سائەتىنى ئىجات قىلغان بۇلۇپ ئابباسىلار خەلىپىلىكى تەرىپىدىن شۇ ۋاقىتتىكى فىرانسىيە پادىشاھىغا بىر قۇم سائەت سوۋغا قىلىنغاندا ئۇلار مىڭىۋاتقان بۇ قۇم سائىتىگە ھەيران قالغانلىقتىن بۇنىڭ ئىچىدە شەيتان بارئۆخشايدۇ دىگەن ئىكەن .... بۇ ۋەقەلىكتىن بىر قانچا يۇز يىلدىن كىيىن غەرىپلىكلەر تەرىپىدىن ئىجات قىلىنغان تىلۋىزىيە مۇسۇلمانلارغا دەسلەپتە يىتىپ كەلگەندىن كىيىن مۇسۇلمانلار بۇ ئاجايىپ نەرسىگە ھەيران قالغانلىقتىن بۇنىڭ سىرىنى بىلمىگەنلىكتىن بۇنىڭ ئىچىدە شەيتان بار ئۆخشايدۇ دىگەن ئىكەن ... دىمەك دەۋىر شۇنداق ئالماشقان شۇنداق ئۆزگەرگەن ..  مۇسۇلمانلار 14- ئەسىردىلام قەغەز ياساش تېخنىكىسىنى ئىگىللىگەن بۇلۇپ باغداتنى مەركەز قىلغان ئاساستا ئۆزلىرىنىڭ پەلىسەپە ، ئەدىبىيات ، ئىسلامىيەت ۋە باشقا ئەسەرلىرىنى قەغەزگە يېزىپ چىقارغان بولسا شۇ ۋاقىتتىكى غەرىپلىكلەر ئەسەرلىرىنى تېرىلەرگە دەرەخلەرنىڭ يۇپۇرماقلىرىغا يازغان ئىكەن ......  تەرەقىيات نۆۋىتى ئالماشقاندىن كىيىن ھازىرقى زامان تېخنىكا يىڭىلىقلىرى ئىسلام ئەللىرى ۋە باشقا 3- دۇنيا ئەللىرىگە ياۋرۇپالىقلاردىن يېرىم ئەسىر كىچىكىپ كەلدى ..
دۇنيا قانداق ئۆزگەرگەن ...؟ .. ۋە مەدىنىيەت پەن تېخنىكا يىڭىلىقلىرى نۆۋىتى قانداق ئالماشقان ..؟   بۇنىڭدا ئىسلام دىنىنىڭ بىرەر كەمچىللىككى بارمۇ   .. ؟ تەبىئى ياق ... ئۇنداقتا كەمچىللىك كىمدە .. ؟ كەمچىللىك مۇسۇلمانلارنىڭ دەل ئۇزىدە .... بۇرۇنقى مۇسۇلمانلار ئىسلام دىنىنىڭ روھىغا ئەمەل قىلغان ئۇلار شۇ زامانلارنىڭ شارائىتلىرىغا نىسبەتەن پەن تېخنىكا ۋە مەدىنىيەتنىڭ ياراتقۇچىلىرى ۋە ئاساس سالغۇچىلىرى بۇلۇپ ياشىغان  ... لىكىن 16- ئەسىرلەردىن كىيىن مۇسۇلمانلارنىڭ ئۆز ئارا بۇلىنىشى قۇرئان كىرىمنىڭ روھىدىن ئاستا ئاساتا يىراقلىشىپ ھەر تۇرلۇك خۇراپاتلىقلارنى دىن دەپ قۇبۇل قىلىشى بۇ دۇنيالىقتىن تېنىپ تەركى دۇنيالىقنى تەرغىپ قىلىپ ئۆمۇر سۇرۇشى نەتىنجىسىدە ئۇلار ئۆزلىرىدىكى تەرەقىيات نۆۋىتىنى باشقىلارغا ئۇتۇنۇپ بەردى ... شۇنىڭ بىلەن ئۇلار سەلتەنەتتىن ، كۇچ قۇۋەتتىن ، خەلىقارالىق ئىتىباردىن ئايرىلىپ قېلىپ بۇگۇنكى ھالغا چۇشۇپ قالدى ... ئۇلار بۇ قىلمىشلىرى بىلەن ئەڭ ئاۋال ئاللاھ تاللاھقا ئاسىلىق قىلغان  بولسا ئاندىن ئەۋلاتلىرىنىڭ كىلەچىكىگە زىيانكەشلىك قىلغان بولدى ... ....
ئەمدى قۇرئان كىرىمدە كىشىلىك ۋە ئىجدىماھى ھەق ھۆقۆق مۇناسىۋەتلەرنىڭ قانۇن پىرىنسىپلىرى ئەڭ ئادالەتلىك ۋە ئەڭ مۇكەممەل دەرىجىدە بىكىتىلگەن تۇرۇپ ئۆزلىرىنى ئىسلام دىنىنى تەتقىق قىلىدىغان ۋە ئىجىرا قىلىدىغان مەملىكەتلەرمىز دەپ سانايدىغان بەزى مۇسۇلمان مەملىكەتلەر ئىنسانلارنىڭ بۇ ھەق ھۆقۆقلىرىنى ئىجىرا قىلىشتا نىمە ئۇچۇن ياۋرۇپالىقلاردىن ئارقىدا تۇرىدۇ دىگەن سۇئالغا كەلسەك  شۇبىھسىزكى ھازىرقى ئىسلام مەملىكەتلىرى گۆرۆھۋازلىق ئاساسىغا قۇرۇلغان دۆۋلەتلەر بۇلۇپ ھەممىسى بىزنىڭ ئاساسلىق دەستۇرىمىز قۇرئان كىرىم دەپ دەۋا قىلشىدۇ  .. مىسىر بۇنىڭ ئەڭ بىشىدا ئەمما مىسىردىكى ئەكۋاللار ئىنىق .. ئەمىلىيەتتە ئەكۋال شۇنداقمۇ ...؟؟  ئەمىلىيەتتە ھەممىسى قۇرئان كىرىمنى ئاساس قىلىپ تۇرۇپ مىڭىۋاتقانلارمۇ ... ؟  كاشكى شۇنداق بولسا ئىدى .. ئۇلار راستىنلارم قۇرئانغا ئەمەل قىلغان بولسا قۇرئاننىڭ پىرىنسىپلىرىنى تەدبىق قىلىشتا ‹‹ مۆھمىنلەر ئۆز ئارا قېرىنداشلاردۇر ›› دىگەن ئايەتنى گۆرۆھۋازلىققا ئۆزگەرتىۋالمىغان ئۇنى خۇدبىلاردىلام ئۆقۇپ قۇيۇپ ئەملىلىيەتتە تەدبىق قىلىنمايدىغان بىر ماۋزۇغا ئايلاندۇرۇپ قويمىغان بۇلار ئىدى .. ‹‹ مۆھمىنلەر كۇپپارلارغا قاتتىقتۇر ئۆز ئارا كۇيۇمچاندۇر ›› دىگەن ئايەتنى  .. ئۆز رىئاللىقىدا  ئۆز - ئارا قاتتىقتۇر ،  كۇپپارارغا يۇمشاقتۇر دىگەندەك مۇئامىلىلەردە بولماس ئىدى ... ئۇلار يىنە ‹‹ سىلەر ھەقىقى مۆزمىن بولساڭلار ئۇستۇنلۇككە ئىگە بۇلىسىلەر ››  دىگەننى چوڭ دۆۋلەتلەرگە قانچە بىقىنساڭلار يۇلانسەڭلەر ئامان قالىسىلەر دىگەنگە ئوزگەرتىپ ئىش كۆرمىگەن بۇلاتتى .. سادداممۇ ئۆز ۋاختىدا ئۇزىنى مۇجاھىد دەپ ئېلان قىلىپ بايرىقىغا ‹‹ ئاللاھۇ ئەكبەر ›› دەپ يىزىپ ئۆتتۇرغا چىققان ... ساددامنىڭ قانداق ئادەملىكى ھەممىگە ئايان .. ساددام ئامېركىنىڭ يىقىن ئادىمى ئىدى ساددام ئۇزىنىڭ ھاكىمىيىتىنى ساخلاش ئۇچۇن نۇرغۇن زىيالىلارنى ئۆلتۇرگەن نۇرغۇن بىگۇنا كىشىلەرنىڭ ھەتتا سەبى بالىلارنىڭمۇ  جىنىغا زامىن بولغان ئىدى ... ئاخىرى ساددامنىڭ ئاقىۋىتى يەنە شۇ ئامېركىلىقلارنىڭ قۇلىدا تۇگەشكەن بولدى ...
ھازىرقى ئىسلام مەملىكەتلىرىنى ياۋرۇپا مەملىكەتلىرىگە سېلشتۇرۇپ كۆرگۇنىمىزدە مەسىلىنىڭ تۇگىنى نەدە ئىكەنلىكىنى تىخىمۇ ئايدىڭ چۇشۇنۇپ يىتىمىز ...
ياۋرۇپا مەملىكەتلىرىدە ئۇزىنىڭ ماددى ۋە مەنىۋى كۇچ قۇۋەتلىرىگە ئىشىنىش روھى بولسا مۇسۇلمانلاردا ئۇزىنىڭ ماددى ۋە مەنىۋى كۇچ قۇۋىتىدىن شۇبھىلىنىش بار ...
ياۋرۇپالىقلاردا ئۇزى پىچىپ ئۇزى تىكىش روھى بولسا بۇلاردا پاشقىلار پىچىپ بارگەن تەييارنى تىكىش روھى بار ....
ياۋرۇپالىقلاردا دىمكىراتىيە ۋە ئەركىن سايلام بولسا  بۇلاردا دىكتاتۇرلۇق بار ...
ياۋرۇپالىقلاردا ئەركىن سايلام بولسا بۇلاردا ساختا سايلام بار ...
ياۋرۇپالىقلاردا ئۆز ئارا ئىتىپاقلىق بولسا بۇلاردا گۆرۆھۋازلىق بار ....
ياۋرۇپالىقلاردا ئۆز ئۇزىگە خۇجا بۇلۇش بولسا بۇلاردا چۆڭ ھاكىم مۇتلەققە بوي سۇنۇش ۋە بېقىنىش بار ....
ياۋرۇپالىقلاردا تىرىشچانلىق بولسا  بۇلاردا ھورۇنلۇق بار ...
ياۋرۇپالىقلاردا بۇگۇنكى ئىشنى ئەتىگە قويماسلىق روھى بولسا  بۇلاردا بۇگۇنكى ئىشنى كىلىدىغان يىنە بىر يىلغىچە  كىچىكتۇرۇش بار ....
ياۋرۇپالىقلاردا ئىنتىزامچانلىق بولسا  بۇلاردا بىپەرۋالىق بار ....
ياۋرۇپالىقلاردا ئىنسانىيەتچىلىك بولسا بۇلاردا قەبىلىۋازلىق ۋە گۆروھۋازلىق بار ...
ياۋرۇپالىقلاردا ئىجادىيەت ۋە كەشپىياتچانلىق روھى بولسا بۇلاردا پاسسىپلق ۋە تەييارتاپلىق بار ....
ياۋرۇپالىقلاردا ۋەدىگە ۋاپا قىلىش سۇزىدە تۇرۇش بولسا بۇلاردا ۋەدىگە ۋاپا قىلماسلىق ۋە سۇزىدە تۇرماسلىق بار ....
ئەكۋال مۇشۇنداق تۇرغان ئىكەن بۇلار قانداقمۇ ياۋرۇپالىقلارغا يىتەلىسۇن ... قانداقمۇ باشقىلارنىڭ ئارقىدا قالمىسۇن ....
يۇقۇرقى سېلىشتۇرمىلارغا قارىساق  بۇ ئىككى پېرقىنىڭ ئۆز ئارا ئالمىىشىپ كەتكەنلىكىنى كۇرىمىز ... ياۋرۇپالىقلار گۆيا ھەزىرىتى مۇھەممەد ساللالاھۇ ئەلەيھىۋەسسەلللەمنىڭ يۇلىنى ۋە ئۇنىڭ ئەخلاقىنى ئۇزىگە دەستۇر قىلغان  مۇسۇلمانلارنى بولسا گۆيا تاش دەۋرىدە ياشاۋاتقاندەك كۇرىمىز ..... ئەمىلىيەتتىمۇ بىزدە ئىسلام دىنىنىڭ پەقەت قۇرۇق شەكلىلام قالغان ھەممىمىز ئۇزىمىزنى مۇسۇلمان دەيمىز يۇ ئەمما مۇسۇلمانچا ياشىغىنىمىز يوق .... ئاللاھ تائاللاھ ھەممىمىزگە ھىدايەت بەرسۇن مۇسۇلمانلارنى بىرلاشتۇرسۇن  ئامىن ...

سۈرەتتە: بۇ جايدىكى ئاياللارنىڭ ھەممىسى ئۆي ئىچىدە تۇرىدۇ، ئۇلارنىڭ سىرىتقا چىقىشى چەكلەنگەن بولۇپ، پەقەت ئۇلارنىڭ ئۇچرىشىشىغا يول قويۇلغان ئەرلەر بىلەن كۆرۈشەلەيدۇ.
جۇڭگو خەۋەر تورىنىڭ 2-ئاينىڭ 1-كۈنىدىكى خەۋىرى، نۇر تورى تەرجىمىسى:
ئامېرىكا «ۋاشىنگىتون كۈندىلىك پوچتا گېزىتى» نىڭ 1-چېسلا خەۋەر قىلىشىچە، بېشىغا گۈل قىسىدىغان بۇ ئەرلەر ئەرەبىستاننىڭ جەنۇبى، يەمەن بىلەن چېگىرلىنىدىغان خابالا تېغى (Haba La mountain) دا ياشايدۇ. بۇ يەردە سىياسى ھاكىمىيەت يوق بولۇپ، بارلىق ھوقۇقنى قەبىلە باشلىقى يۈرگۈزىدۇ، ھەممە جايدا قەبىلە قائىدىسى ئىجرا قىلىنىدۇ.
