مۇشۇ ئامېرىكىنىڭ باشقا دۆلەتلەرنىڭ ئىچكى ئىشىغا ئارىل ...

ئامرىكا ئەركىنلىكنى تەشەببۇس قىلىدىغان جىگىرى باردۆلەت،،،سىزگە دىسەم ئامرىكا نىمشىقا چاۋشىيەن،ئراق،لىۋىيە،سۈرىيە،ئىران،جۇڭگو،روسىيە،كۇبا قاتارلىق دۆلەتلەر بىلەن تازا خۇشى يوق ،،ھەتتا نەچچە دۆلەتكە ئۇرۇش قوزغاپمۇ باقتى،،چاۋشىيەن مەسلىسىگە كەلسەك چاۋشىيەن ھۆكۈمىتى شەكلى ئۆزگەرگەن پادىشاھلىق تۇزۇمدىكى دۆلەت بوۋىسى دۆلەت قۇرۇپ بېرىپتى ھازىر نەۋرىسىگە قالدى،،كىشىلەرئەركىنلىك دەيدىغان نەرسىدىن مۇستەسنا ياشايدۇ،،چاۋشىيەن پۇقراسى رەھبىرىنى ئەۋلىيا بىلىدۇ،كىشىگە چۇقۇنۇش بەك ئېغىر.ئۇلارنىڭ يادرۇ قۇرالىنى تەتقىق قىلىشىنى ئامرىكا ئەمەس پۇتۇن دۇنيا چەكلەۋاتىدۇ،بۇنىسى ھەممىمىزگە ئايان،ئۇلار دەيدۇ،بىز تىنىچلىق ئۇچۇن ئىشلىتىمىز ،،ئېغىزدىكى گەپكە كىم ئىشىنىدۇ،،چاۋشىيەننىڭ كورىيەگە ھۇجۇم قىلمايدۇ سەپ كىم دىيەلەيدۇ،ھازىر دۇنيادىكى ئەڭ خەتەرلىك چىگرا رايۇن دەل بۇ ئىككى دۆلەت ئارسىدكى38-پاراللىل سىزىق چىگرا رايۇنى بۇ جايدا چاۋشىيەنىڭ نۇرغۇن ئەسكەرلىرى بومبىللىرىنى كورىيەگە قارىتىپ قۇيۇقلۇق ،بىر بۇيرۇق كەلسە كورىيەگە ھۇجۇم قوزغايدۇ،،چاۋشىيەن بىر بىكىنمىچلىك ئىچىدىكى قالاق،جائىل،قاتمال دۆلەت ،پۇقرا دىگەندە شۇ نەگە قارىسا شۇ تۇك باش رەھبىرىنىڭ سۆزىنى خاتىرلەپلا ئۇتىدۇ،،قۇلىدا شۇ بىرتال خاتىرە بىلەن قەلەم بار،،تاس قالىدۇ،باشلىقى يۆتەلسە شۇنىمۇ ‹ئۆھۆ›دەپ خاتىرلەپ ماڭغىلى...ئەمدى ئىراققا كەلسەك ئىراقمۇ شۇ سادامدىن ئىبارەت بۇ مەلئۇن نەچچە مىليۇن كۇرت خەلقىنى قىرغىن قىلىدى،تەخىتتىن چۇشمەي نەچچە ئون يىل ھەممە ھۇقۇقنى ئۆزى تۇتۇپ،ئۇرۇق تۇققالىرىنى دۆلەتنىڭ مۇھىم ئاپاراتلىرىغا قۇيۇپ،دۆلەتنى ۋە خەلقنى ئۆزچاڭگىلىغا ئېلىۋېلىپ خەلق ئۇچۇن ئازراق بولسىمۇ ئىش قىلپ بەرمىدى،،ئازراقلا گەپ قىلغاننى دارغا ئاستى،ئۆلتۇردى،،بۇنداك دۆلەت رەھبىرى ئەلۋەتتە ئۆلۇمگە مەھكۇم،چۇنكى ئۇ نۇرغۇن قان قەرىزگە بۇغۇلغان،!لىۋىيەگە كەلسەك كازافى دىگەن بۇ مەلئۇنمۇ شۇنداق ئۆزىنى پادىشاھ قىلىغان خۇمسى ،،ئاخىرى ئۆز دۆلىتىدىكى ھەققانىيەتنى ياقلىغۇچى كىشىلەر تەرىپىدىن ئۆلتۇرۇلدى،،خەلق ھەممىنى قىلالايدۇ،،ئەركىنلىك ئامرىكا تۇتقان يول ئامرىكا كىمدە كىم ئەركىنلىك ئۇچۇن كۆرەش قىلىدىكەن شۇ نى پۇتۇن كۇچى بىلەن قوللايدۇ،،سۇرىيە مۇ شۇنداق بەشىر دىگەن ئىپلاسمۇ پادىشاھ بۇلۇپ نەچچە ئون يىل ھۇقۇق بەرمەي ھازىرغىچە ‹مەن پادىشاھ›دەپ ئىچكى ئۇرۇش قىلىۋاتىدۇ،،ئەمما ئۇ چۇقۇم مەغلۇپ بۇلىدۇ،،چۇنكى خەلق ھامان مەڭگۈ كۈچلىك!!ئىران بولسا ئوتتۇرا شەرىققە خۇجا بولسام ھەممىسى مىنىڭ گېپىمنى ئاڭلىسا دەيدىغان دۆلەت ،بۇ پارىسلار ئەگەر ئامرىكا قارشى تۇرۇپ،نوخنىسىنى تارتىپ تۇرمىسا ئاتوم بومبىنى چۇقۇم ياسايدۇ،،ياسىمايدىغان ئىش بولسا ب د ت نىڭ يادرو سۆھبىتى تۇلا ئېچىلىپ تۇلا مەغلۇپ بولمايتى،،نۇرغۇن ئاشكارلاشقا تىگىشلىك بولغان تەتقىقات ۋە ئىشلاردىن خەلىقئارادىن يۇشۇرۇپ،كەلدى ئەگەر دەپلا قويسا چاۋىسى چىتقا يېيلىشىدىن قوقىدۇ.جوڭگۇ مەسلىسىگە كەلسەك ھەرقانداق دۆلەت باشقىلارنىڭ ئۆزىدىن ئېشىپ كىتىشىنى خالىمايدۇ،،جوڭگۇ ھازىر دۇنيادىكى ئىككىنچى چوڭ ئىقتىسادى چوڭ گەۋدە يۇشۇرۇن بازار كۈچى بەك كۈچلۇك،،كەڭرى بازار بار،،دۇنيا ئىقتىسادى بۆھراندىن قۇتۇلغان ۋە دۇنياغا ئىجابىي تەسىر قوزغاپ ،دۇنيا ئىقتىسادىنىڭ ئەسلىگە كىلىشىگە تۆھپە قۇشۇپ كىلىۋاتىدۇ،،،شۇڭلاشقا ئامرىكا جۇڭگوغا بەكمۇ ئامراق ئەمەس،،لازىملىق بولسا كىلە بالام دەيدۇ،،بولمىسا بالام چاغان كەلدى بىردەم پوجاڭزا ئېتىپ ئويناپ تۇر!!داداڭنىڭ ئىشى بار تۇلا لوقما سالما دەيدۇ!ئۇنىڭدىن باشقا جۇڭگۇ بىلەن ئامرىكىنىڭ تۇزۇلمە ۋە سىياسى ئىشلىرى بەك ئوخشاپ كەتمەيدۇ،ئامرىكدا ھەممە ئىش قانۇنىي تەرتىپ بۇيىچە ماڭىدۇ،راس ھەقىقىي بۇلىدۇ،،جۇڭگودا بولسا يەرلىك سىياسەتلەر ۋە ھۇقۇق ۋە پۇل بىلەن ماڭىدۇ،،تۇزۇللمىسىمۇ ھەرخىل ،،ھەممىدە دىڭ شىياۋپىڭ بۇۋىسى ئوتتۇرغا قويغان «ئىقتىسادنى مەركەز قىلىش» ،ھەممىدە پۇل بىلەن ئۆلچەم قىلىشنى ئاساس قىلىدۇ...شۇڭلاشقا قىسقىغىنە 30يىلدا دۇنيا بۇيىچە پۇلى كۆپ دۆلەت بۇلالىدى،،ئەمما تۆلىگەن بەدەل تاپقان شۇ پۇلدىن نەچچە مىڭ ھەسسە كۆپ بولدى.،روسىيەگە كەلسەك سۆۋىت پارچىلانغاندىن تارتىپ رۇسىيە بۇلۇنۇپ چىقىپ كەتكەن دۆلەتلەرگە ئۆزىنىڭ ئاكا يوللۇق ئىكەنلكىنى بۇلدۇرۇپ كەلدى،دىمەكچى شۇ 18دۆلەت ئاساسەن رۇسىيەگەبېقىندى،، روسلار بىلەن ئامرىكىنىڭ «سۇغۇق مۇناسىۋەتلەر ئۇرۇشى»دىلا ئۆچ ئاداۋىتى بار ھەم روسىيە ئامرىكا بىلەن تېخنىكا جەھەتتە تەڭ تۇرالايدىغان بىردىنبىر دۆلەت گەرچە رۇسىيەنىڭ پۇلى جۇڭگۇدىن ئاز بولسىمۇ ئەمما تېخنىكا سەۋىيسى جۇڭگۇدىن زور ئۈستۇن،،،شۇڭلاشقا ئامرىكىمۇ روسىيەنىڭ ئۆزىدىن ئېشىپ كىتىشنى خالىمايدۇ،،ئەمما روسىيە بىلەن ھەقىقىي ئۇرۇش قىلىغۇسى يوق،،،ئامرىكىنىڭ كۇبا بىلەن ئەمەس،كۇبانىڭ ئامرىكا بىلەن خۇشى يوق،چۇنكى كۇبا ماركىس ئوتتۇغا قويغان سوتسىيالىزىم بىلەن مېڭىۋاتقان دۆلەت (جۇڭگۇ بولسا ئۆزىنىڭ سوتسىيالىزىمى بىلەن مېڭىۋاتىدۇ)شۇڭلاشقا كاپىتالىزىم بىلەن كىلىشەلمەي ئامرىكا بىلەن پۇت تىپىشىپ كەلدى50 يىل ،،ھازىر ئوباما كاباغا قاراتقان ئىقتىسادى جازانى بىكار قىلىدى ،،بۇ ئامرىكا بىلەن كۇبانىڭ بۇندىن كىيىن ياخشى ھەمراھلاردىن بوپ قېلشىدىن بىشارەت بەردى....مەن پەقەت ئۇمۇمى ۋەزىيەتتىن تەھلىل قىلدىم ،ئەلۋەتتە ئامرىكا خاتا قىلىغان ئىشلارمۇ نۇرغۇن ،،بۇ خاتالىقلارنى ئايرىم تەھلىل قىلىشقا توغرا قىلىدۇ ئەلۋەتتە !
