ماۋۇ ئىنكاسىڭىزغا دىققەت قىلماپتىمەن...ئەمدى بولسىمۇ بىلگىنىمنى يازاي'
«يەنى غەربلىكلەرنىڭ تەرەققىي قىلىشى ئىسلام بويىچە ماڭغانلىقىدا دېگەن قىسىمدا»
بۇ دىگەنلىكىم غەرپلىكلەرنىڭ پەن مەدەنىيەت قاتارلىق ئىجابى تەرەپلەردىكى تەرققىياتىنى كۆزدە تۇتۇپ دىدىم.ياخشى ،توغرا نەتىجە چوقۇمكى ياخشى،توغرا يولدىن كىلىدۇ.بارچە ياخشىلىق توغرىلىق،ھەقلىقنىڭ جەمگاھى دەل ئىسلامدۇر(بۇنى ئالدىنقى ئىنكاسىمدا ئېيتقان)بۇ ھەم مۇشۇ تېمىدا دىيلگەندەك ،كىمىكى ئاللاھنىڭ تەقدىرىگە(يەنى:ئەڭ توغرا قانۇنيتىگە) بويسۇنۇپ تىرشىدىكەن(تىرىششنىڭ ئۈزى دۇئادۇر،دۇئانىڭ چىنلىقى ئەمەلىي ھەركەت بىلەن تېخىمۇ ئىسپاتلانغۇسىدۇر،بۇ دۇئانىڭ ئىجابەت بۇلۇشى ھەققىدە بىر ھەدس بارغۇ،ياكى بۇ دۇنيادا ئىجابەت قىلىدۇ،ياكى ئۇ دۇنياغا قۇرامساق قىلىپ ئېلىپ قۇيىدۇ،ياكى شۇ مىقداردا گۇناھىنى مەغغىرەت قىلىدۇ،مادامىكى ئۇ دۇئايىم ئىجابەت بولمىدى دەپ ئالدىراپ كەتمسىە .....دىگەنگە ئوخشاش )،ئۇ مەيلى كىم قايسى مىللەت قايسى دىندىن بۇلۇشىدىن قەتئىنەزەر رەھمانلىق سۈپىتى بىلەن ،يۇقارقى تىرناق ئىچىدىكى ھەدىستە دىيىلگەندەك خالىغان بىر نەتىجىنى بەرگۈسىدۈر(ئاخىرەتكە ئىشەنمەيدىغانلار نەتىجىنى بۇ دۇنيادىلا ئېلىشنى تىلەيدۇ،شۇ سەۋەپتىن زامانىمىزدىكى فىرئەۋنلەرنىڭ تۇرمۇشى ياخشى ئۆتىدۇ ۋە ياكى ئاخىرەتتە ھېچنىمىسى قالماسلىقى ئۈچۈن،مۇكاپاتىنى بۇ دۇنيادىلا بىرىۋىتدۇ(ئۆيلىرىنى ئۆگزىسىنى ئالتۇندىن كۈمۈشتىن قىلىپ بىرەتتۇق دىگەندەك ئايەتلەرمۇ بارغۇ)
