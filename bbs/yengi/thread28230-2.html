رۇزىمۇھەممەت تۇردى ئارتۇچ مۇئەللىمنىڭ ئانچە ياخشى بولمىغان مۇھىت شارائىتى ئاستىدىمۇ توختاۋسىز تىرىشىش ئارقىلىق مەركىزىي شەھەرلەردىكى نى نى زىيالىلار قىلىپ بولالمايۋاتقان ئەھمىيەتلىك ئىشلارنى قىلىۋاتقانلىقىدىن سۆيۈندۈم.
شەخسەن مەن ئارتۇچ مۇئەللىمنىڭ بۇ خىل تىرىشچانلىقىغا ئاپىرىن ئېيتىمەن ھەمدە ھەر ۋاقىت ئەمەلىيىتىم بىلەن قوللاشقا تەييارمەن.
ئارتۇچ مۇئەللىم ئۈمىد غەنىنىڭ ئانچىكى سۆزلىرىنى ئۇنچە بەك كۆڭلىڭىزگە ئېلىپ كەتمەڭ، ئۈمىد غەنى بەلكىم خەلقىمىزنى قەستەن تېرىكتۈرۈش ئارقىلىق «مەدەنىيەت چۆچۈشى» پەيدا قىلىپ، ئۇيغۇر جەمئىيىتىدە ئويغىنىش پەيدا قىلماقچى بولۇشى مۇمكىن، بۇ مېنىڭ شەخسەن قارىشىم، چۈنكى ھەممە ئىشنىڭ ئىككى تەرىپى بولىدۇ ئەمەسمۇ، مەسلىلەرگە كۆپ تەرەپتىن قاراش، ئىمكان بار مەسلىنىڭ ئوبىكتىپ تەرىپىنى ئويلاش بىر قەدەر ئاقىلانىلىكتۇر.
بۇ قاراشقا كېلىشىمگە «ئۈمىد پاراڭلىرى» پىروگراممىسىنىڭ ئۈچۈنچى سانىغا نەق مەيداندىن قاتنىشىشىم سەۋەبچى بولدى، بۇنىڭغا قاتناشمىغان بولسام مەنمۇ ئۈمىد غەنىگە بىر تەرەپلىمە قارىغان بولاتتىم، بۇ پىروگرامما پات ئارىدا سىلەر بىلەن يۈز كۆرىشىدۇ، شۇ چاغدا ئۈمىد غەنىنىڭ ھەممە قاراشلىرىنىڭلا ئۇنداق بىر تەرەپلىمە ئەمەسلىكىنى ھېس قىلىشىڭلار مۇمكىن....
