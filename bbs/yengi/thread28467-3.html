    بۇنداق ئورۇنسىز تەلەپلەرنى ئەمەلىيەتتە بىر قىسىم رايۇن ، بولۇپمۇ جەنۇبىي شىنجاڭدىكى كۆپ قىسىم مەكتەپ ۋە مائارىپ ئىدارىلىرىدىكى پوق قورساق ، نادان ئەمەلدارلار « زور كۈچ » سەرپ قىلىپ ۋۇجۇتقا چىقىرىپ « ئەمەلىيلەشتۈرىۋاتىدۇ »
    ئوقۇتقۇچىغا شۇنداق مەسئۇلىيەتنامىنى ئىمزالاتقان ۋە شۇنداق قارا قورساقلارچە مەسئۇلىيەتنى ئارتقان مەكتەپ مۇدىرى دىگەن رەزگى بىلەنمۇ ئوقۇتقۇچى - مەكتەپ مۇدىرى ئوتتۇرىسىدا مەسئۇلىيەت ئىمزالاپ ، ئوقوغۇچىدىن چاتاق چىقسا ئوقۇتقۇچىنى تۇتقاندىكىن ، ئوقوغۇچىسى چاتاق چىقارغان ئوقۇتقۇچىنىڭ مەكتەپ مۇدىرىغا ، ئاندىن شۇ مەكتەپ قاراشلىق ناھىيەلىك مائارىپ ئىدارىسىغا ، ئاندىن ۋىلايەتلىككە ، ئاندىن ....مۇشۇنداق قىلىپ قاتلاممۇ - قاتلام چارە كۆرۈپ ماڭالىسىمۇ بىر گەپ ئىدى .
    ئەمما بۇ ساراڭ قېتىش مەكتەپ مۇدىرلىرى خۇددى تولۇقسىز ئوتتۇرا مەكتەپ ئەدەبىيات دەرسىلىكىدىكى « گۇناھكار ئېشەك » ھېكايىسىدەك ، شۇ بىچارە ئوقۇتقۇچىلارغا كۆرسەتمىگەن كۈنى ئاساسەن قالمىدى . يا شۇ مەرەزلەرنىڭ تاناۋىنى تارتىپ ، ئەخلاقلىق ، بىلىملىك ، ھەقىقەتەن مائارىپ ئۈچۈن كۈچ چىقىرالايدىغان ئىلىم ئەھلىلىرىنى مەكتەپلەرگە باش قىلىدىغان « شامال » بۇ تەرەپلەرگە يىتىپ كەلمىدى .
    ئەگەر چۇڭقۇرلاپ تەكشۈرسە ئىدى ، بۇ چۈپرەندىلەردىن « يولۋاس » ياكى « چىۋىن » بولمايدىغىنى بەكمۇ ئاز چىققان بولاتتى ...

مۇشۇ باشلىقلارنىڭ بىر گىپى بار : بۇ دىگەن يۇقىرىدىن چۇش ...

  بۇ دىگەن ئۇ نادان ، بىتەلەت مەكتەپ مۇدىرلىرىنىڭ تېرە تاراقشىتقىنى . 2011- يىلى مەن ئىشلىگەن يېزىدا « قوش تىل » دا ئوقۇۋاتقان ئوقۇتقۇچىلارنىڭ كىرا ياردەم پۇلىنى ئورۇنسىز تۇتۇپ قالدى ، سەۋەبى : ئوقۇۋاتقان بولغاچقا كىرا كەتمەسمىششششش .
   شۇنىڭ بىلەن سۈرۈشتۈرسەم ئاتالمىش « يۇقۇرى » ئورۇنلاشتۇرۇپتىمىش . ئۇ « يۇقۇرى » نەدۇ ؟ دىسەم ، بازارلىق ھۆكۈمەت باشلىقىكەن . ئۇ قارا تىل باشلىقتىن سورىسام ، ناھىيىلىك تەشكىلات بۆلۈمنىڭ « ھۆججىتى » شۇنداقمىش . ئۇ بۆلۈمىنىڭ مۇئاۋىنى مىنىڭ ئاغىنەم ئىدى ، سۈرۈشتۈرسەم ، ئۇنداق ئىش يوق .
    شۇنىڭ بىلەن ناھىيىلىك ئىنتىزام تەكشۈرۈش بۆلۈمىگە قاراپ ماڭغاندا مىنىڭ پەيلىمدىن خەۋەر تاپقان مائارىپ ئىدارىسىنىڭ باشلىقى ماڭا تېلىفون قىلىپ ئىدارىغا چاقىرتىپ ، سۆھبەتلىشىش ئارقىلىق 31 نەپەر ئوقۇتقۇچىنىڭ 1000 يۈەندىن قانۇنسىز تۇتۇپ قېلىنغان پۇلى قايتۇرۇلغانىدى .
   شۇڭا ، بۇ رەزگىلەرنىڭ يۇقۇرىنىڭ نامىنى سۈيئىستىمال قىلىپ « يۇقۇرىنىڭ بۇيرۇقى » دىگەنلىرىنىڭ كۆپىنچىسى يالغان . ( بۇ مۇتلەق ئەمەس ، ھەممە جايلار ئۇنداق بولماسلىقى مومكىن ) چۈنكى ھېلىقى ئۆزىنىڭ زىيالىيلىقىنى ئۇنۇتقان ياكى زىيالىي سانالمايدىغان ئوقۇتقۇچىلار ئۇ « يۇقۇرى » نى ئىچكىرلەپ سۈرۈشتە قىلىپ تەكشۈرۈپ باقمايدۇ ئەمەسمۇ؟!
