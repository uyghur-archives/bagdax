ياپونىيىدە مىدېتسىنا ئوقۇۋاتقان لۇشۇن مۇ ئەينى ۋاقىتتا بىر كىنو دىن خەنزۇلارنىڭ مىللى روھىدىكى ئېغىر زەئىپلىكنى كۆرۈپ ،كەسپىنى ئۆزگەرتىش قارارىغا كەلگەن ئىكەن ،ئۇنىڭ شۇ قېتىمقى كەسكىن تاللىشى ،خەن...لەرنىڭ مىللى ئويغۇنىشىدا مىسلىسىز رول ئوينىغان ،،،،ئەركىن ئاكىمىز پۈتۈن زېھنىنى ئالەم قاتنىشى تەتقىقاتىغا قاراتقاندا  ،بىزدەك بۇ پىيادە يۈرگەن قەۋىملەرگە نىسبەتەن ھازىرچە زۆرۈرىيىتى يوق دەپ ھېس قىلدىم ،،،بۇنىڭدا ئاساس نەپ ئالغۇچى ئامرېكا بولىدۇ ،،،ئەلۋەتتە ،،،بىراق ئامېرىكا زوڭتۇڭلىرىنىڭ ئاغزىدىكى مەنتىقىي ئىبارە بىلەن كۆڭلىدىكى پوتېنسىال ئاڭ ئوخشىمايدىكەن ،،،ئامرېكىنىڭ تاشقى سىياسىتىدىكى نەيرەڭۋازلىغىنى ئاساسەن چۈشىنىپ قالدۇق ،،،،ئۇ «دۇنيا ساقچىسى »ئەمەس ،بەلكى پەقەت ئامرىكا مەنپەتىنى قوغداش «ساقچى»سىكەن ،،،ئازراق نەپ كەلسىلا "قاتىل "بىلەنمۇ ئېغىز-بۇرۇن يالىشىپ يۈرىۋېرىدىكەن ،،،،
بىلىمخۇمار ئاكىنىڭ ئەجرىگە رەھمەت .
