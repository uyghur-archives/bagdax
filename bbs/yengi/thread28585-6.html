يوللىغان ۋاقتى 2015-2-27 00:21:11 |ھەممە قەۋەتنى كۆرۈش
ھەقىقەت ئىگىلىدۇ سۇنمايدۇ،ئىنساننىڭ بىر ياراتقۇچىسى بولىدۇ كىشىلەر ئۇ ياراتقۇچىنى تاشلاپ باشقا بىر قولىدىن ئىش كەلمەيدىغان تاشلارغا ئىبادەت قىلسا بۇنى توسۇش كىرەك چۈنكى بۇ دۇنيانى بىر ياراتقۇچى يارىتىپتۇ،ئۇ يارىتىپ بزىگە دىننى يەر يۈزىگە تارقىتىشنى بۇيرۇدى شۇڭا ئۇنىڭ دىگىنىدەك قىلىمز،ئىنسان يا ھەقىقەت ئۈستىدە يا قاھالەت ئۈستىدە ياشايدۇ،ئىسلام دىنى كىشىلەرنى بۇتلارغا چوقۇنۇشتى ئىنسانلارنى ياراتقان تۇپراقنى سۇ بىلەن سۇغارغان كىشىلەرنى ھاۋا بىلەن نەپەسلەندۈرگەن ئۈستىدە ئاستىدا ھىچقنداق كۆتۈرۈپ تۇرىدىغان تىرەك بولمىغان يەرشارىنى كائىناتتا مۇشۇنداق تۇرغۇزالىغان ھەم قۇياشنى يارىتىپ يەرشارىنى ئىسسىقلىق بىلەن تەمىنلىگەن مۇشۇنداق مۆجىزىلىك مۇشۇنداق ئادەمنىڭ ئەقلى يەتمەيدىغان ھاياتلىق سىرىلىرنى ئورۇنلاشتۇرغان،ياراتقۇچىغا ئىبادەت قىلىشقا بۇيرۇيدىغان دىن،مەيلى باتىل دىن بولسۇن ياكى ھەق دىن بولسۇن چوقۇم شۇنىڭ بەرپا بولۇشى ئۈچۈن يەنە بىر دىن يوقايدۇ،مەسلەن جۇگودا گومىنداڭ يوقۇلۇپ كومپارتىيە ھاكىميەتكە چىقتى،ئۇلارنىڭ ئۇرۇش قىلىش شۇ كومپارتىيىنى ھاكىميەت يۈزىگە چىقىرىش ئۈچۈ،بۇمۇ ئىرقى قىرغىنچىلىق بولامدۇ،ىىرىستىيانالر مەيدانغا كەلگەندىمۇ پۇتپەرەسلەر ئالدى بىلەن خىرىستىيانلارغا زۇلۇم قىلىپ ئۇلارنى ئۆلتۈرگەن ئۇلار خىرىستىيان دىنى ئۈچۈن،بۇتپەرەسلەر بىلەن ئۇرۇشقان بۇمۇ ئىرقى قىرغىنچىقما،يەھۇدىيلارمۇ شۇنداق،تارىختىن بۇيان بىر مەنتىقىنىڭ ياكى ئىدىينى چوڭراق قىلساق بىر دىننىڭ پارتىينىڭ بەرپا بولۇشى ئۈچۈن يەنە ىرىس يوقايدۇ،ئاندىن بۇ يەرىدىكى قاسىنىڭ ھەق قايسىنىڭ ناھەق مەسىلىسگە قارايمىز،ئىسلامغا قارىسىان ھەق نمە ئۈچۈن چۈنكى ئىسلام بۇيرۇدىغىنى ھەققە بۇيرۇيدۇ يەنى ئىنساننى ئۆزلىرى ياسىۋالغان تاشقا ئىبادەت قىلىشدىن ئزۆلىرىنى ياراتقان بىرلا ئىلاھقا ئىبادەت قىلىشقا بۇيرۇيدۇ پۇتپەرەسلەر بۇنى قوبۇل قىلماي مەۋجۇداتلارنىڭ ياراتقۇچىسىغا قارشى چىقتى ھەم ئىسلامنىڭ مۇقەددەس جايى بولغان كەبىگە ئاۋال ئاللاھقا شىرىك كەلتۈرىدىغان تاشلارنى تىكلىدى شۇڭا ئۇلارنى چىقىپ تاشلاش كىرەك،چۈنكى ئىسلام ھەقدۇر،يەنە مەككە ئۇرۇشىدا رەسۇلۇللاھ كىچىك بالىلارنى ئاياللارنى قىرىلارنى تەسلىم بولغانلارنى ئۆيىگە كىرۋالغانلارنى ئۆلتۈرمەسلىككە بۇيرۇغان بۇ چاغدا پەقەت شۇ قولىغا قۇرال ئىلىپ قارشىلىق قىلىدىغان ئاز كىشىلەرلا قالغان ھەم رەسۇلۇللاھ مەككە ئىچىدە ئادەم ئۆلتۈرمەسلىكنى بۇيرۇغان بۇ ھەرگىز قىرغىنچىلىق ئەمەس قارشىلىق قىلغانلانى ئۆلتۈرۈشكە بۇيرۇغان يەنە بىرقىسىم ئىلگىرى مۇسۇلمانلارغا ئىنسان قىلىپىدىن چىققان ئەسكىلىكلەرنى قىلغان كىشىلەرنى كەبىنىڭ يۇپۇقىغا ئىسىلۋالسىمۇ تۇتۇپ ئۆلتۈرۈشكە بۇيرۇغان،بۇ ھەرگىزمۇ ئىرقى قىرغىنچىلىق ئەمەس،ئىشنىڭ جەريانىغا قارايمىز بىرسى بزىگە زۇلۇم قىلسا بزى ئۇنىڭدىن قىساس ئىلشىمىز ئۇنىڭغا قارشى چىقىشىمزى بۇ ھەقدۇر ئىرقى قىرغىنچىلىق بولمايدۇ،مۇشرىكلار ئاۋال رەسۇلۇللاھنى ئلتۈرمەكچى بولۇپ ئاخىرى يۇرتىدىن قوغلاپ چىقاردى يەنە بولدى قىلماۇ مۇسۇلمانلارنىڭ مەككىدە قالغان ئائىلىسىدىكىلەرنى ئۆلتۈردى مۇسۇلمانلارنىڭ كارۋانلىرىغا ھۇجۇم قىلدى ئاندىن رەسۇلۇللاھ ئۇلارغا ھۇجۇم قىلىپ بۇتلارنى چاقدى،بۇ رەسۇلۇللانھىڭ تەرجىمھالىدا ئىنىق بار،رەسۇلۇللانھىڭ بۇنداق قىلىشى پۈتۈن ھەق ئۈچۈندۇر ئىسلام نىمىنى سۆزلەيدۇ،بۇ قۇرئاندا ئىنىق كۆرسۈتۈلگەن ھەققانىيەت ئۈچۈن ئۇرۇش قىلغانلار ئىرقى قىرغىنچى بولسا ئۇنداقتا يىقىنىقى زاماندىكى سۆۋەت قىزىل ئارمىيسى چۇڭگۇنىڭ ياپۇنغا قارشى تۇرغان ئارمىيەلىرى كىيىنكى گومىنداڭ بىلەن ئۇرۇشقان ئارمىيەلىرىئىرقى قىرغىنچى بولمامدۇ ھەممىسىنڭ مەخسىتى نىمە، شۇ ئۆزىنىڭ ئىدىيسىگە يول قيمىاچقا ئزكىنى ئىدىيسىنى قوغداش ئۈچۈن ھەم كومىنزىمنى جڭگوغا ھاكىم قىلىش ئۈچۈن ئۇرۇش قىلغان،بۇمۇ ئىرقى قىرغىنچىقما؟ئۇمۇ بىردىن دىن دىگەن كىشىلەرنى ئۆزى ئوتتۇرغا قويغان ئىدىيەگە تشەببۇس قىلىش بۇ مگجبۇرلاش شەكىلدە بولسۇن ياكى ھاكىميەتنى قولىغا ئىلشى شەكىلدە بولسۇن ھەممىسى بىر مەقسەت ئۈچۈن شۇنداق قىلىدۇ،شۇ دىن ھىساپلىنىدۇ،جۇگومۇ گومىنداڭ دەۋىرىدكى كومپارتىيەگە ئىسكىلىك قىلغان جىنايەتچىلەرنى جازاغا ھۆكۈم قىلغن يەنە ياپۇنلارنى ئۇرۇش جىنايەتچىلىرىمۇ بار يەنە گىرماننىڭ جىنايەتچىلىرىچۇ،بۇلارنى جازالاشمۇ ئىرقى قىرغىنچىلىق ھىساپلىنامدۇ؟دۇنيادا كىشىلەر مەلۇم نىشاننى ئۆزىگە باشلامچى قىلىپ ياشايدۇ،لىكىن شۇ كشىلەرنى ىر ئىلاھ ياتقان شۇڭا شۇ ئىلاھ ماڭا ئىبادەت قىل دىگەن ھەم شۇنى باشقىلارغا تارقىتىشقا بۇيرۇغان.بزى شۇنىڭغا بويسۇنىمىز،بۇ دۇنيانىڭ ئۆزلىكىدىن پەيدا بوپ قىلشى قانۇنيەتلىرىنىڭ ئۆزكلىكىدىن تۈزىلىشى مۇمكىنمۇ؟ئىنسانلارنىڭ ئۆزلىكىدىن نەپەسلىنەلەيدىغان بولۇشى ئىچكى ئەزالىرىنىڭ شۇنداق مۇكەممەل بولۇشى،مۇشۇنداق ھەركەت قىلالايدىغان بولۇشى مۇمكىنمۇ؟ئاددى مىسال ئالساق بىر ماشىنا ئۆزلىكىدىن ماڭمايدۇ چوقۇم ئۇنى بىر تىخنىك ياسايدۇ ئاندىن بىر شوپۇر ھەيدەيدۇ ئۇنىڭ شوپىرىنى كۆرمىسەكمۇ بىرسىنىڭ ئۇنى باشقۇرۋاتقانلىقىغا چوقۇم ئىشىنىمزىغۇ.بىر بىنامۇ ئۆزلىكىدىن پەيدا بولمايدۇ ئۇنى چوقۇم بىرىس ياسىغان چوقۇم شۇنىڭغا ئىشىنىمزى ئۇنداق بولغاندىكىن ئىنسانمۇ دۇنيامۇ ئۆزلىكىدىن پەيدا بولۇپ قالمىغان چوقۇم ئىندانغا مۇشۇنداق ھاۋانى بىرەلەيدىغان دۇنيانىڭ مۇشۇنداق قانۇنيەتلىرىنى بىكىتىپ باشقۇرىدىغان بىرسى بولىدۇ،ئۇ بىزنىڭ ياراتقۇچىمىز بولىدۇ ئۇ دەل ئاللاھدۇر چۈنكى ھازىغا قەدەر بىرە زات مەن ئىنسانلارنڭ ئىلاھى دەپ باقمىدى پىرئەۋىن دىگەنىدى ئاخىرى ھالاك بولدى،شۇڭا بزىنىڭ ئىلاھىمىز دەل ئالللاھدۇر،ئاللاھ ئىنسانلارنى توغرا يولغا باشلاش ئۈچۈن پەيغەمبەرلەرنى ئەۋەتتى،بۇ دىگەن بىر سىناق دۇنياسى يەنى ئىنسانلار دۇنياغا يارىلىپ ھەرخىل  سىناقلارغا دۇچ كىلىدۇ بۇنىڭ بەزىسى شەيتان ئارقىلىق بولىدۇ بزىسى ئاللاھنىڭ ئىرادىسى بىلەن بولىدۇ،ئىنسنالر ئۆلگەندىن كىيىن يەنە بىرمۇنچە ئىنسانلار مەيدانغا كىلىدۇ،ئۇلارمۇڭ سىنىلىدۇ،ئەڭ ئاخىرىدا بۇ دۇنيانىڭ ۋاقتى توشىدۇ ۋە ئىنسنالر يارالغاندىن تارتىپ تۇغۇلغان بارلىق ئىنسالار يىغىلىپ دۇنيادا قىلغان ئىشلىرىدىن ئاللاھقا ھىساپ برىدۇ ئىنسان تىنۋالاي دىسە ئۇنىڭ ئىككى دولىسىدىكى پەرىشتىلەر خاتىرلەپ ماڭغان يەنە كىلىپ قىلغان ئىشلىرنى ئىنساننىڭ پۇت قوللىرى كۆزلىرى گۇۋاھلىق بىلىدۇ تىشمزى دىگەن جەسەت مۇھىمى بىزنىڭ روھىمىز،ئاللاھ دۇنيادا ئۆزىنىڭ دىگنىنى قىلغان توسقۇنىدىنيانغان كىشىنى ئەبەدى قالىدغان ئۆزلمەيدىغان جەننەت بىلەن مۇكافتالايدۇ،ئۇنداق قىلمىغان دۇنيادىكى ئاللانھىڭ سىناقلىرىدىن ئۆتەلمەي ئاللانىڭ كۆرسەتمىسىدىن يىراقلاپ كەتگەن كىشىلەرنى لاۋۇلداپ كۆيگەن دوزاخ بىلەن جالايدۇ ئۇ يەردىمۇ ئىنسان ئۆلمەيدۇ،تىرىلىر يىڭىلىنىپ تۇرىدۇ مەڭگۈ ئازاپلىنىدۇ،ئاللاھ كىشلەرنى توغرا يولغا باشلاش ئۈچۈن بەيغەمبەرلكەرنى ئەۋەتتى زامان ئاخىرىدا ئەڭ ئاخىرقى پەيغەمبەرنى ئەۋەتتى ئۇ مۇھەەممەت ئەلەيسالامدۇر،مۇھەممەت ئەلەيسالام ئۆزىنىڭ پەيغەمبەر ئىكەنلىكىنى شۇ دەۋىردە مۆجىزىلەر بىلەن ئىسپاتلىغان مەسلەن ئاينى ئىككىگە بۆلۈش دىگەندەك مۆجىزىلەر چىققان ھەم يەنە ىر مۆجىزىسى بولسا ئۇنىڭ سۆزى ئۇ شۇ چاغدا ئىنساننىڭ ئانىنىڭ قوسقىدىكى ھالىتىنى بىرمۇ بىر سۆزلەپ بەرگەن مەسلەن ئىنسان ئانىنىڭ قوسقىدا 40 كۈن مەنىلىك 40 كۈن لەختىە قانلىق 40 كۈن بىر لوقما گۆشلۈك ھالەتتە تۇرىدغانلىقىنى 1400 يىل ئىگىرى ئىيتىپ بەرگەن،ئۇ چاغدا ھازىرقىدە بىچاۋ سىتى ئاپراتلىرىنىڭ بولۇشى مۇمكىن ئەمەس رەسۇلۇللاھ بۇنى قانداق بىلگەندۇ ئۇنى چوقۇم شۇ ئىنساننى ياراتقان ئىلاھ ئىيتىپ بەرگەن بۇ ئۇنىڭ راس ئىكەنلىكىنىڭ دەلىلى،يەنە ىرىس ئايال ئىرىدىن ئاجىرشىپ كەتسە 40 كۈن ئىددەت تۇتىدۇ 40 كۈندىن كىيىن توي قىلسا بولىدۇ ھازىر ئالىملار تەكشۈرۈپ ھەقىقەتەن 40 كۈندە ئايالنىڭ بەدىنىدن ئىرىنىڭ سۇيۇقلىقى پۈتۈنلەي تازلىنىپ بولىدىغانلىقى ئىسپاتلكاندى شۇنداق بولغان باشقا ئەرگە تەگسە ئۇنىڭ بويىدىكى بالا ئىلگىركى ئىرىنىڭ پۇشتىدىن بولمايدۇ،مۇشۇ سەۋەپتىن بىر كاتتا مىدتسىنا يەھۇدى مۇسۇلمان بولغان،بۇ بىر دەلىل،شۇنداقى دەلىللەر تۇرسا ئاللاھ باركەن رەسۇلۇللاھ ھەقەن دەپ ئىشىنىمىز،ئۇنىڭغا ئىشەنگەندىكىن ئۇ نىمىگە بۇيرۇسا شۇنى قىلىمزى ئۇنىڭ سۆزلىرىگىمۇ ئىشىنىمىز،مۇسۇنداق كاتتا دۇنيا قانۇنيەتلىرىنى ياراتقان ئىنسانلارنى ياراتقان ئاللاھ قانداقمۇ يالغان گەپ قىلسۇن،شۇنىڭ ئۈچۈن رەسۇلۇللاھ ئاللاھنىڭ سۆزى بويچە ئىش قىلىپ ئاۋال دىننى تارقاتتى  ئاندىن بۇھەق دىننى مەككىگە ۋە باشقا جايلارغا ھاكىم قىلدى، بۇ ھەق بىلەن باتىل ئوتتۇرسىدىكى ئۇرۇش شۇڭا بىرسى ھەق بىرسى باتىل ھەقنىڭ قىلغىنى تاجاۋۇزچىلىق بولمايدۇ،ئۇ ئىنسنالارنىڭ خاتا قارشىنى سۈپۈرۈپ ياراتقۇچىغا ئىبادەت قىلىشقا بۇيريدىغان قانۇننى تىكلەيدۇ،بۇ ھەرگىزمۇ تاجاۋزچىلىق بولمايدۇ،ئالللاھقا ئىشەنگندىكىن ئاللاھنىڭ ھەممە سۆزلىرىگە ئىشىنىشىمىز كىرەك.

يوللىغان ۋاقتى 2015-2-27 09:01:52 |ھەممە قەۋەتنى كۆرۈش
كىچىكىمىزدە ھەممىمىز مۇنداق ھادىسەگە  ئاڭلاپ باققان، ياكى يولۇقۇپ باققان بولىشىمىز مۇمكىن. بىر سۇخەنچى ئەركىنگە دەيدۇ: «-ئەركىن، ئەزمەت سەندىن قورقمايمەن دەيدۇ، سەن ئەزمەتتىن قورقامسەن؟» ئەزمەت دەيدۇ: «ياق قورقمايمەن» سۇخەنچى ئەزمەتكىمۇ ئوخشاش گەپ قىلىدۇ. ئۇمۇ «قورقمايمەن» دەيدۇ. شۇنىڭ بىلەن ئەركىن بىلەن ئەزمەت ئۇرۇشۇپ كېتىدۇ. سۇخەنچى بەزىدە ھەتتا ئۇلارنى يۈزمۇ-يۈز تۇرۇپمۇ ئۇرۇشتۇرۇپ قويالايدۇ. شۇ ئۇرۇش تۇپەيلى ئۇلار ئۇزۇنغىچە دۈشمەنلىشىپ ئۆتۈشى مۇمكىن.
ئويلاپ باقايلى بۇ يەردە بىز تىگىنى چۈشىنىپ يىتەلمەيدىغان، ئەرزىمەس بىر مەسىلە ئۈچۈن ئۇرۇشۇشنىڭ ھاجىتى بارمۇ؟ ئۇلار ئىناق دوست بولۇپ ئۆتسە بولمامدۇ؟ ...
سۇخەنچىلەر مەسىلىنىڭ نېگىزىنى ھامان چۈشەندۈرۈش مۇرەككەپ، ياكى ئېنىقلاپ چىقىش تەس نوقتىلارغا باغلايدۇ. ھەم پائالىيىتىنى بۇنداق ئىنچىكە نوقتىلارنى ئانچە چۈشىنىپ كەتمىگەن توپ ئارىسىدا ئېلىپ بارىدۇ. ئۇنىڭ ئۈستىگە: يا يالغان پاكىت ئويدۇرىدۇ، ياكى ئەسلى پاكىتنى يوق قىلىۋېتىپ ئاندىن سۇخەنچىلىك قىلىدۇ.
ئىنسانىيەت ئارىسىدىكى ئىتتىپاقسىزلىق ۋە قان تۆكۈلۈشلەر ئەنە شۇنداق سۇخەنچى شەيتانىزىم شەيتانىستلىرىنىڭ ھىلە ۋە نەيرەڭۋازلىقلىرى تەرىپىدىن بولىۋاتىدۇ.
مۇنداق قاراپ باقسام مۇنبەرلەردە دىن، پەن، پەلسەپە ھەققىدىكى تېمىلا بولسا كىرىپ سۇخەنچىلىك قىلىپ، داىندارلار، پەندارلار، تەپەككۈردارلارنىڭ ئارىسىغا زىددىيەت سېلىپ قويۇپ چىقىپ كېتىدىغانلار كۆپ ئىكەن. ئۇلارنىڭ نە دىندىن، نە پەندىن، نە پەلسەپىدىن خەۋرى يوقكەن. مۇنازىرىگە قېتىلىشتىكى مەقسىدى: يا مۇنازىرىدە «يېڭىش» ئەمەس، (چۈنكى، ئىزچىل پىكىر قاتناشتۇرمايدۇ)؛ يا «كۆرۈش، ئۇچۇر ئىگەللەش» ئەمەس، (چۈنكى، ئۇچۇر ئىگەللەۋاتقان ئادەم قاپ بەلدىلا مازاق گەپلىرگە چۈشمەيدۇ)؛ يا «سوراپ بىلىۋېلىش» ئەمەس، (چۈنكى، مەنبە ياكى مۇناسىۋەتلىك كىتاب-ماتېرىيال سورىغىنى يوق).
مەيلى قەستەن بولسۇن ياكى بولمىسۇن، سۇخەنچىنىڭ قىلغىنى پەقەت بىرلا: «زىددىيەت سېلىش؛ ئايرىش؛ قارىلاش». شۇڭا، كۆپچىلىك مۇنازىرە جەريانىدا سۇخەنچى ۋە سۇخەنچىلىكتىن قاتتىق ھەزەر ئەيلەيلى. «يولداش» سۇخەنچىلەرمۇ بارمىش ئاڭلىسام.
