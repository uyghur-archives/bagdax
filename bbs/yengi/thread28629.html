    دائېبېدو فرانسس كېرې بۇركىنا فاسودىكى كېچىك بىر يېزا ـ گاندودا چوڭ بولغان. ئوقۇش ئۈچۈن دەسلىۋىدە ئۆز دۆلىتىدە يۇرتىدىن ئايرىلىشقا مەجبۇر بولىدۇ، كېيىن گېرمانىيىدىكى بېرلىن تېخنىكا ئۇنۋېرسىتىتىغا ئوقۇشقا بېرىپ يەنە بېر قېتىم ئۆز يۇرتىدىن ئايرىلىدۇ. ئۇ ئۆزىنى بېقىپ چوڭ قىلغان مەھەللە-يۇرتىغا جاۋاب قايتۇرۇشقا بەل باغلايدۇ. بۇنىڭ ئۈچۈن ئۇ ئارخېتېكتورلۇقنىڭ قۇدرىتىدىن پايدىلىنىدۇ.
    ئۇ ئۆزى قۇرغان شىركەت ــ كېرى قۇرۇلۇش شىركىتى ئارقىلىق، ئۇ ئۆز دېققىتىنى يەرلىك قۇرۇلۇش ماتىرياللىرى ۋە تېخنىكىسىنى زامانىۋى ئۇسۇلدا قوللىنىشقا قارىتىپ، شۇ ئارقىلىق ئۆز ئەتىراپىدىكى مۇھىتقا قايتىدىن ئېنىقلىما بەرمەكچى ۋە قايتىدىن لاھىيلەپ چىقماقچى بولىدۇ. بۇركىنا فاسوغا نىسبەتەن ئېيتقاندا، بۇ ئاساسلىقى سېغىز توپىدىن قۇرۇلۇش قىلىشتىن دېرەك بېرىدۇ،  ئۇنىڭ يېڭىچە ئۇسۇلى ئاشۇ ئەتىراپتىكى مۇھىتقا شۇنچە سىڭىشكەن ئاجايىپ گۈزەل، زامانىۋى بىنالارنى ۋۇجۇدقا كەلتۈردى.
ئۇ سالغان تۇنجى قۇرۇلۇش ـ گاندودىكى بىر باشلانغۇچ مەكتەپ ـ 2001 ـ يىلى پۈتكەن بولۇپ، ئاغاخان ئارخېتېكتورلۇق مۇكاپاتغا ئېرىشكەن. شۇنىڭدىن كېيىن ئۇ يۇرتىدا باشقا قۇرۇلۇشلارنىمۇ لاھىيلەپ چىقىدۇ، ئۇنىڭ يۇرتىدىكى باشقا قۇرۇلۇشلىرىمۇ نۇرغۇنلىغان مۇكاپاتلارغا ئېرىشكەن، بۇلارنىڭ ئارىسىدا شىندلېر خەلقئارالىق مۇكاپاتى، ماركوس مۇكاپاتى  ۋە شىۋىتسىيە ئارخېتېكتورلۇق مۇكاپاتى شۇنداقلا يەنە خولكىم خەلقئارالىق ئارخېتېكتورلۇق مۇكاپاتىمۇ بار.
          كېرې يەنە ئارخېتېكتورلۇق ئوقۇتۇشى بىلەنمۇ شۇغۇللىنىدۇ. ئۇ مېلۋوكىدىكى ۋىزكانسىن ئۇنۋېرسىتىتى ۋە خارۋارد ئۇنۋېرسىتىتى قاتارلىق نۇرغۇنلىغان ئالىي مەكتەپلەردە نۇتۇق سۆزلىگەن. 2013 - يىلىدىن باشلاپ،  شىۋىتسارىيىنىڭ مېندېرىسو ئارخېتېكتورلۇق ئاكادېمىيىسىدە ئوقۇتقۇچىلىق قىلىۋاتىدۇ.
