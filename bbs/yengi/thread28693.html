بۇ رەڭ سىزگە قانداق كۆرۈندى؟ ھەققىدەئويلىغانلىرىم


    تاشپۇلات ئاكىمىزنىڭ بۇ رەڭ سىزگە قانداق كۆرۈندى دېگەن تېمىسىنى ئوقۇپ ئۇزۇندىن بىرى كاللامغا كىلىۋالغان ۋە ئۆز ئەمەلىيتىمدە سىناق قىلىپ باققان رەڭ توغرىسىدىكى بۇ سىرغاجاۋاب تاپقاندەك بولدۇم.

بۇرەڭ سىزگە قانداق كۆرۈندى دېگەن تېمىنىڭ ئادرىسى:

./thread-28644-1-1.html

رەڭ توغرىسىدا ئىزدىنىش ئۈچۈن ئاۋال نۇر توغرىسىدا چۈشەنچە ھاسىل قىلىشىمىز كېرەك.رەڭ دەيدىغان نەرسە نۇرغىلا خاس بولغان ئاتالغۇ،ھەممىمىزگە ئايان بولغىنىدەك بىزنىڭ يەرشارىنى نۇر ۋەئىسسىقلىق بىلەن تەمىنلەۋاتقان قۇياش يەتتە خىل رەڭلىك نۇردىن تەركىب تاپقان،يەرشارىدا بىزنىڭ كۆزىمىزگە رەڭگارەك كۆرۈنگەن ھەممە نەرسەئاشۇ قۇياشتىن كەلگەن يەتتە خىل رەڭلىك نۇرنىڭ ئارلاشمىسىنىڭ شۇ جىسىمدا ئەكىس ئىتىپ قايتقاندىن كېيىن،بىزنىڭ مىڭىمىزدە قۇبۇل قىلىنغان سانلىق مەلۇماتلاردۇر،يىغىپ ئەكەلسەك ھەممىسى رەقەملەردىن تۈزۈلگەن پىروگراممىلاردۇر.كۈندۈزى بىزگە رەڭگارەڭ كۆرۈنگەن نەرسىلەر كېچىسى قارا كۆرىنىدۇ،بىزئاشۇ قارا جىسىملارغا نۇرچىقارغۇچى ئۈسكىنىلەردىن پايدىلىنىپ ھەرخىل رەڭدىكى نۇرلارنى چۈشۈرسەك(مەسىلەن:قىزىل،كۆك،يېشىل،ئاق،سېرىق)ئۇنىڭ بىزنىڭ كۆز ئالدىمىزدائەكىس ئىتىدىغان رەڭگى ھەرخىل بۇلىدۇ،نۇرنىڭ ئاساسىي زەرىچىسى فوتۇندۇر،فوتۇن ئېلىكتىر ماگىنىت دولقۇنىدىن شەكىللەنگەن يۇقۇرى ئىنىرگىيەلىك زەرىچە.ئېلىكتىر ماگىنىت دولقۇنىنىڭ تەۋرىنىش قېتىم سانى ۋەئۇزۇن قىسقىلىقىغا ئاساسەن نۇرنىڭ كۈچلۈك ئاجىزلىقىدائۆزگۈرۈش بولىدۇ،بىز  ئادەتتە تۇرمۇشىمىزدا ئىشلىتىۋاتقان،ئىنفىرا قىزىل نۇر،لازىر نۇر،رىنتىگىن نۇرى،ھەممىسىنىڭ مەنبەسى بىر،ئەمما ئېلىكتىرون تەۋرىنىشى ئوخشىمىغانلىقى ئۈچۈن بۇ يەردە ھەرخىل نۇرلار شەكىللىنىدۇ،بىز  ئاشۇنۇرنىڭ تەۋرىنىشى ۋە ئۇزۇن قىسقلىقىنى يىشىپ چىقىپ تورمۇشىمىزدا پايدىلىنىش ئۈچۈن ئوخشىمىغان دولقۇنسۈزگۈچ،تەۋرىنىشنى مۇقۇملاشتۇرغۇچ لامپىلارئارقىلىق بۇ نۇرلارغائىرىشەلەيمىز،ئەمما قۇياشتىن كەلگەن بۇ نۇرلارنى بىز ھېچبىر ئۈسكىنە ئىشلەتمەستىن بىۋاستەقۇبۇل قىلىپ ئىشلىتىمىز،ئېلىكتىر ماگىنىت دولقۇنى قۇياشنىڭ ئۆزىدە پىششىقلاپ ئىشلىنىپ بىزگە يىتىپ كېلىدۇ،بىزگە زىيانلىق نۇرلارئاتمۇسفىرا قاتلىمىدا پارچىلىنىپ چېچىلدۇ،تەۋرىنىشى،ۋە سۈرئىتى ئاستىلاپ بىز قۇبۇلقىلالىغۇدەك دەرىجىدىكى پايدىلىقنۇرغا ئايلىنىدۇ.جىسىم ماددىدىن تەركىب  تاپىدۇ،بۇ مادىدىلار تۈزۈلىشىگە ئاساسەن بەزى نۇرلارنى قايتۇرىدۇ،قارا رەڭنى مىسالغائالساق بۇقۇياشتىن كەلگەن بارلىق نۇرنى ئۆزىگە سۈمۈرۋالىدۇ،شۇنىڭ بىلەن بىزنىڭ مىڭىمىزدە ئاشۇقارا رەڭ ئەكىس ئېتىدۇ.ئالەم بوشلۇقىدىكى ئاشۇ قاراڭغۇ ماددا بىلەن بىز تۇرمۇشىمىزدائۇچراتقان قارا ماددىلار ھەممىسى بىر ماددىلاردۇر،ئالەم بوشلۇقى ئاشۇ قارا ماددىلاردىن تۈزۈلگەنلىكى ئۈچۈن نۇر باشقا رەلىك ماددىغا يۇلۇققانغاقەدەر ئاشۇ قارا مادداتەرپىدىن سۈزۈۋېلىنىدۇ.

بىرقىزىل گۈلنى مىسالغا ئالايلى：
    ئۇنىڭ ئاساسىي تەركىبى قىزىل ماددا،ئۇقۇياشتىن كەلگەن قىزىلنۇرنىلار قۇبۇل قىلىپ باشقا نۇرنى تاقىتىۋېتىدۇ،بۇ قىزىل نۇر بىزنىڭ مىڭىمىزدە قىزىل رەڭلىك سانلىق مەلۇماتنى شەكىللەندۈرىدۇ.بۇجاھاندىكى ھەممە نەرسەئۇچۇر تارقاتقۇچى ۋە ئۇچۇر قۇبۇل قىلغۇچىدۇر،نۇر ۋە ئېلىكتىر ماگىنىت دولقۇنى ئۇچۇر توشىغۇچىدۇر.بۇ ئىككىسى بىر بىرىگەئۆزگىرەلەيدۇ.ئىككىسىنىڭ مەنبەسى بىردۇر.ھاۋادىكى رادىئو دولقۇنىنى بىز ئاڭلىيالمايمىز،ئەمما ئۇنىڭ مەۋجۇتلۇقىنى بىلىمىز.بىز ئالاھىدە ئۈسكىنىلەردىن پايدىلىنىپ بىزگە لازىملىقىنى يىشىپ ئىشلىتىمىز،مەسىلەن يانفۇن شۇنىڭغائوخشاش باشقا مەۋجۇداتلارمۇ ئۆزىگە لازىملىق نۇردولقۇنىنىلا قۇبۇل قىلىدۇ.باشقىسى بىلەن ئىشى يوق.بىز تۇرمۇشىمىزدا ئىشلىتىۋاتقان نۇر كابىل بۇنىڭ مىسالى،يىڭنىنىڭ تۈشىكىدىن كىچىك تۈشۈكتىن نەچچە مىليون نۇرماڭىدۇ،ئەمما بۇ بىربىرىگە ئارلىشىپ  كەتمەي،ئۆزى لازىملىق يەرگەبارىدۇ.

ئەمدى ئاددىي مىساللارنى كەلتۈرەي،ھەممە ئادەمنىڭ چىرايى ئوخشاش

1-  ئۇ قۇبۇل قىلغان نۇرئوخشىمايدۇ.
2-  سىزنىڭ مىڭىڭىز يىشىپچىقارغان ئۇچۇر ئوخشىمايدۇ.
    سىزنىڭ كۆزىڭىزگە چىرايلىق كۆرۈنگەن ئادەم يەنە بىر ئادەمنىڭ كۆزىگە سەت كۆرىنىدۇ.مەلۇم بىر ئادەم سىزنىڭ كۆزىڭىزگە بىرخىل سۈرەتتە كۆرۈنسە،يەنەبىر ئادەمنىڭ كۆزىگە باشقا بىر خىل سۈرەتتەكۈرنىدۇ،سىزنىڭ ئېڭىڭىزدىكى مەلۇم بىرخىل رەڭ ئۇنىڭ كۆزىگە باشقا خىل رەڭدە كۆرىنىدۇ.مەسىلەن بىر قىزىل رەڭ سىزنىڭ كۆزىڭىزگە قىزىل كۆرۈنسە باشقا بىرسىنىڭ كۆزىگەكۆك كۆرىنىدۇ.ئەمما ئۇ يەنىلا ئۇ رەڭنى قىزىل دەپ بىلىدۇ.ھەممىمىز بىلىمىز ئاسماننىڭ رەڭگىنى كۆك دەپ،توغرا ھەقىقەتەنمۇكۆك.ئەمما ئۇ بەزى كىشىلەرنىڭ كۆزىگە ئاق ۋە ياكى يېشىل كۆرىنىدۇ،ئەمما ئۇ يەنىلا ئۇنى كۆك دەپ بىلىدۇ،ئۇنى مىڭىسى قۇبۇل قىلغان ئۇچۇر ئۇنىڭغا ئاشۇ يېشىل ياكى ئاقرەڭدىن كۆك دەپ ئۇچۇر بەردى،ھېچ كىم باشقىلارنىڭ كۆزىگە بىر شەكىلنىڭ قانداق سۈرەتتە كۆرۈنىدىغانلىقىنى بىلمەيدۇ.
     مەسىلەن سىز بىر دوستىڭىزغا بىرنەرسىنى كۆرسىتىپ ئۇنىڭدىن ئۇنىڭ كۆزىگە ئۇ نەرسىنىڭ قانداق سۈرەتتە كۆرىنگەنلىكىنى سورايسىز،ئۇ جاۋاب بىرىدۇ،ئۇنىڭ جاۋابى ھەقىقەتەن توغرا،سىزنىڭ كۆزىڭىزگىمۇشۇنداق كۆرىندى.
    ئەممائىككىڭلارنىڭ مىڭىسى قۇبۇل قىلىدىغان سۈرەت باشقا باشقا،سىز ياخشى كۆرگەن قىز سىزنىڭ كۆزىڭىزگەشۇنداق چىرايلىق كۆرىنىدۇ،ئەمما باشقىلارنىڭ كۆزىگە سەت كۆرىنىدۇ.ياراتقۇچىسىزنىڭ قەلبىڭىزدىكى نۇر قۇبۇل قىلغۇچى پىروگراممىنى ئازراقلا ئۆزگەرتىپ قويۇش ئارقىلىق ئۇنى سىزگە ئەڭ چىرايلىق شەكىلدە كۆرسىتىدۇ،سىز شۇ نۇرنىڭ گۈزەللىكىگەمەھليا بۇلۇپ،شۇ نۇرنىڭ ئىشقىدا ئۇنىڭغا كۆيىسز،ئەگەر ياراتقۇچى ئۇنىڭدىكى ئۇنۇرنى ئېلىۋەتسە ياكى ئازراق ئاجىزلىتىۋەتسە ۋەياكى سىزنىڭ نۇر قۇبۇل قىلغۇچى پىروگراممىڭزنى ئۆزگەرتىۋەتسە ئۇ سىزنىڭكۆزىڭىزگە سەت كۆرۈنىپ كېتىدۇ،ئۇنى تىللاپ كېتىسىز،ئۆچ بۇلۇپ كېتىسىز.چۈنكى سىزدە ئۇ نۇرنى قۇبۇل قىلغۇدەك سېستىما يوق ياكى ئۇنىڭدا نۇر يوق.بەزى كىشىلەرنىڭ كۆزىگە جىنلاركۆرىنىدۇ،بۇنى تەتقىق قىلىش ئۈچۈن يەنە مۇشۇ نۇر ئىلمىغا مۇراجەت قىلىشقا توغرا كىلىدۇ.جىن ئوتنىڭ يالقۇنىدىن يارالغان مەخلۇقاتتۇر،ئوت دىمەك نۇر دىمەك،ئوتنىڭ يالقۇنى ئۇنىڭ ئەڭ كۈچلۈك كۈيۈش نوقتىسىدا،ئوت يالقۇنىنىڭ تىمپىراتۇرسى يۇقۇرى ھەم شۇنىڭغا ئەگىشىپ نۇرىمۇ بىر قەدەر كۈچلۈك.ئوتمۇ كۈچلۈك ئاجىزلىقى تارقىتىدىغان نۇرىغا ئاساسەن نەچچەتۈرگە بۈلىنىدۇ.بىز جىنلارنىڭ قايسى خىل تۈردىكى ئوتتىن يارالغانلىقىنى بىلمەيمىز،ئەمما ئۇمۇ بىر مەخلۇقات بولغانلىقى ئۈچۈن ئۇنىڭ ئۇچۇر تارقىتىدىغان ھەم قۇبۇلقىلىدىغىنىنى بىلىمىز.بىز ئۇنى قايسى چاستوتىدىكى ئېلىكتىر ماگىنىت دولقۇنىدىن پايدىلىنىپ ئۇچۇر ئالماشتۇرىدىغانلىقنى بىلەلىسەك،بىز ئۇ ئاۋاز دولقۇنىنى ئالاھىدەئۈسكىنىلەردىن پايدىلىنىپ يىشىپ چىقالايمىز،بەزى جىنكەش موللىلارئەپسۇنلاردىن پايدىلىنىپ ئۆزىنىڭ ئاڭلاش دائىرسىنى جىنلار ئالاقىلىشىدىغان دولقۇن دائىرسىگە يۆتكەيدۇ.كۆرۈشمۇ شۇنىڭغا ئوخشاش

1- جىنلاركۆزىگە كۆرۈنمەكچى بولغان كىشىنىڭ نۇر قۇبۇل قىلىش سېستىمىسىنى قالايمىقان قىلىدۇ.
2-سىزنىڭنىرۋا سېستىمىڭىزقالايمىقانلىشىپ نۇر قۇبۇل قىلىش سېستىمىڭىز قۇبۇل قىلغان نۇرنىڭ دولقۇن ئۇزۇن قىسقىلىقىدا ئۆزگىرىش يۈز بىرىدۇ.
3-ئاللاھ سىزگە كۆرسىتىشنى ئىرادەقىلىپ سىزنىڭ كۆرۈش پەردىڭىزنى ئېچىۋېتىدۇ.بۇ پەردە نۇر سۈزگۈچى پەردىدۇر.

مۇشۇئۈچ خىل
سەۋەبنىڭ بىرى ھازىرلانغاندا سىز جىننى كۆرەلەيسىز.
