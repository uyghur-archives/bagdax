ئۆزىدىن كېتىپ قالدى ، بۇنىڭغا يەنە نېمە دېگۈلۈك ئەمدى ، مۇشۇنداق تۇرسا يەنە نېمە مىش مىش پاراڭ بۇ . قومۇل تارىخىدا ئەل يۇرت ئۈچۈن قان بەدەلنى جىق تۆككەن ، ئەلكۈننىڭ غېمىنى تولا يېگەن ۋاڭلار بەگلەر تەيجىلەر دېگەن تولا گەپ ، بۇلارنىڭ ئىچىدە مىسالەن ئەبەيدۇللا ۋاڭ ھەم ئۇنىڭ بۈيۈك ئاتىسى مۇھەممەت شاھ غازىبەگ ، يەنە شۇ بېشىر ۋاڭ ھەم ئۇنىڭ سۆيۈملۈك رەپىقىسى مېھرىبانۇ ........ لېكىن ئەپسۇس تۇڭگانلارنىڭ بەييەنخۇ يېغىلىقى باشقىچە نەتىجە يارىتىپ قالدى ، ئەي ...... تۆتۈر پېشانە بېشىر ۋاڭ ......... سەركە يىقىلسا خەلققىمۇ ئاپەت ئېلىپ كېلىدىغان ئىشكەن . شۇ يېغىلىقىدا قومۇلدا تۈزۈك ئاادەم قالماي ئاخىرى قومۇل ئىچى تىنىچلانغاندا ھايات قالغانلارنىڭ ناھايىتى بىر ئاز قىسمىلا ئاندىن قومۇلغا قايتىپ كېلەلىگەن ...... تاراتىمۇ قالغايتىمۇ ئىشقىلىپ مۇشۇ ئىككى تاغلىق يۇرتتىن بىرسىدە ، ئاڭلاشلارغا قارىغاندا بەييەنخۇ ئەسكەرلىرىگە ئىلھاملاندۇرۇش يۈزىسىدىن ھەر بىر ئادەمنىڭ بىردىن قۇلىقىنى كېسىپ كەلسەڭلار ئانداق مانداق مۇكاپاتلايمەن دېگەچكە كىچىككىنە بىر تاغلىق مەھەللىدىلا بىر سۇمال ( يوغان خالتا ، سوبو دابو ) قۇلاق كېسىلگەن ، شۇنىڭدىنلا كۆرۈۋېلىشقا بولىدۇ قومۇلدا ئەسلى ئۇيغۇر سانى ئۇنچىلىك ئاز ئەمەسكەن ، ئەسلى جىقكەن . تۇرپان قومۇل بىلەن قوشنا ھەم تەقدىرداش بولغانلىقى ئۈچۈن كېيىن بەييەنخۇنىڭ قوشۇنى ھەم ئۇلارنىڭ كۆپلىگەن ئۇرۇغ تۇغقانلىرى تۇرپانغا كەلتۈرۈلۈپ .........بىر ئىشلار بولغان . شۇڭا خوجىنىياز ھاجىلار زامانىدا تۇڭگانلارنىڭ بىز ئۇيغۇلارغا ئۆچمەنلىك بىلدۈرۈپ تۇڭگان يېغىلىقىنى ئېلىپ بېرىشى بەلكىم پەرىزىمچە مۇشۇ مەنىدىن بولسا كېرەك دەپ ئويلايمەن ...... بۇغۇ ئەمدى بىر تارىخقا ئايلىنىپ بەئەينى بىر  تىلسىمات بوپكەتتىغۇ ، شۇڭا بۇ توغرىسىدا زىيادە گەپ قىلغۇم يوق ....... بۇ بەختىيار زاماندا ،  ئۇلارمۇ بىزگە ئوخشاشلا مۇسۇلمان ..........

     باركاللاھ ! قۇمۇلدىن خېلى ئەزىمەتلەر چىقىدۇ!...قەشقەر، خوتەن، ئاتۇش، ئاقسۇ، ئىلى، تۇرپان ئالتە شەھەردىن ئوخشاشلا ئەزىمەتلەر چىقتىغۇ : مەھمۇد كاشىغەرى، يۇسۇپ خاس ھاجىپ ، ئەلنامە ئالائىدىن خوتەنى، توغلۇق تۆمۈرخان، ھەزىرتى تەجەللىي، موللا موسا سايرامى، موللا ئەخمەت لۈكچۇنى( قىسۇرى) ، ئابدۇخالىق ئۇيغۇرى، ئەخمەتجان قاسىمى لار...شۆھرەت مۇتەللىپ، خالمۇرات غوپۇر، ئەركىن سىدىق، رىشات ئابباس، تاشپۇلات روزى، تۇرسۇنجان ....لارنىڭ ھەممىسى بىز ئۇيغۇرلاردىنغۇ!
    گەپ بۇيەردە قۇمۇل ۋاڭى ۋە قۇمۇل ھەققىدە بولىۋاتىدۇ، ۋاڭلارنىڭ يۇرتنىڭ ئىشىكىنى ساقلاپ بەرگىنىگە، قۇمۇلدا «مۇساپىرلار قەۋرىستانلىقى» قىلىپ بەرگىنىگە، جەنۇپتىن قۇمۇلغا پانالىنىپ كەپقالغان قېرىنداشلارنى نەچچە ئاي يىللاپ «گادايلار سېڭى» دىن چىقىم قىلىپ باققىنىغا، كەتكىسى كەپقالسا مەخسۇس ئادەم ئاجرىتىپ نەچچە كۈنلۈك يەرلەگىچە ئۇزۇتۇپ قويغىنىغا، قۇمۇلدا قالغانلىرىنىڭ چوڭ بايلاردىن بولۇپ كەتكىنىگە...رەھمەت ئېيىتساق بولىدۇ!  ھازىرچۇ، ئۆتكۈر ئەپەندىنىڭ شېئىرىنى قىبلىنامە قىلىپ ھەممە يەرگە ئېسۋالغاننى دېمەي تۇرايلى، قۇمۇللۇقلار بولمىسا « قىزىل راۋاقتىكى چۈش» قاتارلىق كىلاسسسىكلارنىڭ تەرجىمىسى شۇنچە ۋايىغا يىتەرمىدى، ئۇيغۇرچە بىلىدىغان خەنزۇ ۋە ئېنگىلىز مۇتەخەسسىسلەر ھەيراتتا قالارمىدى!  تويلاردا نەنىڭ ناخشىسىنى ئېيىتساڭلار قىزىۋاتىدۇ، سەھنىلەردە كۆپرەك نەنىڭ لاھىيلىگەن كېيىملىرى يارىشىۋاتىدۇ، تېلۋېزۇرلاردا كىملەر لاھىيەلىگەن سەھىپىلەر قارشى ئېلنىۋاتىدۇ، «ساغلاملىق يېتەكچىسى» نى نەلىكلەر باشقۇرۋاتىدۇ، «باغداش» تورىنىچۇ، «ئۇيغۇر تېبابەتچلىكى» جۇرنىلنى نەلىك تەسسىس قىلىپ باشلاپ بەرگەن، 7جىلىتلىق گىگانىت ئەسەر«ئۇيغۇر تېبابەت قامۇسى» نى نەلىك رۇياپقا چىقارغان، نەچچە يۇز مۇتەخىسسىس نەچچە يىلدا ئاران تاماملىغان، ئاسپرانىتلار ئوقۇپ تاماملىمىسا بولمايدىغان 9 خەنزۇچە گىگانىت ئەسەرنى نەلىك ۋايىغا يەتكۈزۇپ بىۋاستە خەنزۇچە يېزىپ بىر يىلدا تاماملىغان.....؟
     ھەممە يۇرتنىڭ ئۆزىگە تۇشلۇق تۆھپىسى بار، ئارتۇقچىلىق، كامچىلىقلىرى بار! بىر-بىرىمىزنى تولۇقلايلى ، يۇرتۋازلىق قىلىپ بىر-بىرىمىز سوكمەيلى، بىر ئۇيغۇرنىڭ دەپ ئۆز-ئارا ئۈگىنەيلى ، مەدھىيلەيلى!
        ھەرگىزمۇ ئۈمىد غەنىدەك مەھمۇت كاشىغەرى، يۇسۇپ خاس ھاجىپتەك ئۇلۇغ ئالىملىرىمىزغا تىل تەككۈزىدىغان مۇناپىقلاردىن بولمايلى!
        قەھرىمانلار خەلقنىڭ غېمىنى يىسە، قەھرىمانلانىڭ بېشىغا كۈن كەلگەندە بىز خەق قاراپ تۇرمىغان بولساق ، شىڭشىسەي دېگەن جاللات خوجا نىياز ھاجىنى يۈرىكىنى قاپتەك قىلىپ ئۆلتۈرەلمىگەن بولاتتى !
