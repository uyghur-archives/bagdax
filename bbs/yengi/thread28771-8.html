ئۇنداق بولسا يېڭى ئىدىيە دىگەن نىمە؟ئىسلاھاتنى ئۇيغۇرلار ھازىرقى ۋەزىيەتتە ئۆزىچىلا باشلاپ ئۆزئالدىغا بىر ئىشلارنى قىلىش ئىمكانىيتى بارمۇ؟ئۇيغۇرلار زادى نىمىنى ئىسلاھ قىلىدۇ؟خاتىرجەم بىر ئۇسۇرغىلى بولمايۋاتقان بىر جەمىيەتتە ئۇيغۇرلار زادى نىمىنى قىلسا قىلالايدىغان ئىمكانىيەت بارمۇ؟بەزىلەر دىگەندەك ئۇيغۇرلارنىڭ ئىسلام دىنى تەرەققىياتقا راستىنلا پۇتلىشىۋاتامدۇ؟قانداق يېرى پۇتلىشىۋاتىدۇ؟سىز مۇسۇلمان بولسىڭىز (ئۆزىڭىز مەن مۇسۇلمان دەپسىز)مۇسۇلمان نىمە قىلىش كىرەك؟قانداق قىلغاندا ئاللاھقا يارايدىغان مۇسۇلمان بولغىلى بولىدۇ؟پۈتۈن دۇنيادىكى بارلىق ئىنسان ۋە جىنلار گۇرۇھى بىرلىشىپمۇ بارلىققا كەلتۈرەلمەيدىغان،ۋۇجۇتقا چىقىرالمايدىغان قۇرئان كەرىم  قۇلىمىزدا تۇرسا ئۇنىڭغا ئەمەل قىلماي ،يەنە نىمىنى ئىسلاھ قىلىمىز؟................سۇئاللار بەك كۆپ،ئالدى بىلەن مۇشۇ سۇئاللارغا بىر جاۋاپ بېرىڭ،ئاندىن ماۋۇ بىرنەچچە سولتەكنىڭ چاۋىكىغا ئەمەس مۇتلەق كۆپ ساندىكى مۇنبەرداشلارنىڭ ئىنكاسىغا قاراپ داۋامىنى يېزىشنى ئويلىشارسىز،مۇشۇ ماۋزۇ راستىنلا جەلپ قىلارلىق بىر ماۋزۇ بوپتۇ،لىكىن ھازىرقى ۋەزىيەتتە بۇ ھەقتە بىر نىمىلەرنى يېزىش پەقەت  قۇرۇق ئاۋارە بولۇش ياكى مۇسۇلمانلىقنى جايىدا قىلالمىغانلىقنىڭ ئىپادىسى بوپ قېلىشى مۈمكىن،چۈنكى ئۇيغۇرنىڭ ھازىر مۇشۇ باشقىلارنىڭ كۆزىگە سەت كۆرىنىپ قېلىۋاتقان دىنىدىن باشقا بىرنىمىسى قالمىدى.....بۇنىمۇ ئىسلاھ قىلىپ ئۇيغۇرلار تەرەققىيات ئۈچۈن قىپ-يالىڭاچ بوپ كەتسىمۇ ئاخىرىدا ئېرىشىدىغىنى يەنىلا گۇمراھلىق بولىدۇ......چۈنكى پۇل دىسە خۇتۇنىنى سۇلاپ بېرىپ،ھۇقۇق دىسە ئانىسىنى سېتىپ،ئەمەل دىسە ئاچا-سىڭىللىرىنى دەسمى قىلىىۋاتقانلارنىڭ ئېرىشكىنى بۇ دۇنيانىڭ كىچىككىنە لەززىتى بولسىمۇ كۆرىۋاتىمىز....ئادىمىيلىكتىن چىقىپ ھايۋاندەك ياشاۋاتىدۇ،نە ئەدەپ-ئەخلاق،نە ئىنساپ ۋىجدان،نە ئادىمىيلىك...................پۇل تاپقان بىلەن كۈنلىرى دەككە دۈككىدە.....،مۇشۇ ئىگىز بىنانى،ئىسىل پىكاپنى،بانكىدىكى نۇرغۇن پۇلنى تەرەققىيات دەپ چۈشىنىۋالغانلارنى دوراپ ،شۇلارغا ئوخشاش تۇرمۇشنى ئىستەپ بىر بەلەن ئېتىقاتىنى تەرەققىياتقا پۇتلاشتۇرۇپ يەنە تېخى نۇمۇسمۇ قىلماي باشقىلارنى مۇشۇنىڭغا تەشەببۇس قىلىشنىڭ ئۆزىلا مۇرتەدلىككە ماڭغانلىق..................قالغىنىنى يەنە ئىنكاس شەكلىدە ئورتاقلىشارمىز...............

پاسسىپ ئەمەسمەن. گەپنى تومتاق قىلماي، سەۋەبىنى ئوتتۇرىغا قويۇپ باقمامسىز؟ سىز يېزىق ئۆزگەرتىشنىڭ ئاقىۋىتىنى تەسەۋۇر قىلالامسىز؟ ئوتتۇرا ئەسىرلەردىن ئۈزۈلۈپ قالغان كىلاسسىكلىرىمىزنى ئەمدى مۇشۇ يېزىقتا رەتلەپ بىر قۇر تېپىپ چىقتۇق. ئەمدى ئەشۇلارنى يەنە يوق قىلىپ، ئەۋلات بوشلۇقى شەكىللەندۈرەي دىمەكچىمۇسىز؟ كىيىنكىلەرگە سىز نىمە قويماقچى؟ سىزدەكلەردە جەمىيەتكە نىسبەتەن مەسئۇلىيەت دەيدىغان نەرسە مەۋجۇتمۇ؟ سىزدەكلەرنىڭ دىگىنى بىلەنلا ئۆزگىرىپ قالمايدۇ.
خەنزۇلار بىزدەك داۋاملىق يېزىق ئۆزگەرتمىگىنى ئۈچۈنلا بەشمىڭ يىللىق تارىخىم بار دىيەلەيدۇ. ھازىرقى ئەۋلاتلىرى ئۆزىدىن ئۈچ - تۆت ئەۋلات ئىلگىرىكى ئەۋلارلىرىنىڭ تەۋەرۈكلىرىنى تەرجىمىسىز، توسالغۇسىز كۆرەلەيدۇ. سىزنىڭ ھازىر دەۋاتقان گېپىڭىزنىڭ خاراكتېرىنى چۈشۈنۈپ كەلسىڭىز تەرەققىيات ئەمەس، ئۇيغۇرنىڭ جىسمىنى قويۇپ روھىيىتىنى يوق قىلىش دىگەن گەپ. كىم سىزگە پۇل بېرىپ مۇشۇنداق قۇتراتقۇلۇققا سېلىۋاتىدۇ. مۇشۇ ئېقىمنى ياقلايدىغانلىرىڭلارنىڭ ھەممىسى ئېنىقلا ئۇيغۇرغا خائىن.
ئەقىل كۆزىڭىز بىلەن قاراپ، ھەزىم قىلىپ كۆرۈڭ. يېزىق ئۆزگەرسىلا پۈتۈن بىر قەۋىم يېڭىچە تەپەككۈر، يېڭىچە مەدەنىيەتكەن ئېرىشىپ قالىدىغان ئىش يوق. قابىليىتىڭىز يار بەرسە كۆز قارىشىڭلارنى ئىلمىي يوسۇندا ئەمەلىي ۋە نەززەرىيىۋىي يوللار بىلەن ئىسپاتلاڭ. سىزمۇ پاسسىپمۇ؟ دەپ سورالغان تومتاق سوئالىڭىز سىزگە بولغان تونۇشۇمنى 180گرادۇسلۇق يۆنۈلۈشكە بۇراپ قويدى. ئەھۋالدىن قارىغاندا ئارىمىزدا سىز ۋە سىزدەكلەرنى ئۆز ئىچىگە ئالغان تەبرىقچى بۆلگۈنچىلەر باردەك قىلىدۇ.
