  نۇري تېيىپ
كەپتەرلەر -(17)


   بىزنى دېرىزە تەرەپكەجايلاشقان بىر ئۈستەلگە باشلىدى كۈتكۈچى. ھەر خىل تازىملارنى قوبۇل ئەتكەچ،ئۈستىلىمىزگە كېلىپ جايلاشتۇق.
    بۈگۈن ئىككىمىزلا ئايرىم ئولتۇرۇپ، بىر پاراڭلىشايلى دېدىم. بەزى تەلەپلىرىمنى، ئۇنىڭغا قارىتا سېنىڭ پىكىرلىرىڭنى ئاڭلاپ باققۇم بار. جىسمانىي جەھەتتە، ساڭا ھېچقانداق قايىل بولمايدىغان يېرىم يوق. ئەلۋەتتە يۈرەك، جىگەر، جاسارەت جەھەتتىمۇ ھەم .... ھا ھاھا ... باشقىچە ئويلاپ قېلىۋاتامسەن نېمە؟ مەن سېنى يامان ئىشقا سالىمەن دېگىنىم يوق جۇمۇ ...
     ئۇ ئادەم ئورنىدىن يېرىم تۇرۇپ، سوزۇلۇپ كېلىپ مۈرەمگە يېنىك ئۇرۇپ قويدى-دە، قېلىن بىر كونۋېرتنى ئۈستەلگەقويۇپ، ئۇنى ئۈستەلگە سۈركىگىنىچە ئالدىمغا سۈرۈپ ئېلىپ كەلدى.ئاز بولسىمۇ، بۇنى تۇرمۇشۇڭغا ئىشلەت، يەتمىگىنىگە يەنە مەن بار.كېلەر ھەپتىنىڭ ئاخىرى،توكيودا بىر سودا كېلىشىمى بار. ئەگەر "ھە" دېسەڭ، مەن بىلەن بىللەبارغىن. ياپون جەمئىيىتىنى بىر كۆرۈپ باقمامسەن ... ھا ھا ھا... قىزىق ئويۇنلارمۇ بوپ تۇرىدۇ. ئاستا - ئاستا كۆنسەڭ، ئەتە- ئۆگۈن ياپوننىڭ ئوتتۇراقىسمىغىمۇ بىللە بېرىۋېتەمدۇق تېخى ....
    مەن كونۋىرتقا قاراپ ئۇزۇن  ئويلىنىپ كەتتىم، بۇ ئاز دىگەندىمۇ ئون مىڭ دوللارغا يىقىن پۇل بولۇپ مىنىڭ نەچچە ئايلىق ئىش ھەققىمگە تەڭ كىلەتتى.
بۇ پۇلنى ئەپ قويسام، كېيىن بۇ ئادەم نېمە دېسە، شۇنى قىلماي بولمايدىغان ئەھۋال يۈز بېرىدۇ. قانداق قىلىش كېرەك؟ئۇ ئادەمنىڭ قىلىۋاتقان گەپلىرىنى ئاڭلىغاچ، كاللامنى كەلگۈسىدىكى ئىشلارغايۈگۈرتۈپ كۆردۈم. سوئال - جاۋابلار كاللىنى بىر ئالدى.خۇلاسە يەنىلا، "ئېلىشقابولمايدۇ" بولۇپ چىقتى.
مېنىڭ ھازىرچە تاپاۋىتىم ئۆزۈمگە يېتىۋاتىدۇ. تېخى بانكىدا، ئىقتىساد قىلغان ئازراق پۇلۇممۇ بار. قېنى،كېيىنچە تۇرمۇشتا قىينىلىپ قالسام، تارتىنمايلا سىزنى ئىزدەي، دەپ كونۋېرتنى ئۇئادەم تەرەپكە سۈردۈم.
    ھا ھا ھا... كۆپ ئويلاپ كەتتىڭ سەن. بوپتۇ، كۆڭلۈڭنىمۇ چۈشىنىۋاتىمەن. سېنىڭ ئويلاۋاتقىنىڭمۇ توغرا. قېنى ئازراقتىن ئىچىشەيلى ئەمسە … دېدى ئۇ ئادەم،كونۋېرتنى ئۈستەلنىڭ يان تەرىپىگىرەك سۈرۈپ قويۇپ.تاغدىن - باغدىن پاراڭلىشىپ،ئاخىرى يەنىلا ئىككى ئەركەك بىر يەرگە كەلسە قىلىشمىسا بولمايدىغاندەك پاراڭلاردىنمۇ قىلىشىۋەتتۇق.
    – بېيجىڭغا بىللەبارالامسەن؟
    ئۇ ئادەم، مەن  پاراڭغا، باشچىلاپ كىرىپ، بەخىرامان سۆزلەپ تۇرغان ۋاقتىمدا،تۇيۇقسىزلا سوراپ قالدى. مەن بۇ ئادەمنىڭ باياتىن ياقى ئىچىۋاتقان ھاراق -پىۋىلىرىنى ئويلاپ، ئاللىقاچان كەيپى توشۇپ "ئوينايمىز تېخى ... ھە قۇيە ...ئەمدى ئىچىمەن...مەن ..."دېگەن يەرگە كېلىپ بولدى،دەپ ئويلاپتىمەن. ئۇنداق ئەمەسكەن.
    بايىقى شوپۇر يىگىت، سەن،مەن ئۈچىمىز بارىمىز. بېجىرىدىغان ئىشلار بار. سەن ماڭا تەرجىمانلىق قىپ بەرسەڭ،ئۇنىڭدىن كېيىن بەزىبىر ئىشلاردا يېنىمدا تۇرۇپ بەرسەڭلا بولدى. ئىش ھەققىڭنى دېگىنىڭدىن ئاشۇرۇپ بېرىمەن.ئۇ ئادەم ھېچ بىر ھاراق ئىچكەندەك ئەمەسلا، ئەستايىدىل پاراڭغا چۈشۈپ كەتتى. مەن سەل ئېھتىيات قىلىپ سۆزلەشكە ئۆتتۈم.
    بىللە بېرىشقا، ھازىرلا ۋەدەبېرىۋەتسەممۇ بولىدۇ، لېكىن ئىشلەۋاتقان ئاشخانىدىن ئۇزۇن مۇددەت ئايرىلالمايمەن.ئادەم كەمچىل. مەنمۇ ئۇ ئادەمگە ئەستايىدىللىق بىلەن خىزمەتتىكى ئۆزقىيىنچىلىقىمنى ئېيتتىم.
    ئۇ كىچىك ئىشلار. مەن ئورۇنلاشتۇرىمەن. خاتىرجەم بول. تېز بولسا بىر ھەپتە، بولمىسا ئون كۈنگە قالماي قايتىپ كېلىمىز. مەنمۇ ئۇ خەقنىڭ مايلىق تاماقلىرىنى ئىككى كۈنلا يېسەم"ھۆ" بولىدىكەنمەن قارا... يەنىلا مۇشۇ ئۆزىمىزنىڭ تامىقىغا يەتمەيدىكەن... ھا ھا ھا ...خىزمەت ھەققىدىكى پاراڭلارمۇشۇنچىلىكلا بولدى. ئۇ ئادەممۇ مەندىن "زادى قانداق قىلىسەن؟ بارامسەن،بارمامسەن؟ بۈگۈنلا جاۋابىنى بېرىسەن" دەپمۇ كەتمىدى.
ئۇ ئادەم مېنىڭ قارشى چىقمىغىنىمغا قاراپلا، قوشۇلدى دەپ ئويلىدىمۇ ياكى بۇ خىلدىكى ئادەملەرنىڭ سۆھبەتلىرى مۇشۇنچىلىكلا بولامدۇ، ئۇ تەرىپى ماڭا نامەلۇم.باشقاپاراڭلار ئۇلغىيىپ، بۇ تېما ياپوندىن ھالقىپ كورېيە، فىلىپپىن، تەيۋەن،تايلاندلارغىچە كەتتى.
     ئەمما يەنىلا، گەپ ئارىلىقلىرىدا بەزى بىر سوئاللار تۇيۇقسىز سورۇلۇپمۇ تۇردى. ھېلىقى يىگىت يىراقتاتۇرۇپلا ئىشارەت قىلىۋىدى، ئۇ ئادەم:
– ھە ئاياقتىكى سورۇن تەييار بوپ قالغان ئوخشايدۇ. قوپە، يەنە ئازراق ئويناپ، ئاندىن بولدى قىلايلى بۈگۈن، دېدى-دە، ئورنىدىن تۇردى.
رېستوراننىڭ بۇلۇڭ تەرىپىگەجايلاشقان تولىمۇ قاراڭغۇ، تولىمۇ سۈرلۈك ئايرىمخانىغا كىرىپ كەتتۇق....
ئەتىسى باشلىرىم زىڭ - زىڭ،كۆزلىرىم قىپ - قىزىل ھالدا، ئۆزۈمنىڭ ئۆيىدە ئۆزۈم بىلەن ئارانلا كۆرۈشتۈم.ئاخشام قانداق، قانچىدە، كىم بىلەن ئۆيۈمگە قايتىپ كەلگىنىمنى ئەسلىيەلمەي،"ئاچچىق - چۈچۈك سۇيۇقاش بولغان بولسا - ھە، مۇشۇنداق ۋاقىتتا ..." دەپ غودۇڭشىغاچ تاماكامنى ئىزلىدىم.
يانچۇقتا تۇرۇپ پۇرلۇشۇپ كەتكەن تاماكامنى تۈزەشتۈرۈپ، تۇتاشتۇرۇپ:
  ئاخشام ئاخىرى قانداق بولغان بولغىيتتى. ئايرىمخانىغا كىردۇق ... ھە، ئۇ ئېسىمدە ....، "خوشە خوشە"دېدۇق. ئۇمۇ ئېسىمدە. ئۇنىڭدىن كېيىن، ناخشا ئېيتقاندەكمۇ قىلىمىز جۇما. ئاخىرى...؟؟؟
   ئاخىرىنى ھېچبىرئەسلىيەلمىدىم. يانچۇقۇمنى كولىسام، ئىسىم كارتۇشكىلىرى ئوندىن ئېشىپمۇ كېتىپتۇ.بىرىنى تونۇمىدىم.باشلار غاراڭ - غۇرۇڭ،" ئىككىنچى ئىچمەيمەن، ئۈچىنچى ئىچمەيمەن" دېگەن گەپنى قايتىلاپ، ئۆيگەكىرىپ، تالاغا چىقىپ دېگەندەك قىلىپ، بىر كۈننى ئاران ئۆتكۈزدۈم. لېكىن، ئۇ كۈنكى ئىككىنچى، ئۈچىنچى ئىچمەيمەن دېگەن ۋەدىلەر، شۇ كىچىككىنە ئۆيۈمدىلا قالدى.
     شۇنداق قىلىپ، مەن بۇمەھەللىمىزدىكى سۈرلۈك ئادەم بىلەن، خېلى قەدىناس ئاغىنىلەردىن بولۇپمۇ قالدىم.ھەپتىدە بىر، بەزىدە ھەتتا ھەپتىدە ئىككى قېتىم بىر يەرلەرگە بېرىپ قويىدىغانمۇ بولۇپ قېلىۋاتىمەن. توكيونىڭ گۈزەل كوچىلىرىدىكى پۇلدارلار يىغىلىپ، تاماشاقىلىدىغان جايلار، ماڭا ئاستا - ئاستا "ھە ئۆتكەندە كەلدۇققۇ بۇ يەرگە"دەيدىغان يەرگىمۇ كەپ قالدى.كۆپ ۋاقىتلاردا مەن، ئۇئادەم بىلەن يانمۇ - يان ئولتۇرۇشۇپ، بىمالال چاقچاقلار قىلىشىپ يۈرگەچ، بۇئادەمنىڭ سايىسىدا مەنمۇ ئازدۇر - كۆپتۇر "ھۆرمەتلىنىش" كە ئىگە بولۇپ قالدىم دېسە. كۈندۈزلىرى ماڭا، مەن كۆرۈشۈپمۇ باقمىغان، تونۇمايدىغان ئادەملەرسالام بېرىدىغانمۇ بوپ قېلىۋاتىدۇ.مۇشۇنىڭ بىلەن داڭلىق ئادەم بوپ كېتەمدىم نېمە، دەپ ئويلاپمۇ قويىمەن بەزىدە.
     مەن مېڭىۋاتقان يول توغرىمۇياكى قورقۇنچلۇق ئاقىۋەتلەرنى ئېلىپ كېلىدىغان تۇيۇق يولمۇ؟ ئەتە - ئۆگۈن ئانداق بولدى، مانداق بولدى دەپ ئىككى پۇتۇمنى بىر ئۆتۈككە تىقىپلا قويارمۇ؟ بۇنىڭغائۆزۈممۇ جاۋاب بېرەلمەيمەن.خاتامىكىن دەي دېسەم، ئۇسورۇنلاردا ھېچقانداق، قانۇنغا خىلاپ ۋە ياكى يامان ئىشلار ھەققىدە گەپ - سۆز،مۇزاكىرىلەر يوق. ئاساسەن تىجارەت، قانداق قىلىپ پۇل تېپىش، دېگەنگە ئائىتلاگەپلەر. قېنى باشقا كەلگەندە كۆرمەيمۇ ....
ئويۇن - تاماشالار ئۆزيولىدا داۋاملىشىۋەردى. يا بىرەر خىزمەتمۇ تاپشۇرۇپ ئالمىدىم. ئەلۋەتتە خىزمەت قىلمىغانىكەنمەن، ئۇ ئادەمنىڭ ماڭا ھەر قېتىم كۆرۈشكەندە بەرگەن پۇللىرىنىمۇ قوبۇل قىلماي يۈرۈۋەردىم. مۇشۇ بىكارغا يەپ، ئىچىپ يۈرگىنىممۇ يېتەر .... دەپ.
  ئەمما ئۇ ئادەم، مېنىڭ كەلگۈسى ئۇنىڭدا ئىشلىشىم ئۈچۈن، چوقۇم بىرەر قاتناش قورالىنىڭ پراۋىسى بولۇشىنى تەلەپ قىلدى. ھەم نۇرغۇن يوللىرىنى قىلىپ، ماڭا ئوتتۇرا تىپلىق موتوسېكىلىت پراۋىسى ئېلىپ بەردى. موتۇسېكىلىتنى ھەيدەشنى ئۆگىتىدىغان مەكتەپلەرگە، ئۇ ئادەمنىڭ گېپى ئۆتكىنى بىلەن، ئەڭ ئاخىرقى دۆلەت بويىچە بىر تۇتاش ئېلىنىدىغان، يازمىچەئىمتىھان يەنىلا ئۆز كۈچۈم بىلەن ھەل بولدى. شۇنداق قىلىپ، مەنمۇ ياپونىيىدەك بىردۆلەتنىڭ قانۇنلۇق، رەسمىي پراۋىسىغا ئېرىشتىم. ئىككى CC كۈن ئۆتمەيلا يىپ -يېڭى دالا موتوسېكىلىتى بەجا كەلتۈرۈلدى. قارا كۆك، ياماخا لانزا، 250.
پاھ، ... ئوتتۇرا مەكتەپكەچىققاندىن بېرى ئارزۇ قىلىپ يۈرگەن ئۈمىدىم ئەمەلگە ئاشتى. بۇ موتۇسېكىلىتنى، تۈزيوللاردا غوڭۇلدىتىپ ھەيدىگىنىم بىلەن ئانداق پاتقاق يوللاردا چاپتۇرۇشتا، ئاجىزبولغان مەن. ھەر كۈنى دېگۈدەك مەھەللىمىزگە يېقىن بىر دەريا بويىغا بېرىپ، مەشىق قىلدىم. ئېگىز يەردىن موتوسېكىلىت بىلەن سەكرەيمەن دەپ ئۆزۈم ئوڭغا، موتۇسېكىلىتنى سولغا سەكرىتىپ، نەچچە - نەچچە كۈنلەپ قوپالماي يېتىپمۇ كەتتىم. تىك دۆڭلەرگەچىقىمەن دەپ، ئەپلەشتۈرەلمەي موتوسېكىلىتنىڭ ئاستىدا قالغان چاغلىرىم كۈرمىڭ. بۇھەقتىكى كىتاب، ۋىدىيولارنىمۇ كۆردۈم. ئۇنىڭغا قاراپ ئولتۇرسام،ۋىدىيولاردىكىلەردىن ياخشى قىلىمەن خىيالىمدا، لېكىن نەق مەيداندا يەنىلاپالاكەتلىك قىلىپ، ئۆزۈمدىن ئۆزۈم كۆڭۈلسىزمۇ بوپ كېتىمەن بەزىدە. بىر نەچچەقېتىملاپ دېڭىز ساھىلىغا بېرىپ قۇم ئۈستىدە موتوسېكىلىتنىڭ ئالدى چاقىنى كۆتۈرۈپ،ئارقا چاقى بىلەنلا مېڭىشنى ئاران ئۆگەندىم. بۇ جەرياندىكى كۈلكىلىك ئىشلار تېخىمۇكۆپ.
    توكيونىڭ بۇلۇڭ -پۇچقاقلىرىغا كىرىپ چىقىپ، ئۇ يەرگىمۇ خېلىلا كۆندۈم. توكيونىڭ گادىرماچ قالايمىقان يوللىرى، ئۇنى ئەستە تۇتۇش مەن ئۈچۈن خېلىلا قىيىن بىر ئىش بولدى.بارغان يەرلەرنىڭ ئىسمىنى تازا ئوبدان ئېسىمدە تۇتالمىغاچ، مەن بارغان يەرلەرنى يوغان قىلىپ ئىككىگىلا بۆلىۋالدىم. بىرى بىزنىڭ مەھەللە تەرەپ، يەنە بىرى توكيو تەرەپ دەپ.
    دېڭىز ساھىلىدا مېڭىش، ياپون خەرىتىسىنى ئۈجۈرىدىن - بۈجۈرىگىچە ئىگىلەش، يۇقىرى سۈرئەتلىك يوللار بىلەن ئادەتتىكى يوللارنىڭ كېسىشىش تۈگۈنلىرىنى ئىگىلەش خەرىتىلىرىگە پىششىق بولۇش دېگەنگە ئوخشاش ئۆگۈنۈشكە تېگىشلىك ئىشلار رېتى بويىچە بىرىنچى، ئىككىنچى دەپ تاپشۇرۇق بولۇپ قولۇمغا يازمىچە بولۇپ يېتىپ تۇردى. بۇ ئەلۋەتتە ھېلىقى ئادەمنىڭ ئورۇنلاشتۇرۇشى ۋە ماڭا قويغان قاتتىق تەلىپى ئىدى.
    مانا بۇ كوپ ۋاقىتلاردا مېنى بېسىۋالىدىغان موتوسېكىلىت ئاخىرى مەن ئۇچۇن ئېيتقاندا، بىر ئويۇنچۇققا ئايلاندى.ئۇنى مەن بىر ئوبدانلا كونتۇرۇل قىلالايدىغان سەۋىيىگە يەتتىم. پاتقاق، قۇم،تاشلىق يوللاردا قانداق ھەيدەش، قانداق ئايلاندۇرۇش، ئاسفالت يوللاردا قانداق قىلىپ تېز ئېگىلىش، تېز قوزغاش، تېز توختاش .... دېگەنگە ئوخشاش ھۈنەرلەر ئاساسىي جەھەتتىن پېچەت دېگەن يەرگە يېتىپ باردى.
    تۇنجى خىزمەتنى تاپشۇرۇپ ئالدىم. بىر دانە سومكا، يەنى دۈمبىگە ئاسىدىغان ئوقۇغۇچىلار سومكىسى. بۇنى توكيودىن ئادەتتىكى كىچىك يول بىلەن دېڭىز ساھىلىغا چىقىپ، نىشانغا يەتكۈچە، دېڭىزساھىلىدىن چىقماسلىق ۋە سەھەر يولغا چىقىپ، كۈن پاتقۇچە تاپشۇرىۋالغۇچىغا نەقۋاقتىدا يەتكۈزۈش ۋە ئۇنىڭدىن بىر پارچە خەت ئېلىپ، قايتىپ كېلىش شەرتى ئاستىداقوبۇل ئەتتىم. نىشانغا بۇرۇن بېرىشقىمۇ، كېچىكىپ كېيىن بېرىشقىمۇ بولمايدۇ دېيىلدى.
ئوھ ... ئالە شەرىڭنى،كېچىكمە دېگىنىڭنىغۇ چۈشەندىم، بالدۇر بارماسلىق دېگىنىڭنى تازا قوبۇل قىلالمىدىمغۇ ئاداش ... دېگۈم كەلدى.
     سەھەر، موتۇسېكىلىتنى ھەيدەپ ماي ئىسكىلاتىغا بېرىپ، ماينى لىپمۇ - لىپپاڭ قۇيدۇم. دېيىشىۋالغىنىمىز بويىچە،ماينى قۇيۇپ بولۇپ، شۇ ماي ئىسكىلاتىدا قارا رەڭلىك بىر ماشىنىنىڭ كېلىشىنى ساقلاپ تۇردۇم. تەييارلىق پېچەت.دېيىشىۋېلىنغان دەل ۋاقىتتابىر قارا رەڭلىك ماشىنا، ماي ئىسكىلاتىغا ئېگىلىپ كىرىپ كەلدى.مەن تېخى، ئۇ ماشىنىدىكىلەربىلەن سالام - سائەت قىلىشىپ، ئۇيغۇرچە كۆرۈشۈپ، بىرەردىن قەھۋە ئىچىۋېتىپ ئاندىن سومكىنى ئېلىپ، يولۇمغا ماڭىمەن دەپ ئويلاتتىم. نەدىكىنى؟ ماشىنا موتوسېكىلىتنىڭ يېنىغا توختىدى. دېرىزە ئەينىكى چۈشۈرۈلۈپ، ئىچىدىن ئادەمنىڭ بېشى چىقماي، بىرقارا رەڭلىك، دۈمبىگە ئاسىدىغان ئېغىر سومكا چىقتى. ئۇ سومكىنى قولۇمغا تۇتقۇزدى -دە، " غىڭڭىدە" كېتىپ قالدى. گەپ - سۆزمۇ يوق.
ئوھ ... ما گۇيدا ئەدەپ -ئەخلاق دېگەن نەرسە يوقكىنا؟ يېرىم سائەتتەك پاراڭلىشامدۇقكىن دېسەم ....دېگىنىمچە سومكىنى تۇتۇپلا، تۇرغان يېرىمدە تۇرۇپ قاپتىمەن.
     يادلىۋالغان خەرىتە بويىچەتوكيودىن تارتىپ شىمال تەرەپكە سوزۇلغان 16- نومۇرلۇق يولغا چىقىپ، دېڭىز تەرەپكە يۈرۈپ كەتتىم.ئۇچامدا ئادەتتىكى بىر ئاقرەڭلىك قىسقا يەڭ مايكا، پادىچىلار ئىشتىنى. ئىشتان سەل تارمۇ قانداق، سەل راھەتسىز. ئەمما يولغا چىقىپ بولدۇم، ئەمدى ئامال يوق.
    كۆپ - كۆك ئاسمان، كۆزيەتكۈسىز دېڭىز. غۇر - غۇر - غۇر قىلىپ يۈزنى سىيپاپ ئۆتۈپ تۇرۇۋاتقان دېڭىزشامىلى. راھەت راھەت ... ئالدىمدا سوزۇلۇپ ياتقان دېڭىز ساھىلى، دېڭىز دولقۇنىنىڭ ئاخىرلىشىشى بىلەن، تەبىئىي ھالدا تۈزلىنىپ قالغان، نەم قۇمدا، بىرەر كاتاڭمۇ يوق،سىپ - سىلىق. موتۇسېكىلىتنى بولىشىچە ھەيدەپتىمەن. بەزىدە شەيتانلىقىم تۇتۇپ كېتىپ، دېڭىز سۈيىگە موتۇسېكىلىتنى بىر ئەكىرىپ، بىر چىقىرىپ، ئالدى چاقىنى بولۇشىچە كۆتۈرۈپ، ئارقا چاقىنىڭ تېز پىقىرىشىدىن، قۇمغا ئېلان باغرى ئىزلارچىقىرىپ مېڭىپمۇ كۆردۈم. نەچچە ئايلارچە پالاكەتلىك قىلىپ، ئاران ئىگىلىگەن ھۈنەرلىرىمنىڭ ھەممىنى ۋايىغا يەتكۈزۈپ قىلىپ باقتىم. ھەر ھالدا ئوڭدىسىغا، خامكاۋىدەك چۈشۈشتەك كۈلكىلىك ئىشلار يۈز بەرمىدى. كۆڭلۈم ۋاچچىدە كۆتۈرۈلۈپ كەتتى.
    دېڭىز ساھىلى. قۇملۇقتاتۇغۇلۇپ چوڭ بولغان، ئۆمرىدە دېڭىز كۆرۈپ باقمىغان بىر ئۇيغۇر يىگىتى، ھازىردۇنيادىن بىغەم ساپ ھاۋا، تىنچ شارائىتتا ھەممىنى ئۇنۇتقان ھالدا، ئالغائىلگىرىلىمەكتە. ئۇنىڭ كاللىسىدا بارى پەقەت: "مۇشۇ سومكىنى ئاپىرىپ بېرىمەن،ئۇنىڭدىن بىر پارچە خەت ئالىمەن، قايتىمەن" دىن باشقا ھېچنېمە يوق. بۇئادىغىنە كاللا"توختا بۇ ئېغىرسومكىنىڭ ئىچىدە نېمە باردۇ؟" نى ئويلاپمۇ قويمىدى. ئۈچ-تۆت سائەتچە توختىماي مېڭىپ چارچاپمۇ قاپتىمەن. كۈنمۇ بارغانچە ئىسسىپ، يانپاشلىدىم تار ئىشتان ئىچىدەقىينىلىپ، تەرلەپ بىرنېمە بوپلا كەتتى.
   ۋوي توختا، بۇ دېڭىزساھىلىدا بىرەر ئىنسان بولمىسا ماڭا قاراپ تۇرۇۋاتقان، توختاپ بىر سۇغا چۈشمەيمۇ.... دەپ ئويلاپ مۇۋاپىقراق ئورۇننى پايلاپ خېلىغىچە ماڭدىم.ئوڭ تەرەپ چەكسىز كەتكەن دېڭىز، سول تەرەپ ئازراق دۆڭ، دېڭىزدىن 100 مېتىرلارچەيىراقلىقتا ئازغىنە يېشىللىق كۆرۈنۈشكە باشلىدى. يېقىنلاشقانسېرى، ئۇنىڭ پاكار-پاكار يۇلغۇنلاردىن (يۇلغۇن دەپ ئېلىپلاۋەتتىم، كىچىكىمدە كۆرگەن كوس - كوسدەرىخىگە ئوخشىشىپ كېتىدىغان بىر خىل دەرەخ ئىش قىلىپ)، تەركىپ تاپقان ئورمانلىقى ئېنىق كۆرۈندى.ۋاھ ... مەشە تازا بولىدىكەنمانا، ئادەممۇ يوق، ئۆزۈم يالغۇز. بىر پەيزى قىلمايمەنمۇ ...ئىلى دەرياسىدائۈزگەندەك بىر ئۈزۈۋەتمەيمۇ ...

   ئوھۇي ... قىلىپ موتۇسېكىلىتنى ئۇچارتىۋەتتىم. خۇشاللىقىمدا ئورنۇمدىن تىك تۇردۇم. بايا بېسىپ ئولتۇرغان كىيىملىرىممۇ ئۇچۇپ كەتتى. ۋاھ ... موتوسېكلىتنى يالىڭاچ ھالداھەيدىسىمۇ ئالامەت بولىدىكەن جۇمۇ .... بىرەرسى كۆرۈپ قالمىغاندۇ - ھە؟ بۇنداق ئادەم - ئىنسان يوق يەرگىمۇ بىرنېمە دېگىلى بولمايدىكەن ئەمەسمۇ ....ئۇف ... چاقماق تېزلىكىدىكى ئىنكاس، شاپپۇر - شۇپپۇر ھەرىكەتلەر بىلەن ھېلىمۇ، مەيلى ئۇلارنىڭ قانداق ئادەم بولۇشىدىن قەتئىينەزەر، ئۇلاردىن قۇتۇلۇپ قاچالىدىم. ئەمدى نىشانىمغا يەتكۈچەسۇغىمۇ چۈشمەيمەن بولدى.
    ئېغىزلىرىم قۇرۇپ، چۆلدەقالغان ئادەمدەك ئۇسساشقا باشلىدىم. يانداپلا دېڭىز ساھىلىدىن چىقىشنى شۇنچەئويلىساممۇ، زادىلا غەيرەت قىلالمىدىم. تازىمۇ ئوچۇقچىلىق بىر يەرگە كېلىپ،موتوسېكىلىتنىڭ ئوتىنىمۇ ئۆچۈرمەي تۇرۇپلا، ئالدىمدا يەنە قانچىلىك مۇساپىنىڭ بارلىقىنى خەرىتىگە قاراپ ھېسابلىدىم. نورمال تېزلىكتە ماڭسام، يەنە بىرسائەتلەردە نەق بارىدىكەنمەن. مەندە ھازىر ئارتۇق ۋاقىت خېلىلا بار. كېچىكمەي بېرىشىم مۇقىملاشتى، لېكىن بالدۇر بېرىپ قېلىشىم ئېھتىمالغا ناھايىتى يېقىن. چۈنكى مەن، ئۇ يەردىكى مېنى ساقلاپ تۇرغان ئادەمنىڭ ھېچنىمىسىدىن خەۋەرسىز. مېنىڭ بىلىدىغىنىم پەقەت، 250 CCمېنىڭكىگە ئوپمۇ - ئوخشاش موتوسېكىلىتتىنلا ئىبارەت. قارا كۆك، ياماخا، لانزا. دالا موتوسېكىلىتى.
    مەنزىلگە يېقىنلاشتىم. دېڭىزساھىلىدا كىشىنى ھەيران قالدۇرغۇدەك چوقچىيىپ چىققان، تولىمۇ ئۇشلۇق يوغان بىر تاش.كۆلىمى "ۋويجون" دېگۈدەك بولمىسىمۇ، ئۇنىڭ ئېگىزلىكى سىزنى ھەيران قالدۇرىدۇ. ئەتراپىدا ھېچقانداق نەرسە يوق. پەقەت ئوتتۇرا قولدەك چوقچىيىپ چىققان ئېگىز بىر تاللا تاش. تاشقا يېقىنلىشىپ كەلدىم. راست دېگەندەك بىر مېنىڭكىگەئوخشاش موتوسېكىلىت تاشنىڭ يېنىدا تۇرۇپتۇ. موتوسېكىلىتنىڭ يېنىدا قاپ - قارا بىرگەۋدە ئوڭدىسىغا، قالپىقىنى بېشىغا قويۇپ، بىر پۇتىنى ئۇزۇن سوزۇپ، يەنە بىرپۇتىنى ئوماققىنە يىغىپ يېتىپتۇ. قارا خۇرۇم ئىشتان، موتوسېكىلىتچىلار ئۆتۈكى پۇتىدا ... بۇ ئىسسىقتا، بۇ ئادەم ئىسسىمىغانمىدۇ ھوي ... دېگەنلەرنى ئويلىغاچ ئۇنىڭغا يېقىنلاشتىم.
     سىنچىلاپ قارىدىم. قاراكۆزەينەكنىڭ تازىمۇ يوغىنىنى تاقىۋاپتۇ. خۇرۇمدىن تىكىلگەن بەدىنىگە ھىممىدەكەلگەن تالستوپكا، ئىچىدە ئاق مايكا، تالىستوپكىنىڭ ئالدى سېرتمىسى ئوچۇق.قارا دېگەندەك قىلدى.قارىسام، بىر گۈزەل قىز. سەل كەرشنىۋاي ئارىلاشقان رەڭدىكى تاشلىنىپ تۇرغان پارقىراق چېچىنى بىرلا قىلىپ بوغۇۋاپتۇ. خۇرۇم تالىستوپكىسى بولمىغان بولسا،كۆتۈرۈلگەن كۆكسىلىرى، ھىم مايكىسىنى ئېتىپلائېتەركەن كاساپەتنىڭ … ئەمما قىزنىڭ كىيىنىشى، ئۆزىگە شۇنچە يارىشىپتىكى پاھ - پاھ ...
    ئوھ، مەن بىرئازجىددىيلەشتىم. مەن نەگە، كىمگە، نېمە ئېلىپ كېلىۋاتىمەن؟ مەن كىم؟مەن سۇرئىتىمنى ئاستىلىتىپ،ئۇ قىزدىن بىر قېتىم ئۆتۈپمۇ كەتتىم. ئۇ توختىتىپ قويغان موتوسېكىلىتنىڭ رەڭگى،ماركىسى، قانچە چ چ ئىكەنلىكى ئېنىق بولغان بولسىمۇ، موتوسېكىلىتنىڭ نومۇرىنىمۇئېنىق كۆرۈۋالاي جۇمۇ ... ھازىرغىچە دېيىلگەن بەلگىلەر، ئاساسىي جەھەتتىن ئوخشاش چىقىۋاتىدۇ. يەنىلا دىققەت قىلاي، "پوككىدە" خاتا ئىشتىن بىرنى قىلىپسالماي.موتۇسېكىلىتنى ئېگىلدۈرۈپ،قىزنىڭ يېنىغا قايتا كەلدىم. ئوتىنى ئۆچۈرمەي تۇرۇپ، موتۇسېكىلىتنى توختىتىپ،پۇتۇمنىڭ ئۇچىدا قۇمنى دەسسەپ تۇرۇپ، "ھى" دەپ باقتىم.
قىز ئورنىدىن تۇرۇپ مەنتەرەپكە، سول قولىدا موتوسېكىلىت قالپىقى، ئوڭ قولىدا بىر ئاق كونۋېرتنىپۇللاڭلاتقاچ كېلىۋاتىدۇ.
    قىز مەزمۇت قەدەملەر بىلەن تولىمۇ ئىشەنچلىك ھالدا كۆكرىكىنى كېرىپ "گۈس - گۈس" دەسسەپ كېلىۋاتىدۇ."يايرىمنىڭ مېڭىشلارى، ماڭسا قول سېلىشلارى" نىڭ نەق ئۆزى ... مۇنداقلاقارىشىمدا، بۇ گۈزەل قىزدىن قۇسۇر تاپالمىدىم. ئويماقتەك لەۋلىرى، سۈزۈك مەڭزىلىرى... چىرايىدا يا كۈلۈش يوق، يا رەنجىش يوق. تولىمۇ ئەستايىدىل، كەسكىن.
     مەن بۇ ۋاقىتتا، دۈمبەمدىكى سومكىنى ئۇنتۇپ، قىزغا باھا بېرىش بىلەن بوپ كەتكەچ، قىزنىڭ يېنىمغا كەپ بولغىنىنىمۇ سەزمەي قاپتىمەن.ئوھ .... ئاغزىم ئېچىلىپلاقالدى. بۇ قانداق ئىش بوپ كەتتى ئەمدى؟ بۇ ماڭا تونۇشلا چىرايغۇ؟ ئوڭۇممۇ  چۈشۈممۇ؟
     ياق ئوڭۇم. مەن ئۇخلىمىدىم.مەن نەچچە يۈز كىلومېتىر دېڭىز ساھىلىنى بېسىپ كەلدىم. كۆزۈمگە ئىشەنمەيلا قالدىم.بېشىمنى بىر نەچچە رەت سىلكىپ، كۆزلىرىمنى چىمچىقلىتىپمۇ باقتىم. ئەمما بۇ قىزيەنىلا تونۇش چىراي. ئەينى ۋاقىتتىكى گۈزەللىكىدىنمۇ بەدەنلىرى تولۇپ، نەچچە ھەسسەگۈزەللىشىپ كېتىپتۇ.
