يوللىغان ۋاقتى 2015-4-10 13:36:25 |ھەممە قەۋەتنى كۆرۈش
      دوسلار مەن يەنە كەلدىم،سىلەرنىڭ ئاراڭلاردا ئارىلاپ-ئارىلاپ پەيدا بولۇپ قالىمەن،تۇرمۇش ھەلەكچىلىكى ئۈچۈن ئالدىراشمەن،مانا مۇشۇنداق پەيدا بولغان چاغلاردا ئەھۋاللىشىپ تۇرارمىز،سىلەرگە مانا مۇنداق ئاددى ئەسەرنى ئىلىپ كەلدىم،كۆرۇپ سىلەرنىڭ تۇرمۇشۇڭلارغا لازىملىقى بولسا پايدىلناسىلەر.  
   بۇ ئۆيدە يەنە غەۋغا باشلاندى،ئاپام توختىماي ماڭا كايىپ سۆزلەۋاتاتتى..........
  -بولدى،سىنىڭ دەردىڭ بولسا مىنىڭمۇ ئۆزەمگە تۇشلۇق  دەردىم بار،سالامەتلىگىم تىخى ياخشى ئەمەس،سەنمۇ قىرىق ياشقا ئاز قالدىڭ،ئۆزەڭنىڭ ئىشىنى ئۆزوڭ ھەل قىل،ماڭا بۇندىن كىيىن دەرت ئىيىتقىچە بولما.-ئاپام تىلفۇندا تىلفۇن راسىتلا يىرىلىپ كەتكىدەك دەرىجىدە ماڭا ۋاقىردى،تىلفۇن تۇتۇپ تۇرغان قولۇم شۇنچىلىك تىترەۋاتاتتى،ئەگەر بۇ تالا بولمىغان بولسا،مەن توكسىكىلىت ھەيدىمىگەن بولسام ئۈنۈمنى سىلىپ ۋاقىراپ يىغلىغان بولاتتىم،ئچىمدىن "ۋايجان""ۋايجان"دىگەن نىدالا چىقتى،كۆز ياشلىرىم يۇزۇمنى نىداسىز بويدى،بۇنى توختىتىش،بۇ تالاغۇ دەپ نۇمۇس قىلىش،ياق..ياق..ھىچقانداق ئاڭ مەندە يوق ئىدى،مەن پەقەت
-   ماقۇل،مەن دەردىمنى دىمەي،بىراق بۇ دۇنيادا يىغلىغىنىمدا تەسەللى بەرگىدەك كىممۇ قالدى،قىرىق ياشقا كىتىپ بارغان ئادەم دەرت تارىتماسلىقى كىرەككەندە،ماقۇل سىنى ئىزدىمەي،دەردىمنى دىمەي،ماقۇل،ماقۇل،بولدىمۇ ماقۇل......-تىلفۇننى قويىۋەتتىم.
    ئۆيدە كىمنىڭ بارلىقىنىنىمۇ بىلمىدىم،ئىشىكدىن "ۋايجان"دەپ ۋاقىراپ كىرگىمىنچە ئۇن سىلىپ يىغلىدىم،سىز تىلۋوزۇرلاردا كۆرگەنمۇ،ئىگىز تاغنىڭ چوقىسىغا چىقىپ ۋاقىراپ ئازاپلىرىنى يەڭگىللەتكەننى ۋە يەنە دىڭىز بويلىرىغا بىرىپ ۋاقىراپ ئىچىنى بوشاتقانلارنى،مەنمۇ شۇنداق قىلدىم،ئۇن سىلىپ يىغلىدىم،ۋاقىردىم،خۇدا........خۇدا....دەپ توۋلىدىم،ئىككى قولۇم بىلەن مەيدەمنى ئۇردۇم،جىق يىغلىيالمىدىم،يىرىم سائەتلەردىن كىيىن ئۆزلۈگۈمدىن توختاپ قالدىم،سەۋەپ مەندە جان قالمىدىم،قولۇم،پۇتۇم،پۇتۇن بەدىنىم ماغدۇرسىزلىقدىن لاغ...لاغ... تىتىردى،لاسسىدە ئۆزۇمنى تاشلىۋەتتىم.
بىرسى بىلىكىمدىن تۇتۇپ يۆلىدى،قارىدىم يىنىمدىكى ئۆيدىكى خەنزۇ خوشنام ئىدى،ئۇ ماڭا بىر پىيالە مۇزدەك سوغۇق سۇنى ئىچكۇزدى،ئاندىن مەندىن تەپسىلى ئەھۋالنى سورىدى.مەن ئىسەدەپ تۇرۇپ:
    -ئائىلىمىزدە تۆت بالا بىز،كىچىكىمدىن بىر خىل جايىل كەپسىز،دىگىنىم دىگەن بالا ئىدىم،مۇنداقچە قىلىپ ئىيىتقاندا باشباشتاقراق بالا ئىدىم، ئىشلىرىمنى بىرسىگە مەسلىھەت سالمايتىم ئۆزۇم ھەل قىلاتتىم،قانچە قىينالساممۇ،باشقىلارغا يىلىنمايتىم،ھورۇن ئىدىم ئىش قىلمايتىم،كەينىمدىكى سىڭلىم پۇتۇن ئۆي ئىشىنى ئۆزى ئۇستىگە ئالغان بولۇپ شۇلا قىلاتتى،ئۆيدىكىلەرمۇ مىنى بىشەم،ھورۇن ئاتىكاچا بۇنىڭغا گەپ قىلغاننىڭ پايدىسى يوق دەپ مەندىن نىرى تۇراتتى،ئاپاممۇ ئۇكىلىرىمغا بۇ ئەسكىدىن نىرى تۇرۇڭلار بالىلىرىم بۇ سىلەرنى ئۇرۇپ ئۆيدە جىدەل چىقىرىدۇ،دەپ ئۇكىلىرىمنى نەمدىن نىرى قىلاتتى،بارا-بارا مەن تىخىمۇ يالغۇز قالدىم،ئارلىقلىرىمىز يىراقلاشتى،بىر ئىنىم بار ئاپامنىڭ جىنى،كەينىمدىكى سىڭلىم ئاپامنىڭ قولى،ئەڭ كىچىك سىڭلىم بالىنىڭ كىچىكى،دادىسىدىن كىچىك قالغان،شۇنىڭ بىلەن بىلىلارغا بىردىن ئۇنۋان دوپپىسى بىرىلدى،مەن بولسام يالغۇز قىلىۋەردىم،نۇرغۇن دەرتلەرنى تارىتتىم،تۇرمۇش خاپىلىقلىرىغا قالدىم،ئاجىراشدىم،ئاجىرشىپ بولۇپ ئۆيدىكىلەرگە خەۋەر قىلدىم،سىرىتتىن ئۆي تۇتتۇم،قايتا توي قىلدىم،توي قىلىشقا ئاز قالغاندا خەۋەر قىلدىم،تۇغدۇم،ئىككى بالىلىق بولدۇم،بەكلا يالغۇز،بەكلا يالغۇز يەنىلا بەكلا يالغۇز قالدىم .
    ئاپام بىلەن مىنىڭ ئوتتۇرۇمدا يىقىنلىشقىلى بولمايدىغان ھاڭلار قاتلاممۇ-قاتلام ئاۋۇدى،كۇنلەر ئۆتتى،ئاپامدىن كۇندە ئاغرىنىدىغان بولدۇم،" بالىلارغا ئوخشاش مۇئامىلە قىلمىدى،ھەممىزنى تەڭ كۆرمىدى،مەندىن بىرەر قىتىممۇ نىمە قىلىۋاتىسەن،مىنە بولدۇڭ،تۇرمۇشۇڭ قانداقراق، ئاغرىدىڭمۇ، ساقمۇسەن،....."دەپ ئەھۋالمۇ سوراپ قويمايدۇ.گەپ تاپسا كەينىمدىكى سىڭلىمغا دەيدۇ،كۇندە كىچىك سىڭلىمغا تىلفۇن ئۇرۇپ يىگەن-ئىچكەندىن تارتىپ كىچىك ئىشلاردىن چوڭ ئىشلارغىچە ئەھۋال سوراپ ھارمايدۇ،جىنىم ئوغلۇم تالادا تاماق يەپ قالما ساڭا ئازراق ئايرىم تاماق قىلىپ قويدۇم،دىككىدە كىلىپ يەپ كەتكىن،سەن قانداقراق،نىمە ئىشلارنى قىلىۋاتىسەن،كەچ تەرەپتە كەچ قالمىغىن،ئۆيگە بالدۇر قايىتقىن.........بۇ ئانىنىڭ پەرزەنىتلىرىگە كۆڭۇل بۆلۇشلىرى مىنىڭ ئىچىمنى بەكلا سىقتى،بىر كۇنى ئانامنى يوقلاپ چىقتىم،ئاپام مىنىڭ رەڭگىنمى كۆرۇپ"نىمە بولدۇڭ"دەپ سوراپ قالدى،بولدى شۇنىڭ بىلەن مەن تۇكۇلۈپلا كەتتىم بەك يىغلىدىم،بىشىمغا كەلگەن خاپىلىق،دوختۇرنىڭ كۆسىڭىزدە ئۆسمە باركەن ئاپراتسىيە قىلدۇرۇڭ دىگەن دىئاگۇنىزلىرى،بالىلىرىمنىڭ ئەھۋالى،ئىرىمنىڭ ماڭا سىلىۋاتقان زۇلۇمى،ئانامنىڭ مىنى تەڭ كۆزدە كۆرمىگەن خورلۇقلىرىم،ئىنە-سىڭىللارنىڭ مەندىن نىرى تۇرۇشلىرى.....ئىش قىلىپ ئوتتۇز نەچچە يىللىق دىيىشنى ئويلىغان ھەممە گەپنى دەپ،ئاخىرىدا"ئاپا مەنمۇ بىر نۇرمال ئادەم،دەردىم بار،كىسىلىم بار،مەنمۇ يار-يۆلەكچىگە مۇھتاج،ماڭىمۇ كۆڭۇل بولگىن،بەزى چاغلاردا مىنى ئاپام توققاندىمۇ دەپمۇ ئويلاپ قالىمەن،بۇندىنكىيىن ئۇچ پەرزەنتىڭ بىلەن بىللە مىنىمۇ قوشۇپ ئويلاپ قويغىن" دىدىم.ئاپام ئولتۇرۇپ كىتىپ تىللاپلا كەتتى،مەن قاچان سىنى ئويلىماپتىمەن،ئىشىكدىن كىرسەڭ مىھماندەك كۆرپىگە باشلايمەن،تاماقنىڭ ئوبدىنىنى ساڭا بىرىمەن،سىلە ئۆزلىرى بالام دەيمەن،خاپىلىقى جىق بۇ ئۆينىڭ ئىشلىرىنى دەپ بىشىنى ئاغرىتماي دەپ،ھەممىنى سىڭلىڭغا دەپ شۇ بىچارىنى ئاۋارە قىلىمەن،ئوغلۇم بولسا بىرتال ئالدى-ئارقىدا يوق،ئۇنىڭ دەردىگە ھال-مۇڭ بولمىسام،بۇ دۇنيادا يالغۇز قالىدۇ،كىچىكى ئۆزەڭمۇ بىلىسەن داداڭدىن يالغۇز قالدى،ياۋاش،ھىچ ئىشنى قاملاشتۇرالمايدۇ،ئۇنىڭ ھەممە ئىشىنى ئۆزەم قىلمىسام بولمايدۇ،سەن ئۆز بىر تۇققانلىرىڭنىمۇ كۆرەلمەپسەن،سىنىڭ ئەسكىلىكىڭ مانا مۇشۇ،شۇڭا ھەممە ئادەم سەندىن نىرى تۇرىدۇ،مۇشۇنى بىلسەڭ بولدى،ئۆزەڭنىڭ ئىشىنى ئۆزەڭ ھەل قىلالىغاندىكىن ئۆزەڭنى باشقىلارغا سىلىشتۇرغىچە چوڭلىق رولاڭ بىلەن ئىشلىرىڭ تۇگەتمەي مىنى ئەيىپلىگىنى  چىقدىڭمۇ،ماڭ ئۆيۇڭگە قايىت،ماڭا دىگەنلىرىڭنى ئويلۇنۇپ باق دەپ ئويدىن ھەيدەپ چىقاردى،شۇنىڭدىن كىيىن كۆڭلۇم پوق يىۋالغاندەك بولۇپ بىر نەچچە كۇننى ئۆتكەزدىم،كىم بىلسۇن بۇگۇن تاماق يىگىلى ھەممەيلەن يىغىلىپ ئولتۇرۇپ،ئاپام بىلەن ئىككىمىز زادىلا ئوچۇق چىراي ئىچىشالمىدۇق،تاماقدىن كىيىن ئاپامنى كۇيئوغلىمىز ماشىنىدا ئۆيگە ئىلىپ چىقىپ كەتكەن ئىدى،مەن ئۆيگە بارغىچە تىلفۇندا............دەپ بايا يۇز بەرگەن ئىشلارنى سۆزلەپ بەردىم.
    خوشنام ئولتۇرۇپ كەتكەندىن كىيىن (خوشنام تۇڭگان ،ئىرى خەنزۇ شۇڭا ئۆزىنى خەنزۇ دەيتى )ئاپڭ توغرا دەپتۇ،ئۇ سىنى بىر مۇستەققىل ھەممە ئىشنى قىلالايدۇ،بۇنىڭدىن ئەنسىرمىسەممۇ بولىدۇ دەپ قارىغاچقا ساڭا ئالاھىدە كۆڭۇل بۆلمەيدۇ،باشقىلارغا بولسا بايا ئۆزى دىگەندەك مەلۇم سەۋەپلەر تۇپەيلى ئالاھىدە ئىتىبار بىرىدۇ،ئۆزىنى ھەممە بالامنى تەڭ كۆرىۋاتىمەن دەپ قاراپ مىنىڭ كۆڭلۇم بويۇنچە ھەممە ئادەم مىنى چۇشۇنىدۇ دەپ ئويلايدۇ،ئۇ ئانا ئەمەسمۇ،يىشى 60 قا باردى،ئادەم چوڭ بولغانچە،مىجەزدىمۇ ئۆزگىرىدۇ،سەن بۇنى توغرا چۇشەن،دوختۇر سىنى ئاپرايسىيە قىلدۇر دىگەن بولسا تىز ئاپراتسىيە قىلدۇرغىن،بۇندىن كىيىن ئىچىڭ سىقىلسا مىنى ئىزدە،ئۆزۇڭنى ئىككى بالاڭ ئۈچۈن ئاسىرمىساڭ بولمايدۇ،سىنىڭ بايام يىغلىغان يىغاڭنى ئاڭلىغان خوشنا-خۇلۇملارنىڭ ئىچى قان بولۇپ كەتتى،سەن ئۇنچە قەھھىر قىلىپ يىغلىما،سالامەتلىگىڭدىن چاتاق چىقىدۇ،ئوغلۇڭغا قارا بەك بىچارە بولۇپ كەتتى،بالىلارنى ئايا،بايا ئىشىكنى چەكسەم،بىر قۇتقازغۇچى خۇدا كىرگەندەك ماڭا ھىچ نىمە دىمەيلا سەن بار ئۆينى كۆرسەتتى،بۇندىن كىيىنكى تۇرمۇشۇڭدا ئەمدى شۇلارنى ئويلاپ ياشا،قانداق قىلالامسەن.-دىدى. مەن "ھە" دىدىم.كەچ بولدى،بىر كىچە ئۇخلىيالمىدىم،چۇنكى خوشامنى ئۇزتۇپ چىقىپ ئوغلۇمنىڭ رەڭگىنىڭ بەكلا ساغىرىپ كەتكەنلىگىنى كۆرۇپ كۆڭلۇم يىرىم بولدى،ئۆزەمنىڭ خاتا قىلغانلىقىمنى بىلدىم.مەن نىمىنى خاتا قىلدىم،ئۆيگە كىلىپ بۇنچە يىغلىماسلىقىم كىرەكمىدى، ۋەياكى ئاپام بىلەن ھەق-نەھەقنى ئايرىماسلىقىم كىرەكمىدى بىلمىدىم.مەن توغرا چۇشۇنەي....ھەممىنى توغرا چۇشۇنەي،مىنىڭ خاپا بولۇش،يىغلاش،ھاللىنىش،مۇڭلۇنۇش،دەرت ئىيتىش.....ھوقوقۇم بولمىغاندىكىن،ھەممىنى توغرا چۇشۇنەي.

كۆڭلىڭزنى مەن چۈشەندىم. كەچۈرمىشلىرىڭىز مىنىڭكىگە بەك ئوخشىشىپ قالىدىكەن. ئوخشىمايدىغان يېرى مىنى مومام باققان. 7ياشتىن كېيىن ئۆز ئاتا-ئانام ئۆيىگە ئىلىپ كەتكەن. دادامغۇ بەك كۆيۈمچان ئادەم ئىدى. ئۆمرىنىڭ ئاخىرىغىچە ئۆزمىھرى- مۇھەببىتىنى ئايىماي بىرىپ تۇردى. لېكىن ئويلاپ باقسام ئاپام سەل شەخسىيەتچى ئىكەن. نۇمۇس قىماي نەچچە مىليونغا يارايدىغان مىراسلارنى بۆلىدىغان چاغدا ھەر چوڭ ھۆرمەتكە سازاۋەر ھاجىم ئۇرۇق- تۇققانلىرىمىز كېلىپ مىراس ئىشىدا بىزنى كېلىشتۈرسە نۇمۇس قىلماي« مۇشۇ ئىككىسىگە پەقەت مىھرىم يوق» دەپ ئولتۇرىدۇ. شۇ كۈندىن تارتىپ ئاسمان ئۆرۈلۈپ چۈشكەندەك، ھەممە ئىشەنچىم يەرگە ئۇرۇلغاندەك، ھاياتتىكى مىھرى- مۇھەببەتنىڭ يىلتىزى يوقتەك ئويلايدىغان بولۇپ قالدىم.
مۇشۇنداق شەخسىيەتچى ئانىلارنى سوتلايدىغان ۋىجدان سوتى بولۇش كېرەك. مەن ئانا بولمىش ئاشۇ ئىنساننى يالغۇز قويۇش ئارقىلىق جازالاۋاتىمەن. بەلكىم ۇناھلىرى بۇنىڭدىن جقتۇر. 4 بالا  پاتقان قورسىقىدا 4 بالىنىڭ مىھرى پاتماي قالامدۇ؟  ئىشقىلىپ دىسەم گەپ جىق.
ھازىر ئۆزەمگە شۇنداق بىر مىزان بەلگۈلىۋالدىم. ماڭا قىلچىلىق مىھىر ئاتا قىلغان ئادەمگە ئلقانچىلىك مىھرى قايتۇرىمەن. ماڭا كۈلۈمسىرەپ گەپ قىلغىنى يول تازىلايدىغان تازىلىقچى بولسۇمۇ ئۇنىڭغا ئوچۇق چىراي بىلەن پاراڭ قىلىمەن. ماڭا ئىشىنىپ دەردىنى تۆككەن ئادەمگە دەردىمنى تۆكۈمەن.
ئىشقىلىپ مىھرى- مۇھەببەت مەڭگۈ ئۈزۈلۈپ قالمىسۇن.
شەخسىيەتچى ئانىلارغا دەيدىغىنىم پرزەنتىڭلاردىن ئاياپ كەتكەن مىھرى -مۇھەببىتىڭلار سىڭگگەن ئاشۇ جىسمىڭلار ھامان بىركۈنى توپىنىڭ ئاستىدا چىرىپ قاغا قوجغۇنلارغا يەپ بولۇپ تۈگەيدۇ.

سېۋرىيار ئەپەندىم مەنمۇ دەل ئاپامنىڭ ئاشۇ «باياناتى» ئوچۇق ئاشكايە ئىيتىلمىغان چاغدا مۇشۇنداق قاراشتا ئىدىم. لېكىن كېيىن ئويلاپ باقسام ئاپامنىڭ ماڭا يىللاردىن بۇيان قىلغىنى ئېيتقىنى بىلەن ئوپ- ئوخشاش ئىكەن. مەن ئۇنى كەچۈرۈپ ئۇنىڭغا ئۆزەمگە ئۇرۇپ يۈرگىنىمنى بىر ئەخمەقلىق مۇشۇنىڭغا لايىق دەپ قاراپ كەپتىكەن. توغرا ئانا رەھىمسىزلىك قىلغان بالا چىدام، غەيرەتلىك، كۆيۈمچان بولىدىكەن. لېكىن قەلبىچۇ ئانا مىھرىگە قانمىغان ، بەئەينى سۇسىز چۆل. ئۆزبەكنىڭ بىر ناخشىسى بار « ئانام مىنى ھىچكىم سەندەك سۆيۈپ باقمىدى» دىگەن. تۇرمۇشتىكى خوشاللىرىمىزنى ، قايغۇ -ھەسرەتلىرىمىزنى، ئېرىشكەن يوقاتقانلىرىمىزنى ئىيتىپ بەرسەك ئاڭلاپ بىرىدىغان ئانىمىز بولغان بولسا ، كۆز ياشلىرىنى ئاشۇ كورىيەنىڭ قىسىق كۆز ئارتىسلىرى ئىشلىگەن كىنوغا تۆككىچە بىز ئۈچۈن بولسىمۇ بىر ئىككى قىتىم قويغان بولسا ، قەلبىمىز يامغۇردىن كېيىنكى قۇياش نۇرىدەك پارلىغان بولاتتى. ئەپسۇسلىنىدىغان يەرىمىزنىڭ مۇشۇ بولۇپ قىلىشى بەكلا ئىچىنىشلىق.  
مەنمۇ بىر ئانا ، مەن كۆرگەن كۈنلەرنى بالامغا ھەرگىز كۆرسەتمەيمەن دەپ بالامغا بەكلا كۆيۈپ ئۇنى  كارغا كەلمەس قىلىپ قويۇشتىن بەكلا ئەنسىرەيمەن.
