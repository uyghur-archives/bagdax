ئەسسلامۇ -ئەلەيكۇم تورداشلار ،مۇشۇ ۋاقتىڭلار ھەم مۇشۇنىڭدىن كىينىكى ۋاقىتلىرىڭلار كۈڭۈللۈك بولغاي....
مەن مۇشۇ تور سودىسى داۋامىدا ئۆزەمنىڭ كەم ئەقىللىقلىقىدىن يۈز بەرگەن تىپىك قويمىچىلىق مىساللىرى ۋە بۇنىڭ ئاخىرىدىن بىر نەچچە مىسال كەلتۈرۈپ ،سىلەرنىڭمۇ مۇشۇنداق ئىشلارغا ئۇچرىغاندا سەگەك بۇلىشىڭلارنى بىلدۈرۈپ قويماقچىمەن.مىنىڭ تاۋباۋدىكى ھىسابات نۇمۇرۇم :mosa09972010 (دەرىجەم 4 يۇلتۇز ئىدى )جىفۇباۋنىڭ كەينىدە(账户被盗最高可赔100万,立即投保。)

+6062成长值就能尊享特权了哦！دىگەن ئەسكەرتىشمۇ بار .(ئالغان مېلىمنىڭ قىممتىمۇ بەش يىلدىن بىرى 50000يۈئەنگە بېرىپ قالاي دىگەن.(ئۆزەمگە ۋە باشقىلارغا ئەكەلتۈرپ)
مەن توردا ،جۈملىدىن مىسرانىم ۋە باغداشتا نەچچە قېتىم قويمىچىلىقنىڭ ئۇسۇللىرىنى ئاڭلاپ كىلىۋاتقان بولساممۇ ،لىكىن ئۆزەمنىڭ بىخەستىلىكى ۋە ساددىلىقى تۈپەيلىدىن يەنە 2قېتىم چوڭ قۇيۇلۇپ كەتتىم ھەم ئىسىمدىن چىقمىغۇدەك ساۋاق بولدى ،ئۇ بولسىمۇ يەنىمۇ نەچچە قەدەمنى ئارتۇق بېسىپ بەكرەك ئېھتىياتچان بۇلۇش،مەۋھۇم سودىدا يەتتە ئۆلچەپ بىرى كىسىپ ئەمەس ،يەتمىش ئۆلچەپ بىر كەسسىمۇ بولىدىغانلىقى ،ئالدىراقسانلىق بىلەن ،باشقىلارنىڭ زورلىشى بىلەن ،تىز جاۋاب قايتۇرۇش ئۇسۇلى بىلەن بولغان ئىشلارنىڭ ئاقىۋىتىنىڭ ياخشى بولمايدىغانلىقىنى بىلىپ يەتتىم،ھەم يانچۇق خاتىرەم بار ئىدى ،مۇھىىم ئىشلارنى خاتىرلەيدىغان ،شۇنىڭغا يىزىۋالدىم .
ئەمسە ،بۇ بايقۇش قانداق قۇيۇلۇپ كەتكەندۇ دەپ قىزىقىۋاتقان بۇلۇشۇڭلار مۇمكىن،بوپتۇ ،دەۋىتەي ،باشقىلار ئۈچۈن بىر ئويلىنىش بۇلۇپ قالسۇن ،مىنىڭغۇ مەڭگۈ ئىسىمدىن چىقماس زادى ،تور سودىسىدىن بىر بولسا قولنى يۇيۋېتىشىمغۇ مۇمكىن ئەمەس ،ئەمما ئەمما بۇرۇنقىدەك ئاددىي چاغلايدىغان ،ئىشلار قائىدە بويىچە بۇلىدۇ دەپ قارايدىغان ئىشەنچ ،بۇلۇپمۇ تور سودىسىغا بولغان ئىشەنىچ 80پىرسەنىت تۆۋەنلەپ كەتتى دىسەممۇ بولىدۇ.
1-قېتىم مىنى قۇيۇپ كەتكىنى كورلا چەرچەنلىك ئۈچ ئويغۇر مۇناپىق بىرلىشىپ قۇيۇپ كەتتى...خودانى شىپى كەلتۈرۈپ تۇرۇپ قۇيۇپ كەتتى بىر خاتېرە كومپىوتېر سودىسى ۋاستىسى بىلەن (ئۇچۇر تورىغا ئېلان بىرىپتىكەن ،كىين پاش قىلىپ ئەھۋالنى مەلۇم قىلسام ،ئۇچۇر تورى ۋە ساقچىلارنىڭ رازۋىتكا بۇلۈمىدىنمۇ مۇۋاپىق جاۋاپ ئالالمىدىم)ئىككى قېتىمدا 1800يۈئەننى ،مەن ئۇ مۇناپىقنىڭ ماھىرە ھۈسەن دىگەن نامدىكى بىر بانكا كارتىسىغا ،قايتا -قايتا خۇدانى شىپى كەلتۈرشى ۋە مەن چەرچەن يۈئەنتۇڭنىڭ تىز يوللانما خادىمى دەپ بىر شىرىكىدىن كەلگەن تېلفۇن  بىلەن(ئۇ مۇناپىقنىڭ تىز يوللانما شىركىتىنىڭ خىزمەت ئەھۋالىدىن خېلى خەۋىرى بار ،ئىشلەپ باققان بىرىدەك قىلىدۇ،چۈنكى مەن ئۇ چاغدا تاۋباۋدا تور سودىسى قىلىپ نەرسە ئەكەلتۈرۋاتقىلى 3يىللار بۇلۇپ قالغان چاغلار ئىدى)بىر قىسىم ئەھۋاللارنى سوراپ كۈرۈپ ئىشىنىپ كىتىپ قالغىنىنى ئۇرۇپ بەرگەن ئىدىم.شۇنىڭدىن كىيىن باشقىلارغا ،بۇلۇپمۇ تۇنۇشمىغان بىرىگە مالغا پۇل سېلىپ بىرىدىغان ئىشتىن قاتتىق ئىبرەت ئالغان ئىدىم،ھەتتا دىيارىمىزدا شۇ خۇدانىمۇ شىپى كەلتۈرۈپ ساتىدىغان مۇناپىقلارنىڭ بارلىقىنى شۇ چاغدىلا بىلگەنىدىم.
2-قېتىم تاۋباۋدا مۇشۇ 2كۈننىڭ ئالدىدا تاۋباۋدىن  ئايالىمنىڭ دەيدەيگە سېلىشى بىلەن بىر نەچچە قۇر كىيىم سېتىۋالغانىدىم ،لىكىن ئۇ خەنزۇنىڭ شۇ تاۋباۋ دۇكىنىغا 1000يۈئەن ياجىڭ قويغان يېرىنى كۈرۈپ ،ئىشەنچىلىك ئىكەن ،باشقا 10نەچچە ئادەم ئاپتۇ ،باھامۇ بىرىپتۇ دەپ ،7يۈرۈش كىيىم سېتىۋالغان ئىدىم 3 كۈنگىچە ئەۋەتمىدى ،ئاندىن مەن پۇل قايتۇرۇشقا ئىلتىماس قىلغان ئىدىم ،شۇ كۈنى كەچتە ئۇ مۇناپىق تېلفۇن قىپتۇ ،يەتتە قۇر كىيمىڭنى سېلىۋەتتىم ،سەنمۇ پۇل قايتۇرشۇڭنى ئىناۋەتسىز قىلىۋەتسەڭ ،مال ئېۋەتىش ئۇچۇرى كۈرۈنىدۇ دەپ،نەچچە قېتىم تېلفۇن قىلدى ئۇ مۇناپىق لىكىن چولام تەككەندە كۈرۈپ باقاي دەپ ، قۇيىۋەتتىم.لىكىن كەچتە ئۆيگە كىرىپ كۈرۈپ بېقىپ ،ئۇچۇردا ئۆزگىرىش يوقلىقىنى كۈرۈپ كارىم بولمىدى ،لىكىن شۇ دۇكانغا قايتا كىرسەم ئېچىلمىدى ،بۇنداق ئەھۋالدا قانداق بولار دەپ ئاخىرىنى ئىزدىنىپ باقماپتىمەن.
ئەتىسى جىفۇباۋدىن دەپ بىر مۇلازىمەت تېلفۇنىدىن ئۇچۇر كەلدى ۋە بىر سەل تۇرۇپ تېلفۇن قىلدىن ،پۇل قايتۇرۇشتىكى پۇلۇڭنى ئەسلى دۇكان ئۈچۈرۈلگەن بولغاچقا نۇرمال تەرتىپتە قايتۇرۇشقا بولمىدى ،سەن بىز ئەۋەتكەن بىر ئادىرىسقا كىرىپ كىملىك نۇمۇرىڭىنى ،سەن پۇل چىقىم قىلغان بانكا كارتا نۇمۇرىڭنى كۈرگۈزۈپ بەر دەپ ،مەن ئۇ ساراڭنىڭ بەزى يەرلىك شىۋىلىرىنى بىلىپ بولالماي ،ئايالىم خەنزۇچىنى مەندىن پىششىقراق بىلگەندىكىن سەن پاراڭلىشىپ باققىن دىسەم ،ئەخمەق خوتۇن گۆل بۇلۇپ ،شۇ مۇناپىقنىڭ دىگىننى شۇ پېتى ماڭا ئۆگىتىپ نەتىنجىدە شۇ بويىچە ئۇچۇرلانى كۈرگۈزۈپتىمىز ،بۇ جەرياندا خوتۇنۇم دىگەن دەرىجىدىن تاشقىرى ئەخمەقمۇ پۇل چىقىىپ كىتىدۇ دىگەن ئۇچۇرنى كۆرمەي بېشىدىكى دەلىللەش نۇمۇرىنىلا دەپ بەرسە بولىدۇ ماڭا....شۇنىڭ بىلەن 2560يۈئەننى ئۇ مۇناپىق يۆتكەپ كىتىپتۇ...
شۇنىڭ بىلەن تۇرۇپ كاللامنى ئازراق ئىشلىتىپ ،بۇ قېتىم بىر ئاخىرىنى چىقىرىپ باقاي ، بۇرۇنغۇ ئۇدۇلسىغا بىر كارتىغا سېلىپ بەر دىسە سېلىپ بىرىپتىمەن ،بىراق بۇ قېتىم تاۋباۋ ،جىاۇباۋ ۋە ياجىڭ تۆلەنگەن ھالەتتىكى دۇكاندىن ئالغان تۇرسام ،ئۇلار شۇ رەسمىيەتلەرنى تۆلىمەي تۇرۇپ،1000يۈئەن ياجىڭنى قويماي تۇرۇپ ئاچمىغان دۇكان بولمىسا ،.... دەپ ئاخىرى بىر باشتىن 110 غا ،ئاندىن تەۋەلىكتىكى ساقچىخانىغا ،ئاخىرىدا جىنايى ئىشلار رازۋىتكا بۈلۈمنىڭ ،تور ساقچى ئىشخانىسىغا كىرىپ ،زادى مۇشۇ جەرياننىڭ بىر چۈشەندۈرۈپ قويساڭلار دەپ تۇرىۋېلىپ  شۇنى بىلدىمكى ،قويمىچلارنىڭ ھۈنىرى ئىز قويماسلىق ئىكەن .
ئۇنداق قويمىچى مۇناپىقلارنىڭ ھەر بىرسىدە 5-10تال ھەتتا ئۇنىڭدىن ئارتۇق كومپىيوتېر سېستىمىسىنىڭ بېشنى قايدۇرۇپ راسىتكەن دىگۈزۈۋېتەلىگۈدەك كىملىكلىرى بولىدىكەن،تېلفۇن نۇمۇرلىرى ۋە بانكا كارتىسىمۇ شۇنداق بولىدىكەن،بىر -بىرىنى كۈرۈشمەي تۇرۇپ ۋەزىپە تەقسىم قىلىشىپ ،بىرلىشىپ ،توخۇ پۇقىدەك چاپلىشىپ،ئادەم قايمۇقتۇرىدىكەن ،ئاخىرى كىشىنى راست دىگۈدەك ھالغا كەلتۈرۈپ ئاندىن جۆندىۋىتىدىكەن ،بۇ جەريانىدا ھەتتا 110 مۇلازىمەت نۇمۇرلىرىدىن تېلفۇن قىلىشقىچە ،ساقچى فورمىسى ۋە نۇمۇرلىرىنى ياساشقىچە پىلانلىشىدىكەن ....
مەن پۈتۈن ئۇچۇرلانى ۋە تەجىربەمنى ،يىپ ئۇچى ئىزدەش ئويلىرىنى دىسەم ،ئۇلار ماڭا يۇقارقى سەۋەبلەر تۈپەيلى بۇ ئىشنىڭ ئاخىرىنى چىقىرىشنىڭ قىيىنلىقىنى ،بىر كۈنى ئۆزى  بىر سەۋەبلەر بىلەن قۇيرۇقى تۇتۇلۇپ ئىقرار قىلمىغۇچىلىك بۇ ئىشنىڭ ئاخىرىنى چىقارغىلى بولمايدىغانلىقنى ئېيىتتى....
شۇنىڭ بىلەن ئايالىمنىڭ ۋېيشىندا سودا قىلىش پىلانلىرى ،ۋە مىنىڭ ياردەم قىلىش پىلانللىرىم يوق بولدى ۋە دىدىمكى :خوتۇن مىنىڭدەك بەش يىل تور سودىسى قىلغان ،ھەتتا يۈەنتۇڭنىڭ باش شىركىتىگە ئەرىز قىلىپ ،ساراڭقېتىش تىز يوللانما خادىمنى ئادەم قىلىۋەتكەن ،نۇرغۇن قىلتاقلارنى بايقاپ ،نۇرغۇن قىلتاقلارغا دۈچ كىلىپمۇ ئەگىپ ئۈتەلىگەن ئادەمنى قۇيۇپ كەتكەن يەردە ،بولدى ئەمدى ،ئۇ تاۋباۋ ،جىفۇباۋ ،بانكا سېستىمىسىنى گۆللەپ كىتىشكىچە بارالىغان يەردە ئەمدى ساڭا نىمە بار خۇتۇن ،سودا قىلاي دىسەڭ ،بېرىپ كۆر،مېلىنى كۆر ،پۇلنى سانا ،ئاندىن ئېلىپ كەل ،بىر قولدا پۇل ،بىر قولدا مال سودىسىنى قىل،بولمىسا مىنىڭدەك پۇلۇڭنى ئۇچۇرتۇپ يۈرگىنىڭ يۈرگەن ،ئۇنتۇپ قالما ،مىنى ئاۋۇ مۇناپىققا قانداق قۇيۇشۇپ بىرىپ كەتكىنىڭنى ،سەندە تور سودىسى قىلغۇدەك ئىقتىدار ۋە سەگەكلىكنىڭ يېرىممۇ يوق مەن ،جېنمدا ئەمدى تور سودىسدىن چىكىنىشنى ،بولمىغاندا شۇ تور پۇل كارتىسىغا ئەمدى 2-300يۈئەندىن تمىتىپ سۇغۇرۇش (يەتمىگەندە باشقا كارتىدىن ئازراقتىن يۆتكەپ كىلىش)شەكىلىدە ،پۇل قڭيۇپ ،يۆتكەپ ئاندىن ئېلىشنى ئويلىشىۋاتىمەن،دىدىم ...

بىراق بۇ بىر ئىش بىلەنلا توردىكى ھەممىسىنى بىر تاياقتا ھەيدىسەك بولمايدۇ.
يەنىلا ئىھتىياتچان بولساق ئەڭ ياخشى. بىر نەرسە ئېنىق بولسۇنكى دۇنيا بىكارلىق نەرسە يوق.ئوخشاش قىممەتتىكى نەرسىلەرنىڭ باھا پەرقىنىڭ بەك چوڭ بولىشى دەل ئۇنىڭدا بىرەر مەسىلە بارلىقنى بىلدۈرىدۇ.
شۇڭلاشقا ھەممىمىز ئۆزىمىزگە ھۇشيار بولساق،بۇنداق ئىشلارنىڭ ئالدىنى ئالغىلى بولىدۇ.ماڭمىمۇ توردىن ئۇنى بۇنى ئېلىپ تۇرغىلى10يىلدىن ئېشىپتۇ.لىكىن ھازىرغىچە بىرەرسىگە توردا قويۇلۇپ باقمىدىم.
سىزگە تەۋسىيەرىم ئۇنداق بەك ئاچچىقلىنىپ كەتمەڭ،چۈنكى ئادەم يىقىلمىسا يۈگۈرۈشنى ئۈگنەلمەيدۇ.
بۇ قويمىچىلارنىڭ ھۈنىرى ھەربىر يېڭىلىق چىقسا شۇنىڭغا يانداش بىرەر يوچۇق تېپىپ ئادەم قويۇشنى ئىستەيدۇ.خۇددى يېڭى چىققان سىستىمىنىڭ يۇچقىغا ۋىروس چاپلاشقاندەك.ئۇلارنى تىللاپ تۈگەتكىلى بولمايدۇ،بۇ گەپلەرنى ئۇلار كۆرشىمۇ مۇمكىن،لىكىن تەسىر قىلمايدۇ.
شۇڭا ھەممىز ئۆزىمىزگە پۇختا بولساق ئەڭ ياخشى ئامال شۇ.
