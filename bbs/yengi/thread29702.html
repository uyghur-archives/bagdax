مائارىپ ئىدارىسىدىكى بىر قىسىم كادىرلار تۆۋەندىن كەلگەن ئوقۇتقۇچىلارغا نىمە

ئۈچۈن قوپال مۇئامىلە قىلىدۇ ؟



    مائارىپ ئىدارىسى بىرىنچى سەپتە جاپالىق ئىشلەۋاتقان باغۋەنلەر ئۈچۈن خىزمەت قىلىدىغان كۆۋرۈكتىن ئىبارەت .مەسىلەن ئوقۇتقۇچىلارنىڭ جانىجان مەنپەتىگە مۇناسىۋەتلىك بولغان ئۇنۋان ئېلىش ، يۇقۇرى دەرىجىلىك مائارىپ مەمۇرى تارماقلىرىنىڭ ئوقۇتقۇچىلارنىڭ مەنپەتىنى كاپالەتلەندۈرۈشكە ئائىت ھەر تۈرلۈك سىياسەت ، نىزاملىرىنى پۈتكۈل ئوقۇتقۇچىلارغا يەتكۈزۈشتىن ئىبارەت ، يەنە ئوقۇتۇتۇش پىروگرامممىسى ، ئوقۇتۇش پىلانى ، تەتقىقات ۋەزىپىسى ، يەنە مەكتەپلەرنىڭ ئوقۇتۇش -سۈپىتى ۋاھاكازالار . گەپنى يىغساق مائارىپ ئىدارىسى دادا دىگەن گەپ ،  لېكىن بۇ ئاتالمىش «دادا» ، «ئاكا» ئۆز قېرىندىشىغا ئاسانلىق كەلتۈرمەكتە يوق (بۇ يەردە شۇ  ئارىيەتچىلەرنى ئورنىنى كۆتىرىپ ئاشىۇنداق نام بىلەن ئوخشىتىش قىپ قويدۇم )، ئۇلارغا قىيىنچىلىق تۇغدۇرماقتا  ، ئۆزىنى ئاتالمىش ناھىيىتى چوڭ بىر يەردە ئىشلەيدىغان ھاكىم ، ۋالىدىن چوڭ چاغلاپ ئوقۇتقۇچىلانىڭ غەزىپىنى يۇقۇرى كۆتۈرمەكتە .مائارىپ ئىدارىسىغا ئارىيەت نامى بىلەن ،كەلگەن بىر قىسىملارنىڭ مائارىپ ئىدارىسىنىڭ يۈزىنى چۈشۈرۈپ قىلغان ئىشلىرى ئادەمنىڭ ئوغىسىنى قاينىتىدۇ .مەن مائارىپ ئىدارىسىنىڭ يېىڭىدىن تەيىنلەنگەن سېكرىتارىنىڭ تولىمۇ كەمتەر ، بىلىملىك كىشى ئىىكەنلىكىنى بىلىمەن ، بۇ كىشى بىر قېتىملىق ئوقۇتقۇچىلارنىڭ چوڭ يىغىنىدا :بىرىنچى سەپتە ئىشلەۋاتقان جاپاكەش باغۋەنلەر سىلەر جاپا چەكـتىڭلار ، بۈگۈن گەرچە دەم ئېلىش كۈنى بولسىمۇ قىممەتلىك ۋاقتىڭلارنى سەرپ قىلدىڭلار ،مەن بۇنىڭدىن كەچۈرۈم سورايمەن ،يىغىندىن تارقىغاندىن كېيىن يولدا دىققەت قىلىڭلار دەپ كىچىككىنە ھالقىلارغىچە بىز ئوقۇتقۇچىلارغا شۇنداق كۆيۈندى، مەن بۇنىڭدىن قاتتىق تەسىرلەندىم .  ھەم توردا بولسىمۇ سىزگە رەخمەت ئېيتىمەن !ئارىيەتچىلەر سىلەر سېكرىتارىمىزدىن ئۆگۈنۈڭلار .بىزدە: بۇغداي نېنىڭ  بولمىسا بۇغداي سۆزىڭ يوقما دەيدىغان ئېسىل ماقال بار ئەمەسمۇ _؟!
    بۇنىڭغا مىسال بۇ قېتىم ناھىيە مەسسئۇللىرىنىڭ بولۇپمۇ ناھيىلىك ھۆكۈمەتنىڭ «ھاكىمى»نىڭ مائارىپقا يۈكسەك دەرىجىدە كۆڭۈل بۆلىشى بىلەن ،ناھىيىمىز تەۋەسىدىن جەمئى 183نەپەر ئوقۇتقۇچى يىراقتىن -يېقىنغا ، يەنە بىر يەدىن شەھەرگە ، باشلانغۇچتىن - ئوتتۇرا مەكتەپكە دىگەندەك يۆتكەلگەن ئىدى. بۇنىڭ قاتارىدا كەمىنىگىمۇ ئۆچىرەت كىلىپ يۆتكەلگەن ئىدىم .كادىرلار تارماقلىرىدىن يۆتكىلگەنلىك ئۇچۇرىنى بىلگەندىن كېيىن مائارىپ ئىدارىسىگە بارسام مائارىپ ئىدارىسىنىڭ كادىرلار ئىشىغا مەسئۇل مۇئاۋىنى ئۇ ئىدارە باشلىغىمۇ ئەمەس شۇ  مائارىپ ئىدارىسىغا ئارىيەتكە كەلگەن (ئارىيەت نامى بىلەن ئىشلەۋاتقان 50-60يېقىن ئارىيەتچىلەردىن شۇنچىلىكى بار مەن بۇ يەردە ھەممىسىنى دىمىدىم )   مۇئاۋىن كادىرلار خىزمىتىگە مەسئۇل ئوخشايدۇ ئىشقىلىپ مائارىپ ئىدارىسىنىڭ كادىرلار ئىشىخانىسىدا ئشلەيدىغان بىرى كېلىش مەخسىدىمنى ئېيتىۋىدىم، ئۇ ماڭا جاۋابەن: بىز  ھازىر بۇ ئىشنى قىلمايمىز يەنى يۆتكەلگەنلەرنىڭ ئۇقتۇرۇشىنى بۈگۈن بەرمەيمىز ئالدىرساق ئىككى كۈندىن كېيىن ،مەڭا نىمە دەپ بېسىم قىلىسەن دەپ ناھىيىتى قوپاللىق بىلەن مۇئامىلە قىلغانتى غۇززدىدە ئاچىغىم كىلىپ ، مەن ئۇنىڭغا جاۋابەن: كادىرلار ئىدارىسى بۈگۈندىن قالماي دەرھالا  بېجىرىشنى باشلىمىساڭلار ئەسلىدىكى ئۇقتۇرشنى ئەمەلدىن قالدۇرىمىز دەيدۇ، دىيىشىمگىلا يەنە ناھىيىتى قوپاللىق بىلەن سەن ماڭا بۇيرۇق قىلالمايسەن ، دىيىشىگىلا ئاچىغىم تازا كەپتىكتەن: سەن نەدە ئىشلەپ كەلگەن  خىزمەت پوزىتسىيەسى  دىگەننى بىلەمسەن ؟ سەندەك دەرس ئۈتۈش خوشياقمايدىغان ھېلىقىدەك يوللار بىلەن كۈچۈكلىنىپ كەپ قالغان سەندەك بىرنىمىلەرنى جىق كۆرگەنمەن ، دەپ قويۇپ مەكتەپكە قاييتىم  چۈشكە يېقىن  سائەت 11بىلەن بىزنىڭ مەكتەپنىڭ كادىرلار ئىشىغا مەسئۇل يولداش ئۇقتۇرۇشنى ئەكەلگەنلىكىنى ئېيتتى كىرسەم راست دىگەندەك تۇرۇپتۇ ، شۇنىڭ بىلەن ۋەسسالام بولغان بولدى ئۇنىڭدىن كېيىن مائارىپ ئىدارىسىغا بېرىپ مائاش ئارخىپىنى  ئالماقچى بولغاندا (بۇنىمۇ كادىرلار ئىدارىسى لازىم قىلىدىكەن )  ئارخىپنى باشقۇرغۇچى ئارىيەتچىگە ئەھۋالنى ئېيتتسام ئۇ ئارىيەتچىمۇناھىيىتى قوپال بەرمەيدىغانلىىقىنى ئېيتتى (ئۇ  ئارىيەتچىنىڭ دىيىشچە مەكتەپنىىڭ كادىرلار ئىشىغا مەسئۇل مۇدىرى كەلسە بېرەمىش مەن  ئارىيەتچىگە مەن ھۆججەت يىزىپ بىرەي دىسەممۇ زادى ئۇنمىدى    ) بۇ ئىشلارغا زادى چىدىمىدىم  غۇرۇرۇم ، قاتتىق ئازاپلاندى ، چۈنكى مۇشۇ رايۇنىمىز مائارىپىدا نەچچە ئون يىل جاپالىق ئىشلەپ تۇرۇپ مۇشۇنچىلىك يۈزىمىز بولمىسا (ئاشۇ تۆنۈگۈن بۇناكۈن خىزمەتكە چىققان مائارىپ ئىدارىسىغا ئارىيەتكە كەلگەن جانباختى نىمىلەر بىزگە مۇشۇنداقمۇئامىلە قىلىدىغانغا نىمە ھەددى ؟ مەكتەپلەردە بىرەر يېرىم سائات دەرسنىڭ ھۆددىسىدىن چىقالمايدىغان بۇنداق جانباختىلار  )  ماقۇل ئارخىپنىغۇ بىر تۇتاشمۇ باشقۇرسۇنمۇ دەيلى لىكىن بۇ دىگەن ھۆكۈمەت ھەركىتى تۇرسا بۇلارنى ئەسلى ئۆزى بىر تۇتاشلا كادىرلار ئىدارىسىغا ئاپىرىپ گۇۋاھلىقتىن ئۆتكۈزۈپ بولۇپ ئاندىن ياندۇرۇپ ئەكەتسە بولىدىغان ئىش تۇرسا ئۆزى مېڭىپ قىلمىسۇن بىر ئادەم بۇ ئىشنىڭ موھىملىقىنى دەپ ۋاقىت بىلەن مۇناسىۋەتلىك بۇ ئىشتا مۇشۇنداق پوزىتسىيەدە بولسا قانداق بولىدۇ ؟ گەپنىڭ راستىنى دىسەم بۇلارنىڭ  ساپاسى يوق تۆۋەندە دېھقانلىرىمىزنىڭ پەرزەنتىلىرىنى ئاخماق قىلىپ جانباقىدىغان جانباختىلار دىسەك ئارتۇقچە كەتمەيدۇ ئەگەر ئۇلاردا بىزە ساپا ، خىزمەت پوزىتسىيە  ئۇقۇمى ، باشقىلارغا مۇئامىلە قىلىشتىكى ئەخلاق ئۇقۇمى ، ۋىز دان  بولغىنىدا ئىدى ئۇنداق قىلمىغان بولاتتى ، خوش مەنغۇ ناھىيە مەركىزىنىڭ كۆزى بولغان نۇختىلىق بىر مەكتەپتە ئىشلەپ مۇشۇنداق تەڭسىزلىككە ئۇچىراپتىمەن ، ناھىيە مەركىزىدىن يىراق بولغان يېزا - قىشلاقلاردا ئىشلەۋاتقان جاپاكەش ئۇستازلارغا قانداق مۇئامىلە قىلا ؟سىز ئاشۇ جاپاكەش باغۋەنلەرنىڭ خزمەتتە ھاردىم تالدىم دىمەي مىللى مائارىپ ئۈچۈن ئۆچمەس زور -تۆھپىلەرنى قوشىۋاتقانلىقىنى ئويلىغىنىڭىدا سىزنىڭ ئاشۇنداق مەدىئنيةتسئزمۇئامىلىڭىز قانداق تەسىر بېرەر ؟سىزدە بىر ئادەمىلىك مۇئامىلىسىگە خاس ۋىزدان بارمۇ ؟سىزچە سىز ئۆزىڭىز بەك چوڭ يەردە ئىشلەۋاتقان ئوخشىمامسىز ؟ سىز ئۆزىڭىزىنى ئاشۇلارنىڭ ئورنىغا قويۇپ دەڭسەپ باختىڭىزمۇ ؟ سىز  ئارىيەت بولۇپ يۆتكىلىش ئۈچۈن نورمال بولمىغان ھەركەتلەرنى قىلىۋاتقىنىڭىزدا نىمىلەرنى ئويلىدىڭىز ؟سىز بىرەر -يېرىم سائەت مۇنبەرگە چىقىپ ئوقۇغۇچىلارغا يارىغۇدەك دەرس سۆزلەش ساپايىڭىز بارمۇ ؟ مەن ئاپتۇنۇم رايۇنلۇق مائارىپ نازارىتىدىكى بىر قانچە باشقارما باشلىقىنى تونۇيمەن ھەم ئۇلار بىلەن ھەمداستىخان بولۇپ باققان ، سىز ئاشۇلارنىڭ 10دىن بىرچىلىك ھوقۇقنى بېرىپ قويسا ئۆزىڭىزنى تۇتالماي مەن ھەجەپ چوڭ باشلىق بولۇپ كەتتىم دەيدىغان ئوخشىمامسىز  ؟ مەن مائارىپ نازارىتىنىڭ نازىرى يولداش تۇرسۇن ئىبرايىم بىلەنمۇ  ئۆزنىڭ ئىشخانىسىدا سۆھبەتلىشىش پۇرسىتىگە مۇيەسسەر بولغان بىز ئالاھەزەل 30 مىنۇتتەك سۆھبەتلەشكەن ..............................
   قېرندىشلىرىم ئادەمنى قىلغان ئىشى بېزەيدۇ بەك يوغىنغاپ كەتمەيلى ، "ئېلى داۋۇت"تەك كةسئپتة پئشقان ئارىيەتچىلەردىنمۇ ئۆگىنىپ قويايلى !  دوستۇم بەك ھۆمىرەكلەپ كەتمەڭ جاپاكەش باغۋەنلەرنىڭ نەرىزىدىن چۇشسىىڭز ھامان بىر كۈنى نەس بېسىپ ھالىڭىز خاراپلاشقاندا سىزگە تۈكۇرۇپ ئۆتۈپ كېتىدۇ . يادىڭزدا بولسۇن مائارىپ ئىدارىسنى بېشىغا كەيگەن ئۆزىنى ھىچكىمگە تەڭ قىلمايدىغان ھېلىقى پوچى ھازىر بىز بىلەن بىرگە ئىشلەۋاتىدۇ جۇمۇڭ !     مەن بۇنى مەلۇم بىرى دەپ قاراتمىلىقى بار ئوتتۇرغا قويمۇدۇم شۇڭا قېرىنداشلىرىمىزنىڭ توغرا قاراشىنى سورايمەن يەنە بىر جەھەتتن يۇقارقىدەك ئەھۋاللار ھەرقايسى جايلاردا ئوخشاش بولمىغان ھالدا مەۋجۇت بولىشى مۇمكىن چۈنكى ئاپتۇنۇم رايۇنلۇق پارتىكوم چىققارغان بىر ھۆجەتتە  خىزمەت پوزىتسىسىيەسىنى ، خىزمەت ئىستىلىنى ياخشلىلاشقا ئائىت بەلگىلىمە ئوتتۇرغا قويۇلغاندىن بۇيان خىزمەت ئىستىلىدا كۆپلەپ ياخشىلىنىشلار بارلىققا كەلگەن بولسىمۇ يەنە يۇقارقىدەك بىر قىسىم مەسىلىلەر ساخلانماقتا ، بۇنى قەتئى تۈزۈتۈش كېرەك .(بەلكىم ئاشۇ ئارىيەتچىلەر  مائارىپ ئىدارىسىنىڭ كەسپى شىتات خادىمى بويىىنچە يۆتكىلىپ كەپ بولغاندۇ) كادىرلار،ئىزتىمائى  كاپالەت ئىدارىسى دىگەننى   بۇرۇنقى نام بىلەن كادىرلار ئىدارىسى دەپلا ئېلىۋەردىم توغرا چۈشىنىشىڭلارنى سورايمەن.
