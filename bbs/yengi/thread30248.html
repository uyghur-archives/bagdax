تارىخنىڭ ئۆزگۈرىشىنى تېخنىكىنىڭ ئۆزگىرىشىدىن ئايرىپ قارىغىلى بولمايدۇ.گەرچە نۇرغۇن ئامىللار ئۆزگۈرۈشكە سەۋەپچى بولسىمۇ تېخنىكىنىڭ ئۆزگىرىشى بۇ جەرياندا ئىنتايىن مۇھىم رول ئوينايدۇ.مىسىلسىز گۈللەنگەن تارىم ۋادىسى يىپەك يولى ئارقىلىق ئۆزگۈللىنىشىنى خىلى بىر مەزگىل داۋاملاشتۇرغان بولسا ،يىڭى دىڭىز يوللىرىنىڭ ئىچىلىپ ، يىپەك يۇلىنىڭ خاراپلىشىشى تارىم ۋادىسىنى بارا-بارا دۇنيادىن ئايرىلغان يىگانە ئارالغا ئايلاندۇرۇپ،تاكى 20-ئەسىرنىڭ قۇڭغۇراقلىرى جاراڭلاۋاتقان دەۋىرلەرگىچە ئۇتتۇرا ئەسىر مۇھىتىدا قالدۇرغان ئىدى. شۆبھىسىزكى يىڭى دىڭىز يوللىرىنىڭ ئىچىلىشى يەنىلا تېخنىكىنىڭ تەرەققىياتىدىن كەلگەن بۇلۇپ،كومپاس...قاتارلىق يىڭىچە تېخنكىلار ۋە جۇغراپىيەلىك ئامىللار يىڭى دىڭىز يوللىرىنىڭ ئىچىلىپ،يىپەك يولىنىڭ خاراپلىشىشدىكى ۋاستىلىك سەۋەپلەر ھىساپلىنىدۇ.
يىڭى تېخنىكىنڭ دۇنيانى ئۆزگەرتىشىدىكى ئەڭ بۈيۈك نامايەندىلەر يەنلا سانائەت ئىنقىلابىنىڭ يۈز بېرىشىدە بىۋاستە ئەكىس ئىتىدۇ.
ئىنقىلاپ دىگىنىمىز كونىنىڭ ئورنىنى يىڭىلار ئىلىش دىمەكتۇر.كونىنى يۇقىتىپ ،يىڭىنى پەيدا قىلىش دىمەكتۇر . سانائەت ئىنقىلابى بولسا ئشلەپچىقىرىش شەكىللرىنىڭ يىڭى بىر خىل شەكىلگە كۆچكەنلىكىنىڭ ئىپادىسىدۇر.
ھالبۇكى  ئالدىنقى ئۈچ قىتىملىق سانائەت ئىنقىلابى ھارغىن تارىم ۋادىسى بىلەن ئۆز ۋاقتىدا ئۇچىرىشالمىدى .ھەتتا دۇنيا 300يىلغا يىقىن ۋاقىت ئىچىدە بايقىغان ،ئۆزلەشتۈرگەن، ئۆزلىرىنى يۈكسەلدۈرگەن تېخنىكىلارغاتارىم ۋادىسى بىراقلا يۈزلىنىشكە،بىراقلا قۇبۇل قىلىشقا مەجبور بولدى.تەرەققىياتتىن ئىبارەت بۇ ئۇچقۇر ئاتقا تېخنىكىنى ئەڭ ئالدىدا ئىگەللىگەنلەر ،ئۇچۇرنى بىرىنچى بۇلۇپ قۇبۇل قىلغانلار مىنىۋالدىكى ،بىزدەك ئۇنى قۇغلاپ ماڭغانلار تەرەققىياتقا زادىلا يىتەلمىدۇق.
خوش ئۇنداق بولسا بىز تارىخقا نەزەر سىلىپ باقايلى:
1-قىتىملىق سانائەت ئىنقىلابى ئەنگىلىيەدە پارتىلىغان بۇلۇپ،18-ئەسىرنىڭ 60-يىللىرىدىن 19-ئەسىرنىڭ ئۇتتۇرلېرىغىچە ئىلىپ بېرىلغان، بۇ قىتىمقى ئىنقىلاپنىڭ مۇھىم ئالاھىدىلىكى ئىنسانىيەت پار ماشىنىسى دەۋرىگە كىرگەن.
2-قىتىملىق سانائەت ئىنقىلابى 19-ئەسىرنىڭ كىيىنكى يېرىمىدىن 20-ئەسىرنىڭ باشلىرىغىچە ئىلىپ بېرىلغان بۇلۇپ، ئىنسانىيەت ئىلىكتىر دەۋرىگە كىرگەن .
3-قىتىملىق سانائەت ئىنقىلابى 2-دۇنيا ئۇرۇشىدىن كىيىن مەيدانغا كەلگەن بۇلۇپ،ئىنسانىيەتنىڭ پەن-تېخنىكا دەۋرىگە كىرگەنلىكىدىن دېرەك بېرىدۇ،ئۇچۇر تېخنىكىسى،كىلون تېخنىكىسى،ئالەم قاتنىشى تېخنىكىسى قاتارلىق تېخنىكىلار بۇ قىتىمقى سانائەت ئىنقىلابىنىڭ مۇھىم بەلگىسى بۇلۇپ ھىساپلىنىدۇ.
1765-يىلى باشلانغان 1-قىتىملىق سانائەت ئىنقىلابىدىن تاكى ئىككىنچى دۇنيا ئۇرۇشىدىن كىيىنكى دەۋىرگىچە بولغان 300يىل جەريانىدا تارىم ۋادىسى بۇ ئۆزگۈرۈشلەردىن يىراق قالدى. بۇ تارىم ۋادىسىنىڭ تېخىمۇ كەينىدە قىلىشىغا سەۋەپچى بولدى.
بۈگۈنكى ئۇچۇر دەۋرىدە بولىمىش بىزلەر ئۇچۇر ۋاستىلىرىنى تۇغرا يەرگە ئىشلىتىپ، قېرىنداشلىرىمىزنى يىڭى ئۇچۇرلاردىن خەۋەردار قىلىشنى ئۆزىمىزنىڭ مۇھىم ۋەزىپىسى قىلىش ئۆمىدىدە گىرمانىيەدە باشلىنىپ ،كۈنسىرى يىقىنلاپ كىلىۋاتقان يىڭى بىر قىتىملىق سانائەت ئۆزگىرىشىدىن چۈشەنگەنلىرىمنى دوستلار بىلەن ئورتاقلىشىش مەقسىتىدە تۆتىنچى قىتىملىق سانائەت ئىنقىلابىنىڭ تېخنىكا يۈزلىنىشىنى قىسقىچە بايان قىلىپ ئۈتىمەن.
بۇ قىتىملىق سانائەت ئىنقىلابى «سانائەت4.0» دەۋرى دەپ ئاتىلىدىغان بۇلۇپ ، مەنمۇ تۈۋەندە «سانائەت 4.0» دەپ ئاتاشنى مۇۋاپىق كۆردۈم.
«سانائەت4.0» سىتىراتىگىيە تەتقىقات تۈرى بولسا گىرمانىيە فىدىراتسىيە ئۇقۇتۇش -تەتقىقات مىنىستىرلىكى ۋە ئىقتىساد-تېخنىكا مىنىستىرلىكى بىرلىكتە ياردەم مەبلىغى بىلەن تەمىنلىگەن ،گىرمانىيە قۇرۇلۇش ئاكادىمىيەسى ، فىرائۇنخوفىر جەمىئىتى{1}،سىمىنىس شىركىتى قاتارلىق گىرمانىيە ئىلىم-پەن ساھەسى ،ئىشلەپچىقىرىش گېگانىتلىرىنىڭ تەشەببۇسى ۋە تۈرتكىسىدە تۇرغۇزۇلغان بىر تۈر بۇلۇپ،ھازىر ئاللىقاچان دۆلەت سىتىراتىگىيىسى دەرىجىسىگە كۈتۈرۈلۈپ،گىرمانىيە فىدىراتسىيە ھۆكۈمىتىنىڭ 200مىليون ياۋرۇ مەبلەغ سىلىشىغا ئېرىشتى.
گىرمانىيە ھۆكۈمىتى ئوتتۇرىغا قويغان «سانائەت 4.0» سىتىراتىگىيەسىنى 2013-يىلى 4-ئايدىكى خانوۋىر سانائەت يەرمەنكىسىدە رەسمىي يۈرگۈزۈشكە باشلىغان . ئاساسلىق مەقسىتى گىرمانىيەنىڭ سانائەت رىقابەت كۈچىنى ئۆستۈرۈش،يىڭى بىر قىتىملىق سانائەت ئىنقىلابىدا يىتەكچى ئورۇننى ئىگەللەشنى مەقسەت قىلغان بۇلۇپ، مەزكور سىتىراتىگىيە گىرمانىيەنىڭ ئىشلەپچىقىرىش ،ئىلىم-پەن ساھەسىدە بىردەك ئىتىراپ قىلىنىشىقا ئېرىشكەن،  فىرائۇنخوفىر جەمىيىتىنىڭ قارمىقىدىكى 6-7ئىشلەپچىقىرىش ساھەسىدىكى پەن-تەتقىقات مەركەزلىرى دەرھال «سانائەت4.0» ئۇقۇمىنى سانائەت ساھەسىگە رەسمىي ئىلىپ كىرگەن، سىمىنىس شىركىتى ئاللىبۇرۇنلا بۇ ئۇقۇمنى سانائەت يۇمشاق دىتاللىرىنى ئىچىش ۋە ئىشلەپچىقىرىش كونتىرول سېستىمىسىغا ئەكىرىپ قوللىنىشنى باشلىۋەتكەن ئىدى.
«سانائەت4.0»ئۇقۇمى بولسا ئاساسلىقى سانائەت ئىشلەپچىقىرىش جەريانىدىكى مەركەزلىك كونتىرول سېستىمىسى ئەندىزىسىنىڭ  تارقاق ھالەتتىكى كۈچلەندۈرۈلگەن كونتىرول سېستىما ئەندىزىسىگە ئۆزگۈرۈشىنى كۆرسىتىدۇ. بۇنىڭ ئاساسىي مەقسىتى بولسا جانلىقلىقى يۇقىرى بولغان خاسلىققا ئىگە ، سىفىرلاشقان ،مەھسۇلات بىلەن مۇلازىمەت بىرلەشكەن ئىشلەپچىقىرىش ئەندىزىسى قۇرۇپ چىقىشتىن ئىبارەت. بۇگۈنكى زاماندا يىڭى قىممەت يارىتىۋاتقان جەرياندا ئۆزلۈكسىز ئۆزگىرىش بۇلىۋاتىدۇ، كەسىپ زەنجىرىنىڭ ئىش ھالقا تەقسىماتى قايتا تەشكىللەنمەكتە.
گىرمانىيە ئىلىم-پەن ۋە ئىشلەپچىقىرىش ساھەسىنىڭ ئورتاق قارىشى شۇكى،«سانائەت4.0» ئۇقۇمى بولسا  «ئەقلىي ئىقتىدارلىق ياساش»نى ئاساسىي ئىقىم قىلغان تۆتىنچى قىتىملىق سانائەت ئىنقىلابىدۇر ياكى ئىنقىلاپ خاراكتىرىگە ئىگە ئىشلەپچىقىرىش ئۇسولىدۇر.  بۇ سىتىراتىگىيە ئۇچۇر-ئالاقە تېخنىكىسى ، تور بوشلۇقى مەۋھۇم سېستىمىسىدىن تۇلۇق پايدىلىنىش ئارقىلىق بۇنى يەنە  ئۇچۇر فىزىكا سېستىمىسى بىلەن بىرلەشتۈرۈشتىن ئىبارەت شەكىل ئارقىلىق ياساش كەسپىنى ئەقىلىي ئىقتىدارلاشتۇرۇشنى كۆرسىتىدۇ.
«سانائەت4.0» تۈرى ئاساسلىقى ئىككى چوڭ تېمىغا بۈلىنىدۇ. بىرى: «ئەقلىي ئىقتىدارلىق زاۋۇت» بۇلۇپ ،ئاساسلىقى ئەقىلىي ئىقتىدارلىق ئىشلەپچىقىرىش سىستىمىسى ۋە جەريانىنى شۇنداقلا تورلاشقان كەڭ ئىشلەپچىقىرىش ئەسلىھەلىرىنىڭ ئەمەلگە ئىشىشىنى تەتقىق قىلىدۇ؛ يەنە بىرى:«ئەقلىي ئىقتىدارلىق ئىشلەپچىقىرىش» بۇلۇپ ،ئاساسلىقى پۈتكۈل كارخانىنىڭ ئىشلەپچىقىرىش ماددىي ئەشيا ئۇبۇرۇت باشقۇرۇش، ئادەم بىلەن ماشىنىنىڭ ئۆز-ئارا بىرىكىشى ۋە ئۈچ ئۆلچەم تېخنىكىسىنىڭ سانائەت ئىشلەپچىقىرىش جەريانىدا قوللىنىشىنى تەتقىق قىلىدۇ.  بۇ تۈر ئوتتۇرا ۋە كىچىك تىپتىكى كارخانىلارنى جەلىپ قىلىشقا ئىنتايىن ئەھمىيەت بېرىدىغان بۇلۇپ، ئوتتۇرا ۋە كىچىك تىپتىكى كارخانىلارنىڭ يىڭى بىر ئەۋلات ئەقىلىي ئىشلەپچىقىرىش تېخنىكىسىنىڭ قوللانغۇچىسى ۋە نەپ ئالغۇچىسى بۇلۇشىنى شۇنىڭ بىلەن بىللە ئىلغار سانائەت ئىشلەپچىقىرىش تېىنىكىسىنىڭ ياراتقۇچىسى ۋە تەمىنلىگۈچىسىگە ئايلىنىشىنى نىشان قىلىدۇ.
گىرمانىيەنىڭ ياساش كەسپى بولسا  دۇنيادىكى  رىقابەت كۈچى ئەڭ كۈچلۈك  ياساش كەسپىنىڭ بىرى بۇلۇپ ،گىرمانىيە  بۇ ساھەنىڭ سەركىسى ھىسابلىنىدۇ. گىرمانىيەنىڭ بۇ خىل ئەۋزەللىكىنىڭ مەنبەسى گىرمانىيەنىڭ سانائەت- تېخنىكا مەھسۇلاتلىرىنى تەتقىق قىلىش ۋەئىچىش ئىشلېرىغا، مۇرەككەپ سانائەت جەريانىدىكى باشقۇرۇشقا ئالاھىدە ئەھمىيەت بەرگەنلىكىدە. گىرمانىيە كۈچلۈك بولغان ئەسلىھە ۋە سېخ ياساش سانائىتىگە ئىگە ،دۇنيا ئۇچۇر تېخنىكا ساھەسىدىكى ئىقتىدار سەۋىيەسىمۇ ئىنتايىن يۇقىرى، قىستۇرما سېستىما ۋە ئاپتۇماتلاشتۇرۇش قۇرۇلۇشى جەھەتتىمۇ ئىنتايىن كەسىپلەشكەن تېخنىكىسى بار، بۇ ئامىللار گىرمانىيەنىڭ ياساش قۇرۇلۇشى سانائىتىدە باشلامچى ئۇرۇنغا چىقىشىدا ئاساس بولغان. «سانائەت4.0» سىتىراتىگىيەسىنى ئەمەلگە ئاشۇرۇش ئارقىلىق ،گىرمانىيە يىڭى بىر ئەۋلات سانائەت-ئىشلەپچىقىرىش تېخنىكىسىنى تەمىنلىگۈچى دۆلەت ۋە ئاساسلىق بازارغا ئايلىنالايدۇ.شۇنداقلا دۆلەت ئىچىدىكى ياساش كەسپىنى ئىزچىل تەرەققىي قىلدۇرۇشنىڭ ئالدىنقى شەرتى ئاستىدا خەلىقئارادىكى رىقابەت كۈچىنى تېخىمۇ ئۈستۈرەلەيدۇ.

بۇنىڭدىن كۆرەلەيمىزكى يىقىن كەلگۈسىدە يىڭى تېخنىكا يىڭىچە بى
دۇنيانى بارلىققا كەلتۈرۈشى مۇمكىن،يىڭىچە ئىشلەپچىقىرىش ئۇسۇلىغا چۇقۇم يىڭىچە تەپەككۈر بىلەن ماسلىشىش كېرەك.يىڭىچە تەپەككۈر يىتىلدۈرۈش ئۈچۈن توختىماي ئۈگىنىدىغان «مەڭگۈلۈك ئۆگىنىش دەۋرى»ھالىتىگە كىرىش كېرەك.
دۇنيا قەيەردە؟ بىز قەيەردە؟ دىگەن سۇئالنى ھەركۈنى ئۆزىڭىزدىن سوراشنى ئۇنۇتماڭ.شۇندىلا رىقابەت كەسكىن،يىڭىلىنىش تىز بۇ دۇنيادا ئۆز مەۋجوتلۇقىمىزنى ساقلاپ تۇرالايمىز!
