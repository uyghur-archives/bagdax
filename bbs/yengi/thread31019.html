نەشىر قىلىپ تارقاتقۇچى : شىنجاڭ خەلىق باش نەشىرىياتى

   تەخمىنەن بىرەر يىللار ئىلگىرى،مىسرانىم تور بېكىتىنىڭ قۇرۇلغانلىقىنىڭ  يەتتە يىللىقىنى تەبرىكلەش مۇناسىۋىتى بىلەن ئۆتكۈزۈلگەن پائالىيەتتنىڭ قىسقافىلىمىنى كۆرگەن ئىدىم . بۇ سورۇنغا ئاتاقلىق يازغۇچىمىز، مىللىتىمىزنىڭ پەخر-ى جالالىدىن بەھرام ئاكا ئالاھىدە چاقىرىلغان مېھمان سۈپىتىدە لېكسىيە سۆزلىگەن ئىدى.ئەزەلدىن مىنى بەكمۇ قايىل قىلىدىغان بۇ ئۇستاز ئۆزىنىڭ كىتاب ،رومانلارنى يېزىش،كۈندىلىك ئىش-پائالىيەتلىرىنى قانداق تەرتىپلىك ئورۇنلاشتۇرۇش ھەققىدە قىسقىچە سۆز قىلغان ئىدى.شۇ جەرياندىكى تارتقان جاپا-مۇشەقەتلىرى ،ئۆزىگە قويغان تەلىپىنىڭ يۇقىرىلىقى،ھاياتقا،ئىنسانىيەتكە بولغان قىممەت قارىشىنىڭ يۇقىرىلىقى مىنى تېخىمۇ قول قۇيۇشقا مەجبۇر قىلدى.شۇ لېكسىيەسىدە ئۆزىنىڭ مىسرانىم تور بېكىتىنىڭ قۇرغۇچىسى ئابابەكرى مۇختەر ۋە سۈرەت ڧىلىم ئالغۇچىسى جۈمەخۇنلار بىلەن تونۇشقاندىن كىيىن ،ئۇلارنىڭ ئىش-ئىزلىرىدىن چوڭقۇر تەسىرلەنگەنلىكىنى ،شۇلار توغرىسىدا يازغان   <تىۋىشسىز دۇنيا > ناملىق روماننىڭ پات ئارىدا نەشىرىدىن چىقىدىغانلىقىدىن خەۋەر بەرگەن ئىدى.مەن شۇ ۋاقىتتىن باشلاپ مۇشۇ كىتابنى تەقەززالىىق بىلەن كۈتتۈم.مەلۇم سەۋەپلەر تۈپەيلى بۇ كىتاب مۇشۇ ئايدا ئاخىرى قولۇمغا يېتىپ كەلدى.
   تېز پوچتىدىن كىتابنى ئالغاندىن كىيىن ،ئوراقلىق قەغەزنى ئاستا سويدۇم،كىتابنىڭ مۇقاۋىسى كۈلرەڭ ئىدى .تۇنجى قېتىم كۈرۈپلا ئنتايىن ئارەمبەخىش  راھەت  تۇيغۇغا كېلىپ قالدىم.كۈلرەڭ،سېرىق، قىزىل رەڭلەرگە ئوخشاش كۆزنى قاماشتۇرمايتتى .كۆڭۈلنىڭ ئەڭ چوڭقۇر قاتلاملىرىدىكى داۋالغۇپ تۇرغان دولقۇنلارغا سۈرلۈك ھەم ئارەمبەخىش ئاتا قىلاتتى..پۈتۈن ئەتراپىمدىكى شاۋقۇن-سۇرەنلەر توختاپ ،تىۋىشسىز دۇنياغا كىرىپ كىتىۋاتاتتىم .
  ئۆزىنىڭ ئەقىل-پاراسىتى ،باتۇر -شىجائەتچانلىقى،قەھرى ھەم مېھرىنىڭ تۇرتكىسىدە يۇرتنىڭ ئاتاقلىق سەركىسى ئاتالغان خاسىيەتلىك ۋادا قىزى توختى بورىچى نەچچە ئەسىر باشقىلارنىڭ ئاغزىدا تەرىپلىنىپ كەلدى .بۇ ئىتائەتچان موماينىڭ تەربىيىسى ئاستىدا چوڭ بولغان جۈمەتاي تەبىئەت بىلەن ئۆچەكىشىپ ئىنتايىن زېرەك چوڭ بولدى.ئۇ كىچىكىدىن ئۆگۈنۈشكە ھېرىسمەن بۇلۇپ يېشى توشماي تۇرۇپلا باشلانغۇچ مەكتەپكە بېرىپ دەرس ئاڭلايتتى.قۇشلار ۋە ھەرخىل جانىۋارلار بىلەن پاراڭلىشاتتى.ئەتراپىدىكى ھەممە نەرسىگە ئالاھىدە ئشتىياقى بار بۇلۇپ ،ئسمى جىسمىغا لايىق تەبىئەت ئوغلى ئىدى. ئۇنىڭغا مومىسى ھېكايە چۆچەك ئېيتىش،بۇرۇنقىلارنىڭ كەچۇرمىشلىرىنى سۆزلەپ بېرىش ھەتتا ئۆز ئەمەلىيىتىدە كۆرسىتىش ئارقىلىق ئۇنىڭغا كىچىكىىدىن باشلاپ ياخشى تەربىيە قىلاتتى.بالا تەربىيلەش ھەھىقەتەن ئاپپاق قەغەزگە گۈزەل رەسىىملەرنى سىزىپ چىققانغا ئوخشاش ھەربىر چىكىت ھەربىر سىزىقلارغا سەل قاراشقا بولمايتتى.مەن بۇ توختى بورى مومىمىزدىن ياخشى ئلھام ئېلىپ كىىينكى تۇرمۇشۇمغا قانداق تەدبىقلاش توغرىسىدا كۆپ ئويلىنىشلارنى قىلدىم،بۇ موماينىڭ ئەخلاق پەزىلىتى ئىش بېجىرىش پىرىىنسىپى ماڭا ھەقىقەتەن ياخشى ئۈلگە بولدى.جۈمەتاي شۇنداق ياخشى چوڭ بۇلىۋاتقان كۈنلەرنىڭ بىرىدە ،ئۇنىڭ تىۋىشسىز ھاياتى باشلاندى.<يىتىمنىڭ ئاغزى ئاشقا ئاشقا يەتكەندە بۇرنى قاناپتۇ> <تەلەيسىز تۆگىگە مىنسە ،پۇتىنى ئىت چىشلەپتۇ...... >دېگەندەك ئسسىق ئۆتۈپ كېتىش سەۋەپلىك تەبىئەتتىكى بارلىق تىۋىشلاردىن مەھرۇم قېلىپ ،بىر ئۆمۈر تىمتاسلىقتا ئۆتۈش نىمىدىگەن تاساددىپىيلىق ھە...........
  ھەر قانداق ئنسان يوقاتقاندىلا شۇ نەرسىنىڭ قىممەتلىكىنى ھېس  قىلىدۇ،مۇشۇ پەيىتتە مەن ھېس قىلدىم،مەن نېمىدىگەن بەخىتلىك ،مەن نېمىدىگەن تەلەيلىك ھە،مەن ئەركىن نەپەس ئېلىپ ،تەبىئەتنىڭ شاۋقۇن-سۈرەنلىرىنى ئاڭلىيالايمەن،كۆڭلۈم خالىغان جايلارغا بېرىپ ،روھىم ۋە جىسمسمدىن ھۇزۇر ھالاۋەت ھېس قىلالەيمەن.مەن بۇرۇن ئويلاپ باقمىغان ،ئۆزۈمنىڭ بۇنچىىلىك بەخىتلىك ئىكەنلىكىمنى ھەرگىزمۇ ئويلاپ باقمىغان.
    ئوقۇپ مۇشۇ  يەرگە كەلگەندە تولىمۇ ئېچىندىم،كىچىك ناتىۋاننىڭ ئاخىرقى ھاياتىدىن ئەنسىرىشكە باشلىدىم،ئۇ مۇشۇنداقلا تۈگۈشۇپ كېتەرمۇ؟جېنىنى قانداق باقار؟ھەي،،مەن ئەسلىدە جۈمەتايدىن ،قۇلىقى ئاجىز ،تىلى ئاجىز ،پۇتى ئاجىز ،ئاتىلىشقا مېيىپ ئەمىلىيەتتە ھەممىمىزدىن  ساق كىشىلەردىن ئەنسىرىمەي ئۆز-ئۆزۈمنىڭ كىيىنلىكىدىن ئەندىشە قىلسام بوپتىكەن.جۇمەتاي ھەرگىزمۇ مېيىپ ئەمەس ،ئۇ ئۆز تىرىشچانلىقىغا تايىنىپ ،كۆپلەپ مەشىق قىلىپ مېڭىش ئىقتقدارىنى ئەسلىگە كەلتۈرىۋالغاندىلا قاتتىق قايىل بولغان ئىدىم .ئوقۇشقا بولغان ھېرىسمەنلىكى ،ئۆزىگە بولغان تەلەپچانلىقى ئۇنى مەكتەپ قوينىغا قايتا ئېلىپ كىردى.شۇ چاغدا جۈمەتاي قاتتىق ھاياجانلاندى،ئۈمۈد-ئارزۇلىرىنىڭ يېقىلغان چىرىقىنى كۆرگەندەك بولدى،قەشقەر گاس-گاچىلار مەكتىپى بىرغايىبانە  ئۇچقۇن بولۇپ مىسكىن ھاياتىنى نۇرلۇق گۇلخانلارغا تۇتاشتۇردى.مەن شۇ چاغدا ئۆزۈمدىن تولىمۇ نۇمۇس قىلدىم ،نەچچە قېتىم ھازىرقى ئالىي مەكتەپكە قانچىلىك ئاددىي كېلىپ قالغانلىقىمنى ساۋاقداشلىرىمغا سۆزلەپ بېرىپ بۇنداق ياخشى ئۆگۈنۈش سورىنىغا كېلەلىگەنلىكىمدىن ئازراقمۇ  مىننەتدارلىق تۇيغۇسىدا بولۇپ باقمىغان  ئىدىم.لېكىن  بىزنىڭ مەڭگۈ ھۆرمەتلىشىمىزگە ئەرزىيدىغان ئاتالمىش مېيىپلار خەلقىمىز ،ھۆكۈمىتىمىز ،ئىللىق ماكانىمىز يارىتىپ بەرگەن ئۆگۈنۈش سۇرۇنلىرىنى جېنىدىن ئەزىز كۆرۇپ قەدىرلەپ ،ئۆز قەدىر-قىممىتىنى يارىتىۋاتمامدۇ؟!!!!لېكىن ھازىرقى ئالىي مەكتەب ئوقۇغۇچىلىرى ،ئوتتۇ را مەكتەب ئوقۇغۇچىلىرى مەكتەب بىلىم سورۇنلىرىنى ئاغرىنىش رازى بولماسلىقلار بىلەنلا تەرىپلەپ دۆلەتكە ،جەمىئيەتكە ئائىلىگە ،ئاتا-ئانىمىزغا مىننەتدارلىق تۇيغۇسىدا بىرەر قېتىممۇ كۈلۈپرەك قاراپ قويمايمىز.جۈمەتاي ،مەرزىيەلەر ئاڭلاش ،سۆزلەش ئىقتىقدارىدىن مەھرۇم قالغانلىقى ئۈچۈن مېيىپ سانالسا ،ئۇنداقتا بىز شۈكۇر قىلىش ،مىننەتدارلىق بىلدۇرۇش ،رازى بۇلۇشتىن مەھرۇم قالغان مېيىپلارمۇ ؟زادى كىم مېيىپ ؟
  جۈمەتاي گاس-گاچىلارمەكتىپىدە ئۆز تىرىشچانلىقىغا تايىنىپ ساۋاقداشلىرى ،ئوقۇتقۇچى ئۇستازلىرى ھەم يۇرتتىكى تۇنۇش -بىلىشلىرى ئارىسىدا خېلى كۆزگە كۈرۈنۇپ قالغان ئىدى.م.گەرچە نۇرغۇنلىغان سۇيقەست ۋە توسالغۇلارغا ئۇچرىغان بولسىمۇ ،ئۆز ئەخلاق پەزىلىتى ،ئادىمىيلىكى بىلەن ئۆز سۆيگۈسىگە ئېرىشتى،دېمەك ،ئادەم  ئۆز قىممىتىنى ياراتقاندىلا ئاندىن باشقىلارنىڭ ئېتىبارىغا ئېرىشەلەيدىكەن،مەيلى مېيىپ مەيلى بېجىرىم،مەيلى گاداي مەيلى باي؛ئەڭ ياخشى ئۆزى بۇلۇش ؛ھەقىقىي ئادەم بۇلۇپ ئۆز مەسئۇلىيىتىنى ئادا قىلغانلىق ھېسابلىنىدىكەن.؛؛؛؛؛؛؛؛؛؛؛    مۇنۇ بىر جۇملە سۆز ئىنتايىن ئاددىي لېكىن ماڭا ئىنتايىن رومانتىك تويۇلىدۇ .” ھەيھات ،ياخشى كۆرگەن ئادىمىڭ  يېنىڭدا ئولتۇرسا ،نېمىدىگەن ھۇزۇر ،بىراق تەرلەپ كېتىدىكەن كىشى.”ھەممە قېرىنداشلىرىمنىڭ مۇشۇنداق روماتنىك مۇھەببەت ھېكايىسىنى ساداقەتمەن بىر جۈپى بىلەن تەڭ روياپقا چىقىرىشىغا تىلىكداشمەن .
  ماڭا ئەڭ تەسىر قىلغىنى جۇمەتاينىڭ  ئۆزلىكىدىن ئۆگۈنۈش روھى ۋە ئۆگەنگەنلىرىدىن،ئاڭلىغان ئىلىم ھېكمەتلىرىدىن زېرىكمەي- تېرىكمەي خاتىرە قالدۇرۇش روھى بولدى.مەكتەپتىكى چاغدا ئوقۇتقۇچى سۆزلىگەن نوقتىلارنى ئەستايىدىللىق بىلەن يېزىپ مەشىق قىلاتتى ،جەمىئىيەتكە چىققاندىن كىيىن ئۇستازى لاۋتىيەن دىن ئاپىرات رېمونىت قىلىشنى ئۆگەندى .جۈمەتاينىڭ بۇ روھى ھەممىمىزنىڭ ئۆگۈنۈشىگە ئەرزىيدۇ .لېكسىيە ،مۇھىم يىغىنلارغا قاتناشقاندىمۇ قەغەز قەلەم ئالماي چىقىدىغان مەن ئۈچۈن تېخىمۇ ئەرزىيدۇ.
   جۈمەتاي بار كۈچى بىلەن تىرىشقان بولسىمۇ ،بىردىنبىر ئارزۇسى _گاس-گاچىلار مەكتىپى ئېچىش _ئەمەلگە ئاشمىدى..بىرىنچى قېتىم ئۆز يۇرتىدىكى ئەمالدارلارنىڭ جۈمەتايغا ئىشەنمەسلىكى ،مېيىپ دەپ كۆزگە ئىلماسلىقى سەۋەبىدىن ۋەيران قىلىندى.ئىككىنچى قېتىم قەشقەر گاس-گاچىلار مەكتىپىنىڭ مۇدىرى ئۆز نەپسىگە چوغ تارتىپ يول قويمىدى.گاس-گاچىلار ساۋابلىق ئىشلارنى قىلاي دېسە ئاتالمىش ساق-سالامەت كىشىلەر تەرىپىدىن نابۇت قىلىندى.ئاپتورنىڭ بۇنداق يېزىشى بىز ساقلارغا قىلنغان خىقابنامىمۇ ؟ھەقسىز مەكتەپ قۇرۇش،جەمىيەتنىڭ يۈكىنى يېنىكلىتىشتەك ئېغىر ۋەزىپىلەرنى ئۇرۇنداش ئۇلارغا قارىغاندا بىزگە بەكرەك ماس كېلىدۇ.شۇڭا جۈمەتاينىڭ ئارزۇسىنى بىزلەر ئەمەلگە ئاشۇرايلى،غايە يېرىم ئەمەللىشىشتۇر.بەلكىم بىز بۇنداق ئارزۇ-ئارمانلارنى ئويلاپمۇ باقمىغاندىمىز تايىنلىق .ئۇنداقتا،يەنە بىر قېتىم زادى كىم مېيىپ؟
  بىز مېيىپ سانىلىشنى خالىمىغان ئىكەنمىز،ئۇرۇنسىز ۋايساش،قاقشاشلاردىن قول ئۈزۈپ،ئۆز قىممىتىمىزنى يارىتىش ئۇچۇن تىرىشايلى.جۈمەتاي ،مەرزىيە ،سىلەر ھەرگىز مېيىپ ئەمەس،مېنىڭ مەڭگۈلۈك ئۆگۈنۈش ئۈلگەم.

مەزكۇر تەسىرات ماقالىسى ئىزدىنىش كىتابخانىسى 2015-يىلى كىتاب ئوقۇش يىلى مۇناسۋتى بىلەن كىتاب سوۋغا قىلىش پائالىيتىگە يوللانغان كىتاب ئوقۇش تەسىرات  ماقالىسى  بۇلۇپ،بۇ يىل يېڭىدىن نەشىر قىلىنغان،جۈمە سۈرەتچىلىكنىڭ قۇرغۇچىسى،زامانىمىزدىكى ياشلارنىڭ ئۈلگىسى جۈمەخۇننىڭ ھاياتى تەسۋىرلەنگەن "تىۋىشسىز دۇنيا"ناملىق روماندىن ئالغان تەسىرات.
