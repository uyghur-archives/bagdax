ناسا مۇكاپاتىغا ئېرىشكەن ئۇيغۇر ئالىمى رەيھان بەكتۇرغا مۇبارەك بولسۇن!




2015-يىلى 8-ئاينىڭ 3-كۈنى ئامېرىكا دۆلەتلىك ئالەم قاتناش ئىدارىسى (NASA) ئالەم تېخنىكىسى ۋەزىپە باشقارمىسى بۇ يىللىق كىچىك ئانتېننا تېخنىكا ھەمكارلىقى تۈرى بويىچە ناساغا سۇنۇلغان تەتقىقات تېما ئىلتىماسلىرىنى باھالاش نەتىجىسىنى ئېلان قىلدى. ناسانىڭ بۇ يىللىق مۇكاپاتىغا ئامېرىكا بويىچە جەمئىي 100 دىن ئارتۇق ئالىم ئىلتىماس قىلغان بولۇپ، بۇنىڭ ئىچىدە پەقەت 8 نەپەر كىشى بۇ يىللىق مۇكاپاتقا تاللانغان. بۇ سەككىز نەپەر ئالىمنىڭ بىرى - دەل ئامېرىكا يۇتا شىتاتلىق ئۇنىۋېرستېتى كومپىيۇتىر ۋە ئېلېكتىر ئېنژىنىرلىقى فاكۇلتېتىدىكى ئۇيغۇر پىروفېسسور رەيھان بەكتۇر ئىكەن. ئالىمىمىز رەيھان بەكتۇرنىڭ ئېرىشىش ناھايىتى تەس، رىقابەت كۈچلۈك بولغان بىر ساھەدە ناسانىڭ تەتقىقات مۇكاپاتىغا ئېرىشىشى كىشىنى خۇشال قىلىدۇ. بۇ مۇكاپاتنىڭ بىر قىسمى بىر نەپەر دوكتور ئوقۇغۇچىنى مەبلەغ بىلەن تەمىنلەشنىمۇ ئۆز ئىچىگە ئالىدىغان بولۇپ، دوكتور رەيھان بەكتۇر يېڭىدىن ئالماقچى بولغان بىر نەپەر ئۇيغۇر دوكتور ئوقۇغۇچىسىنىڭ كېيىنكى ئىككى يىللىق ئوقۇش پۇلى ۋە تۇرمۇش چىقىملىرىنى مۇشۇ مەبلەغ بىلەن تەمىنلىمەكچى ئىكەن. مەن دوكتور رەيھان بەكتۇرنى چىن دىلىمدىن تەبرىكلەيمەن!
تۆۋەندىكىسى NASA دىن دوكتور رەيھان بەكتۇرغا ئەۋەتىلگەن مۇكاپاتقا تاللانغانلىق ئۇقتۇرۇش:

دوكتور رەيھان بەكتۇر كېيىنكى 2 يىلدا بۇ ناسا تەرىپىدىن مەبلەغ ئاجرىتىلغان بۇ تەتقىقات تېمىسىغا ئاساسلىق يېتەكچىلىك ۋەزىپىسىنى ئۈستىگە ئالىدىكەن ۋە ناسانىڭ "Goddard ئالەم ئۇچۇش مەركىزى" بىلەن ھەمكارلىشىپ، يەر شارىغا يېقىن ۋەزىپىلەرنى ئۈستىگە ئالىدىغان ئالەم ئۇچقۇرلىرىغا ئىشلىتىدىغان يېڭى تىپتىكى ئانتېننا ياساپ چىقىدىكەن. بۇ يېڭى تەتقىقات ئەڭ يېڭى، ئەڭ ئۈنۈملۈك ۋە تەننەرخى تۆۋەن بولغان ئالەم تەكشۈرۈشتە ئىشلىتىلىدىغان ئانتېننانى بارلىققا كەلتۈرىدىكەن.

بۇ يىلنىڭ بېشىدا دوكتور رەيھان بەكتۇر يەنە ئامېرىكا دېڭىز ئارمىيىسىنىڭ بىر تەتقىقات مۇكاپاتىغا ئېرىشكەن بولۇپ، ئاساسلىقى يۇقىرى تېخنىكىلىق سېنتىزلەنگەن ئۆمۈچۈك يىپىكى ئانتېنناسى ياساش ۋەزىپىسىنى ئۈستىگە ئالغان. بۇ تەتقىقاتتا، ئۇ قاتناش ۋاستىلىرىنىڭ ئالدى ئەينىكى، كىيىم ياكى تېرىگە بىۋاستە ئانتېننانى يازىدىغان بولۇپ، بۇ خىل تەتقىقات ئارقىلىق ئەڭ يەڭگىل ۋە كىيگىلى بولىدىغان ئانتېننا ياساپ چىققىلى بولىدىكەن.

دوكتور رەيھان بەكتۇر ھازىرغىچە ئامېرىكا تەبئىي پەن فوندى قاتارلىق ئورۇنلار مەبلەغ سالغان كۆپلىگەن تەتقىقات تۈرلىرىنى مۇۋاپىققىيەتلىك تاماملىغان بولۇپ، ئۇنىڭ بۇ يىل ئارقا-ئارقىدىن ئامېرىكا دېڭىز ئارمىيىسى ۋە ئامېرىكا ئالەم قاتناش ئىدارىسىنىڭ تەتقىقات مۇكاپاتىغا ئېرىشكەنلىكى ئۇنىڭ كەسپىدە ئۈزلۈكسىز يۈكسىلىۋاتقانلىقىنى كۆرسىتىپ بېرىدۇ. ئۇنىڭ بۇندىن كېيىنكى تەتقىقاتلىرىغا تېخىمۇ روز ئۇتۇقلار يار بولسۇن!

تۇرسۇنجان نۇرمۇھەممەت بىلگە
2015-يىلى 8-ئاينىڭ 5-كۈنى


دوكتور رەيھان بەكتۇر ھەققىدە

دوكتور رەيھان بەكتۇر ئاقسۇدا تۇغۇلغان بولۇپ، ئوتتۇرا مەكتەپنى ئاقسۇدا تاماملىغاندىن كېيىن ئەلا نەتىجە بىلەن چىڭخۇا ئۇنىۋېرستېتى ئېلېكتىر ئېنژىنىرلىقى فاكۇلتىتىغا قوبۇل قىلىنغان. تولۇق كۇرسنى پۈتتۈرگەندىن كېيىن جۇڭگو پەنلەر ئاكادىمىيىىسىدە ماگىسىتىر ئاسپىرانتلىقتا ئوقۇغان ۋە ئوقۇشىنى پۈتتۈرۈپلا ئامېرىكا كلەمسون ئۇنىۋېرستېتى (Clemson University) ئېلىكتىر ئېنژىنىرلىقى كەسپىگە تولۇق ئوقۇش مۇكاپات پۇلى بىلەن دوكتورلۇققا قوبۇل قىلىنغان. ئۇ 2005-يىلى دوكتورلۇق ئوقۇشىنى تاماملاپ ئۇزۇن ئۆتمەي ئامېرىكا يۇتا شىتاتلىق ئۇنىۋېرستېتى كومپىيۇتىر ۋە ئېلېكتىر ئېنژىنىرلىقى فاكۇلتېتىنىڭ مەڭگۈلۈككە تەكلىپ قىلىش سىستىمىسى بويىچە ياردەمچى پىروفىسورلۇققا تەكلىپ قىلىنغان ۋە تەبئىي پەنلەر ساھەسىدە  ئامېرىكا تەتقىقات ئۇنىۋېرستېتلىرىدىكى بۇ خىل ئۇسۇلدا پىروفىسورلۇق خىزمىتىگە ئېرىشكەن تۇنجى ئۇيغۇر بولۇپ قالغان. ئۇ قىسقىغىنە ئون نەچچە يىللىق تەتقىقات ھاياتىدا ئامېرىكىدىكى نوپوزلۇق ئىلمىي ژورناللاردا نۇرغۇنلىغان ئىلمىي ماقالىلەرنى ئېلان قىلغان ۋە ئۆز ساھەسىدىكى ئامېرىكا ئىلمىي جەمئىيەتلىرى تەرىپىدىن تەتقىقات نەتىجىلىرى ئۈچۈن كۆپ قېتىم مۇكاپاتلانغان. ئالاھىدە دېققەت قىلىشىمىزغا ئەرزىيدىغىنى، ئۇ ئامېرىكا ئۇنىۋېرستېتلىرىدا ئۇيغۇر دوكتور ۋە ماگىستىر ئوقۇغۇچىلارنى يېتشىتۈرگەن تۇنجى ۋە بىردىنبىر ئۇيغۇر پىروفىسور بولۇپ ھېسابلىنىدۇ.  دوكتور رەيھان بەكتۇر ئامېرىكىدىكى ئايال تەتقىقاتچىلار ئىچىدە نەتىجىسى ئالاھىدە كۆزگە كۆرۈنەرلىك بولۇپ، 2009-يىلى ۋە 2011-يىلى ئىككى نۆۋەت يۇتا شىتاتىنىڭ يىللىق «تېخنىكا ساھەسىدىكى ئايال» مۇكاپاتىنىڭ ئاخىرقى 16 نامزاتى تىزىملىكىگە كىرگەن. ئۇ 2011-يىلى «يىللىق مۇنەۋۋەر پىروفىسور مۇكاپاتى» ۋە «مۇنەۋۋەر ئاسپىرانىت يېتەكچىسى مۇكاپاتى»غا ئېرىشكەن. دوكتور رەيھان بەكتۇر ئېرىشكەن كاتتا مۇكاپاتلار ئىچىدە، ئۇ 2013-يىلى ئېرىشكەن «دون دۇدلىي مائارىپ مۇكاپاتى» ئالاھىدە ئورۇندا تۇرىدۇ. ئامېرىكا تەتقىقات ئۇنىۋېرستېتلىرىدىكى مەڭگۈلۈككە تەكلىپ قىلىش سىستىمىسىنىڭ تەلىپى بويىچە ئۇ 6 يىل ياردەمچى پىروفىسورلۇق خىزمىتىنى تاماملىغاندىن كېيىن ئۇنىۋېرستېتنىڭ باھالىشىدىن ئۆتۈپ، يۇتا شىتاتلىق ئۇنىۋېرستېتىنىڭ رەسمىي «مەڭگۈلۈككە تەكلىپ قىلىنغان پىروفىسورى» بولۇپ قالغان ۋە ئورۇنباسار پىروفىسورلۇق ئۇنۋانىغا كۆتۈرۈلگەن. دوكتور رەيھان بەكتۇرنىڭ تەتقىقاتلىرى يۇقىرى تېخنكىلىق ئانتېنا ياساشقا مەركەزلەشكەن بولۇپ، ھازىر ئۇ ئۆز ساھەسىدە ئامېرىكىدىكى، شۇنداقلا دۇنيادىكى ئالدىنقى قاتاردا تۇرىدىغان مۇتەخەسىسلەرنىڭ بىرىدۇر.

دوكتور رەيھان بەكتۇرنىڭ تەرجىمھالى
تەربىيىلىنىش ئەھۋالى
1998-يىلى چىڭخۇۋا ئۇنىۋېرستېتى ئېلىكتېر ۋە ئېلىكتېر ئىلىمى فاكۇلتىتىدا باكلاۋۇرلۇق ئوقۇشىنى پۈتتۈرگەن
2000-يىلى جۇڭگو پەنلەر ئاكادىمىيىسى ئاۋاز ئىلىمى ئىنىستىتۇتىدا ماگىستىرلىق ئوقۇشىنى پۈتتۈرگەن.
2005-يىلى ئامېرىكا كىلەمسېن ئۇنىۋېرستېتى ئېلىكتېر ۋە كومپىيۇتېر ئېنژىنىرلىقى فاكۇلتىتىدا دوكتورلۇق ئۇنۋانىغا ئېرىشكەن.

خىزمەت تارىخى
2005-2006 ئامېرىكا ئىللىنوي شىتاتىدىكى موتورولا شىركىتىدە ئىنژىنىر بولۇپ ئىشلىگەن.
2006-2012 ئامېرىكا يۇتاھ شىتاتلىق ئۇنىۋېرستېتى ئېلىكتېر ۋە كومپىيۇتېر ئىنژىنىرلىقى فاكۇلتېتىدا ياردەمچى پىروفېسسورلۇق خىزمىتىنى قىلغان.
2012- يۇتاھ شىتاتلىق ئۇنىۋېرستېتى ئېلىكتېر ۋە كومپىيۇتېر ئىنژىنىرلىقى فاكۇلتېتىنىڭ مەڭگۈلۈككە تەكلىپ قىلىشىغا ئېرىشىپ ئورۇنباسار پىروفېسسورلۇققا ئۆستۈرلۈپ ئىشلەپ كەلمەكتە.

تەتقىقات تۈرلىرى (ئۇزۇن تىزىملىك بولۇپ، بۇ يەردە پەقەت يېقىنقى بىر نەچچە يىلدىكى ئاساسلىقلىرى تىزىلدى)
دۆلەتلىك مېيىلىق ۋە ئەسلىگە كەلتۈرۈش تەتقىقات ئىنىستىتۇتى: ئىجتىمائىي ئالاقە ئىقتىدارىغا ئىگە روبوتلار (ماشىنا ئادەملەر)نىڭ سىمسىز توك سىتىمىسى (2014)، ھەمكار تۈر ئىگىسى، جەمئىي سوممىسى 106،424 دوللار، دوكتور بەكتۇرغا بۆلۈنگىنى 40 مىڭ دوللار.
ئامېرىكا دۆلەتلىك تەبئىي پەن فوندى (#1128622) : كۆپ فۇنكىسيىلىك قۇياش ئېنېرگىيىلىك ئانتېنا ئوقى تەتقىقاتى (2012-2015). باش تۈر ئىگىسى، جەمئىي سوممىسى 198،963 دوللار.
ئامېرىكا دۆلەتلىك تەبئىي پەن فوندى (# 0801426): كىچىك سۈنئىي ھەمراھلارغا ئىشلىتىدىغان قۇياش ئېنېرگىيىلىك باتارىيە ئانتېناسى (2008-2011). جەمئىي سوممىسى 192،210 دوللار.
