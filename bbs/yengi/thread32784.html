ئامېرىكىنى بىرىنچى قېتىم توغرىسىغا كېسىپ ئۆتۈۋاتقان ئىككى ئۇيغۇر يىگىتى

تاشپولات روزى


مەن ئىككى ھەپتىنىڭ ئالدىدا ماسساچۇسېتس شتاتىنىڭ پايتەختى بوستوندا، تولۇق ئوتتۇرىنىڭ 3-يىللىقىنى پۈتتۈرۈپ مۇشۇ ئاينىڭ ئاخىرى تولۇق ئوتتۇرا 4-يىللىققا ئوقۇشقا كىرىدىغان كورشات سىراجىدىن ئۇكامدىن ئۇچۇر تاپشۇرۇپ ئالدىم. ئۇ ئۇچۇرىدا، بوستوندىكى يېقىن تونۇش ئاكىسى ئارافات ئابدۇۋەلى بىلەن 40 كۈنلۈك ئامېرىكىنى توغرىسىغا ماشىنا بىلەن كېسىپ ئۆتۈش سەپىرىگە چىققانلىقىنى، ھازىر سەپەرگە چىققىنىغا 20 نەچچە كۈن بولغانلىقىنى، يەنە 3 ياكى 4 كۈن ئىچىدە كالىفورنىيەنىڭ لوس ئانجېلېس شەھىرىگە يېتىپ بارىدىغانلىقىنى، مۇمكىن بولسا مەن ۋە بىزنىڭ ئائىلىمىزدىكى بارلىقى بىلەن كۆرۈشۈپ، سۆھبەتتە بولۇپ ئۆتۈپ كەتمەكچى بولغانلىقىنى ئېيتىپتۇ. مەن بۇ خەۋەرنى ئۇققاندىن كېيىن دەررۇ كورشات ئۇكامغا خەت يېزىپ، ئۇلارنى قىزغىن قارشى ئالىدىغانلىقىمىزنى، ئۆيىمىزدە ئۇلارنىڭ قانچىلىك تۇرسىمۇ بولىدىغانلىقىنى، ئۇلار ئۈچۈن قولىمىز كۆكسىمىزدە ئىكەنلىكىنى ئېيتىپ سۆز قالدۇردۇم. بىزدە يەتتە ياشلىق بالا مېھمانغا كەلسىمۇ ئالدىغا چىقىپ كۈتۈۋال دەيدىغان ئېسىل پەزىلەتلىرىمىز بار. يات دۆلەتلەردە ئۆز قېرىنداشلىرىمىز بىلەن ئۇچرىشىش بىز ئۈچۈنمۇ تېپىلماس غەنىيمەت!


بۇلتۇر قارلۇق شىركىتىگە بارغاندا تارتىلغان رەسىم. ئوڭدىن كورشات، سىراجدىن(دادىسى)، گۈلنار(ئاپىسى)، پەخىردىن، مەن، مېنىڭ ئوغلۇم فىرات.

مەن بۇ ئىككى يىگىتنى تونۇشتۇرۇۋېتەي، ھەممىڭلارغا تونۇشلۇق بولغان قارلۇق شىركىتىنىڭ قۇرغۇچىسى ئاكا-ئۇكا سىراجىدىن ۋە پەخىردىن بولۇپ، كورشات سىراجىدىن بولسا سىراجىدىن ئىنىمنىڭ چوڭ ئوغلى بولىدۇ.  ئۇ ئامېرىكىغا 14 يېرىم يېشىدا يەنى ئوتتۇرا مەكتەپنىڭ بىرىنچى يىللىقىدا ئوقۇشقا كەلگەن.  مۇشۇ ئاينىڭ ئاخىرىدىن باشلاپ ئۇ ئوتتۇرا مەكتەپنىڭ ئاخىرقى يىللىقىدا ئوقۇيدىغان بولۇپ، كېلەر يىلى ئوقۇش پۈتتۈرۈپ، ئالىي مەكتەپكە ئوقۇشقا چىقىدۇ. مەن كورشات ئۇكام بىلەن بۇلتۇر ئوغلۇم بىلەن ۋەتەنگە بارغىنىمدا ئۈرۈمچىدىكى قارلۇق شىركىتىدە كۆرۈشكەن. ئۇنىڭ ناھايىتى ئەدەبلىك، ئەقىللىق، پاراڭچى يىگىت ئىكەنلىكى بۇلتۇر قارلۇق شىركىتىنى زىيارەت قىلغىنىمدا ئۇققان ئىدىم. بۇلتۇر ئۇنىڭ بىلەن ئۇچراشقاندا ئۇنىڭ قارلۇق شىركىتىدە دەرس ئۆتۈۋاتقانلىقىنىمۇ ئاڭلاپ بۇ كىچىك ئەزىمەتكە تېخىمۇ ئاپىرىن ئوقۇغان ئىدىم. ئارافات ئابدۇۋەلى، بوستوندىكى بىر ئالىي مەكتەپتە ئوقۇپ يېقىندا ئىقتىسادتىن ماگىستىرلىق ئۇنۋانى ئالغان بولۇپ، بۇ يىل 28 ياشلارغا كىرىپتۇ. مەن بۇ ئىنىمىزنى بۇ قېتىم كۆرىشىم بىرىنچى قېتىم بولىدۇ. ئۇ ئىنىمىزمۇ پاراڭچى، چاقچاقچى، كۆڭلىگە كۆپلىگەن غايىلەرنى پۈككەن، تىرىشچان يىگىت ئىكەن.


مەن 8-ئاينىڭ 9-كۈنى يەكشەنبە ئەتىگەن كورشات ئىنىمدىن تېلېفون تاپشۇرۇپ ئالدىم، ئۇلار كالىفورنىيە سانتىياگو شەھىرىگە يېتىپ كەپتۇ، بىزنىڭ ئۆيگە كەچ سائەت 6 ئەتراپىدا كېلىدىغانلىقىنى ئۇقتۇردى. ئۇ تېلېفوندا يەنە بىزنىڭ بۈگۈن يەكشەنبە كۈنى بەك ئالدىراش بولساق دۈشەنبە كۈنى كۆرۈشسەكمۇ بولىدىغانلىقىنى دەپ تۈزۈت قىلدى. بۇ يىگىتنىڭ تېلېفوندا چوڭ كىشىلەردەك تەمكىن ۋە قائىدىلىك سۆزلەشلىرى مېنى تېخىمۇ خۇشال قىلدى. بولۇپمۇ ئامېرىكىدەك بىر دۆلەتتە كىچىكىدىن ئاتا-ئانىسىدىن ئايرىلىپ ئۇيغۇرنىڭ سالاملىشىش  ئادەتلىرىنى ئۇنتۇماي ئۇيغۇرچىنى تولۇق مۇكەممەل سۆزلىشى مېنىڭ بۇ يىگىتكە بولغان ھۆرمىتىمنى تېخىمۇ ئاشۇردى. مەن بۇ ئۇكامغا تۈزۈت قىلماسلىقىنى بىزنىڭ تەييارلىقلارنى ئاللىقاچان تەييارلاپ بولغانلىقىمىزنى، تارتىنماي ئۆيىمىزگە ئۆز ئۆيىگە كەلگەندەك كېلىۋېرىشىنى ئېيتتىم. كورشات ئىنىمنىڭ ئۇيغۇرچىنى بۇنداق راۋان سۆزلىشى ۋە سالاملىشىش ئادەتلىرىمىزنى ئۇنتۇپ قالماسلىقى ئۇنىڭ ئاتا-ئانىسىنىڭ ئۇنىڭغا قانداق تەلەپلەر بىلەن ئوقۇتۇپ چىققانلىقىدىن ئايرىپ قاراشقا بولمايدۇ. كورشاتنىڭ ئانىسى ئۇيغۇر دىيارىدا بالا تەربىيىلەشتە كۆزگە كۆرۈنگەن ئانىلاردىن بىرى. مەن بۇ ئانىنىڭ يازغان كىتابلىرى ۋە يازغان ماقالىلىرىدىن ئۇنىڭ بالىسىنى قانداق تەربىيىلەپ چىققانلىقىنى كۆز ئالدىمغا كەلتۈرەلەيمەن. ئانا بىر بالىنىڭ دەسلەپكى ئوقۇتقۇچىسى بولۇپ، ئۇنىڭ بالىغا بولغان تۆھپىسى ھەممىدىن ئۈستۈن تۇرىدۇ. بالا كىچىكىدە يېڭىدىن ئۆسۈپ يېتىلىۋاتقان بىر تۈپ دەرەخكە ئوخشايدۇ. ئەگەر بىز ئاشۇ دەرەخنىڭ ياۋايى شۇملىرىنى پۇتاپ تۇرمىساق دەرەخ ساغلام ئۆسۈپ يېتىلەلمەيدۇ. شۇڭا بالىنىڭ كەلگۈسىدە قانداق ئادەم بولۇپ چىقىشتا ئانىنىڭ رولى بەكمۇ چوڭ بولىدۇ.  ئائىلە تەربىيىسى ياخشى بولسا، بالىنىڭ قانداق بىر جەمئىيەتتە ياشىشىدىن قەتئىي نەزەر، ئۆز ئانا تىلىنى پۇختا ئىگىلەپ جەمئىيەت ئۈچۈن ۋە خەلقى ئۈچۈن كۆپ ياخشى ئىشلارنى قىلالايدۇ.


بۇ ئىككى يىگىتنىڭ بۇ قېتىم بېسىپ ئۆتىدىغان يولى تەخمىنەن 25،000 كىلومېتىر بولۇپ، جەمئىي 15 شتاتنى بېسىپ ئۆتىدۇ. ئۇلار كالىفورنىيەگە كەلگۈچە 10 شتاتنى بېسىپ ئۆتكەن. ھەر بىر چوڭ شەھەرگە كەلگەندە 2 كۈن ياكى 3 كۈن تۇرغان. ئۇلار كالىفورنىيەگە كەلگۈچە 26 كۈننى ئۆتكۈزگەن. مۇشۇ 26 كۈن ئىچىدە ئۇلار ماشىنا ئىچىدە ياكى باغچىلاردا ئۆي ياساپ ئۇخلاپ كەلگەن. بۇنداق قىلىش ئارقىلىق ئىقتىسادنى ئەڭ تۆۋەن دەرىجىدە تىزگىنلىگەن. ئىقتىسادنى تىزگىنلەشلا ئەمەس ئۆزلىرىنىڭ تەبىئەت بىلەن ئۇچراشقاندا قانداق مۇستەقىل تۇرمۇش كەچۈرۈش يوللىرىنى ئۆگەنگەن. ئۇلار يول ئۈستىدىكى داڭلىق ساياھەت رايونلىرى، تارىخى مۇزېي، داڭلىق ئالىي مەكتەپلەرگە بېرىپ ئېكسكۇرسىيىدە بولغان. ئۇلار تېكساس شتاتىغا كەلگەندە بىزنىڭ ياش ھاياتلىق ئىلمى ئالىمىمىز تۇرسۇنجان نۇرمەمەت ۋە بالىلار پسىخولوگىيە مۇتەخەسسىسى زەينۇرە سابىر بىلەن كۆرۈشۈپ چوڭقۇر پىكىردە بولغان.


كەچ سائەت 6 ئەتراپىدا بۇ ئىككى يىگىت بىزنىڭ ئۆيىمىزگە ساق-سالامەت يېتىپ كەلدى. بىز ئۇلار بىلەن ئاتا-بالا ۋە ئانا-بالا كۆرۈشكەندەك قۇچاقلىشىپ قىزغىن كۆرۈشتۇق. ئۇلار بىلەن قىزغىن سەپەر ئۈستىدىكى ھەرخىل قىززىق گەپلەر بىلەن پاراڭلاشقاچ، ئۆيدە سىنچاي ئىچتۇق. ئۇلار شۇنچە يولنى بېسىپ ئۆتسىمۇ چىرايىدىن ھاردۇق دېگەن نەرسىنى پەقەتلا كۆرگىلى بولمايتتى. كورشات ئۇكامنىڭ مەن كۆرگەن ۋاقىتتىنمۇ بويى ئېگىزلەپ، خۇش-خۇي پاراڭچى بولۇپ كېتىپتۇ. يېڭى كۆرۈشكەن ئارافات ئۇكام پاكار بويلۇق چىرايىدىن كۈلۈپلا تۇرىدىغان ئىللىق چىراي بالا ئىكەن. مەن چاي ئىچكەچ ئۇلارغا بىزنىڭ ئۆينىڭ ئەتراپىدا داڭلىق ئاشخانىلار بەكمۇ كۆپ سىلەرنى ئاشۇ ئاشخانىغا ئاچىقىپ پاراڭلاشقاچ مېھمان قىلاي دەيمەن دەپ ئۇلاردىن سورىدىم. ئۇلارمۇ بىزنىڭ ئۆيدىكى ئىللىق چىرايلارنى كۆرگەندىن كېيىن، بىزنى ئۆز قېرىنداشلىرىدەك كۆرۈپ، بولىدۇ شۇ يەرگە چىقايلى، ئاشۇ يەردە تېخىمۇ كۆپ پاراڭلىشايلى دەپ ماقۇل بولدى. مەن ئۇلارنى ئادەتتە يېرىم سائەت ياكى بىر ئىككى سائەت ساقلاپ كىرىدىغان ئامېرىكىلىقلارنىڭ داڭلىق ئاشخانىسىغا ئەكىردىم. بىز بۇ يەردە تاماق يېگەچ قىزغىن پاراڭلاشتۇق. ئۇلار سەپەر ئۈستىدىكى قىزىق ھېكايىلەرنى سۆزلەپ بەردى ھەم مېنىڭدىن ئامېرىكادا قانداق قىلىپ مۇۋەپپەقىيەت قازانغانلىقىمنى سورىدى. مەن ئامېرىكىغا كەلگەندىن بېرى بېشىمدىن ئۆتكەن ئىشلارنى قىسقا قىلىپ سۆزلەپ بەردىم. ئۇلارنىڭ قەتئىي ئۆز نىشان ۋە مەقسەتلىرى ئۈچۈن توختىماي تىرىشىشىنى، مەغلۇبىيەت ئالدىدا ئىككىلەنمەي بىلىم ئېلىشىنى داۋاملاشتۇرسا، ئاشۇنداق قىلغاندا ئۆز مەقسەتلىرىگە چوقۇم يېتىدىغانلىقىنى سۆزلىدىم. سىز ئۆز ۋەتىنىڭىزدىن ئايرىلىپ باشقا دۆلەتكە كېلىپ، مۇۋەپپەقىيەت قازىنىمەن دەيدىكەنسىز ئالدى بىلەن ئاشۇ دۆلەتنىڭ قانۇنىنى بىلىشىڭىز، ئاشۇ دۆلەتنى قۇرۇپ چىققان ئىنسانلار بىلەن يېقىن ئۆتەلەيدىغان بولىشىڭىز كېرەك. مەن ئۇلارغا بولۇپمۇ ياشلارنىڭ بۇ دۆلەتتە چوڭراق مۇۋەپپەقىيەتلەرگە ئېرىشىش ئۈچۈن مۇشۇ دۆلەتتە چوڭ بولۇپ ئۆسۈپ يېتىلگەن ئۇيغۇردىن باشقا مىللەتلەر مەسىلەن، ئاق تەنلىك، قارا تەنلىك ۋە باشقا مىللەتلەر ياشلىرى بىلەن ئىناق ئۆتىدىغان، ئۇلار بىلەن پىكىرلىشەلەيدىغان ئەھۋالنى يارىتالىشى كېرەكلىكىنى دېدىم.  يەنە 3 يۈز 60 كۈن ئۆز قەۋمدىكىلەر بىلەن بولسىڭىز يېڭى تەپەككۇرلىرىنىڭ يارىتىلمايدىغانلىقىنى شۇڭا پۇرسەت بولسا مۇشۇ دۆلەتنى قۇرۇپ چىققان مىللەتلەر بىلەن يېقىنلىشىشى كېرەكلىكىنى دېدىم. مەن بۇ ئىككى ياشقا ۋەتەندىن بىر ئۇيغۇر ياشنىڭ ئوتتۇرا مەكتەپنى پۈتتۈرۈپ بۇ دۆلەتكە كېلىپ بىر يىل ئىچىدە ئېنگلىزچىنى ۋە كەسپىنى ياخشى ئۆگىنىش بىلەن بىرگە بىر شىركەتنى مۇشۇ دۆلەتتىكى ئەڭ ئاساسلىق مىللەتلەر بىلەن قۇرۇپ چىققانلىقىنى، ھەم ئاشۇ شىركەتنىڭ باش دىرېكتورى ئىكەنلىكىنى، قىلغان ئىشلىرى شتات ۋە شىركەتلەر تەرىپىدىن مۇكاپاتلانغانلىقىنى سۆزلەپ بەردىم. بۇ ئىككى ياشنىڭ ئۆز ئارزۇلىرىنى ئاڭلاپ باقتىم ۋە ئۇلارنىڭ كەلگۈسىدە قىلماقچى بولغان ئىشلىرىنى ۋە غايىلىرىنى ئاڭلاپ ئۇلارغا ئاپىرىن ئوقۇدۇم. قىيىن ئىش يوق ئالەمدە كۆڭۈل قويغان ئادەمدە دېگەندەك، قەتئىي ئىرادىگە كېلىپ تىرىشسىڭىز ۋەتەن ئىچى ۋە سىرتىدا بولامسىز چوقۇم غەلىبە قىلالايسىز.


بىز رېستۇراندىكى تاماقلارنىمۇ يەپ پاراڭلارنىمۇ تۈگىتىپ ئۆيگە قايتىپ كەلدۇق. ئۆيدە ئوغلۇم بىزنى ساقلاپ تۇرۇپتۇ. بۇ ئىككى يىگىت ئۆيىمىزگە كەلگەندە ئوغلۇم چېنىقىش ئورنىغا كەتكەن ھەم ئەتىسى مەكتىپى باشلىنىدىغانلىقى ئۈچۈن مەكتەپكە كېرەكلىك نەرسىلەرنى ئالغىلى كەتكەن ئىدى. بۇلار ئۆيگە كىرىشى بىلەن ئوغلۇم ئۆز دوستلىرىنى تېپىۋالغاندەك قىزغىن كۆرۈشۈپ، بۇلتۇر ئۈرۈمچىدە كورشات بىلەن كۆرۈشكەن ۋاقتىدىكى ھېكايىلىرى ۋە خەت ئارقىلىق پاراڭلىشىپ تۇرغان ھېكايىلىرىنى ئېيتىشىپ پاراڭلىشىپ كەتتى. ئۆيدە ئايالىم بىز كەلگۈچە يەيدىغان مېۋە-چېۋە قاتارلىقلارنى تەييارلاپ تاق قىلىپ قويۇپتۇ. بىز كالفورنىيەنىڭ مېۋە-چېۋىلىرىنى يېگەچ ئائىلە بويىچە يەنە قىززىق پاراڭغا چۈشۈپ كەتتۇق. پاراڭمۇ خېلى ئۇزۇنغا سوزۇلدى، يېڭى كەلگەندە بۇ ئىككى يىگىتنىڭ چىرايىدىن پەقەت چارچاش ئالامەتلىرى كۆرۈلمىگەن بولسىمۇ، ئەمدى 26 كۈنلۈك يول ھاردۇقى كۆرۈلۈشكە باشلىدى. شۇڭا بىز ئۇلارنىڭ ياتىدىغان ئۆيلىرىنى ئورۇنلاشتۇرۇپ، ياخشى دەم ئېلىشىنى تىلەپ ئۇخلاپ قېلىشتۇق. ئەتىسى بۇ ئىككى يىگىت لوس ئانجېلېستىكى دۇنياغا داڭلىق ھوللىۋودنى ۋە ئۇنىڭ ئەتراپىدىكى داڭلىق ئورۇنلارغا بارماقچى بولدى. مېنىڭ ئەتىسى ئەتىگەندە دەرسىم بولغاچقا ئەتىگىنى ئۇلار بىلەن كۆرۈشەلمەي مەكتەپكە كېتىپ قالدىم. مەكتەپتىكى دەرسىمنى ئۆتىۋېتىپ كەلسەم ئۇلار ئەتىگەنلىك تاماقلىرىنى يەپ ئۆيدىن يولغا چىقىپتۇ. ئۇلار مېنى كۆرۈپ تۈنۈگۈن بەك ياخشى ئۇخلىدۇق 26 كۈندىن بۇيان بۇنداق دەم ئېلىپ باقماپتىكەنمىز دەپ ئۆز تەسىراتلىرىنى سۆزلەپ بەردى. بىز ئۇلارنى يولدا ناھايىتى دىققەت قىلىپ ماشىنا ھەيدىشىنى تەكىتلەپ يولغا سېلىپ قويدۇق.


ئۇلار كەچكە ئۈلگۈرۈپ ئۆيگە ساق-سالامەت قايتىپ كەلدى. ئايالىم ئۇلارغا لەڭمەن تەييارلاپ ساقلاپ ئولتۇرۇپتىكەن. بۇ ئىككى يىگىتنىڭ ھەر ئىككىلىسى لەڭمەننى بەك ياخشى كۆرىدىكەن. قىزىم ئۇلارغا لەڭمەننى سوزۇپ سېلىپ بەردى. تۈنۈگۈن قىزىم شىركىتىنىڭ ئىشلىرى بىلەن كەچ كەلگەچكە ئۇلار بىلەن تۈنۈگۈن كۆرۈشەلمەي ئۇخلاپ قالغان ئىدى. ئۇلار ئەتىگىنى ناشتىلىق قىلغاچ شىركەت كۆرۈش ۋە مۇۋەپپەقىيەت قازىنىش توغرىلىق تازا پاراڭلىشىپتۇ. كەچتە ئۇلار لەڭمەن يېگەچ قىززىق پاراڭغا چۈشۈپ كەتتى. مەن ئەتە ئۆتىدىغان دەرسلىرىمنى تەييارلىغاچ بۇ ياشلارنىڭ يېنىدىن ئايرىلىپ ئۆز ئىشخانامغا كىرىپ كىتاب كۆردۈم. ئەتىسى ئەتىگەن دەرسىم سائەت 10 بولغاچقا ئۇلار بىلەن ئەتىگىنى پاراڭلىشىپ، ناشتا قىلىپ مەكتەپكە كېتىپ قالدىم. بۇ ئىككى يىگىت دۇنياغا داڭلىق سانتامونىكا دېگەن دېڭىز بويلىرىغا بېرىپ كېلىش ئۈچۈن كەتتى.


كەچتە ئايالىم ئۇلارغا ئوخشىغان پولو ئېتىپ ساقلاپ ئولتۇردى. ئۇلار كېلىشى بىلەن پولونى تارتتۇق، پولونى يېگەچ بۇ ئىككى يىگىت بىلەن يەنە پاراڭلىشىپ كەتتۇق. ئۇ ئىككىلىسى «تاشپولاتكا، ھەدىمىز تاماققا بەك ئۇستا ئىكەن» دەپ ئايالىمنى ماختاپ كەتتى. مەن ئۇلارغا توي قىلىشتىن بۇرۇن ئايالىمغا «تاماقنى ياخشى ئېتەلمىساڭ توي قىلمايمەن» دەپ شەرت قويغان شۇڭا تاماق ئېتىشنى ياخشى ئۆگەنگەن دەپ چاقچاق قىلدىم. بۇ ئىككى يىگىت ھەم ئايالىم پاراققىدە كۈلۈشۈپ كەتتى. ھايات دېگەندە ئانچە مۇنچە چاقچاق، قىززىق گەپ بولمىسىمۇ ھاياتنىڭ پەيزى ۋە ئائىلىنىڭ مەنىسى بولمايدۇ-دە! مەن چاقچاق قىلغاچ بۇ ئىككى يىگىتتىن نېمە ئۈچۈن بۇ 40 كۈنلۈك ئۇزۇن سەپەرگە چىقىشنى ئويلاشتىڭلار دەپ سورىدىم. ئارافات ئۇكام مۇنداق دەيدۇ، «مەن 28 ياشقا كىردىم ئويلاپ باقسام تېخىچە ئۆزۈمگە زادى نېمە مۇۋاپىق بولىدىكەن قانداق يولنى باسسام بولغۇدەك دەپ ئويلاپ يۈرەتتىم. مانا مۇشۇ 26 كۈن ئىچىدە مەن كۆپ نەرسىلەرنى ئۆگەندىم، مۇستەقىل تۇرمۇش كەچۈرۈش يوللىرىنى ئۆگەندىم. ئۇستازلاردىن كۆپ بىلىم ئالدىم. ئادەملەرنىڭ سىرتقى قىياپىتىگە قاراپ مۇئامىلە قىلسا بولمايدىغانلىقىنى ئۆگەندىم. مەن ئۆزۈمگە قانداقتۇر بىر نەرسىنىڭ مۇۋاپىق ئىكەنلىكىنى تېپىپ چىققاندەك بولدۇم. بۇ قېتىمقى ئۇزۇن يوللۇق سەپەر مېنىڭ ھاياتىمغا چوڭ تەسىر قوزغايدۇ. مەن ئاللىقاچان بىر نىشانى بار كىشىگە ئايلانغاندەك بولدۇم دېدى». كورشات ئۇكام مۇنداق دېدى «مەن يول يۈرۈش ئارقىلىق ئامېرىكانىڭ زادى قانداق بىر دۆلەت ئىكەنلىكىنى بىلىش بىلەن بىرگە كىچىكىمدىن باشلاپلا مۇستەقىل تۇرمۇش كەچۈرۈش، ئامېرىكىدىكى ئالىي مەكتەپلەرگە ئۆز پۇتۇم بىلەن مېڭىپ كىرىپ ئۇلارنى چۈشىنىش بولدى. مەن بۇ قېتىم كۆرگەن مەكتەپلەر ئارقىلىق كەلگۈسىدە قانداق بىر مەكتەپكە كىرىپ ئوقۇسام بولىدىغانلىقىنى بىلگەندەك بولدۇم. ئۇستازلاردىن كۆپ ئېسىل گەپلەرنى ۋە رىغبەتلەندۈرۈش سۆزلىرىنى ئاڭلىدىم، بۇلار مېنىڭ بۇندىن كېيىنكى ھاياتىمغا چوڭ تەسىر قوزغايدۇ. نىشان ۋە مەقسەت بولمىسا كۆزلىگەن نىشانغا يېتىپ بارغىلى بولمايدىغانلىقىنى تېخىمۇ بىلىپ يەتتىم».


بۇ ئىككى يىگىت ياش ئالىمىمىز تۇرسۇنجان نۇرمەمەت بىلەن بىرگە.

مەن بۇ ئىككى ياشنىڭ 26 كۈنلۈك ئۇزۇن سەپەردە ئۆزى يەتمەكچى بولغان مەقسەتلىرىگە يەتكەنلىكىنى ئۇقۇپ بەك خۇش بولدۇم.  ئۇلارنىڭ قالغان 14 كۈن ئىچىدە تېخىمۇ كۆپ بىلىملەرگە ئېرىشىشىنى چىن كۆڭلۈمدىن ئۈمىد قىلدىم. مەن بۇ ئىككى ياشنىڭ 26 كۈن ياتاقلاردا ياتماي ئۆزلىرى تەييارلىغان ئۆيلەردە ۋە ماشىنا ئىچىلىرىدە ئۇخلاپ بۇ سەپەرنى ئاساسىي جەھەتتىن تاماملىغانلىقىدىن تېخىمۇ تەسىرلەندىم. مانا ئۇلارنىڭ مۇشۇنداق قىلالىغانلىقى ئۇلاردا مەقسەت ۋە نىشانىنىڭ بولغانلىقىنى كۆرسىتىدۇ. يېڭى يەرلەرنى كۆرگەنلىكنىڭ ئۆزى ئىنساننىڭ تەپەككۇرىنى ئاشۇرۇپ، يېڭى مۇۋەپپەقىيەتلەرنى قولغا كەلتۈرۈشكە يېڭى يوللارنى ئېچىشىدا گەپ يوق. بىز 3 كۈن ئىچىدە بۇ ئىككى يىگىت بىلەن بىر ئائىلىنىڭ ئادەملىرىدەك بولۇپ كەتتۇق. بۇنداق بولالىشى بۇ ئىككى يىگىتنىڭ تەنلىرىدە قانداقتۇر بىر ئېسىل نەرسىلەرنىڭ بارلىقىنى سېزىپ ئالغىلى بولىدۇ. بىز ئائىلىمىز بويىچە بۇ ئىككى يىگىتنىڭ ئاخىرقى نىشانىغا ساق-سالامەت يېتىپ بېرىشىنى چىن كۆڭلىمىزدىن ئۈمىد قىلىمىز. ئۇكىلىرىم سىلەرگە ئاق يول بولسۇن! بىز سىزلەرنىڭ ھەر ۋاقىت بۇ ياققا كېلىپ ئويناپ كېتىشىڭلارنى ئۈمىد قىلىمىز! بۇ ئىككى ياش ئەتىسى ئەتىگەن يەنى 8-ئاينىڭ 12-كۈنى ئورۇنلىرىدىن تۇرۇپ ئەتىگەنلىك ناشتىلىرىنى ئېچىپ بولۇپ، بىز بىلەن خوشلىشىپ كېيىنكى نىشانى سان فرانسىسكوغا قاراپ يول ئالدى. خەيرى-خوش!
