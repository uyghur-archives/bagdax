بۇنداق ئەھۋالنى مەنمۇ كۆرگەن . مەنمۇ ئاپتورغا ئوخشاش ئۇرۇمچىگە تەربىيىلىنىشكە چىققان . مەنمۇ شۇنداق ئەھۋالنى بايقىغان . كۇرسقا قاتناشقان ھەربىر كۇرسانتقا بىردىن سومكا ، خاتىرە دەپتەر ، قەلەم تارقىتىپ بەردى . لېكىن شۇ خاتىرە دەپتەرگە خاتىرە يازىدىغان ، قەلەمنى ئىشقا سالىدىغاندىن قانچىسى چىقتى ؟ ئۆزەمنى بۇ يەردە ماختىغۇم كېلىپ ئەمەس ، بەلكى گېزى كېلىپ قالغانلىغى ئۈچۈن دىيىشكە توغرا كىلىپ قالدى . مەن بىر ئادەم خاتىرە يازدىم . شۇ تارقىتىپ بەرگەن خاتىرە دەپتەرنى تولدۇرۇپ بولۇپ، ئۆزەم ئويدىن ئېلىپ چىققان ، خاتىرە يېزىشنى نىشان قىلغان دەپتەرگىمۇ داۋاملىق خاتىرە يازدىم . باشقىلارنىڭ دەپتەرلىرى ئۇستەلنىڭ ئاستىدا  ياكى ياتاقلىرىدا قالدى . مېنىڭ داۋاملىق خاتىرە يېزىۋاتقىنىمنى كۆرگەن ، كۇندىلىك ئوتۇلگەن دەرسلەرنى خاتىرىلەپ يېزىشقا مەسئۇل بولغان ئۆگىنىش باشلىقىمىز بۇنى كۆرۇپ ، :ھەي بۇ خاتىرە دەپتەرنى ئەسلى سىز يازسىڭىز بوپتىكەن -دەيدۇ ، بەلكىم يېزىش ئېغىر كەلگەن ئوخشايدۇ . بىزمۇ ساۋاقداشلار بىلەن بىللە رىستۇرانلارغا بېرىپ بىللە غىزالىنىپ، كوڭۇل ئېچىپ كەلدۇق . لېكىن ئىشلار ئىنتايىن نورمال ، ئەخلاق -پەزىلەتتىن چەتنىگەن ئىشلار ، ھاياسىزلىقلار قەتئى بولمىدى . لېكىن مەن قاتناشقان ساۋاقداشلار ئولتۇرىشى بىر -ئىككى قېتىملىق بىلەن چەكلەندى ، خالاس. ئالدىنقى يىلى قاتناشقان  بۇ بىلىم ئاشۇرۇشتا مەن مەن سىنىپتىكى كۇرسانتلارنىڭ ئىچىدىكى يېشى ئەڭ چوڭلار قاتارىدىن ئورۇن ئالغانلارنىڭ بىرى ئىدىم . لېكىن ساۋاقداشلىرىم مېنىڭ يېشىمنى خاتا پەرەز قىلىۋاپتۇ ، ھەتتە ئۇلار مېنىڭ ھەقىقى يېشىمنى ئاڭلاپ ھەيران قېلىشتى ، مېنىڭ خەنزۇ تىلى سەۋىيەمنى كۆرۇپ سىز خەنزۇ تىلى  كەسپىدە ئوقۇغانمۇ دەپ سوراپ قالدى . بۇ ئۇچ ئايلىق ئۆگىنىش جەريانىدا مەن ماتىريال توپلاشقا ئالاھىدە ئەھمىيەت بەردىم . شۇ مۇناسىۋەت بىلەن مېنىڭ يېنىمدىكى بارلىق بارماق دېسكىلىرىم توشۇپ كېتىپ ، ئاخىرىدا ئۆز يېنىمدىن 500 يۇەن پۇل چىقىرىپ، قاتتىق دېسكا سېتىۋالدىم . ھېلى ھەم ئۇرۇمچىدە تەربىيىلىنىشكە چىققان چاغدا يىغىۋالغان ماتىرياللاردىن پايدىلىنىپ  دەرسخانىدا دەرس ئوتۇۋاتىمەن . شۇ چاغدا پۇرسەتنى غەنىمەت بىلىپ يىغىۋالغىنىمغا بەك خۈش بولىمەن . لېكىن شۇنچىلىك تىرىشقانلارنى رەھبەرلىك قانچىلىك قوللاپ ، مۇئەييەنلەشتۇرۇشنى ، نەزىرىگە ئىلىپ قويۇشنى ئەسلا بىلمەيدۇ . ئەمىلىيەتتە ئۇلار كوڭلىدە كىمنىڭ قانداقلىغىنى ، قانداقراق ئىشلەيدىغانلىغىنى بىلىدۇ . مۇرەككەپ مۇناسىۋەت چەمبىرىكىدە قوللىغۇچى ئىزدەش ، ئورنىنى مۇستەھكەملەش سەۋەپلىرىدىن جورىسى باشلىقلاردىن ، ئاز ئىشلەپ ،كوپ كۆرسۇتۇشكە ئۇرۇنىدىغانلارنىڭ كونتىروللىغىدىن قۇتۇلالماي قونچاق بولۇپ يۇرۇيدۇ. خالاس. لېكىن بىزدە مۇستەھكەم ئېتىقاد ، قەتئى ئىرادە بولغانلىغى ئۈچۈن ئۇلارنىڭ مۇئەييەنلەشتۇرۇشلىرىگە ئېرىشەلمىسەكمۇ ، سىنىپتىكى شۇ سەبىلىرىمىزنىڭ ۋاقتىنى بىباھا بىلىم بىلەن تولدۇرۇش ئۈچۈن ھارماي ،زېرىكمەي داۋاملىق تىرىشىۋاتىمەن . مەن پىشقەدەم بولۇپ قالدىم ، ئۆگەنمىسەممۇ بولىدۇغۇ دېگەننى ئويلاپ باقمىدىم . شۇ تىرىشچانلىقلىرىم سەۋەپلىك يېڭى باش مەكتەپ مۇدىرى مەكتىۋىمىزگە كەلگەندىن كېيىن تۇزىگەن باھالاش ئولچەملىرى بويىچە ئەڭ يۇقۇرى ئۇنۋاننى ئېلىش سالاھىتىگە ئىگە بولدۇم . بۇمۇ مەن ئۈچۈن زور روھى يولەك بولدى، گەرچە ئوقۇتۇش ئىشلىرىدا مەنپەئەتنى كۆزلىمىگەن بولساممۇ ،ئۆزەمنىڭ ئەمگەكلىرىنىڭ جۇغلانمىسى نەتىجىسىدە ئۆز كوڭلۇمدە ئۆزەمنى مۇئەييەنلەشتۇرۋالدىم .
