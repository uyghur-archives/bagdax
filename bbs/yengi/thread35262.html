     سۆيۇملۇك خاتىرەم . مەن سەن بىلەن تونۇشقىلى نەچچە يىللاربولدى . شۇنچە يىللاردىن بىرى سەن ماڭا سىرداش دوسىت ، ھەمرا بولۇپ كەلدىڭ . قانچىلىغان ئويقۇسىز كىچىلەرنى سەن بىلەن مۇڭدىشىپ - سىردىشىپ ، قەلىب  پىنھاندا يۇشۇرۇن ساقلىغان سىرلىرىمنى خوشاللىق - خاپىلىق ۋە دەرىتلىرىمنى رەببىمگە تۆكۇپ سىنىڭ ئاپئاق بەتلىرىڭنى رەڭ بىلەن بوياپ ئۆتكەزگىنىمنى ساناپ بىرەلمەيمەن . قانچە قىتىم مىنىڭ تارام - تارام كۆزياشلىرىم ، سىنىڭ ئاپئاق مەڭزىڭگە تامچىلىغاندا سەن مىنىڭ قەلبىمدىكى ، دەرىت ئەلىمىمنى ھېىس قىلدىڭمۇ ! مىنىڭ غۇبارسىز قەلبىمنى چۇشەنگەنسەنمۇ ؟
    سىردىشىم ، دوستۇم سۆيۇملۇك خاتىرەم . مەن بۇگۇن يەنە سىنى قولۇمغا ئالدىم . سەن بىلەن سىردىشىش ۋە دەتلىرىمنى ساڭا توكۇش ئۇچۇن ئاپئاق بەتلىرىڭنى قەلېمىمنىڭ رىڭى بىلەن بوياپ ئاق بەتلىرىڭ ئوستىدە قەلەم تەۋرەتتىم . بۇ نىمە ئۇچۇن ؟ چۇنكى مىنىڭ ئەتىراپىمدىكى ئۇرۇق تۇققان دوسلىرىمنىڭ ھەممىسىنىڭ دەردى ئۆزگە يەتكۇچە بار ! مەن قەلېمىم ئارقىلىق سىنىڭ ئاق بەتلىرىڭگە ، قەلىىب سىرىمنى يىزىپ قەلبىمنىڭ ئۇنسىز تىۋىشلىرىنى ساڭا سۆزلەپ . رەببىمدىن ئاسانلىق تىلەپ سەن بىلەن سىردىشىپ قەلبىمنى قەلىېمىم ئارقىلىق سىنىڭ بەتلىرىڭدە قالدۇردۇم . ھەم شۇ ئارقىلىق سەن بىلەن سىردىشىپ ، مۇڭ دۇشۇپ ئۆز-ئۆزۇمگە تەسەللى بىرىپ ، كىشىلىك ھاياتقا ئۆمۇتلىك قاراپ ئۆز ئارزۇ- ئسـتەكلىرىم ئۇستىدە ئىزدىنىپ داۋاملىق جاسارەت بىلەن  ئالغا ئىنتىلىش ئويى بىلەن غەيرەتكە كىلىپ كەلگۇسىۇمگە ئۆمۇت ئىچىدە تەلپۇنۇب كۆڭلۇم ئارام تىپىپ رەببىمگە شۇكرى ئىتىپ  يەڭگىللەپ قالىمەن . نەچچە يىللاردىن بىرى ، سەنلام مىنىڭ قەلىم سىرلىرىمنى ئۇن تۇن سىز ئاڭلاپ . ماڭا تەسەللى بولۇپ مەن بىلەن يىقىن سىرداش بولۇپ كەلدىڭ .
     سۆيۇملۇك خاتىرەم . سەن مەن ئۇچۇن قەدىرلىك ھەم سۆيۇملۇك دوستۇم . رەببىم  نىمىشقا سىنىڭ سىناقلىرىڭ دائىم مىنىڭ ئوستۇمدە بولىدۇ ! سىنىڭ سىناقلىرىڭدا مىنىڭ تەغدىر - قىسمەتلرىم سەۋەبىمدىن ئائىلەمگە ئاتا-ئانامغا كۆڭۇل ئاغىرقى كىلىدۇ ! ، نىمىشقا ؟ بۇ ، ئاتا-ئانامغا ئائىلىمىزگە كەلگەن كۆڭۇلسزلىكلەرگە مىنىڭ تەغدىر قىسمىتىم  سەۋەپچى بولۇپ قالىمەن ! ھە چۇشەندىم ھايات ئىنسان ئۇچۇن ئىنتاھام مەيدانى ئىكەن . سەن مىنىڭ ھاياتىمغا سۇئال ؟ تاشلاپ مەن دىن جاۋاپ كۆتۇپسەن . ئەمدى مىنى مۇشكۇل سىناقلىرىڭ بىلەن سىنىمىغىن چۇنكى مەن ئىنسان ئازغۇچى سىنىڭ سىناقلىرىڭغا بەرداشلىق بىرەلمەي ئىزىپ قىلىشىم مۇمكىن .
       ماڭا توغىرا يولنى كۆرسەت . مىنى شۇ توغىرا يولنىڭ ئۇستىدە قىل . ماڭا ئۆزۇڭ ياردەم قىل . مەن سەندىنلام ياردەم تىلەيمەن . ئاتا -ئانامنى ئويلىسام ئۇلارغا يۇز كىلەلمىگەنلىگىم ئۇچۇن قاتتىق ئازاپلىنىمەن ، شۇڭا قەلبىم دائىم ئازاپ ئىچىدە ، ئۇنسىز ئىغىر تىنىق ئالىمەن . قەدىرلىك ئاتا -ئانام مەنسىلەرگە مىسكىن قىزىڭلارنى كەچۇرۇڭلار!!! دەپ سىلەردىن ئۆمۇت بىلەن كەچۇرۇم سورايمەن . مىنى كەچۇرۇڭلار !!! قەدىردانلىرىم . سۆيۇملۇك خاتىرەم سەن مىنىڭ قەلىبسىرلىرىمنىڭ بىردىن بىرگۇۋاچىسى ، مىنىڭ ھاياتىمدىكى ئارزۇ ئارمانلىرىمنىڭ شائىدى . سۆيگۇ ھاياتىمدىكى ئاچچىق ھەم تاتلىق ئۇنتۇلغۇسىز ، سۆيگۇ ئەسلېىمەمنىڭ مۇھەببەت قىسىسى . مىنىڭ كەچمىشلىرىم . سىنىڭ ئاپئاق بەتلىرىڭدە ئىز قالدۇرغان . مىنىڭ قەلبىمدە پىنھان ساقلىغان سىرلىرىم ساڭا سىرئەمەس . سەن مىنىڭ سىردىشىم ، بۇگۇن مەن ساڭا ئۆزۇمنىڭ خۇشال بولغۇنۇم ياكى قايغۇرغىنىمنى ئىنىق ئېىتالمايمەن .  چۇنكى ئۆزۇممۇ مەن خۇشال بولدۇممۇ ؟ ياكى قايغۇردۇممۇ ! . ئۆزۇمگە ئۆزۇم ئىنىق جاۋاپ بىرەلمەيمەن . چۇنكى مەن بۇگۇن نىكاھدىن ئاجراشتىم ، ھاياتىمغا بەخىت تىلىگەن ماڭا دۇئالىرىدا رازىلىق بەرگەن ئاتا-ئانامنى غەمكىن قىلدىم . مىنى خارلىغان ئاياللىق ئىززەت ھۆرمۇتۇمنى تۆككەن ، قەدىرىمنى قىلمىغان . ئائىلىسىگە ئايالىغا مەسئۇل بولمىغان ، نىكاھقا خىيانەت قىلغان . مەسئۇلىيەتسىز . شەرىپسىز ئەرگە مەڭگۇ خوش دىدىم . ئۇ نىكاھدىن كىيىن ئاتا- ئانامنى ھۆرمەت قىلمىغان . ئۆز ئاتا - ئانىسى ئورنىدا كۆرمىگەن . مەن ئۇچۇن قەدىرلىك بولغان قەدىردان ئاتا-ئانامنى ھاقارەتلىگەن . ئۇ ، شەرەبسىز ئىنساندىن ھايات سەپىرىمدە مەڭگۇلۇك ئايرىلدىم ، مەن سۆيگەن ، مىھرىمنى بەرگەن ئۇ ئىنساندىن مىنىڭ سۆيگۇمنىڭ - ساداقىتىمنىڭ قەدىرىنى بىلمىدى .
    نىكاھىمىزغا خىيانەت قىلىپ  يەر بىلەن يەكسان قىلدى . سەبىر قاچامنى توشقۇزدى . ئايرىلدىم .  شۇڭا ئۆزۇمنىڭ خۇشال بولۇشۇمنى ياكى قايغۇرشۇمنى بىلمەيمەن . قەلبىم ئازاپ بىلەن تولدى ، بىركۇنلەردە ئۆلۇپ كىتىمەن ، قەلبىمدىكى سۆيگۇمنى ساداقىتىمنى  بىللە ئىلىپ كىتىمەن . ئارزۇ ئارمانلىرىم مەن بىلەن بىللە كىتىدۇ ، بۇنداق سۆزنى دىيىش گەرچە يارام سىزلىق بولسىمۇ . بۇ ، سۆزلەرنى ساڭا دىيىشكە مەجبۇرمەن . خاتىرەم سەن ، مەندىن سەن يارامسىزمۇ ؟ دەپ سوراۋاتام سەن ! ياق مەن يارامسىز ئەمەس . ئۆمۇت ئۇچقۇنلىرى كۆزۇمدە چاقناپ تۇرىدۇ ،  يۇرۇگۇم ئۆمۇت ، ئىرادە ، جاسارەت ، قەيسەرلىك سۆيگۇ ساداقەت ساڭغا مەڭگۇ ھەمرا دەپ پىچىرلا ۋاتىدۇ . مەن بۇنى سەزدىم .  شۇڭا سىنىڭ سۇئالىڭغا ، ياق دەپ كەسكىن جاۋاپ بەردىم . مەن بۇگۇن ھازىردىن باشلاپ ئۆزھاياتىمنى  ھەممىنى قايتىدىن باشلايمەن . مەن ئۇچۇن قايتىدىن باشلاش ھەقىقەتەن قىيىن تاللاش بولغان بولسىمۇ ، بىراق مەن يەنلام  ئۆمۇت بىلەن ھاياتىمنى ، تۇرمۇش يولۇمنى قايتىدىن باشلاشنى تاللىدىم ، ئەسلىمەلىرىمنى ساڭا ھەدىيە قىلدىم . ئۇنى سەن مەڭگۇ سىر قىلىپ ساقلاپ قوي . مەن يىڭى ھاياتقا يىڭى تۇرمۇشقا مۇھتاج ! سەن ھېس قىلمىدىڭمۇ ؟ ئادەملەرئۆزگۇر ۋاتىدۇ ، مەنمۇ ئۆزگۇرۇشكە مۇھتاج !
      مەن ئۆزگۇرشۇم يىڭى ھاياتقا ئۆمۇت بىلەن يۇزلۇنۇشۇم كىرەك . كىملىكىمنى تۇنۇپ ،  بىلىشىم ۋە كىملىكىم بىلەن ياشىشىم كىرەك . ئۆزقەدىر قىممىتىمنى ، ئۆزۇم - ئۆزۇم دىن باشلاپ قىلىپ باشقىلارغا ھىېس قىلىدۇرشىم كىرەك . شۇڭا مەن ئوقۇشقا ئۇگۇنۇشكە بىلىم ئىلىشقا مۇھتاج ! ئوقۇ دىگەن خىتاب بىلەن ئوقۇش ئارقىلىق رەببىمنى ۋە ئۆزۇمنى بىلىپ كىملىگىمنى تونۇيمەن . ئوقۇش ئارقىلىق تەپەككۇر قىلىپ بىلىم ئالسام يىڭى ھاياتىمدا ، قايتىدىن باشلاپ قەدەم باسقان تۇرمۇشۇمدا ، ئىلىم ، ئىلىش ئۇگۇنۇش بىلەن  بەخىت ئىچىدە ياشايمەن . ھەم  ، بىلىم ئارقىلىق ئىنسانى قەدىر قىممىتىمى ساقلاپ رەببىمنىڭ ئىبادىتىدە ، بىشىمنى كۆتۇرۇپ ، يۇزۇم - يۇرۇق ياشىيالايمەن . بىلىم ئارقىلىق ئائىلەمگە پەرزەتلىرىمگە ئانىلىق مىھرىمنى بىرىپ ، پەرزەنىتلىرىمنى ياخشى تەربىيەلەپ ، ئائىلەمنى پەرۋىش قىلالايمەن . ھەم شۇ ئارقىلىق جەمىيەتكە ، ئىنسانىيەتكە تۆھبە قوشالايمەن سۆيۇملۇك خاتىرەم بۇگۇن ساڭا مۇشۇنچىلىك سىرلىرىمنى ئىھتاي . سەن بۇنى مەن ئۇچۇن سىر قىلىپ ساقلاپ بەر باشقىلارغا ئىتىپ قويما . سىرلىرىمنى باشقىلاردىن سىرتۇت . چۇنكى مىنىڭ ھەسەتخور دۇشمۇنۇم دوسلىرىمنىڭ ئىچىدە بار . ئىنساننىڭ دۇشمەننىڭ ئەڭ يامىنى شۇ ئىنساننىڭ قەلىېپ سىرلىرىنى بىلگەنلىرى . قەلىپ سىرلىرىمنى سىر تۇت .  مىنىڭ سىرداش دوستۇم .سۆيۇملۇك خاتىرەم .  مىسكىن . كۇندىلىك خاتىرەمدىن .

  سىردىشىم ، دوستۇم سۆيۇملۇك خاتىرەم . مەن بۇگۇن يەنە سىنى قولۇمغا ئالدىم . سەن بىلەن سىردىشىش ۋە دەتلىرىمنى ساڭا توكۇش ئۇچۇن ئاپئاق بەتلىرىڭنى قەلېمىمنىڭ رىڭى بىلەن بوياپ ئاق بەتلىرىڭ ئوستىدە قەلەم تەۋرەتتىم . بۇ نىمە ئۇچۇن ؟ چۇنكى مىنىڭ ئەتىراپىمدىكى ئۇرۇق تۇققان دوسلىرىمنىڭ ھەممىسىنىڭ دەردى ئۆزگە يەتكۇچە بار ! مەن قەلېمىم ئارقىلىق سىنىڭ ئاق بەتلىرىڭگە ، قەلىىب سىرىمنى يىزىپ قەلبىمنىڭ ئۇنسىز تىۋىشلىرىنى ساڭا سۆزلەپ . رەببىمدىن ئاسانلىق تىلەپ سەن بىلەن سىردىشىپ قەلبىمنى قەلىېمىم ئارقىلىق سىنىڭ بەتلىرىڭدە قالدۇردۇم . ھەم شۇ ئارقىلىق سەن بىلەن سىردىشىپ ، مۇڭ دۇشۇپ ئۆز-ئۆزۇمگە تەسەللى بىرىپ ، كىشىلىك ھاياتقا ئۆمۇتلىك قاراپ ئۆز ئارزۇ- ئسـتەكلىرىم ئۇستىدە ئىزدىنىپ داۋاملىق جاسارەت بىلەن  ئالغا ئىنتىلىش ئويى بىلەن غەيرەتكە كىلىپ كەلگۇسىۇمگە ئۆمۇت ئىچىدە تەلپۇنۇب كۆڭلۇم ئارام تىپىپ رەببىمگە شۇكرى ئىتىپ  يەڭگىللەپ قالىمەن . نەچچە يىللاردىن بىرى ، سەنلام مىنىڭ قەلىم سىرلىرىمنى ئۇن تۇن سىز ئاڭلاپ . ماڭا تەسەللى بولۇپ مەن بىلەن يىقىن سىرداش بولۇپ كەلدىڭ .
رەخمەت : ياخشى يىزىلغان ئەسەركەن !
