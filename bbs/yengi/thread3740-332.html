ئۇنۋان دەرىجىسى يوقۇرلىمىسا ئوقۇتقۇچىلارمائاشى نىڭ ئۈسىشى مۇمكىن ئەمەس ،چوشقا مۇدىر باغۋەننىڭ ئىززەت ھۈرمىتىنى دەپسەندە قىلىپلا قالماي ، ئۇنىڭ ئىقتىسادىي كىرىمىگىمۇ زىيان سالدى دىگەن ،
مېنىڭ ئاچاممۇ ئوقۇتقۇچى ، ئاچامنىڭ ئۇنۋان ئېلىش ئىشىنى سۈزلەپ بەرگۈم كەلدى ،
بۇلتۇر ئاچامنىڭ ئۇنۋان ئېلىش  مەسىلىسىدە ئاچام مەكتەپ مۇدىرىنىڭ تۇققىنى بىلەن رىقابەتلىشىپ قالغان ( ئالىي دەرىجىلىك ئۇنۋان )         ،   مەكتەپ مۇدىرى ھوقۇقىدىن پايدىنىلىنىپ ئۇنۋاننى ئۈزىنىڭ تۇققىنىغا بەرمەكچى بولدى ، ئىككەيلەننىڭ ئۇنۋان ئېىلش شەرتلىرىنى سېلىشتۇرغاندا ئاچامنىڭ شەرتى   ئۇنۋان   ئېلىشقا تولۇق توشىدۇ ، مۇدىرنىڭ تۇققىنى بىر نەچچە خىل شەرتنى ھازىرلىمىغان ئىدى     ئەمما مەكتەپ مۇدىرى جاھىللىق بىلەن تۇققىنىىنىڭ ماتىرياللىرىنى مائارىپ ئىدارىسىگە يوللىدى ، ئاچام ناھەقچىلىققا چىدىماي مۇدىر بىلەن دىيىشىپ ئۇنى نىيىتىدىن ياندۇرالمىدى ،  ئاندىن مائارىپ ئىدارىسىغا بېرىپ باشلىقلارغا ئەرز قىلغاندىن كىيىن ئۇلارمۇ ئانچە بەك پەرۋا قىلمىدى ، مۇدىر باشلىقلارنىمۇ ئوڭشاپ بولغان ئىكەن ،  ئاچام ئۇنۋانغا ئېرىشىش ئۈچۈن باشلىقلار بىلەنمۇ قاتتىق دىيىشىپ ئۇلاردىىن ئادىل ، ئوچۇق ئاشكارە ئىش بېجىرىشنى تەلەپ قىلغان ، باشلىقلار ئاچامنىڭ پوزىتسىيەسىگە قاراپ <<ئەمىسە بىز قايتىدىن تەكشۈرۈپ كۈرەيلى ، سىز ھازىرچە قايتتىپ تۇرۇڭ >> دەپ يولغا سېلىپ قويغان ، ئاچام 10نەچچە كۈن ساقلاپ يەنە خەۋەر يوق مائارىپ ئىدارىسىغا بارغاندا باشلىقلارنىڭ ئەلپازى تېغىمۇ يامانلاشقان ، ئۇلار << يەنە بىر ئوقۇتقۇچىنىڭ شەرتى ئۇنۋان ئېلىشقا توشىدىكەن ، سىز نىڭ شەرتلىرىڭىز توشمايدۇ كىلەر يىلى ئىلتىماس قىلىڭ >> دەپ مۇتھەملىك قىلغان ، ئاچام ئاچچىقىدا ئايلىنىپ كىتەيلە دەپ قاپتۇ ، ئاندىن كىيىن ئۇلارغا ئۇنۋاننى بەرمىسە ۋىلايەتلىك مائارىپ ئىدارىسىغا ئەرز قىلىدىغانلىقىنى ئېيتقاندا ئۇلار سەل بوشاپ قالغان ، ھازىرچە يوقۇرغا مەلۇم قىلماسلىقىنى ئېيتىپ ئاچامنى كەتكۈزۋەتكەن ، 1ھەپتە ئۈتكەندىن كىيىن مائارىپ ئىدارىسى ئاچامنى چاقىرتىپ مەكتەپ رەھبەرلىرىدىن نەتىجىنى كۈتىشىنى خاتىرجەم بولىشىنى ئېيتقان ، شۇنداق قىلىپ ئاچام مائارىپ ئىدارىسىگە بىر نەچچە قېتىم چېپىپ يۈرۈپ ئۇنۋان مەسىلىسىنى ھەل قىلغان .
مائارىپتىكى سامان تىگىدىن سۇ يۈگۈرتىدىغان ئىشلار قومۇلدىمۇ يوق ئەمەس بۇنداق ئىشلار ھەممە يەردە مەۋجۈت ، ئەمما ئاچامنىڭ ئۇنۋان ئىشىنى باغۋەننىڭ ئىشى بىلەن سېلىشتۇرغاندا ئاسمان زىمىن پەرىقلىندۇ ،
قومۇلدا پەقەت تەلەپ پىكىرلەر يوللۇق ھەق دۇرۇس بولسىىلا ھەققانىيەت يولىدا مەقسىتىگە يىتىش ئۈچۈن ئاخىرغىچە چىداپ بەل قويىۋەتمىسىلا ھەممە ئىشلار ئاساسەن كۈڭۈلدىكىدەك ھەل بولىدۇ ،  

شۇنى دەڭە ! بەزىلەر ئەركىشى تۇرۇپ ئاچىڭىز دەك بىر ئايال كىشىدە بولغانچىلىك يۈرەك ھەم ۋىژدان بولمىسا ئەلۋەتتە ئۇنۋاننى مائارىپ كادىرلىرى ئىشىك ئالدىغا ئەكلىپ تارقاتمايدۇ دە  !
