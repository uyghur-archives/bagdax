بىز ئۇيغۇرلارنىڭ باتۇر قىزى نۇزۇگۇم  ـئۆزى ياشاپ ئۆتكەن ئەينى ۋاقىتلاردا ، ئۆزىنىڭ مول بىلىمى ، گۈزەللىكى ، يۈكسەك ئەخلاق  ـ پەزىلىتى بىلەن ، خەلقىمىز ئارىسىدا  
« نازۇگۇم » دېگەن مۇھەببەتلىك نام بىلەن ئاتالغان ئىكەنتۇق .
شائىر ، موللا بىلال موللا يۈسۈپ ئوغلى ( نازىم ) ئەپەندىم ، ئۆزىنىڭ ( « نۇزۇگۇم قىسسىسى » ) ناملىق ئەسىرىدە ، نۇزۇگۇم ھەققىدە توختېلىپ مۇنداق ئېيتىدۇ :
« بۇ نازۇگۇم قىز ، موللا ئايال ئىدى . ئۆزى ئىلباغدەك ئۆڭلۈك ، قارا قاش ، قاڭشارلىق نازۇك بەدەن بولغاچقا ، قەشقەر خەلقى « نازۇگۇم » دەپ ئاتىغان ئىدى . »
نۇزۇگۇمنىڭ باتۇرلىقى ھەم قەھرىمانلارچە ئۆلۈمى ، بىز ئۇيغۇر قىزلىرىنىڭ مىللي روھىمىزنى تىكلىۋىلىشىمىز ئۈچۈن ، نادىر ئۈلگە بولالايدۇ .

كاللامنى سوغوق سۇدا چاقاپ زۇكامداپ قاسام دورا پۇلىنى ئاتچوت قىپ بېرىدىغاندەكلا گەپ قىپسىز بۇرادەر ئەلۋەتتە شۇ دەۋېرنى مەن باشتىن كەچۈرمىدىم،سىز ھەم شۇنداق بولىشىڭىز ئېنىق ،سىزمۇ ئاڭلىغىنىڭىزنى ئوتتۇرغا تاشلاپ شۇ دەۋېردىكىلەرگە قوشۇپ ئۆزىڭىزنى ئاقلاۋاتىسىز،شۇ دەۋېردە كارغا كەلگۈدەك ئەركەك زاتى قالمىسا نۇزۇگۇممۇ نۇزۇك باشىغا دەت كەلدى ،ئەپ قاچارغا ئەر بامۇ دىگەن قوشاقنى توۋلاپ پادىچى ئەركەكلېرىدىن پاناھلىق سورىماس بوغۇيتى دەپ مەنمۇ كاللامدىن ئۆتمىگىنىنى دىدىم،ساختا تارىختىن كاللىسى ساق ئادەم چىنلىقنىمۇ ھېس قىلىپ قىياس قىلالايدۇ،يېقىنكى كۈنلەردىمۇ ئەر بامۇ دىگەن ناخشا مەيدانغا چىقىپ، ماشىندا ئاۋازنى بولىشىغا قۇياپ باشلېرىنى تازا مۇڭ قىپ گىلدىڭشىتىپ ئاڭلاپ،رەڭلىك سورۇنغا ماڭغان ئەركەكلىنى تۇلا كۆرەپ كەتتىم،مەسىلە شۇ دەۋېردىكى ئەركەكلە قوي بېقىپ گۆشسىز قېلىپ يىلىكى سۇيۇلۇپ كىتىپ ئەركەكلىك بۇرچىنى پادىچىلىقىنى ئادا قىلالىغۇدەك قەھرى قالمىغان دەپ توغرا چۈشۈنۈپ شۇ دەۋېردىكى ئەركەكلەنى ئاقلىدۇقمۇ دەيلى، سىز بىز ياشاۋاتقان بۇگۈنكى كۈندىكى ئەر بامۇ دىگەن ناخشا قوشاققا نىمىنى بانا قىلىمىز؟
قارا قىشتا سەگەك كاللامنى سوغوق سۇدا چايقايمە دەپ ئاۋارە بوغۇچە قۇڭامنى چايقاپ ،ئاپتاپ كۆرمىگەن ئەر دىگەن ئاۋازلىق ئەسەرنى خوشنا خۇلۇم ئاڭلىغۇدەك ۋاڭشىتىپ قۇياپ قۇيۇپ، يولۋاس ئۆلتۇرگەن ئەزمەت ۋە كىر يۇيۋاتقان ئەر دىگەن كىتاپنى يەنە بى ئوقىۋىتەيچۇ---ئەركەكلىكىم بېشىنى كۇتەرەپ قاسا ئەجەپ ئەمەس
