     ئىلاۋە:ئەسلى بۇتىمىنى مىسرانىم ۋە روھلان تۇرىدا ئېلان قىلدۇرغان ،مىسرانىم تۇرىدا «ئەلدوست شىركىتى»دىكىلەر خېلى ئۇمۇتلەنگۇدەك گەپلەرنى قىلغان ئىدى.كىيىن داۋالىنىشنى مۇۋاپىق كۇرۇپ خىراجەت قاتارلىق جەھەتلەردىن مەسلەھەت سورىسام جاۋاپ بەرمەي يۇقاپ كەتتى.شۇڭا باغداش مۇنبىرىدىكى مۇنبەرداشلارنىڭ ماڭا بىرەر ياخشى دوختۇرخانا ياكى دۇرا تەۋىسىيە قىلىپ قالامدىكى دەپ ئويلاپ بۇ مۇنبەرگە يوللاشنى مۇۋاپىق كوردۇم ،ئوقۇپ بولغانلارنىڭ باشقىچە خىياللاردا بۇلۇپ قالماسلىقىنى ئۇمىت قىلىمەن.
      ھازىرمۇ پۇتۇمدىكى جاراھەتنىڭ ئاغرىق ئازاۋىدىن ھىچ ئىش قىلالماي ،نەگە بېرىشنى ،قانداق داۋالىنىشنى بىلمەي تۇرۋاتىمەن،سەمىمى مەسلەھەتىڭلارغا مۇھتاجمەن.


       ئەتە چوڭ ئوغلۇمنىڭ تەتىللىك ئارام ئېلىش ۋاقتى تۇشۇپ مەكتىۋىگە بارىدىغان كۇن. مەن دادىسى بولغاندىن كىيىن ئەلۋەتتە ئۇنى مەكتىۋىگە ئاپىرىپ قۇيۇش كىرەك،ئۇنىڭغا تىرىشىپ ياخشى ئوقۇش توغرىلىق تەربىيىلەرنى بىرىپ مۇستەققىل ياشاش ئىقىتىدارىنى يىتىلدۇرۇشنى ،ئۇزىنىڭ ئىشلىرىنىڭ ھۆددىسىدىن ياخشى چىقىشنى ،ساۋاقداشلىرى بىلەن ئىتىپاق ئۈتۈپ بىزنى ئەنسىرەتمەسلىگىنى ئۇزنىڭ كەلگۇسى توغرىلىق ئويلىنىپ بېقىشنى يەنە ئىشقىلىپ دادىسى بالىسىغا قىلىدىغان ئۇزۇن تەربىيىلەرنىڭ ھەممسىنى قىلىپ مەكتىپىگە ئەكىرىپ قۇيۇشۇم كىرەك ئىدى.ئەمما،،،،،،،،،،،،،،،،،،،،،،،،،،،،
  12يىل بولدى ئۇزۈمنىڭ بۇكىسەلگە قانداق گىرىپتار بۇلۇپ قالغىنىمنى بىلمەيمەن.بەلكىم ئۆي كوچكەندە ياغاچ تىگىپ پۇتۇمنىڭ  سۇنۇپ كىتىشى سەۋەپ بولغاندۇ،ياكى ئاكتىپلىق قىلىپ رەھبەرلىرىم داڭلاپ قويسا پۇتۇم سۇنۇپ كىتىپمۇ ياخشى ئارام ئالماي 8كۇندىن كىيىن ھاسا تاياق بىلەن ئىشلەش جەريانىدا تۇمۇرلار گېپىس ئارىدا ئىلستىكىلىقى ياخشى بولماي پەيدا بۇلۇپ قالغان بۇلىشىمۇ مۇمكىن دوختۇرلارمۇ شۇنداق دىگەن ئىدى ئەينى ۋاقىتتا.
ساقايماس كىسەل ياغاچ قۇرتىدەك ئىچ باغرىمنى يەپ بولماقتا.ئۈزۈمنى روھلۇق تۇتۇپ كىسەلنى قەتئى ئويلىماسلىققا تىرىشىپ نەچچە ۋاقىتلارنى ئۆتكۇزۇپ كەلدىم .چۇقۇم ساقىيىپ كىتىدۇ دەپ نۇرغۇنلىغان دوختۇرخانىلارغا بېرىپ داۋالىنىپ باقتىم،لىكىن ياخشىلىنىش يوق كۇندىن كۇنگە ئېغىرلىشىپ 2يىلدىن بىرى پۇتۇمدا جاراھەت بولسا ساقايماي يول مېڭىشىم قىيىنلىشىپ ئويدە ئولتۇرۇشقا مەجبۇر بۇلۇپ قالدىم .ئويلىغان نۇرغۇن ئارزۇ ئارمانلىرىم ،قىلماقچى بولغان نۇرغۇن ئىشلىرىم ئۆز جايىدا قالدى ۋە كەينىگە كىچىكىپ كىتىپ بارىدۇ.سۇنماس ئىرادىلىرىم سۇنۇشقا ،جەسۇر قامەتلىرىم ئىگىلىشكە باشلىدى.بالىلىرىمغا قارىسام يۇرۇكۇم پۇچلىنىپ كىتىدۇ.ئۇزۇمنى يارىماس دادىدەك ھىس قىلىمەن.ئۇلاردىن ئىچىمدە كەچۇرۇم سوراپ كىتىمەن.چوڭ ئوغلۇم 14 ياشقا كىرەي دەپ قالدى .كىچىكى 7ياشقا كىردى .ئىككىپەرزەنتىمنى ئېلىپ باغچىلارغا ئاپىرىپ ئوينىتىپ باقمىغانلىقىمنى ئويلىسام ئۇزۇمنى قۇيىدىغان يەر تاپالماي كىتىمەن.ماڭالماسلىق ئىچ باغرىمنى تىلغىماقتا.
  ئايالىمغا نىمە دىيىشىمنى بىلەلمەيمەن.«جاپا تارتىۋاتىسىز»دىگەن بۇ بىر ئاددى سوزنى مىڭ قېتىم دىگىنىم بىلەنمۇ ئۇنىڭ جاپاسىنى يەڭگىللىتىشكە ئاجىزلىق قىلىۋاتاتتىم .ئۇنىڭ كۇندىزى مەكتەپتە ئىشلەپ ئەتىسى ئاخشىمى ئاش -تاماق ئىتىپ يەنە ئويدىكى قوي قوتانلارنىڭ ھالىدىن خەۋەر ئېلىپ شەنبە -يەكشەنبە كۇنلىرى يىغىلىپ قالغان كىرلارنى يۇيۇپ ،مەكتەپتە تۇگىتىپ بۇلالمىغان تاپشۇرۇقلارنى ئۆيگە ئېلىپ كىلىپ تەكشۇرۇپ قوشكىشىلىك جاپا تارتىۋاتقانلىقىغا قارىسام ئۆينى تاشلاپ يۇقاپ كەتكۇم كىلىپ كىتىدۇ.ئەڭ چاتاق يېرى يېقىنقى ئىككى يىلدىن بىرى پىسخىكامدا كۇرىلىۋاتقان مەسىللەر ئائىلە تۇرمۇشىمىزغا بەزى سوغوقچىلىقلارنى ئېلىپ كىلىشكە باشلىدى.
  كۇنلىرىمنى كۇلكە چاقچاق بىلەن ئوتكۇزۇپ كەلگەن مەندەك بىر ئادەمنىڭ ماڭالماس بۇلۇپ قېلىشى دوستلىرىم ۋە ئۇرۇق تۇققانلىرىمنى ھەيران قالدۇرماقتا ۋە ئەنسىرەتمەكتە ئىدى.10يىلدىن بىرى جىق ماڭالمىغان بىلەن ئوز ئىشىمنىڭ ھوددىسىن چىقىپ كىتەلىگۇدەك ھالەتتە ئىدىم .ئەمما 2يىلدىن بۇيان ھاجەتكە چىقىپ كىرىپ بولغىچە پۇتۇمدىكى جاراھەت بىر ئاغرىغىلى تۇرسا ئېغىزلىرىمغا گەپمۇ كەلمەي ،پىشانەمدىن مۇنچاق- مۇنچاق تەرلەر چىقىپ نەپەسلىرىم تىزلىشىپ بىر -ئىككى سائەتكىچە يىلاندەك تولغۇشۇپ كىتەتتىم .ئەزەلدىن بىر يىرى ئاغرىپ ۋايجانلاپ ياتقان ئادەمنى كورسەم ئاچچىقىم كىلەتتى ،ئەمما ھازىر ئۇزۇم بىلىپ بىلمەي بۇۋايجان دىگەنلەرنى نەچچىنى دىۋەتكەنلىكىمنىمۇ بىلمەي قالىمەن.كۇندىن كۇنگە سەپرا مىجەز بۇلۇپ ئايالىمنىڭ بازاردىن پۇكۇنى نەرسىنى ئەكىرسىڭىزچۇ دىگەن گەپلىرى قۇلۇقۇمغا مۇش بىلەن ئۇرغاندەك ئاڭلىنىپ ئۇرۇشقىلى تەييار بۇلۇپ تۇرىدىغان بۇلۇپ كاللامغا كەلمىگەن خىياللار كىلىپ توختاپ قالىدىغان بۇلۇپ قالدىم .ئۇزۇن يىل داۋالىنىۋەرگەنلىكتىن كىسىلىمگە نىسبەتەن ئايالىمنىڭ كاللىسىدا سىزىم قالمىغاندەك تۇراتتى.باللىرىمغا بىر گەپنى دەپ ئىجرا قىلدۇرالمىسام ئۇلارنىمۇ ئۇرىۋىتىشكە تاسلا قالاتتىم .ئايالى ۋە پەرزەنىتلىرىنى ئۇرۇپ باشقۇرۇش بۇتىپىك ئىقتىدارسىز ئەرلەرنىڭ قىلىدىغان ئىشى بۇلۇپ مەنمۇ مۇشۇ ئىقتىدارسىز ئەرلەر قاتارىغا كىرىش ئالدىدا تۇرماقتا ئىدىم .ياق!ياق! دەيتىم ئۆزۇمگە كاللامنى سىلكىۋېتىشىمنى دائىم ئەسكەرتىپ تۇراتتىم .ماشىنا ھەيداپ ماڭغان ۋاقىتلىرىمدا ئالدىمدىكى ماشىنىغا سىگنال بەرسەم يول بەرمىسە سۇقۇۋېتىشنى ئويلىغان ۋاقىتلىرىممۇ بەك كۆپ.بولسا ماشىنا ئالدىدىكى ماشىننى كۇتۇرۇپ يوتكەپ قۇيىدىغان ئىقتىدارلارنى سەپلەپ قويسا دەپ ئويلاپ كىتەتتىم .كىچسى چىراقتا پاس بەرسەم پاس بەرمىسە مەخسۇس ماشىنامنىڭ ئىچىگە قۇيۇپ قويغان ئەينەكنى چىقىرىپ ئالدىمدىكى ماشىنىنىڭ چىرىقىغا توغرىلاپ تۇتۇپ ئۇلارنى تاس قالتتىم ئازگالغا چۇشۇر ۋەتكىلى .ئۇزۇمنىڭ نىمە بۇلۇپ كىتىۋاتقانلىقىمنى بىلمەي قېلىۋاتاتتىم.
  نۇرمال ماڭغان ئاەملەرگە قاراپ ھەۋەس قىلىدىغان بۇلۇپ قالغىنىمنىڭ قاچاندىن باشلانغانلىقى ئىسىمدە يوق .بۇرۇنقى ۋاقىتلىرىمنى ئەسلەيتىم ،ئوقۇتقۇچىلار بىلەن ئوينىغان توپلىرىم ،مەكتىۋىمىزدىكى توي -توكۇنلەردە ئۇيناپ كەتكەن ئۈچ پەدە ،ئاتلاشكۇلىرىم .قېنى ئۇلار ،ئۇئىشلار ئويدە ئولتۇرغان ۋاقىتلىرىمدا بىربىرلەپ ئىسىمگە كىلىشكە باشلايتى .روھىمنى غاجىلايتى ،پىسخىكامنى تېخىمۇ قالايمىقانلاشتۇرۋىتەتتى .بۇنىڭدىن ئەنسىرەپ مەكتەپكە چىقىپ ئىشلەشكە مەجبۇر بۇلاتتىم لىكىن 2كۇندىن كىيىن يەنىلا ئۆيدە ئولتۇرۇشقا مەجبۇر ئىدىم .مىنىڭ كىسىلىم «قان تۇمۇر توسالغۇغا ئۇچراپ قان تۇمۇر دىۋارىدا نوكچا قان پەيدا بۇلۇپ  قېلىش كىسىلى» بۇلۇپ بۇكىسەل ئارام ئېلىشنى تەلەپ قىلاتتى سەل كۆپرەك يول مېڭىپ قالسام پۇتلىرىم كۇپتەپ ئىششىپ ئاياقلىرىم پاتماي قالاتتى .
   2003-يىلىدىن ھازىرغىچە ئاپتۇنۇم رايۇنلۇق تىببى ئۈنۋېرسىت دوختۇرخانىسى،جوڭيى دوختۇرخانىسى،شىجىياجۇاڭ تىنىچلىق دوختۇرخانىسى قاتارلىق دوختۇرخانىلاردا داۋالىنىپ بولدۇم .خوتەندىمۇ مىللى تىبابەت بىلەن داۋالىنىپ باقتىم ،دىگەندەك كۇرۇنەرلىك ئۇنۇمگە ئىرىشەلمىدىم.
  توردا مەسلىھەت سوراش جەريانىدا بىرقانچە مۇنبەرگە ئەزا بۇلۇپ توردىكى ئەسەرلەرنى ئوقۇپ ۋاقىت ئۆتكۇزۇشكە تىرىشىپ باقتىم ، ئەمما كومپىيوتتىر ئالدىدا ئۇزۇن ئولتۇرۇشمۇ ئوخشاشلا پۇتۇمغا زىيان قىلاتتى.بۇگۇنمۇ تونۇگۇن بىركۇن جاپالىق ئىشلىگەننىڭ پايدىسىنى كۇرۇپ يەنە ئۆيدە ئولتۇرۇپتىمەن . ئەتە ئوغلۇمنى تىنىچ -ئامان مەكتىپىگە ئاپىرىپ قۇيۇشنى ئويلايتىم .شۇڭا بۇگۇن تىنىچ ياخشى ئارام ئېلىشنى ئويلىدىم ۋە دەرتلىرىمنى سىلەر بىلەن ئورتاقلىشىش ئۇچۇن بۇتىمامنى يوللاپ ئولتۇرۇپتىمەن .
  ئاللا ئىگەم ماڭا سەۋىر تاقەت ئاتا قىلغايسەن ،ساراڭ بۇلۇپ قېلىشىمنىڭ ئالدىنى ئېلىشقا ياردەم قىلغايسەن .
