پاي بازىرىدا سودىنى توختىتىش مىخانىزىمى ھەققىدە قارىشىم:ئادىل بايقال(موللام)
ئاكسيە كومىتىتى 2016-يىلى1-ئاينىڭ1-كۈنىدىن باشلاپ پاي سودىسىنى ئالاھىدە ئەھۋالدا ۋاقىتلىق توختىتىش مىخانىزىمىنى رەسمى ئورنىتىشنى قارار قىلدى،شۇنداقلا تىراگىدىيەلىك تۇس ئالغان ئىشمۇ  يۈز بەردى،يەنى 4-چىسلا پاي بازىرى شىددەت بىلەن چۇشۈپ بۇ تۇزۇمنى تۇنجى يۇلغا قويغان كۈندىلا بازار تەرىپىدىن رەسمى قوللىنىلدى،ئاۋۋال 5%چۈشكەندە 15مىنۇت سودا توختىىلدى،ئاندىن 7%چۈشكەندە پۇتۇن كۈن بازار سودىدىن رەسمى توختىتىلدى،بۇ جوڭگۇ بازىرى يۇلغا قۇيۇلغان 20نەچچە يىلدىن بۇيانقى يېڭىلىق بولۇپ،ئاساسلىق مەخسەد پاي بازىرىنىڭ چوڭ داۋالغۇشىنىڭ ئالدىنى ئېلىش ۋە پاي ئاپىتىگە تاقابىل تۇرۇش مەخسەد قىلىنغان،ئامېرىكا بازىرىمۇ 1987-يىلى بۇ تۇزۇمنى رەسمى يۇلغان قويغاندىن بۇيان پەقەت 1997-يىلى بىرلا قېتىم ئىجرا قىلىنغان،جوڭگۇدا بولسا تۇنجى كۈنىلا ئېجرا قىلىنغان،تۈنۈگۈنكى چوڭ تەۋرىنىشتە جۇڭگۇ بازىرىدىن تەخمىنەم 4تىرلىيۇن پايلىق كۆپۈككە ئايلانغان،نۇرغۇن ئۇششاق مەبلەغ سالغۇچىلار بۇ تۇزۇمنى چۈشەنمەسلىك،چۆچۈش،تەشۋىشلىنىش پىسخىكىسى پۇتۇن بازارغا ھەتتا فوند دوكا قاتارلىقلارغىمۇ پاسسىپ تەسر كۆرسەتكەن،چۈنكى جۇڭگۇ بازىرى ئۇششاق پايچىلار تەرىپىدىن ئاساسى قۇرۇلمىنى ئىگىلىگەن،95%ئۇششاق پايچى،ئامېرىكا بازىرىنىڭ بولسا 95%دوكا ،پەقەت 5%ئۇششاق پايچى،شۇڭا ھەرقانداق تۇزۇلمىنى قوللانغاندا ئامېرىكا بىلەن جۇڭگۇنىڭ قۇبۇل قىلىشى ،بازار ئىنكاسى ئوخشاش بولماسلىقنىڭ سەۋەبى نەق مۇشۇ يەدە،
دۆلىتىمىز پاي بازرىنى رەتلەپ تۇزەش جەھەتتىن 2016-يىلى چوڭ خىزمەتلەرنى ئىشلەۋاتىدۇ،بولۇپمۇ ئەنگە ئالدۇرۇش تۈزىمى،پاي سودىسىنى ئالاھىدە ئەھۋالدا توختىتىش تۈزىمى،يېڭى پاينى تاقىتىش ئىسلاھاتى قاتارلىق زۇر ئىسلاھاتلار پاي بازىرىمىزنىڭ ساغلام ئىزچىل تەرەقىياتىغا چوڭ تۆھپە قوشىشى مۈمكىن،يۇقارقى ئسلاھاتلارمۇ پىشىپ يېتىلگەن پاي بازارلىرىدا مۇۋەپپەقىيەت قازانغان تەجرىبىلەردۇر،بۇ يىل جۇڭگۇ مەركىزى بانكىسى ئاز دىگەندە 3-5قېتىمغىچە ئۆسۈم چۈشىرىشى مۈمكىن،پاي بازىرى يەنىلا كاپىتال بازىرىدىكى ئاڭ جانلىق جەڭ مەيدانى بولىشى مۈمكىنچىلىگى بار،چۇنكى ئۆسۈمنىڭ ئۇدا چۈشىشى ،چەتئەل قىززىق پۇللىرىنىڭ جۇڭگۇ بازىرىدىن چېكىنىشى قاتارلىقلار سەۋەبى بىلەن ئامانەت پۇل پاي بازىرىغا كۆچىشى مۈمكىن،بۇنداق بولغاندا سۇ كەلگىچە توغان سال دەپ ئاكسيە كومىتىتى مۇشۇ مۇناسىۋەت بىلەن بارلىق قايدە تۇزۇملەرنى مۇكەممەللەشتۇرۇپ ،كاپىتال بازىرىنىڭ ساغلام تەرەقىياتىغا ئاساس سېلىشى ئىرادىسى كۈچلۈك،
يېقىندا دىيارىمىزدىكى پايچىلارنىڭ كەسپى ساپاسىمۇ تېزدىن ئۆسۈم كاپىتال بازىرىنىڭ ئۆزگىرىشلىرىگە ۋاقتىدا ئاكتىپ ئىنكاس قايتۇردىغان ھالەتكە كەلدى،ئاكسيە مۇنبىرىدە تۈگۈنكى پاي سودىسىنى ۋاقتىلىق توختىتىش چارىسى ھەققىدە كۆپ مۇنازىرە بولۇندى،مۇنازىرە  خۇلاسىسى بۇ قېتىملىق چوڭ چۇشۈشنىڭ ئاساسلىق سەۋەپكارى  پاي سودىسىنى ۋاقتىلىق توختۇتۇش مىخانىزىمى،ئەنگە ئالدۇرۇش سېستىمىسى،يېڭى پاي تاقىتىش ئىسلاھاتى،جوڭگۇ ئىقتىسادىنىڭ ئىزچىل چېكىنىشى،ۋاقتى توشقان پايلارنىڭ بازارغا كىرىشى قاتارلىقلارغا يىغىنچاقلاندى،بولۇپمۇ پاي سودىسىنى ۋاقتىلىق توختىتىش بىلەن ۋاقتى توشقان پايلارنىڭ بازارغا كىرىشى پىسخىكا جەھەتتىن چوڭ ئىنكاس بولدى دەپ خۇلاسە قىلىندى،
2016-يىلىغا نىسبەتەن ئېيتقاندا قارىشىم شۇكى بازار 3000-4500گىچە تەۋرىنىشى،قىززىق نۇقتىلار تەنتەربىيە،كىنۇ تىياتىر،ساغلاملىق ۋە نازۇك تېخنىكىغا ئىگە ھەربى سانائەت كارخانلىرى بولىشى مۈمكىن....

پاي چېكى بازىرىدا مۇشۇنداق بىر چوڭ ئىقتىسادى ۋەيرانچىلىقنىڭ يۈز بېرىدىغانلىقىنى بىر نەچچە ئاي بۇرۇن ئېيتىپ، پايچى دەللال، ئالدامچىلارنىڭ پۇقرالارنى ئالدىماسلىقى توغۇرلۇق ئاگاھلاندۇرغان ئىدۇق.  بۇ دەللال قويمىچىلارنى بەزەن تورداشلار  خېلە ئېغىر گەپلەر بىلەن ئەدەپلىگەن. بىراق ئالدامچىلار ھىلە مىكىردىن ۋاز كەچمىگەن. مانا ئەمدى نومۇس قىلماستىن قايسى يۈزى بىلەن بۇ خەۋەرنى يازغاندۇر؟ 《جۇڭگودا پاي بازىرىنى 95‰ ئۇششاق پايچىلار، 5 ‰ دۆكالار  ئىگىلەيدۇ، ئامېرىكىدا 95‰ دۆكالار، 5 % ئۇششاق پايچىلار ئىگىلەيدۇ... 》پالان پۇستان، دەپ جۇڭگودىكى پاي چېكى بازىرىنىڭ ۋەيران بولىشىنى ئالدانغان  ئۇششاق پايچىلارغا ئارتىپ قويماقچى. جۇڭگولۇق پۇقرالار پاي چېكى ۋە پاي سودىسىنىڭ قانداق شەكىلدىكى قىمار ئىكەنلىكىنى بىلمەيدۇ. ئۇلارنى پاي دەللاللىرى ئالداپ ئەكىرىدۇ. جۇڭگودىكى چوڭ پۇلدارلار ، چوڭ شىركەت ئىگىلىرى مەبلىغىنىڭ مەلۇم قىسمىنى پاي چېكى بازىرىغا دەسمى سېلىپ، چەتئەللىك چوڭ قىمارۋازلار بىلەن ئورتاقلىشىپ، جۇڭگودىكى  ئۇششاق پايچىلارنى يەنى كامبىغەل پۇقرالارنى ئالدايدۇ. جۇڭگودىكى چوڭ پايچىلار زىيان تارتسا دۆلەتكە تۆلەيدىغان باژغا سۇندۇرۇپ قۇتىلىدۇ. كامبىغەل پايچىلار ۋەيران بولۇپ، ئېسىلىپ ئۆلىۋالىدۇ، ياكى ساراڭ بولۇپ تۈگىشىدۇ. غەرب ئەللىرىدە كامبىغەل پۇقرالارنى دۆكا پايچىلار ئالدىمايدۇ. شۇڭا ئۇششاق پايچىلار 5 % نى تەشكىل قىلىدۇ. يىغىپ ئېيتقاندا پاي چېكى سودىسى دۇنياۋىلاشقان قىمارنىڭ دەل ئۆزىدۇر. بىچارە جۇڭگولۇق نامرات پۇقرالار ئۆمۈر بويى ئىشلەپ، تېجەپ تاپقان تەرگنلىرىنى ، بانكىدىن ئالغان قەرز پۇللىرىنى قوشۇپ، پاي چېكى قىمارغا دو تىكىپ بىراقلا ۋەيران بولدى. بەش كۈن ئىچىدە يوقالغان 4 تىرلىيون پۇلنىڭ 95 % قىسمى كامبىغەل پايچىلارنىڭ پۇلى ئىدى. بۇ كىچىك نەرسە ئەمەس.
