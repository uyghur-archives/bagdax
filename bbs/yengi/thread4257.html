  مەن جۇمە نامىزىنىڭ خۇتبىلىرىدە ئاڭلىغان ئەھمىيەتلىك كۆرسەتمە ۋە بەلگىلىمىلەرنىڭ ئاساسلىق مەزمۇنىنى تورداشلارنىڭ ھوزۇرىغا سۇنۇشنى توغرا تاپتىپ،بەلكىم بۇ ھەدىس ۋە ئايەتلەر سىزنىڭ تۇرمۇشىڭىزغا تەسىر كۆرسەتسە ئەجەپ ئەمەس.
1.بەخت دېگەن نىمە؟
جاۋاب:بەخت دېگەن ئىماندىن(ئىشەنچتىن ئىبارات)،ئۇ ھەرگىزمۇ بايلىق ياكى ھوقۇق ۋە باشقىسى ئەمەس.ئاللاغا ھەقىقى ئىشەنگەن كىشىلا خاتىرجەملىككە بەخت تۇيغۇسىغا ئېرىشەلەيدۇ. بايلىقنىڭ قوغلاشقا كىشىنىڭ كۆڭلى دەككە-دۈككە ئىچىدىلا بولىدۇ،ئۇ كىشىدە كۆڭۇل ئارامى بولمايدۇ.

2.جەننەت ۋە دوۋزاق سىلەرگە ئايىقىڭلارنىڭ بوقۇچىدىنمۇ يېقىن.
ئىككى مۇسۇلمان سەپەر داۋامىدا بۇتپەرەسلەرنىڭ يۇرتىدىن ئۆتۈشكە توغرا كەپتۇ،بۇ چاغدا بۇتپەرەسلەر دەپتۇكى :سىلەر بىزنىڭ بۇتلىرىمىزغا يېقىنچىلىق قىلساڭلار ئاندىن بۇ جايدىن ئۆتكۈزىمىز دەپتۇ. مۇسۇلمانلار ئۆزلىرىنىڭ يېنىدا ھىچقانداق نەرسىنىڭ يوقلىقىنى بۇتلارغا يېقىنچىلىق قىلىشقا ئامالسىزلىقىنى ئېيتقاندا ،بۇتپەرەسلەر بىرەر چىۋىننى تۇتۇپ بولسىمۇ بۇتلىرىغا يېقىنچىلىق قىلىشنى ئېيتىپتۇ،مۇسۇلمانلاردىن بىرى چىۋىننى تۇتۇپ بۇتقا سوۋغا قىلىپ بۇ جايدى ئۆتۈپتۇ.لېكىن دوزىقى بولۇپتۇ.يەنە بىر مۇسۇلمان ئۆزىنىڭ بىر ئاللادىن باشقىغا يېقىنچىلىق قىلمايدىغانلىقىنى ئېيتىپتىكەن ،ئۆلتۈرىلىپتۇ لېكىن جەننىتى بولۇپتۇ.
تەھلىل شۇكى:بىز تۇرمۇشتا چىۋىن ئۆلتۈرۈپ بۇتقا يېقىنچىلىق قىلىشتەك ئىشلەرگە قانچىلىك يولۇقتۇق؟!
3.ئەڭ ياخشىڭلار مېنىڭ دەۋرىمدە ياشىغانلىرىڭلار، ئاندىن قالسا ساھابى-كېرەملەرنىڭ دەۋرىدە ياشىغانلىرىڭلار..................شۇنداق بىر كىشىلەر مەيدانغا كېلۇدۇكى:ئۇلار ئېغىزىدا دېگىنىنى ئەمىلىيەتتە قىلمايدۇ. يەنى ئاللانىڭ قانۇنىنى ۋە ئاللانىڭ ئەلچىسىنىڭ كۆرسەتمىلىرىنى ئېغىزىدا سۆزلەيدۇ ئەمىلىيەتتە شۇ بويىچە ئىش قىلمايدۇ.
4.بۇ دۇنيادىكى ئازاب ئاللاھنىڭ بىرلىك بارلىقىنى ئىنكار قىلىش ئەمەس بەلكى ئاللاغا شىرىك كەلتۈرۈش.
5.دىندا ئوتتۇرىھال يول تۇتۇڭلار. يەنى دىندا بەلگىلەنگەن ئەمەللەرنى ۋە ئىشلارنى ئاشۇرۇۋېتىشمۇ، كېمەيتىۋېتىشمۇ ئوخشاشلا خاتا. دەل ئۆزىنى تۇتۇش توغرا.
6.دىنىي مەسىلىگە ھۆكۈم قىلغاندا كىتاب ئاساسىدا ھۆكۈم قىلىڭلار، ناۋادا كىتابتىن كۆرمىگەن بولساڭلار ئەڭ ياخشىسى مەن كۆرمىدىم دەڭلار. ئۇنداق بولمايدىكەن بەك چوڭ خاتالاشقان بولىسىلەر.
7.يالغان مۇسۇلماننىڭ ھۆكمى نىمە؟
 يالغان مۇسۇلمان مەسچىتكە كىرسە ئەڭ تەقۋادار كىشىگە ئايلىنىدۇ،مەسچىتتىن چىقسا بارلىق يامان ئىشلارنىڭ ھەممىسىنى قىلىدۇ.
8.تىلەكنىڭ ئىجابەت بولۇشىغا تىلەككە لايىق ھەرىكەت بولۇشى كېرەك. يەنى يىمەك-ئىچمەك، كىيىم- كېچەك پاكىز بولۇشى، نىيەت توغرا بولۇشى. ئاللاغا ھەقىقى يۈزلەنگەن بولۇشى كېرەك.  مۇشۇ ئاساستا تىرىشىپ مەقسىتىنى ئەمەلگە ئاشۇرۇش ئۈچۈن ئەمگەك قىلىشى ئىزدىنىشى لازىمدۇر.

9.ھەممە نىيەتكە باغلىق،يەنى قىلغان ئەمەل ۋە ياخشى ئىشلىرىڭىز پەقەت ئاللانىڭ رازىلىقى ئۈچۈنلا بولۇشى كېرەك ،ئۇنداق بولماي مال-دۇنيا ياكى ئەمەل......ئۈچۈن بولسا سىزنىڭ قىلغان ئەمەللىرىڭىزنىڭ ھىچقانداق پايدىسى يوق(ئاللانىڭ دەرگاھىدا).
ئاللانىڭ ئەلچىسىنىڭ زامانىدا ،ئاللانىڭ ئەلچىسى مۇھەممەتنىڭ ئەڭ ياراملىق باتۇرلىرىدىن بىرى سەكراتقاچۈشۈپ قالىدۇ،بۇ چاغدا ھەمرالىرى بۇ قېرىندىشىغا«بىر ئاللادىن باشقا ھىچ ئىلاھ يوق» (ئەرەپچە لائىلاھە ئىللالاھۇ) دىگۇزەلمەيدۇ،يەنى بۇ باتۇر تەقۋادارنىڭ تىلى تۇتۇلۇپ قالىدۇ.بۇ ئىش ئاللانىڭ ئەلچىسىگە مەلۇم قىلىنغاندا ،ئاللانىڭ ئەلچىسى ھەيران بولۇپ بۇ نىمە سەۋەپتىندۇر ؟!دەپ ئويلۇنۇپ قالىدۇ ھەمدۇ بۇ باتۇرنىڭ ئانىسىنى چاقىرتىپ كېلىپ سىز ئوغلۇڭىزدىن رازىمۇ؟دەپ سورىغاندا:
ئانا:مەن ئوغلۇمدىن رازى ئەمەس،چۇنكى ئۇ ئايالىنى مەندىن ئۈستۈن كۆردى،مەن بۇ ئىشقا چىدىماي ئۇنىڭ ئويىدىن كۆچۇپ كەتكەن .
ئاللانىڭ ئەلچىسى:ئۇنداقتا ئوتۇن توپلاپ  بۇ ئانىسى رازى بولمىغان بالىنى كۆيدۇرۇڭلار ئۇ ئانىسىنىڭ رازىلىقىنى ئالمىغاننىڭ جازاسىنى بۇ دۇنيادىلا  تارتسۇن دىگىنىدە.
ئانا:ئۇنداق قىلماڭلار !مەن ئوغلۇمدىن رازى !
دەپ رازىلىق بەرمەك ھامان،ھېلىقى تەقۋادار باتۇر«بىر ئاللادىن باشقا ھىچ ئىلاھ يوق»دەپ زىۋانغا كېلىدۇ.
يەكۇن شۇكى:ئانىنىڭ رازىلىقىنى ئالمىغان پەرزەت جەننەت يۈزى كۆرمەيدۇ.
10.قېرىنداشلىقنى ئۈزۈپ قويغان كىشى جەننەت يۈزى كۆرمەيدۇ.
11.ئۆلۇپ كەتكەن ئاتا ئانىغا نىمە تېگىدۇ؟
ئاللانىڭ ئەلچىسى بۇ سۇئالغا مۇنداق جاۋاپ بەرگەن
1،ئۈزۈلۇپ قالغان قېرىنداشلىقنى ئۇلىساڭ شۇنىڭ ياخشىلىقى(ساۋابى) تېگىدۇ.
2،ئاتا -ئانا قىلغان ياخشى ئىشلارنى داۋاملاشتۇرساڭ شۇنىڭ ياخشىلىقى تېگىدۇ.
3،ئاتا-ئاناڭ قىلماقچى بولۇپ قۇربى يەتمىگەن ياخشى ئىشلارنى قىلساڭ شۇنىڭ ياىشىلىقى تېگىدۇ.
4،ئاتا-ئاناڭغا ئاتا تىلەك(دۇئا) تىللىسەڭ شۇنىڭ ياخشىلىقىتېگىدۇ.
12،ئۆلگەندى كېيىنمۇ داۋاملىق ياخشىلىقى بولۇدىغان ئىشلار قايدى؟
1.ئىلىم ئۈگىتىش
2.ياخشى پەرزەت تەربىيىلەش
3.سەدىقە (ئاجىز-ئورۇقلارنىڭ ھەققى) بېرىش
13.تىلىكى ئىجاۋەت بولۇدىغان كىشىلەر كىم؟
1.روزا تۇتقان كىشىنىڭ ئېغىز ئېچىشتىن بۇرۇن قىلغان تىلىكى ئىجاۋەت بولىدۇ.
2.زۇلۇمغا يولۇققان كىشىنىڭ تىلىكى ئىجاۋەت بولىدۇ.
3.مەككە-مەدىنىنى تاۋاپ قىلغان كىشىنىڭ بەيتۇل مۇققەددەسنىڭ ئالدىدا قىلغان تىلىكى ئىجاۋەت بولىدۇ

1.بەخت-ئىماندىن ئىبارەت(تەھرىرلەش رەت قىلىنغاچقا تەھرىرلىيەلمىدىم)
