   ھەممىمىزگە مەلۇم بولغۇنىدەك ئىت ئىنسانلار ھاياتىدا  مۇھىم بولغان ھايۋانلارنىڭ بىرى. كۆپلىگەن ئىنسانلارنىڭ  كۈندۈلۈك تۇرمۇشى ئىت بىلەن زىچ  باغلىنىپ كەتكەن، بولۇپمۇ  غەرىپ ئەللىرىدىكى بەزى ئىنسانلار ئىتنى ئەتىۋارلايدۇ ۋە ئۇنى ئۆينىڭ بىر ئەزاسى قاتارىدا كۆرىدۇ، ئىت بىلەن بىرگە تاماق يەيدۇ، ئىت بىلەن بىرگە ئۇخلايدۇ. شۇنىڭ بىلەن بىر ۋاقىتتا ئىنسانلار ئەڭ يامان كۆرگەن ۋىجدانى يوق، ھورۇن ئادەملەرنى لالما ئىت دەپ تىللايدۇ. ئىنسانلارنىڭ قايسى خىلدىكى ئىتنى ئەتىۋارلىشى، ياكى ياخشى كورۇشى، قايسى خىلدىكى ئىتتىن سەسكىنىشى، ۋە ئۇنىڭدىن نەپىرەتلىنىشى، يالغۇز ئىتنىڭ تۈرىگە ۋە سورتىغا باغلىق بولۇپ قالـماستىن، بەلكى ئەڭ مۇھىمى ئىتنىڭ تەربىلىنىشىگە، مىجەزىگە، كۆنگەن خۇيى پەيلى ۋە ئادەتلىرىگە باغلىق.
   ئىنسانلار توۋەندىكى ئىككى خىل ئىتنى ياخشى كۆرىدۇ ۋە ئۇلارنى ئەتىۋارلاپ، ئاھىلىنىڭ بىر ئەزاسى كۆرۇشىدۇ.
    بىرىنجى تۇردىكىسى، قەددى قامەتلىك، ھەيۋىسى بار، باققان ئىگىسىنى تونىدىغان، ئاھىلە كىشىلىرى بىلەن ياتنى، ياخشى بىلەن ياماننى پەرىقلەندۇرەلەيدىغان ۋە ئۇلارنىڭ بىخەتەرلىگىنى قوغدايدىغان ئىتلاردۇر. بۇ خىلدىكى ئىتلار خالىغان ئادەمگە قاۋاپ، ئۇلارغا ھۇجۇم قىلمايدۇ، بىراق ئوزىگە ۋە ئىگىسىگە چىقىلغان ئادەمگە بوش كەلمەيدۇ. ئىنسانلار بۇخىلدىكى ئىتلارنى ئۆيگە قاراش، بىخەتەرلىك قوغداش قاتارلىق موھىم خىزمەتلەرگە سىلىش ئۇچۇن باقىدۇ، ۋە ئۇلارغا ئالاھىدە ھوقۇق ئىمتىيازلارنى بىرىدۇ.
    ئىككىنجى تۇردىكىسى گەرچە بىرىنجى تۈردىكى ئىتلاردەك ئالاھىدە قەددى قامەتلىك بولمىسىمۇ، بىراق كۆرۈنىشى ئوماق، دوستانە، ئادەمنىڭ زوقىنى كەلتۈرىدىغان، ئەدەپ ئەخلاقنى بىلىدىغان، خالىغان يەرگە تەرەت قىلمايدىغان، كۆرگەنلە نەرسىنى يىمەيدىغان، كۆرگەنلە ئادەمگە قاۋاپ، باشقىلارغا خالىغانچە ھۇجۇم قىلمايدىغان ئىتلاردۇر. ئىنسانلار بۇ خىلدىكى ئىتلارنى ئۆزىلىرىگە كۆڭەل خوشاللىقى ئىلىپ كىلىدۇ دەپ باقىدۇ، ۋە ئۇلارنى زىرىكىپ قالغاندا ئىشى پۇشىقىنى چىقىرىپ كوڭلىنى ئاچىدىغان يىقىن دوستى، ھەمرايى قاتارىدا ئەتىۋارلايدۇ.
   ئىنسانلارنىڭ يۇقارقى ئىككى تۇردىكى ئىتلارنى ياخشى كۆرىشىدىكى ئەڭ مۇھىم سەۋەپ، ئۇلارنىڭ ئىنسانلارغا مەلۇم دەرىجىدە مەنپەت يەتكۇزگەنلىگى ۋە خۇشاللىق ئىلىپ كەلگەنلىگىدەدۇر.
   بۇ ئىككى تۇردىكى ئىتلار كۈنلەرنىڭ بىرىدە ھەر خىل سەۋەپلەر تۇپەيلىدىن ئىگە چاقىسىز ۋە ئۆي ماكانسىز  قىلىپ، كوچىغا چىقىپ قالغان تەقدىردىمۇ، ۋە ياكى ئىگە چاقىسىز ئىتلارنى باقىدىغان ئورۇنلارغا كىرىپ قالغان تەقدىردىمۇ، بۇ ئىتالار خارلانمايدۇ، ۋە ئۇزۇنغا قالماي بۇ ئىتلارنى بىقىۋالىدىغانلار چىقىدۇ، نەتىجىدە بۇ ئىتلار يىڭىدىن ئۆي ماكانلىق بولىدۇ، ۋە بىقىۋالغۇچىلارغا مەلۇم دەرىجىدە مەنپەت ئىلىپ كىلىدۇ، ۋە ئۇلارنى ئۆزلىرىنى بىقىۋالغانلىقىغا پۇشايمان قىلدۇرمايدۇ.
   ئۇندىن باشقا يەنە بىر خىلدىكى ئىتلار بولۇپ، بۇ ئىتلارنىڭ تۇرى ۋە سۇرتى ئالدىنقى ئىككى خىل ئىتلاردىن ئالاھىدە پەرىقلىق بولمىسىمۇ، بىراق بۇ خىلدىكى ئىتلار ياخشى تەربىلەنمىگىنى ئۈچۈن، خالىغان ئادەمگە قاۋاپ قويدىغان، ياماننى كۆرسە قورقۇپ قاچىدىغان، ياۋاشنى كۆرسە قاۋاپ، چىشلەپ قويدىغان، كىم بىر نەرسە بەرسە ئۇنىڭ كەينىدىن كىتىدىغان، بەرمىسە ئوغۇرلاپ يەيدىغان، ئەتىۋارلاپ قويسا كورپىگىمۇ چىچىپ قويىدىغان ئادەتلىرى بولغاچقا، ئىنسانلار ئۇلارنى ياخشى كۆرمەيدۇ، ۋە لالما ئىت دەپ تىللىشىدۇ.
   كۈنلەرنىڭ بىرىدە بىرەرسىنىڭ بۇ ئىتلارغا ھىسىداشلىقى كىلىپ، ئۇلارنى بىقىۋالماقچى بولغاندا، ۋە ياكى ئۇلارغا ئۆي ماكان بەرگەندە، بۇ ئىتلار يەنىلا بۇرۇن كۆنگەن ئادەتلىرى ۋە خۇينى تاشلىماسلىقى، كۆرگەنلە ئادەمگە قاۋاپ، بىقىۋالغۇچىلارغا ھىچقاندا پايدا مەنپەت يەتكۇزمەي، ئەكسىنچە ئۇلارنى زىرىكتۇرۇپ قويىشى مۇمكىن، نەتىجىدە قايتىدىن ئىگە چاقىسىز ۋە ئوي ماكانسىز قىلىپ، كوچىغا چىقىپ بۇرۇنقى سەرگەردانلىق ھاياتىنى داۋام قىلىشى مۈمكىن.
