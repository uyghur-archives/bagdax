پېقىر سەئۇدى توغرىسىدا يەنە ئازراق چۈشەنچە بىرەي: 100 يىل بۇرۇن ئەرەبلەرئوسمان تۈرىك خەلىپىلىكىگە تەۋە ئىدى، ئوسمانلار پەن تېخنىكا جەھەتتىن ياۋروپانىڭ جىق ئارقىسىدا قالغان لىكىن تىرىتورىيەسى بەك چوڭ ئىدى. شۇڭا ئەنگىلىيە ئوسمان ئىمپىرىيەسىنى يىقىتىش ئۈچۈن ئەرەبلەرنى يەنى ئەرەب جەمەتلىرىنى كۈشكۈرىتتى...ۋاي تىرىسى ئاق، كۆزى قارا كىلىشكەن ئەرەبلەر،،قاچانغۇچا دۆلىتىڭ يوق تۈرىككە بېقىنىپ يۇرۇيسىلەر،،سىلى دىگەن باي ئاكامنىڭ بالىتسىتىڭىلىغۇ،،ھەرەمنىڭ غوجىسىتىڭىلىغۇ ..مەن بىلەن بىرلىشىڭىلا...خەلىپىنى يوقىتىپ ھەممىڭلارغا ئەرەبىستان قۇرۇپ بىرىمىز دىدى..شۇنىڭ بىلەن ئەرەبلار خەلىپىنىڭ كەينىدىن ۋاسسىدا پىچاق ئۇرۇپ ئوسمانلارنى غۇلاتتى..لىكىن ئەرەبىستان قۇرالمىدى..ئەكىسچە ياۋروپا دۆلەتلىرى ئوتتۇرا شەرىقىنى بۆلىشىۋېلىپ يەرلىك غوجىلارنى قورچاق قىلىپ تىكلەپ ئۆزىنىڭ مۇستەملىكىسگە ئايلاندۇردى. شۇلارنىڭ بىرى سەئۇدى ئائىلەسى! كيىن بۇ ئائىلە ئۆز ئالدغا دۆلات قۇرغانلىقىنى جاكارلاپ سەئۇدى ئەرەبىستان دىگەن دۆلەتنى جاكارلىدى..بۇ دىگىنى سەئۇدى ئائىلىسىنىڭ ئەرەبىستانى،،باشىقىلارنىڭ ئەمەس! شۇنداق بولغاچ ھازىر دۆلەت خەزىنىسى سەئۇدى جەمەتىگە تەۋە! دۆلەتكە ئەمەس! سەئۇدىدا ئىشلەيدىغان ياكى ئوقۇيدىغانلارنى تونىساڭلار ھازىر سەئۇدىدا تەسىرى چوڭ نەچچە جەمەت بار دەپ سوراپ بېقىڭلار...دەپ بىرىدۇ...بىرىنجىسى سەئۇدى جەمەتى،،ئىككىنجىسى پەيغەمبىرنىڭ جەمەتىگە تەۋەلەر (ئىككىنجى جۇما!!)، يەنە ئىككسىسا بار...مۇشۇ تۆت جەمەت سورۇننىڭ تورىسىدە..بىنلادىن جەمەتىمۇ سەئۇدى جەمەتى بىلەن ئالاقىسى قويۇق كۈچلۈك جەمەت، مۇھىم شەھەرلەرنىڭ تازلىق ئشچىلىرىنىڭ كىيىمىنىڭ ئۈستىدا بىنلادىن جەمەتى دەپ يوغان خەت بار تېخى. سەئۇدىدا ئىشلىمەكچى بولسىڭىز بىلىپ قېلىڭ: 90 پىرسەنىت تېخنىكا تەلەپ قىلىدىغان ئىشنى چەتئەللىك قىلىدۇ، يەرلىكلەر ئىشخانىدا ئولتۇرۇپ ۋاراڭ چۇرۇڭ قىلماي يۈشىچى ئوينىسا مەيلى..كەلمىسىمۇ بولاۋىرىدۇ، ھېچ ئىش قىلاممايدۇ..ئۇلارنىڭ كۆڭلىنى ئاغرىتىشقا تېخىمۇ بولمايدۇ، چۈنكى ئۇ ئەرەب! ئادەتتىكى ئەرەب ئەمەس بەلكى دۆلىتىنىڭ تېپىۋاتقان تىرلىيون پۇلىنىڭ ئۇدۇل سەئۇدى جەمەتىنىڭ يانجۇقىغا كىرىشىگە ماقۇل دەۋاتقان ئەرەب. ئەرەبىستاندا ئىشلىسە جىق پۇل تاپارمىش،،،راسمۇ؟ ناتايىن! سىز ئامىرىكىلىق، ياۋروپالىق بولسىڭىز ھەئە! ئەڭ جىق مۇئاشنى شۇلارغا بىرىدۇ، سىز ئاسىيالىق ياكى ئافرىقىلى، لىكىن ئامېرىلىكىدا ئوقۇغان،،سىزگىمۇ ئوخشاش مۇئاش بىرىلەمدۇ؟ نو!! سوررى!! خاپا بولماڭ...مۇئاشنى پاسپورتىڭىزغا قاراپ بىرىمىز دەيدۇ، سىز ئامېرىكىلىقتىن ئاشۇرۇپ ئىش قىلسىڭىزمۇ خەق 30000 كوي مۇئاش ئالسا سىز 7000 كوي ئېلىشىڭىز مۇمكىن..پەرىق راسىتنلا چوڭ! ئەرەبىستاندا ئىرقى كەمسىتىش بارمۇ؟ ئەپسۇس،،با!!! لىكىن ئافىرىقىلىقلارنى كەمسىتمەيدۇ،،،ئەمسە چىمنى؟ پاكىستانلىق، ھىندىستانلىق مۇسۇلمانلارنى...چۈنكى بۇلارنىڭ ئىشلەمچىلىرى بەك جىق ئۇ يەردە! ئاندىن، ئەرەبلەر ئاق تەنلىك، تىرىسى بەك ئاق! ياۋروپالىقتىنمۇ ئاق! پاكىستانلىق ، ھندىستانلىقلارنىڭ تىرىسى قوڭۇر بولغاچ سەت بىر گەپ بىلەن تىللايدۇ ئۇ يەردە..2 يىل بۇرۇن پادىشاھ تېخى بىر قانۇن ماقۇللىدى،،،دەپ بەرسەم ئىشەنمەي قالىسىز! 2013 تىن باشلاپ سەئۇدىلىقلار پاكىستانلىق خوتۇن ئېلىشقا بولمايدۇ! سەۋەبى ھازىر پاكىستانلىق خوتۇن ئالىدىغانلار جىقلاپ باللىرى يۇندا رەڭدە چۈشۈۋېتىپتىمىز...ئەرەبلەرنىڭ پاختىدەك ئاقلقىنى ساقلاش كىراككەن. شۇندىن كىيىن توردا جىق تالاش تارتىش بولدى لىكىن پاكىستان ھۆكۈمىتى تىنىپ قويمىدى..چۈنكى نۇرغۇن پاكىستانلىقلار ئەرەبىستاندا ئەرزان ئىشلەپ ئاران جان باقىدۇ،،بىر ئېغىز گەپ قىلىپ قويسا پادىشاھ ئاچچىقلاپ ھەممىسىنى ھەيدىۋەتمىسۇن دىدىمىكىن،،يەنە بىر يىرى سىياسى ھەمكارلىقىمۇ بار...شۇڭىمۇ دەيمىز...سەئۇدى دۇنىيادىكى ئاز ئۇچىرايدىغان قۇللۇق جەمىيەتنىڭ ئاسارىتى تىخىچە بار، لىكىن ياۋروپا بىلەن يېقىن ئۆتكەچ ھازىرغۇچە ھېچقانداق ئىش بولمىغان، 90 پىرسەنىت نۇپۇسى كارغا كەلمەس، ئىشلەپقىچىرىش كۈچى نۆلگە يېقىن ، تاسادىپى بىر ئىش بولسا پادىشا جەمەتى ياۋروپاغا تىكىۋىتىدۇ..پۇللىرى ئاساسەن ياۋروپا ۋە ئامېرىكىغا مەبلەخ سېلىخلىق ۋە ياكى بانكىدا. پادىشاھ جەمەتىنىڭ باللىرى كىچىكىدىن ئامېرىكىدا ئوقۇيدۇ..لىكىن دۆلەت ئىچىدە خەلىق نادان، نادان بولسا جەمەتكە تەھدىت يوق..لىكىن نىفىت تۈگىسە پادىشاھ جەمەت قاچىدۇ، پۇل با!!! خەلىق نىمە بولا!! ئەرەبلەر مۇشۇنىڭدىن قورقىدۇ ھازىر!
