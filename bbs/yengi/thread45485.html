  قەشقەردە نۇرغۇن تۇرمۇشتا ئاجىز قىرىنداشلىرىمنى كۆردۇم  ، يىڭى بازاردا بىر بوۋام سۇپەت كىشىنىڭ قىلىن ئاق جىلتكىگە يەڭنى ياماپ قويغان بىر چاپاننى كىيىپ ئولتۇرۇپ بىردىن بىر پۇلغا يارىغۇدەك نەچچە توخۇلىرىنى ساتقىنى ئەكەلگىنىنى كۆرۇپ ئىچىم ھەقىقەتەن سىيرىلدى، كىچىلىك بازار ، تاختا كۆۋرۈك ئەتراپىدا بىردىن بىر تىرىكچىلىك قۇرالى بولغان ئۇچ چاقلىق ھارۋىلىرىنى ئالامماي چارىسىز قالغان قىرىنداشلىرىمنى كۆرۇپ ھەقىقەتەن ئىچىم ئاغرىدى .،خىلى ياشقا بىرىپ قالسىمۇ جۇمە يەكشەنبە بازارلاردا چىلانلىرىنى ، زاغرىلىرىنى سىتىش ئۈچۈن سوغۇقتا مەزمۇت تۇرغان بۇۋا مومىلىرىمنى  كۆرۈپ ئىچىم ھەقىقەتەن  ..ئاچچىق بولدى....  قەھرىتاندا خۇددى قۇياشنى ئىتەك بىلەن توسقىلى ، ياپقىلى بولمىغاندەك ،  يوقسۇللۇقنىمۇ ياپقىلى بولمايدىكەن. ئىچىمدە ئۇيغۇر يوقسۇل قىرىنداشلىرىمىز ئەسلى قىشتا بىلىنىدىكەن دەپ ئويلىدىم دە ، بۇ ئويۇمنى ئانامغا ئىيتسام ئانام يوقسۇللۇق دىگەن ھەر ۋاقىت ھەر جايدا بىلىنىدۇ دىدى.
   ئاللاھ ھەق تائاللا بايلارنى ياراتتى ، كەمبەغەل ئاجىز يوقسۇلنى ياراتتى ، بايلارنىڭ بايلىقىدا يوقسۇللارنىڭ ھەققىنى قىلدى ، زاكات مۇسۇلمان دۇنياسىدىكى باي كەمبەغەللىك پەرقىنى ئوڭشايدىغان قۇرال ئىدى ، ھەزىرتى ئەلى قەيەردە كەمبەغەل يوقسۇل كىشىلەرنى كۆرسەڭ ، شۇ يەرنىڭ بايلىرى ، تورمۇشى ياخشى كىشىلەرنىڭ زاكاتنى بەرمىگەنلىكىنى بىلسەڭ بولىدۇ دىگەنىكەن ، مۇسۇلمانلار ھەم قۇرئان كەرىمدە قايتا قايتا تىلغا ئىلىنغان مال مۆلكىنى ئاللاھ رىزاسىنى ئىزدەپ سەرپ قىلىش دىگەن ئايىتىنى بەجانىدىل ئىجرا قىلمايدىكەن ، مۇسۇلمانلار دۇنياسىدا پاراۋانلىق مەڭگۇ ھەم مەڭگۇ روياپقا چىقمايدۇ،.ھەم شۇنىڭغا ئوخشاش مۇئمىن مۇئمىنگە نىسبەتەن ھەقىقەتەن بىنانىڭ قىشلىرىغا ئوخشايدۇ ، ئۇلار بىر بىرىنى چىڭداپ مۇستەھكەملەپ تۇرىدۇ ، مۇسۇلمانلار ھەقىقەتەن بىر بەدەنگە ئوخشايدۇ ، بىر ئەزا ئاغرىسا قالغان ئەزالارمۇ بىئارامسىزلىنىدۇ ، بۇ يەردە شۇنداق بىر قانۇنىيەت تەكىتلىنىدۇكى بىر مۇسۇلمان ئاچ قالسا يەنە بىر مۇسۇلماننىڭ ئىغىزىدىن ھەرگىز غىزا ئۆتمەيدۇ ، بىر مۇسۇلماننىڭ قىشتا كىيىدىغان قىلىن كىيىنى بولمىسا يەنە بىر مۇسۇلمان كۆندە بىردىن كىيىم ئالماشتۇرۇپ يۇرمەيدۇ ، ناۋادا ئۇ بىر قىرىندىشىنىڭ يوقسۇل ھالىتىنى كۆرۇپ تۇرۇپ ھىچقانداق ھىسداشلىق ، شۇنىڭغا شاپائەت قىلىش ياخشىلىق قىلىش پوزىتسىيەسىدە بولمايدىكەن ، ئۇ ھەقىقەتەن بىنانىڭ قىشى ،بولالمايدۇ، نەبىگە ئۇممەت ھەم بولالمايدۇ. شۇنداقلا بىز ھەرگىزمۇ ئۆز خوشنىمىزنىڭ ، ئۆز مەھەللىمىزدىكى بىر تونۇشىمىزنى يوقسۇللۇق ، خىراجەتسىزلىك سەۋەبلىك شىرىن چۇشتە كۆرىشىمىز قاملاشمايدۇ ، بۇنىڭ ھاجىتىنى راۋا قىلىش بىزدىكى باشتارتىپ بولالمايدىغان موسۇلمانچىلىق، ھەممىنى قولى ئوچۇق كارخانىچى قىرىنداشلىرىمىزغا ئىتتىرىپ قويىۋەرسەك بۇمۇ قاملاشماس.
   ھەقىقەتەن يوقسۇلغا بەرگىدەك ئىشىنغان خىراجىتىمىز بولمىسا   چىرايلىق كۆلۈمسىرەش ، ۋە ياكى بىر ياخشى سۆزمۇ ئۇنىڭغا قىلغان ياردىمىمىز بولىدۇ ، رەسۇلىمىز مۇئممىنگە چىرايلىق كۆلۈمسىرەشمۇ سەدىقە ئورنىدا دىگەن ، قىلغان ياخشى سۆزلىرىڭلارمۇ سەدىقە دىگەن ،   ئەقىللىق   ئەجدادلىرىمىز بۇغداي نىنىڭ بولمىسا ، بۇغداي سۆزۈڭ يوقمىدى دىگەن ئەمەسمىدى،  بەلكىم بىزنىڭ قىلغان ئازغىنە ياردىمىمىز بىلەن بىر يوقسۇل نامراتلىقتىن قۇتۇلالماسلىقى مومكىن ، يەنى ھىچ تىلغا ئالغىدەك ئىش قىلمىغان بولىشىمىز مومكىن  لىكىن بىزنىڭ ئىرىشىدىغىنىمىز ئاشۇ يوقسۇلنىڭ خەيىرلىك دۇئاسى، ئاللاھنىڭ سىلە رەھىمنى ئۆزۈپ قويماڭلار ، بىر بىرىڭلارغا ياخشىلىق قىلىڭلار ، شاپائەت قىلىڭلار دىگەن چاقىرقىغا ئاۋاز قوشۇپ رەببىمىز بىرىدىغان ئەجىر ، رەسۇلۇلاھنىڭ بىر مۇسۇلماننىڭ ئىغىرچىلىقىنى كۆتۈرۋەتكەن كىشىنىڭ ئاللاھ تائالا ئىنسان ئىنسانغا ياردەم قىلالمايدىغان كۆندە بىر ئىغىرچىلىقىنى كۆتۇرۋىتىدۇ دىگەن ھەدىسىگە ئەمەل قىلىپ ئىگە بولىدىغان كاتتا ساۋاپ ئەمەسمۇ؟ بەلكىم ھەر بىرىمىزنىڭ قىلغان زەررىچە ياخشىلىقىمىزغا ئاللاھ ئۆز پەزلى بىلەن بەرىكىتىنى چۇشۇرسە ، بىر ئۈزۈمنى قىرىق ئادەم يىگەندەك ، ئاشۇ ئۇيغۇر يوقسۇل قىرىندىشىمىز   يوقسۇللۇقنىڭ تۆمۇر كىشىنىنى  سىلكىنىپ  چېقىپ تاشلىسا بۇ ئالەمچە خۇشاللىق ئەمەسمۇ؟ ياخشىلىق چوڭ كىچىك دەپ ئايرىلمايدۇ ، ھەممىمىز قولىمىزدىن كەلگىنىنى قىلىپ ئەتراپىمىزدىكى ، مەھەللىمىزدىكى ، ئارقىدىن جەمئىيىتىمىزدىكى يوقسۇل قىرىنداشلارنى ئاشۇ زىيادە نامراتلىق كۆچىسىغا كىرىپ قىلىشىدىن ساقلايلى، ھەمدە ئۇلارغا  ياردەم قولىمىزنى سۇنايلى ، ئۆز قىرىندىشىمىزدىن، ئاللاھنىڭ ئاجىز يوقسۇل بەندىسىدىن ياردەمنى ئايىغىنىمىز ،ئاللاھنىڭ بىزدىن ياردەمنى ئايىشىغا سەۋەپ بولۇپ قالمىسۇن، رەسۇلىمىز ئاخىرەتتە ھەم قولىمىزدىن تۇتۇپ تارتىپ چىقىرىشنى خالىماي قالمىسۇن.

ناھايتى ياخشى تېما يوللاپسىز ! بۇنداق ئىشلار ئەتراپىمىزدا كۆپ . مۇھىمى مەن نىمە قىلدىم ، نىمە قىلالايمەن دەپ ئۈزىمىزگە سوئال قۇيۇپ باقايلى.
ئاساسەن ياشتا چوڭراق كىشىلىرىمىز تۇرمۇش بېسمىدا مەيلى قەھرىتان سوغۇق ياكى ئىسسىق بولسۇن باققاللىق ۋە باشقا ئۇششاق تىجارەتلەرنى قىلىدۇ . بىز ئۇلارنىڭ ئالدىدىن ئۈتۈپ تاللا بازارلىرىغا يۈگىرەيمىز.

ەيغەمبەر ئەلەيھىسسالام: زاكات ئىنسانلارنىڭ كىرىدۇر، دېگەن. (مۇسلىم)

سەدىقە ئاللاھنىڭ غەزىپىنى ئۆچۈرىدۇ ۋە يامان ئۆلۈمدىن ساقلايدۇ. (رەزىن)

ئەگەر بىر مۇسۇلمان ئاللاھتىن ساۋاب ئۈمىد قىلغان ھالدا ئائىلە ۋە بالا-چاقىسىغا خىراجەت قىلىپ بەرگەن بولسا، (قىلغان خىراجىتى)ئۇ كىشىنىڭ سەدىقىسى بولىدۇ. (بۇخارى)

يېرىم خورما(سەدىقە قىلىش)ئارقىلىق بولسىمۇ ئۆزەڭلارنى جەھەننەم ئوتىدىن قۇتقۇزۇڭلار. ئەگەر ئۇنىمۇ تاپالمىساڭلار ياخشى سۆز قىلىش ئارقىلىق جەھەننەم ئوتىدىن قوغدىنىشقا تىرىشىڭلار. (بۇخارى)

ئۈستىدىكى قول(بەرگۈچى قول) ئاستىدىكى قول(ئالغۇچى) قولدىن ياخشىدۇر. سەدىقە بېرىشنى ئالدى بىلەن ھالىدىن خەۋەر ئېلىش سېنىڭ مەجبۇرىيىتىڭ بولغانلاردىن باشلىغىن؛ سەدىقىنىڭ ئەڭ ياخشىسى ئېھتىياجدىن ئاشقاندا بېرىلگىنىدۇر. كىمكى(تىلەمچىلىك ۋە يامان ئىشلاردىن) ساقلىنىشنى ئارزۇ قىلسا، ئاللاھ ئۇ كىشىنى ئۇنداق ئىشلاردىن ساقلايدۇ. كىمكى ئىنسانلارنىڭ ياردىمىگە مۇھتاج بولماسلىقنى ئارزۇ قىلسا، ئاللاھ ئۇ كىشىنى ھېچكىمگە مۇھتاج قىلمايدۇ. (بۇخارى)
ەبۇ ھۈرەيرە رەزىيەللاھۇ ئەنھۇدىن رىۋايەت قىلىنىدۇكى، پەيغەمبەر ئەلەيھىسسالام مۇنداق دېگەن: پۇل-مالنىڭ كۆپ بولغىنى بايلىق ئەمەس، بەلكى نەپسنىڭ قانائەتچان بولغىنى بايلىقتۇر. (بۇخارى)

مال-مۈلك ۋە يارىتىلىشتا سىلەردىن ئۈستۈن كىشىلەرنى كۆرگەن ۋاقتىڭلاردا، ئۆزەڭلەردىن تۆۋەن كىشىلەرگە نەزەر سېلىڭلار. (بۇخارى)

تاماق يەپ شۈكرى قىلغۇچىنىڭ ئەجرى، روزا تۇتۇپ سەۋر قىلغۇچىنىڭ ئەجرى بىلەن ئوخشاشتۇر. (ئىبنى ماجە)
ئەجەل بەندىنى قانداق ئىزدىسە، رىزىقمۇ ئۇنى شۇنداق ئىزدەيدۇ. (بەززار)

ئەي تىجارەت ئەھلى! سودا دېگەندە يالغان گەپ، بىھۇدە قەسەم قىلىشلار بولۇپ تۇرىدۇ. شۇڭا پات-پات سەدىقە قىلىپ تۇرۇڭلار. (ئەبۇ داۋۇد ۋە نەسەئى)
