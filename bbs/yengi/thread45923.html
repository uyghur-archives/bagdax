ئامېرىكىنىڭ 90 مىڭ ئەسكىرى چاۋشىيەننى 56 كۈندە ھۇجۇم قىلىپ ئالالامدۇ

يەن چىياۋ

نۇرجاھان تەرجىمىسى



يېقىندا تورداشلار ئامېرىكا CNN ئىشلىگەن تېلېۋىزىيە پروگىراممىسىنىڭ كەسمە سۈرىتىنى ئاشكارىلىدى، بۇ پروگىرامما «شىمالىي كورىيەگە قارىتىلغان ئۇرۇش پىلانى»نى چۆرىدەپ مۇزاكىرە قىلغان، ئامېرىكا 90 مىڭ ئەسكەرنى ئىشقا سېلىپ 56 كۈندە شىمالىي كورىيە (چاۋشىيەن)نى ھۇجۇم قىلىپ ئېلىشنى پىلانلىغان.
يېقىندا شىمالىي كورىيە مەسىلىسى يەنە قىزىپ كەتتى، نۇرغۇن كىشىلەر بۇ قېتىم ئامېرىكا شىمالىي كورىيەنىڭ ئىغۋاگەرچىلىكىگە ئەستايىدىل تاقابىل تۇرىدۇ، دەپ قارىدى، ئۇرۇش قاش بىلەن كىرپىك ئارىسدا قالدى دەپ قارىدى. شۇنداق دېيىشكە بولىدۇكى، ئامېرىكا CNNنىڭ تېلېۋىزىيە پروگىراممىسىدا «شىمالىي كورىيەگە قارىتىلغان ئۇرۇش قىلىش پىلانى» پەقەت ھەربىي كۆڭۈل ئېچىش خالاس، خۇددى تورداشلارنىڭ ئۆز ئارا مۇنازىرە قىلىشقىنىغا ئوخشايدۇ، رۇخسەت قىلىنغان دائىرىگە تەۋە، ئەمما ئۇنى ئامېرىكا ئارمىيەسىنىڭ ھەقىقىي ئۇرۇش قىلىش پىلانى قىلىش پۈتۈنلەي خاتا.

ئالدى بىلەن ئامېرىكا شىمالىي كورىيەگە قاراتقان ئۇرۇش قىلىش پىلانىنى مۇددەتتىن بۇرۇن ئاشكارىلىغۇدەك ئۇنداق ئەخمەق ئەمەس.
شىمالىي كورىيەگە قارىتىلغان بۇ ئۇرۇش پىلانى دېگەن نەرسە بۇرۇنلا بار ئىدى، ئامېرىكىغا قارشى تۇرۇپ شىمالىي كورىيەگە ياردەم بېرىش ئۇرۇشى ئاخىرلاشقان ئالدىنقى ئەسىرنىڭ 50-يىللىرىدىن بۈگۈنگە قەدەر توختاپ قالغىنى يوق. چۈنكى ئۇرۇش گەرچە بەزىدە سىياسىي كىلماتنىڭ ئۆزگىرىشى سەۋەبلىك ئۇنچە زۆرۈر بولمىسىمۇ، ئەمما زۆرۈر ئىستراتېگىيەلىك مۇھىم نۇقتىغا بولغان ھەربىي زەربە بېرىش پىلانىدىن ئەزەلدىن ۋاز كەچمەيدۇ ھەمدە ھەربىي زەربە بېرىش تېخنىكىسىنىڭ يۇقىرى كۆتۈرۈلىشى سەۋەبلىك ھەربىي پىلانىنى ھەرۋاقىت ئۆزگەرتىدۇ، بۇلارنىڭ ھەممىسى ئادەتتىكى ھەربىي بىلىملەر، شۇڭا ئامېرىكىنىڭ شىمالىي كورىيەگە ھەربىي زەربە بېرىش پىلانى بار بولۇپلا قالماستىن، چوقۇم نەچچە خىلى، ئاددىي ھالدا ئومۇميۈزلۈك زەربە بېرىش ۋە قىسمەن تاقابىل تۇرۇش دەپ ئىككى خىلغا يىغىنچاقلاشقا بولىدۇ.


ئومۇميۈزلۈك زەربە بېرىش چوقۇملا شىمالىي كورىيەنىڭ ھازىرقى ھاكىمىيىتىنى ئاغدۇرۇپ تاشلاشنى ھەربىي نىشان قىلىدۇ، لېكىن قىسمەن تاقابىل تۇرۇش بولسا بىر قىسىم تۇيۇقسىز يۈز بەرگەن ۋەقە ياكى شىمالىي كورىيەگە بىر قىسىم قاراتمىلىق، تاشقى ئوپېراتسىيەگە ئوخشاش ھەربىي زەربە بېرىشتۇر، بۇلار خەلقئارا ۋەزىيەت ۋە ھەرخىل ئەھۋالللارغا ئاساسەن بېكىتىلىدۇ، ئۆزگەرمەس نەرسە ئەمەس.
شىمالىي كورىيەگە قارىتىلغان ئومۇميۈزلۈك زەربە بېرىش پىلانى چوقۇملا ئامېرىكا ئارمىيەسىنىڭ مۇستەقىل ھەرىكىتى ئەمەس، ئەگەر ئامېرىكا خەلقئارا ۋەزىيەتكە ھۆكۈم قىلىپ شىمالىي كورىيەگە قول سالسا، جۇڭگو ۋە رۇسىيەنىڭ ھەيۋىسىنىڭ تەلتۆكۈس سۇنىدىغانلىقىنى چۈشەندۈرۈپ بېرىدۇ.

سوغۇق ئۇرۇش ئاخىرلاشقاندىن بۇيان، ئامېرىكا رۇسىيە ۋە جۇڭگونى ئىستراتېگىيەلىك قورشاشتىن، ئىستراتېگىيەلىك بوشلۇقىنى قىسىشتىن ئەزەلدىن ۋاز كېچىپ باققىنى يوق. نۆۋەتتە ئىران يادرو مەسىلىسى ئاللىقاچان ھەل بولدى، ئامېرىكا ئېھتىمال بىر دۈشمىنىم ئازلىدى دەپ قارىشى مۇمكىن. ئۇكرائىنا مەسىلىسى يەنىلا رۇسىيەنى چەكلەپ تۇرماقتا، بولۇپمۇ رۇسىيە سۇرىيەگە ئەسكەر كىرگۈزۈشىنى ئامېرىكا ئېھتىمال پۇتىن ۋاقتىنچە دىققىتىنى بۆلۈپ شىمالىي كورىيەگە يېتىشىپ بولالمايدۇ، دەپ قارىشى مۇمكىن. جۇڭگوغا قارىتا ئامېرىكا ئېھتىمال ئېنىق كۆرۈپ يەتتىم دەپ قارىشى مۇمكىن، جۇڭگونىڭ ئىزچىل تىنچ قەد كۆتۈرۈشكە ئەمەل قىلىدىغانلىقىنى، شىمالىي كورىيەگە ھۇجۇم قىلغان تەقدىردىمۇ جۇڭگو ئىككىنچى قېتىم شىمالىي كورىيەگە ياردەم بېرىپ ئامېرىكىغا قارشى تۇرۇشى مۇمكىن ئەمەسلىكىنى بىلىدۇ، شۇڭا بۇ ئامېرىكىنىڭ شىمالىي كورىيەگە زەربە بېرىشكە تەييارلىنىشىدىكى ئاساسىي ھۆكۈمى. ئامېرىكا 3-ئايدا جەنۇبىي كورىيە بىلەن ئۆتكۈزىدىغان بىرلەشمە ھەربىي مانېۋىر پۇرسىتىدىن پايدىلىنىپ ئامېرىكا ئارمىيەسىنىڭ قاتنىشىش سالمىقىنى زورايتماقچى، بىرلا ۋاقىتتا بۇ قېتىمقى پۇرسەتتىن پايدىلىنىپ ھەربىي مانېۋىر ۋاقتىدا شىمالىي كورىيەگە داۋاملىق ئىغۋاگەرچىلىك قىلغاندىن كېيىن راست تاقابىل تۇرۇش تەييارلىقىنى قىلماقچى. شۇنداق دېيىشكە بولىدۇكى، ئامېرىكا ئەگەر شىمالىي كورىيەگە ئۇرۇش قوزغاشقا تەييارلانسا ئۇرۇشماقچى بولسىلا ئۇرىشىدۇ، ئاددىي ھالدىلا شىمالىي كورىيەنىڭ يادرو ئەسلىھەسىگە زەربە بېرىش بىلەنلا بولدى قىلمايدۇ ھەمدە شىمالىي كورىيە بىلەن مۇنداقلا ئېتىشىپ قويۇشىنىڭ ئۆزىلا يېرىم ئارالدىكى پورۇخ تۇڭىنىڭ پارتلايدىغانلىقىدىن دېرەك بېرىدۇ، ھېچكىم كونترول قىلالمايدۇ.

راستنى ئېيتقاندا، ھازىرقى تېخنىكا شارائىتىدىكى ئۇرۇشتا، شىمالىي كورىيەگە قارىتا ئۇرۇشقا ئامېرىكا ئارمىيەسى 90 مىڭ ئەسكىرىنى ئىشقا سالسا يېتەرلىك، بىرىنچىدىن ئامېرىكا ئارمىيەسى چوقۇملا جەنۇبىي كورىيە ئارمىيەسىگە ئاساسلىق تايىنىدۇ، ئامېرىكا پەقەت ھاۋادىن زەربە بېرىش ئارقىلىق جەنۇبىي كورىيەگە ياردەم بەرسىلا بولىدۇ. 56 كۈندە شىمالىي كورىيەنى ھۇجۇم قىلىپ ئېلىش ئالالماسلىقتا، مەن ۋاقىت بەك ئۇزۇن بوپ كەتتى دەپ قارايمەن، 30 كۈنمۇ كەتمەيدۇ، شىمالىي كورىيەدەك بۇنداق سىرتىدىن قاتتىق قول كۆرۈنىدىغان ھاكىمىيەتتە ساختا تەركىب بەك كۆپ، ھازىرقى زامان ئۇرۇشىدىكى تېخنىكا ۋاستىسىدا ئامېرىكىنىڭ ئامالى نۇرغۇن، پەقەت شىمالىي كورىيەنىڭ ئالاقە ئەسلىھەسى، رادار ئەسلىھەسى قاتارلىق مۇھىم ھەربىي ئەسلىھەلىرىنى بىر كۈن ئىچىدە بومباردىمان قىلىۋەتسە، شىمالىي كورىيە ئاساسىي جەھەتتىن مالىمانچىلىققا پېتىپ قالىدۇ. شىمالىي كورىيە سېئۇلنى دەرھال ئوت دېڭىزىغا ئايلاندۇرۇۋېتىدۇ دېيىش ئۆزىنى ماختاپ ئۇچۇرغىنىدۇر، ئەگەر ئامېرىكا بىلەن جەنۇبىي كورىيە شىمالىي كورىيەنى يوقاتماقچى بولسا چوقۇم ئالدىن قول سالىدۇ، ئالدى بىلەن شىمالىي كورىيەنىڭ 38-پارالېل تەرىپىدىكى بارلىق زەمبرەك ئىستىھكاملىرىغا ئالدىن قول سېلىپ زەربە بېرىدۇ، شىمالىي كورىيەنى ۋەيران قىلغۇچ تويۇنما بومباردىمان قىلىش كۆپ قىسىم ئۈنۈملۈك كۈچىنى ۋەيران قىلىدۇ، قالغىنىغا يەر يۈزى قىسىمى ئەۋەتىپ پارتىزانلارنى بىر تەرەپ قىلىۋەتسىلا بولىدۇ.

كورىيە ئۇرۇشى زادى پارتلامدۇ يوق دېگەندە، مېنىڭچە ئەسلا مۇمكىن ئەمەس، بۇنىڭدا ئەڭ مۇھىم بىر نۇقتا بولسا خەلقئارا ۋەزىيەت تېخىمۇ ئۇ دەرىجىدە تەرەققىي قىلمىدى، شىمالىي كورىيە ئەمەلىيەتتە چوڭ دۆلەتلەر ئويۇنىدىكى پېچكا، جۇڭگو ۋە رۇسىيە تېخى زاۋال تاپمىدى، ھەتتا بەزى جەھەتلەردە كۈچەيمەكتە. راستنى ئېيتقاندا ئامېرىكىغىمۇ ئېنىق، ئەگەر جۇڭگو ۋە رۇسىيەنى بۇلۇڭغا قىستاپ قويسا مۇقەررەر قايتۇرما ھۇجۇم قىلىدۇ، جۇڭگو گەرچە تىنچلىققا «ئۆز ئارزۇسىنى ئاساس قىلسىمۇ»، ئەمما راستىنلا جۇڭگونى قىستىسا ئاچچىقى كەلگەن توشقانمۇ ئادەم چىشلەيدۇ، ھازىر ئامېرىكا جەنۇبىي جۇڭگو دېڭىزىدا جۇڭگوغا بېسىم قىلماقتا، ئەگەر يەنە شىمالىي كورىيەگە ھۇجۇم قىلسا جۇڭگو تېخىمۇ پاسسىپ ھالغا چۈشۈپ قالىدۇ. رۇسىيە سۇرىيەدە سەئۇدى ئەرەبىستان ۋە تۈركىيەنىڭ بېسىمىغا ئۇچرىماقتا، چوقۇملا ئامېرىكا ئارقىسىدا قۇتراتماقتا، مەقسىتى رۇسىيەنى چەكلەشتۇر، ئەمما بۇلارنىڭ ھەممىسى پەقەتلا ئىستراتېگىيەلىك قورشاۋدۇر، جۇڭگو ۋە رۇسىيەنىڭ ھازىرقى ئامالى بولسا ئېغىر-بېسىق بولۇشتۇر، چۈنكى مەيلى شىمالىي كورىيە تەرەپتە بولسۇن ياكى سۇرىيە تەرەپتە بولسۇن دۇنيا ئۇرۇشى پارتلىشى مۇمكىن، بۇ ئىككى رايون چوڭ دۆلەتلەر ئويۇنىنىڭ قىزىق نۇقتىسى، بۇ خىل تىركىشىشنىڭ ئۇزۇن مەزگىللىك بولىدىغانلىقىغا، قىسقا مەزگىل ئىچىدە ھەل بولۇشى مۇمكىن ئەمەسلىكىگە ئىشىنىمەن.
