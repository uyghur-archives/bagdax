ئەسسالامۇ ئەلەيكۇم مۇنبەر باشقۇر غۇچى ۋە باشقا رەھبەرلەر ھەم تۇرداش قېرىنداشلار بۇدىلونىڭ داۋامىنىڭ ئىلگىرلەش نەتىجىسىنى بىلىشىنى ئارزۇ قىلغانلىقىڭلار ئۈچۈن (سوز تاۋلاشتا يىتەرسىزلىك  ، رېقىپ تاشقى كىسەللىكلەر  دوختۇر خانىسى سەھيەنىڭ تەستىقى بىلەن قۇرۇلغان دوختۇر خانا تۇرۇپ، قانۇنسىز دوختۇرلۇق قىلغان دەپ باھالاندى ؟نىمە ئۈچۈن جاۋاپكار تەرەپلەر، دەۋاگەر نىڭ مىڭىسى نۇرمال ئىشلەيدۇ شۇنىڭغا قارغاندا، ساق ،دىيىشىدىكى ئاساسى بارمۇ؟ دەپ سورغىنىمنىڭ جاۋابى چۈشۈپ قالغانلىق ئۈچۈن تەھىرلەپ  بۇتىمىنى يۇللاپ قويدۇم، شۇنىڭ ئۈچۈن تەستىقلاپ بىرشىڭلارنى شۇنداقلا ھەققانى ئىنكاس قايتۇرۇپ بىرشىڭلارنى چىن كۆڭلۈمدىن ئۈمۈت قىلىمەن، ھەممىڭلارنىڭ مۇشۇ ۋاختى خەيرىلىك ئۇتۇقلۇق بۇلىشىغا تىلەكداشمەن.رەخمەت ئەۋارە قىلىپ سالدىم سىلەرنى

ئەرزىمنىڭ 129-قېتىملىقى باشتىن ئاخرى جىنايى ئىشلار قۇشۇمچە ھەقتەلەپ ئەرزى
ئاپتۇنۇم رايۇنلۇق يۈقۇرى خەلىق سوت مەھكىمىسگە نارازىلىق ئەرىزى بولدى   
   مەن 1-سوتتىكى نارازى بولغۇچى دەۋاگەر: ئۇبۇلھەسەن ئابدۇكىرەم ئەر ئۇيغۇر 1970يىلى تۇغۇلغان.باشلانغۇچ مەدىنيەت سەۋىسىگە ئىگە، خوتەن، شەھەر ،لاسكۈي بازار، ئاقتاش كەنتىدىن ،ھازىر بۇداۋا تۈپەيلىدىن 5 يىل داۋامىدا ئۈرۈمچى شەھەر جىدىي مىڭخۋا 12-كۈچا 37-قورو 301-نۇمۇرلۇق ئۆيدە ئولتۇرۋاتىمەن،ئەھۋالىم كەينى –كەينىدىن  زۇلۇمغا ئۇچراپ مېيىپ بولۇپ باشتىن ئاخىرى ئالدىنىپ، قانۇنى ھوقۇقۇمنى قوغدىيالماي ئىقتىساتتىن  ئايرىلغان ئۇبۇلھەسەن ئابدۇكىرەم باشچىلىقىداھەرجەھەتتىن خارلانغان بىرئائىلە كىشىلىرى بۇلىمىز .ئەسلىدە بىر يەككە تىجارەتچى ئىدىم. تېلىفۇن: 13150498983 -18999896351
ئاپتۇنۇم رايۇنلۇق ئەدىلىيەنىڭ ئۇرۇنلاشتۇرشىغا ئاساسەن ئۈرۈمچى ئەدىلىيەنىڭ قانۇننى ياردەم ھاۋالىسى بىلەن ، شىنجاڭ جۇدىڭ ئەدىلىلىك باھالاش ئورنىنىڭ 2016-يىلى 1-ئايدا چىقارغان 430-نۇمۇرلۇق باھالاشنىڭ يەكۇنىدە، زودۇڭ ئەدىلىيە باھالاش ئورنى : جاۋاپكارلارغا  جاۋاپكارلىقنى بىكىتكەن مەزمۇنىدا ئېغىر زەخمىنىڭ ئىچىدىكى پىرسەنتىنى %75پىرسەن %10پىرسەن %15پىرسەن دەپ ئايرىم –ئايرىم ھالدا جاۋاپكارلىقنى بىكىتتى، شۇنىڭ ئۈچۈن ئەينى ۋاقىتتا نارازىلىق بۈلدۈرگەن ئەرزىم ئالاھىدە تۈزۈتتۈرۈش كۈرگۈزىلدى شۇڭا بۇئەرزىمنى قۇبۇل قىلىپ ئەسلى ئەرزىمنى قايتۇرۇپ بىرشىڭلارنى تەلەپ قىلىمەن
  كەينى –كەينىدىن زەنجىر سىمان  ئەگىشىپ بىزگە زىيان سالغان تەرتىپ چاتما ( جاۋاپكارلانىڭ )  تۇنۇشتۇرۇش سالاھىتى ئىسمى تۈۋەندىكىچە
