   ئەسسالامۇ ئەلەيكۇم  ، ئەزىزلەر ، ھەممىڭلارغا خۇشاللىق يار بولسۇن .
    بۇ تېمامدا مەدەنىيەتلىك ئادەم زادى قانداق بولىدۇ  دېگەن مەزمۇندا قىسقىچە توختىلىمەن .  
    ھەممىمىز بىلىمىز ، مەدەنىيەت بولسا ئىقتىساد ، سىياسەت قاتارلىقلار بىلەن بىر قاتاردا تۇرىدىغان ئەڭ  مۇھىم ئۈچ  ئۇقۇمنىڭ بىرى ، شۇنداقلا چوڭىدىن تارتىپ دۆلەتتىن كىچىكىدىن تارتىپ شەخىسكىچە باشقىلار تەرىپىدىن قويۇلىدىغان باھالارنىڭ بىرى . بىز ھەممىمىز مىللىتىمىزنى  قەدىمدىن تارتىپ ئېسىل مەدەنىيەتلەرنى ياراتقان دەپ پەخىرلىنىمىز ، مەنمۇ شۇنداق . ئەمما ، بىز مۇشۇ پەخىرلىنىش ھېسسىياتىمىزدا تۇرۇۋاتقان چېغىمىزدا ئۆزىمىز ھەققىدە ئويلىنىپ باقتۇقمۇ  ، بىز ھازىر مۇشۇ ھالىتىمىز بىلەن قانچىلىك مەدەنىيەتلىك ؟
    يەر شارى كىچىككىنە بىر كەنىتكە ئوخشاپ قالغان بۈگۈنكى كۈندە ، پۈتۈن دۇنيا مىقياسىدا مەدەنىيەت ئالمىشىش ھادىسىسىناھايىتى كۆپ بولۇۋاتىدۇ ، بىزمۇ يېقىن زامانلاردا نۇرغۇن يېڭىچە مەدەنىيەتلەرنى قوبۇل قىلدۇق .   ئەلۋەتتە مەدەنىيەتكە بېرىلگەن تەبىرلەر بىر بىرىگە ئوخشىمايدىغان بولغاچقا زادى كىم مەدەنىيەتلىك ، كىم قالاق دىگەنلەرنى ئايرىشمۇ قىيىن بىر  مەسىلىگە ئايلىنىپ قالدى . نۇرغۇن تالاش تارتىشلار بولدىيۇ ئەمما ھېچكىم ھېچكىمنى قايىل قىلالمىدى . مەنمۇ دائىم بۇ تېما توغرىسىدا ئويلىنىدىغان بولغاچقا ، ئۆزۈمنىڭ كۆز قاراشلىرىمنى ئورتاقلىشىش ئۈچۈن  بۇ كىچىك تېمىنى يېزىپ ئولتۇرۇپتىمەن .
    ھازىرقى  ياشلىرىمىزنىڭ  كاللىسىدا بۇ توغرىسىدا ئويلىنىش كەمچىل ، چۈنكى ھېچكىم ئۆزىنى قالاق دىيىشنى ياقتۇرمايدۇ ئەمەسمۇ . نۇرغۇن ئوغۇللىرىمىز ئۆزىنىڭ تاماكا چىكەلىشىنى ۋە ھاراق ئىچەلەيدىغانلىقىنى مەدەنىيەت ساناپ ئۆزىنى ناھايىتى مەدەنىيەتلىك دەيدۇ ، ئىچمەيدىغان ، چەكمەيدىغانلارنى بولسا قالاق دەپ ئارىغا ئالمايدۇ ؛ نۇرغۇن قىزلىرىمىز ئۆزىنىڭ ھەددىدىن قىسقا ھەم تەر كىيىم كىيەلىشىنى مەدەنىيەت دەپ قارايدۇ ، كىيىمنى ئۇزۇن ، كەڭراق كەيگەنلەرنى بولسا مەتۇ ، زاماننىڭ ئارقىدا قالغان دەپ تىللايدۇ . راستلا شۇنداقمۇ  ؟  ؟  ؟ مەن ھەرگىز مۇنداق قارىمايمەن ، بۇ خىل كۆز قاراش ئىنتايىن نادانلىق بىلەن چىقىرىلغان دەپ قارايمەن  . ئۇنداق بولسا مېنىڭ ئويلىشىمدىكى مەدەنىيەتلىك ئادەم زادى قانداق ئادەم ؟
     بۇ سوئالغا ئاددىي قىلىپ جاۋاب بىرىدىغان بولساق، مەدەنىيەتلىك ئادەم ، باتىل ئەقىدە ، قۇرۇق نەزەرىيەلەرگە قارىغۇلارچە ئەگىشىپ شاپاشلىغان كىشى ئەمەس بەلكى ھەق دىننى قوبۇل قىلىپ ئاللاھنى ۋە پەيغەمبىرىنى تۇنۇغان كىشىدۇر .
   مەدەنىيەتلىك ئادەم  ، ھەددىدىن ئېشىپ پۈتۈن ئىنسانىيەت بىر يەرگە كىلىپمۇ بىر ئايىتىنى مەيدانغا كەلتۈرۈشكە ئاجىزلىق قىلىدىغان مۇقەددەس كىتاب قۇرئان كەرىمگە تەتۈر قاراپ يوغانچىلىق قىلىدىغان ئەمەس ، ھەقىقەتكە بوي ئېگىپ ئاجىزلىقىنى ئېتىراپ قىلغان ھالدا قۇرئانغا ئەگىشىپ ، مەنىسىنى چۈشىنىپ ئەمەل قىلغان كىشىدۇر .
   مەدەنىيەتلىك ئادەم،  مىللىي بىرلىكىدىن ، مۇقەددەس ئەقىدىسىدىن ، پۈتۈنلەي ۋاز كېچىپ ئۆزىنى ئىنكار قىلغان كىشى ئەمەس ، ۋەتەننىڭ تەقدىرى ، مىللەتنىڭ ئىستىقبالىغا ئىزچىل كۆڭۈل بۆلگەن ، قوبۇل قىلىش جەھەتتىكى پايدىلىق ۋە پايدىسىز تەرەپلەرنى ئەستايىدىل پەرىقلەندۈرۈپ ، <مېغىزىنى ئېلىپ شاكىلىنى چىقىرىۋېتىش > پىرىنسىپلىرى بويىچە مىللەتنىڭ ۋۇجۇدىدىكى روھىي ئىللەتلەرنىڭ تۈپ يىلتىزىنى قۇرۇتۇپ تاشلىيالايدىغان ، مىللىتىنىڭ مۇقەددەساتلىرىنى جان تىكىپ قوغدىيالايدىغان كىشىدۇر .                                                                                                                                             
   مەدەنىيەتلىك ئادەم ، بازارلاردا ھايا قىلماستىن ئاياللىرى بىلەن قول تۇتۇشۇپ يۈرگەن ، تانسىخانىلاردا ئۆزى باشقا ئاياللار بىلەن تانسا ئوينىغان ، ئايالىنى باشقا ئەر بىلەن  تانسىغا قويۇپ بەرگەن كىشى ئەمەس ، مۇسۇلمانلارچە ئەخلاق ئۆلچىمى ۋە ھەرىكەت مىزانى بويىچە ئاللاھتىن قورقۇپ ھارامدىن  نومۇسىنى ساقلىغان ئىخلاسمەن ، تەقۋادار ، سەۋرچان ، سەمىمىي كىشىدۇر .  
   مەدەنىيەتلىك ئادەم ، زالىمنىڭ چوماقچىسى ، نامەرتنىڭ يالاقچىسى بولغان كىشى ئەمەس ، ئۆزىنىڭ ئىنسانىي قەدىر -قىممىتىنى تونۇپ ھەقىقەتتە تۇرغان ، ھەقسىزلىككە باش ئەگمىگەن ئالىيجاناب ، باتۇر ، بەخىتسىزلىك ئىچىدىن بەخىت ئىزدەيدىغان كۈرەشچان كىشىدۇر .
    قىسقىسى ، مەدەنىيەتلىك ئادەم ، ئەقىل پاراسىتىنى ئىشقا سېلىپ ئۆزىنىڭ دۇنياغا كىلىش غايىسىنىڭ نېمە ئىكەنلىكىنى تونۇپ يەتكەن ۋە شۇ غايىسىنى رېئاللىققا ئايلاندۇرۇش ئۈچۈن تىرىشقان ئادەمدۇر .
  يۇقارقىلار مېنىڭ مەدەنىيەتلىك ئادەم توغرىسىدا ئويلىغانلىرىم ، ئەلۋەتتە ھەر بىرىمىزنىڭ ئويلىغانلىرىمىز ، كۆز قاراشلىرىمىز ئوخشىمايدۇ ، بۇ توغرىدا پاراڭلىشىشنى خالايدىغانلار بولسا پىكىر ئالماشتۇرۇشنى قىزغىن قارشى ئالىمەن .

بۇ تىمىڭىز سوبىكتىۋ كۆز قاراشقا تويۇنغان، نەزەريىۋى ئاساسى ئاساسى جەھەتتىن بولمىغان قۇرۇق پاراڭ دۆۋىسى بولۇپ قاپتۇ.
1.ئالدى بىلەن بىر تىمىنى ئوتتۇرىغا تاشلىغاندا، مۇنازىرە تىمىسى بولغان يادرولۇق ئۇقۇمنىڭ ھەممە ئادەم ئورتاق قوبۇل قىلالايدىغان تېرمىنىنى ئاۋال ئوتتۇرىغا قويشىڭىز، ھىچ بولمىغاندا ئىتراپ قىلشىڭىز كىرەك.
2. سىز مۇلاھىزە قىلماقچى بولغان سىلىشتۇرما ئوبىكىتلارنىڭ دائىرىسىنى ئاۋال ئەسكەرتىشىڭىز كىرەك. شۇ چاغدا سىز زادى مەھەللىۋى پاراڭ سالماقچىمۇ، ياكى ئىلمى يەكۈننى ئىلگىرى سۈرمەكچىمۇ، ئايدىڭ بولۇپ كەتسۇن.
3. پىكىرىڭىزنىڭ دەلىلى ئاساسلىرى ۋە لوگىكىلىق يەكۈنلىرىڭىزنى 1، 2، 3-دەپ سىستىمىلىق ئانالىز شەكلىدە ئوتتۇرىغا قويشىڭىز، ھىچ بولمىسا توپلۇم، قاتلام، ياكى زەنجىرسىمان مۇناسىۋەتلەر ئەكىس ئەتكەن گىرەف ياكى جەدۋەللەر بىلەن پىكىر ئانالىزلىرىڭىزنى مۇكەممەل ئەكىس ئەتتۈرۈشكە تىرىشىشىڭىز كىرەك.
4. ھەممە ئادەم ھەرخىل قارايدىغان بىر مەسىلىگە، بىر جۈملە ئاددى سۆز بىلەن جاۋاب بىرىپسىز، شۇڭا توغرىسى، شۇل نۇقتىنەزىرىڭىزنى «مىنىڭ چۈشەنچەمچە» دەپ ئىلىشىڭىز كىرەك.
5. تىمىڭىز «بىز»نىڭ (زادى كىمنى دىگەنلىكىڭىز ئىنىق ئەمەس) خارەكتىرى بەك پىروتوتايىپلاشتۇرۇۋىتىلگەن (تىپى سوبىكتىپ قىلىپلاشتۇرۇۋىتىلگەن). بۇ دىگەنلىرىڭىزنى ئۇيغۇرلارغا قارتىلغانمىكىن، دىسە، ئون نەچچە مىليون ئۇيغۇرلارنىڭ ھەممىسىنىڭ سىز دىگەندەك ئويلىنىشى، سىز دىگەندەك كۆز قاراشتا بولۇشى، ياشلارنىڭ ھەممىسىنىڭ سىز دىگەندەك بولۇشى ناتايىن. شۇڭا سىزنىڭ مۇلاھىزە قىلۋاتقان ئوبىكتىڭىزمۇ خىيالى بولۇپ قالغان.
6. تىمىڭىز خىيالى بىر توپنىڭ مەدەنيەتلىشىش دەرىجىسىدىن ئىبارەت بىر ئىنتايىن يۇقىرى ئۇنۋىرسال بىلىم تەلەپ قىلدىغان مۇنازىرىدىن باشلىنىپ، بىر سىزنىڭ سوبىكتىپ قىممەت قارشىڭىزدىكى «مەدەنيەتلىك ئادەم»نىڭ بىرنەچچە خۇسۇسيىتىنى ئوتتۇرىغا قويۇش بىلەن خۇلاسىلەنگەن. يىغىپ ئىيىتقاندا باش-ئاخىرى ماسلاشمىغان.
گەپنى قىسقارىتقاندا، بىر كىشىنىڭ باشقىلارنى كۈزىتىش جەريانىدا چىقارغان سوبىكتىۋ يەكۈنى، تەنقىد ئوبىكتى بولغان باشقىلاردىن، بەكرەك شۇ كىشىنىڭ ئۆزىنىڭ تەپەككۇر ئۇسۇلى ۋە ئىچكى دۇنياسىنى ئەكىس ئەتتۈرۈپ بىرىدۇ. سىزنىڭ بۇ «يازما»ڭىزنىڭ بەكرەك ئەكىس ئەتتۈرگىنى بىر توپ تەتۇر، قىتىپ كەتكەن كاللىغا ئىگە بولغان، تەپەككۇرسىز قارسىغا ئەگىشىدىغان كىشىلەر توپى، كاللىسىدا ئويلىنىش كەمچىل بولغان، ئازراقلا تىيىز چۈشەنچىگە ئىگە، باشقىلار بىلىدىغان نەرسىنى بىلمەيدىغان، ئەقىلسىز ئىش قىلىپ، ئەقىلسىز پاراڭ سالىدىغان جاھىل ياشلار... ئوبدانراق كاللىڭىزنى سىلكىۋىتىڭ!
