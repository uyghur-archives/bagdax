بىرەر سوئالغا جاۋاپ بىرەلمىسە ،ياكى جاۋابى تولۇق بولمىسا ئۆرە تۇرغۇزۇپ قويدىغان،ياكى باشقا ئىشلارغا بۇيرىيدىغان ،ئوقۇغۇچى جاۋاپ بىرىۋاتقاندا خاتا بولۇپ قالسا ئوقۇغۇچىنىڭ گىپىنى توختىتىۋىتىدىغان ئىشلار ئوقۇغۇچىغا بەك ئېغىر كىلىدۇ ،
يەنە تاپشۇرۇقلارمۇ بەك كۆپ بىر كۇندە ئاز دىگەندە 4 قېتىملىق تاپشۇرۇق ئىشلەيمىز ،ھەر بىرىنى ئىشلەشكە 20 مىنۇتتىن ۋاقىت كەتسىمۇ 80 مىنۇت ،
تەخمىنەن 40مىنۇت ئالدىن دەرىس كۆرمىز،تەخمىنەن 30 مىنۇت دەرىس تەكىرارلايمىز ،بۇ ئىشلارغا تەخمىنەن ئىككى يېرىم سائەت ۋاقىت كىتىدۇ ،
ئۇندىن باشقا 8 سائەت دەرىس ،تازىلىق،تاماقلىنىشقاتارلىقلارغىمۇ ۋاقىت كىتىدۇ،
بەزىدە ئەتىگەنلىك تاماقنى 1 سائەتلىك دەرىستىن كىيىنكى ئارامدا يەيمىز ،
يەنە مەشىق ئىشلەشلەرمۇ بار ،ئوقۇتقۇچىلار مۇشۇلارنىمۇ ئويلاپ قويسا
بىرەر سوئالغا جاۋاپ بىرەلمىسەك ياكى تولۇق جاۋاپ بىرەلمىسەك بىزنى خىجىل قىلدۇرۇپ ساۋاقداشلارنىڭ ئالدىدا كۆڭلىمىزنى رەنجىتمىسە ،مانابۇ بىر ئوقۇغۇچىنىڭ يۇرەك سۆزى

بۇ كىتاپنى سىزگە تەۋسىيە قىلاي. بۇ ئوقۇتقۇچىلا ئەمەس، باشلىق، ساقچى، سوتچى، جىنايەتچى، ئوقۇغۇچى، ئۇرىشىدىغان ئەر- ئاياللار... ئىشقىلىپ ھەرقانداق ئادەمگە ئالاقە قوللانمىسى بولالىغۇدەك بىر كىتاپ. بۇنىڭ ئاپتورى مارشال روزېنبىرگ ئۆزى شەكىللەندۈرگەن «Nonviolent Communication -چىرايلىق مۇئامىلە»پەلسەپەسى بىلەن دۇنياغا تونۇلغان پىسخولوگ. كىتابىمۇ نۇرغۇن تىللارغا تەرجىمە قىلىنغان. ئەگەر خەنزۇچىسىغا كۈچىڭىز يەتسە مەنچە مۇشۇنى ئوقۇڭ.  ئۇيغۇرچىسى بارمۇ- يوق بىلمىدىم. ئەمما بىرسى تەرجىمە قىلىۋاتىدۇ. بەلكى بىرەر يىلدىن كېيىن چىقىپ قېلىشى مۇمكىن.  
بۇ ئادەمنىڭ كىتابىدا بالىلارغا تەن جازاسىلا ئەمەس باشقا ھېچقانداق جازا بېرىشكە بولمايدۇ، دەپ بار. چۈنكى ئۇنىڭ پەلسەپەسىدە مۇكاپات- جازا تۈزۈمى ئىنسانلارنى باشقىلارغا زۇلۇم سالىدىغان بىر مەخلۇققا ئۆزگەرتىۋېتىدۇ.
