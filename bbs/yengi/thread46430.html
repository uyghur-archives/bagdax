رۇسىيە نېمىشقا يېڭى سېبىرىيەگە S-400 باشقۇرۇلىدىغان بومبا ئورۇنلاشتۇرىدۇ

سەن شەن

نۇرجاھان تەرجىمىسى



S-400 يىراق مۇساپىلىق ھاۋا مۇداپىئە باشقۇرۇلىدىغان بومبىسىنىڭ ئىشلەپچىقىرىش مىقدارىنىڭ ئۈزلۈكسىز ئېشىشىغا ئەگىشىپ، رۇسىيە ئارمىيەسى پۈتۈن مەملىكەت ھەمدە چېگرا سىرتىدىكى مۇھىم رايونلاردا بۇ يېڭى تىپتىكى ھاۋا مۇداپىئە قورالىنى ئورۇنلاشتۇرۇشقا باشلىدى. نۆۋەتتە، موسكۋا ئەتراپىغا S-400 باشقۇرۇلىدىغان بومبا سىستېمىسى ئورۇنلاشتۇرغاندىن سىرت، رۇسىيە ئارمىيەسى يېقىندا يەنە يېڭى بىر تۈركۈم ھاۋا مۇداپىئە باشقۇرۇلىدىغان بومبا سىستېمىسىنى يېڭى سېبىرىيەگە ئورۇنلاشتۇردى.
يېڭى سېبىرىيە 1893-يىلى قۇرۇلغان شەھەر. مەمۇرىي رايوندا يېڭى سېبىرىيە بولسا يېڭى سېبىرىيە ئوبلاستىنىڭ مەركىزى، يېڭى سېبىرىيە ئوبلاستى رۇسىيەنىڭ يېڭىدىن ئايرىلغان يەتتە يۇقىرى دەرىجىلىك مەمۇرىي رايونىدا سېبىرىيە فېدېراتسىيە باشقۇرۇش رايونىنىڭ بىر قىسىمى ھەم پۈتكۈل سېبىرىيە رايونىدىكى ئەڭ چوڭ شەھەر. يېڭى سېبىرىيە رۇسىيەدە موسكۋا ۋە سانت پېتىربۇرگدىن قالسىلا، نوپۇسى ئۈچىنچى كۆپ شەھەر. مەزكۇر شەھەردە زاۋۇتلار ئىنتايىن كۆپ ھەمدە رۇسىيەدىكى ئەڭ ياخشى بىر قىسىم ئۇنىۋېرسىتېت، موزىي ۋە تىياتىرخانىلار بار.
يېڭى سېبىرىيە ئوبلاستى 1937-يىلى 9-ئاينىڭ 28-كۈنى قۇرۇلغان، 1943، 1944-يىلى كېمېپوۋو ئوبلاستى ۋە تومسىك ئوبلاستى ئىلگىرى-ئاخىرى بولۇپ يېڭى سېبىرىيە ئوبلاستىدىن ئايرىلىپ چىقىپ، كېمېپوۋو ئوبلاستى ۋە تومسىك ئوبلاستى بولۇپ قۇرۇلغان، ئالتاينىڭ بىر قىسىم زېمىنى يېڭى سېبىرىيە ئوبلاستىغا قوشۇلۇپ، بۈگۈنكى يېڭى سېبىرىيە ئوبلاستىنى شەكىللەندۈرگەن.
يېڭى سېبىرىيە سېبىرىيە رايونىدىكى ئەڭ چوڭ سانائەت، مەدەنىيەت ۋە ئىلىم-پەن مەركىزى. بىرلا ۋاقىتتا، يېڭى سېبىرىيە رۇسىيە، ياۋروپا ۋە يىراق شەرق رايونىنى تۇتاشتۇرىدىغان مۇھىم مەركەز، رۇسىيە فېدېراتسىيە رايونىدىكى ئەڭ مۇھىم ئىككىنچى چوڭ ئۇنىۋېرسال مەركەز. شۇڭا، يېڭى سېبىرىيە رۇسىيە فېدېراتسىيە رايونىدىكى ئەڭ تەرەققىيات يوشۇرۇن كۈچى بار شەھەر دەپ قارىلىدۇ. ھازىر يېڭى سېبىرىيە مۇھىم ئىقتىساد مەركىزى بولۇپ قالدى، ماشىنسازلىقى ياسىمىچىلىقى، دۆلەت مۇداپىئە، ئايروپىلان ياساش، مېتال تاۋلاش، ئېلېكتر قۇرۇلىشى، ئېلېكترون، خىمىيە، دورا ياساش سانائىتى، يېمەكلىك پىششىقلاپ ئىشلەش، يېزا ئىگىلىك ماشىنىلىرىنى ياساش ۋە قۇرۇلۇش ماتېرىياللىرى جەھەتتە ئالاھىدە گەۋدىلىك.
يېڭى سېبىرىيەنىڭ پۇل-مۇئامىلە كەسپى ئىنتايىن تەرەققىي قىلغان، 60دىن ئارتۇق بانكا يەرلىكتە تارماق ئورگىنى ۋە ۋەكىل ئورگىنى تەسىس قىلغان، ئۇلارنىڭ يېرىمى رۇسىيە فېدېراتسىيە رايونىدىكى ئەڭ چوڭ خەلقئارا پۇل-مۇئامىلە ۋە كرېدىت شىركەتلىرى، رۇسىيە رايونىدىكى ئەڭ چوڭ شىركەت ۋە خەلقئارا سودا شىركەتلىرى بىردەك يېڭى سېبىرىيەدە ئاكتىپ سودا مەبلەغ سېلىش ۋە سودا پائالىيەتلىرى بىلەن شۇغۇللانماقتا، قۇرۇلۇش كەسپى ئۇلارنىڭ ئالدىن تاللىشى.
يېقىنقى يىللاردا، يېڭى سېبىرىيە جەلىپ قىلغان مەبلەغدە قۇرۇلۇش مەبلىغىنىڭ ئومۇمىي سوممىسى رۇسىيەدە ئەڭ يۇقىرى رايونلارنىڭ بىرى ھەمدە مەبلەغ ئومۇمىي سوممىسى يەنە داۋاملىق ئاشماقتا.
2014-يىلى 10-ئاينىڭ 16-كۈنى، جۇڭگو باش مىنىستىرى لى كېچىياڭ رۇسىيە باش مىنىستىرى مېدىۋىدېۋ بلەن بىرگە جۇڭگو-رۇسىيە باش مىنىستىرلىرى 19-قېتىملىق قەرەللىك ئۇچرىشىش يىغىنىغا رىياسەتچىلىك قىلدى. يىغىندا رۇسىيە بىلەن جۇڭگو «موسكۋا-بېيجىڭ» يۇقىرى سۈرئەتلىك تۆمۈر يول تەرەققىيات ھەمكارلىق ئەسلەتمىسىنى ئىمزالىدى، ئىككى تەرەپنىڭ بۇ تۈرنى تەرەققىي قىلدۇرۇش نىيىتى بار، ئەڭ ئاخىرىدا «موسكۋا-يېڭى سېبىرىيە-قازاقىستان-شىنجاڭ-بېيجىڭ» يۇقىرى سۈرئەتلىك تۆمۈر يول لىنىيەسىنى ئاچىدۇ.
يېڭى سېبىرىيە رۇسىيەنىڭ مۇھىم ئىلىم-پەن مەركىزى.
ئىككىنچى دۇنيا ئۇرۇشى مەزگىلىدە، سوۋېت ئىتتىپاقى ئىستراتېگىيەلىك يۆتكىلىپ، يېڭى سېبىرىيەدە ياۋروپادىن قىسىمەن كۆچۈپ كەلگەن نۇرغۇن ئىشچى ۋە ئاممىنى ئورۇنلاشتۇرغان، شەھەر قۇرۇلىشى قەدىمىنى تېزلەشتۈرگەن. مەزكۇر شەھەرنىڭ پەن تەتقىقات ئەمەلىي كۈچى ئىنتايىن مول: بۇ يەردە رۇسىيە ئاكادېمىيەسىنىڭ سېبىرىيە تارمىقى، رۇسىيە تېببىي ئاكادېمىيەسىنىڭ سېبىرىيە تارمىقى، رۇسىيە دېھقانچىلىق ئاكادېمىيەسىنىڭ سېبىرىيە تارمىقى تەسىس قىلىنغان.
بۇنىڭدا، يېڭى سېبىرىيە ئىلىم-پەن شەھىرى دۇنيادىكى داڭلىق يۇقىرى پەن-تېخنىكا رايونلىرىنىڭ بىرى، يېڭى سېبىرىيە شەھىرىنىڭ جەنۇبىغا 20 كىلومېتىر كېلىدىغان يەرگە جايلاشقان، نۆۋەتتە تەبىئىي پەن، تېخنىكا پەنلىرى قاتارلىقلارنى ئۆز ئىچىگە ئالغان 30 ئۇنىۋېرسال پەن تەتقىقات ئەمەلىي گەۋدىسى بار، رۇسىيەنىڭ شەرقىي رايونىدىكى ئەڭ چوڭ پەن تەتقىقات مەركىزى.
يېڭى سېبىرىيە پەن-تېخنىكا رايونى بىردىنبىر رۇسىيە پرېزىدېنتى بۇيرۇققا ئىمزا قويۇپ قۇرۇلغان پەن-تېخنىكا رايونى. پەن-تېخنىكا رايونىنىڭ باش دېرىكتورلىقىنى دۇنيادىكى داڭلىق ئالىم، ئاكادىمىك ئۈستىگە ئالغان. پەن-تېخنىكا رايونى ئاساسلىقى يېڭى سېبىرىيە پەن-تېخنىكا شەھىرىنىڭ ئالىملىرىنىڭ كارخانا قۇرۇشىغا ياردەم بېرىش، ئۇلارغا خەلقئارا ھەمكارلىق شېرىكى تېپىپ بېرىش، خەلقئارا ھەمكارلىق دائىرىسىنى كېڭەيتىش، كارخانىلارنىڭ خەلقئارا بازارغا مۇلازىمەت تەمىنلىشىگە يۈزلىنىشىگە ياردەم بېرىدۇ. بۇ جايغا رۇسىيەدىكى %50دىن كۆپ ئەمەلىي كۈچى بار ئالىم توپلاشقان، پەن-تەتقىقات ئىقتىدارى رۇسىيە بويىچە ئۈچىنچى ئورۇندا تۇرىدۇ.
2015-يىلى 11-ئاينىڭ 19-كۈنى، ئىككىنچى نۆۋەتلىك يېڭى سېبىرىيە ئىقتىساد مۇنبىرى رۇسىيەنىڭ يېڭى سېبىرىيە كۆرگەزمە مەركىزىدە داغدۇغىلىق ئېچىلدى. يېڭى سېبىرىيە ئىقتىساد مۇنبىرى رۇسىيەنىڭ يىراق شەرق رايونىدا ئۆتكۈزۈلگەن ئەڭ چوڭ كۆلەمدىكى خەلقئارا ئىقتىساد مۇنبىرى، ئۆتكۈزگۈچى تەرەپنىڭ تەكلىپىگە بىنائەن، جۇڭگونىڭ ئىشلەپچىقىرىش كەسپىنى چەتئەلدە تەرەققىي قىلدۇرۇش جەمئىيىتى جۇڭگو ۋەكىلى سۈپىتىدە يىغىنغا قاتناشتى. مۇنبەرنىڭ ئاساسىي مەقسىتى جۇڭگو ئوتتۇرىغا قويغان «بىر بەلباغ بىر يول» ئىستراتېگىيەلىك ھەمكارلىق پىلانىغا ماسلىشىپ، جۇڭگو بىلەن رۇسىيەنىڭ پۇل-مۇئامىلە كارخانىلىرى، سانائەت كارخانىلىرى، سودا شىركەتلىرى، سودا-سانائەت ھەمدە ساياھەت شىركەتلىرىنىڭ خەلقئارالاشقان ئالماشتۇرۇش ھەمكارلىق سۇپىسى قۇرۇپ، ئىككى تەرەپلىك ھەمدە كۆپ تەرەپلىك قىممىتى بار سودا ھەمكقارلىقى بەرپا قىلىشنى ئىلگىرى سۈرۈشتۇر.

بۇنىڭدىن كۆرۈۋېلىشقا بولىدۇكى، رۇسىيە نۆۋەتتە غەربنىڭ بېسىمىغا دۇچ كېلىۋاتقان ئەھۋالدا، بۇنداق مۇھىم شەھەرنى قوغداشنى كۈچەيتىشىنىڭ مەقسىتىنى ئويلاپلا بىلگىلى بولىدۇ. يېڭى تىپتىكى S-400 ھاۋا مۇداپىئە باشقۇرۇلىدىغان بومبا سىستېمىسى ئورۇنلاشتۇرۇش رۇسىيە دائىرلىرىنىڭ يېڭى سېبىرىيەگە بولغان يۈكسەك كۆڭۈل بۆلۈشى ۋە ئەھمىيەت بېرىشىنى كۆرۈۋالغىلى بولىدۇ.
رۇسىيە ئارمىيەسىنىڭ دېيىشىچە، S-400 باشقۇرۇلىدىغان بومبا سىستېمىسى سۈرئەت، دەلىللىكى قاتالىق جەھەتلەردە ئامېرىكىنىڭ «ۋەتەنپەرۋەر» PAC-3 يەردىن ھاۋاغا بولغان باشقۇرۇلىدىغان بومبا سىستېمىسىدىن ئۈستۈن ئىكەن، بۈگۈنكى دۇنيادا ئىقتىدارى ئەڭ ياخشى ھاۋا مۇداپىئە باشقۇرۇلىدىغان بومبا سىستېمىسى.
S-400 باشقۇرۇلىدىغان بومبا سىستېمىسى تۇنجى قېتىم 3 خىل يېڭى تىپتىكى باشقۇرۇلىدىغان بومبا ۋە ماتورلۇق نىشان ئىزدەش سىستېمىسى قوللانغان، ھەرخىل ئۇرۇش ئايروپىلانلىرى، سىگنال ئايروپىلانلىرى، تاكتىكىلىق باشقۇرۇلىدىغان بومبا ھەمدە باشقا توغرا يېتەكلىنىدىغان قوراللارغا تاقابىل تۇرالايدۇ، ھەم ئەنئەنىۋى ھاۋادىن مۇداپىئەلىنىش ۋەزىپىسىنى ئۈستىگە ئالالايدۇ ھەم غەيرىي ئىستراتېگىيەلىك باشقۇرۇلىدىغان بومبىدىن مۇداپىئەلىنىش ۋەزىپىسىنى ئىجرا قىلالايدۇ.
S-400 سىستېمىسىنىڭ ئەڭ چوڭ ئالاھىدىلىكلىرىنىڭ بىرى تۆۋەن بوشلۇق، ئوتتۇرا بوشلۇق، يۇقىرى بوشلۇق، يېقىن مۇساپە، ئوتتۇرا مۇساپە، يىراق مۇساپىدىكى ھەرخىل باشقۇرۇلىدىغان بومبىلارنى ئاتالايدۇ. S-400 سىستېمىسى يېڭى تىپتىكى 40N6 يىراق مۇساپىلىق باشقۇرۇلىدىغان بومبىنى ئىشلەتكەندا، S-400 ھاۋا مۇداپىئە باشقۇرۇلىدىغان بومبا سىستېمىسىنىڭ ئېتىش ئېگىزلىكى 400 كىلومېتىرغا يېتىدۇ، بۈگۈنكى دۇنيادا يەردىن ھاۋاغا بولغان باشقۇرۇلىدىغان بومبا ئېتىش مۇساپىسى ئەڭ يۇقىرى.
شۇڭا، رۇسىيەنىڭ يېڭى سېبىرىيەدە ئىلغار S-400 باشقۇرۇلىدىغان بومبا سىستېمىسى ئورۇنلاشتۇرۇشى رۇسىيەنىڭ بۇنىڭدىن كېيىن «بىر يول بىر بەلباغ» ئىقتىساد تەرەققىياتىدا مەزكۇر رايوننىڭ مۇھىم ئورنىغا ئاساسەن چىقارغان قارار.

   سۈرىيە تىنىچلىق سۆھبىتىگە قايتىپ كەلگەن بىلەن روسىيە ۋە سۈرىيە مەقسىتىگە يەتتى .ئۆتىچىلەرنىڭ تەلىپى ئىجرا بولامدۇ بۇ تىخى ئىنىق ئەمەس .يەنە ئامىرىكىنىڭ ئاسادنى ئاغدۇرۇش پىلانى ئەمەلگە ئاشمىدى .تۈركىيەنىڭ سۈرىيە تۈركمەنلىرىدىن پايدىلىنىپ سۈرىيەنى پارچىلاش خىيالى يەنىلا خام خىيال بولماقتا .سەئۇدى ئاجىز دەپ قارىغان سۈرىيە شىئەلىرى يەنىلا ھاكىمىيەت ئۈستىدە .مۇنداقچە ئىيتقانداق سۈرىيە يەنىلا چوڭ خەۋىپ ۋە كىرىزىس ئاستىدا .
   روسىيەنىڭ يەنە داۋاملىق ئوتتۇرا شەرق ۋە ئىسلام مەزھەپ كۆرىشىگە ئارىلاشقىسى يوق .يەنە قىلىپ ،پۇتېن بارلىق ئىستىيىگىلىك مەقسىتىگە يەتتى .يەنە ئامىرىكىنىڭ چاۋشىيەن مەسىلىسى جەڭ پۇرىقى كۈچلۈك .ھىچبولمىسا چاۋشىيەن روسىيە بىلەن بىۋاستە چىگرىلىنىدۇ .شۇڭا ...ئامىرىكىنىڭ بۇياقتىن قىلىۋاتقان ئويۇنىغا تاقابىل تۇرىشى كىرەك .
    س-400تۈركىيە ۋە سەئۇدىنىڭ بارلىق ھاۋا ھۇجۇمىغا تاقابىل تۇرايلايدۇ .ئەگەر يەر يۈزى قسىمى بىلەن سۈرىيەگە كىرسە ،سۈرىيىگە تاقاۋۇز قىلغانلىق بولۇپ ،ئەگەر ئاساد ھەربىي ياردەم تەلەپ قىلسا ،روسىيە قانۇنلۇق ھالدا تاجاۋۇزچى دۆلەتلەر بىلەن ئۇرۇش قىلالايدۇ يەنە تۈركىيەدىن ئۆچىنى ئالالايدۇ .يەنە ئوتتۇرا شەرق ئۇرۇش ئوتى يانغان ھامان نىفىت باھاسى شىددەت بىلەن ئۆسىدۇ .بۇ ئىشلارغا ئامىرىكا ئارلاشقان بىلەن  ياۋروپا ئارلاشمايدۇ .چۈنكى ياز پەسلى كۆچمەنلەر ئۈچۈن ئىنتايىن پايدىلىق بولۇپ ،كۆچمەنلەر ھەسسىلەپ ئىشىپ ياۋروپا كۆچمەنلەر كىرىزىسىدىن قۇتۇلالمايدۇ .شۇڭا ياۋروپا روسىيە بىلەن بىۋاستە دۈشمەنلەشمەيدۇ .

پۇتىننىڭ تەلەتىگە قارىسا ئۈچىنچى دۇنيا ئۇرۇشىنىڭ قوزغىلىپ كېتىش ئېھتىماللىقى يۇقىرى، روسىيەنىڭ غەربىدە ئۇكرائىنا ئۇرۇشى روسىيەنىڭ شەرقىدە شەرقىي ئاسىيا ۋەزىيىتى ئېغىر دەرىجىدە جىددىيلىشىپ كەتتى. ئەگەر روسىيە ئۆزىنى بىلەلمەي يەنە ھەددىدىن ئاشسا ئامېرىكا باشقىچە تەدبىر قوللىنىشى مۇمكىن، يەنى ئىككىنچى دۇنيا ئۇرۇشىدا ئامېرىكا باشچىلىقىدا ئۇرۇش قىلىپ، مىڭبىر جاپادا مەغلۇپ قىلغان، دۇنيادىكى ئەڭ يامان، ئەڭ يىرتقۇچ يولۋاس بولغان ياپونىيە بىلەن گېرمانىيەنى ئۆز مەيلىگە قويۇۋېتىشى مۇمكىن. ئەينى ۋاقىتتا ئامېرىكا ۋە سوۋېت ئىتىپاقى ناھايىتى ئېغىر بەدەل تۆلەش بەدىلىگە بۇ ئىككى يولۋاسنى ئۇرۇشتا يەڭگەن، كېيىن ياپونىيە ۋە گېرمانىيەدىن ئىبارەت ئىككى يولۋاسنى ئامېرىكا قەپەزگە سولاپ قويغان، قاچانكى دۇنيا ۋەزىيىتى ئامېرىكىغا پايدىسىز ياكى ئامېرىكىغا بېسىم ئېشىپ كەتكەن ۋاقىتتا بۇ ئىككى يولۋاسنى قەپەزدىن چىقىرىۋېتىدۇ. بۇ ئىككى يولۋاسنىڭ ئالاھىدىلىكى ئىنتايىن يۇقىرى، ئۇلارنىڭ مائارىپ، داۋالاش، ئۆتكۈر پەن- تېخنىكا ۋە ياكى شۇنىڭغا ئوخشايدىغان ھەرقانداق ساھەدە دۇنيانىڭ ئەڭ ئالدىدا تۇرىدۇ، لېكىن ھەربىي قۇرۇلۇشى بۇنىڭ سىرتىدا، بۇنىڭغا چەكلىمە قويۇلغان. ياپونىيە دېگەن يولۋاس روسىيەگە چىش-تىرنىقىغىچە ئۆچ، بۇ يولۋاس قەپەزدىن چىققان ھامان روسىيە بىلەن ئۇرۇشۇش ئۈچۈن كەڭ كۆلەملىك ھەربىي تەييارلىق قىلىدۇ، ھەربىي تەييارلىق پۈتسىلا شىمالدىكى تۆت ئارالغا ھۇجۇم قىلىدۇ. شۇنىڭ بىلەن روسىيە بىلەن ياپونىيە ۋەھشىيلەرچە ئۇرۇشۇپ ئۇرۇشنى ئەۋجىگە چىقىرىدۇ. يەنە بىر كۈچلۈك يولۋاس بولغان گېرمانىيە قەپەزدىن چىققان ھامان ياپونىيەگە ئوخشاشلا كەڭ كۆلەملىك ھەربىي تەييارلىق قىلىدۇ، شۇنىڭغا ماس ھالدا مىللەتچىلىك ئېغىر دەرىجىدە كۈچىيىپ كېتىپ، بۇرۇنقى ئۇرۇشتا مەغلۇپ بولغان تارىخنى ئەسكە ئېلىپ، بۇرۇنقى ھاقارەتنى يۇيۇشقا كۈچلۈك بەل باغلايدۇ. ھەمدە ياپونىيەگە ماسلىشىپ روسىيەگە ئۆچ بولغان نۇرغۇن دۆلەتلەرنىڭ ھەربىي قىسىمىلىرىنى تەشكىللەپ روسىيەگە شىددەتلىك ھۇجۇم قىلىدۇ، شۇنىڭ بىلەن ئۈچىنچى دۇنيا ئۇرۇشى باشلىنىپ كېتىش ئېھتىماللىقى ئىنتايىن يۇقىرى. دۇنيادا روسىيەنى چەكلىيەلەيدىغان دۆلەتتىن پەقەتلا ئامېرىكىلا بار، قالغان دۆلەتلەر روسىيەنى چەكلىيەلمەيدۇ، لېكىن مەيلىگە قويۇۋەتسە ياپونىيە بىلەن گېرمانىيەنىڭ روسىيەنى چەكلەش ئىقتىدارى بار، ھازىرچە ياپونىيە بىلەن گېرمانىيەگە چەكلىمە بار، ئەگەر بۇ چەكلەشنى پۈتۈنلەي بىكار قىلىۋەتسە بۇ ئىككىسى بىرلىشىپ روسىيەنى چاناپ پېيىنى قىرقىۋېتىدۇ. مېنىڭچە ئامېرىكا ياپونىيە دېگەن يولۋاسنى قەپەزدىن چىقىرۋېتىشى مۇمكىن، چۈنكى شەرقى ئاسىيا ۋەزىيىتى شۇنى تەلەپ قىلىۋاتىدۇ، ئەگەر ياپونىيە دېگەن يولۋاس قەپەزدىن چىقىپ قالسا بىرلا ۋاقىتتا ئىككى زور دۆلەتنى، يەنى جۇڭگو بىلەن روسىيەنى چەكلىيەلەيدۇ. ئەگەر ئامېرىكا ياپونىيەنى قەپەزدىن چىقىرىۋەتسە روسىيە بىلەن جۇڭگودىن كېلىدىغان تەھدىتتىن، بېسىمدىن قۇتۇلالايدۇ. روسىيەنىڭ سىبىرىيە رايونىغا ئېس 400 نى ئورۇنلاشتۇرۇشى ئېھتىمال شەرقىي ئاسىيا ۋەزىيىتى جىددىيلىشىپ كەتكەنلىكىدىن بولغانلىقى بولۇشى مۇمكىن، ھەمدە جۇڭگوغىمۇ قارىتىلغان بولۇشى مۇمكىن، چۈنكى روسىيەدە جۇڭگولۇقلار سىبىرىيەگە كۆچۈپ كېلىپ نوپۇسىنى كۆپەيتىۋاتىدۇ دەيدىغان ئەندىشە بار، جۇڭگولۇقلار سىبىرىيەنى ئىگەللىۋېلىشى مۇمكىن دەپ قارايدىغانلارمۇ كۆپتۇر. روسىيە ئىزچىل جۇڭگولۇقلاردىن ئەندىشە قىلىۋاتىدۇ، چۈنكى سىبىرىيە رايونىدا رۇسلارنىڭ نوپۇسى ئاز، لېكىن جۇڭگولۇقلار كۈندىن-كۈنگە كۆپىيىۋاتىدۇ. روسىيەنىڭ سىبىرىيەگە ئېس 400 نى ئورۇنلاشتۇرۇشى ئاساسلىقى شىمالىي كورېيە ئۇرۇشىنىڭ ۋەھىمىسى بولغان ئامېرىكىغا قارىتىلغان، ئاندىن قالسا جۇڭگوغا، ياپونىيەگە قارىتىلغان.

  روسنىڭ ئارمىيىسىنى سۈرىيەدىن چىكىندۈرۈشى :
1 .  بۇ ئۇرۇش ئۆزى ئويلىغاندەك نىفىتنىڭ باھاسىنى ئۆرلىتەلمىدى ، ئەكسىچە روسقا قارشى كۈچلەر بىرلىشىپ نىفىت باھاسىنىڭ تېخىمۇ چۈشۈشىنى كەلتۈرۈپ چىقاردى .
2 . ئەگەر روس داۋاملىق سۈرىيەدە ئويناۋەرسە ، باشقا دۆلەتلەرنىڭ قۇرۇقلۇق ئارمىيىسىنىڭ سۈرىيەگە كىرىش ئېھتىماللىقى بارغانسىرى ئەمىلىيەتكە ئايلىنىش ئېھتىمالىقى بار . بۇنداق ئەھۋالدا تۇقۇنۇشۇپ قېلىشتىن ساقلانغىلى بولمايدۇ ، ئەگەر شۇنداق بولسا روس پۈتۈن يەر شارى بىلەن جەڭ قىلىشقا توغرا كىلىدۇ ، بۇ بىر قورقۇنۇچلۇق ئاقىۋەت .
3 . ئۇرۇش قىلىپ نىفىت باھاسىنى ئۆستۈرۈش پىلانى ئەمەلگە ئاشمىدى ، بۇ ئۇرۇشقا پۇل ساماندەك كەتتى ، بەشىرنىڭ ئەتە - ئاخشاملىرى تامنىڭ كەينىدە بىرىدىغان ئالتۇن - كۈمۈش ، دوللارلىرى يەنە باشقا نەرسىللىرى تۈگۈدى ، ئەمدى روسنى نىسىگە ئىشلىتىدۇ .
4 . مۇشۇ ئۇرۇش ئارقىلىق باشقىلارنى ئۆزى بىلەن سۆھبەتلىشىشكە قىستىماقچى ئىدى ، پەقەت مۇشۇ جەھەتتىلا 20 % ئۈنۈمنى قولغا كەلتۈردى ، يەنى ئىمبارگو ، جازالار كەمەيتىلمىگەن ئەھۋال ئاستىدا ، روسنىڭ سۈرىيەدىن چىقىپ كىتىش توغرىسىدىلا سۆھبەت بولدى .

  دىمەك روسنىڭ قىلغان ئىشى ، بەشىرنىڭ مەھشەر كۈنىنى سەل سوزدى شۇ ، ئەمدى كۆرىمىز سۈرىيەدە كىلىشتۈرۈش ئەمەلگە ئاشماسلىقى سەۋەپلىك ، بۇ ئۇرۇش قايتىدىن يەنە باشلىنىدۇ ، روس ئەمدى ئارمىيىسىنى سۈرىيەگە ئەۋەتمەيدۇ ، بەشىر مەغلۇپ بولۇدۇ ۋە دارغا ئېسىلىدۇ .
