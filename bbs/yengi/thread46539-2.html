دۆلەتنىڭ تۈرلۈك سىياسىي تۈزۈلمىللىرى يۇرت كۆرگەن بەزى بايلارغا تۇسالغۇ بوپ قالىدۇ،ئۇلارنىڭ ئىدىۋى ھالىتى ئادەتتىكى كىشىلەرگە قارىغاندا يۇقۇرى ئۇرۇندا بولغاچقا بىركۈن بولسىمۇ خۇشال ياشاشنى تەشەببۇس قىلىدۇ شۇڭلاشقا ئۇلار شارائىت ۋە ياشاش مۇھىتى تېخىمۇ ياخشى بولغان جايلارغا بىرىپ تاپقان پۇلىنىڭ چىشىنى چاقىدۇ.بۇنىڭ ئەڭ ئاددا مىسالى نۇرغۇن چوڭ قۇرۇقلۇق چولپانلىرىنىڭ سىنگاپور،مالايسىيا،ياپۇنىيە،تەيۋەن،شىياڭگاڭ،ئامرىكا ۋە ياۋرۇپا دۆلەتلىرىگە چقىپ كىتىشى ئىنىق مىسالى.ئۇلار پۇل مالنى دىمىسە ھىچ بولمىغاندا سىياسىي تۇزۇلمىنى دەپ چىقىپ كىتىدۇ (نۇرغۇن چولپانلار پىلانلىق تۇغۇت چەكلىمىسى تۇپەيلى چەتئەل دۆلەت تەۋەلىكىگە ئۆتۇپ پەرزەنىت كۆرۋاتىدۇ)ھازىر بالدۇر بېيىغان بايلار قان-تەر بىلەن تاپقان پۇلىنى بىكاردىن خەققە بىرىشكە رازى ئەمەس.بۇمۇ نۇرمال ھادىسە.ئەمما شۇ پۇل تېپىش يۇلىغا قارتا دۆلەتنىڭ تۇرلۇك ئىتىبار ۋە يۆلەش سىياسەتلىرىنىڭ ياردىمىنى ئۇنتۇپ قالسا بولمايدۇ.بىراق بۇ ئاغزىمىزدىكى شۇئار خالاس...خەنزۇلار ئەزەلدىن پۇلنى يۇقۇرى ئۇرۇنغا قۇيۇپ كەلگەن،ھەممە ئىشتا پۇلنى بىرىنچى ئۆلچەم قىلىدۇ ،،شى رەئىس بۇ قېتىم شۇ پۇلنىڭ كۈچىدىن پايدىلنىۋاتقان يولۋاسلارنى خېلى ئوڭشىدى ،،ئەمما تۇلۇق ئەمەس.ھازىر نامراتلىققا ھۇجۇم قىلىش سېپىگە ئاتلاندۇق،ھەربىر كادىرغا كەم دىگەندە بىردىن نامرات ئائىلىنى يۇلەيسەن دەپ پەرمان چۇشتى،،بۇنىڭغا ئاغزىمىزدا ياردەم بىرىمىز دىگىنىمىز بىلەن ئوتتۇرا تەبىقىدىكى كادىرلارغا نىسپەتەن بۇ زور بېسىم بۇلىدۇ.كادىرلار چىن كۆڭلىدىن رازى ئەمەس پەقەت بېسىم تۇپەيلى يېنىدىن پۇل چىقىرىپ ياردەم بىردىغان گەپ،باشلىق كادىرلارغا تەس كەلمىگەن بىلەن ئادەتتىكى كادىرلارغا تۇرمۇشىغا راسلا تەسىر كۆرسىتىدۇ(شەخسەن مېنىڭ تۇرمۇشۇمغىمۇ تەسىر قىلىدى،بۇ يىللىق پارتىيە بەدەل پۇلىغا 550 يۇەن ئەتراپىدا تۆلىدىم،ئىككى نامرات پارتىيە ئەزاسىنىڭ پۇلىنى تۆلەيسەن دەپ چىقىرىپ قۇيغاچقا،يەنە تۆت جانلىق بىر ئائىلىنى بىر يىل ياردەم بىرىپ يۆلەيدىكەنمەن ۋاھاكازالار)بۇ يەردىكى گەپ نۇرغۇن ئىشلارغا دۆلەت يەنە سىياسەت ئارقىلىق ئوتتۇرا-تۆۋەن كىرىملىك كىشىلەرنى يۆلەيدىكەن.كىيىنچە بايلارنىڭ تىجارەت باج سوممىسىنى يۇقىرى كۆتۇرەمدۇ بۇنى بىرنىمە دىمەك تەس....
