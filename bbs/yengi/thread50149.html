ئەتىگەندە ئورنۇمدىن تۇرۇپ دەرىزىنى ئاچتىم ،قوياشنىڭ ھەممە يەرگە نۇرىنى چىچىپ يوپيورۇق قىلۋەتكىنى كۆرسەم باشقىچە روھلىنىپ كىتەتتىم ،چۇنكى مەن يورۇقلۇقنى ياخشى كۆرەتتىم ،قاراڭغۇدىن قورقاتتىم، شۇنىڭ ئۇچۇنمۇ مەن ھەركۇنى چىراقنى ئۆچۇرمەي ئۇخلايتىم ،بەزىلەر بۇ ئادىتىمگە ھەيران قالاتتى ،بەلكىم ئۆزۇم يالغۇز بالا بولغاچقا كىچىكىمدىن ئانام مىنى قارا بىسىپ قالمىسۇن دەپ چىراقنى ئۆچۇرمەي ئۇخلىتىشى سەۋەپ بولغانمىكىن  .
سىرىتقا چىقىنىمدا  ھەممە يەر ياپيىشىل ھەرخىل گۇللەر ئىشىك ئالدىمدا ئىچىلىپ كەتكىنى كۆرسەم ئۆزۇمنى باشقىچە بەخىتلىك ھىس قىلاتتىم ،ئۇ گۇللەرگە تويماي قارايتىم ،بالىلارنىڭ چۇرقۇرىشىپ ئويناشلىرى چىملىقلارنىڭ مەيىن شامانلدا يەلپىنىشلىرى ماڭا بىرخىل راھەت ئىلىپ كىلەتتى ،،،تۇيۇقسىز ئالدى بىنادىكى سادىق ئالدىمغا يۇگىرەك كىلىپ  مىنى چۇچىتىۋەتتى ،
ھاشىم جېجىق نىمدانداق ھاڭۋاختىدەك قاراپ تۇرىسەن ،بۇگۇن كەچتە ئادىلنىڭ تۇغۇلغان كۇنى بىللە ئولتۇرشىمىز جۇما ،،ئەمسە ھىلى تىلفۇن قىلاي دەپ ئالدىراپ كىتىپ قالدى ،،،
ھەممە ئاغىنە ساۋاقداشلىرىم مىنى ھاشىم جېجىق دەپ ئاتىشاتتى ،نىمىشقا بۇنداق ئاتايدىغانلىقىنى بىمىسەممۇ بۇ لەقەمگىمۇ كۆنۇپ قالدىم ،قانچە تىرىككەنسىرى  شۇنچە كۇچەپ كىتىدىكەن ،ھازىر كارىم بولمىغاندىن بىرى چاقچاق قىلغان  ۋاقىتتىن باشقىدا ھاشىمجان دەپ چاقىرىدىغان بولدى .
ئادىل كىمۇ  .....بىر قىتىم كۆرۇشكەندىن باشقا ھىچنىمىنى ئەسلىيەلمىدىم ...
سائەت 6بولۇپ ئانام ئەتكەن تامىقىنى يەپ ئولتۇرسام يەنە سادىق ئالدىراڭغۇ ئىشىكنى قاتتىق قىقىپ ھاشىمجان دەپ تولىغىنى تۇردى ،مەن تاماقنى قوياپ قويۇپ ئۇنىڭ بىلەن ئۆزۇممۇ  ئانچە تونۇشمايدىغان ئادىلنىڭ تۇغۇلغان كۇنىگە ماڭدۇق ،،،
ئۆزۇم بۇ شەھەرلىك بولساممۇ لىكىن بىرىپ باقمىغان يوللاردا مىڭىۋاتىمىز ،چوڭ كوچىنىڭ دالدىسىغا جايلاشقان بىر يەر ئاستى دۇككانغا كىردۇق ،ئىشىكتىن كىرشىمىزگە قاراڭغۇ تار يول ئۇ مىنى بىر ئايرىم خانىغا باشلاپ كىردى ،،تاماكا ئىسى ئاچچىق ھاراق گۇپۇلدەپ پۇراپ تۇراتتى ،،،ھەر بىرىنىڭ قىشىدا بىردىن قىز   ، ۋاراڭ چۇرۇڭلار بىر ئالغان ،مەن كىرىپ ھەممىسى بىلەن كۆرۇشۇپ بولۇپ بىر چەتتىكى كىرىسلودا  ئولتۇردۇم ،،ئادىل ھاراق تۇتتى ،ياق ئاداش مەن ئىچەلمەيمەن ،ھاراق پۇرىسام ئانام مەندىن قاتتىق رەنجىيدۇ ،ۋاي ئەركىشىدەك يۇرە ھاشىم جېجىق  ئىچە دەپ ۋاقىرىدى يىنىمدا ئولتۇرغان ئادىل ،شىرە ئۇستى قالايمىقان ئۇلار بىردەم قىزلارغا قالايمىقان چاقچاق قىلسا بىردەم ئاغزىنى بۇزۇپ تىللاپ باقاتتى ،شۇ ئەسنادا بىر بالا ئىشىكتىن  كىرىپ كەلدى ،سادىق ئالدىغا بىرىپ قاتتىق تىللاپ ئارقىدىن بىرنىملەرنى دەپ چىقىرۋەتتى،
مەن ھەيران بولۇپ سورۇدۇم ،سادىق ئۇ كىمۇ نىمىشقا تىلللاپ كىتىسىلە ...
ۋاي ئۇ ئەيدىز گۇي ....نىمە  مەن ھەيران قالغىنىمدىن  ئاۋازىم چىقىپ كەتتى ،
بۇرۇن يىقىن ئاغىنىلەر ئىدۇق كىيىن شۇنداق بولغاندىن بىرى ئارىمىزغا قاتمىدۇق ،لىكىن ئۇنىڭ پۇلى بولمىغاندىكىن بىزگە قىتىلىپ ئوينىغۇسى كىلىدۇ ،بۇگۇن بۇنداق خۇشاللىق سورۇنغا ئۇنى ئەكرسەك بولمايدۇ ..
قاچان شۇنداق بولۇپ قالغان ،،،،بىزمۇ بىلمەيتۇق ،،،بىر قىتىم قاتتىق مەس بولۇپ كىتىپ بىر ھازا يىغلاپ كەتتى ،ئىنچكىلەپ سورىساق ئۆزىنىڭ ئەيدىز ئىكەنلىكىنى دەپ سالغان ...
بۇلارنى ئاڭلاپ بىر خىل قورقۇنۇچ باستى، شۇ سورندا ئولتۇرغانلارمۇ كۆزۇمگە باشقىچە كۆرۇندى ...بىردەم ئولتۇرۇپ ماڭماقچى بولدۇم  ...سادىق...  ھاشىم جېجىق نەگە ،يەنە سېزىپ قويغان جېجىقتەك مەلەڭنىڭ دوقمۇشىدا تۇرامسەن ،،بۇگۇن ياتاق ئاچىمىز  .ئەركەكتىك يۇرە ..ئۆيەڭگە  بىرەر كۇن كىرمىسەڭ ھىچنىمە بولمايدۇ ،،،
ئۇ مىنى زورلاپ تۇتۋالغانسىرى مىنى قورقۇنۇش باسقىنى تۇردى ،،،ئەيدىزنىڭ نىمە بىلەن يۇقىدىغانلىقىنى شۇلار ئىنىق بىلىپ تۇرسىمۇ ،لىكىن نىمىشقا ئۆزىنى يەنە شۇ يولغا زورلايدۇ ...ئەيدىزشۇلارنى بىربىرلەپ يەپ كىتەمۇ ،،،مەنمۇ قۇربانى بولۇپ كىتەرمەنمۇ ؟؟
خۇددىي بۆرە توپىغا كىرىپ قالغان قوزىدەك ھىس قىلسام ،يەنە تۇرۇپ  قانداق قۇتۇلۇشنى ئويلايتىم ،،بىردەمدىن كىيىن ھەممىسى مەس بولۇپ ئالاجوقا سۆزلىگىنى تۇردى ،،،مەن تازلىق ئۆيىگە بىرىپ  كىلىشنى باھانە قىلىپ ئاستا قاچتىم ...
ئىشىكنى ناھايىتى ئاستا ئاچتىم ئانامنىڭ ئويغۇنۇپ كىتىشىنى خالىمايتىم ،لىكىن ئۆيگە كىرسەم ئانام ئۇخلىماي مىنى ساقلاپ ئولتۇرۇپتۇ ...
......
كۆزۇمنى يۇمۇپ ياتتىم ،ئۇيقۇ يوق كاللام ناھايىتى  چىگىش بولۇپ كەتتى ،بىر تۇرۇپ  ئىنسانلارنىڭ ئۆزىنىڭ قولىدا ئۆزىنى نىمىشقا  نابۇت قىلشىنى ئويلىسام بىردەم  تۇرۇپ  قانداق قىلىپ شۇلارنى توغرا يولدا باشلىغىنى بولار دىگەنلەرنى ئويلايتىم ،،،مەن كىم ؟ئۇلار مىنىڭ گىپىمنى ئاڭلامۇ ،،،
ئەلۋەتتە مەن دۇنيانى ئۆزگەرتەلمەيمەن ،لىكىن ئۆزۇمنى ئۆزگەرتەلەيمەن ...شۇلارنى ئويلاپ قانداق ئۇخلاپ قالغىنىمنى بىلمەيمەن ،،،ئەتىسى يەنە ئورنۇمدىن تۇرۇپ ناشتا قىلىپ بولغاندىن كىيىن  يەنە بۇرۇنقىدەك سىرىتقا چىقىپ قوياشنىڭ ئىسىسىق ھارارىتىدە گۇل گىياھلارنىڭ خۇش پۇراق ھىدىنى پۇراپ،بەزىدە بالىلارنىڭ چۇرقىرىشىپ ئويناشلىرى بوۋاي مومايلارنىڭ بىربىرگە ھال مۇڭ بولاشلىرىنى كۆرۇپ، ئۆزۇمنى يەنىلا بەخىتلىك، دۇنيارىمنىڭ يەنىلا شۇنداق گۇزەل ئىكەنلىكىنى ھىس قىلدىم.....
