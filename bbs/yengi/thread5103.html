    يېقىندىن بۇيان بىرقانچە تورداشلارنىڭ بىرقىسىم توركۆرگۈلەردە باشقا جايلاردى خەتلەر نورمال ، ئەمما ئىنكاس يېزىش كۆزنىكىدە كونا يېزىقنى نورمال يازغىلى بولمايۋاتىدۇ دىگەن ئىنكاسلارنى تاپشۇرۋالدىم ،    ھەم بۇتوغرىسىدا ئىزدىنىپ كۆرگەنتىم . (دەسلەپكى قەدەمدە css دتەڭشىكىدىكى مەسىلە دەپ يەكۈن چىقاردىم )بۈگۈن مانا ie8دا بۇمەسىلىگە يولۇقۇپ بىرقىسىم ئەمەلىي تەجىرىبىلەرنى ئېلىپ بارغاندىن كېيىن بىر ئامالنى تېپىپ چىقتىم . گەرچە مەسىلىنى تۈپتىن ھەل قىلغىلى بولمىسىمۇ ، تورداشلارنى باشقا جايلارغا يېزىۋېلىپ ئاندىن مۇنبەرگە چاپلاش ئاۋارىچىلىقىدىن قۇتۇلدۇرۇدۇ .  مەسىلىنى ھەل قىلغۇچە تورداشلارنىڭ بۇ ئامالنى قولللىنىپ تۇرۇشىنى ئۈمۈدقىلىمەن .
   مەشخولات تەرتىپى تۆۋەندىگىچە :
      مۇنبەردا ئىنكاس يوللاش ئىككى خىل بولۇپ بىرىنچىسى تورداشلار كۆپ ئىشلىتىپ كۆنۈپ كەتكەن تېز ئىنكاس يوللاش يەنى بىرتېمىنى ئاچقاندىن كېيىن شۇ بەتنىڭ تېگىدە چىقىدىغان ئىنكاس يېزىش رامكىسى (ئەلۋەتتە سىز تىزىملىتىپ كىرگەندىن كېيىن ئاندىن كۆرىنىدۇ) ، ھازىر مەسىلە دەل مۇشۇ تېز ئىنكاس يوللاش كۆزنىكىدە كۆرىلىۋاتىدۇ .
   ئىككىنچى خىل ئىنكاس يوللاش ئۇسولى بولسا ، شۇتېمىنى كۆرىۋاتقان بەتتە  بىرىنچى يازمىنىڭ سول تەرەپ بېشىدا ئىككى تاللاش تۈرى بار بىرى تېما يوللاش ، يەنە بىرى ئىنكاس يوللاش .  بۇجايدىكى ئىنكاس يوللاشنى باسسىڭىز ئالىي ئىنكاس يوللاش كۆزنىكى ئېچىلىدۇ بۇجايدا سىز ئىنكاسىڭىزنى ئىلگىرلىگەن ھالدا تەھرىرلىيەلەيسىز . 
   مەن تېز ئىنكاس رايونىدا كاشىلا كۆرۈلگەن بولسا ئالىي ئىنكاس رايونى نورمالمىدۇ دېگەن ئوي بىلەن سىناپ باققان  ئىدىم نەتىجىدە نورمال كونا يېزىقتا خەت ئۇرغىلى بولدى . ئەلدىن ئەلگە نەپ دەپ  دەرھال بۇتېمىنى يوللاپ ئولتۇرۇپتىمەن ، يوقۇرقىدەك مەسىلىگە يولۇققانلار بولسا مۇش ئۇسولنى قوللىنىپ بېقىڭلار .

   مۇنبىرىمىزنىڭ تېخىمۇ مۇكەممەل بولىشى ئۈچۈن تورداشلارنىڭ مۇنبەرنى ئىشلىتىش داۋامىدا  يەنە باشقا يولۇققان مەسىلىلىرى بولسا بىزگە ۋاقتىدا ئىنكاس قىلىشىنى ئۈمۈد قىلىمىز .قوللىغىنىڭلارغا رەھمەت ! باغداشنىڭ ئەتىسى سىلەرنىڭ قوللىشىڭلاردىن ئايرىلالمايدۇ .
