         ئەنەس ئىبنى مالىك رەزىيەللاھۇ ئەنھۇ رىۋايەت قىلغان ھەدىس (ئاللاھ ئۇنىڭدىن رازى بولسۇن) رەسۇلۇللاھ 62يىشىدا يەنى ۋاپات بولۇشقا بىريىل قالغاندا ئاللاھنىڭ ئەڭ قاتتىق سىناقلىرىدىن ئۆتۈپ بولغاندا، ئىنسان بالىسى كۆتۈرەلمەيدىغان، بەرداشلىق بىرەلمەيدىغان ئىمتىھانغا دۇچ كەلدى. رەسۇلۇللا پەيغەمبەرلىككە ۋەھيى كېلىشتىن بۇرۇن خەدىچە ئانىمىزدىن ئككى ئوغۇل پەرزەنت كۆرگەن ئىدى. بىرىنىڭ ئىسمى قاسىم يەنە بىرىنىڭ تاھىر ئىدى. لېكىن ئۇلار كىچىكىدە ئالەمدىن ئۆتكەن ئىدى.رەسۇلۇللاھ پەيغەمبەر بولغاندىن كېيىن ئوغۇل پەرزەنت كۆرمىگەنلىكى ئۈچۈن دائىم ئوغۇل پەرزەنتتى بولۇشنى ئارزۇ قىلاتتى، شۇچاغلاردا كافىرلارنىڭ 10-15تىن خوتۇنلىرى بولغاچقا ئوغۇللىرى جىق ئىدى ھەم شۇنىڭدىن پەخىرلىنەتتى ۋە رەسۇلىللاھقا لەقەم قويۇپ (ئەبتەر) يەنى (دىنى كېسىك،ئاخىرى يوق) ئوغلى يوق، ئۇ ئالەمدىن ئۆتكەندىن كېيىن ئۇنىڭ ئىشلىرى توختاپ قالىدۇ،دەپ رەسۇلۇللاھنىڭ دىلىغا ئازار بەردى. بۇنى رەسۇلۇللاھ ئاڭلاپ كىرىپ ئى ئاللاھ بۇندىن كىيىن ئشلىرىم توختاپ قالمىسا ئوغۇل پەرزەنت بەرسەڭ دەپ يىغلاپ دۇئا قىلدى. شۇندا جەبرائىل ئەلەيھسسالام قۇرئاندىكى ئەڭ كىچىك سۈرە بولغان (سۈرە كەۋسەر) دىن ۋەھيى ئېلىپ كەلدى. رەسۇلۇللاھنىڭ بۇ سۈرىدىن كۆڭلى يورىغاندەك بولدىيۇ لېكىن ئاتىلىق مېھرى يەنە جۇش ئۇرۇپ تۇراتتى. كېيىنچە ئاللاھنىڭ ھېكمىتى بىلەن مارىيە ئانىمىزدىن بىر ئوغۇل پەرزەنت كۆردى ۋە ناھايىتى خۇرسەن بولۇپ پۈتۈن قۇرەيىشلەرگە خەۋەر يەتكۈزدى، ھەممەيلەنگە دۇئالارنى قىلدى. 7 كۈنلۈك بولغاندا ئوغۇلغا (ئبراھىم) دەپ ئىسىم قويدى. يەنى رەسۇلىللاھ ئوغلۇم ئىبراھىم خەلىيلۇللاھدەك بولسۇن دەپ ئبراھىم دەپ ئىسىم قويدى. بۇ ئوغۇل 5 ياشقا كىرگەن يىلى يەنى رەسۇلۇللاھ 62 ياشقا كىرگەن يىلى قاتتىق ئاغرىپ قالدى. بۇنىڭدىن رەسۇلۇللاھنىڭ كۆڭلى پەرىشان بولۇپ يىغلاپ«يا رەببىم 5 كە كىرگەن شۇ ئوغلۇمدىن ئايرىلالمەنمۇ؟ يا رەببىم! ئۆزۈڭ شىپالىق بەرگەيسەن!» دەپ دۇئالار قىلدى، ۋە ئورنىدىن تۇرۇپ سىرتقا قارىسا ئوغلى يۈگۈرۈپ تالادا يۈرگەنلىكىنى كۆرۈپ«سۇبھەناللاھ رەببىم ! دۇئايىمنى ئىجاۋەت قىلىپسەن شۈكرى» دەپ، ئوغلى يېنىغا ماڭدى. دەل شۇ پەيىتتە جەبرائىل ئەلەيھىسسالام كېلىپ قالدى، يەنى جەبرائىل ئەلەيھىسسالام رەسۇلۇللاھقا بەزىدە ئەرەب شەكلىدە كۆرۈنەتتى، بەزىدە كۆرۈنمەي رەسۇلۇللاھنى بىر يۈك باسقاندەك بولاتتى. بۇ قېتىم ئۇ كۆرۈنمەي رەسۇلۇللاھنىڭ يەلكىسىدىن باستى. رەسۇلۇللاھ جەبرائىلغا قاراپ« ئەي جەبرائىل بۇ قېتىم يىغلاپ كېلىپسىزغۇ؟ نىمە بولدىڭىز؟»دەپ سورىدى. جەبرائىل دېدىكى« يا رەسۇلۇللاھ! قىيامەتنىڭ خەۋىرى كەلدى.» دېدى. ۋە ئاز ۋاقىتتىن كىيىن «يا رەسۇلۇللاھ قىيامەتنىڭ خەۋىرى كەلگىنى يوق! ئوغلۇم شىپا تاپتى، دەپ خۇرسەن بولۇپسىز»دېدى. رەسۇلۇللاھ« شۇنداق ئەي جەبرائىل، ئوغلۇمنى دۇنياغا تېگىشمەيمەن، ئۇنى ھەقىقەتەن ياخشى كۆرىمەن!» دېدى. شۇ زامان جەبرائىل يەنە يىغلىدى . «يا رەسۇلۇللاھ! مەن بىلەن كىمنىڭ كەلگىنىنى بىلەمسىز؟»دېدى. رەسۇلۇللاھ:« كىم كەلدى؟ ئەي جەبرائىل؟»دېدى. « ئەزرائىل ئەلەيھسسالام كەلدى، بۈگۈن ئوغلىڭىز روھى قەبىز قىلىنىدۇ يا رەسۇلۇللاھ! ئاللاھ سىزگە ھېچقاچان ئوغۇل پەرزەنت بەرمەيدىكەن، چۈنكى سىزدىن كېيىن ئۈممەتلىرىڭىز ئوغلىڭزنى پەيغەمبەرلىك ماقامىغا كۆتۈرۈپ قويىدۇ دەپ تەشۋىشلىنىمىز ۋە بىر قەلبتە ئىككى مۇھەببەت بولمايدۇ، يا رەسۇلۇللاھ! سىز ئۈممىتىڭىزنى تاللاڭ ياكى ئوغلىڭىزنى» دېدى. رەسۇلۇللاھنىڭ ۋۇجۇدلىرى تىترەپ «يا رەببىم! قېرىغان چېغىمدا قانداق ئىمتىھان بولدى ئەمدى بۇ ماڭا؟» دەپ يىغلاپ كەتتى.جەبرائىلمۇ يىغلاپ تۇرۇپ« يا رەسۇلۇللاھ بارچە پەرىشتىلەرنىڭ قوللىرىنى دۇئاغا ئېچىپ قويدۇم، كېلىڭ،بىرگە ئاللاھقا دۇئا قىلايلى! ھېچ بولمىسا سىز دۇنيادىن كەتكىچە ئوغلىڭىز تۇرۇپ تۇرسۇن» دېدى. رەسۇلۇللاھ« شۇنداقمۇ؟ يا جەبرائىل،مەن ئوغلۇمنىڭ ئۆمرىنى سوراشقا كەلگەن پەيغەمبەرئەمەسمەن، ماڭا ئىختىيار بېرىلسە مەن ئۈممەتنى تاللىدىم. سىز ھەم دۇئا قىلىڭ مەن ھەم دۇئا قىلاي ئاللاھ ئوغلۇمنىڭ ئورنىغا ئىسلامغا خىزمەت قىلىدىغان پەرزەنتلەردىن ساھابە ئۈممەتلىرىمگە، ئاخىر زامان ئۈممەتلىرىمگە، ئوغلۇمنىڭ ئورنىغا ساداقەتمەن، ئىنساپلىق پەرزەنتلەردىن بەرسە ئەجەپ ئەمەس، دۇئا قىلىڭ» دەپ يىغلاپ تۇرۇپ دۇئا قىلدى ۋە ئەزرائىل ئەلەيھسسالامغا« ئەي ئەزرائىل! شۇ كۆزۈمنڭ قارىسى بىلەن خوشلىشىۋالغىلى رۇخسەت قىلىڭ!» دېدى. شۇچاغدا جەبرائىل ئەلەيھسسالام يىغلاپ تۇرۇپ دېدىكى«يا رەسۇلۇللاھ! ئەزرائىل ئەلەيھسسالام سىزنىڭ ئىشارىڭىزنى كۈتۈپ تۇرۇۋاتىدۇ.» رەسۇلۇللاھ ئوغلى يېنىغا چىقتى. ئوغلى ئىبراھىم ئۆز تەڭتۇش ساھابە بالىلىرى بىلەن ئويناپ يۈرەتتى ۋە شۇنچە ساغلام بوپقالغان ئىدى. رەسۇلۇللاھ قارىسا شۇنچە كۆپ بالىلار ئارىسىدا ئەڭ چىرايلىقى يەنە ئۆز ئوغلى. گۈزەل خۇلقلىق، گۈزەل سۈپەتلىك. رەسۇلۇللاھ ئۇزاق ئوغلىغا تىكىلىپ قاراپ تۇردى ۋە« ئەي كۆزۈمنىڭ قارىسى بالام بۇ ياققا كېلىڭ» دېدى. ئوغلى يۈگرەپ كەلگەن ئىدى. رەسۇلۇللاھ ئۇنىڭ باشلىرىنى سىلاپ، باغرىغا باستى ۋە«ئوغلۇم، سىز سەپەرگە چىقىدىغان بولۇپ قالدىڭىز!» دېدى. ئوغلى خۇرسەن بولۇپ« ئۇنداق بولسا يېڭى كىيىملىرىمنى كىيىپ تەييار بولايمىكىن!» دېدى. رەسۇلۇللاھ ئوغلىنىڭ سۆزلىرىنى ئاڭلاپ يىغلاپ كەتتى ۋە« ئەي ئوغلۇم سىزگە كىيىم كېرەك بولمايدىغان، دۇنيا كېرەك بولمايدىغان جايغا كېتىدىغان بولۇپ قالدىڭىز، ئۆيگە كىرىپ بىر مېھمان بىلەن كۆرۈشۈپ چىقىڭ»دېدى. ئەقلى پاراسەتتە رەسۇلۇللاھنىڭ قېنى بولغانلىقى ئۈچۈن ئوغۇل ئاتىسى رەسۇلۇللاھنىڭ نېمە دېگىنىنى دەرھال چۈشەندى ۋە قولىدىكى ئوينىغان ئويۇنچىقىنى كۆتۈرۈپ« ئۇنداق بولسا بۇنى ئامانەت ئالغان ئىدىم، بۇنى بېرىپ قوياي»دېدى. بۇ گەپلەرنى رەسۇلۇللاھ ئاڭلاپ «ئوغلۇم دەۋەتتى ۋە يا رەببىم! شۇنداق ئامانەتچى، شۇنداق پاراسەتلىك ئوغلۇم بار ئىدى» دېدى
