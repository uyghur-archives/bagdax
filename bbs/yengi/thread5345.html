مۇنبەردىكى قۇمۇللۇق مۇسۇلمان قېرىنداشلاردىن سەمىمىي ياردەم سورايمىز!

ئەسسالامۇ ئەلەيكۇم ۋەرەھمۇتۇللاھى ۋەبەركەتىھۇ!
     مۇنداق گەپ ئىدى قېرىنداشلار، بىر  ئەر-ئايال قېرىندىشىمىز قۇمۇلغا ھىجرەت قىلماقچى بولغان ئىدى. ئۇلار قۇمۇلنىڭ ئوي-چۇققۇرىنى بىلمەيدۇ. ئاتايىتىن كىرا قىلىپ بېرىپ ئەھۋال ئېگەللەيمىز دېسە كىرا، ياتاق، تاماق دېگەنلەرگە پالان پۇل كېتىدىكەن. ئۇلار ئاساسلىقى مۇسۇلمان ئاياللارنىڭ تۇرمۇش بويۇملىرى ۋە شۈبھىدىن خالىي، خىمىيىۋىي تەركىۋى يوق ئۆزىمىزنىڭ ئىشەنچلىك ساغلاملىق مەھسۇلاتلىرىنى ساتماقچى. شۇڭا ئۇلارغا مۇسۇلمانلار توپلىشىپ ئولتۇراقلاشقان ئاۋات بازاردىن 20 كۇۋادرات مېتىر ئەتراپىدا دۇكان، 50 كۇۋادرات مېتىر ئەتراپىدىكى ئۆي لازىم. ئالدى بىلەن بۇ قېرىنداشلىرىمىز  يۇرتۇڭلارغا بېرىپ يۇقىرىدىكى تىجارەت بىلەن شۇغۇللانسا ئىستىقبالى قانداق بولىدىغانلىقى، ئەگەر مۇۋاپىق بولسا دۇكان ۋە ئۆينىڭ ئىجارىسىنىڭ قانچە پۇل ئەتراپىدا ئىكەنلىكىنى اللە رىزالىقى ئۈچۈن ئەتراپلىق ئىزدىنىپ خەۋەر بەرگەن بولساڭلار قۇمۇلدىكى ئېمانى كامىل ئەنسار قېرىنداشلاردىن تولىمۇ خۇرسەن بولار ئىدۇق! 

      ئەسكەرتىش:   بىز مۇسۇلمانلار يالغان گەپ قىلمايمىز ھەم بىر-بىرىمىزگە ئاسان ئىشىنىمىز. دېمەكچى، گول كېلىمىز! مۇنبەرداشلارنىڭ ئىنكاسقا چاقچاق ئارىلاشتۇرماسلىقىنى، ئىشەنچلىك بولمىغا ئۇچۇرلارنى يەتكۈزمەسلىكىنى اللەنىڭ نامى بىلەن ئۆتۈنۈپ سوايمىز!!!...

ئەسسالامۇ ئەلەيكۇم ۋە رەھمەتۇللاھى ۋە بەركاتىھۇ !!! قېرىندىشىم ،اللاھۇ ھەق سۇبھانۇ ۋەتەئەلانىڭ ياردىمى ۋە بەركىتى سىزگە ۋە پۈتۈن دۇنيا مۇسۇلمانلىرىغا بولسۇن ،، ئامىن      قېرىندىشىم اللاھ خالىسا سىزگە ياردەم قىلساق بولىدۇ ،ئەمما<ھىىجرەت> بۇ سۆزۈڭىزنىڭ مەنىسىنى تازا چۈشۈنەلمىدىم، سىز بۇ سۆزنى كىنايە ياكى بولمىسا تەنەئورنىدا قوللاندىڭىزمۇ بىلەلمىدىم . ئىنشا اللاھ سىز بۇ سۆزنى ئۆز مەنىسىدە قوللانغان بولىشىڭىزنى ئۈمۈد قىلىمەن، <ھىجرەت> بۇ سۆزنى ئاڭلاپلا كۆڭلۈم باشقىچە بىئارامسىزلاندى .ھەم سىز قېرىندىشىمنىڭ ئۇچۇراۋاتقان بېسىملىرىنى ھېس قىلغاندەك بولدۇم . ئەمما قېرىندىشىم <ھىجرەت>سىزنىڭ يۇرتىڭىزدىن بىزنىڭ يۇرتلىرىمىزغا ھەرگىز بولمايدۇ ، ھەقىقى بولۇدۇ دېيىلسە چەتئەللەردىكى ئىسلام دۆلەتلىرىگە بولىدۇ ،ئوخشاش جاھىليەت ۋە كۇفۇرلۇق قاپلىغان  ئىككى يۇرتتىن بىر بىرىگە ھىجرەت قىلىشقا بولمايدۇ. ناۋادا تىجارەت ياكى باشقا ئامىللارنى كۆزدە تۇتقان بولسىڭىز  سىزگە اللاھ خالىسا ياردەم قىلايلى . ۋەياكى بولمىسا بۇ يۇرتقا بىر كېلىڭ ھەم ئەمەلىي ئەھۋالنى كۆرۈپ بېقىڭ ، سىزگە ماس كېلەمدۇ ياكى ئەكسىچە . لېكىن بۇ يەرنىڭ ئادەملىرى بەك مىجەزى ئوچۇق ،  باشقا قېرىنداشلار بۇ قېرىندىشىمىزغا ياردەم قىلايلى ،  مەن ئىچكىردە بولۇپ قالدىم بولمىغان بولسا رەببىم خالىغان بولسا ئۆزۈم ياردەم قىلاتتىم ، ئەمما ئۈمۈدسىزلەنمەڭ ، مەنمۇ دوسلىرىمدىن سۈرۈشتۇرۇپ باقاي . اللاھ خالىسا ئىشلىرىڭىز چوققۇم ياخشى بولۇپ كېتىدۇ  ئۇ دۇنيادا ھەم بۇ دۇنيادا .                    قېرىنداشلار  ھىجرەت قىلىش دېگەن ---- ئۆز يۇرتلىرىدا ئېتىقادى، ئىمان-ئسلامىنى ، ئەقىدىسىنى ، اللاھقا بولغان سۆيگۈسىنى ساقلاپ قېلىشقا ،  ئەتىراپىدىكى مۇھىت،شارائىت ، تۈزۈم ھەم كۇفرىلىق ،بۇزۇقچىلىق، جاھىليەت قاتارلىق ئامىللار تەسىر يەتكۈزگەن ۋاقىتتا ، ئۆز يۇرتىدىن باشقا  ھەقىقى ئىسلامىيەت يۇرتلىرىغا چىقىپ كېتىشنى كۆرسىتىدۇ . . اللاھ ئىگەم  ھەممىزگە مۇستەھكەم ئىمان ،ئېتىقاد،ئككى دۇنيالىق بەخت-سائادەت ئاتا قىلسۇن        ئامىن
