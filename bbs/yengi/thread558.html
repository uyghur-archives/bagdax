     قومۇل شەھىرى شەھەرئىچى يېزىسى مەگەش كەنتىدە تۇرۇشلۇق ئۈمۈتجان تېخى ئەمدىلا 14 ياشنىڭ قارسىنى ئالغان ، تېخى نەچچە ۋاقىتلار ئىلگىرىلا دوستلىرى بىلەن شاد -خورام ئويناۋاتقان ، تولۇقسىز ئوتتۇرا مەكتەپنىڭ 2 يىللىقىدا ئوقىۋاتقان بىر ئۆسمۈر ئىدى .
 لېكىن تۇيۇقسىز بىر قېتىملىق كېسەل تەكشۈرتۈش ھەممىنى ئۆزگەرتىۋەتتى. ئۈمۈدجان ئاق قان كېسىلى (قان راكى ) كىگەگىرىپتار بولۇپ قالغانىدى ، تەغدىرنىڭ بۇ رەھىمسىز چاقچىقى بۇ ئائىلە كىشىلىرىنى ساراسىمىگە سېلىۋەتتى .بۇ كېسەلنى داۋالىتىشقا نۇرغۇن پۇل كېتەتتى ، بۇ دېھقان ئائىلىسدە نەدىمۇ ئۇنچىلىك كۆپ پۇل بولسۇن !
    پەرزەنتىنى داۋالىتىش ئۈچۈن ئۈمىدجاننىڭ ئاتا- ئانىسى تاپقان -تەرگىنىنى يىغدى .باشقىلاردىن قەرز ئالدى. .لېكىن بۇ پۇل ئۈمىدجاننىڭ بىر قېتىملىق خىمىيلىك داۋالاش ئېلىپ بېرىشىغا ئاران يەتتى .ئامال يوق ئۈمىدجان ئائىلىدە بىر مەزگىل كۈتۈلگەندىن كېيىن بۇ يىل 1-ئاينىڭ 3-كۈنى كېسىلى قايتا قوزغىلىك ئاپتونۇم رايونلۇق خەلق دوختۇرخانىسىنىڭ قان كېسەللىكلەر بۆلىمگە بالنىستقا ئېلىندى.مۇشۇ قىسقىغىنە ۋاقىتتىلا 100 مىڭ يۈەن چىقىم كەتتى .
 داۋاملىق داۋالىتشقا يەنە  نۇرغۇن پۇل كېتەتتى . ئۈمىدجاننىڭ ئاتا-ئانىسىنىڭ يۈرىكى پۇچىلاندى.بارچە ئامالنى قىلىپ بالىسىنى داۋالىتىشنى ئويلىغان بولسىمۇ ئىقتىساد ئۇلارنىڭ بېشىنى قاتتىق قاتۇردى. دوختۇرلار ئېيتىشىچە خىمىيىلىك داۋالاش ۋە يىلىك ئالماشتۇرۇشقا 500 مىڭ (50تۈمەن) يۈەندىن كۆپرەك پۇل كېتىدىغانلىقىنى ئېيتتى
      500 مىڭ ، بۇ ئاز پۇلمۇ ، بۇنى قانداق ھەل قىلىش كېرەك ، ئۈمىدجاننىڭ ئاتا- ئانىسى كۆپ ئويلاپ ئاخىرى خەلققە مۇراجات قىلىشنى ئويلىدى.
تورداشلار بىز سىلەرگە شۇنى ھاۋالە قىلماقچى ، بۇ ئۇكىمىزنىڭ تېزدىن ساقىيىپ ساۋاقداشلىرى يېنىغا قايتىشى ئ.ۈچۈن ھەرقايسىڭلارنىڭ  ياردەم قولىشىنى ئۈمىد قىلىمىز !
 تاما -تاما كۆل بولار ! ھەمممىز ئازراقتىن ياردەم قىلساق بۇ پۇل چوقۇم ھەل بولىدۇ  . ئەتراپىڭىزدىكى تونۇشلىرىڭىزغا ،خىزمەتداشلىرىڭىزغا ھاۋالە قىلىپ قويۇشۇڭلارنى ئۈمىد قىلىمىز !

  ئەگەر ئالاھىدە يوقلاپ بېرىشنى خالايدىغانلار باغداش تورى بىلەن ئالاقىلىشڭ !
     
ئەسكەرتىش: ئەسلىدە ئۈمىدجاننىڭ دادىسىنىڭ بانكا كارتا نومۇرىنى ئېلان قىلغان ئىدۇق . تور ئەزالىرى : ئۆزىمىز يىغىپ تورداشلار نامىدىن ئۈرۈمچىدىكى بىر قانچە تورداشلار يىغىلىشىپ بىللە يوقلاپ بارساق دىگەن تەكلىپىگە ئاساسەن مۇنبەر باشقۇرغۇچىسى نىجاد نىڭ  كارتا نومۇرىنى ئىشلىتىشنى قارار قىلدۇق .
