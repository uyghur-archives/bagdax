            ئەسسالامۇ ئەلەيكۇم تورداشلار ،ئالدى بىلەن ھەممىڭلارنىڭ تېنىنىڭ سالامەت ئىشلىرىنىڭ ئوڭۇشلۇق بولۇشىنى ئۈمىد قىلىمەن . 
      ھەممىمىزنىڭ بۆرەك زەئىپلىشىش كىسىلىگە گىرىپتار بولغان قېرىندىشىمىزغا ياردەم قولۇڭلارنى ئۈمىد قىلىمەن .
      ئىدرىس تۇرسۇن ئاقسۇ ئاۋاتتىن 36ياش ئاستا خاراكتىرلىك بۆرەك زەئىپلىشىش كىسىلىگە گىرىپتار بولغان ھازىر جىددى قانغا ئېھتىياجلىق ئۇنىڭ  قان تىپى ئاز كۆرىلىدىغان (RH阴性ab型)قان تىپى بولۇپ ئوخشاش قان تىپىدىكى قېرىنداشلار بولسا ياردەم قولىنى سۇنىشىنى ئۈمىد قىلىمىز .ئۇنىڭ سۆيۈملۈك ئايالى ، ئۇماق بالىلىرى ھەممىمىزگە ئوخشاش بەخىتلىك ئائىلىسى بار ، بەلكىم سىزنىڭ ئۇنىڭغا بەرگەن قېنىڭىز ئۇنىڭ ھاياتىغا يېڭى ئۈمىد ئاتا قىلىشى مۇمكىن . قېرىنداشلار ياردەم قىلايلى ، ئائىلىنىڭ تۈۋرۈكى بولغان ئاتىنىڭ ھاياتى ئۈچۈن ، شۇ سەبىيلەرنىڭ مۇكەممەل بىر ئائىلىدە ئاتا-ئانا مېھرىدە ساغلام ئۆسۈپ يىتىشى ئۈچۈن . ئۇلارغا ھەممىمىزنىڭ ئىللىق مېھرىنى ھېس قىلغۇزالايلى .
