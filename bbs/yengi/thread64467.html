    -سەن نىمىشقا خىزمەتتىن ۋاز كېچىسەن ؟ئالىي مەكتەپتە بىكارغا ئوقۇغانمىدىڭ ؟ توۋۋا .......

مەمۇرلۇق ئىمتاھانىدىن ئۆتكەندىن كېيىن نەچچە كۈن ئويلىنىپ چىقارغان قارارىمنى ئاڭلىغان ئۇرۇق - تۇققان ، تۇنۇش - بىلىشلىرىمنىڭ ھەممىسى كۆزلىرىنى چەكچەيتىپ تۇرۇپ ، ماڭا يۇقىرىقىدەك سوئاللارنى قۇيۇشتى . ھەتتا مېنى ئەخمەقتىن - ئەخمەققە چىقىرىشىپ ، سۆز - چۆچەك قىلىپ يۈرۈشتى . ئەگەر ئۇلار نەچچە ۋاقىتتىن بۇيان ئويلىغانلىرىمنى ، تۈزگەن پىلانىمنى بىلىپ قالىدىغان بولسا ، بەلكىم مېنى قىپقىزىل ساراڭغا چىقىرىۋېتىدىغان ئوخشايدۇ . مەن ئويلايمەنكى ، ھەرقانداق بىر ئىنسان ئۆز ھاياتىدا ئۆزىگە ھەم باشقىلارغا يۈز كېلەلىگۈدەك ھالەتتە ياشىشى كېرەك .

ئالى مەكتەپتە ئوقۇۋاتقان ۋاقتىمدا مەنمۇ باشقا ساۋاقداشلىرىمغا ئوخشاش ئوقۇش پۈتتۈرسەم چۇقۇم ياخشى خىزمەت تاپىمەن دەپ ئويلايىتتىم . كېيىنكى ۋاقىتلاردا ئەتراپىمدىكى تەڭتۈش قىزلارنىڭ توي قىلغاندىن كېيىنكى نىكاھ پاجىئەلىرى مېنىڭ ئىلگىرىكى كۆز - قاراشلىرىمنى پۈتۈنلەي ئۆزگەرتىۋەتتى . رىئال تۇرمۇشتىكى نۇرغۇن ئىشلار ماڭا ‹‹ نىمە ئۈچۈن قىزلارنىڭ كۈلكىسىدىن يىغىسى شۇنچە كۆپ بولىدۇ ؟ زادى نىمە ئۈچۈن ؟ بۇنىڭدىكى سەۋەب نىمە ؟ ›› دېگەن سوئاللارنى قويدى . قىزلارنىڭ تۇرمۇشى ، قىزلارنىڭ تەلىم - تەربىيسىگە ئائىت كىتابلارنى كۆپ ئوقۇش ، پىشقەدەم ئوقۇتقۇچىلار بىلەن پىكىر ئالماشتۇرۇش نەتىجىسىدە ھاسىل بولغان خۇلاسەم ‹‹ قىز - ئاياللارنىڭ تەلىم - تەربىيسىنى كۈچەيتىپ ، بالىلار تەربىيسىگە ئەھمىيەت بەرگەندىلا ئاندىن كۆزلىگەن مەقسەتكە يەتكىلى بولىدۇ . قىز - ئاياللارغا تەربىينى ، ئەخلاق تەربىيسىنى ياخشى ئىشلىمەي تۇرۇپ ، قىز - ئاياللار توغرىسىدا باش قاتۇرۇش بىھۇدە ئىش ›› دېگەن كۆز - قاراش بولدى .

ئېھتىمال ، يۇقىرىقى كۆز - قاراشلىرىم ئارقىلىق جەمىيئەتتىكى بارلىق قىز - ئاياللارغا ئىجابى تەسىر قىلالىشىم ، ياخشى يولغا يېتەكلىيەلىشىم سەل مۈشكۈل ئىش بولسىمۇ بىراق مەن ‹‹ لاياقەتلىك ئايال ، لاياقەتلىك ئانا ›› بۇلۇشقا تەييارلانغان ، بۇ ئۇلۇغ يولدا ئۈزلۈكسىز تىرىشىۋاتقان ھەم ئىزدىنىۋاتقان بىر ئىنسان بۇلۇش سۈپىتىم بىلەن كەلگۈسىدە پەرزەنىتلىرىمنى كۆڭلۈمدىكىدەك ياخشى تەربىيلەپ چىقالىسام ، مەن ئۈچۈن شۇنىڭ ئۆزى كۇپايە . مانا شۇنىڭ ئۆزى مەن ئۈچۈن ئەڭ ياخشى خىزمەتتۇر .

جەمىيئىتىمىزدە ھەرقايسى ساھەلەردە ئەرلەرگە ئوخشاش تىرىشچانلىق كۆرسىتىپ نەتىجە قازىنىۋاتقان ئاياللار خېلى كۆپ .خىزمەتتە ئۇتۇق قازانغان ئاياللارنىڭ پەرزەنىتلىرى تىرىشىپ ئۇقۇپ ئاتا - ئانىسىنىڭ يۈزىنى يۇرۇق قىلسا ، بەزى پەرزەنىتلەر ئاتا - ئانىسنىڭ خىزمەت بىلەن ئالدىراش بۇلۇپ كەتكەنلىكىدىن پايدىلىنىپ يامان يوللاردا مېڭىپ قالدى ھەم مېڭىۋاتىدۇ . دەسلەپتە ئاتا - ئانىلار بالىلىرى ئۆزىنىڭ قىلمىشلىرىنى يۇشۇرغان سەۋەبلىك بالىلارنىڭ ئەسكى خۇي - مىجەزى ، ئەسكى قىلمىشلىرىنى سەزمەيدۇ . بىلگەن ۋاقتىدا ئاللىقاچان كېچىككەن بولىدۇ . بالىنى چىڭ تۇتمىغىنىغا پۇشايمان قىلىشىدۇ . بالىلار كەلتۈرۈپ چىقارغان ئەسكى قىلمىشلارنىڭ يىلتىزىنى ئۆزلىرىدىن ئەمەس بەلكى ، مەكتەپتىن ، ئوقۇتقۇچىلاردىن ، مۇھىتتىن ، بالىلارنىڭ دوستلىرىدىن ئىزدەيدۇ . ھالبۇكى ، ئۆزلىرىنىڭ بالىلارنىڭ ئۆسۈپ - يېتىلىشتىكى ئەڭ مۇھىم ئامىل ئىكەنلىكىنى ئەسلا خىيالىغا كەلتۈرۈشمەيدۇ .

پىلانلىق تۇغۇت سىياسىتىنىڭ يولغا قۇيۇلىشى بىلەن يېقىنقى 20 نەچچە يىل مابەينىدە نۇرغۇن ئائىلىلەردە ‹‹ ئەتىۋارلىق بەگزادە ، ئەركە مەلىكە ›› لەر كۆپلەپ مەيدانغا كەلدى . بەزى ئاتا - ئانىلار بالىلىرىم بىر تال ئوغلۇم ، بىر تال قىزىم دەپ بالىلارنىڭ رايىغا بېرىپ ، نىمە دېسە قىلىپ بېرىپ چوڭ قىلغاچقا ، قىزلار كىيىم - كېچەك ، زىبۇ -زىننەت، ياسىنىشقا قەۋەتلا ئامراق بۇلۇپ قالغىنى بىلەن ئۆي ئىشلىرىغا تولىمۇ خام بولغاچقا توي قىلغاندىن كېيىن ئەرنىڭ ئۆيىدە يەرگە قاراپ قالىدىغان ئىشلار كۆپەيدى . ئوغۇللار بولسا بىكار تەلەپلىككە ، ئويۇن - تاماشا خۇمار بۇلۇپ قالغاندەك كەيىپ - ساپادىن مېھرىنى ئۆزەلمەيدىغان بۇلۇپ قېلىۋاتىدۇ . بۇنداق ئەركە چوڭ بولغان بالىلار مەسئۇلىيەتسىز يېتىلىپ قالغاچقا بالاغەتكە يېتىپ توي قىلغاندىن كېيىن ، دەسلەپتە ئايرىلماسقا ۋەدە قىلىشىپ توي قىلسىمۇ ، بىر ئايغا بارماستىنلا ‹‹ ئاجرىشىش ›› غەۋغاسىنى قىلىشىدۇ . بۇنداق بۇلۇشتىكى سەۋەب شۇكى ، قىز ياسىنىشنى بىلىدۇيۇ، تاماق ئېتىشنى بىلمەيدۇ . ئەينەككە قاراپ ئولتۇرۇشنى بىلىدۇيۇ ، ئۆينى پاكىز تۇتۇشنى بىلمەيدۇ . پۇل خەجلەشنى بىلىدۇيۇ، ئىقتىسادچانلىقنى بىلمەيدۇ .گەپ سېتىشنى بىلىدۇيۇ، ئېرىنىڭ تۇققانلىرى بىلەن ئەپ ئۆتۈشنى بىلمەيدۇ . ئۆزىنىڭ ئاتا - ئانىسىغا ئالەمچە كۆيىنىدۇ يۇ، ئېرىنىڭ ئاتا - ئانىسىنى ھۆرمەت قىلىشنى ، كۆيىنىشنى بىلمەيدۇ ، بىلگەن تەقدىردىمۇ بۇ مەجبۇرىيەتتىن باش تارتىدۇ . كىچىككىنە ئىش بولسىلا ئاتا - ئانا ، تۇققانلىرىغا داتلايدۇ . ‹‹ قول سۇنسا يەڭ ئىچىدە ›› دېگەننى بىلمەيدۇ .  ئاز - تولا نەسىھەت قىلىپ ، ئىش ئۆگەتكەن قېيىن ئانىلىرىنى يامان كۆرىدۇ ، ئىككى ئېغىز نەسىھەت قىلىپ قويسا ، ‹‹ مېنى خارلىدى ئۇ قىلدى ، بۇ قىلدى ›› دەپ ئاتا - ئانىسىغا بىرنى ئىككى ، ئىككىنى ئۈچ كۆرسىتىپ گەپ توشۇيدۇ . ئاتا - ئانا بولغۇچىلارمۇ قىزنىڭ ئېرى ياكى قۇدا بولغۇچىلار بىلەن ئەتراپلىق پاراڭلىشىپ باقماستىن ‹‹ بىز ئەتىۋارلاپ باقساق ، ئۇلار نىمىشقا قىزىمىزغا ئۇنداق مۇئامىلە قىلىدىكەن ›› دەپ ئۆكتەملىك بىلەن جېدەل قىلىشىدۇ . ئاقىۋەتتە بىر قاتار نىكاھ پاجىئەلىرىنى كەلتۈرۈپ چىقىرىدۇ . شۇنىڭغا ماس ھالدا بەزى يىگىتلەرمۇ باركى ، توي قىلغاندىن كېيىن ئائىلىدە ئۆتەيدىغان مەسئۇلىيىتىدىن ئۆزىنى قاچۇرۇشقا ئۇرىنىدۇ . بۇ گەپلەرنى قىلىۋېتىپ مۇنداق بىر ئىشنى قىستۇرۇپ قۇيۇشقا توغرا كەلدى . بۇ ھەقىيقى بولغان بىر ۋەقە . كىچىكىمدىن تارتىپ بىرگە ئويناپ چوڭ بولغان  بىر دوستۇم بار ئىدى . ئۆزى ئالى مەكتەپتە ئوقۇغان ، كىچىكىدىن تارتىپ شۇنچىلىك ئەخلاقلىق چوڭ بولغان ، تولىمۇ ھايالىق ، كىتاب ئوقۇشقا ھېرىسمەن قىز ئىدى . ئاتا - ئانىسىنىڭ تەسىرى بىلەن ناھايتى ئىشچان ھەم چېۋەر بولغان بۇ قىز ئۆزىنىڭ قىزلارغا خاس ئەدەپ - ئەخلاقلىقى بىلەن مەھەللىدىكى چوڭ - كىچىك ھەممەيلەننىڭ ياقتۇرىشىغا ، ھۆرمەتلىشىگە ئېرىشكەن ئىدى . دوستۇم يىگىرمە بىر ياشقا كىرگەن ۋاقتىدا دادىسى كېسەل سەۋەبى بىلەن ئالەمدىن ئۆتتى . ئارىدىن ئۈچ ئاي ئۆتكەندىن كېيىن ئۇرۇق - تۇققانلىرىنىڭ ھەممىسى  <<بالىنىڭ پەرزىنى ئادا قىلمىساق ، دادىسى رەھمەتلىكنىڭ روھى قورۇنۇپ قالىدۇ ›› دەپ ئۇنى توي قىلىشقا قىستىغاچقا ، ئۇ چوڭلارنىڭ مەسلىھەتىگە قۇشۇلۇپ ، چارۋا - مال دوختۇرخانىسىدا ئىشلەيدىغان بىر يىگىت بىلەن توي قىلدى . بۇ يىگىت دەسلەپتە ئۇنىڭغا ھەرجەھەتتىن كۆيۈنۈپ ، قىزنىڭ بېشىغا كەلگەن كۈلپەتلەرگە ھېسداشلىق قىلغان بولغاچقا دوستۇممۇ يىگىتكە قارىتا ‹‹ ئۆزى ئەخلاقلىق ھەم كۆيۈمچان ئىكەن . خىزمىتىنى ئوبدان قىلىپ ، مۇشۇ يىگىت بىلەن ئۆي تۇتۇپ كېتىشكە نىيەت قىلسام بولغۇدەك ›› دېگەن قارارغا كەلگەن . لېكىن توي قىلىپ تۆت ئاي ئۆتكەندىن كېيىن ، دوستۇم ھامىلدار بۇلۇپ قالغان ۋاقىتتا ، دوستۇمنىڭ ئېرى ئۆزىنىڭ ئەسلىدىكى بۇزۇق تەبىئىتىنى ئاشكارىلاپ ، تالادىكى يامان يولغا مېڭىپ قالغان چوكانلار بىلەن ئارىلىشىپ، ئۇنى روھى جەھەتتىن خارلاپتۇ .دوستۇم سېزىكلىك ئازابىدا بەكمۇ قىينىلىپ كەتكەن ۋاقىتلاردىمۇ ئۇنىڭ بىلەن قىلچە كارى بولماي ئۆزىنىڭ سەنىمىگە دەسسەپ يۈرۈپتۇ . دوستۇمنىڭ كىيىم - كېچىكى ، يىمەك - ئىچمىكى بىلەن چاتىقى بولماي ، ئۇزۇقلۇق يېتىشمەسلىك سەۋەبىدىن كۈنسېرى ئاجىزلىشىپ كەتكەن ، قان بېسىمى ئۆرلەپ بېشى ئاغرىيدىغان بۇلۇپ قالغان بىر ئاجىزە ئايالنى دوختۇرخانىغا تەكشۈرتۈپ بېقىش ئۇياقتا تۇرسۇن ، دوستۇم بىر ئىشنى قىلىپ دېسىلا ھۈرپىيىپ ، تىل سېلىپ ‹‹ مەن ساڭا بالىلىق بولايلى دېگەنمۇ ؟ ئۆزۈڭدىن كۆر ، تەتۈر پىشانە يېتىم ئوغلاق ›› دەپ قاتتىق سىلكىۋېتىپتۇ . پات - پات باش قىسمىدا پەيدا بولىدىغان ئاغرىقتىن ئىڭراپ كەتسە ، بىزارلىقىنى ئىپادىلەپ ئاغزى - بۇرنىنى ياستۇق بىلەن ئىتىۋالغان . ھەتتا ئوتتەك قىززىپ كەتكەن كۈنلەردىمۇ بىر پىيالە قايناقسۇ تەڭلىيەلمەي 12 - ئايدىكى قاتتىق سۇغۇقتا ئۆيدىن چىقىرىۋېتىپتۇ . دوستۇمنىڭ جەمەتىدىكىلەر بالا يېتىم قىلىشنى خالىمايدىغان ، قىز - ئوغۇللىرى ئاجرىشىمەن دەپمۇ ئاجرىشالمايدىغان بولغاچقا ، دوستۇم ‹‹ مېنى ئاجراشتۇرىۋېتىڭلار ›› دەپ يالۋۇرۇپ يۈرۈپمۇ ئاجرىشالمىدى . ئاخىرىدا دوستۇمنىڭ ئېرى ئۇنىڭغا شۇنچىلىك زەھەرخەندىلىك بىلەن ئادەم ئېغىزىغىمۇ ئېلىشقا ئۇيىلىدىغان تىللار بىلەن ھاقارەتلىگەنلىك سەۋەبىدىن قورساقتىكى بالىسى ئالتە ئايلىق بولغاندا بويىدىن ئاجراپ كەتتى . شۇ ئىشتىن كېيىن ئەسلى تالاغا قاراپ قېلىپ ئۆزگىرىپ كەتكەن دوستۇمنىڭ ئېرى خاتالىقىنى تونۇپ ، بىر قېتىم پۇرسەت بېرىشىنى ئۆتۈنگەن بولسىمۇ ، دوستۇم ئۇنىماي ئاجرىشىپ كەتكىنىچە ھازىرغىچە توي قىلىشنى خالىماي يۈرۈۋاتىدۇ .

ئادەتتە ئاياللارنىڭ ھامىلدارلىق ۋاقتى ئەڭ نازۇك ۋاقىت بۇلۇپ ، نىكاھتىكى سۈركۈلۈشلەرنىڭ كۆپىنچىسى مۇشۇ ۋاقىتلاردا يۈز بېرىدۇ . ھامىلدار ئاياللاردىكى ڧڧڧڧڧڧڧڧ ڧڧپىزىئولوگىيلىك ئۆزگىرىش سەۋەبلىك شۇ ئاياللارنىڭ مىجەزى خۇددى تاغنىڭ ھاۋاسىدەك ئۆزگىرىشچان بۇلۇپ قالىدۇ . باشقا ئائىلە ئەزالىرى بىلەنمۇ چىقىشالمايدىغان ، دېگىنىنى دېگەن قىلىشقا ئۇرىنىدىغان ، تولا قېيدايدىغان بۇلۇپ قالىدۇ . بەزى كىشىلەر ئاياللارنىڭ ئاشۇنداق نازۇك ئەھۋالىنىمۇ توغرا چۈشىنەلمەي كەينىدە غوتۇلدايدىغان ، ئالدىدا تاكاللىشىدىغان ئىشلارمۇ تولا كۆرىلىدۇ . بەزى ئەرلەرمۇ ھامىلدار ئاياللىرى قېيىدىسا تەڭلا قېييداپ كېتىدۇ تېخى . مانا مۇشۇنداق دوستۇمنىڭ ئېرىگە ئوخشاش ئۆز ئاياللىرىنىڭ مېھىر - مۇھەببەتكە ، چۈشىنىشكە ، كۆيۈنۈشكە ئېھتىياجلىق ۋاقتى ئىكەنلىكىنى ئويلىمايدۇ ، ئويلىغان تەقدىردىمۇ قىلالماي قالىدۇ . ئەسلى بالىلارنى ياخشى تەربىيلەشنىڭ ئالتۇندەك پۇرسىتى مانا مۇشۇنداق چاغلاردىن باشلىنىدىغان بۇلۇپ ، بالىلارنىڭ تۇغۇلغاندىن كېيىنكى سالامەتلىك ئەھۋالى ، مىجەز - خاراكتىرى ئاياللارنىڭ ھامىلدارلىق مەزگىلىدىكى كۈتۈنۈشىگە باغلىق .

يېقىنقى يىللاردىن بۇيان ، نۇرغۇنلىغان كىشىلەر بالىلارنى ياخشى تەربىيلەشكە تىرىشىۋاتىدۇ . بالىلارنى تەربىيلەش ئىشلىرىغا مەبلەغ سېلىۋاتىدۇ . بۇنداق ئىشلار كۆڭۈللەرنى ئىنتايىن سۆيۈندۈرىدۇ . مىللەتنىڭ ساپاسىغا نىسبەتەن بۇ ئىنتايىن ياخشى باشلىنىش . لېكىن شۇنى ئۇنتۇماسلىق كېرەككى ، بالىلارنى ياخشى تەربىيلەشنى - قورساقتىكى چاغدىن باشلىغاندا ئۈنۈمى تېخىمۇ ياخشى بولىدۇ . بۇ ئىشلار قانچە بالدۇر باشلانسا ، كۆزلىگەن مەقسەتكە ئاسانلا يەتكىلى بولىدۇ . ھامىلدار ئاياللارنىڭ ھامىلدارلىق ۋاقتىدا كۈتۈنۈشى ھەم روھى كەيپىياتى ئىزچىل ياخشى بولسا ، ئېرى ۋە ئائىلىسىدىكىلەر تەرىپىدىن يېتەرلىك مېھىر - مۇھەببەتكە ئېرىشەلىسە ھامىلىمۇ ساغلام يېتىلىپلا قالماستىن تۇغۇلغاندىن كېيىنمۇ سالامەتلىكى ياخشى بۇلۇپ ھەم كۆڭۈل خاتىرجەم بولىدۇ ھەم تەلىم - تەربىيە ئىشلىرىدا ئاكتىپ پوزىتسىيدە بولىدۇ . بۇنىڭ ئەكسىچە بولسا دوستۇمنىڭ بېشىغا كەلگەن قىسمەتتىن قىلچە پەرقى بولمايدۇ .

بۇ ماقالىنى يېزىش جەريانىدا تاغدىن - باغدىن سۆزلەپمۇ قالدىم . ئاساسى مەقسەتنى يىغىنچاقلىغاندا ، بالىلارغا تەلىم - تەربىينى ياخشى ئىشلىگەندە ھەم ئەخلاقلىق ھەم ياراملىق بالىلارنى يېتىشتۈرۈپ چىققىلى بولىدۇ . بۇنداق بالىلار ھەرقانداق ئىشتا مەسئۇلىيەتچان ھەم تىرىشچان بولىدۇ . ئادەتتە دادىنىڭ بالىغا تەسىر قىلىدىغان مۇھىم تەرىپى كۆپرەك . ئاياللارنىڭمۇ ھەم شۇنداق . ئەر - ئاياللار بالا تەربىيلەشتە بىر - بىرىنى قوللاپ ، بىر - بىرىگە ياخشى ماسلاشقاندىلا بالىلار ھەرقانداق ئىشقا مەسئۇلىيەتچانلىق بىلەن مۇئامىلە قىلىپلا قالماستىن ، كەلگۈسىدىكى ئائىلىسىگىمۇ مەسئۇلىيەتچان بولىدۇ . ئوغۇللار يىگىتلىك جاسارىتىنى ، قىزلار قىزلىق نازاكىتىنى ساقلاپلا قالماي، قەدىر - قىممىتىنى ، غۇرۇرىنى قەدىرلەيدىغان بولىدۇ .

شۇنىڭ ئۈچۈن بارلىق ئاتا - ئانىلار ، ئاتا - ئانا بۇلۇش ئالدىدا تۇرغان ياشلار ‹‹ ھەم ئەخلاقلىق ھەم ياراملىق پەرزەنتىم بولسىكەن ›› دەپ ئويلىساڭلار ، ئالدى بىلەن ئاياللارنىڭ ھامىلدارلىق ۋاقتىدىكى كۈتۈنۈشىگە دىققەت قىلىڭلار ، ئۇنىڭدىن كېيىن 0 ياشتىن 10 ياشقىچە بولغان ۋاقتىنى چىڭ تۇتۇڭلار . بولمىسا ھېچ ئىش قىلماي تۇرۇپ ياخشى بالا تەربىيلەش ئارزۇيۇڭلار خۇددى تېرەكتىن شاپتۇل تەمە قىلغاندەك ئىش بۇلۇپ قالىدۇ . بالىلارنى كىچىك ۋاقتىدىن باشلاپ ياخشى تەربىيلەشكە تىرىشساڭلار ، ئوغۇللىرىڭلار چوڭ بولسا ، بىر ئۆينىڭ يۈكىنى غەم قىلماي كۆتۈرەلەيدىغان ، ئېغىر ئىشلارغا پىسەنىت قىلمايدىغان ، بېشىغا ئېغىر كۈن چۈشسىمۇ ئاھ ئۇرمايدىغان بولىدۇ . قىزلىرىڭلار ياتلىق قىلىنغاندىن كېيىنىمۇ ئۆينىڭ ئىشلىرىنى جايىدا قىلالايدىغان ، تاماق ئېتىش قاتارلىق ئىشلارنى قاملاشتۇرۇپ قىلالايدىغان بولىدۇ . ئائىلىسىنى ، ئېرىنى قەدىرلەپ ، قېيىن ئاتا- ئانىسىنىڭ ھۆرمىتىنى قىلىپ دۇئاسىنى ئالالايدىغان بولىدۇ .

ھۆرمەتلىك تورداشلار، ئۇشبۇ قۇرلارنى ئوقۇۋېتىپ ، مېنىڭ نىمە ئۈچۈن خىزمەتتىن ۋاز كەچكەنلىكىمنىڭ سەۋەبىنى چۈشەنگەندەك بولغانسىلەر . بەلكىم سىلەرمۇ تۇققانلىرىمغا ئوخشاش ، مېنى ئەخمەقتىن - ئەخمەققە چېقىرىۋېتەرسىلەر ، قۇلۇڭدىكى تەييار ناننى تارتقۇزۇپ قويۇپسەنغۇ ؟ ئەمدى جېنىڭنى قانداق باقارسەن ؟ قولۇڭدا مائاش كارتىسى بولمىسا توي قىلاي دېسەڭمۇ يىگىتلەر ئاسانلىقچە ساڭا نەزىرىنى سالمايدۇ ›› دەپ ئويلايدىغانسىلەر . غەيرەتلىك ئادەم ھامان ئۆزىنىڭ يولىنى تېپىپ ماڭالايدۇ . مەن ئالدى بىلەن ئۆزۈمنى ھەم كەلگۈسىدىكى پەرزەنىتلىرىمنى ياخشى تەربىيلەشكە مەبلەغ سېلىش قارارىغا كەلدىم . مەن قىسقىغىنا ھاياتىمىدا ‹‹ لاياقەتلىك ئايال ، لاياقەتلىك ئانا ›› بولالىساملا شۇنىڭ ئۆزى كۇپايە . مەن ئويلايمەنكى ، ئۆيدىكى ۋاقىتتا ئەرلەر كومپىيوتېر ، يانپۇن بىلەن ھەپىلىشىپ ئولتۇرماستىن بالىلار بىلەن قىززىقارلىق ، ئەھمىيەتلىك ئويۇنلارنى بىرگە ئوينىشى كېرەك ، بۇنداق ئويۇنلار دادا- بالا ئوتتۇرىسىدىكى مېھىر - مۇھەببەتنى كۈچەيتىپلا قالماستىن بالىلارنىڭ ئۆسۈپ - يېتىلىشىگە ئاكتىپ تەسىر كۆرسىتىدۇ .

شۇنىڭغا ئوخشاش ئاياللارمۇ خىزمەتتىن بوشىسىلا سودا ساراي ، پەردازخانا ، تاماقخانىلارنى ئارىلاپ يۈرمەستىن كۆپرەك ۋاقتىنى بالىلارنىڭ تۇرمۇشىغا سەرپ قىلىشى لازىم .تامنىڭ ھۇلىمۇ پۇختا قۇرۇلسا  ئۇزاققىچە قەد كۆتۈرۈپ تۇرغىنىغا ئوخشاش بالىلارغا تەربىينى ياخشى ئىشلىسەك ، بالىلارنىڭ تۇرمۇش ئەھۋالى بىزنى ئاخىرقى ئۆمىرىمىزگىچە سۆيۈندۈرىدۇ . شۇنداقلا بىز ئۆزىمىزگە ھەم بالىلارغا ئىككىلا دۇنيادا يۈز كېلەلەيمىز . بالىلار - ئىنسانلارغا بېرىلگەن ئەڭ قىممەتلىك ئامانەتتۇر . ھاياتىمىزدىكى بۇ ئامانەتنىڭ ياخشى ساقلانغان - ساقلانمىغانلىقى ئۈستىدە بىزگە ھامان سورىقى بار . يەنە كېلىپ بۇ دۇنيادا بۇ سوراقنى بېرىپ بولىمىز .

مەن مۇشۇ نۇقتىلارنى بالىلار تەلىم - تەربىيسى ئۈستىدە ئىزدىنىش جەريانىدا ھېس قىلغانلىقىم ئۈچۈن ئۆزۈمنىڭ كۆپ ۋاقتىمنى بالا تەربىيلەشتەك ئۇلۇغۋار خىزمەتكە سەرپ قىلماقچى بولىۋاتىمەن . بۇنىڭ سەۋەبى شۇكى ، ‹‹كەسىپتە ئۇتۇق قازىنىپ ، ئائىلىدىن بەختىنى تاپالمىغان ئاياللارنىڭ ئەھۋالى ئەڭ ئېچىنارلىق بولىدۇ ›› دەپ ئويلايمەن .
