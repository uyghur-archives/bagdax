روماننىڭ باشلىنىش بېتىدە ياچى بەگ داستانىدىن بىر كوپلىت ئارىيە بىرىلگەن بولۇپ،

ئىگىسى يوق يەر بارمۇ ؟
قۇمدا سوققان شەر بارمۇ ؟
ياچى بىگىىم ئۆلدى دەپ ،
ئىسەنكىرەر ئەر بارمۇ ؟

    شۇنداق يېزىق بىلەن ئىپادىلەش ئۇرغۇ ھىسيات بىلەن ئېيتىلغان ياچى بەگ داستانىدىن پەرىقلىندۇ .  مەن ھازىرغىچە كىتاپنىڭ يۈزدەك بېتىنى ئوقۇپ بولغان بولساممۇ ئەمما قومۇل شۋىلىرىگە بىرىلگەن ئايرىم چۈشەندۇرۇش ئىزھاتلارنىڭ بىرىلمىگەنلىكىنى ھىس قىلىدىم . بۇ شېئىردىكى ئىسەنكىرەر ئەر بارمۇ دىگەن مىسرادىكى ئىسەنكىرەر دىگەن سۆزنىڭ مەنىسىنى  باغداشلىقلار بىلەن ئورتاقلاشقۇم بار .بەلكىم پالۋان ئۆلدى دەپ ئىسىدەپ يىغلايدىغان ئەر بار بارمۇ دەپ چۇشۈنىۋالغان تورداشلىرىمىز بولۇپ قېلىشى مۇمكىن .نىھايەتتە بۇ ئىسەدەش دىگەن مەنىنى بەرمەيدۇ ، ئىسەنكىرەش دىگەن بۇ قومۇل شىۋىسى ئاياش قەدىرلەش چىن يۇرىكىدىن كۆيۇنۇش ، ھەر ۋاقىت مۇھىم بىلىش ، قوغداش دىگەندەك  يانداش مەنىلەرنى بىرىش مۇمكىن ،ئەمما ئىسەنكىرەشنىڭ ئۆز مەنىسىنى  ئۆز ئاتا ئانىسىدىن مەھەلىداشلىرىدىن ئاڭلىماي تۇرۇپ چۇشۇنۇش سەل تەس .
.مېنىڭچە ئارىسلان تالىپ ئاكىمىزنىڭ بۇ ياچى بەگ رۇمانىدىن ئىبارەت بۇ كاتتا ئەمگىكى ياچى بەگنى ئەڭ جايىدا ئەڭ ياخشى ئەڭ مەنىلىك ئىسەنكىرگەنلىك بولۇپ قاپتۇ . ئەمدىكى گەپ موشۇنداق زور ئەمگەكنى قەدىرلەپ ئوقۇپ ئىسەنكىرەپ ئوقۇپ ئۆزلىرىگە ھايات ئىنىرگىيەسى توپلاشنى خالايدىغان ئوقۇرمەنلەرنىڭ ئىسەنكىرىىشىگە قاپتۇ دەپ ئويلۇدۇم .

    روماننىڭ باشلىنىشى بىلەنلا قىزىقىشىمىزنى قوزغايدۇ ، يەنى ئەرۋاھنىڭ ئىككى قالماق ئەسكىرىنى ئىككى يا ئوقى بىلەن يوشۇرۇن ئوق ئۈزۈپ دەل يۈرەككە تەككۇزۇپ ئۆلتۇرۇشى .روماننىڭ بۇنداق باشلىنىشى بىزنىڭ قىزىقىشىمىزنى قوزغايدۇ .يوشۇرۇن قەستلەش دۇشمەن كۇچ سېلىشتۇرمىسادا  پەرىق چوڭ  بولغاندا كۆپ قوللىنىدىغان بىر خىل جەڭ شەكلى بىز ئاستا ئاستا ئەرۋاھ  بىلەن بىللە ئۇنىڭ  قالماق ئەسكەرلەرنى يوشۇرۇن ئوق ئۈزۈپ جايلىغان يېرىدىن باشلاپ رومانغا تەبئى ھالدا شۇڭغۇشقا باشلايمىز ، ئاپتۇر ئوقۇرمەنلەرنى تارىخى رىئاللىق بىلەن ئۆزىنىڭ بەدئى كۈچىنى بىرلەشتۇرۇپ باشلىنىش دىتالىنى ۋايىغا يەتكۇزۇپ پىلانلاپ بىزنى ئوق ياچى بەگ سۇلتانشاھنىڭ ھاياتىغا باشلاپ كىردى.

    ياچى بىگىىم ئەسلى ئوق ياچى بەگ سۇلتانشاھ . ياچى دىگەن گەپنىڭ ئۆزىدىنلا چىقىپ تۇرۇپتىكى يانىڭ ئىگىسى يانى ۋايىغا يەتكۇزۇپ ئاتقۇچى ، قومۇلدا قەھرىمان ياچى بەگنىڭ ئىسمى يېڭى تۇغۇلغان بۇۋاقلارغا قويۇلۇش بىلەن ئەسلەنگەن ئىسەنكىرلەنگەن . ئەلۋەتتە ھازىر بەك ئاز قويۇلۇدۇ بۇ ئىسىم ، شۇنداق بولسىمۇ ئەجدات ئۇدۇملىرىنى تېپىۋېلىش ھەر يەردە ھەر ھالەتتە باشقىلارنىڭ تولۇق قايىلىقىنى  قوزغايدۇ .

    بۇ يەردە چۇشەندۇرۇپ قوپۇشقا تىگىشلىك يەنە بىر نۇقتا  ماۋۇ ياچى بەگ بىلەن ياچىۋەك دىگەن ئىككى ئىسىمنى ئارلاشتۇرۇۋېلىش ، مەسخىرۋازلىق پەيدا بولۇش ئىېھتىماللىقى بار رېھىم قاننىڭ بىر ئىتوتىدا رېھىم  قانننى ھەي ياچىۋەك دەپ مەسخىرە قىلىدۇ . بۇ يەردە ياچىۋەك بولسا يەنى قومۇل يەرلىك تىلىدىكى بۆشۈك تەۋەتتى نى ھالىغا باقماي ھارۋىنىڭ چاقىنى توسايمەن دەپ مىجىلىپ كەتكەن بىر كىچىك ھاشارەتنى كۆرسىتىدۇ .ئەمما بۇ روماندا تەسۋىرلەنگەن ياچى بەگ بولسا ئوقيا چى بەگ بولۇپ ئۇ ئوقيا ئېتىشقا ماھىرلىقى بىلەن ياچى بەگ ئاتالغان .بۇ ئىككى نۇقتىنى چوقۇم ئايدىڭلاشتۇرۋېلىش كېرەك ، بۇ كىتاپنىڭ ئۆزىدىمۇ تىلغا  ئېلىنغان .

    روماننىڭ 81 -بېىتدە ياچى بەگنىڭ ئانىسىنىڭ نامىدا ياچى بەگنىڭ چۇشىدە ئانىسى مۇنداق دەيدۇ :سەنمۇ ئەر بولدۇڭمۇ ؟   قومۇلنىڭ داڭلىق پالۋىنى ، ئوقياچىسى ، داستان يېزىپ تەرىپلەنگەن قەھرىمان ياچى بەگمۇ ئەر ھىساپلانمىسا قومۇلدا يەنە باشقا ئەر قالامدۇ دەپ ئويلاپ قېلىشىڭىز ئېھتىمال .شۇنداق مەن ئاپتۇرنىڭ رۇماننىڭ  كىرىش سۆز قىسمىدىكى تۆرۇك تاشاۋ غوللاردا تۇرمۇش ئۆگەندىم دەپ چۇشەنچە بەرگەنلىكىنى موشۇ بىر جۈملە سۆزدىن كۆرۋالدىم ، دەر ھەقىقەت بۇ سەنمۇ ئەر بولدۇڭمۇ دىگەن سۆز تۇركى قەۋىملەر ئىچىدىكى ئەرلەرنى ئەرلىك بۇرچىغا تارتىپ تۇرۇدىغان مۇھىم بىر ئەسكەرتمە سۆز بولۇپ ماقال تەمسىل كەبى تۇراقلاشقان بىر سېھىرلىك جۈملە .ئۇنداق گەپنى ھازىر شەھەر مۇھىتىدا يۇمشاق دەسسەپ يۇرگەنلەر ئارىسدا ناھايىتى ئاز ئېيتىلسىمۇ  سەھرا قىشلاقلاردا تاغ تېمىزلاردا يەنىلا مۇۋاىپىق پەيىتىتە ئەنە شۇنداق ئانىلىرىمىز  ئەنە شۇنداق ئاچا سىڭىلىلىرىمىز تەرپىدىن يۇز خاتىرە قىلىنماي ئېيتىلىپ ئەرلەرنى ئەرلىك بۇرچىغا تارتىدۇ . بۇ گەپ چىڭ دىسىڭىز چىڭ ، بوش دىسىڭىز بوش ، راس دىسىڭىز راس ، چاقچاق دىسىڭىز چاقچاق بولۇدىغان ئەمما ئەرلەردىن كۈتىۋاتقان ئۈمۈت ئارزۇ ئارمانلارنى نەق جايىدا جايىغا يەتكۇزۇپ  ئەرلەنىڭ يۇرەك بېغىشىغا تىگىدىغان بىر گەپ .ئەلۋەتتە ئاپتۇر بۇ سۆزنى ياچى بەگكە ئىشلىتىشتىن بۇرۇن ناھايىتى ياخشى بىر دىتال تەييارلاپ ياچى بەگنىڭ چۇشىدە ئاندىن ئوتتۇرغا قويغان .

    بىز تاراتقۇلاردىن  ئۆز بىكىستان مەلىكىسى دىلنارەنىڭ  ئۆز بىكىستان بىخەتەرلىك ئىدارىسىنىڭ باشلىقى رۇستەمنىڭ يۇزىگە تۇكۇرۇپ تۇرۇپ سەنمۇ ئەر بولدۇڭمۇ ؟ دىگەن ياڭراق ساداسىنى ئىشتكەن . ئۇيغۇر پەلسەپەسى ۋە ئۇيغۇر تىبابەت پەلسەپەسى ئۇچۇن ئۆزىنىڭ يۇرەك قېنىنى سىڭدۇرۇپ ئىجتىھات بىلەن ئەمگەك  مىۋىسىنى سۇنۇپ  كىلۋاتقان ئۇستاز داۋۇت ئوبۇلقاسىم ئەپەندىمنىڭ بىز ئەر ئەمەس مۇسۇلمانمۇ ؟ دىگەندەك تەپەككۇر مىتودلىرىنى ئوقۇپ كەلگەن . تۇركى مىللەتلەر ئىچىدە ئۇيغۇر لارنىڭ ئوغۇللارنى ئەر ئاتاش ئىنسان كامالىتىنى ئەرگە يۈكلەپ مەنە بىرشى مانا موشۇنداق ئاپتۇر تەرپىلىگەندەك ئائىللەرگىچە سىڭىپ كىرىپ ئانىلىرىمىز ، ئاچا سىڭىللىرىمىزنىڭ ئاغىزىدىن ئەسكەرتىش ، تارتىش ، ئەرلىك مەجبورىيەت ۋە بۇرچىنى ھىس قىلدۇرۇش مەنسىدە چىقىشى ئۇيغۇرلاردا ئەر بولۇشنىڭ مەنىسىنى باشقا مىللەتلەرگە قارىغاندا خېلىلا مول بولغان مەنىۋى مەزمونغا ئىگە قىلغان .

    شۇنداق بىرەر ئامرىكىلىق ئەرنىڭ ئالدىغا بېرىپ سەنمۇ ئەر بولدۇڭمۇ دىسىڭىز ئېھىتمال ئۇ ھىجىيىپ تۇرۇپ بۇبەك مۇھىم ئەمەس مۇھىمى مەن ئىلمى دىمىكراتىك دىيىشى مۇمكىن ، بىرەر ياپونىيەلىك داخى مىللىتى ئۇچۇن ھەسسە قوشۇش ئەقىل بىلەن ئىش قىلىش مۇھىم دەپ جاۋاپ بىرىشى مۇمكىن ،مەنغۇ ئامرىكا ياپونغىمۇ چىقىپ باقمىغان ، ئىشقىلىپ سەنمۇ ئەر بولدۇڭمۇ ،دىسە ئەڭ ھار ئالدىغىنى بىزنىڭ قەۋمىمىزمىكىن دەپ ئويلايمەن ئۆزەمچە .
. ئاپتۇر ياچى بەگنىڭ ئانىسى نامىدىن ياچى بەگكە دىگەن سەنمۇ ئەر بولدۇڭمۇ دىگەن جۈملە بىر ئىنچىكە تۇرمۇشنى كۆزىتىشتىن ،ئۇيغۇرلارنىڭ ئەڭ ئاساسى بىرلىكى بولغان ئائىللەردىكى ئەرلەرنى  ئەرلىك بۇرچىغا تارتىپ سۆزلىنىدىغان تۇراقلاشقان بىر سۆزنى ئىنتايىن ئۇستىلىق بىلەن ياچى بەگنىڭ چۈشگە ئورۇنلاشتۇرغان .

    كىتاپنى تېخى تولۇق ئوقۇپ بولالمىغان بولساممۇ ياچى بەگنىڭ ئالدىدىكى ئىسلام بىگىم ، ۋە ياچى بەگنىڭ كەينىدىكى قاسىم پالۋاننىڭ جاڭپېييۇەن شىڭشىسەي نىڭ 6500 كىشلىك قوشۇنى قومۇلغا يۇرۇش قىلغاندا ئۆزى يالغۇز سۇمقاغىغا بېرىپ  كىچىدە يوشۇرۇن ئوق ئېىتپ بىر چىرىكنى ئۆلتۇرگەنلىكى ، نىمانچە ئوخشۇشۇپ كىتىدىغاندۇ ھە(داۋامى بار )
