ماۋجۇشى قانداق سىياسەت قوللاندىكىن بىلمىدىم .بىز ئۇ دەۋىردە تۇغۇلغان لېكىن ئۇ تۇرمۇشلارنى كۆرمىگەن .
چوڭلارنىڭ ئاغزىدىن دايىم ماۋجۇشى ئۇنداق ياخشىتتى مۇنداق ياخشىتتى دەيدىغان گەپلەرنى ئاڭلايمەن تولاراق .
بوپمۇ پىشقەدەم ئوقۇتقۇچىلار ،پىشقەدەم كادىرلار ماۋجۇشىنى بەك ياخشى دەپ ماختاپ كېتىدۇ ..ماۋجۇشى بۇ
دادىلىرىمىزغا قانداق دارا يىگۈزۈپ قويغاندۇ؟
مېنىڭ پەرىزىم .ئالدى بىلەن كوپراتسىيەلەشتۈردى .بايلارنىڭ پۇللىرىنى ماللىرىنى يەرلىرىنى ئېلىۋېلىپ كەمبەغەللەرگە
تەڭ تەقسىم قىلىپ بەردى بىر قىسمىنى ..يەنە بىر قىسمىنى كوپراتسىيە مۈلكى قىلدى ..ئاندىن شەخسىلەرنىڭ قولىدا
كەتمەن گۈرجەكتىن باشقا پۇلغا يارىغۇدەك نەرسىنى قويمىدى .ئاڭلىسام مەدەنىيەت ئىنقىلابىنىڭ بەزى زۇلمەت مەزگىللىرىدە
بىر كىشىنىڭ مورىسىدىن تۈتۇن چىقسىمۇ ئۇنى تارتىپ چىقىرىپ كاپىتالىزىمنىڭ قالدۇغى قىلارمىش .
شۇڭا بەزى كىشىلەر يېرىم كىچىدىن ئاشقاندا يۇشۇرۇن موما سېلىپ يەرمىش .مانا مۇشۇنداق شەخسىنى كوپراتسىيەگە
يۈكسەك دەرىجىدە قارايدىغان قىلىۋالغان ماۋجۇشىنى ماختاپ بەرمىسە دادىلىرىمىز ئاچ قالمامدۇ ..
ماختاپ ماختاپ ماختىۋېرىپ تۆۋەندىكى دادىمىزدەك بولۇپ قالغان دادىلىرىمىز يەنە قانچىلىك باردۇر؟
لېكىن ماۋجۇشىنىڭ ئىككى سىياسىتىدا خاتالىق يۈز بەرمىدى دەپ قارايمەن .بىرسى كۆپ پەرزەنتلىك بولۇش سىياسىتى ..
بۇ بولۇپمۇ بىز ئۇيغۇرلارنىڭ تەبىئىي كۆپيىشىگە پايدىلىق بولدى .ئىككىنچىسى مىللەتلەر مەسىلىسىنى توغرا ھەل
قىلغان سىياسىتى .يەنى ئاپتونومنىڭ رەئىسىنى ھەتتا بەزىدە سېكرىتارىنى مىللىلاردىن قىلىش .يەنە بارلىق
نازىرلارنىڭ يەرلىكتىكى ئىدارىلارنىڭ بىرىنچى قول رەھبەرلىرىنى مىللىدىن قىلىش سىياسىتى .
بۇ سىياسەت خېلى ياخشى ئىدى .لېكىن ھازىر تەتۈرسىچە
