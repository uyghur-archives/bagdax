بۇ نۆۋەتتىكىسىنىڭ باھالىغۇچىلار سەۋۋىيەسى   بىرىنجى قېتمغا يىتەلمىدى !

ھەممە نەرسىىنىڭ چىكى بولىشى كېرەك ،توغرا سىز دىگەندە ...

دىگىنىڭىز مىنىڭ دىگىنىم بىلەن بىردەك ،،ئەمما سىز بىراقلا يۇسۇندا سەنئەتنى ئىنكار قىلسىڭىز بولمايدۇ،مىنىڭ ئىنكاسىمنى قايتا ئۇقۇڭ،مەن سەنئەتنى مىللەتنىڭ بۇندىن كىيىنكى تەرەققىياتىنىڭ ئاساس دىمەكچى ئەمەس پەقەت بىرقىسمى ،،،بىر مىللەتلە دەرۋەقە ھەر ساھە دىن تالانتلىق ئىختىساس ئىگىللىرى لازىم ،بىزدىن ئالىم،ئىنژىنىر،دوكتۇر،كارخانىچى،تەنتەربىيچى،يازغۇچى،دوختۇر،سىياسىيون،قاتارلىق ھەرساھە زاتلىرى ئىنتايىن مۇھىم مۇشۇ يۇسۇندا سەنئەتچىمۇ مۇھىم.گەپنىڭ پوس كاللىسىنى دىسەم باشتىمۇ دىگەن ئىدىم بۇ يەردىكى گەپ،ھەممە ئۆز يۇلىدا مۇۋاپىق بولغىنى ئەڭ ئەۋزەل سىىز دىگەندەك ھازىر دىمىسىمۇ ھەممە يەردە شۇ ناخشا -ئۇسۇل ئىشلىرى ئەۋىج ئېلىپ كەنىت-مەھەللە ھەممە ياق شۇ ئۇسۇلغا مەستابا بولدى بۇ راس ئىش،،بۇ ئىش سىياسەت خاراكتىرلىك  بۇلىۋاتقان ئىش ئىشەنمىسىڭىز كەنىت يېزىلارغار بېرىڭ ،ناخشا-ئۇسۇل مەدەنىي كۆڭۇل ئېچىش پائالىيتى كۆپ بۇلۇش ‹ئۈچ ئەل خىزمەت گۇرۇپپىسى›نىڭ ئالتە تۇرلۇك قىلمىسا بولمايدىغان مۇھىم ۋەزىپىسى،ھەپتىدە چۇقۇم شۇنداق ئىش قىلمىسا بولمايدۇ.كىچىك ئالما ئۇسۇلىنى ھاجەتخانىدا ئوينىمىدۇق قالغان يەرنىڭ ھەممە يەردە ھەر ساھەدىكى كىشىلەر ئويناپ تۇگەتتۇق،،ئۇدا ئىككى يىل ئوينىدۇق..زامانىۋى ئىدىيەنى سىڭدۇرۇش ماھىيەت جەھەتتىن توغرا ئىش ،خەلىقنى ئويغۇتۇش،نازار ئىدىيەلەردىن يىراق تۇرۇش،ئىلغار مەدەنىيەتنى ئۆگۇتۇش ۋە تەشۋىق قىلىش بۇ دۆلەت ئوتتۇرغا قويغان بەك ياخشى سىياسەت..مېنىڭ كاللامدىن ئۆتمەيدىغان بىر ئىش ،،كۈندە ناخشا-ئۇسۇل ئويناپ يۇرۇش بىز دەۋەتقان ئاتالمىش‹ئىلغار مەدەنىيەتنى ئۆگۇنۇش ۋە تەشۋىق قىلىش ›بۇلامدىغاندۇ؟ياشلار ياشلىق ئارزۇ  ئارمانلىرى ئۇرغۇپ تۇردىغان جەزبىدار قۇشۇن بۇلارنى كۇندا ناخشا -ئۇسۇل ئوينىتىۋەرمەت بولسا ئىلغار تېخنىكىلارنى،ئىلىم -پەنگە ئائىت تەكشۇرۇش ۋە ئىكىسكۇرسىيەلەرگە ،كىتاپ ئۇقۇش،ئىگىلىك تىكلەش،قانۇن -تۇزۇم،تارىخ،مىللىي مەدەنىيەت ،ئۆرۇپ-ئادەت قاتارلىق جەھەتلەردىنمۇ تەربىيەلىسەك يېزىلاردىكى شۇ ياشلارنىڭ ئىدىيەسىدىمۇ زور ئۆزگۇرۇش بولغان بۇلاتتى،،،مائارىپ ئەلۋەتتە مۇھىم شۇنداقلا بەك مۇھىم ،،شۇڭلاشقا شۇ دىھقان ياشلارنى دۆلەت تىلى ۋە باشقا جھەتتىكى تەربىيەلەش كۇرۇسلىرىغا ئۇيۇشتۇرۇپ ئۇلارنىڭ ئاڭ سەۋىيەسىنى يۇقۇرى كۆتۇرۇش كىرەك،،،
