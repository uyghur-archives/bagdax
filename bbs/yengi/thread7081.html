           تۈن كېچىدىكى مەخپىي يارلىق : " مەركەزدىن بىر چوڭ ئايرۇپىلان قاچتى "

      1971 - يىلى 13 - سىنتەبر تۈن كېچىدە نەنجىڭ ھەربىي رايۇنىنىڭ 2 - نومۇرلۇق ئەربابى شاۋ يۇڭيىڭنى دژورنى كاتىپ تاتلىق ئۇيقۇدىن ئۈيغىتىۋەتتى .
       بىر نەچچە مىنۇتتىن كېيىن ، بىر جىپ ماشىنا زىجىڭشەن تېغىنىڭ شىمالى ئېتىگىدىكى فۇگۇيشەن 7 - نومۇرلۇق قۇرۇدىن يولغا چىقتى . ماشىنىدا نەنجىڭ ھەربىي رايۇنىنىڭ 1- مۇئاۋىن قۇماندانى قوشۇمچە سەنمۇجاڭى شاۋ يۇڭيىڭ ئولتۇراتتى . . .
       جىپ ماشىنا جۇڭشەنلىڭدىكى 8 - نومۇرلۇق قۇرۇغا يېتىپ كەلدى . شاۋ يۇڭيىڭ شوپۇرغا ماشىنىنىڭ چىرىغىنى ئۈچىرىۋېتىپ ئۇدۇل قۇرۇغا كىرىشنى بۇيرىدى .
       شۈي شيۇنىڭ دەرۋازىسى ئۇنىڭغا كېچە - كۈندۈز ئوچۇق تۇراتتى ، بۇ شۇي شيۇنىڭ ئالاھىدە بۇيرۇقى ئىدى .
       شاۋ يۇڭيىڭ پەلەمپەيدىن چىقىۋېتىپ ، شۇي شيۇنىڭ ياتاقخانا ئالدىدا تۇرغىنىنى بايقىدى .
       _ ھېلى مەركىزى كومىتېت ئىشخانىسىدىن تېلېفۇن كەلدى ، مەركەزدىن بىر چوڭ ئايرۇپىلان قېچىپتۇ ، قايسى يۆنىلىشكە قاراپ قاچقانلىقى ئېنىق ئەمەس ئىكەن . مەركەز بىزگە : قاتتىق كۆزىتىڭلار ،ئەگەر سىلەر بار يەرگە قونىدىغان بولسا قايتا ئۇچۇشقا رۇخسەت قىلماڭلار ؛ بارلىق ئايرۇپىلانىڭلار ئۇچمىسۇن ، پاراخوتلىرىڭلار يولغا چىقمىسۇن ، بىر ئايرۇپىلان ، بىر پاراخوتنىڭمۇ قېچىپ كېتىشىگە يول قويۇلمىسۇن ، دەپ بۇيرۇق قىلدى .
       بىر نەچچە مىنۇتتىن كېيىن ، شاۋ يۇڭيىڭ ھەربىي رايۇن ئۇرۇش قىلىش بۆلىمىگە كىردى .
       . . .      . . .
       ئەتىسى _ 14 - سىنتەبر ئەتتىگەن سائەت ئالتىلەردە ، شۇي شيۇ بىلەن شاۋ يۇڭيىڭ شۇي شيۇ بىناسىنىڭ 2 - قەۋت پەلەمپەي يېنىدىكى تاماق جوزىسىدا ناشتا قىلىۋاتقاندا تېلېفۇن جىرىڭلىدى . شاۋ يۇڭيىڭنىڭ كېيىن ئۇقۇشىچە ، مەخپىي تېلېفۇننى جۇ ئېنلەي زۇڭلى بەرگەنىكەن .
       بىر ئازدىن كېيىن ، شۇي شيۇ چىقىپ كەلدى .
       _ سىزگىلا دەيمەن ، باشقىلارغا دېيىشكە بولمايدۇ ، لىن بياۋ قېچىپتۇ !

            جىددى ھەرىكەت قىلىش : تىغ ئۇچىنى "3 دۆلەت ، 4 تەرەپ " كە قارىتىش

       "مۇئاۋىن باش قۇماندان قېچىپ كېتىۋېتىپ ئايرۇپىلان دۇمىلاپ چۈشۈپ ھالاك بولغاندىن كېيىنكى 3 - كۈنى شۇي شيۇنى ماۋزېدۇڭ ، جۇئېنلەيلەر جۇڭنەنخەيگە مەخپىي چاقىرتتى ، تۈن نىسپى سائەت 2 لەردە شۇي شيۇ نەنجىڭگە يەنە مەخپىي قايتىپ كەلدى . 10 نەچچە مىنۇت ئۆتكەندىن كېيىن ، ئۇرۇش قىلىش بۆلىمىدە ساقلاپ تۇرىۋاتقان شاۋ يۇڭيىڭ شۇي شيۇنىڭ ئۆيىگە چاقىرتىلدى .
       _ رەئىس بىزگە "3 دۆلەت ، 4 تەرەپ " نى تۇتۇڭلار دېدى . كىمنى بىللە ئېلىپ مېڭىشنى خالىسىڭىز شۇنى ئېلىپ مېڭىڭ ، دەرھال يولغا چىقىڭ ! ئاۋۋال شاڭخەيگە بېرىپ ۋاڭ ۋېيگونى تۇتۇڭ . ئاندىن خاڭجۇغا بېرىپ چېن لىيۈننى تۇتۇڭ .
       ئالدىراپ - تېنەپ ئادەم تاللاندى :  نەنجىڭ ھەربىي رايۇن قوغداش بۆلىمىنىڭ باشلىقى ، قوغداش كادىرى ، ئۆزىنىڭ كاتىپى ۋە كانىۋىيى بولۇپ جەمئىي بەش كىشى تاللاندى . نەچچە مىنۇتتىن كېيىن ، بىر جىپ ماشىنا ئۈن - تۈنسىز ھەربىي رايۇن قۇرۇسىدىن چىقىپ ، نەنجىڭ ۋوگزالىغا يېتپ باردى . . .
       " 3  دۆلەت ، 4 تەرەپ " دېگەن نام لىن بياۋ ئۆزىنىڭ " 571 قۇرۇلىشى خاتىرىسى " دە ئۆزىنىڭ جاھىل مۇرتلىرىغا ۋە ئۇ تىزگىنلەپ تۇرىۋاتقان جايلارغا قويغان نام ئىدى .
       " 3  دۆلەت " __  نەنجىڭ ، شاڭخەي ، خاڭجۇ .
       " 4 تەرەپ "  __  بېيجىڭدىكى ھاۋا ئارمىيە سىياسى بۆلۈمىنىڭ مۇئاۋىن مۇدىرى جياڭ تىڭجياۋ ؛ شاڭخەيدىكى  ھاۋا ئارمىيىسىنىڭ سىياسى كومسسارى ۋاڭ ۋېيگو ؛ خاڭجۇدىكى ھاۋا ئارمىيە 5 - جۈنىنىڭ سىياسى كومسسارى چېن لىيۈن ؛ نەنجىڭ ھەربىي رايۇنى ھاۋا ئارمىيە مۇئاۋىن قۇماندانى جۇ جيەنپىڭ .
       تۈن كېچە ، تۈتەك . شاۋ يۇڭيىڭ ۋاگۇن دېرىزە پەردىسىنى قايرىپ تاشقىرىغا قارىدى . ئۇنىڭ كۆڭلى غەش ئىدى . بۇنىڭدىن ئۈچ كۈن بۇرۇن لىن بياۋ چۆل - جېزىرىدە گۇمران بولغان بولسىمۇ ، لېكىن ئۇنىڭ قانىتى تا ھازىرغىچە كېسىپ تاشلانمىغان ئىدى . . .
       قەپەسگە چۈشكەن يىرتقۇچ ھايۋان ئۈلۈم ئالدىدا جان تالىشىدۇ . " باش قۇماندان "ىدىن ئايرىلغان " 3  دۆلەت ، 4 تەرەپ " ئالاقزادىلىكتە قالغان يىرتقۇچ  ھايۋانغا ئوخشاش ئۈلۈم ئالدىدا جان تالىشارمۇ ؟ جاننى ئالقانغا ئېلىپ قويۇپ ئېلىشارمۇ ؟
       ئۇنى كۈتۈپ تۇرغان تۈن كېچىدىكى شاڭخەي كۆرۈنۈشتە جىمجىت تۇرغىنى بىلەن ، ئەمەلىيەتتە يوشۇرۇن قىرغىنچىلىقنى پىلانلاۋاتقانمىدۇ ؟ ئۆزلىرى ئاران ئالتە كىشى ، لىن بياۋنىڭ جاھىل مۇرتلىرى تېزگىنلەپ تۇرىۋاتقان شاڭخەي گارنىزوندا بەش مىڭدىن ئارتۇق ئەسكىرى كۈچ بار ، ئۇنىڭ ئۈستىگە شاڭخەيدە ۋاڭ ۋېيگو تېزگىنلەپ تۇرىۋاتقان ھاۋا ئارمىيە 4 - جۈنىمۇ بار . . .

شاڭخەيگە مەخپىي بېرىش :  ۋاڭ ۋېيگونى تۇتۇش

        تاڭ سەھەر . شاڭخەي گارنىزون شتابىنىڭ تېلېفۇنى جىرىڭلىدى .
        _ قۇماندانىڭلار تېلېفۇنغا كەلسۇن !
        قۇماندان ئاۋازنى ئاڭلاپ ، كۆزلىرىنى بىر نۇقتىغا تىككەنچە تۇرۇپلا قالدى . ئۇ جىددى قىياپەتتە زەڭ قويۇپ ئاڭلىدى .
       _ ياخشى ماشىنا ئەۋەتمەڭ ، كونا جىپ ئەۋەتسىڭىز بولىدۇ .
       گارنىزون قۇماندانى تۇرۇپكىنى قويۇپ قويۇپ ، دەرھال ماشىنىغا ئولتۇرۇپ شاڭخەي ۋوگزالىغا يۈرۈپ كەتتى .
        _ تېزلىكتە ئىككى شى يۆتكەپ كېلىپ ، شاڭخەينىڭ مۇھىم يوللىرىنى تېزگىنلەڭ ، ھاۋا ئارمىيە 4 - جۈنىنىڭ يول يۈرۈشىگە رۇخسەت قىلىنمىسۇن !
       يۈز كۆرۈشۈش بىلەنلا مۇئاۋىن قۇماندان شاۋ يۇڭيىڭ كەسكىن بۇيرۇق بەردى . شاڭخەي  گارنىزون قۇماندانى بۇيرۇققا بىنائەن يول ئالدى .
       ئاۋات شەھەر رايۇنىدىكى ماشىنىلار ئېقىنىدا ، سىرى چۈشۈپ كەتكەن ھېلىقى ھەربىي جىپ ماشىنا كالا ھارۋىسىدەك غىچىلداپ كېتىۋاتاتتى . ھېچكىممۇ ئۇنىڭغا كۆز قىرىنى سېلىپ قويمىدى .
       قارىماققا ، بۈگۈن چۈشتىن كېيىن ، جىڭجياڭ مېھمانخانىسى ئادەتتىكىدەك جىمجىت ئىدى . ۋاھالەنكى ، جىق سۇغا دولقۇن يوشۇرۇنغان بۇلىدۇ . جۇڭگونىڭ سىياسى ۋەزىيىتىدە يۈز بېرىۋاتقان چوڭ بىر ئىش مانا مۇشۇ بىنادا ھەل قىلىندى .
       شاڭخەي  گارنىزوننىڭ مۇئاۋىن  قۇماندانى باشلىغان بىر يىڭ ئەسكىرى كۈچ مۇشۇ بىناغا يوشۇرۇندى . . .
        10 - قەۋەتتىكى ياسىداق بىر كىچىك مېھمانخانىدا مەخپىي يىغىن ئېچىلىۋاتاتتى .  يىغىن جوزىسى ئەتىراپىدا نەنجىڭ ھەربىي رايۇننىڭ مۇئاۋىن قۇماندانى شاۋ يۇڭيىڭ ، شاڭخەي شەھەرلىك ئىنقىلابى كومىتېتنىڭ مۇئاۋىن مۇدىرى ۋاڭ خۇڭۋېن ، ماتيەنشۈي ۋە شاڭخەي  گارنىزوننىڭ قۇماندانى ۋە سىياسى كومىسسارى ئولتۇراتتى .
       _ رەئىس بىزنى ۋاڭ ۋېيگونى تۇتۇشقا بۇيرىدى . مېنىڭ پىكرىمچە ، يىلاننى ئوۋىسىدىن چىقىرىش كېرەك ... ئەڭ ياخشىسى ئەسكىرى كۈچ ئىشلەتمەي ، تېنچ يول بىلەن ھەل قىلىش كېرەك . كونكىرېت چارىسى : ۋاڭ خۇڭۋېن ۋاڭ ۋېيگوغا تېلېفۇن بەرسۇن . . . _ دېدى شاۋ يۇڭيىڭ خەرىتىنى جوزا ئۈستىگە قويۇپ .
       30 نەچچە ياشلىق ۋاڭ خۇڭۋېن دەرھال قەددىنى رۇسلاپ ، بېشىنى لىڭشىتىپ ماقۇل بولدى . ئەينى چاغدا ئۇ بىر « ئىنقىلابى كومىتېتنىڭ مۇئاۋىن مۇدىرى» ئىدى ، چوڭ ھەربىي رايۇننىڭ مۇئاۋىن قۇماندانى ئالدىدا بۇيرۇققا بويسۇنىدىغان كىچىك بىر ئەسكەر ئىدى ، خالاس .
      ھاۋا ئارمىيىسى 4 - جۈنىنىڭ سىياسى كومىسسارى ۋاڭ ۋېيگو پارقىراق پىكاپتا ئولتۇرۇپ جىڭجياڭ مېھمانخانىسىغا كېلىپ ، چاققانلىق بىلەن پەلەمپەيگە چىقتى ، ئۇنىڭ بىلەن كاتىپى ۋە تاپانچا ئاسقان توت نەپەر كانۋاي بىللە كەلگەنىدى .
       _ توختاڭلار ، _ خىتاب قىلدى ئىشىك تۈۋىدىكى ساقچى ۋاڭ ۋېيگونىڭ « ھەمراھ » لىرىغا ۋە ئۇلارنى يېنىدىكى دەم ئېلىش ئۈيىگە ئېلىپ كىردى .
       ۋاڭ ۋېيگو ئىشىكتىن ئاتلاپ يىغىن مەيدانىغا كىردى . ئوينىڭ ئوتتۇرىسىدا شاۋ يۇڭيىڭ ئولتۇراتتى . ۋاڭ ۋېيگو ھاڭ - تاڭ بولۇپ قېتىپلا قالدى . مۇئاۋىن قۇماندان شاۋ يۇڭيىڭنىڭ توساتتىن جىڭجياڭ مېھمانخانىسىنىڭ بۇ ئۈيىدە پەيدا بولغانلىقى ۋاڭ ۋېيگو ئۈچۈن ئەزرائىلنى كۆرگەندەك ئىش بولدى _ ھەممە ئىش ئايدىڭلاشتى ، توزاققا چۈشتۈم ، ئەمما كېچىكتىم !
       _ قولغا ئېلىندىڭ ! _ دەپ جاكارلىدى شاۋ يۇڭيىڭ سوغۇققىنە . گەپ ئارىلىقىدا ، نەنجىڭ ھەربىي رايۇنى قوغداش بۆلۈمىنىڭ باشلىقى بىلەن قوغداش بۆلۈمچىسىنىڭ باشلىقىنىڭ تاپانچىسى بىرلا ۋاقىتتا ئۇنىڭ كۆكسىگە قادالدى . قوغداش كادىرى كويزىنى « جاق » قىلىپ ئۇنىڭ قولىغا سالدى .
       ۋاڭ ۋېيگو چىرايى تاتارغان ھالدا يەر ئاستى ئۆيگە يالاپ ئېلىپ كىرىلدى . بىر ھەربىي جىپ ماشىنا سىرتتىكى يول ياقىسىدا توختۇتۇپ قويۇلغانىدى . ۋاڭ ۋېيگو يەر ئاستى ئۆيدىن يالاپ ئېلىپ چىقىرىلىشى بىلەن تەڭ ، ماشىنىنىڭ يېنىدا تۇرغان ئىككى كىشى تىزلىكتە ئۇنىڭ بىلىكىنى قايرىپ ماشىنا ئىچىگە ئېلىپ كىردى .
       ماشىنا ئۇچقاندەك تېزلىكتە تيەنما لەخمىسىگە قاراپ يۈرۈپ كەتتى .

              ئۇستىلىق بىلەن ئالداش :  چېن لىيۈننى تورغا چۈشۈرۈش

       _ باشلىق قۇماندان شۈي شيۇدىن تېلېفۇن .
       ۋاڭ ۋېيگونى ئەمدىلا « ئۇزىتىپ » تۇرۇشىغا ، تېلېفۇن جىرىڭلىدى . شاۋ يۇڭيىڭ تروپكىنى قولىغا ئالدى . مەخپىي تېلېفۇندىن شۈي شيۇنىڭ ئاۋازى ئاڭلاندى :
       _ سىز خاڭجۇغا بارماڭ ! بارسىڭىز ئۇلار سىزنى قاماپ قويىدۇ ! _ شۈي شيۇنىڭ ئاۋازىدىن كۆيۈنۈش ۋە ئەندىشە قىلىش ھېسياتى بىلىنىپ تۇراتتى .
       خاڭجۇدا تۇرۇشلۇق ھاۋا ئارمىيە 5 - جۈنى لىن بياۋنىڭ جاھىل مۇرتلىرى تېزگىنلەپ تۇرىۋاتقان بىر قۇرغان بولۇپ ، ماۋزېدۇڭنى « ھاۋادا ئۆلتۈرۈش ۋاقەسى » نى پىلانلىغانىدى . ماۋزېدۇڭنى پارتىۋېتىشقا ئۇرۇنغان يەردە ، نەنجىڭ ھەربىي رايۇنىنىڭ بىر مۇئاۋىن قۇماندانى ئۇلارنىڭ نەزىرىدە قانچىلىك ئىش بولسۇن ؟
       _ سىز بارماڭ ! _ دېدى شۈي شيۇ كەسكىن بۇيرۇق قىلىپ ، _ سىز ئامال قىلىپ چېن لىيۈننى شاڭخەيگە كەلتۈرۈڭ !
      شاۋ يۇڭيىڭ تېلېفۇننى قويۇپ ۋاڭ خۇڭۋېنغا مۇنداق دېدى :
      قانداق ؟ يەنىلا كونا ئۇسۇلنى قوللىنىپ ، چېن لىيۈننى تەكلىپ قىلىپ كەلتۈرەيلى .
      _ ئەمما ، _ دېدى ۋاڭ خۇڭۋېن بىر ئاز ئويلىنىۋېلىپ ، _ 2 _ شۇجىنىلا كېلىشكە تەكلىپ قىلساق ، قارشى تەرەپنىڭ گۇمانلىنىشىنى كەلتۈرۈپ چىقىرارمىكىن . . .
      _ ئۇنداقتا ، يولداش نەن پىڭنىمۇ تەكلىپ قىلايلى !
      نەن پىڭ 20 _ جۈننىڭ سىياسى كومېسسارى بولۇپ ، ئەينى چاغدا قوشۇمچە جېجياڭ ئۆلكىلىك پارتىكومنىڭ 2 - شۇجىسى ئىدى ، ئۇلارنىڭ ھەر ئىككىلىسى لىن بياۋنىڭ جاھىل مۇرتلىرى بولسىمۇ ، بۇ قېتىمقى ھەرىكەتنىڭ نىشانى چېن لىيۈن ئىدى .
      ۋاڭ خۇڭۋېن قولىغا تروپكىنى ئېلىپ ، « كونا » ئويۇننى ئوينىدى .
      _ يولداش نەن پىڭمۇ ؟ مەركەزنىڭ بىر يوليۇرۇقى بار چېن لىيۈن بىلەن ئىككىڭلار كەلسەڭلار . . .
      كەچ كىردى ، ئۆيدىكى چىراق نۇرى سارغۇچ چاقنىدى . خاڭجۇدىن ماشىنا بىلەن شاڭخەيگە كېلىش ئۈچۈن ئاز دېگەندە توت سائەت ۋاقىت كېتەتتى . شاۋ يۇڭيىڭ ھەقىقەتەن بەك ھېرىپ ، چارچاپ كەتكەنىدى ! ئۇنىڭ كارۋات يېنىغا بېرىپ باقمىغىنىغا بىر نەچچە كېچە بولدى _ ئەمما ئۇنىڭ ئۇخلىشىغا بولمايتتى . . .
      خاڭجۇدىن شاڭخەيگە قاتنايدىغان بىر پىكاپ چاقماق تېزلىكىدە ئۇچقاندەك كېلەتتى . بۇ پىكاپتىكى ھاۋا ئارمىيە 5 - جۈنىنىڭ سىياسى كومېسسارى چېن لىيۈن كۆزىنى يۇمۇپ ئۇيقىغا كەتتى .
      يەنە بىر پىكاپتا  20 _ جۈننىڭ سىياسى كومېسسارى نەن پىڭ ئەنسىز ھالدا ماشىنا چېرىقىنىڭ نۇرىغا قارىدى . . .
       چېن لىيۈننى كۈتۈپ تۇرغىنى يەنىلا شاۋيۇڭيىڭنىڭ « قولغا ئېلىندىڭ » دېگەن سوغۇققىنە سۆزى بولدى .
       چېن لىيۈن قولغا ئېلىنىپ ، تيەنما لەخمىسىگە ئېلىپ بېرىلدى .
       نەن پىڭ ھۇدۇققان ھالدا قارا تەرگە چۈمۈپ ، پارقىراپ تۇرغان كويزىنىڭ ھاۋا ئارمىيە 5 - جۈنىنىڭ سىياسى كومېسسارىنىڭ قولىغا سېلىنغانلىقىغا تەشۋىشلىك نەزىرى بىلەن قاراپ تۇرۇپلا قالدى .
      _ يولداش نەن پىڭ ، سىزنى چۆچۈتىۋەتتۇق ، _ دېدى شاۋيۇڭيىڭ كۈلۈمسىرىگەن ھالدا .

قۇماندانلىق شتابىدا  :  جۇجيەنپىڭ باش ئەگدۈرۈلدى

         ۋاڭ ۋېيگو ، چېن لىيۈنلەر نەنجىڭغا يالاپ ئېلىپ كېلىنگەن كۈننىڭ ئەتىسى چۈشتىن بۇرۇن نەنجىڭ ھەربىي رايۇنىنىڭ مۇئاۋىن قۇماندانى جۇ جيەنپىڭ شاۋ يۇڭيىڭنىڭ كاتىپىدىن كەلگەن تېلېفۇننى ئالدى .
        _ مۇئاۋىن قۇماندان شاۋ يۇڭيىڭ سىزنى كەلسۇن ، سۇرايدىغان ئىش بار ، دەيدۇ .
        جۇجيەنپىڭ ئىشخانىسىدىن ئايرىلىپ ، نەنجىڭ ھەربىي رايۇننىڭ قۇرۇسىغا يېتىپ كەلدى .
        _ ھوي ، لاۋجۇ ، ئولتۇرۇڭ ، ئولتۇرۇڭ .
        شاۋيۇڭيىڭنىڭ كەيپىياتى ناھايىتى ياخشى ئىدى .
        _ لاۋجۇ سىزدىن بىر ئىشنى سۇرايمەن . ئاڭلىشىمچە ، شاڭخەيدىكى جۈيلۇ كوچىسىدا بىر كىچىك يىغىن ئېچىلغانىكەن ، سىز قاتناشقانمۇ - يوق ؟
        جۇجيەنپىڭ بۇنى ئاڭلاش بىلەنلا كۆز ئالدى قاراڭغۇلىشىپ ، لاسسىدە بولۇپ قالدى .
        چاتاق چىقتى .
        ئاشۇ يىل 31 - مارتتا جۈيلۇ كوچىسىدىكى ھاۋا ئارمىيىسى 4 - جۈن مېھمانخانىسىنىڭ يەر ئاستىدىكى مەخپىي ئۆيىدە  لىن لىگو رىياسەتچىلىكىدە ، جياڭ تىڭجياۋ ، چېن لىيۈن ، جۇجيەنپىڭ ، ۋاڭ ۋېيگو ، يۈشىنيې قاتارلىقلار جەم بولۇپ ، « 571 قۇرۇلىشى» دىكى ئالاقىدار ماددىلارنى ئىجرا قىلىشنى مۇزاكىرە قىلغان ، شۇ قېتىمقى مەخپىي يىغىندا لىن لىگو : جۇڭگودىكى  " 3  دۆلەت ، 4 تەرەپ " نىڭ شاڭخەي جۈيلۇ كوچىسىدىكى مەخپىي ئۆيىدە دۇنياغا كەلگەنلىكىنى ، ئۇ جۇڭگو ئىنقىلابى تارىخىدىكى ئىككىنچى قېتىملىق « ھاكىمىيەتنى قۇراللىق تارتىۋېلىش » يېڭى ئەسىرىنى ئاچىدىغانلىقىنى جاكارلىغان ! . . .
       جۇجيەنپىڭ بۇ ئەسنادا يا ئاسمانغا چىقىش ، يا يەرگە كىرىپ كېتىشنىڭ ئامالى يوق ئىكەنلىكىنى بىلدى . . .
       جۇجيەنپىڭ نۇرسىز كۆزلىرىنى چەكچەيتىپ تۇرغان ئۈلۈك بېلىققا ئوخشاپ قالغانىدى .
       شاۋيۇڭيىڭ يەنىلا كۈلۈمسىرەپ تۇراتتى ، جۇجيەنپىڭنىڭ كۆزىدىن بولسا ياش تاراملاپ ئېقىۋاتاتتى .
       شاۋيۇڭيىڭ كۈلۈمسىرىگىنىچە تۇرىۋەردى ، جۇجيەنپىڭ گاھ ھۆركىرەپ ، گاھ ئاستا يىغلاپ ، ئاخىرى يىغىسىنى ماتەم موزىكىسىغا ئوخشاش تۈگەتتى .
       _ بۇ يەردىن چىقىپ ، ئۇبدان ئويلىنىڭ ، ئويلانغاندىن كېيىن ، قەغەزگە يېزىپ چىقىڭ ، _ دېدى شاۋيۇڭيىڭ سوغۇققانلىق بىلەن قەتئىي تۈردە .
       جۇجيەنپىڭ دەلدەڭشىگەن پېتى ئورنىدىن تۇرۇپ ، ئۆزىنى ئىدارە قىلالمىغان ھالدا پەلەمپەينىڭ يۆلەنچۈكىگە تايىنىپ بىنادىن چۈشتى . شۇ يەردە تۇرغان بىر نەچچە جەڭچى پارقىراپ تۇرغان كويزىنى ئۇنىڭ كۈچسىز قولىغا سالدى .
        ئەينى چاغدا يەنە بىر « تەرەپ » ھېسابلانغان جياڭ تىڭجياۋ بېيجىڭدا قولغا چۈشتى .

لىن بىياۋ كومپارتىيەنىڭ ئاتاغلىق ھەربى ئالىمى، ئىستىراتىگىيەچى، داڭلىق مارشال، ئەينى زاماندا خۇاڭفۇ ھەربىي مەكتەپنىڭ مۇئاۋىن مۇدىرى بولغان. گومىنداڭ بىلەن بولغان ئىچكى ئۇرۇشتا شەرقىي شىمال، شىمالىي جۇڭگو، بېيجىڭ-تىيەنجىن ئۇرۇشىدا ھەل قىلغۇچ خىزمەت كۆرسەتكەن، سابىق سوۋېتلەر ئىتتىپاقىغا تەربىيەلىنىشكە بارغاندا 2-دۇنيا ئۇرۇشىنىڭ ئالدىنقى سېپىدە جەڭگە قاتنىشىپ، فاشىسىت ستالىننىڭ ئوردىنىغا ئېرىشكەن. يېڭى جۇڭگو قۇرۇلغاندىن كېيىن پارتىيە ۋە دۆلەتنىڭ مۇئاۋىن رەئىسى ھەم ماۋ زېدۇڭنىڭ ئەڭ ئىشەنچلىك ئىز باسارى بولغان.
