    توكيو تەتقىقاتچىلىرى يېقىندا ئېلان قىلغان تەتقىقات نەتىجىسىدە كۆرسىتىلىشچە، ئەرلەر تاماقنى ھەدىدىن زىيادە تويۇپ يىگەندە، ھۆجەيرىلەرنىڭ راكقا ئايلىنىشىنى كونترول قىلغۇچى ئېرسىي ماددىلارنىڭ ئىقتىدارىنىڭ تۆۋەنلىتىۋېتىدىكەن. بۇ تەتقىقات 58 نەپەر ئەر ئۈستىدە ئېلىپ بېرىلغان بولۇپ، تەتقىقاتچىلار ئۇلارنىڭ يىمەكلىك ئادىتىنى تەكشۇرۈش ئارقىلىق، "ھەر ۋاقلىق تاماقنى بەك تويۇپ يەيدىغان" كىشىلەر بىلەن "80%  تۇيۇپ تاماق يەيدىغان" كىشىلەرنى سېلىشتۇرغاندا ئالدىنقى كىشىلەرنىڭ ھۆجەيرىلىرىدە ئۆزگىرىش بولۇپ، ھەرىكەت ئىقتىدارىنى يوقاتقان. بۇ راك پەيدا بولۇشنىڭ ئاساسلىق سەۋەبىكەن. 
        تەكشۈرۈشتە يەنە بايقىلىشچە، ئادەتتە يېشىل چاي ئىچىدىغانلار ۋە كاللەك بەسەينى كۆپ يەيدىغانلارنىڭ ھۆجەيرىلىرىنىڭ ھەركىتى ناھايىتى جانلىق بولىدىكەن، ئەمما نىمە ئۈچۈن ئادەم ھۆجەيرىسىگە نىسبەتەن جانلاندۇرۇش رولى بارلىقى تېخى ئېنىق ئەمەسكەن.
        بەك تويۇپ يىيىشنىڭ نىمە خەۋىپى بار؟ تەھلىل قىلىپ كۆرسەك: بىرىنچىدىن، ئاشقازان كېسىلىنى كەلتۈرۈپ چىقىرىدۇ. ئادەمنىڭ ھەزىم قىلىش سېستىمىسى قەرەللىك ھالدا ئارام ئېلىشى، ئۆز-ئۆزىنى يېڭىلىشى كېرەك، شۇنداق بولغاندىلا نورمال خىزمەت ئىقتىدارىنى ساقلاپ قالالايدۇ. ئادەم ئاشقازىنى شىللىق پەردىسىنىڭ ئۆمۈرى ناھايىتى قىسقا بولىدۇ، ھەر ئىككى-ئۈچ كۈندا بىر قېتىم يېڭىلىندۇ، ئالدىنقى تاماق تېخى تولۇق ھەزىم قىلىنىپ بولمايلا ئۈستىگە يەنە تاماق يىگەندە ئاشقازاننىڭ ئۆزىنى يېڭىلىشىغا پۇرسەت بولمايدۇ.
       ئىككىنچىدىن، تاماق يەپلا ئۇخلاشنىڭ زىيىنى كۆپ. بەزىلەر تاماق يەپلا ئۇخلاشقا ئادەتلەنگەن، بۇ شۇبھىسىز ئۇيقۇ بىلەن كېسەل تاپقانلىق. چۈنكى ئادەم ئۇخلىغاندا كۆپۈنچە ئەزالار ماددا ئايلىنىشى ئاستىلاپ، "دەم ئېلىش" باسقۇچىغا كىرىدۇ، ئەمما تاماقتىن كېيىنلا ئۇخلىغاندا، ئاشقازان، ئۈچەيلەر مەجبۇرى ھالدا"جىددى خىزمەت" ھالىتىدە تۇرىدۇ. تاماق ئۇزاق ۋاقىت ئاشقازاندا تۇرىۋەرسە ئاشقازان كۆپلەپ ئاشقازان كىسلاتاسى ئاجرىتىپ چىقىرىدۇ، بۇنىڭ بىلەن ئاشقازان شىللىق پەردىسى بۇزغۇنچىلىققا ئۇچراپ، ئاسانلا ئاشقازان ئىچكى دىۋارى شەلۋەرەش، ياللۇغلىنىشنى كەلتۈرۈپ چىقىرىدۇ. بۇنىڭ بىلەن راكقا ئايلىنىشنى تېزلىتىۋېتىدۇ.
       ئۈچىنچىدىن،غەيرىي تەرەققىي قىلىش. ئوزۇقلۇق نورمىدىن كۆپ بولۇش بەدەندىكى ھەرقايسى ئەزالارنىڭ يۈكىنى ئېغىرلاشتۇرىۋېتىدۇ. يۈرەك-مىڭە كېسەللىكلىرى، دىئابىت، جىگەرنى ماي قاپلاش، سېمىزلىك قاتارلىق "بايلار كېسەللىكلىرى" نىڭ ئاساسلىق سەۋەبى كۆپ يىيىشتە. بۇندىن باشقا، قالقان بەز ھومۇنى مىقتارى ئادەتتىكى يىمەكلىك مىقتارى بىلەن ئوڭ تاناسىپ بولىدۇ. ئۇزۇن ۋاقىت بەك كۆپ يىيىشتىن بەدەندىكى قالقان بەز ھورمۇنىنىڭ مىقدارنى ئاشۇرىۋېتىپ، سۆڭەكنىڭ كالتسىي خورىتىشىنى كۆپەيتىۋېتىدۇ، بۇنىڭ بىلەن سۆڭەك شالاڭلىشىشنى كەلتۈرۈپ چىقىرىدۇ. ياش چاغلىرىدىلا كۆپ ئوزۇقلىنىدىغان كىشىلەر ياشانغاندا، بەدىنىدىكى قالقان بەز ھورمۇنىنىڭ مىقدارى كۆپ بولغاچقا، كالتسينى قانچە كۆپ تولۇقلىغان بىلەنمۇ سۆڭەككە قوبۇل قىلىنىشى قىيىن بولىدۇ. شۇڭا سۆڭەك شالاڭلىشىش ئەھۋالىنىڭ كۆرۈلىش نىسبىتىمۇ يوقۇرى بولىدۇ.
