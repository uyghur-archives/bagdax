شىنجاڭ دىلدوسىت كارخانا باشقۇرۇش مەسلىھەتچىلىك چەكلىك شىركىتى تەييارلىدى . تور ئالاقىمىز www.dildost.com

     بۈگۈن بىز ئۇچۇرلىشىش تەرەققىي قىلغان بىر دەۋردە ياشاۋاتىمىز،بولۇپمۇ ئېلېكتىرونلۇق سودىنىڭ كېلىپ چىقىشى كىشىلەرنىڭ ياخشى مەھسۇلات سېتىۋالالماسلىقتەك توسالغۇسىنى بۆسۈپ تاشلىدى. كارخانىغا نىسبەتەن ئۆزىنىڭ مەھسۇلاتىنى تېخىمۇ كەڭ دائىرىدە تەشۋىق قىلالايدىغان بىر يول شەكىللەندى، ئىستېمالچىلارغا نىسبەتەن ئىستېمال قىلىشتىن ئىبارەت بىرىنچى ئامىلنى قانائەتلەندۈردى. ئېلېكتىرونلۇق سودا بىزنىڭ تۇرمۇشىمىزغا نۇرغۇنلىغان قولايلىقلارنى ئىلىپ كەلدى، پەقەت ئويلىيالماسلىق بولىدۇ ھەتتا سېتىۋالالمايدىغان مەھسۇلات بولمايدۇ.

     ئېلېكتىرونلۇق سودىنى چۈشىنىش ئىنتايىن ئاسان بولۇپ، ئاساسلىقى ئېلېكتىر ئەسلىھەلىرى بىلەن تور بىخەتەرلىك تېخنىكىسىغا تايىنىپ يۈرگۈزىدىغان بىرخىل سودا-سېتىق شەكلى، بۇنىڭ ئىچىدە يەنە ئەشيا-ئوبورۇتىغا مۇناسىۋەتلىك مۇلازىمەتلەرمۇ بار، مۇشۇلار ھەممىسى مۇكەممەل بىر سىستېمىنى تۈزۈپ چىققان. دۆلىتىمىزنىڭ ئېلېكتىرونلۇق سودا تەرەققىياتى ناھايىتى تىز بولۇپ، ئىشلەپچىقىرىش كۆلىمى ناھايىتى تىزلا كېڭەيدى، دۆلەت قاتلىمى جەھەتتىن ئىلىپ ئېيتقاندا دۆلەتنىڭ GDP ئۈچۈن ئاز بولمىغان شان-شەرەپلەرنى ئىلىپ كەلدى.

     تىجارەت بولسا تاۋار باشقۇرۇش تەپەككۇرىنىڭ يادروسى. چۈنكى پارچە سېتىش سودىسىنىڭ ھەممە كەسىپلىرىنىڭ باشلىنىشى تاۋارنى راۋاجلاندۇرۇشنى چۆرىدىگەن بولۇپ، ھەرقانداق پارچە ساتقۇچىلار تاۋارنىڭ ۋەزىيىتىگە قاراپ ئورۇن بېكىتىدۇ، تاۋارنىڭ بېكىتىلگەن ئورۇنىنى چۆرىدەپ ئۆزگىرىدۇ، تاۋارنىڭ ئۆزگىرىشىگە ئەگىشىپ تەڭشىلىدۇ. كەڭ مەنىدىن ئىلىپ ئېيتقاندا تاۋار باشقۇرۇش تاۋار بىلەن مۇناسىۋەتلىك بولغان بىر قاتار باشقۇرۇش پائالىيەتلىرىنى ئۆز ئىچىگە ئالىدۇ، كەسىپ تىجارىتىنىڭ ھەر بىر ھالقىسىغا چېتىلىدۇ؛ تار مەنىدىن ئىلىپ ئېيتقاندا تاۋار باشقۇرۇش ئاساسلىقى ماركىنىڭ باشقۇرۇلۇشى شۇنداقلا تىپ نومۇرىنى باشقۇرۇشىنى كۆرسىتىدۇ.

      ناۋادا نۆۋەتتە سىز بىر ئېلېكتىرونلۇق سودا شىركىتىدە تىجارەت قىلىۋاتقان بولسىڭىز چوقۇم ھەرخىل مەسىلىلەرگە يولۇقىسىز، ئېقىم مىقدارىغا ئېرىشىش ئۈچۈن ئامال ئىزدەيسىز. ئېقىم مىقدارى بولغاندىن كىيىن ئۆزگىرىشنى ئويلىشىسىز، ئۆزگىرىشنىڭ غەلىبىلىك بولۇش-بولماسلىقى مەھسۇلاتنىڭ نەقلىشىشكە باغلىق بولىدۇ، بۇ يەردە سېتىش پىسخولوگىيەسى شۇنداقلا ئىجتىمائىي توپ جەھەتتىكى بىلىملەرگە ئېھتىياجلىق بولىسىز. ئېلېكتىرونلۇق سودا رىقابىتى كەسكىنلىشىۋاتقان بۈگۈنكىدەك كۈندە بىز چوقۇم سانلىق مەلۇماتنى ئاساسلىق يېتەكلىگۈچى قىلىپ تىجارەت خىزمىتىنى يۈرگۈزۈشىمىز، ھەرقانداق بىر قارارنى چىقىرىشتىن بۇرۇن چوقۇم نەچچىلىگەن نېمىشقىلارنى سورىشىمىز بۇ مەسىلىلەر ئارقىلىق سانلىق مەلۇماتلارنى قىزىشىمىز، چوڭ سانلىق مەلۇماتتىن ئابۇنتنىڭ ھەرىكەت شەكلىنى قىزىپ چىقىپ، خېرىدارنىڭ قەلبىدىكى ئېھتىياجغا يېقىنلىشىشىمىز كېرەك، بۇ كىتابنىڭ مۇھىم نۇقتىسى بولسا سانلىق مەلۇمات تىجارەتلەشتۈرۈش نۇقتىسىدىن چىقىش قىلىپ تۇرۇپ ئېلېكتىرونلۇق سودىنى تەھلىل قىلىشتۇر.

     1.ئېلېكتىرونلۇق سودا بولسا توردا سېتىش ئەمەستۇر

     ئېلېكتىرونلۇق سودىنىڭ يۈزەكى مەنىسىدىن قارىغاندا ئۇ توردا مەھسۇلات سېتىشتۇر، ئەمما ئېلېكتىرونلۇق سودىنىڭ مەنىسى ئىنتايىن كەڭرى بولۇپ، چېتىلىدىغان تېخنىك سودا شەكلىمۇ بىر قەدەر كۆپ، مەھسۇلات سېتىش بولسا ئېلېكتىرونلۇق سودىدىكى بىر ھالقا خالاس.

     بۈگۈنكى ئېلېكتىرونلۇق سودىنىڭ ئالتە خىل شەكلى بار.

     1. B2C شەكىل

     كارخانا بىلەن ئىستېمالچى ئوتتۇرىسىدىكى ئېلېكتىرونلۇق سودا، ئاددىي قىلىپ ئېيتقاندا يەنى ئارىلىقتىكى ۋاسىتىچى سودىگەر ھالقىسى قىسقارتىلغان، كارخانا بىلەن ئىستېمالچى بىۋاسىتە يۈزلىنىدۇ، ئارىلىقتا ھېچقانداق قوشۇمچە ھالقا يوق، تەننەرخىمۇ ئازلىتىلغان بولۇپ، ئىستېمالچىلار باھاسى تېخىمۇ ئەرزان بولغان ھەمدە كاپالەتلەندۈرۈلگەن مەھسۇلاتلارنى سىتىۋالالايدۇ. B2C سودا شەكلىنى بىر قەدەر ياخشى قىلالىغىنى بولسا تيەنماۋ بىلەن جىڭدۇڭ.

      2. B2B شەكىل

     كارخانا بىلەن كارخانا ئوتتۇرىسىدىكى سودا شەكلى بولۇپ، كارخانا تور ياكى باشقا تورنىڭ ھەربىر سودا-سېتىققا نىسبەتەن ئەڭ ياخشى بولغان ھەمكارلاشقۇچى تېپىشنى قوللىنىپ، زاكاز قىلىشتىن تارتىپ راسچوت قىلىشقىچە بولغان بارلىق سودا-سېتىق ھەرىكەتلىرىنى تاماملايدۇ. B2B ئاساسلىقى كارخانىنىڭ ئىچكى قىسمى شۇنداقلا كارخانا بىلەن ھەمكارلاشقۇچى زاۋۇتلارغا قارىتا بىرىكتۈرۈش ئىلىپ بارىدۇ، شۇنداقلا ئىنتېرنېتتا كارخانا بىلەن كارخانا ئوتتۇرۇسىدەك سودا-سېتىقنى يۈرگۈزىدۇ. كارخانىنىڭ ئىچكى قىسمىدىكى تور قۇرۇلمىسى ئۇچۇرىنىڭ ئۆتۈشۈش ئاساسىنى ئىشلىتىپ، شۇنداقلا سىرتقى ئىنتېرنېت ئىشلەپچىقىرىش زەنجىرىدىكى ھەمكارلاشقۇچى زاۋۇتلارنى بىرىكتۈرۈپ، تەمىنلەش زەنجىرىدىكى بىرلەشتۈرۈشكە يىتىدۇ. شۇڭلاشقا، B2B سودا شەكلى ئارقىلىق كارخانا ئىچكى قىسمىدىكى ئۇچۇرنىڭ ئۆتۈشىشىدىكى تەننەرخنى ئاددىيلاشتۇرغىلى بولۇپلا قالماستىن، كارخانا بىلەن كارخانا ئوتتۇرىسىدىكى سودا-سېتىق تەرتىپىنى تېزلەشتۈرۈپ تەننەرخنىڭ خورىشىنى ئازايتىدۇ. ھازىر بۇ شەكىلنى ياخشىراق قىلىۋاتقىنى ئالى بابا.

     3. C2C شەكىل

     ئىستېمالچىلار بىلەن ئىستېمالچىلار ئوتتۇرىسىدىكى ئېلېكتىرونلۇق سودا، C2C سودا سۇپىسى بولسا ساتقۇچى ۋە سېتىۋالغۇچى ئىككى تەرەپ ئۈچۈن تەمىنلىگەن توردا بار بولغان سودا-سېتىق سۇپىسى ئارقىلىق، ساتقۇچى تورغا ئۆزىنىڭ مەھسۇلاتىنى چىقىرىدۇ، سېتىۋالغۇچى ئۆز ئىختىيارى بويىچە مال تاللاپ سېتىۋالىدۇ. ئاددىي قىلىپ چۈشەندۈرگەندە ئادەم بىلەن ئادەم ئوتتۇرىسىدىكى ئېلېكتىرونلۇق سودىدۇر. مەسىلەن، ئىستېمالچىدا كونا بىر كومپيۇتېر بار بولۇپ، ئۇ ئىنتېرنېت ئارقىلىق مەھسۇلاتنى باشقا بىر ئىستېمالچىغا سېتىپ چىقىرىدۇ، بۇ خىل شەكىلدىكى سودا -سېتىق ئېلېكتىرونلۇق سودىسى دەپ ئاتىلىدۇ. نۆۋەتتە C2C نى بىر قەدەر ياخشى قىلىۋاتقىنى تاۋباۋ.

     4. C2B شەكلى

     ئىستېمالچىلار بىلەن كارخانا ئوتتۇرىسىدىكى ئېلېكتىرونلۇق سودا بولۇپ،ئىستېمالچى ئاۋۋال ئېھتىياجىنى ئوتتۇرىغا قويىدۇ، ئاندىن ئىشلەپچىقارغۇچى كارخانا ئىستېمالچىنىڭ ئېھتىياجىغا ئاساسەن ئىشلەپچىقىرىشنى تەشكىللەپ ئۇ ئېھتىياجلىق بولغان مەھسۇلاتىنى ياساپ تەمىنلەيدۇ، ياكى ئىستېمالچىنىڭ ئۆزىنى ئىشلەپچىقىرىش ۋە مەھسۇلات لايىھسىگە قاتناشتۇرۇش ئارقىلىق ئىستېمالچى ئېھتىياجلىق بولغان مەھسۇلاتىنى ياساپ چىقىدۇ. ئومۇمەن قىلىپ ئېيتقاندا C2B نىڭ يادروسى بولسا ئىستېمالچىنى مەركەز قىلغان، ئىستېمالچى باش بولىدۇ، C2B مەھسۇلاتنىڭ تۆۋەندىكىدەك بىر قانچە خىل ئالاھىدىلىكى بار:

      n ئوخشاش بىر ئىشلەپچىقىرىش  زاۋۇت ئىشلەپچىقارغان مەھسۇلات مەيلى قايسى خىل يول بىلەن سېتىلغان بولسۇن باھاسى ئوخشاشتۇر، يەنى پۈتۈن دۆلەت خەلقى ئوخشاش بىرخىل باھادا سېتىۋالىدۇ.

     n مەھسۇلات باھا قۇرۇلمىسى مۇۋاپىق.

     n يول ئېنىق.

     n تەمىنلەش زەنجىرى ئېنىق.

     5. O2O شەكلى

     ئەمەلىي بازار بىلەن تور بازىرى بىرىكتۈرۈلگەن ئېلېكتىرونلۇق سودا شەكلى، O2O توردىكى خېرىدار باشلىغۇچى ئارقىلىق، ئەمەلىي دۇكان بىلەن ئىنتېرنېتنى مۇكەممەل ئۇلاشتۇرۇپ، ئىنتېرنېتنىڭ يەرگە چۈشۈشىنى ئەمەلگە ئاشۇرۇپ، خېرىدارنى توردىكى ئېتىبار باھادىن بەھرىلىنىش پۇرسىتىگە ئىگە قىلىش بىلەن بىرگە ئەمەلىي دۇكاندىكى غەمخورلۇق قىلىش مۇلازىمىتىدىنمۇ بەھرىمەن قىلىدۇ. O2O كوللېكتىپ سېتىۋېلىش ئەمدىلا گۈللەنگەن ۋاقىتتا پەيدا بولغان بولۇپ، خېرىدارلار كوللېكتىپ سېتىشنىڭ ئۇقۇمىنى تېخىمۇ بەكرەك چۈشەنگەن بولدى خالاس، كوللېكتىپ سېتىۋېلىش بولسا ۋاقىتلىق سېتىشنى تېزلىتىشتۇر، ئەمما O2O تور پونكىتىدا پەقەت سودىگەر بىلەن تور پونكىتى ھەمكارلىقنى داۋاملاشتۇرسىلا سودىگەرنىڭ مەھسۇلاتى داۋاملىق "سېتىشنى تىزلىتىش" ھالىتىدە بولىدۇ، O2O نىڭ سودىگەرلىرىنىڭ ھەممىسى ئەمەلىي بازاردا دۇكانغا ئىگە بولغان بولىدۇ، ئەمما كوللېكتىپ سېتىۋېلىشنىڭ سودىگىرىنىڭ ئەمەلىي دۇكىنى بولۇشى ناتايىن.

     ئاددىي ئېنىقلىمىسى بولسا بىر شىركەتنىڭ توردا ھەم ئەمەلىي بازاردا دۇكىنى بولسا شۇنداقلا بۇ ئىككى دۇكاندىكى باھا ئوخشاش بولسا O2Oدەپ ئاتىلىدۇ.

     6. ھۆكۈمەت شەكلى

     كارخانا بىلەن ھۆكۈمەت ئوتتۇرىسىدىكى ئېلېكتىرونلۇق سودا ھۆكۈمەت بىلەن كارخانا ئوتتۇرىسىدىكى ھەرخىل ئىشلارنى قاپلىغان بولۇپ، ھۆكۈمەتنىڭ سېتىۋېلىش، باج يىغىش، تاۋار تەكشۈرۈش، باشقۇرۇش نىزامنامىسىنى تارقىتىش شۇنداقلا قانۇن-بەلگىلىمىلەرنى تارقىتىشنى ئۆز ئىچىگە ئالىدۇ. بىرتەرەپتىن ھۆكۈمەت ئىستېمالچى بولۇش سۈپىتى بىلەن ئىنتېرنېتتا ئۆزىنىڭ سېتىۋېلىش تىزىملىكىنى ئېلان قىلىپ، ئېنىق، ئاشكارا، يۇقىرى ئۈنۈم، ساغلام ھالدا ئېھتىياجلىق بولغان مالنى سېتىۋېلىشنى ئىشقا ئاشۇرىدۇ. يەنە بىر جەھەتتىن ھۆكۈمەت ئىنتېرنېت ئارقىلىق ئېلېكتىرونلۇق سودا ئۇسۇلىنى ئاساس قىلىپ يېتەرلىك ۋە دەل ۋاقتىدا كارخانىلارنى كونترول قىلىپ تەڭشەش ئىلىپ بارىدۇ ھەمدە ئۆلچەم ۋە نازارەت قىلىپ باشقۇرۇشقا يېتەكچىلىك قىلىدۇ. ئىنتېرنېت ۋە باشقا تور تېخنىكلىرىنى ئىشلىتىپ، ھۆكۈمەت دەل ۋاقتىدا تولۇق ئۇچۇرلارغا ئېرىشىپ، دەل ۋاقتىدا قارار چىقىرىدۇ، تىزدىن ئىنكاس قايتۇرىدۇ، سىياسەت بەلگىلىمىلەرنى تىز ھەم بىۋاسىتە ھالدا كارخانىلارغا يەتكۈزىدۇ، ئېلېكتىرونلۇق سودىدا ھۆكۈمەتنىڭ مۇھىم رولى بار، يەنى ئېلېكتىرونلۇق سودىغا نىسبەتەن ھەرىكەتلەندۈرۈش، باشقۇرۇش ۋە ئۆلچەملەشتۈرۈش رولىدىن ئىبارەت.

     بەلكىم كەلگۈسى بىر كۈنلەردە ئاغدۇرۇلغان بىرخىل ئېلېكتىرونلۇق سودا شەكلى پەيدا بولۇشى مۇمكىن، مەيلى قانداقلا ئۆزگەرسۇن بۇ شەكىل خېرىدارنى ئاساس قىلىدۇ، كارخانا مەھسۇلات،مۇلازىمەت ساتىدۇ، ئىستېمالچى ئىستېمال، بەھرىلىنىشنى سېتىۋالىدۇ، ئىككىسىنىڭ مۇناسىۋىتى يەنىمۇ بىر قەدەم ئىلگىرىلەپ باشقا ساھەلەرگە يېڭى ئۆسۈش ئىلىپ كېلىدۇ، مەسىلەن، ئېلانچىلىق كەسپى، لايىھەلەش كەسپى، ئەشيا-ئوبوروتى كەسپى قاتارلىقلار.
